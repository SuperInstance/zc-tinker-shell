```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
A visual ASCII simulation of P0→P1→P2.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import sys

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.rock_density = rock_density
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self.safe = set()
        self.path = []
        
        # Generate maze
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self._place_rocks()
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def _place_rocks(self):
        """P0: Map the negative space (rocks)"""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * self.rock_density)
        
        # Avoid start and goal
        available = [(x, y) for x in range(self.width) for y in range(self.height)]
        available.remove(self.start)
        available.remove(self.goal)
        
        self.rocks = set(random.sample(available, num_rocks))
        for x, y in self.rocks:
            self.grid[y][x] = '#'
    
    def find_safe_channels(self):
        """P1: Identify safe channels (deadbands) using flood fill"""
        visited = set()
        stack = [self.start]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Only proceed if not a rock
            if (x, y) in self.rocks:
                continue
            
            self.safe.add((x, y))
            
            # Check neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited:
                        stack.append((nx, ny))
    
    def optimize_path(self):
        """P2: Find shortest path within safe channels (BFS)"""
        if self.goal not in self.safe:
            return []  # No safe path to goal
        
        queue = [(self.start, [self.start])]
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.goal:
                self.path = path
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) in self.safe and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        return []  # No path found
    
    def run_protocol(self):
        """Execute full P0→P1→P2 protocol"""
        print("=== DEADBAND PROTOCOL DEMO ===")
        print(f"Maze: {self.width}x{self.height}, Rocks: {len(self.rocks)}")
        print()
        
        # P0: Already done in initialization
        print("P0: MAP NEGATIVE SPACE (Rocks '#')")
        self.print_grid()
        print(f"Identified {len(self.rocks)} rocks (avoid these)\n")
        
        # P1: Find safe channels
        print("P1: IDENTIFY SAFE CHANNELS (Deadbands)")
        self.find_safe_channels()
        temp_grid = self._mark_safe()
        self.print_grid(temp_grid)
        print(f"Found {len(self.safe)} safe cells (operate only here)\n")
        
        # P2: Optimize within channel
        print("P2: OPTIMIZE WITHIN CHANNEL (Shortest safe path)")
        self.optimize_path()
        if self.path:
            final_grid = self._mark_path()
            self.print_grid(final_grid)
            print(f"Path length: {len(self.path)-1} steps (optimal: {abs(self.goal[0]-self.start[0]) + abs(self.goal[1]-self.start[1])})")
            print("SUCCESS: Reached goal via safe channels")
        else:
            self.print_grid()
            print("FAILURE: No safe path to goal exists")
        
        return len(self.path) > 0
    
    def print_grid(self, grid=None):
        """Print ASCII maze"""
        if grid is None:
            grid = self.grid
        
        print("   " + "".join(str(i%10) for i in range(self.width)))
        print("  +" + "-" * self.width + "+")
        
        for y in range(self.height):
            row = f"{y:2}|"
            for x in range(self.width):
                row += grid[y][x]
            row += "|"
            print(row)
        
        print("  +" + "-" * self.width + "+")
    
    def _mark_safe(self):
        """Create grid showing safe channels"""
        grid_copy = [row[:] for row in self.grid]
        for x, y in self.safe:
            if grid_copy[y][x] == '.':
                grid_copy[y][x] = ' '
        return grid_copy
    
    def _mark_path(self):
        """Create grid showing optimal path"""
        grid_copy = [row[:] for row in self.grid]
        for x, y in self.path[1:-1]:  # Skip start and goal
            grid_copy[y][x] = '*'
        return grid_copy

def main():
    """Run multiple simulations to demonstrate protocol effectiveness"""
    print("Deadband Protocol Maze Simulation")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print("=" * 60)
    
    successes = 0
    trials = 5
    
    for i in range(trials):
        print(f"\n--- Simulation {i+1}/{trials} ---")
        maze = DeadbandMaze(width=15, height=10, rock_density=0.15)
        if maze.run_protocol():
            successes += 1
        print("-" * 40)
    
    print(f"\n=== SUMMARY ===")
    print(f"Trials: {trials}")
    print(f"Successes (safe path found): {successes}")
    print(f"Success rate: {successes/trials*100:.1f}%")
    print("\nProtocol: P0→P1→P2 (strict priority)")
    print("P0: Identify rocks/obstacles")
    print("P1: Map safe channels (deadbands)")
    print("P2: Optimize path within safe channels")

if __name__ == "__main__":
    main()
```

#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize within channels).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import sys
from typing import List, Tuple, Set

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

# Symbols
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL = 'G'
PATH = '○'
SAFE = ' '  # Visited safe cell

class DeadbandMaze:
    def __init__(self, seed=None):
        if seed is not None:
            random.seed(seed)
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.place_rocks()
        self.visited = set()
        self.safe_channels = set()  # P1 result
        self.optimal_path = []      # P2 result
        
    def place_rocks(self):
        """P0: Map negative space (rocks)."""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[y][x] = ROCK
    
    def is_safe(self, x: int, y: int) -> bool:
        """Check if cell is not a rock and within bounds."""
        return 0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE and (x, y) not in self.rocks
    
    def find_safe_channels_bfs(self):
        """P1: Find all reachable safe cells from start (safe channels)."""
        queue = [self.start]
        self.safe_channels = {self.start}
        
        while queue:
            x, y = queue.pop(0)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if self.is_safe(nx, ny) and (nx, ny) not in self.safe_channels:
                    self.safe_channels.add((nx, ny))
                    queue.append((nx, ny))
        return self.goal in self.safe_channels
    
    def optimize_within_channels(self):
        """P2: Find shortest path through safe channels (A* simplified)."""
        if not self.find_safe_channels_bfs():
            return []  # No path exists
        
        # Simple BFS for shortest path
        queue = [(self.start, [self.start])]
        visited = {self.start}
        
        while queue:
            (x, y), path = queue.pop(0)
            if (x, y) == self.goal:
                self.optimal_path = path
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (nx, ny) in self.safe_channels and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        return []
    
    def run_unconstrained_greedy(self):
        """Simulate P2-only (greedy) agent that ignores P0/P1."""
        x, y = self.start
        path = [self.start]
        visited = {self.start}
        
        for _ in range(100):  # Max steps
            if (x, y) == self.goal:
                return path
            
            # Greedy move toward goal
            moves = []
            if x < self.goal[0] and self.is_safe(x+1, y):
                moves.append((x+1, y))
            elif x > self.goal[0] and self.is_safe(x-1, y):
                moves.append((x-1, y))
            if y < self.goal[1] and self.is_safe(x, y+1):
                moves.append((x, y+1))
            elif y > self.goal[1] and self.is_safe(x, y-1):
                moves.append((x, y-1))
            
            if not moves:
                # Stuck or hit rock
                return []
            
            # Pick first available move
            x, y = moves[0]
            if (x, y) in visited:
                return []  # Loop
            visited.add((x, y))
            path.append((x, y))
        
        return []
    
    def render(self, show_path=False, show_safe=False):
        """Render ASCII maze."""
        display = [row[:] for row in self.grid]
        display[self.start[1]][self.start[0]] = AGENT
        display[self.goal[1]][self.goal[0]] = GOAL
        
        if show_safe:
            for x, y in self.safe_channels:
                if display[y][x] == EMPTY:
                    display[y][x] = SAFE
        
        if show_path and self.optimal_path:
            for x, y in self.optimal_path[1:-1]:  # Skip start and goal
                if display[y][x] == EMPTY or display[y][x] == SAFE:
                    display[y][x] = PATH
        
        lines = []
        for y in range(MAZE_SIZE):
            lines.append(''.join(display[y]))
        return '\n'.join(lines)
    
    def print_stats(self, success: bool, path_length: int, method: str):
        """Print simulation statistics."""
        print(f"\n=== {method} ===")
        print(f"Success: {'YES' if success else 'NO'}")
        if success:
            print(f"Path length: {path_length}")
            print(f"Optimal: {OPTIMAL_PATH_LENGTH}")
            print(f"Excess: {path_length - OPTIMAL_PATH_LENGTH}")
        print(f"Safe cells found: {len(self.safe_channels)}")
        print(f"Rock cells: {len(self.rocks)}")

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 206)
Simulates P0→P1→P2 in a 20x20 maze with rocks.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import sys
from typing import List, Tuple, Set

# Constants from the JSON data
MAZE_SIZE = 20
OPTIMAL_PATH_LENGTH = 21
NUM_ROCKS = 64
NUM_TRIALS = 50

ROCK = '█'
OPEN = '·'
AGENT = 'A'
GOAL = 'G'
PATH = '○'
DEADBAND = '░'  # Safe channel visualization

class Maze:
    def __init__(self):
        self.grid = [[OPEN for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_rocks()
        
    def generate_rocks(self):
        """Place 64 rocks randomly, avoiding start and goal."""
        possible = [(x, y) for x in range(MAZE_SIZE) for y in range(MAZE_SIZE)]
        possible.remove(self.start)
        possible.remove(self.goal)
        
        self.rocks = set(random.sample(possible, NUM_ROCKS))
        for x, y in self.rocks:
            self.grid[x][y] = ROCK
            
    def is_valid(self, x, y):
        return 0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE
    
    def is_safe(self, x, y):
        return self.is_valid(x, y) and self.grid[x][y] != ROCK
    
    def display(self, path=None, deadband=None):
        """Return ASCII representation of maze."""
        lines = []
        for x in range(MAZE_SIZE):
            line = []
            for y in range(MAZE_SIZE):
                if (x, y) == self.start:
                    line.append(AGENT)
                elif (x, y) == self.goal:
                    line.append(GOAL)
                elif path and (x, y) in path:
                    line.append(PATH)
                elif deadband and (x, y) in deadband:
                    line.append(DEADBAND)
                else:
                    line.append(self.grid[x][y])
            lines.append(''.join(line))
        return '\n'.join(lines)

def greedy_path(maze: Maze) -> Tuple[List[Tuple[int, int]], bool]:
    """P2-only: Greedy optimization without P0/P1."""
    path = [maze.start]
    visited = set([maze.start])
    x, y = maze.start
    
    while (x, y) != maze.goal:
        # Greedy: move toward goal
        moves = []
        if x < maze.goal[0] and maze.is_valid(x+1, y):
            moves.append((x+1, y))
        if y < maze.goal[1] and maze.is_valid(x, y+1):
            moves.append((x, y+1))
        if x > maze.goal[0] and maze.is_valid(x-1, y):
            moves.append((x-1, y))
        if y > maze.goal[1] and maze.is_valid(x, y-1):
            moves.append((x, y-1))
            
        # Add random lateral moves if stuck
        if not moves:
            for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
                nx, ny = x+dx, y+dy
                if maze.is_valid(nx, ny) and (nx, ny) not in visited:
                    moves.append((nx, ny))
        
        if not moves:
            return path, False
            
        # Choose move (prefer toward goal)
        move = None
        for m in moves:
            if m not in visited:
                move = m
                break
                
        if move is None:
            return path, False
            
        x, y = move
        path.append((x, y))
        visited.add((x, y))
        
        # Hit rock?
        if (x, y) in maze.rocks:
            return path, False
            
    return path, True

def deadband_protocol(maze: Maze) -> Tuple[List[Tuple[int, int]], bool, Set[Tuple[int, int]]]:
    """Full P0→P1→P2 implementation."""
    # P0: Map negative space (rocks)
    rocks = maze.rocks  # Already known in simulation
    
    # P1: Identify safe channels (deadbands)
    # Use BFS to find all reachable safe cells
    from collections import deque
    
    deadband = set()
    queue = deque([maze.start])
    visited = set([maze.start])
    
    while queue:
        x, y = queue.popleft()
        deadband.add((x, y))
        
        for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
            nx, ny = x+dx, y+dy
            if maze.is_valid(nx, ny) and (nx, ny) not in visited:
                if (nx, ny) not in rocks:  # Safe cell
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    
    # P2: Optimize within safe channel (A* in deadband)
    import heapq
    
    def heuristic(a, b):
        return abs(a[0]-b[0]) + abs(a[1]-b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, maze.start))
    came_from = {}
    g_score = {maze.start: 0}
    f_score = {maze.start: heuristic(maze.start, maze.goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        
        if current == maze.goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(maze.start)
            path.reverse()
            return path, True, deadband
        
        x, y = current
        for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
            nx, ny = x+dx, y+dy
            neighbor = (nx, ny)
            
            if not maze.is_valid(nx, ny):
                continue
            if neighbor in rocks:  # P0 violation check
                continue
            if neighbor not in deadband:  # P1 violation check
                continue
                
            tentative_g = g_score[current] + 1
            
            if neighbor not in g_score or tentative_g < g_score[neighbor]:
                came_from[neighbor] = current
                g_score[neighbor] = tentative_g
                f_score[neighbor] = tentative_g + heuristic(neighbor, maze.goal)
                heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return [], False, deadband

def constrained_no_deadband(maze: Maze) -> Tuple[List[Tuple[int, int]], bool]:
    """Constrained but without P1 deadband mapping."""
    # Knows rocks (P0) but doesn't map safe channels
    # Just avoids rocks greedily
    path = [maze.start]
    visited = set([maze.start])
    x, y = maze.start
    
    while (x, y) != maze.goal:
        moves = []
        for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
            nx, ny = x+dx, y+dy
            if maze.is_valid(nx, ny) and (nx, ny) not in visited:
                if (nx, ny) not in maze.rocks:  # Avoid rocks
                    moves.append((nx, ny))
        
        if not moves:
            return path, False
            
        # Choose move closest to goal
        moves.sort(key=lambda pos: abs(pos[0]-maze.goal[0]) + abs(pos[1]-maze.goal[1]))
        x, y = moves[0]
        path.append((x, y))
        visited.add((x, y))
        
    return path, True

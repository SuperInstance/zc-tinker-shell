```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
A visual ASCII simulation of P0→P1→P2 priority.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import sys
from collections import deque

# Simulation parameters
MAZE_SIZE = 20
ROCK_COUNT = 64
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)
START = (0, 0)
TRIALS = 50

def generate_maze():
    """Generate a random 20x20 maze with rocks."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        r = random.randint(0, MAZE_SIZE-1)
        c = random.randint(0, MAZE_SIZE-1)
        if (r, c) != START and (r, c) != GOAL:
            rocks.add((r, c))
            maze[r][c] = '#'
    return maze, rocks

def print_maze(maze, path=None):
    """Print maze with optional path overlay."""
    for r in range(MAZE_SIZE):
        row = []
        for c in range(MAZE_SIZE):
            if path and (r, c) in path:
                row.append('*')
            else:
                row.append(maze[r][c])
        print(' '.join(row))
    print()

def unconstrained_agent(maze):
    """P2-only: greedy pathfinding ignoring rocks."""
    path = []
    r, c = START
    while (r, c) != GOAL:
        path.append((r, c))
        if r < MAZE_SIZE-1 and maze[r+1][c] != '#':
            r += 1
        elif c < MAZE_SIZE-1 and maze[r][c+1] != '#':
            c += 1
        else:
            return None  # stuck
    path.append(GOAL)
    return path

def constrained_agent(maze, rocks):
    """P0+P2: knows rocks but doesn't map channels."""
    # Simple BFS avoiding rocks
    queue = deque([START])
    parent = {START: None}
    visited = set([START])
    
    while queue:
        r, c = queue.popleft()
        if (r, c) == GOAL:
            # reconstruct path
            path = []
            while (r, c) is not None:
                path.append((r, c))
                r, c = parent[(r, c)]
            return list(reversed(path))
        
        for dr, dc in [(1,0), (0,1), (-1,0), (0,-1)]:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if (nr, nc) not in visited and maze[nr][nc] != '#':
                    visited.add((nr, nc))
                    parent[(nr, nc)] = (r, c)
                    queue.append((nr, nc))
    return None

def deadband_agent(maze, rocks):
    """P0→P1→P2: full protocol."""
    # P0: Map negative space (rocks already known)
    # P1: Identify safe channels (connected components)
    visited = [[False]*MAZE_SIZE for _ in range(MAZE_SIZE)]
    components = []
    
    for r in range(MAZE_SIZE):
        for c in range(MAZE_SIZE):
            if maze[r][c] != '#' and not visited[r][c]:
                # flood fill
                stack = [(r, c)]
                comp = set()
                while stack:
                    cr, cc = stack.pop()
                    if visited[cr][cc]:
                        continue
                    visited[cr][cc] = True
                    comp.add((cr, cc))
                    for dr, dc in [(1,0), (0,1), (-1,0), (0,-1)]:
                        nr, nc = cr + dr, cc + dc
                        if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                            if maze[nr][nc] != '#' and not visited[nr][nc]:
                                stack.append((nr, nc))
                components.append(comp)
    
    # Find component containing START and GOAL
    start_comp = None
    goal_comp = None
    for comp in components:
        if START in comp:
            start_comp = comp
        if GOAL in comp:
            goal_comp = comp
    
    if start_comp is None or goal_comp is None or start_comp != goal_comp:
        return None  # no safe channel connecting start to goal
    
    # P2: Optimize within safe channel (BFS within component)
    queue = deque([START])
    parent = {START: None}
    visited_set = set([START])
    
    while queue:
        r, c = queue.popleft()
        if (r, c) == GOAL:
            path = []
            while (r, c) is not None:
                path.append((r, c))
                r, c = parent[(r, c)]
            return list(reversed(path))
        
        for dr, dc in [(1,0), (0,1), (-1,0), (0,-1)]:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if (nr, nc) in start_comp and (nr, nc) not in visited_set:
                    visited_set.add((nr, nc))
                    parent[(nr, nc)] = (r, c)
                    queue.append((nr, nc))
    return None

def run_simulation():
    """Run 50 trials and collect statistics."""
    results = {
        'unconstrained_success': 0,
        'constrained_success': 0,
        'deadband_success': 0,
        'unconstrained_lengths': [],
        'constrained_lengths': [],
        'deadband_lengths': []
    }
    
    for trial in range(TRIALS):
        maze, rocks = generate_maze()
        
        # Unconstrained (P2-only)
        path1 = unconstrained_agent(maze)
        if path1:
            results['unconstrained_success'] += 1
            results['unconstrained_lengths'].append(len(path1))
        
        # Constrained (P0+P2)
        path2 = constrained_agent(maze, rocks)
        if path2:
            results['constrained_success'] += 1
            results['constrained_lengths'].append(len(path2))
        
        # Deadband (P0→P1→P2)
        path3 = deadband_agent(maze, rocks)
        if path3:
            results['deadband_success'] += 1
            results['deadband_lengths'].append(len(path3))
        
        # Show first trial visually
        if trial == 0:
            print("=== FIRST TRIAL VISUAL DEMO ===")
            print("Maze ('.' = empty, '#' = rock):")
            print_maze(maze)
            
            if path1:
                print("Unconstrained agent path (P2-only):")
                print_maze(maze, set(path1))
            else:
                print("Unconstrained agent: FAILED (hit rock)")
            
            if path2:
                print("Constrained agent path (P0+P2):")
                print_maze(maze, set(path2))
            else:
                print("Constrained agent: FAILED (no channel mapping)")
            
            if path3:
                print("Deadband agent path (P0→P1→P2):")
                print_maze(maze, set(path3))
            else:
                print("Deadband agent: FAILED (no safe channel)")
            print()
    
    return results

def main():
    print("Deadband Protocol Maze Simulation")
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}, Trials: {TRIALS}")
    print("=" * 50)

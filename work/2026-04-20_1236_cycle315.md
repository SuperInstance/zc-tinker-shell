**Task:** Complete the Deadband Protocol visual demo (ASCII maze simulation).

**Status:** I have written a Python script that generates a random 20x20 maze and visually demonstrates the strict P0→P1→P2 priority order of the Deadband Protocol. The script simulates an agent navigating from start (S) to goal (G) with obstacles (rocks). It shows three distinct phases:

1. **P0 (Map Negative Space):** Agent scans and marks all obstacle cells (rocks) — what NOT to do.
2. **P1 (Identify Safe Channels):** Agent performs flood-fill from start, only moving through non-rock cells, to find all reachable safe cells — where it CAN be.
3. **P2 (Optimize Within Channel):** Agent runs A* pathfinding, but constrained to only the safe cells identified in P1, to find the shortest safe path.

The simulation prints each phase visually with ASCII characters and reports success/failure and path length. It validates the protocol: if P0+P1 are skipped (greedy P2-only), the agent often hits rocks and fails; following P0→P1→P2 yields a safe, optimal path.

**Output:** The script is saved as `deadband_maze_demo.py`. Below is the code and an example run.

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker
Visual demonstration of P0→P1→P2 priority in a 20x20 maze.
"""

import random
from collections import deque
import heapq

def generate_maze(width=20, height=20, rock_density=0.15):
    """Generate a random maze with start, goal, and rocks."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    # Place rocks
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
    # Ensure start and goal are clear
    start = (0, 0)
    goal = (width-1, height-1)
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    return maze, start, goal

def print_maze(maze, title="Maze"):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))

def p0_map_rocks(maze):
    """P0: Map negative space — identify all rock cells."""
    rocks = []
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start, rocks_set):
    """P1: Identify safe channels via flood-fill from start."""
    width, height = len(maze[0]), len(maze)
    safe = set()
    queue = deque([start])
    visited = set([start])
    while queue:
        x, y = queue.popleft()
        safe.add((x, y))
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and (nx, ny) not in rocks_set:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    return safe

def p2_optimize_path(maze, start, goal, safe_set):
    """P2: Find shortest path within safe channels using A*."""
    width, height = len(maze[0]), len(maze)
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            path.reverse()
            return path
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            neighbor = (nx, ny)
            if 0 <= nx < width and 0 <= ny < height:
                if neighbor not in safe_set:
                    continue  # Not in safe channel — deadband constraint
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    return None  # No safe path

def greedy_p2_only(maze, start, goal):
    """Greedy P2-only (no P0/P1) for comparison."""
    width, height = len(maze[0]), len(maze)
    path = []
    x, y = start
    while (x, y) != goal:
        path.append((x, y))
        # Simple heuristic: move toward goal
        if x < goal[0]:
            nx, ny = x + 1, y
        elif y < goal[1]:
            nx, ny = x, y + 1
        else:
            break
        if nx < 0 or nx >= width or ny < 0 or ny >= height:
            break
        if maze[ny][nx] == '#':
            return None  # Hit a rock
        x, y = nx, ny
    path.append(goal)
    return path

def run_demo():
    """Run the full Deadband Protocol demonstration."""
    random.seed(42)  # For reproducibility
    maze, start, goal = generate_maze(20, 20, 0.15)
    print_maze(maze, "Initial Maze (S=start, G=goal, #=rock)")
    
    # --- P0: Map Negative Space ---
    rocks = p0_map_rocks(maze)
    rocks_set = set(rocks)
    print(f"\nP0: Mapped {len(rocks)} rocks (negative space).")
    
    # --- P1: Identify Safe Channels ---
    safe = p1_find_safe_channels(maze, start, rocks_set)
    # Visualize safe channels
    maze_p1 = [row[:] for row in maze]
    for x, y in safe:
        if maze_p1[y][x] == '.':
            maze_p1[y][x] = '~'  # Safe water
    print_maze(maze_p1, "P1: Safe Channels (~ = safe, # = rock)")
    print(f"P1: Found {len(safe)} safe cells reachable from start.")
    
    # --- P2: Optimize Within Safe Channels ---
    path = p2_optimize_path(maze, start, goal, safe)
    if path:
        maze_p2 = [row[:] for row in maze]
        for x, y in path[1:-1]:  # Exclude start and goal
            maze_p2[y][x] = '*'
        print_maze(maze_p2, "P2: Optimal Path Within Safe Channels (* = path)")
        print(f"P2: Safe path length = {len(path)-1} steps.")
        success = True
    else:
        print("P2: No safe path to goal.")
        success = False
    
    # --- Compare with Greedy P2-only (violation) ---
    greedy_path = greedy_p2_only(maze, start, goal)
    if greedy_path:
        print(f"\nGreedy P2-only: Path length = {len(greedy_path)-1} steps (but may have been lucky).")
    else:
        print("\nGreedy P2-only: Failed — hit a rock (typical outcome).")
    
    # Summary
    print("\n=== Deadband Protocol Summary ===")
    print("P0→P1→P2: " + ("SUCCESS" if success else "FAIL"))
    print("Strict priority ensures agent avoids rocks, finds safe channels, then optimizes.")
    return success

if __name__ == "__main__":
    run_demo()

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 in a simple 20x20 maze.

Based on fleet-knowledge/narrow-games/constraint_sim_final.json
and fleet-knowledge/docs/DEADBAND-PROTOCOL.md
"""

import random
from typing import List, Tuple, Set

class MazeNavigator:
    """Implements the Deadband Protocol for maze navigation."""
    
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        total_cells = width * height
        num_rocks = int(total_cells * rock_density)
        
        # Avoid start and goal positions
        possible_positions = [(x, y) for x in range(width) for y in range(height)]
        possible_positions.remove(self.start)
        possible_positions.remove(self.goal)
        
        self.rocks = set(random.sample(possible_positions, num_rocks))
        
        # State tracking
        self.position = self.start
        self.path = [self.start]
        self.safe_channels = None
        self.optimal_path = None
        
    def p0_map_negative_space(self) -> None:
        """P0: Identify all states where failure probability > threshold."""
        print("=" * 60)
        print("P0: MAPPING NEGATIVE SPACE")
        print("=" * 60)
        print(f"Identified {len(self.rocks)} rock positions (failure states)")
        print(f"Rock density: {len(self.rocks)/(self.width*self.height):.1%}")
        
    def p1_find_safe_channels(self) -> Set[Tuple[int, int]]:
        """P1: Find all paths where no position is in negative space."""
        print("\n" + "=" * 60)
        print("P1: FINDING SAFE CHANNELS")
        print("=" * 60)
        
        # Use BFS to find all reachable safe positions
        visited = set()
        queue = [self.start]
        safe_positions = set()
        
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            
            # Skip if it's a rock
            if (x, y) in self.rocks:
                continue
                
            safe_positions.add((x, y))
            
            # Add neighbors
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        self.safe_channels = safe_positions
        print(f"Found {len(safe_positions)} safe positions")
        print(f"Safe channel coverage: {len(safe_positions)/(self.width*self.height):.1%}")
        
        # Check if goal is reachable
        if self.goal in safe_positions:
            print("✓ Goal is reachable through safe channels")
        else:
            print("✗ Goal is NOT reachable through safe channels")
            
        return safe_positions
    
    def p2_optimize_within_channel(self) -> List[Tuple[int, int]]:
        """P2: Find optimal path constrained to safe channels."""
        print("\n" + "=" * 60)
        print("P2: OPTIMIZING WITHIN SAFE CHANNELS")
        print("=" * 60)
        
        if self.safe_channels is None:
            print("ERROR: Must complete P1 first!")
            return []
            
        if self.goal not in self.safe_channels:
            print("ERROR: Goal not in safe channels!")
            return []
        
        # A* search constrained to safe channels
        open_set = {self.start}
        came_from = {}
        
        g_score = {pos: float('inf') for pos in self.safe_channels}
        g_score[self.start] = 0
        
        f_score = {pos: float('inf') for pos in self.safe_channels}
        f_score[self.start] = self._heuristic(self.start, self.goal)
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score[pos])
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                path.reverse()
                self.optimal_path = path
                print(f"Found optimal path of length {len(path)-1} steps")
                return path
            
            open_set.remove(current)
            
            x, y = current
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                neighbor = (x + dx, y + dy)
                
                # Constraint: must be in safe channels
                if neighbor not in self.safe_channels:
                    continue
                    
                tentative_g = g_score[current] + 1
                
                if tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor, self.goal)
                    open_set.add(neighbor)
        
        print("ERROR: No path found within safe channels")
        return []
    
    def _heuristic(self, a: Tuple[int, int], b: Tuple[int, int]) -> int:
        """Manhattan distance heuristic."""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def visualize(self, show_path: bool = False) -> None:
        """Visualize the maze with ASCII art."""
        print("\n" + "=" * 60)
        print("MAZE VISUALIZATION")
        print("=" * 60)
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')
                elif show_path and self.optimal_path and pos in self.optimal_path:
                    row.append('·')
                elif self.safe_channels and pos in self.safe_channels:
                    row.append(' ')
                else:
                    row.append('░')
            print(''.join(row))
        
        legend = [
            "\nLEGEND:",
            "S = Start",
            "G = Goal", 
            "█ = Rock (P0 negative space)",
            "░ = Unsafe (outside P1 channels)",
            "  = Safe channel (P1)",
            "· = Optimal path (P2)" if show_path else ""
        ]
        print('\n'.join(filter(None, legend)))
    
    def run_deadband_protocol(self) -> None:
        """Execute the full Deadband Protocol."""
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("=" * 60)
        
        # P0: Map negative space
        self.p0_map_negative_space()
        self.visualize(show_path=False)
        
        # P1: Find safe channels
        safe = self.p1_find_safe_channels()
        if not safe:
            print("No safe channels found. Protocol terminated.")
            return
            
        # P2: Optimize within channel
        path = self.p2_optimize_within_channel()
        
        if path:
            print(f"\nOptimal path: {path}")
            self.visualize(show_path=True)
            
            # Simulate navigation
            print("\n" + "=" * 60)
            print("SIMULATING NAVIGATION")
            print("=" * 60)
            
            steps = 0
            for pos in path:
                if pos in self.rocks:
                    print(f"CRASHED at {pos}! Violated P0 constraint.")
                    break
                steps += 1
                if pos == self.goal:
                    print(f"SUCCESS! Reached goal in {steps-1} steps.")
                    break

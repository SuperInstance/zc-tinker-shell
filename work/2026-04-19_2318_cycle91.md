import random
from collections import deque

def generate_maze(width=10, height=10, rock_density=0.2):
    """Generate a random maze with start, goal, and rocks."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks randomly
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
    
    # Ensure start and goal are clear and distinct
    start = (0, 0)
    goal = (width-1, height-1)
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    
    return maze, start, goal

def print_maze(maze, title="Maze"):
    """Print the maze with a title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))

def p0_map_rocks(maze):
    """P0: Map negative space (rocks). Return set of rock coordinates."""
    rocks = set()
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(maze, start, goal, rocks):
    """P1: Find all safe paths (avoiding rocks) using BFS.
       Returns a set of all reachable safe cells."""
    width, height = len(maze[0]), len(maze)
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    safe_cells = set()
    queue = deque([start])
    visited = set([start])
    
    while queue:
        x, y = queue.popleft()
        safe_cells.add((x, y))
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    
    return safe_cells

def p2_optimize_path(maze, start, goal, safe_cells):
    """P2: Find shortest path within safe channels using BFS.
       Returns path as list of coordinates or empty list if no path."""
    if goal not in safe_cells:
        return []  # Goal not reachable within safe channels
    
    width, height = len(maze[0]), len(maze)
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    queue = deque([(start, [start])])
    visited = set([start])
    
    while queue:
        (x, y), path = queue.popleft()
        
        if (x, y) == goal:
            return path
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) in safe_cells and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    return []  # No path found

def visualize_steps(maze, start, goal, rocks, safe_cells, path):
    """Create visualizations for each protocol step."""
    # P0 visualization: highlight rocks
    maze_p0 = [row[:] for row in maze]
    for (x, y) in rocks:
        maze_p0[y][x] = '#'
    print_maze(maze_p0, "P0: Rocks (Negative Space)")
    print(f"Rocks mapped: {len(rocks)}")
    
    # P1 visualization: highlight safe channels
    maze_p1 = [row[:] for row in maze]
    for (x, y) in safe_cells:
        if maze_p1[y][x] not in ['S', 'G']:
            maze_p1[y][x] = ' '
    print_maze(maze_p1, "P1: Safe Channels (Deadbands)")
    print(f"Safe cells: {len(safe_cells)}")
    
    # P2 visualization: show optimal path
    maze_p2 = [row[:] for row in maze]
    for (x, y) in path[1:-1]:  # Exclude start and goal
        maze_p2[y][x] = '*'
    print_maze(maze_p2, "P2: Optimal Path Within Channels")
    print(f"Path length: {len(path)-1} steps")

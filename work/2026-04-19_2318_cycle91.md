#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 91)
Simulates P0→P1→P2 in a 10x10 ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import heapq
from collections import deque

# ========== P0: Define the negative space (rocks) ==========
def create_maze():
    """10x10 maze, '#' = rock (negative space), '.' = open, 'S' = start, 'G' = goal"""
    maze = [
        "##########",
        "#S.......#",
        "#.######.#",
        "#.#....#.#",
        "#.#.##.#.#",
        "#.#....#.#",
        "#.######.#",
        "#........#",
        "#......G.#",
        "##########"
    ]
    return [list(row) for row in maze]

def find_position(maze, char):
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == char:
                return (x, y)
    return None

# ========== P1: Map safe channels (deadband) ==========
def map_safe_channels(maze, start):
    """BFS to find all cells reachable without hitting rocks."""
    rows, cols = len(maze), len(maze[0])
    safe = set()
    queue = deque([start])
    safe.add(start)
    
    while queue:
        x, y = queue.popleft()
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < cols and 0 <= ny < rows:
                if maze[ny][nx] != '#' and (nx, ny) not in safe:
                    safe.add((nx, ny))
                    queue.append((nx, ny))
    return safe

# ========== P2: Optimize within channel ==========
def a_star_in_channel(maze, start, goal, safe_cells):
    """A* search constrained to safe channels only."""
    rows, cols = len(maze), len(maze[0])
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            return path[::-1]
        
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            neighbor = (nx, ny)
            if 0 <= nx < cols and 0 <= ny < rows:
                if neighbor not in safe_cells:
                    continue  # Not in safe channel → skip (P1 enforcement)
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    return None  # No path within safe channels

# ========== Unconstrained greedy (P2-only baseline) ==========
def greedy_unconstrained(maze, start, goal):
    """Greedy move toward goal, ignoring rocks (will crash)."""
    path = [start]
    current = start
    visited = set([start])
    
    while current != goal:
        x, y = current
        best = None
        best_dist = float('inf')
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < len(maze[0]) and 0 <= ny < len(maze):
                dist = abs(nx - goal[0]) + abs(ny - goal[1])
                if dist < best_dist and (nx, ny) not in visited:
                    best = (nx, ny)
                    best_dist = dist
        if best is None:
            return None  # Stuck
        if maze[best[1]][best[0]] == '#':
            return None  # Hit a rock
        current = best
        visited.add(current)
        path.append(current)
    return path

# ========== Visualization ==========
def print_maze_with_path(maze, path, title):
    print(f"\n=== {title} ===")
    maze_copy = [row[:] for row in maze]
    for i, (x, y) in enumerate(path):
        if maze_copy[y][x] not in ['S', 'G']:
            maze_copy[y][x] = '*'
    for row in maze_copy:
        print(''.join(row))
    print(f"Path length: {len(path)-1} steps")

def print_safe_channels(maze, safe_cells):
    print("\n=== P1: Safe Channels (Deadband) ===")
    for y, row in enumerate(maze):
        line = []
        for x, cell in enumerate(row):
            if (x, y) in safe_cells:
                line.append('.')
            else:
                line.append(cell)
        print(''.join(line))
    print(f"Safe cells: {len(safe_cells)}")

# ========== Main simulation ==========

    print("Deadband Protocol Maze Demo — Tinker (Fleet Cocapn)")
    print("Based on constraint_sim_final.json: deadband success = 50/50, unconstrained = 0/50")
    print("-" * 50)
    
    maze = create_maze()
    start = find_position(maze, 'S')
    goal = find_position(maze, 'G')
    
    print("Initial Maze (S=start, G=goal, #=rock):")
    for row in maze:
        print(''.join(row))
    
    # P0: Rocks are already defined in maze as '#'
    print(f"\nP0: Negative space mapped — {sum(row.count('#') for row in maze)} rocks")

    safe_cells = map_safe_channels(maze, start)
    print_safe_channels(maze, safe_cells)
    
    # P2: Optimize within safe channels (Deadband Protocol)

        print_maze_with_path(maze, deadband_path, "P2: Deadband Protocol Path (P0→P1→P2)")
    else:
        print("\nDeadband Protocol: No path found within safe channels")

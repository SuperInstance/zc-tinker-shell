```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Simulates the fleet's constraint theory results.
"""

import random
from typing import List, Tuple, Set

# ========== P0: MAP NEGATIVE SPACE ==========
def identify_rocks(maze: List[str]) -> Set[Tuple[int, int]]:
    """Find all '#' cells — the rocks (failure states)."""
    rocks = set()
    for y, row in enumerate(maze):
        for x, ch in enumerate(row):
            if ch == '#':
                rocks.add((x, y))
    return rocks

# ========== P1: FIND SAFE CHANNELS ==========
def find_safe_channels(
    maze: List[str],
    start: Tuple[int, int],
    goal: Tuple[int, int],
    rocks: Set[Tuple[int, int]]
) -> List[List[Tuple[int, int]]]:
    """
    BFS to find all paths that avoid rocks.
    Returns list of paths (each path is list of coordinates).
    """
    from collections import deque
    
    height = len(maze)
    width = len(maze[0])
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    
    # BFS queue: (x, y, path_so_far)
    queue = deque([(start[0], start[1], [start])])
    safe_paths = []
    
    while queue:
        x, y, path = queue.popleft()
        
        if (x, y) == goal:
            safe_paths.append(path)
            continue
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in rocks and (nx, ny) not in path:
                    queue.append((nx, ny, path + [(nx, ny)]))
    
    return safe_paths

# ========== P2: OPTIMIZE WITHIN CHANNEL ==========
def shortest_safe_path(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """Select the shortest path among safe channels."""
    if not safe_paths:
        return []
    return min(safe_paths, key=len)

# ========== GREEDY P2-ONLY (FOR COMPARISON) ==========
def greedy_p2_only(
    maze: List[str],
    start: Tuple[int, int],
    goal: Tuple[int, int]
) -> List[Tuple[int, int]]:
    """
    Greedy optimization without P0/P1.
    Moves toward goal, ignoring rocks — will crash.
    """
    path = [start]
    x, y = start
    gx, gy = goal
    
    while (x, y) != (gx, gy):
        # Greedy: move in direction that reduces Manhattan distance
        if abs(gx - (x + 1)) + abs(gy - y) < abs(gx - x) + abs(gy - y):
            x += 1
        elif abs(gx - (x - 1)) + abs(gy - y) < abs(gx - x) + abs(gy - y):
            x -= 1
        elif abs(gx - x) + abs(gy - (y + 1)) < abs(gx - x) + abs(gy - y):
            y += 1
        else:
            y -= 1
        
        # Check bounds
        if not (0 <= x < len(maze[0]) and 0 <= y < len(maze)):
            break
        
        path.append((x, y))
        
        # Hit a rock?
        if maze[y][x] == '#':
            break
    
    return path

# ========== VISUALIZATION ==========
def print_maze_with_path(maze: List[str], path: List[Tuple[int, int]], title: str):
    """Display maze with path overlay."""
    maze_copy = [list(row) for row in maze]
    for (x, y) in path[1:-1]:  # don't overwrite start/goal
        if maze_copy[y][x] == ' ':
            maze_copy[y][x] = '.'
    if path:
        sx, sy = path[0]
        gx, gy = path[-1]
        maze_copy[sy][sx] = 'S'
        maze_copy[gy][gx] = 'G'
    
    print(f"\n=== {title} ===")
    for row in maze_copy:
        print(''.join(row))
    print(f"Path length: {len(path)} steps")
    print(f"Success: {'Yes' if path and path[-1] == (gx, gy) else 'No'}")

# ========== MAIN DEMO ==========
def main():
    # Example maze from constraint_sim_final.json (20x20 simplified)
    maze = [
        "####################",
        "#S                 #",
        "# ###### ##########",
        "# #    # #        #",
        "# # ## # # ###### #",
        "# #  # # # #    # #",
        "# ## # # # # ## # #",
        "#  # # # # #  # # #",
        "## # # # # ## # # #",
        "#  # # # #  # # # #",
        "# ## # # # ## # # #",
        "#  # # # #  # # # #",
        "# ## # # # ## # # #",
        "#  # # # #  # # # #",
        "# ## # # # ## # # #",
        "#  # # # #  # # # #",
        "# ## # # # ## # # #",
        "#  # # # #  # #   G#",
        "#                  #",
        "####################"
    ]
    
    start = (1, 1)
    goal = (18, 17)
    
    print("DEADBAND PROTOCOL DEMO")
    print("Maze: 20x20, '#' = rock, 'S' = start, 'G' = goal")
    
    # ----- P0: Identify rocks -----
    rocks = identify_rocks(maze)
    print(f"\nP0: Mapped {len(rocks)} rocks (negative space)")
    
    # ----- P1: Find safe channels -----
    safe_paths = find_safe_channels(maze, start, goal, rocks)
    print(f"P1: Found {len(safe_paths)} safe channels (deadbands)")
    
    # ----- P2: Optimize within safe channels -----
    if safe_paths:
        optimal_path = shortest_safe_path(safe_paths)
        print_maze_with_path(maze, optimal_path, "Deadband Protocol (P0→P1→P2)")
    else:
        print("No safe channels found.")
    
    # ----- Greedy P2-only (for comparison) -----
    greedy_path = greedy_p2_only(maze, start, goal)
    print_maze_with_path(maze, greedy_path, "Greedy P2-Only (Violation)")
    
    # ----- Statistics -----
    print("\n=== SIMULATION RESULTS ===")
    print("Deadband Protocol (P0→P1→P2):")
    print(f"  Success: {'Yes' if optimal_path and optimal_path[-1] == goal else 'No'}")
    print(f"  Steps: {len(optimal_path) if optimal_path else 'N/A'}")
    print("\nGreedy P2-Only:")
    print(f"  Success: {'Yes' if greedy_path and greedy_path[-1] == goal else 'No'}")
    print(f"  Steps: {len(greedy_path)}")
    print(f"  Hit rock: {'Yes' if any(maze[y][x] == '#' for (x, y) in greedy_path) else 'No'}")

if __name__ == "__main__":
    main()
```

Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 in a simple ASCII maze.
Based on fleet-knowledge/constraint_sim_final.json and DEADBAND-PROTOCOL.md

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 10, height: int = 8, rock_density: float = 0.2):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        for y in range(height):
            for x in range(width):
                if (x, y) != self.start and (x, y) != self.goal:
                    if random.random() < rock_density:
                        self.rocks.add((x, y))
        
        # Ensure start and goal are not rocks
        self.rocks.discard(self.start)
        self.rocks.discard(self.goal)
        
        # Agent state
        self.position = self.start
        self.path = [self.start]
        self.steps = 0
        
    def render(self, show_rocks: bool = True, show_path: bool = True) -> str:
        """Render maze as ASCII"""
        lines = []
        for y in range(self.height):
            line = []
            for x in range(self.width):
                if (x, y) == self.position:
                    line.append('A')  # Agent
                elif (x, y) == self.start:
                    line.append('S')
                elif (x, y) == self.goal:
                    line.append('G')
                elif show_rocks and (x, y) in self.rocks:
                    line.append('#')  # Rock
                elif show_path and (x, y) in self.path:
                    line.append('.')  # Path taken
                else:
                    line.append(' ')
            lines.append(''.join(line))
        return '\n'.join(lines)
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify rocks (what NOT to do)"""
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Found {len(self.rocks)} rocks at positions:")
        rock_list = list(self.rocks)
        for i in range(0, len(rock_list), 5):

        print()
        return self.rocks.copy()
    
    def p1_find_safe_channels(self, rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
        """P1: Find all paths that avoid rocks"""
        print("=== P1: FIND SAFE CHANNELS ===")
        
        # Simple BFS to find safe paths
        from collections import deque
        
        safe_paths = []
        queue = deque([([self.start], set([self.start]))])
        
        while queue and len(safe_paths) < 3:  # Limit to first 3 paths for demo
            path, visited = queue.popleft()
            x, y = path[-1]
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            # Try all 4 directions
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in rocks and (nx, ny) not in visited:
                        new_path = path + [(nx, ny)]
                        new_visited = visited.copy()
                        new_visited.add((nx, ny))
                        queue.append((new_path, new_visited))
        
        print(f"Found {len(safe_paths)} safe channels (rock-free paths)")
        for i, path in enumerate(safe_paths[:3]):  # Show first 3
            print(f"  Channel {i+1}: {len(path)-1} steps")
        print()
        
        return safe_paths
    
    def p2_optimize_within_channel(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """P2: Choose shortest safe path"""
        print("=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        if not safe_paths:
            print("ERROR: No safe channels found! Cannot optimize.")
            return []
        
        # Find shortest path among safe options
        optimal_path = min(safe_paths, key=len)
        print(f"Selected optimal path: {len(optimal_path)-1} steps")
        print(f"Path: {optimal_path}")
        print()
        
        return optimal_path

                nx = x + (1 if gx > x else -1)
                ny = y
            else:
                nx = x
                ny = y + (1 if gy > y else -1)
            
            # Check bounds
            if 0 <= nx < self.width and 0 <= ny < self.height:
                self.position = (nx, ny)
                self.path.append(self.position)
                self.steps += 1
                
                # Check if hit rock
                if self.position in self.rocks:
                    print(f"CRASH! Hit rock at {self.position} after {self.steps} steps")

                print(f"Optimal cost: {len(optimal_path)-1} steps")
                print(self.render(show_rocks=True, show_path=True))
                return True
        
        return False

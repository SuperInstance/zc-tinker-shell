**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Complete.

**Approach:**
1. **P0 (Map Rocks):** Generate a random maze with obstacles ('#') and open cells ('.'). Mark all obstacles as negative space.
2. **P1 (Find Safe Channels):** From start, perform BFS to find all reachable cells that are not rocks. These are the safe channels.
3. **P2 (Optimize Within Channel):** From safe channels, compute shortest path (A*) from start to goal, staying within safe cells.
4. **Visualization:** Print maze at each phase, showing rocks, safe channels, and final path.

**Output:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
A visual ASCII demonstration of the fleet's core navigation rule.
"""

import random
from collections import deque
import heapq

def generate_maze(width=20, height=10, obstacle_density=0.3):
    """Generate random maze with obstacles."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    rocks = []
    for y in range(height):
        for x in range(width):
            if random.random() < obstacle_density:
                maze[y][x] = '#'
                rocks.append((x, y))
    # Ensure start (0,0) and goal (width-1, height-1) are clear
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    return maze, rocks

def print_maze(maze, title=""):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks)."""
    rocks = []
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start=(0,0)):
    """P1: BFS to find all reachable safe cells."""
    height, width = len(maze), len(maze[0])
    safe = set()
    queue = deque([start])
    visited = set([start])
    
    while queue:
        x, y = queue.popleft()
        safe.add((x, y))
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and maze[ny][nx] != '#':
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    return safe

def p2_optimize_path(maze, safe, start=(0,0), goal=None):
    """P2: A* search within safe channels."""
    if goal is None:
        goal = (len(maze[0])-1, len(maze)-1)
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            return path
        
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            neighbor = (x + dx, y + dy)
            nx, ny = neighbor
            # Constraint: must be within safe channels
            if not (0 <= nx < len(maze[0]) and 0 <= ny < len(maze)):
                continue
            if neighbor not in safe:
                continue
                
            tentative_g = g_score[current] + 1
            if neighbor not in g_score or tentative_g < g_score[neighbor]:
                came_from[neighbor] = current
                g_score[neighbor] = tentative_g
                f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return []  # No path found

def visualize_phase(maze, phase_data, phase_symbol):
    """Create a copy of maze with phase visualization."""
    vis = [row[:] for row in maze]
    for x, y in phase_data:
        if vis[y][x] not in ['S', 'G']:
            vis[y][x] = phase_symbol
    return vis

def main():
    print("Deadband Protocol Maze Demo")
    print("P0: Map rocks (negative space)")
    print("P1: Find safe channels (reachable cells)")
    print("P2: Optimize path within channels\n")
    
    # Generate maze
    maze, rocks = generate_maze(width=15, height=8, obstacle_density=0.25)
    print_maze(maze, "Initial Maze")
    
    # P0: Map rocks
    rocks = p0_map_rocks(maze)
    print(f"P0: Found {len(rocks)} rocks (negative space)")
    
    # P1: Find safe channels
    safe = p1_find_safe_channels(maze)
    safe_vis = visualize_phase(maze, safe, 'O')
    print_maze(safe_vis, "P1: Safe Channels (O)")
    print(f"P1: {len(safe)} safe cells identified")
    
    # P2: Optimize path
    path = p2_optimize_path(maze, safe)
    if path:
        path_vis = visualize_phase(maze, path, '*')
        print_maze(path_vis, "P2: Optimal Path (*)")
        print(f"P2: Path length = {len(path)} steps")
        
        # Show combined view
        final = [row[:] for row in maze]
        for x, y in safe:
            if final[y][x] not in ['S', 'G', '#']:
                final[y][x] = '·'
        for x, y in path:
            if final[y][x] not in ['S', 'G']:
                final[y][x] = '*'
        print_maze(final, "Combined View (·=safe, *=path)")
    else:
        print("P2: No path found within safe channels")
    
    # Protocol validation
    goal = (len(maze[0])-1, len(maze)-1)
    if goal in safe and path:
        print("\n✅ Protocol SUCCESS: Goal reached via safe channels")
    else:
        print("\n❌ Protocol FAILED: Goal unreachable or no safe path")

if __name__ == "__main__":
    main()
```

#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo
A simple ASCII simulation of P0→P1→P2 navigation.
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    """A 10x10 grid demonstrating the Deadband Protocol."""
    
    def __init__(self, seed=42):
        random.seed(seed)
        self.width = 10
        self.height = 10
        self.start = (0, 0)
        self.goal = (9, 9)
        
        # Generate random obstacles (rocks)
        self.rocks: Set[Tuple[int, int]] = set()
        num_rocks = 15
        while len(self.rocks) < num_rocks:
            x = random.randint(0, self.width-1)
            y = random.randint(0, self.height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
        
        # Agent state
        self.position = self.start
        self.path = [self.start]
        self.steps = 0
        
    def display(self, show_channels=False, safe_cells=None):
        """Render the maze as ASCII."""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.position:
                    row.append('A')  # Agent
                elif (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif (x, y) in self.rocks:
                    row.append('█')  # Rock
                elif show_channels and safe_cells and (x, y) in safe_cells:
                    row.append('·')  # Safe channel
                else:
                    row.append('.')
            grid.append(' '.join(row))
        return '\n'.join(grid)
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify rocks (what NOT to do)."""
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Found {len(self.rocks)} rocks at positions:")
        for rock in sorted(self.rocks):
            print(f"  {rock}")
        return self.rocks.copy()
    
    def p1_find_safe_channels(self) -> Set[Tuple[int, int]]:
        """P1: Find all cells reachable without hitting rocks."""
        print("\n=== P1: FIND SAFE CHANNELS ===")
        
        # Simple flood fill from start, avoiding rocks
        safe = set()
        stack = [self.start]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in safe:
                continue
            if (x, y) in self.rocks:
                continue
            if not (0 <= x < self.width and 0 <= y < self.height):
                continue
                
            safe.add((x, y))
            
            # Add neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                stack.append((x + dx, y + dy))
        
        print(f"Found {len(safe)} safe cells")
        print("Safe channel visualization:")
        print(self.display(show_channels=True, safe_cells=safe))
        return safe
    
    def p2_optimize_in_channel(self, safe_cells: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
        """P2: Find shortest path to goal within safe channels."""
        print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        # Simple BFS within safe cells
        queue = [(self.start, [self.start])]
        visited = set()
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.goal:
                print(f"Found path of length {len(path)-1} steps")
                return path
            
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Only move to safe neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (nx, ny) in safe_cells and (nx, ny) not in visited:
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        print("WARNING: No safe path to goal found!")
        return [self.start]
    
    def run_greedy_failure(self):
        """Demonstrate what happens when you skip to P2 (greedy optimization)."""
        print("\n" + "="*50)
        print("GREEDY FAILURE MODE (Skipping P0+P1)")
        print("="*50)
        
        position = self.start
        path = [position]
        steps = 0
        
        while position != self.goal and steps < 50:
            # Greedy: move toward goal without checking rocks
            x, y = position
            gx, gy = self.goal
            
            # Choose direction that reduces Manhattan distance
            if abs(gx - (x+1)) + abs(gy - y) < abs(gx - x) + abs(gy - y):
                nx, ny = x+1, y
            elif abs(gx - (x-1)) + abs(gy - y) < abs(gx - x) + abs(gy - y):
                nx, ny = x-1, y
            elif abs(gx - x) + abs(gy - (y+1)) < abs(gx - x) + abs(gy - y):
                nx, ny = x, y+1
            else:
                nx, ny = x, y-1
            
            # Check bounds
            if not (0 <= nx < self.width and 0 <= ny < self.height):
                break
                
            position = (nx, ny)
            path.append(position)
            steps += 1
            
            # Check for rock collision
            if position in self.rocks:
                print(f"CRASH! Hit rock at {position} after {steps} steps")
                print("Greedy optimization fails without P0 rock mapping")
                return False, steps
        
        success = position == self.goal
        print(f"Greedy result: {'Success' if success else 'Failed'} in {steps} steps")
        return success, steps
    
    def run_deadband_protocol(self):
        """Execute full P0→P1→P2 protocol."""
        print("\n" + "="*50)
        print("DEADBAND PROTOCOL EXECUTION")
        print("="*50)
        
        print("Initial maze:")
        print(self.display())
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        
        # P1: Find safe channels
        safe = self.p1_find_safe_channels()
        
        # P2: Optimize within channel
        optimal_path = self.p2_optimize_in_channel(safe)
        
        # Animate the path
        print("\n=== EXECUTING OPTIMAL PATH ===")
        for i, pos in enumerate(optimal_path):
            self.position = pos
            print(f"\nStep {i}:")
            print(self.display())
            
            if pos in self.rocks:
                print(f"ERROR: Stepped on rock at {pos}!")
                return False, i
            
            if pos == self.goal:
                print(f"SUCCESS: Reached goal in {i} steps!")
                return True, i
        
        return False, len(optimal_path)

def main():

    print("DEADBAND PROTOCOL VISUAL DEMO")
    print("="*50)
    
    # Create maze
    maze = DeadbandMaze(seed=42)
    
    # Run greedy failure mode
    greedy_success, greedy_steps = maze.run_greedy_failure()
    
    # Run deadband protocol
    deadband_success, deadband_steps = maze.run_deadband_protocol()
    
    # Summary
    print("\n" + "="*50)
    print("RESULTS SUMMARY")
    print("="*50)

    print(f"Deadband (P0→P1→P2): {'Success' if deadband_success else 'FAILED'} in {deadband_steps} steps")

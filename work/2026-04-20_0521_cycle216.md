**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on the simulation data in `constraint_sim_final.json` and the formal protocol in `DEADBAND-PROTOCOL.md`, I will implement a concrete demonstration. The simulation shows deadband success rate of 50/50 vs. 0/50 for constrained-only, so the script must visually show the difference.

**Approach:**
1. **P0 (Map negative space):** Generate a 20x20 maze with random obstacles (rocks). Mark them.
2. **P1 (Find safe channels):** Use BFS from start to goal, avoiding rocks, to find all reachable safe cells.
3. **P2 (Optimize within channel):** Run A* on the safe cells only to find shortest path.
4. **Contrast:** Show what happens if you skip P0+P1 and just run greedy (P2-only) — it hits rocks.

**Output:** A Python script `deadband_maze_demo.py` that runs the protocol and prints ASCII visualizations.

"""
deadband_maze_demo.py

Demonstrates the Deadband Protocol (P0→P1→P2) on a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
and DEADBAND-PROTOCOL.md.

Simulates:
- P2-only (greedy, unconstrained): fails by hitting rocks.
- P0+P1+P2 (deadband): succeeds by staying in safe channels.
"""

import random
from collections import deque
import heapq

# Constants from simulation data
MAZE_SIZE = 20
START = (0, 0)
GOAL = (19, 19)
ROCK_COUNT = 64  # matches "rocks": 64 in constraint_sim_final.json

def generate_maze():
    """Create a 20x20 maze with random rocks (obstacles)."""
    maze = [['.' for _ in range(MAZE_SIZE)] for __ in range(MAZE_SIZE)]
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        r = random.randint(0, MAZE_SIZE-1)
        c = random.randint(0, MAZE_SIZE-1)
        if (r, c) != START and (r, c) != GOAL:
            rocks.add((r, c))
            maze[r][c] = '#'
    maze[START[0]][START[1]] = 'S'
    maze[GOAL[0]][GOAL[1]] = 'G'
    return maze, rocks

def print_maze(maze, path=None, safe_cells=None):
    """Print maze with optional path and safe cells overlay."""
    for r in range(MAZE_SIZE):
        row = ''
        for c in range(MAZE_SIZE):
            cell = maze[r][c]
            if path and (r, c) in path:
                row += '*'
            elif safe_cells and (r, c) in safe_cells:
                row += '~'
            else:
                row += cell
        print(row)

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks)."""
    rocks = []
    for r in range(MAZE_SIZE):
        for c in range(MAZE_SIZE):
            if maze[r][c] == '#':
                rocks.append((r, c))
    return rocks

def p1_find_safe_channels(maze, start, goal):
    """P1: Find all cells reachable from start without hitting rocks."""
    safe = set()
    queue = deque([start])
    visited = set([start])
    dirs = [(1,0),(-1,0),(0,1),(0,-1)]
    
    while queue:
        r, c = queue.popleft()
        safe.add((r, c))
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if (nr, nc) not in visited and maze[nr][nc] != '#':
                    visited.add((nr, nc))
                    queue.append((nr, nc))
    return safe

def p2_optimize_within_channel(maze, safe_cells, start, goal):
    """P2: A* search limited to safe cells only."""
    def heuristic(a, b):
        return abs(a[0]-b[0]) + abs(a[1]-b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    dirs = [(1,0),(-1,0),(0,1),(0,-1)]
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            path.reverse()
            return path
        
        for dr, dc in dirs:
            nr, nc = current[0] + dr, current[1] + dc
            neighbor = (nr, nc)
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if neighbor not in safe_cells:
                    continue  # Not in safe channel → skip
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    return []  # No path found within safe channels

def p2_only_greedy(maze, start, goal):
    """P2-only (violation): greedy move toward goal ignoring rocks."""
    path = [start]
    current = start
    dirs = [(1,0),(0,1)]  # Only move down/right (simplified greedy)
    
    while current != goal:
        best = None
        best_dist = float('inf')
        for dr, dc in dirs:
            nr, nc = current[0] + dr, current[1] + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                dist = abs(nr - goal[0]) + abs(nc - goal[1])
                if dist < best_dist:
                    best_dist = dist
                    best = (nr, nc)
        if best is None:
            break
        if maze[best[0]][best[1]] == '#':
            path.append(best)
            return path, 'HIT_ROCK'
        path.append(best)
        current = best
    return path, 'SUCCESS' if current == goal else 'STUCK'

**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on the simulation data in `constraint_sim_final.json` and the formal protocol in `DEADBAND-PROTOCOL.md`, I will implement a concrete demonstration. The simulation shows deadband success rate of 50/50 vs. 0/50 for constrained-only, so the script must visually enforce the strict P0→P1→P2 priority.

**Approach:**
1. **P0 (Map negative space):** Generate a random 10x10 maze with rocks (`#`), start (`S`), goal (`G`). Identify all rock coordinates.
2. **P1 (Identify safe channels):** Use BFS from start to find all reachable cells that avoid rocks. These are the safe channels.
3. **P2 (Optimize within channel):** Run A* search *only within safe channels* to find shortest path.
4. **Visualization:** Print maze at each stage, highlight safe channels, show final path.

**Output:** A standalone Python script that runs the demonstration.

## File: `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 strict priority in pathfinding.
Based on Cocapn fleet's Deadband Protocol v1.0.
"""

import random
from collections import deque
import heapq

def generate_maze(width=10, height=10, rock_density=0.2):
    """Generate a random maze with rocks, start, and goal."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks randomly
    rocks = []
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
                rocks.append((x, y))
    
    # Ensure start and goal are not rocks
    start = (0, 0)
    while start in rocks:
        start = (random.randint(0, width-1), random.randint(0, height-1))
    goal = (width-1, height-1)
    while goal in rocks or goal == start:
        goal = (random.randint(0, width-1), random.randint(0, height-1))
    
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    
    return maze, start, goal, rocks

def print_maze(maze, title="Maze"):
    """Print maze with borders."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' ' + ' '.join(row))
    print()

def p0_map_negative_space(maze, rocks):
    """P0: Identify rocks (negative space)."""
    print("P0: MAP NEGATIVE SPACE")
    print(f"Identified {len(rocks)} rock cells (coordinates: {rocks[:5]}...)" if rocks else "No rocks found.")
    return rocks

def p1_find_safe_channels(maze, start, rocks):
    """P1: Find all cells reachable without hitting rocks (safe channels)."""
    width, height = len(maze[0]), len(maze)
    safe = set()
    queue = deque([start])
    visited = set([start])
    
    while queue:
        x, y = queue.popleft()
        safe.add((x, y))
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and maze[ny][nx] != '#':
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    
    print(f"P1: IDENTIFY SAFE CHANNELS")
    print(f"Found {len(safe)} safe cells (including start).")
    
    # Visualize safe channels
    viz = [row[:] for row in maze]
    for x, y in safe:
        if viz[y][x] not in ('S', 'G'):
            viz[y][x] = '~'  # Safe water symbol
    print_maze(viz, "Safe Channels (P1)")
    
    return safe

def p2_optimize_within_channel(maze, start, goal, safe):
    """P2: Find shortest path within safe channels using A*."""
    if goal not in safe:
        print("P2: Goal is not in safe channels! Cannot proceed.")
        return []
    
    width, height = len(maze[0]), len(maze)
    
    # Manhattan heuristic
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            print(f"P2: OPTIMIZE WITHIN CHANNEL")
            print(f"Found path length {len(path)}")
            return path
        
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            neighbor = (nx, ny)
            if 0 <= nx < width and 0 <= ny < height:
                if neighbor not in safe:
                    continue  # Not in safe channel → skip
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    print("P2: No path found within safe channels.")
    return []

def visualize_final(maze, path, start, goal):
    """Show final maze with path."""
    viz = [row[:] for row in maze]
    for x, y in path:
        if (x, y) not in (start, goal):
            viz[y][x] = '*'  # Path marker
    
    print_maze(viz, "Final Path (P2)")
    print(f"Path coordinates: {path}")

def deadband_protocol_demo():
    """Run the full Deadband Protocol demonstration."""
    print("=" * 50)
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("P0 → P1 → P2 (strict priority)")
    print("=" * 50)
    
    # Initialize
    random.seed(42)  # Deterministic for demo
    maze, start, goal, rocks = generate_maze(width=10, height=10, rock_density=0.2)
    print_maze(maze, "Initial Maze")
    print(f"Start: {start}, Goal: {goal}")
    
    # P0: Map negative space
    rocks = p0_map_negative_space(maze, rocks)
    
    # P1: Find safe channels
    safe = p1_find_safe_channels(maze, start, rocks)
    
    # P2: Optimize within channel
    path = p2_optimize_within_channel(maze, start, goal, safe)
    
    # Results
    if path:
        visualize_final(maze, path, start, goal)
        print("SUCCESS: Path found following Deadband Protocol.")
    else:
        print("FAILURE: No path possible within safe channels.")
    
    # Protocol reminder
    print("\n" + "=" * 50)
    print("PROTOCOL REMINDER:")
    print("P0: Identify rocks (what NOT to do)")
    print("P1: Map safe channels (where you CAN be)")
    print("P2: Optimize within channels (best path)")
    print("Violation: Optimizing before mapping → hit rocks.")
    print("=" * 50)

if __name__ == "__main__":
    deadband_protocol_demo()
```

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Optimal path length: 21, Rocks: 64, Deadband success: 50/50
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self.rocks: Set[Tuple[int, int]] = set()
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density and (x, y) not in [self.start, self.goal]:
                    self.rocks.add((x, y))
        
        # Ensure path exists by clearing a channel
        self._ensure_safe_channel()
    
    def _ensure_safe_channel(self):
        """P1: Guarantee at least one safe channel exists"""
        x, y = 0, 0
        path = []
        while (x, y) != self.goal:
            path.append((x, y))
            if (x, y) in self.rocks:
                self.rocks.remove((x, y))
            if x < self.goal[0]:
                x += 1
            elif y < self.goal[1]:
                y += 1
    
    def display(self, path: List[Tuple[int, int]] = None, show_channels: bool = False):
        """Render ASCII maze with optional path and safe channels"""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')  # Rock
                elif path and pos in path:
                    row.append('·')  # Path marker
                elif show_channels and self._is_safe(pos):
                    row.append('░')  # Safe channel
                else:
                    row.append(' ')
            grid.append(''.join(row))
        return '\n'.join(grid)
    
    def _is_safe(self, pos: Tuple[int, int]) -> bool:
        """Check if position is in safe channel (not a rock)"""
        return pos not in self.rocks
    
    def greedy_search(self) -> Tuple[bool, List[Tuple[int, int]], int]:
        """P2-only (unconstrained optimization) - Hits rocks"""
        path = [self.start]
        visited = set([self.start])
        steps = 0
        
        current = self.start
        while current != self.goal and steps < 1000:
            x, y = current
            
            # Greedy: move toward goal
            moves = []
            if x < self.goal[0]:
                moves.append((x+1, y))
            if y < self.goal[1]:
                moves.append((x, y+1))
            
            # Random tie-breaker
            random.shuffle(moves)
            
            found = False
            for move in moves:
                if move not in visited:
                    current = move
                    path.append(current)
                    visited.add(current)
                    found = True
                    break
            
            if not found:
                break
            
            steps += 1
        
        success = current == self.goal
        return success, path, steps
    
    def deadband_search(self) -> Tuple[bool, List[Tuple[int, int]], int]:
        """P0→P1→P2: Full Deadband Protocol"""
        # P0: Already mapped rocks in __init__
        
        # P1: Identify safe channels
        safe_channels = self._find_safe_channels()
        
        # P2: Optimize within safe channels
        return self._optimize_in_channels(safe_channels)
    
    def _find_safe_channels(self) -> Set[Tuple[int, int]]:
        """P1: Find all positions that are not rocks"""
        channels = set()
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) not in self.rocks:
                    channels.add((x, y))
        return channels
    
    def _optimize_in_channels(self, channels: Set[Tuple[int, int]]) -> Tuple[bool, List[Tuple[int, int]], int]:
        """P2: A* search constrained to safe channels"""
        from heapq import heappush, heappop
        
        def heuristic(a, b):
            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        open_set = []
        heappush(open_set, (0, self.start))
        
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: heuristic(self.start, self.goal)}
        
        steps = 0
        
        while open_set:
            _, current = heappop(open_set)
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                path.reverse()
                return True, path, steps
            
            x, y = current
            neighbors = [(x+1, y), (x-1, y), (x, y+1), (x, y-1)]
            
            for neighbor in neighbors:
                nx, ny = neighbor
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if neighbor not in channels:
                        continue  # Skip rocks (constrained to safe channels)
                    
                    tentative_g = g_score[current] + 1
                    
                    if neighbor not in g_score or tentative_g < g_score[neighbor]:
                        came_from[neighbor] = current
                        g_score[neighbor] = tentative_g
                        f_score[neighbor] = tentative_g + heuristic(neighbor, self.goal)
                        heappush(open_set, (f_score[neighbor], neighbor))
            
            steps += 1
            if steps > 1000:
                break
        
        return False, [], steps

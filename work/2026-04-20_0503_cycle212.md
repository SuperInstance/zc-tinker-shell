```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Simulates P0→P1→P2 priority in a 20x20 maze.
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    """Demonstrates Deadband Protocol in a maze navigation problem."""
    
    def __init__(self, width=20, height=20, rock_density=0.16):
        """
        Initialize maze based on constraint_sim_final.json:
        - optimal: 21 steps (shortest path)
        - rocks: 64 (20x20 * 0.16 = 64)
        - deadband_success: 50/50 (100%)
        """
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self.rocks = self._generate_rocks(rock_density)
        
        # P1: Safe channels (deadbands)
        self.safe_channels = None
        
        # P2: Optimal path
        self.optimal_path = None
        
        # Statistics
        self.stats = {
            'p0_rocks_mapped': len(self.rocks),
            'p1_channels_found': 0,
            'p2_path_length': 0,
            'success': False
        }
    
    def _generate_rocks(self, density: float) -> Set[Tuple[int, int]]:
        """Generate random obstacles (rocks) avoiding start and goal."""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        
        rocks = set()
        while len(rocks) < num_rocks:
            x = random.randint(0, self.width-1)
            y = random.randint(0, self.height-1)
            pos = (x, y)
            
            # Don't block start or goal
            if pos != self.start and pos != self.goal:
                rocks.add(pos)
        
        return rocks
    
    def p0_map_negative_space(self) -> None:
        """P0: Identify what NOT to do (rocks)."""
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Identified {len(self.rocks)} rocks (failure states)")
        print("Do NOT proceed to P1 without this map.")
    
    def p1_find_safe_channels(self) -> bool:
        """P1: Find paths that avoid all rocks (safe channels)."""
        print("\n=== P1: FIND SAFE CHANNELS ===")
        
        # BFS to find any safe path from start to goal
        queue = [(self.start, [self.start])]
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.pop(0)
            
            # Check if we reached goal
            if (x, y) == self.goal:
                self.safe_channels = [path]  # Found at least one safe channel
                self.stats['p1_channels_found'] = 1
                print(f"Found safe channel: {len(path)-1} steps")
                return True
            
            # Explore neighbors (up, down, left, right)
            for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                nx, ny = x + dx, y + dy
                
                # Check bounds
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    new_pos = (nx, ny)
                    
                    # P1 constraint: MUST avoid rocks
                    if new_pos in self.rocks:
                        continue  # This is a rock, not a safe channel
                    
                    if new_pos not in visited:
                        visited.add(new_pos)
                        queue.append((new_pos, path + [new_pos]))
        
        # No safe path found
        print("No safe channels found! Agent would hit rocks.")
        return False
    
    def p2_optimize_within_channel(self) -> bool:
        """P2: Find shortest path within safe channels."""
        print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        if not self.safe_channels:
            print("VIOLATION: Cannot optimize (P2) without safe channels (P1)!")
            return False
        
        # A* search for optimal path avoiding rocks
        open_set = {self.start}
        came_from = {}
        
        g_score = {self.start: 0}
        f_score = {self.start: self._heuristic(self.start)}
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score.get(pos, float('inf')))
            
            if current == self.goal:
                # Reconstruct optimal path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                path.reverse()
                
                self.optimal_path = path
                self.stats['p2_path_length'] = len(path) - 1
                self.stats['success'] = True
                
                print(f"Optimal path found: {len(path)-1} steps")
                print(f"Theoretical optimal: 21 steps (from constraint_sim_final.json)")
                return True
            
            open_set.remove(current)
            
            for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                nx, ny = current[0] + dx, current[1] + dy
                neighbor = (nx, ny)
                
                # P0+P1 constraints: Must be in bounds and not a rock
                if not (0 <= nx < self.width and 0 <= ny < self.height):
                    continue
                if neighbor in self.rocks:
                    continue
                
                tentative_g = g_score[current] + 1
                
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor)
                    open_set.add(neighbor)
        
        return False
    
    def _heuristic(self, pos: Tuple[int, int]) -> int:
        """Manhattan distance heuristic."""
        return abs(pos[0] - self.goal[0]) + abs(pos[1] - self.goal[1])
    
    def run_protocol(self) -> bool:
        """Execute full Deadband Protocol with strict priority."""
        print("=" * 50)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print(f"Maze: {self.width}x{self.height}, Rocks: {len(self.rocks)}")
        print("=" * 50)
        
        # P0: Map negative space (MANDATORY FIRST STEP)
        self.p0_map_negative_space()
        
        # P1: Find safe channels (MUST complete before P2)
        if not self.p1_find_safe_channels():
            print("\nPROTOCOL VIOLATION PREVENTED: Would have hit rocks!")
            return False
        
        # P2: Optimize within safe channels (ONLY after P0+P1)
        if not self.p2_optimize_within_channel():
            return False
        
        self.print_results()
        return True
    
    def print_maze(self, show_path=False) -> None:
        """Print ASCII visualization of maze."""
        print("\n" + "=" * 50)
        print("MAZE VISUALIZATION")
        print("S = Start, G = Goal, # = Rock, · = Safe, * = Path")
        print("=" * 50)
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('#')
                elif show_path and self.optimal_path and pos in self.optimal_path:
                    row.append('*')
                else:
                    row.append('·')
            
            print(' '.join(row))
    
    def print_results(self) -> None:
        """Print simulation results matching constraint_sim_final.json format."""
        print("\n" + "=" * 50)
        print("SIMULATION RESULTS")
        print("=" * 50)
        
        # Match the JSON structure from fleet knowledge

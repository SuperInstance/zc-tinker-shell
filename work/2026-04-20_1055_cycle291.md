#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Simulates the 20×20 maze from constraint_sim_final.json
Shows P2-only (greedy) vs P0+P1+P2 (deadband) navigation.
"""

import random
import time
from typing import List, Tuple, Set

# Constants from simulation
MAZE_SIZE = 20
ROCK_COUNT = 64  # 16% density
START = (0, 0)
GOAL = (19, 19)

class Maze:
    def __init__(self):
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_rocks()
        self.path = []
        self.steps = 0
        
    def generate_rocks(self):
        """Generate random rocks, avoiding start and goal."""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != START and (x, y) != GOAL:
                self.rocks.add((x, y))
    
    def is_rock(self, x: int, y: int) -> bool:
        return (x, y) in self.rocks
    
    def is_valid(self, x: int, y: int) -> bool:
        return 0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE
    
    def display(self, path: List[Tuple[int, int]] = None):
        """Display maze with ASCII art."""
        print("\n" + "="*60)
        print("MAZE (20×20) - '#' = rock, 'S' = start, 'G' = goal")
        print("="*60)
        
        for y in range(MAZE_SIZE):
            row = []
            for x in range(MAZE_SIZE):
                pos = (x, y)
                if pos == START:
                    row.append('S')
                elif pos == GOAL:
                    row.append('G')
                elif self.is_rock(x, y):
                    row.append('#')
                elif path and pos in path:
                    row.append('·')
                else:
                    row.append('.')
            print(' '.join(row))
        print()

def p2_only_greedy(maze: Maze) -> Tuple[bool, List[Tuple[int, int]]]:
    """P2-only: Greedy optimization toward goal, ignoring rocks."""
    x, y = START
    path = [(x, y)]
    visited = set(path)
    
    while (x, y) != GOAL:
        # Greedy: move toward goal (right/down priority)
        moves = []
        if x < GOAL[0]:
            moves.append((x+1, y))
        if y < GOAL[1]:
            moves.append((x, y+1))
        if x > GOAL[0]:
            moves.append((x-1, y))
        if y > GOAL[1]:
            moves.append((x, y-1))
        
        # Shuffle to add some randomness
        random.shuffle(moves)
        
        moved = False
        for nx, ny in moves:
            if maze.is_valid(nx, ny) and (nx, ny) not in visited:
                x, y = nx, ny
                path.append((x, y))
                visited.add((x, y))
                moved = True
                break
        
        if not moved:
            # Stuck
            return False, path
        
        # Check if hit rock
        if maze.is_rock(x, y):
            return False, path
    
    return True, path

def deadband_protocol(maze: Maze) -> Tuple[bool, List[Tuple[int, int]]]:
    """Full Deadband Protocol: P0→P1→P2."""
    # P0: Map negative space (rocks)
    rocks = maze.rocks  # Already known
    
    # P1: Identify safe channels (BFS avoiding rocks)
    from collections import deque
    
    queue = deque([START])
    parent = {START: None}
    visited = set([START])
    
    while queue:
        x, y = queue.popleft()
        
        if (x, y) == GOAL:
            break
        
        # Four possible moves
        for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
            nx, ny = x + dx, y + dy
            if maze.is_valid(nx, ny):
                # Check P0: not a rock
                if (nx, ny) in rocks:
                    continue
                # P1: safe channel
                if (nx, ny) not in visited:
                    visited.add((nx, ny))
                    parent[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    
    # Reconstruct path if goal reached
    if GOAL not in parent:
        return False, []
    
    # P2: Optimize within safe channel (shortest path)
    path = []
    current = GOAL
    while current is not None:
        path.append(current)
        current = parent[current]
    path.reverse()
    
    return True, path

def run_demo():
    """Run the demonstration."""
    print("\n" + "="*60)
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("Based on constraint_sim_final.json simulation")
    print("="*60)
    
    # Create maze
    random.seed(42)  # Deterministic for demo
    maze = Maze()
    
    print("\n1. MAZE GENERATED")
    print(f"   Size: {MAZE_SIZE}×{MAZE_SIZE}")
    print(f"   Rocks: {len(maze.rocks)} (16% density)")
    print(f"   Start: {START}, Goal: {GOAL}")
    
    # Show initial maze
    maze.display()
    
    print("\n2. P2-ONLY (GREEDY OPTIMIZATION)")
    print("   Strategy: Move toward goal, ignore rocks")
    print("   Expected: Hits rocks, fails")
    time.sleep(1)
    
    success1, path1 = p2_only_greedy(maze)
    print(f"   Result: {'SUCCESS' if success1 else 'FAILED'}")
    print(f"   Path length: {len(path1)}")
    maze.display(path1)
    
    if not success1 and path1:
        last_pos = path1[-1]
        if maze.is_rock(*last_pos):
            print(f"   Failed at rock position: {last_pos}")
    
    print("\n3. DEADBAND PROTOCOL (P0→P1→P2)")
    print("   P0: Map rocks (negative space)")
    print("   P1: Find safe channels (BFS avoiding rocks)")
    print("   P2: Shortest path within safe channels")
    time.sleep(1)
    
    success2, path2 = deadband_protocol(maze)
    print(f"   Result: {'SUCCESS' if success2 else 'FAILED'}")
    print(f"   Path length: {len(path2)} (optimal: 39 steps)")
    maze.display(path2)
    
    # Summary
    print("\n" + "="*60)
    print("SUMMARY")
    print("="*60)
    print(f"P2-only (greedy): {'FAILED' if not success1 else 'SUCCESS'}")
    print(f"Deadband Protocol: {'FAILED' if not success2 else 'SUCCESS'}")
    
    if success2:
        print(f"\nOptimal path length: {len(path2)-1} steps")
        print("Path:", " → ".join([f"({x},{y})" for x, y in path2[:5]]), "...")
    
    # Compare to simulation results
    print("\n" + "="*60)
    print("SIMULATION DATA (from constraint_sim_final.json)")
    print("="*60)
    print("Unconstrained (P2-only): 0/50 success")
    print("Deadband Protocol: 50/50 success")
    print("Optimal path: 39 steps")
    print("\nThis demo shows one instance of the pattern.")

if __name__ == "__main__":
    run_demo()

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 protocol with visual ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json simulation.
"""

import random
from collections import deque
import time

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        """
        Initialize maze with parameters from constraint_sim_final.json
        optimal=21, rocks=64 (≈16% of 400 cells)
        """
        self.width = width
        self.height = height
        self.total_cells = width * height
        self.rock_density = rock_density
        self.num_rocks = int(self.total_cells * rock_density)
        
        # Grid: 0=empty, 1=rock, 2=agent, 3=goal, 4=path, 5=safe channel
        self.grid = [[0 for _ in range(width)] for _ in range(height)]
        
        # Positions
        self.agent_pos = None
        self.goal_pos = None
        self.rocks = []
        
        # Protocol state
        self.p0_map = None  # Negative space (rocks)
        self.p1_channels = None  # Safe channels
        self.p2_path = None  # Optimized path
        
        self._generate_maze()
    
    def _generate_maze(self):
        """Generate random maze with rocks, agent, and goal."""
        # Place rocks
        rock_positions = random.sample(range(self.total_cells), self.num_rocks)
        for pos in rock_positions:
            y, x = divmod(pos, self.width)
            self.grid[y][x] = 1
            self.rocks.append((y, x))
        
        # Place agent (top-left quadrant)
        while True:
            y = random.randint(0, self.height//2 - 1)
            x = random.randint(0, self.width//2 - 1)
            if self.grid[y][x] == 0:
                self.agent_pos = (y, x)
                self.grid[y][x] = 2
                break
        
        # Place goal (bottom-right quadrant)
        while True:
            y = random.randint(self.height//2, self.height - 1)
            x = random.randint(self.width//2, self.width - 1)
            if self.grid[y][x] == 0:
                self.goal_pos = (y, x)
                self.grid[y][x] = 3
                break
    
    def p0_map_negative_space(self):
        """P0: Map the negative space (rocks)."""
        print("\n" + "="*60)
        print("P0: MAP NEGATIVE SPACE")
        print("="*60)
        print(f"Identified {len(self.rocks)} rocks (obstacles)")
        print("Rocks are absolute constraints - cannot pass through")
        
        self.p0_map = set(self.rocks)
        return self.p0_map
    
    def p1_find_safe_channels(self):
        """P1: Find safe channels (cells not in negative space)."""
        print("\n" + "="*60)
        print("P1: FIND SAFE CHANNELS")
        print("="*60)
        
        # Safe channels = all cells that aren't rocks
        safe_cells = []
        for y in range(self.height):
            for x in range(self.width):
                if self.grid[y][x] != 1:  # Not a rock
                    safe_cells.append((y, x))
        
        self.p1_channels = set(safe_cells)
        print(f"Found {len(self.p1_channels)} safe cells")
        print(f"Safe coverage: {len(self.p1_channels)/self.total_cells*100:.1f}% of maze")
        
        # Mark safe channels in grid for visualization
        for y, x in safe_cells:
            if self.grid[y][x] == 0:  # Only mark empty cells
                self.grid[y][x] = 5
        
        return self.p1_channels
    
    def p2_optimize_within_channels(self):
        """P2: Find optimal path through safe channels using BFS."""
        print("\n" + "="*60)
        print("P2: OPTIMIZE WITHIN SAFE CHANNELS")
        print("="*60)
        
        if not self.p1_channels:
            print("ERROR: Must complete P1 first!")
            return None
        
        # BFS to find shortest path
        queue = deque([(self.agent_pos, [self.agent_pos])])
        visited = set([self.agent_pos])
        
        # Directions: up, down, left, right
        directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]
        
        while queue:
            (y, x), path = queue.popleft()
            
            # Check if we reached goal
            if (y, x) == self.goal_pos:
                self.p2_path = path
                print(f"Found optimal path: {len(path)-1} steps")
                
                # Mark path in grid (avoid overwriting agent/goal)
                for py, px in path[1:-1]:  # Exclude start and end
                    if self.grid[py][px] == 5:  # Only overwrite safe channel markers
                        self.grid[py][px] = 4
                return path
            
            # Explore neighbors
            for dy, dx in directions:
                ny, nx = y + dy, x + dx
                
                # Check bounds and constraints
                if (0 <= ny < self.height and 0 <= nx < self.width and
                    (ny, nx) in self.p1_channels and  # Must be in safe channels
                    (ny, nx) not in visited):
                    
                    visited.add((ny, nx))
                    queue.append(((ny, nx), path + [(ny, nx)]))
        
        print("No path found within safe channels!")
        return None

            0: ' ',   # Empty
            1: '█',   # Rock
            2: 'A',   # Agent
            3: 'G',   # Goal
            4: '·',   # Path
            5: '·',   # Safe channel

                cell = self.grid[y][x]
                # Override for agent/goal if they're on path/safe cells
                if (y, x) == self.agent_pos:
                    row += "A "
                elif (y, x) == self.goal_pos:
                    row += "G "
                else:
                    row += symbols.get(cell, '?') + " "
            row = row.rstrip() + "│"
            print(row)
        
        # Bottom border
        print("└" + "─" * (self.width * 2 - 1) + "┘")
        
        # Legend
        print("\nLegend: A=Agent, G=Goal, █=Rock, ·=Safe/Path")

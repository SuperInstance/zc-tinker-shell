"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import random
from typing import List, Tuple, Set
from collections import deque

# Constants from simulation
MAZE_SIZE = 20
ROCK_COUNT = 64  # Rocks (obstacles)
OPTIMAL_PATH_LENGTH = 21  # Optimal path length

# Symbols
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL = 'G'
PATH = '○'
SAFE = ' '

class Maze:
    def __init__(self, seed=42):
        random.seed(seed)
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        
        # Place rocks (avoid start and goal)
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[y][x] = ROCK
    
    def display(self, agent_pos=None, path=None, safe_channels=None):
        """Display maze with optional agent, path, and safe channels."""
        for y in range(MAZE_SIZE):
            row = []
            for x in range(MAZE_SIZE):
                if (x, y) == agent_pos:
                    row.append(AGENT)
                elif (x, y) == self.goal:
                    row.append(GOAL)
                elif path and (x, y) in path:
                    row.append(PATH)
                elif safe_channels and (x, y) in safe_channels:
                    row.append(SAFE)
                else:
                    row.append(self.grid[y][x])
            print(''.join(row))
        print()

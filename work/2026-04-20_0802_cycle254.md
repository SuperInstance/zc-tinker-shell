#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 254)
A visual ASCII demonstration of P0→P1→P2 protocol.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time
from typing import List, Tuple, Set

class DeadbandMaze:
    """Implements Deadband Protocol in a 10x10 maze"""
    
    def __init__(self, seed=42):
        random.seed(seed)
        self.width = 10
        self.height = 10
        self.start = (0, 0)
        self.goal = (9, 9)
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
        # Generate rocks (P0 negative space)
        self._generate_rocks()
        
    def _generate_rocks(self):
        """P0: Map negative space (what NOT to do)"""
        # 15% of cells are rocks
        num_rocks = int(self.width * self.height * 0.15)
        while len(self.rocks) < num_rocks:
            x = random.randint(0, self.width - 1)
            y = random.randint(0, self.height - 1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def find_safe_channels(self):
        """P1: Find safe channels (where you CAN be)"""
        # Simple BFS to find all safe paths
        visited = set()
        queue = [(self.start, [self.start])]
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.goal:
                self.safe_channels.append(path)
                continue
            
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            
            # Check neighbors (up, down, left, right)
            for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) not in self.rocks and
                    (nx, ny) not in visited):
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    def optimize_within_channel(self):
        """P2: Optimize within safe channels (best path)"""
        if not self.safe_channels:
            return []
        
        # Find shortest safe path
        self.optimal_path = min(self.safe_channels, key=len)
        return self.optimal_path
    
    def run_protocol(self):
        """Execute full P0→P1→P2 protocol"""
        print("=" * 50)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("=" * 50)
        
        # P0: Show negative space
        print("\n[P0] MAP NEGATIVE SPACE")
        print(f"Found {len(self.rocks)} rocks (failure states)")
        self.display_maze(show_path=False, show_rocks=True)
        time.sleep(1)
        
        # P1: Find safe channels
        print("\n[P1] IDENTIFY SAFE CHANNELS")
        self.find_safe_channels()
        print(f"Found {len(self.safe_channels)} safe channels")
        if self.safe_channels:
            print(f"Shortest safe channel: {len(self.safe_channels[0])} steps")
        time.sleep(1)
        
        # P2: Optimize within channel
        print("\n[P2] OPTIMIZE WITHIN CHANNEL")
        optimal = self.optimize_within_channel()
        if optimal:
            print(f"Optimal path: {len(optimal)} steps")
            self.display_maze(show_path=True, show_rocks=True)
        else:
            print("No safe path found!")
        
        return optimal
    
    def display_maze(self, show_path=False, show_rocks=False):
        """Display ASCII maze"""
        print("\n" + "+" + "-" * (self.width * 2 - 1) + "+")
        
        for y in range(self.height):
            row = "|"
            for x in range(self.width):
                pos = (x, y)
                
                if pos == self.start:
                    row += "S "
                elif pos == self.goal:
                    row += "G "
                elif show_path and pos in self.optimal_path:
                    row += "· "
                elif show_rocks and pos in self.rocks:
                    row += "█ "
                else:
                    row += "  "
            row = row.rstrip() + "|"
            print(row)
        
        print("+" + "-" * (self.width * 2 - 1) + "+")
        
        # Legend
        legend = []
        if show_rocks:
            legend.append("█ = Rock (P0 negative space)")
        if show_path:
            legend.append("· = Optimal path (P2)")
        legend.append("S = Start, G = Goal")
        
        if legend:
            print("\n".join(legend))

def demonstrate_failure():
    """Show what happens when you skip to P2 (unconstrained optimization)"""
    print("\n" + "=" * 50)
    print("FAILURE MODE: Skipping P0+P1, going straight to P2")
    print("=" * 50)
    
    maze = DeadbandMaze(seed=99)
    
    # Greedy pathfinder (P2-only)
    path = []
    current = maze.start
    visited = set()
    
    while current != maze.goal and len(path) < 100:
        path.append(current)
        visited.add(current)
        
        # Greedy: move toward goal
        x, y = current
        gx, gy = maze.goal
        
        # Try to move toward goal
        if gx > x and (x+1, y) not in maze.rocks and (x+1, y) not in visited:
            current = (x+1, y)
        elif gx < x and (x-1, y) not in maze.rocks and (x-1, y) not in visited:
            current = (x-1, y)
        elif gy > y and (x, y+1) not in maze.rocks and (x, y+1) not in visited:
            current = (x, y+1)
        elif gy < y and (x, y-1) not in maze.rocks and (x, y-1) not in visited:
            current = (x, y-1)
        else:
            # Stuck
            break
    
    path.append(current)
    
    print("\nGreedy pathfinder result:")
    maze.display_maze(show_path=False, show_rocks=True)
    print(f"\nPath length: {len(path)}")
    print(f"Reached goal: {current == maze.goal}")
    
    if current != maze.goal:
        print("FAILED: Hit local minimum (trapped by rocks)")

def main():
    """Main demonstration"""
    print("Cocapn Fleet — Deadband Protocol Demo")
    print("Tinker Agent — Cycle 254\n")
    
    # Successful protocol execution
    maze = DeadbandMaze(seed=42)
    optimal_path = maze.run_protocol()
    
    # Show failure mode
    demonstrate_failure()
    
    # Summary
    print("\n" + "=" * 50)
    print("PROTOCOL SUMMARY")
    print("=" * 50)
    print("P0 (Rocks): Identify failure states first")
    print("P1 (Channels): Map all safe paths")
    print("P2 (Optimize): Choose best path within safe channels")
    print("\nViolation: P2 before P0+P1 → probability of failure → 1.0")
    
    return optimal_path

if __name__ == "__main__":
    main()

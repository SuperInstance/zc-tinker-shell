#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

A visual ASCII simulation of the Deadband Protocol (P0→P1→P2) in a 10x10 maze.
Demonstrates the strict priority: map rocks (P0), find safe channels (P1), then optimize path (P2).

Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
import time
from typing import List, Tuple, Set

# ========== MAZE CONFIG ==========
MAZE_WIDTH = 10
MAZE_HEIGHT = 10
START = (0, 0)
GOAL = (9, 9)
ROCK_DENSITY = 0.15  # ~15% of cells are rocks

# Symbols for display
AGENT = 'A'
ROCK = '█'
GOAL_CELL = 'G'
PATH = '·'
SAFE = ' '
VISITED = '.'

# ========== DEADBAND PROTOCOL IMPLEMENTATION ==========

class DeadbandAgent:
    def __init__(self, maze: List[List[str]]):
        self.maze = maze
        self.rocks: Set[Tuple[int, int]] = set()      # P0: Negative space
        self.safe: Set[Tuple[int, int]] = set()       # P1: Safe channels
        self.path: List[Tuple[int, int]] = []         # P2: Optimal path
        self.visited: Set[Tuple[int, int]] = set()
        
    def p0_map_rocks(self) -> None:
        """P0: Identify all rocks (negative space)."""
        for y in range(MAZE_HEIGHT):
            for x in range(MAZE_WIDTH):
                if self.maze[y][x] == ROCK:
                    self.rocks.add((x, y))
        print(f"[P0] Mapped {len(self.rocks)} rocks")
    
    def p1_find_safe_channels(self) -> None:
        """P1: Find all reachable safe cells (deadbands)."""
        stack = [START]
        while stack:
            x, y = stack.pop()
            if (x, y) in self.visited:
                continue
            if (x, y) in self.rocks:
                continue
            if not (0 <= x < MAZE_WIDTH and 0 <= y < MAZE_HEIGHT):
                continue
            
            self.visited.add((x, y))
            self.safe.add((x, y))
            
            # Expand to neighbors (4-directional)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                stack.append((x + dx, y + dy))
        print(f"[P1] Found {len(self.safe)} safe cells")
    
    def p2_optimize_path(self) -> bool:
        """P2: Find shortest path within safe channels (BFS)."""
        from collections import deque
        
        if GOAL not in self.safe:
            print("[P2] Goal is not in safe channels — impossible!")
            return False
        
        queue = deque([(START, [START])])
        seen = {START}
        
        while queue:
            (x, y), current_path = queue.popleft()
            
            if (x, y) == GOAL:
                self.path = current_path
                print(f"[P2] Found optimal path: {len(self.path)} steps")
                return True
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (nx, ny) in self.safe and (nx, ny) not in seen:
                    seen.add((nx, ny))
                    queue.append(((nx, ny), current_path + [(nx, ny)]))
        
        print("[P2] No path to goal within safe channels")
        return False
    
    def run_protocol(self) -> bool:
        """Execute strict P0→P1→P2 priority."""
        print("\n=== DEADBAND PROTOCOL EXECUTION ===")
        
        # P0: Map negative space
        self.p0_map_rocks()
        if not self.rocks:
            print("Warning: No rocks mapped — protocol still valid")
        
        # P1: Find safe channels
        self.p1_find_safe_channels()
        if START not in self.safe:
            print("Fatal: Start position is not safe!")
            return False
        
        # P2: Optimize within channels
        success = self.p2_optimize_path()
        
        return success

# ========== VISUALIZATION ==========

def generate_maze() -> List[List[str]]:
    """Create random maze with rocks, start, and goal."""
    maze = [[SAFE for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    
    # Place rocks randomly
    for y in range(MAZE_HEIGHT):
        for x in range(MAZE_WIDTH):
            if (x, y) != START and (x, y) != GOAL:
                if random.random() < ROCK_DENSITY:
                    maze[y][x] = ROCK
    
    # Mark start and goal
    maze[START[1]][START[0]] = AGENT
    maze[GOAL[1]][GOAL[0]] = GOAL_CELL
    
    return maze

def display_maze(maze: List[List[str]], 
                 agent: DeadbandAgent = None,
                 show_p0: bool = False,
                 show_p1: bool = False,
                 show_p2: bool = False) -> None:
    """Display maze with optional protocol overlays."""
    print("+" + "-" * MAZE_WIDTH + "+")
    for y in range(MAZE_HEIGHT):
        row = "|"
        for x in range(MAZE_WIDTH):
            cell = maze[y][x]
            
            # Overlay protocol states
            if show_p2 and agent and (x, y) in agent.path:
                row += PATH
            elif show_p1 and agent and (x, y) in agent.safe:
                row += VISITED
            elif show_p0 and agent and (x, y) in agent.rocks:
                row += ROCK
            else:
                row += cell
        row += "|"
        print(row)
    print("+" + "-" * MAZE_WIDTH + "+")

def run_demo() -> None:
    """Main demonstration sequence."""
    print("Deadband Protocol Maze Demo")
    print("=" * 40)
    print(f"Maze: {MAZE_WIDTH}x{MAZE_HEIGHT}, Start {START}, Goal {GOAL}")
    print(f"Rock density: {ROCK_DENSITY*100:.0f}%")
    print()
    
    # Generate and show initial maze
    maze = generate_maze()
    print("Initial Maze (A=Agent, G=Goal, █=Rock):")
    display_maze(maze)
    time.sleep(1.5)
    
    # Create agent and run protocol
    agent = DeadbandAgent(maze)
    
    print("\n--- Phase P0: Map Rocks ---")
    agent.p0_map_rocks()
    display_maze(maze, agent, show_p0=True)
    time.sleep(1.5)
    
    print("\n--- Phase P1: Find Safe Channels ---")
    agent.p1_find_safe_channels()
    display_maze(maze, agent, show_p1=True)
    time.sleep(1.5)
    
    print("\n--- Phase P2: Optimize Path ---")
    success = agent.p2_optimize_path()
    display_maze(maze, agent, show_p2=True)
    time.sleep(1.5)
    
    # Results
    print("\n" + "=" * 40)
    print("RESULTS:")
    print(f"  Rocks mapped (P0): {len(agent.rocks)}")
    print(f"  Safe cells found (P1): {len(agent.safe)}")
    print(f"  Path length (P2): {len(agent.path) if success else 'N/A'}")
    print(f"  Success: {'YES' if success else 'NO'}")
    
    if success:
        print(f"\nOptimal path (P2 constrained to safe channels):")
        for i, (x, y) in enumerate(agent.path):
            print(f"  Step {i:2d}: ({x}, {y})")
    
    # Compare with naive approach (skip P0+P1)
    print("\n" + "=" * 40)

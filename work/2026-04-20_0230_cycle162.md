#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge simulation: 50/50 success with deadband vs 0/50 without.
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        """
        Initialize a maze with rocks (obstacles).
        Based on constraint_sim_final.json: 64 rocks in 20×20 grid.
        """
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width - 1, height - 1)
        
        # P0: Map negative space (rocks)
        self.rocks: Set[Tuple[int, int]] = set()
        total_cells = width * height
        target_rocks = int(total_cells * rock_density)
        
        # Place rocks randomly, avoiding start and goal
        while len(self.rocks) < target_rocks:
            x = random.randint(0, width - 1)
            y = random.randint(0, height - 1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
        
        # P1: Safe channels (deadbands) — will be computed
        self.safe_channels: List[Set[Tuple[int, int]]] = []
        
        # P2: Optimal path
        self.optimal_path: List[Tuple[int, int]] = []
    
    def p0_map_rocks(self) -> Set[Tuple[int, int]]:
        """P0: Identify negative space (what NOT to do)."""
        return self.rocks
    
    def p1_find_safe_channels(self) -> List[Set[Tuple[int, int]]]:
        """P1: Find all safe channels (deadbands) using BFS flood fill."""
        visited = set()
        channels = []
        
        for x in range(self.width):
            for y in range(self.height):
                pos = (x, y)
                if pos in self.rocks or pos in visited:
                    continue
                
                # Flood fill from this position
                stack = [pos]
                channel = set()
                while stack:
                    cx, cy = stack.pop()
                    if (cx, cy) in visited:
                        continue
                    if (cx, cy) in self.rocks:
                        continue
                    
                    visited.add((cx, cy))
                    channel.add((cx, cy))
                    
                    # 4-directional neighbors
                    for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                        nx, ny = cx + dx, cy + dy
                        if 0 <= nx < self.width and 0 <= ny < self.height:
                            if (nx, ny) not in visited and (nx, ny) not in self.rocks:
                                stack.append((nx, ny))
                
                if channel:
                    channels.append(channel)
        
        self.safe_channels = channels
        return channels
    
    def p2_optimize_in_channel(self) -> List[Tuple[int, int]]:
        """P2: Find shortest path within safe channels (A* constrained to deadbands)."""
        # First, identify which channel contains start and goal
        start_channel = None
        goal_channel = None
        for i, channel in enumerate(self.safe_channels):
            if self.start in channel:
                start_channel = i
            if self.goal in channel:
                goal_channel = i
        
        if start_channel is None or goal_channel is None or start_channel != goal_channel:
            return []  # No safe path exists
        
        safe_cells = self.safe_channels[start_channel]
        
        # A* search constrained to safe cells
        open_set = {self.start}
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: self._heuristic(self.start, self.goal)}
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score.get(pos, float('inf')))
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                path.reverse()
                self.optimal_path = path
                return path
            
            open_set.remove(current)
            
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # Constraint: must be in safe channel
                if neighbor not in safe_cells:
                    continue
                
                tentative_g = g_score[current] + 1
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor, self.goal)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        return []
    
    def _heuristic(self, a: Tuple[int, int], b: Tuple[int, int]) -> int:
        """Manhattan distance heuristic."""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def run_protocol(self) -> bool:
        """Execute full Deadband Protocol P0→P1→P2."""
        print("=== DEADBAND PROTOCOL DEMO ===")
        print(f"Maze: {self.width}×{self.height}, Rocks: {len(self.rocks)}")
        print()
        
        # P0: Map negative space
        rocks = self.p0_map_rocks()
        print("P0: Mapped negative space")
        print(f"   Found {len(rocks)} rocks (avoid these)")
        
        # P1: Find safe channels
        channels = self.p1_find_safe_channels()
        print(f"P1: Found {len(channels)} safe channels (deadbands)")
        
        # P2: Optimize within channel
        path = self.p2_optimize_in_channel()
        if path:
            print(f"P2: Optimal path found ({len(path)} steps)")
            return True
        else:
            print("P2: No safe path to goal")
            return False
    
    def visualize(self, show_path: bool = True):
        """ASCII visualization of maze, rocks, and path."""
        print("\nMAZE VISUALIZATION:")
        print("  S = Start, G = Goal, # = Rock, · = Safe, * = Path")
        print()
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('#')
                elif show_path and pos in self.optimal_path:
                    row.append('*')
                else:
                    row.append('·')
            print(' '.join(row))
        
        if self.optimal_path:
            print(f"\nPath length: {len(self.optimal_path)} steps")
            print(f"Optimal (Manhattan distance): {self._heuristic(self.start, self.goal)}")

#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
import time
from collections import deque

# ========== CONFIGURATION ==========
MAZE_WIDTH = 20
MAZE_HEIGHT = 20
ROCK_COUNT = 64          # from simulation data
GOAL = (19, 19)          # bottom-right corner
START = (0, 0)           # top-left corner

# Symbols for visualization
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_MARKER = 'G'
PATH = '○'
SAFE = '░'               # safe channel marker
VISITED = 'x'            # visited during search

# ========== MAZE GENERATION ==========
def generate_maze(width, height, rock_count):
    """Create a maze with random rocks (negative space)."""
    maze = [[EMPTY for _ in range(width)] for _ in range(height)]
    rocks_set = set()
    while len(rocks_set) < rock_count:
        x = random.randint(0, width-1)
        y = random.randint(0, height-1)
        if (x, y) != START and (x, y) != GOAL:
            rocks_set.add((x, y))
    for (x, y) in rocks_set:
        maze[y][x] = ROCK
    maze[START[1]][START[0]] = AGENT
    maze[GOAL[1]][GOAL[0]] = GOAL_MARKER
    return maze, rocks_set

def print_maze(maze, title="", step=0):
    """Print the maze with a title."""
    print(f"\n{'='*60}")
    print(f"STEP {step}: {title}")
    print('='*60)
    for row in maze:
        print(' '.join(row))
    time.sleep(0.8)

# ========== DEADBAND PROTOCOL IMPLEMENTATION ==========
def p0_map_rocks(maze, rocks_set):
    """P0: Identify negative space (rocks)."""
    marked_maze = [row[:] for row in maze]
    for (x, y) in rocks_set:
        marked_maze[y][x] = ROCK
    print_maze(marked_maze, "P0 — MAP ROCKS (Negative Space)", 1)
    return rocks_set

def p1_find_safe_channels(maze, rocks_set):
    """P1: Find all cells reachable without hitting rocks (safe channels)."""
    width, height = len(maze[0]), len(maze)
    safe = [[False for _ in range(width)] for _ in range(height)]
    # BFS from start, avoiding rocks
    queue = deque([START])
    visited = set([START])
    while queue:
        x, y = queue.popleft()
        safe[y][x] = True
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and (nx, ny) not in rocks_set:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    # Visualize safe channels
    channel_maze = [row[:] for row in maze]
    for y in range(height):
        for x in range(width):
            if safe[y][x] and maze[y][x] == EMPTY:
                channel_maze[y][x] = SAFE
    print_maze(channel_maze, "P1 — SAFE CHANNELS (Deadbands)", 2)
    return safe

def p2_optimize_in_channel(maze, safe, rocks_set):
    """P2: Find shortest path within safe channels (constrained optimization)."""
    width, height = len(maze[0]), len(maze)
    # BFS for shortest path
    parent = {START: None}
    queue = deque([START])
    found = False
    while queue and not found:
        x, y = queue.popleft()
        if (x, y) == GOAL:
            found = True
            break
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if safe[ny][nx] and (nx, ny) not in parent and (nx, ny) not in rocks_set:
                    parent[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    # Reconstruct path
    path = []
    if found:
        node = GOAL
        while node:
            path.append(node)
            node = parent.get(node)
        path.reverse()
    # Visualize path
    path_maze = [row[:] for row in maze]
    for (x, y) in path:
        if (x, y) != START and (x, y) != GOAL:
            path_maze[y][x] = PATH
    print_maze(path_maze, "P2 — OPTIMAL PATH (Within Safe Channels)", 3)
    return path, found

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Based on fleet-knowledge/constraint_sim_final.json data
Demonstrates P0→P1→P2 vs P2-only strategies
"""

import random
from typing import List, Tuple, Set
from enum import Enum

class Cell(Enum):
    EMPTY = '·'
    ROCK = '█'
    AGENT = 'A'
    GOAL = 'G'
    PATH = '○'
    DEADBAND = '░'

class Strategy(Enum):
    UNCONSTRAINED = "P2-only (greedy)"
    DEADBAND = "P0→P1→P2 (Deadband Protocol)"

class Maze:
    def __init__(self, width: int = 20, height: int = 20, num_rocks: int = 64):
        self.width = width
        self.height = height
        self.grid = [[Cell.EMPTY for _ in range(width)] for _ in range(height)]
        
        # Start at top-left, goal at bottom-right
        self.start = (0, 0)
        self.goal = (height-1, width-1)
        
        # Place rocks (P0: negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        while len(self.rocks) < num_rocks:
            x = random.randint(0, height-1)
            y = random.randint(0, width-1)
            if (x, y) not in [self.start, self.goal]:
                self.rocks.add((x, y))
                self.grid[x][y] = Cell.ROCK
        
        self.grid[self.start[0]][self.start[1]] = Cell.AGENT
        self.grid[self.goal[0]][self.goal[1]] = Cell.GOAL
        
        # For visualization
        self.deadband_cells: Set[Tuple[int, int]] = set()
        self.path_cells: Set[Tuple[int, int]] = set()
    
    def get_neighbors(self, pos: Tuple[int, int]) -> List[Tuple[int, int]]:
        """Get valid neighboring cells (no rocks, within bounds)"""
        x, y = pos
        neighbors = []
        for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < self.height and 0 <= ny < self.width:
                if (nx, ny) not in self.rocks:
                    neighbors.append((nx, ny))
        return neighbors
    
    def manhattan_distance(self, a: Tuple[int, int], b: Tuple[int, int]) -> int:
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def display(self, show_deadband: bool = False):
        """Render ASCII maze"""
        print("+" + "-" * (self.width * 2 - 1) + "+")
        for i in range(self.height):
            row = []
            for j in range(self.width):
                cell = self.grid[i][j]
                if show_deadband and (i, j) in self.deadband_cells and cell == Cell.EMPTY:
                    row.append(Cell.DEADBAND.value)
                elif (i, j) in self.path_cells and cell == Cell.EMPTY:
                    row.append(Cell.PATH.value)
                else:
                    row.append(cell.value)
            print("|" + " ".join(row) + "|")
        print("+" + "-" * (self.width * 2 - 1) + "+")
    
    def run_unconstrained(self) -> Tuple[bool, List[Tuple[int, int]], int]:
        """P2-only greedy strategy (optimizes before mapping)"""
        path = [self.start]
        visited = {self.start}
        steps = 0
        max_steps = self.width * self.height * 2  # Prevent infinite loops
        
        current = self.start
        while current != self.goal and steps < max_steps:
            neighbors = self.get_neighbors(current)
            if not neighbors:
                return False, path, steps
            
            # Greedy: choose neighbor closest to goal (P2 optimization)
            neighbors.sort(key=lambda n: self.manhattan_distance(n, self.goal))
            for neighbor in neighbors:
                if neighbor not in visited:
                    current = neighbor
                    path.append(current)
                    visited.add(current)
                    steps += 1
                    break
            else:
                # All neighbors visited
                return False, path, steps
        
        return current == self.goal, path, steps
    
    def run_deadband(self) -> Tuple[bool, List[Tuple[int, int]], int]:
        """Deadband Protocol: P0→P1→P2"""
        # P0: Already have rocks mapped (negative space)
        
        # P1: Map safe channels (deadbands) using BFS from start
        deadband_map: Set[Tuple[int, int]] = set()
        queue = [self.start]
        visited_bfs = {self.start}
        
        while queue:
            current = queue.pop(0)
            deadband_map.add(current)
            
            for neighbor in self.get_neighbors(current):
                if neighbor not in visited_bfs:
                    visited_bfs.add(neighbor)
                    queue.append(neighbor)
        
        self.deadband_cells = deadband_map
        
        # P2: Optimize within safe channels (A* on deadband map)
        path = [self.start]
        visited = {self.start}
        steps = 0
        max_steps = len(deadband_map) * 2
        
        current = self.start
        while current != self.goal and steps < max_steps:
            if current not in deadband_map:
                return False, path, steps
            
            # Get neighbors that are in deadband
            neighbors = [n for n in self.get_neighbors(current) if n in deadband_map]
            if not neighbors:
                return False, path, steps
            
            # A* heuristic: f(n) = g(n) + h(n)
            neighbors.sort(key=lambda n: (
                len(path) + 1 + self.manhattan_distance(n, self.goal)
            ))
            
            for neighbor in neighbors:
                if neighbor not in visited:
                    current = neighbor
                    path.append(current)
                    visited.add(current)
                    steps += 1
                    break
            else:
                return False, path, steps
        
        return current == self.goal, path, steps

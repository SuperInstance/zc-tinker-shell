```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 147)
Demonstrates P0→P1→P2 priority in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # from constraint_sim_final.json
OPTIMAL_PATH_LENGTH = 21  # optimal path length

class DeadbandAgent:
    def __init__(self, maze_size=20):
        self.maze_size = maze_size
        self.rocks = set()
        self.start = (0, 0)
        self.goal = (maze_size-1, maze_size-1)
        self.safe_channels = []
        self.optimal_path = []
        
    def generate_maze(self):
        """P0: Map negative space (rocks)"""
        # Generate random rock positions
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, self.maze_size-1)
            y = random.randint(0, self.maze_size-1)
            # Don't place rocks on start or goal
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
        
        print(f"P0: Mapped {len(self.rocks)} rocks (negative space)")
        return self.rocks
    
    def find_safe_channels(self):
        """P1: Identify safe channels (paths without rocks)"""
        # Simple BFS to find all safe paths
        from collections import deque
        
        visited = set()
        queue = deque([(self.start, [self.start])])
        safe_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            
            # Check neighbors (up, down, left, right)
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.maze_size and 0 <= ny < self.maze_size and
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.safe_channels = safe_paths
        print(f"P1: Found {len(safe_paths)} safe channels")
        return safe_paths
    
    def optimize_within_channel(self):
        """P2: Find optimal path within safe channels"""
        if not self.safe_channels:
            print("P2 ERROR: No safe channels mapped (P1 not completed)")
            return []
        
        # Find shortest safe path
        self.optimal_path = min(self.safe_channels, key=len)
        print(f"P2: Optimal path length: {len(self.optimal_path)} steps")
        return self.optimal_path
    
    def visualize_maze(self, show_path=False):
        """Visualize the maze with ASCII art"""
        print("\n" + "="*50)
        print("MAZE VISUALIZATION")
        print("="*50)
        
        for y in range(self.maze_size):
            row = []
            for x in range(self.maze_size):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')
                elif show_path and pos in self.optimal_path:
                    row.append('·')
                else:
                    row.append(' ')
            print(' '.join(row))
        
        if show_path:
            print(f"\nOptimal path: {len(self.optimal_path)} steps")
            print(f"Rocks avoided: {len(self.rocks)}")
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2"""
        print("="*60)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*60)
        
        # P0: Map negative space
        print("\n[P0] MAPPING NEGATIVE SPACE")
        self.generate_maze()
        time.sleep(1)
        
        # P1: Find safe channels
        print("\n[P1] IDENTIFYING SAFE CHANNELS")
        self.find_safe_channels()
        time.sleep(1)
        
        # P2: Optimize within channel
        print("\n[P2] OPTIMIZING WITHIN SAFE CHANNELS")
        self.optimize_within_channel()
        time.sleep(1)
        
        # Results
        print("\n" + "="*60)
        print("RESULTS")
        print("="*60)
        
        if self.optimal_path:
            success = len(self.optimal_path) <= OPTIMAL_PATH_LENGTH * 1.5  # Allow some slack
            print(f"✓ Success: {success}")
            print(f"✓ Path length: {len(self.optimal_path)} (optimal: {OPTIMAL_PATH_LENGTH})")
            print(f"✓ Rocks avoided: {len(self.rocks)}")
            print(f"✓ Safe channels found: {len(self.safe_channels)}")
            
            # Show visualization
            self.visualize_maze(show_path=True)
        else:
            print("✗ Failed: No safe path found")
            self.visualize_maze(show_path=False)
        
        return self.optimal_path is not None

def demonstrate_violation():
    """Show what happens when violating P0→P1→P2 order"""
    print("\n" + "="*60)
    print("VIOLATION DEMO: Skipping P0+P1, going straight to P2")
    print("="*60)
    
    agent = DeadbandAgent()
    agent.generate_maze()  # Rocks exist but not mapped
    
    # VIOLATION: Try to optimize without mapping safe channels
    print("\n[VIOLATION] Attempting P2 without P0+P1...")
    try:
        # This will fail because safe_channels is empty
        path = agent.optimize_within_channel()
        print(f"Result: {'Path found' if path else 'No path'}")
    except Exception as e:
        print(f"Result: ERROR - {e}")
    
    agent.visualize_maze(show_path=False)
    print("\nLesson: Optimization without constraint mapping → failure")

def main():
    """Run the demonstration"""
    random.seed(42)  # For reproducibility
    
    # Run proper protocol
    agent = DeadbandAgent()
    success = agent.run_protocol()
    
    # Show violation example
    demonstrate_violation()
    
    # Summary
    print("\n" + "="*60)
    print("DEADBAND PROTOCOL SUMMARY")
    print("="*60)
    print("P0 (Map negative space): Identify rocks/failures")
    print("P1 (Find safe channels): Constrain to known-safe paths")
    print("P2 (Optimize within): Find best path in safe space")
    print("\nFrom fleet data:")
    print(f"- Unconstrained success: 33/100")
    print(f"- Deadband success: 50/50")
    print(f"- Optimal path: {OPTIMAL_PATH_LENGTH} steps")

if __name__ == "__main__":
    main()
```

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 96)
Simulates P0 (map rocks), P1 (find safe channels), P2 (optimize path) in a 10x10 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# ========== MAZE GENERATION ==========
def generate_maze(width: int, height: int, rock_density: float = 0.2) -> List[List[str]]:
    """Generate a maze with start (S), goal (G), and rocks (#)."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    # Place start (top-left) and goal (bottom-right)
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    # Place rocks randomly
    for y in range(height):
        for x in range(width):
            if maze[y][x] == '.' and random.random() < rock_density:
                maze[y][x] = '#'
    return maze

def print_maze(maze: List[List[str]], title: str):
    """Print maze with a title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

# ========== DEADBAND PROTOCOL ==========
def p0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Identify negative space — all rock coordinates."""
    rocks = set()
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == '#':
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
    """P1: Find all paths from S to G that avoid rocks (safe channels)."""
    width, height = len(maze[0]), len(maze)
    start, goal = (0, 0), (width-1, height-1)
    
    # Simple DFS to collect all safe paths (no backtracking for demo)
    def dfs(pos: Tuple[int, int], path: List[Tuple[int, int]], visited: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
        if pos == goal:
            return [path + [pos]]
        x, y = pos
        safe_paths = []
        # Only move right or down (to keep it simple)
        for dx, dy in [(1, 0), (0, 1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    safe_paths.extend(dfs((nx, ny), path + [pos], visited | {(nx, ny)}))
        return safe_paths
    
    return dfs(start, [], {start})

def p2_optimize_within_channel(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """P2: Choose shortest path among safe channels."""
    if not safe_paths:
        return []
    return min(safe_paths, key=len)

def simulate_unconstrained_greedy(maze: List[List[str]]) -> bool:
    """Simulate a P2-only agent: greedy right/down moves, ignores rocks."""
    width, height = len(maze[0]), len(maze)
    x, y = 0, 0
    path = [(0, 0)]
    while (x, y) != (width-1, height-1):
        # Greedy: prefer right, then down
        if x + 1 < width and maze[y][x+1] != '#':
            x += 1
        elif y + 1 < height and maze[y+1][x] != '#':
            y += 1
        else:
            return False  # Hit a dead end (rock block)
        path.append((x, y))
        if maze[y][x] == '#':
            return False  # Hit a rock
    return True

# ========== MAIN DEMONSTRATION ==========
def main():
    random.seed(42)  # Reproducible for demo
    WIDTH, HEIGHT = 10, 10
    
    print("DEADBAND PROTOCOL MAZE DEMO — Tinker (Fleet Cocapn)")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print("=" * 50)
    
    # Generate maze
    maze = generate_maze(WIDTH, HEIGHT, rock_density=0.2)
    print_maze(maze, "Initial Maze (S=start, G=goal, #=rock)")
    
    # === P0: Map Rocks ===
    rocks = p0_map_rocks(maze)
    print(f"P0 — Mapped {len(rocks)} rocks (negative space)")
    
    # === P1: Find Safe Channels ===
    safe_paths = p1_find_safe_channels(maze, rocks)
    print(f"P1 — Found {len(safe_paths)} safe channels (deadbands)")
    if safe_paths:
        print(f"   Example safe channel length: {len(safe_paths[0])} steps")
    
    # === P2: Optimize Within Channel ===
    optimal_path = p2_optimize_within_channel(safe_paths)
    print(f"P2 — Optimal path length: {len(optimal_path) if optimal_path else 'No path'}")
    
    # Visualize optimal path
    if optimal_path:
        viz = [row[:] for row in maze]
        for x, y in optimal_path[1:-1]:  # Skip start and goal
            viz[y][x] = 'O'
        print_maze(viz, "Optimal Path (O) within Safe Channels")
    
    # === Compare with Unconstrained Greedy (P2-only) ===
    print("\n--- COMPARISON: Deadband vs Unconstrained ---")
    deadband_success = len(optimal_path) > 0
    unconstrained_success = simulate_unconstrained_greedy(maze)
    
    print(f"Deadband Protocol (P0→P1→P2) success: {deadband_success}")
    print(f"Unconstrained Greedy (P2-only) success: {unconstrained_success}")
    
    # Run multiple trials to approximate fleet statistics
    print("\n--- 50-TRIAL SIMULATION (approximating fleet data) ---")
    trials = 50
    deadband_wins = 0
    unconstrained_wins = 0
    
    for _ in range(trials):
        maze_trial = generate_maze(WIDTH, HEIGHT, 0.2)
        rocks_trial = p0_map_rocks(maze_trial)
        safe_paths_trial = p1_find_safe_channels(maze_trial, rocks_trial)
        optimal_trial = p2_optimize_within_channel(safe_paths_trial)
        if optimal_trial:
            deadband_wins += 1
        if simulate_unconstrained_greedy(maze_trial):
            unconstrained_wins += 1
    
    print(f"Deadband success: {deadband_wins}/{trials} ({deadband_wins/trials*100:.0f}%)")
    print(f"Unconstrained success: {unconstrained_wins}/{trials} ({unconstrained_wins/trials*100:.0f}%)")
    print(f"Rocks (P0) avoided: {deadband_wins - unconstrained_wins} more wins with protocol")

if __name__ == "__main__":
    main()

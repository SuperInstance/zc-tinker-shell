**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on reading `fleet-knowledge/narrow-games/constraint_sim_final.json` and `fleet-knowledge/docs/DEADBAND-PROTOCOL.md`, I will implement a concrete simulation that mirrors the fleet's constraint theory results.

**Plan:**
1. **P0 (Map Rocks):** Generate a random 10x10 maze with obstacles ('#') and open cells ('.').
2. **P1 (Find Safe Channels):** Use BFS from start to goal to find all reachable paths, then filter to only those that avoid rocks.
3. **P2 (Optimize Within Channel):** From the safe channels, select the shortest path (Manhattan distance).
4. **Visualization:** Print the maze at each phase, marking start ('S'), goal ('G'), rocks ('#'), safe path cells ('*'), and chosen optimal path ('O').

**Output:** A single, runnable Python file.

**File:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Simulates the fleet's constraint theory results in a navigable ASCII maze.
"""

import random
from collections import deque

def generate_maze(width=10, height=10, rock_density=0.2):
    """Generate a random maze with rocks as obstacles."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    rocks = 0
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
                rocks += 1
    # Ensure start (0,0) and goal (width-1, height-1) are clear
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    return maze, rocks

def print_maze(maze, title="Maze"):
    """Print the maze with a title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))

def p0_map_rocks(maze):
    """P0: Identify all rock positions (negative space)."""
    rocks = []
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start, goal):
    """P1: Find all safe paths from start to goal (BFS that avoids rocks)."""
    width, height = len(maze[0]), len(maze)
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    # BFS to find all reachable positions and their parents
    queue = deque([start])
    visited = {start: None}
    
    while queue:
        x, y = queue.popleft()
        if (x, y) == goal:
            break
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if maze[ny][nx] != '#' and (nx, ny) not in visited:
                    visited[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    
    # Reconstruct all safe paths? For simplicity, we find one safe channel (reachable goal)
    if goal not in visited:
        return []  # No safe channel
    
    # For demo: return the single safest channel (the BFS path)
    path = []
    node = goal
    while node != start:
        path.append(node)
        node = visited[node]
    path.reverse()
    return [path]  # Return as list of one channel for simplicity

def p2_optimize_within_channel(safe_channels):
    """P2: From safe channels, pick the shortest (Manhattan distance)."""
    if not safe_channels:
        return None
    # In this simple BFS, the first channel is already shortest. For multiple channels, we'd compare.
    return safe_channels[0]

def run_deadband_simulation():
    """Main simulation loop."""
    random.seed(42)  # Deterministic for demo
    maze, rock_count = generate_maze(10, 10, 0.2)
    start = (0, 0)
    goal = (9, 9)
    
    print_maze(maze, "Initial Maze (P0: Rocks are '#')")
    
    # P0: Map rocks
    rocks = p0_map_rocks(maze)
    print(f"\nP0: Mapped {len(rocks)} rocks (negative space).")
    
    # P1: Find safe channels
    safe_channels = p1_find_safe_channels(maze, start, goal)
    if not safe_channels:
        print("P1: NO SAFE CHANNEL FOUND — Goal unreachable.")
        return
    
    # Mark safe channel on maze
    maze_p1 = [row[:] for row in maze]
    for x, y in safe_channels[0]:
        if maze_p1[y][x] not in ['S', 'G']:
            maze_p1[y][x] = '*'
    print_maze(maze_p1, "P1: Safe Channel Found ('*' = safe path)")
    
    # P2: Optimize within channel
    optimal_path = p2_optimize_within_channel(safe_channels)
    maze_p2 = [row[:] for row in maze]
    for x, y in optimal_path:
        if maze_p2[y][x] not in ['S', 'G']:
            maze_p2[y][x] = 'O'
    print_maze(maze_p2, "P2: Optimal Path Within Channel ('O' = optimal)")
    
    # Metrics
    print("\n=== Simulation Results ===")
    print(f"Rocks (P0 negatives): {rock_count}")
    print(f"Safe channel length: {len(optimal_path)} steps")
    print(f"Optimal path cost (Manhattan): {abs(goal[0]-start[0]) + abs(goal[1]-start[1])}")
    print("Status: SUCCESS — Deadband protocol followed (P0→P1→P2).")

if __name__ == "__main__":
    run_deadband_simulation()
```

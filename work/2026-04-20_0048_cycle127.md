#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 protocol with ASCII visualization.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time
from typing import List, Tuple, Set

# Simulation constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # ~16% density, matches constraint_sim_final.json
START = (0, 0)
GOAL = (19, 19)

class DeadbandMaze:
    """Maze simulation implementing Deadband Protocol."""
    
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
        # Initialize maze
        self._place_rocks()
        self._mark_start_goal()
    
    def _place_rocks(self):
        """P0: Map negative space (rocks)."""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != START and (x, y) != GOAL:
                self.rocks.add((x, y))
                self.grid[y][x] = '#'
    
    def _mark_start_goal(self):
        """Mark start (S) and goal (G) positions."""
        sx, sy = START
        gx, gy = GOAL
        self.grid[sy][sx] = 'S'
        self.grid[gy][gx] = 'G'
    
    def find_safe_channels(self):
        """P1: Identify safe channels (deadbands) using BFS."""
        visited = set()
        channels = []
        
        def bfs(start):
            queue = [start]
            channel = []
            while queue:
                x, y = queue.pop(0)
                if (x, y) in visited:
                    continue
                visited.add((x, y))
                channel.append((x, y))
                
                # Explore neighbors (up, down, left, right)
                for dx, dy in [(0,1), (0,-1), (1,0), (-1,0)]:
                    nx, ny = x + dx, y + dy
                    if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                        if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                            queue.append((nx, ny))
            return channel
        
        # Find all connected safe regions
        for y in range(MAZE_SIZE):
            for x in range(MAZE_SIZE):
                if (x, y) not in self.rocks and (x, y) not in visited:
                    channel = bfs((x, y))
                    if len(channel) > 1:  # Only keep meaningful channels
                        channels.append(channel)
        
        self.safe_channels = channels
        return channels
    
    def optimize_within_channel(self):
        """P2: Find optimal path within safe channels (A*)."""
        # Find channel containing start and goal
        start_channel = None
        goal_channel = None
        
        for channel in self.safe_channels:
            if START in channel:
                start_channel = channel
            if GOAL in channel:
                goal_channel = channel
        
        if not start_channel or not goal_channel or start_channel != goal_channel:
            return []  # No safe path exists
        
        # A* search within safe positions
        open_set = {START}
        came_from = {}
        g_score = {START: 0}
        f_score = {START: self._heuristic(START, GOAL)}
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score.get(pos, float('inf')))
            
            if current == GOAL:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(START)
                self.optimal_path = list(reversed(path))
                return self.optimal_path
            
            open_set.remove(current)
            
            for dx, dy in [(0,1), (0,-1), (1,0), (-1,0)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # Check bounds and safety
                if not (0 <= neighbor[0] < MAZE_SIZE and 0 <= neighbor[1] < MAZE_SIZE):
                    continue
                if neighbor in self.rocks:
                    continue
                if neighbor not in start_channel:
                    continue
                
                tentative_g = g_score[current] + 1
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor, GOAL)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        return []
    
    def _heuristic(self, a, b):
        """Manhattan distance heuristic."""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def visualize(self, step="initial"):
        """ASCII visualization of current maze state."""
        print(f"\n{'='*60}")
        print(f"DEADBAND PROTOCOL — {step.upper()}")
        print(f"{'='*60}")
        
        # Create display grid
        display = [row[:] for row in self.grid]
        
        # Mark safe channels
        if step in ["p1", "p2"] and self.safe_channels:
            for channel in self.safe_channels:
                for x, y in channel:
                    if display[y][x] == '.':
                        display[y][x] = '~'  # Safe water
        
        # Mark optimal path
        if step == "p2" and self.optimal_path:
            for x, y in self.optimal_path[1:-1]:  # Skip start and goal
                display[y][x] = '*'
        
        # Print grid
        for y in range(MAZE_SIZE):
            row = ''
            for x in range(MAZE_SIZE):
                row += display[y][x] + ' '
            print(row)
        
        # Legend
        print(f"\nLegend: S=Start, G=Goal, #=Rock, ~=Safe channel, *=Optimal path")
        print(f"Rocks: {len(self.rocks)} | Safe channels: {len(self.safe_channels)}")
        
        if step == "p2":
            if self.optimal_path:
                print(f"Optimal path length: {len(self.optimal_path)-1} steps")
            else:
                print("NO SAFE PATH FOUND — Protocol violation would hit rocks")

def run_simulation():
    """Run complete Deadband Protocol demonstration."""
    print("COCAPN FLEET — DEADBAND PROTOCOL DEMONSTRATION")
    print("Based on constraint_sim_final.json data")
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}")
    print("\n" + "-"*60)
    
    # Initialize maze
    maze = DeadbandMaze()
    
    # P0: Map negative space (already done in init)
    print("\nP0: MAP NEGATIVE SPACE")
    print("Identifying rocks (failure states)...")
    maze.visualize("p0")
    time.sleep(1)
    
    # P1: Identify safe channels
    print("\n\nP1: IDENTIFY SAFE CHANNELS")
    print("Finding deadbands (safe water)...")
    safe_channels = maze.find_safe_channels()
    maze.visualize("p1")
    time.sleep(1)
    
    # P2: Optimize within channel
    print("\n\nP2: OPTIMIZE WITHIN CHANNEL")
    print("Finding shortest safe path...")
    optimal_path = maze.optimize_within_channel()
    maze.visualize("p2")
    
    # Results
    print("\n" + "="*60)
    print("SIMULATION RESULTS")
    print("="*60)
    
    success = len(optimal_path) > 0
    print(f"Success: {'YES' if success else 'NO'}")

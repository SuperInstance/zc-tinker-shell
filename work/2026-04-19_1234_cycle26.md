**Action:** I'll write `deadband_maze_demo.py`. It will:
1. Generate a random 10x10 maze with rocks (obstacles), start, and goal.
2. Step P0: Map negative space (identify all rock coordinates).
3. Step P1: Find safe channels (BFS to find all reachable cells avoiding rocks).
4. Step P2: Optimize within channel (A* pathfinding only through safe cells).
5. Print ASCII visualization at each step.

**Output:**

```python
#!/usr/bin/env python3
"""
deadband_maze_demo.py
Demonstrates the Deadband Protocol (P0→P1→P2) in a navigable ASCII maze.
Fleet knowledge source: constraint_sim_final.json, DEADBAND-PROTOCOL.md
"""

import random
from collections import deque
import heapq

# Constants
MAZE_WIDTH = 10
MAZE_HEIGHT = 10
ROCK_PROBABILITY = 0.25
START = (0, 0)
GOAL = (9, 9)

def generate_maze():
    """Generate random maze with rocks (#), start (S), goal (G), empty (.)."""
    maze = [['.' for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    rocks = []
    for y in range(MAZE_HEIGHT):
        for x in range(MAZE_WIDTH):
            if (x, y) == START:
                maze[y][x] = 'S'
            elif (x, y) == GOAL:
                maze[y][x] = 'G'
            elif random.random() < ROCK_PROBABILITY:
                maze[y][x] = '#'
                rocks.append((x, y))
    return maze, rocks

def print_maze(maze, title="Maze"):
    """Print ASCII maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))

def p0_map_negative_space(maze, rocks):
    """P0: Identify negative space (rocks)."""
    print("\n--- P0: MAP NEGATIVE SPACE ---")
    print(f"Rocks (failure states) found: {len(rocks)}")
    rock_maze = [row[:] for row in maze]
    for x, y in rocks:
        rock_maze[y][x] = 'X'  # Mark mapped rocks
    print_maze(rock_maze, "Negative Space (X = rock)")

def p1_find_safe_channels(maze, rocks):
    """P1: Find all reachable cells avoiding rocks (safe channels)."""
    print("\n--- P1: IDENTIFY SAFE CHANNELS ---")
    safe = set()
    queue = deque([START])
    visited = set([START])
    rock_set = set(rocks)
    
    while queue:
        x, y = queue.popleft()
        safe.add((x, y))
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                if (nx, ny) not in visited and (nx, ny) not in rock_set:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    
    # Visualize safe channels
    channel_maze = [row[:] for row in maze]
    for y in range(MAZE_HEIGHT):
        for x in range(MAZE_WIDTH):
            if (x, y) in safe:
                if channel_maze[y][x] == '.':
                    channel_maze[y][x] = '~'  # Safe water
    print_maze(channel_maze, "Safe Channels (~ = safe, # = rock)")
    print(f"Safe cells reachable from start: {len(safe)}")
    return safe

def p2_optimize_within_channel(maze, safe):
    """P2: Find shortest path from start to goal using only safe cells."""
    print("\n--- P2: OPTIMIZE WITHIN CHANNEL ---")
    # A* heuristic (Manhattan distance)
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    frontier = []
    heapq.heappush(frontier, (0, START))
    came_from = {START: None}
    cost_so_far = {START: 0}
    
    while frontier:
        _, current = heapq.heappop(frontier)
        if current == GOAL:
            break
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            neighbor = (nx, ny)
            if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                if neighbor not in safe:
                    continue  # Skip unsafe cells (enforce channel)
                new_cost = cost_so_far[current] + 1
                if neighbor not in cost_so_far or new_cost < cost_so_far[neighbor]:
                    cost_so_far[neighbor] = new_cost
                    priority = new_cost + heuristic(neighbor, GOAL)
                    heapq.heappush(frontier, (priority, neighbor))
                    came_from[neighbor] = current
    
    # Reconstruct path
    path = []
    current = GOAL
    while current != START:
        if current not in came_from:
            print("No safe path to goal within channel.")
            return []
        path.append(current)
        current = came_from[current]
    path.append(START)
    path.reverse()
    
    # Visualize path
    path_maze = [row[:] for row in maze]
    for x, y in path[1:-1]:  # Exclude start and goal
        path_maze[y][x] = '*'
    print_maze(path_maze, "Optimal Path (* = path, ~ = safe)")
    print(f"Path length: {len(path)-1} steps")
    return path

def run_deadband_demo():
    """Execute full Deadband Protocol demonstration."""
    print("DEADBAND PROTOCOL DEMO")
    print("=" * 40)
    print(f"Maze: {MAZE_WIDTH}x{MAZE_HEIGHT}, Start {START}, Goal {GOAL}")
    
    # Generate environment
    maze, rocks = generate_maze()
    print_maze(maze, "Initial Maze")
    
    # P0: Map negative space
    p0_map_negative_space(maze, rocks)
    
    # P1: Find safe channels
    safe = p1_find_safe_channels(maze, rocks)
    
    # P2: Optimize within channel
    path = p2_optimize_within_channel(maze, safe)
    
    # Summary
    print("\n=== PROTOCOL SUMMARY ===")
    print(f"P0 rocks identified: {len(rocks)}")
    print(f"P1 safe cells: {len(safe)}")
    print(f"P2 path found: {'Yes' if path else 'No'}")
    if path:
        print(f"Path steps: {len(path)-1}")
    print("\nNote: This mirrors fleet simulation results:")
    print("- Unconstrained (P2-only) often fails (hits rocks)")
    print("- Deadband (P0→P1→P2) finds safe optimal path")

if __name__ == "__main__":
    random.seed(42)  # Deterministic for demo
    run_deadband_demo()
```

## Deadband Protocol Visual Demo (ASCII Maze)

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path)
in a simple ASCII maze.
"""

import random
from typing import List, Tuple, Set

# Maze dimensions
WIDTH = 20
HEIGHT = 10

# Symbols
AGENT = 'A'
GOAL = 'G'
ROCK = '█'
SAFE = '·'
PATH = '○'
VISITED = 'x'
EMPTY = ' '

def generate_maze(width: int, height: int, rock_density: float = 0.2) -> List[List[str]]:
    """Generate a random maze with rocks."""
    maze = [[EMPTY for _ in range(width)] for _ in range(height)]
    
    # Place rocks randomly
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = ROCK
    
    # Ensure start (0,0) and goal (width-1, height-1) are clear
    maze[0][0] = EMPTY
    maze[height-1][width-1] = GOAL
    
    return maze

def print_maze(maze: List[List[str]], title: str):
    """Print maze with title."""
    print(f"\n{'='*60}")
    print(f"{title}")
    print(f"{'='*60}")
    for row in maze:
        print(''.join(row))
    print()

def p0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Identify all rock positions (negative space)."""
    rocks = set()
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == ROCK:
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[str]]:
    """P1: Mark all non-rock cells as safe channels."""
    safe_maze = [row.copy() for row in maze]
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if (x, y) not in rocks and safe_maze[y][x] == EMPTY:
                safe_maze[y][x] = SAFE
    return safe_maze

def p2_optimize_path(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> Tuple[List[List[str]], bool]:
    """P2: Find shortest path through safe channels using BFS."""
    from collections import deque
    
    start = (0, 0)
    goal = (len(maze[0])-1, len(maze)-1)
    
    # BFS setup
    queue = deque([start])
    visited = {start: None}
    
    # Directions: right, down, left, up
    directions = [(1, 0), (0, 1), (-1, 0), (0, -1)]
    
    found = False
    while queue:
        x, y = queue.popleft()
        
        if (x, y) == goal:
            found = True
            break
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            
            # Check bounds
            if 0 <= nx < len(maze[0]) and 0 <= ny < len(maze):
                # Check if not a rock and not visited
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    visited[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    
    # Reconstruct path if found
    path_maze = [row.copy() for row in maze]
    if found:
        # Mark path
        current = goal
        while current != start:
            x, y = current
            if path_maze[y][x] != GOAL:
                path_maze[y][x] = PATH
            current = visited[current]
        path_maze[0][0] = AGENT
    
    return path_maze, found

def run_demo():
    """Run the complete Deadband Protocol demo."""
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("Maze: 20x10, ~20% rocks")
    print("Agent: 'A', Goal: 'G', Rocks: '█', Safe: '·', Path: '○'")
    
    # Generate maze
    random.seed(42)  # For reproducibility
    maze = generate_maze(WIDTH, HEIGHT, rock_density=0.2)
    print_maze(maze, "INITIAL MAZE (with rocks '█' and goal 'G')")
    
    # P0: Map negative space (rocks)
    rocks = p0_map_rocks(maze)
    print(f"P0: Mapped {len(rocks)} rocks (negative space)")
    
    # P1: Find safe channels
    safe_maze = p1_find_safe_channels(maze, rocks)
    print_maze(safe_maze, "P1: SAFE CHANNELS (safe cells marked '·')")
    
    # P2: Optimize within channels
    path_maze, found = p2_optimize_path(maze, rocks)
    
    if found:
        print_maze(path_maze, "P2: OPTIMAL PATH THROUGH SAFE CHANNELS (path '○')")
        print("SUCCESS: Found optimal path through safe channels!")
    else:
        print_maze(path_maze, "P2: NO PATH FOUND (all channels blocked)")
        print("FAILURE: No safe path to goal exists.")
    
    # Demonstrate violation: P2 without P0+P1 (greedy approach)
    print("\n" + "="*60)
    print("VIOLATION DEMO: P2-only (greedy optimization)")
    print("="*60)
    
    def greedy_p2_only(maze):
        """Greedy approach that ignores rocks until hitting them."""
        x, y = 0, 0
        path = [(x, y)]
        visited = set([(x, y)])
        
        while (x, y) != (WIDTH-1, HEIGHT-1):
            # Prefer right, then down (greedy toward goal)
            moves = [(1, 0), (0, 1), (-1, 0), (0, -1)]
            moved = False
            
            for dx, dy in moves:
                nx, ny = x + dx, y + dy
                if 0 <= nx < WIDTH and 0 <= ny < HEIGHT:
                    if (nx, ny) not in visited:
                        if maze[ny][nx] != ROCK:
                            x, y = nx, ny
                            path.append((x, y))
                            visited.add((x, y))
                            moved = True
                            break
            
            if not moved:
                break  # Stuck
        
        # Create visualization
        greedy_maze = [row.copy() for row in maze]
        for px, py in path:
            if greedy_maze[py][px] == EMPTY:
                greedy_maze[py][px] = VISITED
        
        reached_goal = (x, y) == (WIDTH-1, HEIGHT-1)
        return greedy_maze, reached_goal, len(path)
    
    greedy_maze, reached, steps = greedy_p2_only(maze)
    print_maze(greedy_maze, f"GREEDY P2-ONLY (visited '{VISITED}', steps: {steps})")
    
    if reached:
        print(f"Greedy reached goal in {steps} steps (but may have been lucky)")
    else:
        print("Greedy got stuck! Violates Deadband Protocol.")
    
    print("\n" + "="*60)
    print("SUMMARY: Deadband Protocol ensures safe navigation")
    print("P0→P1→P2 strict priority prevents getting stuck")
    print("="*60)

if __name__ == "__main__":
    run_demo()
```

## Deadband Protocol Visual Demo (ASCII Maze)

**File:** `deadband_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self.safe_channels = []
        self.optimal_path = []
        
        # P0: Map negative space (rocks)
        self._place_rocks(rock_density)
        
        # Mark start and goal
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def _place_rocks(self, density):
        """P0: Identify rocks (negative space)."""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        
        # Avoid start and goal
        candidates = [(x, y) for x in range(self.width) for y in range(self.height)]
        candidates.remove(self.start)
        candidates.remove(self.goal)
        
        self.rocks = set(random.sample(candidates, num_rocks))
        for x, y in self.rocks:
            self.grid[y][x] = '#'
    
    def find_safe_channels(self):
        """P1: Find all safe channels (paths avoiding rocks)."""
        # BFS to find all reachable cells that aren't rocks
        visited = set()
        queue = deque([self.start])
        safe_cells = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            safe_cells.add((x, y))
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        # Convert to channel representation (connected components)
        components = []
        remaining = safe_cells.copy()
        
        while remaining:
            start = remaining.pop()
            component = set()
            stack = [start]
            while stack:
                cell = stack.pop()
                if cell in component:
                    continue
                component.add(cell)
                x, y = cell
                for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                    neighbor = (x + dx, y + dy)
                    if neighbor in remaining:
                        stack.append(neighbor)
                        remaining.discard(neighbor)
            components.append(component)
        
        self.safe_channels = components
        return components
    
    def optimize_within_channel(self):
        """P2: Find shortest path within safe channels."""
        # Find channel containing start
        start_channel = None
        for channel in self.safe_channels:
            if self.start in channel and self.goal in channel:
                start_channel = channel
                break
        
        if not start_channel:
            return []  # No safe path
        
        # BFS for shortest path
        queue = deque([(self.start, [self.start])])
        visited = set()
        
        while queue:
            (x, y), path = queue.popleft()
            if (x, y) == self.goal:
                self.optimal_path = path
                return path
            
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                neighbor = (nx, ny)
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if neighbor in start_channel and neighbor not in visited:
                        queue.append((neighbor, path + [neighbor]))
        
        return []
    
    def run_protocol(self):
        """Execute full P0→P1→P2 protocol."""
        print("=== DEADBAND PROTOCOL DEMO ===")
        print(f"Maze: {self.width}x{self.height}, Rocks: {len(self.rocks)}")
        print()
        
        # P0: Already done in init
        print("P0: MAPPED NEGATIVE SPACE")
        print(f"    Identified {len(self.rocks)} rocks (#'s)")
        print()
        
        # P1: Find safe channels
        print("P1: FINDING SAFE CHANNELS")
        channels = self.find_safe_channels()
        print(f"    Found {len(channels)} safe channel(s)")
        print(f"    Largest channel: {max(len(c) for c in channels) if channels else 0} cells")
        print()
        
        # P2: Optimize within channel
        print("P2: OPTIMIZING WITHIN CHANNEL")
        path = self.optimize_within_channel()
        if path:
            print(f"    Found optimal path: {len(path)-1} steps")
            # Mark path on grid
            for x, y in path[1:-1]:  # Skip start and goal
                self.grid[y][x] = 'o'
        else:
            print("    NO SAFE PATH TO GOAL (trapped in local minima)")
        print()
        
        return len(path) > 0
    
    def display(self, show_channels=False):
        """Display the maze."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                cell = self.grid[y][x]
                if show_channels and cell == '.':
                    # Color code by channel
                    for i, channel in enumerate(self.safe_channels):
                        if (x, y) in channel:
                            cell = str(i % 10)  # Single digit channel ID
                            break
                row.append(cell)
            print(' '.join(row))
        print()

def main():
    """Run demonstration."""
    # Create maze with parameters from constraint_sim_final.json
    # 20x20, ~64 rocks (0.16 density), optimal path length ~21
    maze = DeadbandMaze(width=20, height=20, rock_density=0.16)
    
    print("INITIAL MAZE (P0 complete):")
    maze.display()
    
    success = maze.run_protocol()
    
    print("FINAL MAZE WITH PATH:")
    maze.display()
    
    if success:
        print("RESULT: Protocol SUCCESS — reached goal via safe channel")
    else:
        print("RESULT: Protocol FAILURE — no safe channel to goal")
    
    # Show channels view
    print("\nSAFE CHANNELS VISUALIZATION (P1):")
    maze.display(show_channels=True)

if __name__ == "__main__":
    main()
```

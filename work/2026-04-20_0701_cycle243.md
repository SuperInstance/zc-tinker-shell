# deadband_maze_demo.py
# Demonstrates Deadband Protocol (P0→P1→P2) in a 20x20 ASCII maze.
# Fleet: Cocapn, Agent: Tinker, Cycle: 241

import random

def generate_maze(width=20, height=20, rock_density=0.16):
    """Generate a maze with start (S), goal (G), and rocks (#)."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    # Place rocks
    rocks = 0
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
                rocks += 1
    # Ensure start and goal are clear
    start = (0, 0)
    goal = (width-1, height-1)
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    return maze, rocks

def print_maze(maze, path=None):
    """Print the maze, optionally highlighting a path with *."""
    for y, row in enumerate(maze):
        line = ''
        for x, cell in enumerate(row):
            if path and (x, y) in path:
                line += '*'
            else:
                line += cell
        print(line)

def p0_map_rocks(maze):
    """P0: Map negative space — identify all rock coordinates."""
    rocks = []
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.append((x, y))
    return set(rocks)

def p1_find_safe_channels(maze, start, goal, rocks):
    """P1: Find safe channels — BFS to find all paths avoiding rocks."""
    width, height = len(maze[0]), len(maze)
    directions = [(1,0), (0,1), (-1,0), (0,-1)]
    queue = [(start, [start])]
    visited = {start}
    safe_paths = []
    
    while queue:
        (x, y), path = queue.pop(0)
        if (x, y) == goal:
            safe_paths.append(path)
            continue
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    return safe_paths

def p2_optimize_within_channel(safe_paths):
    """P2: Optimize within channel — pick the shortest safe path."""
    if not safe_paths:
        return None
    return min(safe_paths, key=len)

def run_deadband_protocol():
    """Main demonstration of P0→P1→P2."""
    print("=== Deadband Protocol Maze Demo ===")
    print("P0: Map rocks (negative space)")
    print("P1: Find safe channels (avoid rocks)")
    print("P2: Optimize within channel (shortest safe path)")
    print()
    
    # Generate maze matching simulation parameters (20x20, ~16% rocks)
    maze, rocks_count = generate_maze(20, 20, 0.16)
    print("Generated Maze (20x20):")
    print_maze(maze)
    print(f"Rocks: {rocks_count}")
    print()
    
    # P0: Map rocks
    rocks = p0_map_rocks(maze)
    print(f"P0: Mapped {len(rocks)} rock cells.")
    
    # P1: Find safe channels
    start = (0, 0)
    goal = (19, 19)
    safe_paths = p1_find_safe_channels(maze, start, goal, rocks)
    print(f"P1: Found {len(safe_paths)} safe channel(s) from S to G.")
    
    # P2: Optimize within channel
    if safe_paths:
        optimal_path = p2_optimize_within_channel(safe_paths)
        print(f"P2: Optimal safe path length: {len(optimal_path)} steps")
        print()
        print("Maze with optimal safe path (*):")
        print_maze(maze, set(optimal_path))
        
        # Verify against simulation results
        if len(optimal_path) - 1 == 21:  # Steps = path length - 1 (start counted)
            print("\n✓ Matches simulation optimal: 21 steps")
        else:
            print(f"\nNote: Path length {len(optimal_path)-1} steps (simulation optimal: 21)")
    else:
        print("P2: No safe path found (all channels blocked).")
    
    # Demonstrate violation (P2-only greedy)
    print("\n--- Violation Example: P2-only (Greedy) ---")
    print("Attempting direct path without P0/P1...")
    # Simple greedy move right/down
    path = []
    x, y = start
    while (x, y) != goal:
        path.append((x, y))
        if x < goal[0] and maze[y][x+1] != '#':
            x += 1
        elif y < goal[1] and maze[y+1][x] != '#':
            y += 1
        else:
            path.append((x, y))
            print("  Hit a rock! Greedy path fails.")
            break
    if (x, y) == goal:
        path.append(goal)
        print(f"  Greedy path length: {len(path)-1} steps (likely suboptimal/unsafe)")
    print()

if __name__ == "__main__":
    run_deadband_protocol()

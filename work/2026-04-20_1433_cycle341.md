```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Tinker — Cycle 341
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
"""

import random
import time

# ====================
# MAZE CONFIGURATION
# ====================
MAZE_WIDTH = 20
MAZE_HEIGHT = 20
START = (0, 0)
GOAL = (19, 19)

# Rocks (obstacles) — from simulation data: 64 rocks in 20x20 maze
NUM_ROCKS = 64

# Symbols
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_MARKER = 'G'
PATH = '○'
SAFE = '░'

# ====================
# DEADBAND PROTOCOL
# ====================
class DeadbandAgent:
    def __init__(self, maze):
        self.maze = maze
        self.width = len(maze[0])
        self.height = len(maze)
        self.visited = [[False for _ in range(self.width)] for _ in range(self.height)]
        self.safe_map = [[False for _ in range(self.width)] for _ in range(self.height)]
        self.path = []

    def p0_map_rocks(self):
        """P0: Identify negative space (rocks)."""
        rocks = []
        for y in range(self.height):
            for x in range(self.width):
                if self.maze[y][x] == ROCK:
                    rocks.append((x, y))
        return rocks

    def p1_find_safe_channels(self, start, goal):
        """P1: Find all safe cells reachable from start without hitting rocks."""
        stack = [start]
        self.safe_map[start[1]][start[0]] = True
        directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]

        while stack:
            x, y = stack.pop()
            for dx, dy in directions:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if not self.safe_map[ny][nx] and self.maze[ny][nx] != ROCK:
                        self.safe_map[ny][nx] = True
                        stack.append((nx, ny))

        # Return whether goal is in safe channel
        return self.safe_map[goal[1]][goal[0]]

    def p2_optimize_in_channel(self, start, goal):
        """P2: Find shortest path within safe channel (BFS)."""
        if not self.safe_map[goal[1]][goal[0]]:
            return []  # Goal not reachable

        queue = [(start, [start])]
        visited = set([start])
        directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]

        while queue:
            (x, y), path = queue.pop(0)
            if (x, y) == goal:
                return path
            for dx, dy in directions:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited and self.safe_map[ny][nx]:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        return []

    def run_protocol(self):
        """Execute full P0→P1→P2 sequence."""
        print("\n=== DEADBAND PROTOCOL DEMO ===")
        print(f"Maze: {MAZE_WIDTH}x{MAZE_HEIGHT}, Rocks: {NUM_ROCKS}")
        print(f"Start: {START}, Goal: {GOAL}\n")

        # P0: Map rocks
        rocks = self.p0_map_rocks()
        print("P0 — MAP ROCKS (Negative Space)")
        print(f"Identified {len(rocks)} rock cells.")
        self.print_maze(phase=0)
        time.sleep(1.5)

        # P1: Find safe channels
        reachable = self.p1_find_safe_channels(START, GOAL)
        print("\nP1 — FIND SAFE CHANNELS")
        print(f"Goal reachable via safe channels? {reachable}")
        self.print_maze(phase=1)
        time.sleep(1.5)

        # P2: Optimize within channel
        if reachable:
            path = self.p2_optimize_in_channel(START, GOAL)
            print("\nP2 — OPTIMIZE PATH WITHIN SAFE CHANNEL")
            print(f"Shortest safe path length: {len(path)-1} steps")
            self.path = path
            self.print_maze(phase=2)
        else:
            print("\nP2 — SKIPPED (Goal not in safe channel)")

        # Summary
        print("\n=== PROTOCOL SUMMARY ===")
        print(f"P0 rocks mapped: {len(rocks)}/{MAZE_WIDTH*MAZE_HEIGHT}")
        print(f"P1 safe cells: {sum(sum(row) for row in self.safe_map)}")
        print(f"P2 path length: {len(self.path)-1 if self.path else 'N/A'}")
        print(f"Success: {'YES' if reachable and self.path else 'NO'}")

    def print_maze(self, phase=0):
        """Print maze with appropriate symbols for each phase."""
        for y in range(self.height):
            line = []
            for x in range(self.width):
                cell = self.maze[y][x]
                if (x, y) == START:
                    line.append(AGENT)
                elif (x, y) == GOAL:
                    line.append(GOAL_MARKER)
                elif phase == 0:
                    line.append(ROCK if cell == ROCK else EMPTY)
                elif phase == 1:
                    if self.safe_map[y][x]:
                        line.append(SAFE)
                    else:
                        line.append(ROCK if cell == ROCK else EMPTY)
                elif phase == 2:
                    if (x, y) in self.path:
                        line.append(PATH)
                    elif self.safe_map[y][x]:
                        line.append(SAFE)
                    else:
                        line.append(ROCK if cell == ROCK else EMPTY)
            print(''.join(line))
        print()

# ====================
# MAZE GENERATION
# ====================
def generate_maze():
    """Create a random maze with rocks, ensuring start and goal are clear."""
    maze = [[EMPTY for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    
    # Place rocks randomly
    rocks_placed = 0
    while rocks_placed < NUM_ROCKS:
        x = random.randint(0, MAZE_WIDTH-1)
        y = random.randint(0, MAZE_HEIGHT-1)
        if (x, y) not in [START, GOAL] and maze[y][x] != ROCK:
            maze[y][x] = ROCK
            rocks_placed += 1
    
    return maze

# ====================
# MAIN
# ====================
if __name__ == "__main__":
    print("Deadband Protocol Visual Demo")
    print("Reference: constraint_sim_final.json")
    print(f"Optimal path: 21 steps, Deadband success: 50/50")
    print("-" * 40)
    
    # Generate and run
    maze = generate_maze()
    agent = DeadbandAgent(maze)
    agent.run_protocol()
    
    print("\n" + "="*40)
    print("Demo complete. Training tile ready for submission.")
```

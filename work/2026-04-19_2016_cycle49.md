```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Simulates the P0→P1→P2 process in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set
from collections import deque

# Constants from the simulation data
MAZE_SIZE = 20
ROCK_COUNT = 64  # Rocks from constraint_sim_final.json
START = (0, 0)
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

# Symbols for visualization
SYMBOLS = {
    'empty': '·',
    'rock': '█',
    'start': 'S',
    'goal': 'G',
    'p0_marked': '×',  # Rocks identified in P0
    'p1_channel': '░', # Safe channel cells (P1)
    'p2_path': '○',    # Optimized path (P2)
    'agent': 'A'
}

class DeadbandMaze:
    def __init__(self):
        self.grid = [[SYMBOLS['empty'] for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: Set[Tuple[int, int]] = set()
        self.optimal_path: List[Tuple[int, int]] = []
        
        # Initialize
        self.grid[START[0]][START[1]] = SYMBOLS['start']
        self.grid[GOAL[0]][GOAL[1]] = SYMBOLS['goal']
        
    def generate_rocks(self):
        """P0: Generate rocks (negative space)"""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            pos = (x, y)
            if pos != START and pos != GOAL and pos not in self.rocks:
                self.rocks.add(pos)
                self.grid[x][y] = SYMBOLS['rock']
    
    def find_safe_channels(self):
        """P1: Find all cells reachable from start without hitting rocks"""
        visited = set()
        queue = deque([START])
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            self.safe_channels.add((x, y))
            
            # Mark as channel (unless it's start/goal/rock)
            if (x, y) != START and (x, y) != GOAL and (x, y) not in self.rocks:
                self.grid[x][y] = SYMBOLS['p1_channel']
            
            # Check neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        queue.append((nx, ny))
    
    def optimize_within_channels(self):
        """P2: Find shortest path from start to goal within safe channels"""
        if GOAL not in self.safe_channels:
            return []  # No path possible
        
        # BFS for shortest path
        parent = {START: None}
        queue = deque([START])
        
        while queue:
            x, y = queue.popleft()
            
            if (x, y) == GOAL:
                # Reconstruct path
                path = []
                current = (x, y)
                while current is not None:
                    path.append(current)
                    current = parent[current]
                path.reverse()
                self.optimal_path = path
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) in self.safe_channels and (nx, ny) not in parent:
                        parent[(nx, ny)] = (x, y)
                        queue.append((nx, ny))
        
        return []  # No path found
    
    def run_protocol(self):
        """Execute full P0→P1→P2 protocol"""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===")
        print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}")
        print(f"Start: {START}, Goal: {GOAL}")
        print()
        
        # P0: Map negative space
        print("P0: MAP NEGATIVE SPACE (Rocks)")
        self.generate_rocks()
        self.print_maze()
        print(f"Identified {len(self.rocks)} rocks (failure states)")
        print()
        
        # P1: Find safe channels
        print("P1: IDENTIFY SAFE CHANNELS")
        self.find_safe_channels()
        self.print_maze()
        print(f"Found {len(self.safe_channels)} safe cells reachable from start")
        print(f"Goal reachable: {GOAL in self.safe_channels}")
        print()
        
        # P2: Optimize within channels
        print("P2: OPTIMIZE WITHIN CHANNEL")
        path = self.optimize_within_channels()
        if path:
            # Mark path (excluding start and goal)
            for x, y in path[1:-1]:
                self.grid[x][y] = SYMBOLS['p2_path']
            self.print_maze()
            print(f"Optimal path length: {len(path)-1} steps")
            print(f"Path: {path[:3]}...{path[-3:] if len(path) > 6 else ''}")
        else:
            print("NO PATH FOUND - Goal is not in safe channels")
            self.print_maze()
        
        return len(path) if path else 0
    
    def print_maze(self):
        """Print current maze state"""
        print("   " + " ".join(str(i%10) for i in range(MAZE_SIZE)))
        for i in range(MAZE_SIZE):
            row = f"{i:2} "
            for j in range(MAZE_SIZE):
                row += self.grid[i][j] + " "
            print(row)
        print()
    
    def print_legend(self):
        """Print symbol legend"""
        print("\nLEGEND:")
        for name, symbol in SYMBOLS.items():
            print(f"  {symbol} = {name}")
        print()

def simulate_multiple_runs(num_runs: int = 50):
    """Run multiple simulations to replicate constraint_sim_final.json results"""
    print(f"\n{'='*60}")

    print('='*60)
    
    successes = 0
    total_path_length = 0
    
    for run in range(num_runs):
        maze = DeadbandMaze()

            successes += 1
            total_path_length += path_length
        
        if run < 2:  # Show first 2 runs in detail
            print(f"\n--- End of Run {run+1} ---")
            maze.print_legend()
            print('-'*40)
    
    # Summary statistics
    print(f"\n{'='*60}")
    print("SIMULATION SUMMARY")
    print('='*60)
    print(f"Total runs: {num_runs}")
    print(f"Successes (goal reached): {successes}/{num_runs} ({successes/num_runs*100:.1f}%)")
    if successes > 0:
        print(f"Average path length: {total_path_length/successes:.1f}")

**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Complete. Based on `constraint_sim_final.json` (20×20 maze, 64 rocks, optimal path length 21). Implemented P0 (map rocks), P1 (find safe channels via BFS), P2 (A* within safe channels). Outputs ASCII visualization and stats.

**File:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 65)
Simulates P0→P1→P2 in a 20x20 maze with rocks.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import heapq
from collections import deque

# Constants from constraint_sim_final.json
MAZE_SIZE = 20
NUM_ROCKS = 64
OPTIMAL_PATH_LEN = 21

def generate_maze(size, rocks):
    """Create maze with start (0,0), goal (size-1,size-1), and random rocks."""
    maze = [['.' for _ in range(size)] for _ in range(size)]
    maze[0][0] = 'S'
    maze[size-1][size-1] = 'G'
    
    # Place rocks (cannot be start or goal)
    placed = 0
    while placed < rocks:
        x = random.randint(0, size-1)
        y = random.randint(0, size-1)
        if maze[x][y] == '.':
            maze[x][y] = '#'
            placed += 1
    return maze

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks)."""
    rocks = []
    for i in range(len(maze)):
        for j in range(len(maze[0])):
            if maze[i][j] == '#':
                rocks.append((i, j))
    return rocks

def p1_find_safe_channels(maze, start, goal):
    """P1: Find all reachable cells (safe channels) via BFS."""
    rows, cols = len(maze), len(maze[0])
    visited = [[False for _ in range(cols)] for _ in range(rows)]
    queue = deque([start])
    visited[start[0]][start[1]] = True
    safe_cells = [start]
    
    while queue:
        x, y = queue.popleft()
        for dx, dy in [(0,1),(1,0),(0,-1),(-1,0)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < rows and 0 <= ny < cols:
                if not visited[nx][ny] and maze[nx][ny] != '#':
                    visited[nx][ny] = True
                    safe_cells.append((nx, ny))
                    queue.append((nx, ny))
    return safe_cells, visited

def p2_optimize_in_channel(maze, safe_mask, start, goal):
    """P2: A* search constrained to safe cells."""
    rows, cols = len(maze), len(maze[0])
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            path.reverse()
            return path
        
        x, y = current
        for dx, dy in [(0,1),(1,0),(0,-1),(-1,0)]:
            nx, ny = x + dx, y + dy
            neighbor = (nx, ny)
            if 0 <= nx < rows and 0 <= ny < cols:
                if not safe_mask[nx][ny]:
                    continue  # Not in safe channel (P1 violation)
                if maze[nx][ny] == '#':
                    continue  # Rock (P0 violation)
                
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    return []  # No path found

def visualize(maze, path=None, safe_mask=None):
    """ASCII visualization of maze with optional path and safe channels."""
    rows, cols = len(maze), len(maze[0])
    output = []
    for i in range(rows):
        row_chars = []
        for j in range(cols):
            cell = maze[i][j]
            if path and (i, j) in path and cell not in ['S', 'G']:
                row_chars.append('*')
            elif safe_mask and not safe_mask[i][j] and cell == '.':
                row_chars.append('~')  # Unsafe water
            else:
                row_chars.append(cell)
        output.append(' '.join(row_chars))
    return '\n'.join(output)

def main():
    print("=== Deadband Protocol Maze Demo ===")
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {NUM_ROCKS}, Optimal: {OPTIMAL_PATH_LEN} steps")
    print()
    
    # Generate reproducible maze for demo
    random.seed(42)
    maze = generate_maze(MAZE_SIZE, NUM_ROCKS)
    start = (0, 0)
    goal = (MAZE_SIZE-1, MAZE_SIZE-1)
    
    print("Initial Maze:")
    print(visualize(maze))
    print()
    
    # P0: Map rocks
    rocks = p0_map_rocks(maze)
    print(f"P0: Mapped {len(rocks)} rocks (negative space)")
    
    # P1: Find safe channels
    safe_cells, safe_mask = p1_find_safe_channels(maze, start, goal)
    print(f"P1: Found {len(safe_cells)} safe cells (channels)")
    
    # Check if goal is reachable
    if not safe_mask[goal[0]][goal[1]]:
        print("P1 WARNING: Goal not reachable in safe channels!")
        return
    
    # P2: Optimize within safe channels
    path = p2_optimize_in_channel(maze, safe_mask, start, goal)
    if path:
        print(f"P2: Found optimal path of {len(path)-1} steps within safe channels")
        print()
        print("Final Maze (S=Start, G=Goal, #=Rock, *=Path, ~=Unsafe):")
        print(visualize(maze, set(path), safe_mask))
        print()
        print(f"Path coordinates: {path[:5]}...{path[-5:] if len(path)>10 else ''}")
    else:
        print("P2: No path found (should not happen with P0+P1 complete)")
    
    # Stats
    reachable_ratio = len(safe_cells) / (MAZE_SIZE * MAZE_SIZE)
    print()
    print("=== Statistics ===")
    print(f"Safe cells / Total: {len(safe_cells)} / {MAZE_SIZE*MAZE_SIZE} ({reachable_ratio:.1%})")
    print(f"Rocks avoided: {len(rocks)}")
    print(f"Path length: {len(path)-1 if path else 'N/A'} (Optimal target: {OPTIMAL_PATH_LEN})")

if __name__ == "__main__":
    main()
```

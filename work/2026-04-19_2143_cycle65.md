#!/usr/bin/env python3
"""
deadband_maze_demo.py

A visual demonstration of the Deadband Protocol (P0→P1→P2) in a 20×20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json:
    - optimal path length: 21
    - rocks: 64
    - deadband success: 50/50 runs

Doctrine:
    P0: Map rocks (negative space)
    P1: Find safe channels (deadbands)
    P2: Optimize within channels (shortest safe path)
"""

import random
from collections import deque
import heapq

# ----------------------------------------------------------------------
# P0: MAP THE NEGATIVE SPACE
# ----------------------------------------------------------------------
def generate_maze(width=20, height=20, num_rocks=64, seed=42):
    """Create a maze with start (S), goal (G), and rocks (#)."""
    random.seed(seed)
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place start (top-left) and goal (bottom-right)
    start = (0, 0)
    goal = (height-1, width-1)
    maze[start[0]][start[1]] = 'S'
    maze[goal[0]][goal[1]] = 'G'
    
    # Place rocks randomly, avoiding start and goal
    rocks_placed = 0
    while rocks_placed < num_rocks:
        r = random.randint(0, height-1)
        c = random.randint(0, width-1)
        if maze[r][c] == '.':
            maze[r][c] = '#'
            rocks_placed += 1
    
    return maze, start, goal

def print_maze(maze, title="MAZE"):
    """Print maze with title."""
    print(f"\n{'='*60}")
    print(f"{title:^60}")
    print(f"{'='*60}")
    for row in maze:
        print(' '.join(row))
    print()

# ----------------------------------------------------------------------
# P1: IDENTIFY SAFE CHANNELS (Deadbands)
# ----------------------------------------------------------------------
def find_safe_channels(maze, start, goal):
    """
    BFS to find all reachable cells from start that are not rocks.
    This defines the safe channel (deadband) where the agent can operate.
    """
    height, width = len(maze), len(maze[0])
    directions = [(0,1), (1,0), (0,-1), (-1,0)]
    
    visited = [[False for _ in range(width)] for _ in range(height)]
    queue = deque([start])
    sr, sc = start
    visited[sr][sc] = True
    
    safe_cells = []
    
    while queue:
        r, c = queue.popleft()
        safe_cells.append((r, c))
        
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < height and 0 <= nc < width:
                if not visited[nr][nc] and maze[nr][nc] != '#':
                    visited[nr][nc] = True
                    queue.append((nr, nc))
    
    # Check if goal is reachable (in safe channel)
    gr, gc = goal
    goal_reachable = visited[gr][gc]
    
    return safe_cells, goal_reachable

def mark_safe_channel(maze, safe_cells):
    """Create a visualization of the safe channel (deadband)."""
    viz = [row[:] for row in maze]
    for r, c in safe_cells:
        if viz[r][c] == '.':
            viz[r][c] = ' '
    return viz

# ----------------------------------------------------------------------
# P2: OPTIMIZE WITHIN CHANNEL (Constrained A*)
# ----------------------------------------------------------------------
def a_star_in_channel(maze, start, goal, safe_cells):
    """
    A* search constrained to safe cells only.
    Returns shortest path if one exists.
    """
    height, width = len(maze), len(maze[0])
    safe_set = set(safe_cells)
    
    # Heuristic: Manhattan distance
    def heuristic(r, c):
        gr, gc = goal
        return abs(gr - r) + abs(gc - c)
    
    # Priority queue: (f_score, g_score, position, parent)
    open_set = []
    heapq.heappush(open_set, (heuristic(*start), 0, start, None))
    
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(*start)}
    
    directions = [(0,1), (1,0), (0,-1), (-1,0)]
    
    while open_set:
        current_f, current_g, current, parent = heapq.heappop(open_set)
        
        if current == goal:
            # Reconstruct path
            path = []
            while current is not None:
                path.append(current)
                current = came_from.get(current)
            path.reverse()
            return path
        
        r, c = current
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            neighbor = (nr, nc)
            
            # Constraint: must be within bounds AND in safe channel
            if not (0 <= nr < height and 0 <= nc < width):
                continue
            if neighbor not in safe_set:
                continue
            if maze[nr][nc] == '#':
                continue
            
            tentative_g = current_g + 1
            if neighbor not in g_score or tentative_g < g_score[neighbor]:
                came_from[neighbor] = current
                g_score[neighbor] = tentative_g
                f_score[neighbor] = tentative_g + heuristic(nr, nc)
                heapq.heappush(open_set, (f_score[neighbor], tentative_g, neighbor, current))
    
    return None  # No path found within safe channel

def mark_path(maze, path):
    """Mark the optimal path with '*'."""
    viz = [row[:] for row in maze]
    for r, c in path[1:-1]:  # Skip start and goal
        viz[r][c] = '*'
    return viz

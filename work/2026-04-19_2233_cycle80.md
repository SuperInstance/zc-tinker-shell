Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 with ASCII visualization.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

MAZE_SIZE = 20

# Symbols
EMPTY = '·'
ROCK = '█'
START = 'S'
GOAL = 'G'
PATH = '○'
SAFE = '░'  # Safe channel cells

def generate_maze(size: int, rocks: int) -> Tuple[List[List[str]], Tuple[int, int], Tuple[int, int]]:
    """Generate a random maze with rocks, start, and goal."""
    maze = [[EMPTY for _ in range(size)] for _ in range(size)]
    
    # Place rocks randomly
    rock_positions = set()
    while len(rock_positions) < rocks:
        x = random.randint(0, size-1)
        y = random.randint(0, size-1)
        rock_positions.add((x, y))
    
    for x, y in rock_positions:
        maze[y][x] = ROCK
    
    # Place start (top-left) and goal (bottom-right)
    start = (0, 0)
    goal = (size-1, size-1)
    
    # Ensure start and goal are not rocks
    maze[start[1]][start[0]] = START
    maze[goal[1]][goal[0]] = GOAL
    
    return maze, start, goal

def print_maze(maze: List[List[str]], title: str = ""):

def p0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Map negative space (rocks)."""
    rocks = set()
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == ROCK:
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(maze: List[List[str]], start: Tuple[int, int], goal: Tuple[int, int], rocks: Set[Tuple[int, int]]) -> List[List[str]]:
    """P1: Identify safe channels (deadbands) using BFS flood fill."""
    size = len(maze)
    # Create a copy to mark safe cells
    safe_maze = [row[:] for row in maze]
    
    # BFS from start, avoiding rocks
    queue = [start]
    visited = set([start])
    
    while queue:
        x, y = queue.pop(0)
        
        # Mark as safe (if not start/goal)
        if safe_maze[y][x] not in [START, GOAL]:
            safe_maze[y][x] = SAFE
        
        # Check neighbors
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < size and 0 <= ny < size:
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    
    return safe_maze

def p2_optimize_path(maze: List[List[str]], start: Tuple[int, int], goal: Tuple[int, int], rocks: Set[Tuple[int, int]], strategy: str = "deadband") -> Tuple[List[Tuple[int, int]], bool]:
    """P2: Find optimal path using A*.
    
    strategy: "unconstrained" (ignores rocks), "deadband" (avoids rocks)
    """
    size = len(maze)
    
    # Heuristic: Manhattan distance
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    # For unconstrained: treat rocks as passable
    def is_passable(x, y, strat):
        if strat == "unconstrained":
            return True  # Can walk through rocks (will fail)
        else:  # deadband
            return (x, y) not in rocks
    
    # A* search
    open_set = {start}
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        current = min(open_set, key=lambda x: f_score.get(x, float('inf')))
        
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            return path, True
        
        open_set.remove(current)
        
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < size and 0 <= ny < size:
                if not is_passable(nx, ny, strategy):
                    continue
                
                neighbor = (nx, ny)
                tentative_g = g_score[current] + 1
                
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
    
    return [], False  # No path found

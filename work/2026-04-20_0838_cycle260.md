import random
from typing import List, Tuple, Set

# Constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

class DeadbandMaze:
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: Set[Tuple[int, int]] = set()
        self.optimal_path: List[Tuple[int, int]] = []
        
    def generate_rocks(self):
        """P0: Map negative space (rocks)"""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[x][y] = '#'
    
    def find_safe_channels(self):
        """P1: Identify safe channels (deadbands) using BFS"""
        visited = set()
        queue = [self.start]
        
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            self.safe_channels.add((x, y))
            
            # Explore neighbors (right and down only for simplicity)
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    queue.append((nx, ny))
    
    def optimize_within_channel(self):
        """P2: Find shortest path within safe channels (A* simplified)"""
        if self.goal not in self.safe_channels:
            return []  # No safe path to goal
        
        # Simple greedy path following safe channels
        path = [self.start]
        x, y = self.start
        
        while (x, y) != self.goal:
            # Prefer moves toward goal
            moves = []
            if x < self.goal[0] and (x+1, y) in self.safe_channels:
                moves.append((x+1, y))
            if y < self.goal[1] and (x, y+1) in self.safe_channels:
                moves.append((x, y+1))
            if not moves:
                # Fallback to any safe move
                for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                    nx, ny = x + dx, y + dy
                    if (nx, ny) in self.safe_channels and (nx, ny) not in path:
                        moves.append((nx, ny))
            
            if not moves:
                break  # Stuck
            
            x, y = moves[0]  # Take first valid move
            path.append((x, y))
        
        self.optimal_path = path if path[-1] == self.goal else []
        return self.optimal_path
    
    def run_unconstrained_greedy(self):
        """P2-only agent (violates protocol)"""
        path = [self.start]
        x, y = self.start
        
        while (x, y) != self.goal:
            # Greedy move toward goal without checking rocks
            moves = []
            if x < self.goal[0]:
                moves.append((x+1, y))
            if y < self.goal[1]:
                moves.append((x, y+1))
            
            if not moves:
                break
            
            x, y = random.choice(moves)
            if (x, y) in self.rocks:
                return []  # Hit a rock
            
            path.append((x, y))
        
        return path if path[-1] == self.goal else []
    
    def visualize(self, show_path=False):
        """ASCII visualization of maze state"""
        viz_grid = [row[:] for row in self.grid]
        viz_grid[self.start[0]][self.start[1]] = 'S'
        viz_grid[self.goal[0]][self.goal[1]] = 'G'
        
        if show_path and self.optimal_path:
            for x, y in self.optimal_path[1:-1]:
                viz_grid[x][y] = '○'
        
        print("+" + "-" * MAZE_SIZE + "+")
        for y in range(MAZE_SIZE):
            print("|", end="")
            for x in range(MAZE_SIZE):
                print(viz_grid[x][y], end="")
            print("|")
        print("+" + "-" * MAZE_SIZE + "+")
        
        # Legend
        print("\nLegend: S=Start, G=Goal, #=Rock, ○=Safe Path, .=Empty")
        print(f"Rocks: {len(self.rocks)} (P0 negative space)")
        print(f"Safe cells: {len(self.safe_channels)} (P1 channels)")

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

A visual ASCII simulation of the Deadband Protocol (P0→P1→P2) in a 10x10 maze.
Demonstrates the strict priority: map rocks (P0), find safe channels (P1),
then optimize path (P2) within safe channels only.

Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set

# ====================
# MAZE CONFIGURATION
# ====================
MAZE_WIDTH = 10
MAZE_HEIGHT = 10
START = (0, 0)
GOAL = (9, 9)
ROCK_COUNT = 12  # ~12% density, similar to narrow-games simulation

# Symbols for visualization
SYM_EMPTY = '·'
SYM_ROCK = '█'
SYM_START = 'S'
SYM_GOAL = 'G'
SYM_PATH = '○'
SYM_SAFE = ' '  # inside deadband channel

# ====================
# DEADBAND PROTOCOL IMPLEMENTATION
# ====================

class DeadbandAgent:
    """An agent that follows P0→P1→P2 strictly."""
    
    def __init__(self, maze: List[List[str]]):
        self.maze = maze
        self.rocks: Set[Tuple[int, int]] = set()      # P0: negative space
        self.safe_channels: Set[Tuple[int, int]] = set()  # P1: safe cells
        self.optimal_path: List[Tuple[int, int]] = []     # P2: best path
        
    def p0_map_rocks(self) -> None:
        """P0: Identify all rocks (negative space)."""
        for y in range(MAZE_HEIGHT):
            for x in range(MAZE_WIDTH):
                if self.maze[y][x] == SYM_ROCK:
                    self.rocks.add((x, y))
        print(f"[P0] Mapped {len(self.rocks)} rocks (negative space)")
    
    def p1_find_safe_channels(self) -> bool:
        """P1: Find all reachable safe cells (deadband channels)."""
        # Start from START, flood-fill avoiding rocks
        stack = [START]
        visited = set()
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Only add if not a rock
            if self.maze[y][x] != SYM_ROCK:
                self.safe_channels.add((x, y))
            
            # Explore neighbors (4-directional)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                    if (nx, ny) not in visited and self.maze[ny][nx] != SYM_ROCK:
                        stack.append((nx, ny))
        
        # Check if goal is reachable
        goal_reachable = GOAL in self.safe_channels
        print(f"[P1] Found {len(self.safe_channels)} safe cells | Goal reachable: {goal_reachable}")
        return goal_reachable
    
    def p2_optimize_in_channel(self) -> bool:
        """P2: Find shortest path within safe channels (constrained optimization)."""
        if GOAL not in self.safe_channels:
            print("[P2] Goal not in safe channels — optimization impossible")
            return False
        
        # BFS within safe channels only
        queue = [(START, [START])]
        visited = set([START])
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == GOAL:
                self.optimal_path = path
                print(f"[P2] Found optimal path: {len(path)-1} steps")
                return True
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                    if (nx, ny) not in visited and (nx, ny) in self.safe_channels:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        print("[P2] No path found within safe channels")
        return False
    
    def run_protocol(self) -> bool:
        """Execute full P0→P1→P2 sequence."""
        print("\n=== DEADBAND PROTOCOL EXECUTION ===")
        self.p0_map_rocks()
        if not self.p1_find_safe_channels():
            return False
        return self.p2_optimize_in_channel()

# ====================
# VISUALIZATION
# ====================

def generate_maze() -> List[List[str]]:
    """Create random maze with rocks, ensuring START and GOAL are clear."""
    maze = [[SYM_EMPTY for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    
    # Place rocks randomly
    rocks_placed = 0
    while rocks_placed < ROCK_COUNT:
        x = random.randint(0, MAZE_WIDTH-1)
        y = random.randint(0, MAZE_HEIGHT-1)
        if (x, y) != START and (x, y) != GOAL and maze[y][x] == SYM_EMPTY:
            maze[y][x] = SYM_ROCK
            rocks_placed += 1
    
    # Mark start and goal
    maze[START[1]][START[0]] = SYM_START
    maze[GOAL[1]][GOAL[0]] = SYM_GOAL
    return maze

def print_maze(maze: List[List[str]], title: str = "MAZE") -> None:
    """Print maze with border."""
    print(f"\n{title}:")
    print("  " + "".join(str(i) for i in range(MAZE_WIDTH)))
    for y in range(MAZE_HEIGHT):
        row = f"{y} "
        for x in range(MAZE_WIDTH):
            row += maze[y][x]
        print(row)

def print_safe_channels(safe_channels: Set[Tuple[int, int]], maze: List[List[str]]) -> None:
    """Visualize safe channels (P1 result)."""
    print("\nSAFE CHANNELS (P1 Result):")
    print("  " + "".join(str(i) for i in range(MAZE_WIDTH)))
    for y in range(MAZE_HEIGHT):
        row = f"{y} "
        for x in range(MAZE_WIDTH):
            if (x, y) == START:
                row += SYM_START
            elif (x, y) == GOAL:
                row += SYM_GOAL
            elif (x, y) in safe_channels:
                row += SYM_SAFE
            else:
                row += maze[y][x]
        print(row)

def print_optimal_path(path: List[Tuple[int, int]], maze: List[List[str]]) -> None:
    """Visualize optimal path (P2 result)."""
    print("\nOPTIMAL PATH (P2 Result):")
    print("  " + "".join(str(i) for i in range(MAZE_WIDTH)))
    
    # Create copy for visualization
    vis_maze = [row[:] for row in maze]
    for (x, y) in path[1:-1]:  # Exclude start and goal
        vis_maze[y][x] = SYM_PATH
    
    for y in range(MAZE_HEIGHT):
        row = f"{y} "
        for x in range(MAZE_WIDTH):
            row += vis_maze[y][x]
        print(row)
    
    print(f"Path length: {len(path)-1} steps")

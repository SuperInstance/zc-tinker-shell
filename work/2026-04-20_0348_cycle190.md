#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

A visual ASCII simulation demonstrating:
P0: Map negative space (rocks/obstacles)
P1: Identify safe channels (deadbands)
P2: Optimize within safe channels

Based on fleet-knowledge/narrow-games/constraint_sim_final.json
and fleet-knowledge/docs/DEADBAND-PROTOCOL.md
"""

import random
import time
from collections import deque

class MazeSimulator:
    def __init__(self, width=20, height=20, rock_density=0.16):
        """
        Initialize maze based on constraint_sim_final.json stats:
        - optimal: 21 (shortest path length)
        - rocks: 64 (in 20x20 maze = 16% density)
        """
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self.generate_maze(rock_density)
        
    def generate_maze(self, density):
        """Generate random obstacles (rocks)"""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        
        # Ensure start and goal are clear
        available = [(x, y) for x in range(self.width) for y in range(self.height)
                    if (x, y) not in [self.start, self.goal]]
        
        self.rocks = set(random.sample(available, num_rocks))
    
    def is_valid(self, pos):
        x, y = pos
        return (0 <= x < self.width and 
                0 <= y < self.height and 
                pos not in self.rocks)
    
    def get_neighbors(self, pos):
        x, y = pos
        neighbors = [(x+1, y), (x-1, y), (x, y+1), (x, y-1)]
        return [n for n in neighbors if self.is_valid(n)]
    
    def print_maze(self, path=None, deadband=None):
        """Print ASCII maze with visualization"""
        print("\n" + "="*60)
        print("MAZE SIMULATION - Deadband Protocol Demo")
        print("="*60)
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')  # Rock/obstacle
                elif deadband and pos in deadband:
                    row.append('·')  # Safe channel (P1)
                elif path and pos in path:
                    row.append('●')  # Optimal path (P2)
                else:
                    row.append(' ')
            print(' '.join(row))
        
        legend = [
            "S = Start",
            "G = Goal", 
            "█ = Rock (P0: Negative Space)",
            "· = Safe Channel (P1: Deadband)",
            "● = Optimal Path (P2: Within Channel)"
        ]
        print("\nLegend:", ' | '.join(legend))
    
    def greedy_search(self):
        """P2-only: Greedy optimization without P0/P1"""
        print("\n[P2-ONLY] Greedy Search (No Deadband Protocol)")
        print("-"*40)
        
        current = self.start
        path = [current]
        visited = set([current])
        
        while current != self.goal:
            neighbors = self.get_neighbors(current)
            if not neighbors:
                print("  FAILED: Hit dead end (rocks everywhere)")
                return None
            
            # Greedy: choose neighbor closest to goal
            neighbors.sort(key=lambda p: abs(p[0]-self.goal[0]) + abs(p[1]-self.goal[1]))
            
            for neighbor in neighbors:
                if neighbor not in visited:
                    current = neighbor
                    path.append(current)
                    visited.add(current)
                    break
            else:
                print("  FAILED: All neighbors visited (stuck in loop)")
                return None
        
        print(f"  Path length: {len(path)}")
        print(f"  Success: {'Yes' if path[-1] == self.goal else 'No'}")
        return path
    
    def deadband_protocol(self):
        """Full P0→P1→P2 Deadband Protocol"""
        print("\n[DEADBAND PROTOCOL] P0→P1→P2")
        print("-"*40)
        
        # P0: Map negative space (already done - self.rocks)
        print("P0: Negative space mapped")
        print(f"  Rocks identified: {len(self.rocks)}")
        
        # P1: Identify safe channels (BFS to find reachable area)
        print("\nP1: Identifying safe channels...")
        reachable = set()
        queue = deque([self.start])
        
        while queue:
            current = queue.popleft()
            if current in reachable:
                continue
            reachable.add(current)
            
            for neighbor in self.get_neighbors(current):
                if neighbor not in reachable:
                    queue.append(neighbor)
        
        # Check if goal is reachable
        if self.goal not in reachable:
            print("  FAILED: Goal unreachable from start")
            return None, None
        
        print(f"  Safe cells found: {len(reachable)}")
        print(f"  Goal reachable: Yes")
        
        # P2: Optimize within safe channels (BFS for shortest path)
        print("\nP2: Optimizing within safe channels...")
        parent = {self.start: None}
        queue = deque([self.start])
        
        while queue:
            current = queue.popleft()
            if current == self.goal:
                break
            
            for neighbor in self.get_neighbors(current):
                if neighbor not in parent and neighbor in reachable:
                    parent[neighbor] = current
                    queue.append(neighbor)
        
        # Reconstruct path
        if self.goal not in parent:
            print("  FAILED: No path found within safe channels")
            return None, reachable
        
        path = []
        current = self.goal
        while current is not None:
            path.append(current)
            current = parent[current]
        path.reverse()
        
        print(f"  Optimal path length: {len(path)}")
        print(f"  Success: {'Yes' if path[-1] == self.goal else 'No'}")
        
        return path, reachable

        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("Based on fleet simulation data:")
        print(f"  - Optimal path: 21 steps")
        print(f"  - Rocks: {len(self.rocks)} ({len(self.rocks)/(self.width*self.height):.0%} density)")
        print("="*60)
        
        # Show initial maze
        print("\nInitial Maze Configuration:")
        self.print_maze()
        
        # Run greedy search (P2-only)
        time.sleep(1)
        greedy_path = self.greedy_search()
        if greedy_path:
            print("\nGreedy Path Visualization:")
            self.print_maze(path=greedy_path)
        
        # Run Deadband Protocol
        time.sleep(1)
        deadband_path, safe_channels = self.deadband_protocol()
        if deadband_path and safe_channels:

            self.print_maze(path=deadband_path, deadband=safe_channels)

        print("\n" + "="*60)
        print("RESULTS SUMMARY")

        greedy_success = greedy_path is not None and greedy_path[-1] == self.goal

**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Implementing based on fleet-knowledge data. Using the simulation results (`optimal: 21`, `rocks: 64`) and the strict P0→P1→P2 priority.

**Approach:**
1. **P0 (Map negative space):** Generate a 20x20 maze with 64 rock positions (obstacles).
2. **P1 (Identify safe channels):** Use BFS from start to goal, avoiding rocks, to find all reachable safe cells.
3. **P2 (Optimize within channel):** Find shortest path (length 21) through safe cells.
4. **Visualize:** ASCII grid showing start (S), goal (G), rocks (#), safe path (.).

**Output:** Python script `deadband_maze_demo.py` with clear step-by-step comments.

---

```python
#!/usr/bin/env python3
"""
deadband_maze_demo.py

Demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.
Based on fleet-knowledge simulation data:
- optimal path length: 21
- rocks: 64
- 20x20 grid

Strict priority:
P0: Map rocks (negative space)
P1: Find safe channels (reachable cells avoiding rocks)
P2: Optimize shortest path within safe channels
"""

import random
from collections import deque

# ========== P0: MAP NEGATIVE SPACE ==========
def generate_rocks(grid_size=20, num_rocks=64):
    """Place rocks randomly (negative space)."""
    rocks = set()
    while len(rocks) < num_rocks:
        x = random.randint(0, grid_size - 1)
        y = random.randint(0, grid_size - 1)
        # Avoid start (0,0) and goal (19,19)
        if (x, y) != (0, 0) and (x, y) != (grid_size - 1, grid_size - 1):
            rocks.add((x, y))
    return rocks

# ========== P1: IDENTIFY SAFE CHANNELS ==========
def find_safe_channels(rocks, grid_size=20, start=(0,0), goal=(19,19)):
    """
    BFS to find all cells reachable from start without hitting rocks.
    Returns:
        reachable: set of safe cells (channel)
        parent_map: for path reconstruction
    """
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    queue = deque([start])
    reachable = {start}
    parent = {start: None}
    
    while queue:
        x, y = queue.popleft()
        if (x, y) == goal:
            break
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < grid_size and 0 <= ny < grid_size:
                if (nx, ny) not in rocks and (nx, ny) not in reachable:
                    reachable.add((nx, ny))
                    parent[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    return reachable, parent

# ========== P2: OPTIMIZE WITHIN CHANNEL ==========
def shortest_path_in_channel(reachable, parent, start=(0,0), goal=(19,19)):
    """Reconstruct shortest path if goal is reachable."""
    if goal not in reachable:
        return None
    path = []
    curr = goal
    while curr is not None:
        path.append(curr)
        curr = parent[curr]
    path.reverse()
    return path

# ========== VISUALIZATION ==========
def print_maze(rocks, path, grid_size=20):
    """ASCII visualization: S=start, G=goal, #=rock, .=path, ·=empty."""
    grid = [['·' for _ in range(grid_size)] for _ in range(grid_size)]
    for (x, y) in rocks:
        grid[y][x] = '#'
    for (x, y) in path:
        if (x, y) == (0, 0):
            grid[y][x] = 'S'
        elif (x, y) == (grid_size - 1, grid_size - 1):
            grid[y][x] = 'G'
        else:
            grid[y][x] = '.'
    # Ensure start/goal not overwritten by rocks
    grid[0][0] = 'S'
    grid[grid_size - 1][grid_size - 1] = 'G'
    
    print("=" * 40)
    print("MAZE (20x20) - Deadband Protocol Demo")
    print("=" * 40)
    for row in grid:
        print(' '.join(row))
    print(f"Rocks: {len(rocks)}")
    print(f"Path length: {len(path)-1} steps")
    print("=" * 40)

# ========== MAIN DEMONSTRATION ==========
def main():
    random.seed(42)  # Deterministic for demo
    grid_size = 20
    start = (0, 0)
    goal = (grid_size - 1, grid_size - 1)
    
    print("[P0] Mapping negative space (rocks)...")
    rocks = generate_rocks(grid_size, num_rocks=64)
    print(f"   → Mapped {len(rocks)} rock positions.")
    
    print("[P1] Identifying safe channels (BFS avoiding rocks)...")
    reachable, parent = find_safe_channels(rocks, grid_size, start, goal)
    print(f"   → Safe cells reachable from start: {len(reachable)}")
    
    print("[P2] Optimizing shortest path within safe channels...")
    path = shortest_path_in_channel(reachable, parent, start, goal)
    if path is None:
        print("   ❌ Goal unreachable — need new rock map (P0).")
        return
    print(f"   → Optimal path length: {len(path)-1} steps")
    
    # Verify against fleet-knowledge optimal (21)
    if len(path) - 1 == 21:
        print("   ✅ Matches fleet-knowledge optimal (21).")
    else:
        print(f"   ⚠️  Path length {len(path)-1} differs from optimal 21 (seed variation).")
    
    print("\n" + "="*40)
    print("PROTOCOL PRIORITY VERIFICATION:")
    print("  P0 done: rocks mapped")
    print("  P1 done: safe channels identified")
    print("  P2 done: shortest path found within channels")
    print("="*40 + "\n")
    
    print_maze(rocks, path, grid_size)
    
    # Show a few path coordinates
    print("Path coordinates (first 5 + last 5):")
    for i, (x, y) in enumerate(path[:5]):
        print(f"  {i}: ({x},{y})")
    print("  ...")
    for i, (x, y) in enumerate(path[-5:], start=len(path)-5):
        print(f"  {i}: ({x},{y})")

if __name__ == "__main__":
    main()
```

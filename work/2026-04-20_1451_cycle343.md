#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 343)
A visual ASCII simulation of the Deadband Protocol (P0→P1→P2).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json.
"""

import random
import time
import sys

# ========== CONFIGURATION ==========
MAZE_WIDTH = 20
MAZE_HEIGHT = 20
ROCK_COUNT = 64          # from JSON: "rocks": 64
OPTIMAL_PATH_LEN = 21    # from JSON: "optimal": 21
AGENT_START = (0, 0)
GOAL = (19, 19)

# Symbols
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_SYM = 'G'
PATH = '○'
SAFE = '░'
UNSAFE = '▒'

# ========== MAZE GENERATION ==========
def generate_maze():
    """Create a maze with random rocks (negative space)."""
    maze = [[EMPTY for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    rocks_placed = 0
    while rocks_placed < ROCK_COUNT:
        x = random.randint(0, MAZE_WIDTH-1)
        y = random.randint(0, MAZE_HEIGHT-1)
        if (x, y) != AGENT_START and (x, y) != GOAL and maze[y][x] == EMPTY:
            maze[y][x] = ROCK
            rocks_placed += 1
    maze[AGENT_START[1]][AGENT_START[0]] = AGENT
    maze[GOAL[1]][GOAL[0]] = GOAL_SYM
    return maze

def print_maze(maze, title="", delay=0.05):
    """Print the maze with a title."""
    print(f"\n{title}")
    print("═" * (MAZE_WIDTH * 2 + 2))
    for row in maze:
        print('│ ' + ' '.join(row) + ' │')
    print("═" * (MAZE_WIDTH * 2 + 2))
    if delay > 0:
        time.sleep(delay)

# ========== DEADBAND PROTOCOL ==========
def p0_map_negative_space(maze):
    """P0: Identify all rock locations (negative space)."""
    rocks = []
    for y in range(MAZE_HEIGHT):
        for x in range(MAZE_WIDTH):
            if maze[y][x] == ROCK:
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, rocks):
    """P1: Map safe channels (cells not adjacent to rocks)."""
    safe = [[True for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    for rx, ry in rocks:
        for dx in [-1, 0, 1]:
            for dy in [-1, 0, 1]:
                nx, ny = rx + dx, ry + dy
                if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                    safe[ny][nx] = False
    # Start and goal are always safe
    safe[AGENT_START[1]][AGENT_START[0]] = True
    safe[GOAL[1]][GOAL[0]] = True
    return safe

def p2_optimize_within_channel(maze, safe):
    """P2: Find shortest path within safe channels (BFS)."""
    from collections import deque
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    queue = deque([(AGENT_START, [AGENT_START])])
    visited = set([AGENT_START])
    
    while queue:
        (x, y), path = queue.popleft()
        if (x, y) == GOAL:
            return path
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                if safe[ny][nx] and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    return None  # No safe path

def greedy_unconstrained_path(maze):
    """P2-only (greedy) for comparison — violates protocol."""
    x, y = AGENT_START
    path = [(x, y)]
    while (x, y) != GOAL:
        # Move toward goal, ignoring rocks
        if x < GOAL[0]:
            x += 1
        elif x > GOAL[0]:
            x -= 1
        elif y < GOAL[1]:
            y += 1
        elif y > GOAL[1]:
            y -= 1
        path.append((x, y))
        if maze[y][x] == ROCK:
            return path  # Hit a rock
    return path

# ========== VISUALIZATION ==========
def visualize_safe_channels(maze, safe):
    """Create a visualization of safe channels (P1 result)."""
    viz = [[EMPTY for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    for y in range(MAZE_HEIGHT):
        for x in range(MAZE_WIDTH):
            if maze[y][x] == ROCK:
                viz[y][x] = ROCK
            elif not safe[y][x]:
                viz[y][x] = UNSAFE
            else:
                viz[y][x] = SAFE
    viz[AGENT_START[1]][AGENT_START[0]] = AGENT
    viz[GOAL[1]][GOAL[0]] = GOAL_SYM
    return viz

def visualize_path(maze, path, symbol=PATH):
    """Draw the path on the maze."""
    viz = [row[:] for row in maze]
    for (x, y) in path[1:-1]:  # Skip start and goal
        if viz[y][x] == EMPTY:
            viz[y][x] = symbol
    return viz

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo
A simple ASCII maze simulation demonstrating P0→P1→P2.
Fleet Doctrine: P0 map rocks, P1 find safe channels, P2 optimize within channels.
"""

import random
from collections import deque

def generate_maze(width=10, height=10, rock_density=0.2):
    """Generate a random maze with start, goal, and rocks."""
    grid = [['.' for _ in range(width)] for _ in range(height)]
    # Place rocks
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                grid[y][x] = '#'
    # Place start (S) and goal (G) ensuring they are not rocks
    start = (0, 0)
    while grid[start[1]][start[0]] == '#':
        start = (random.randint(0, width-1), random.randint(0, height-1))
    goal = (width-1, height-1)
    while grid[goal[1]][goal[0]] == '#':
        goal = (random.randint(0, width-1), random.randint(0, height-1))
    grid[start[1]][start[0]] = 'S'
    grid[goal[1]][goal[0]] = 'G'
    return grid, start, goal

def print_grid(grid, title=""):
    """Print the grid with a title."""
    print(f"\n{title}")
    for row in grid:
        print(' '.join(row))

def p0_map_rocks(grid):
    """P0: Map negative space (rocks). Return set of rock coordinates."""
    rocks = set()
    for y, row in enumerate(grid):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(grid, start, goal, rocks):
    """P1: Find all safe paths (channels) from start to goal using BFS.
       Returns a list of all safe paths (as coordinate lists)."""
    width, height = len(grid[0]), len(grid)
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    queue = deque()
    queue.append([start])
    safe_paths = []
    visited = {start}
    
    while queue:
        path = queue.popleft()
        x, y = path[-1]
        if (x, y) == goal:
            safe_paths.append(path)
            continue
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    new_path = path + [(nx, ny)]
                    queue.append(new_path)
    return safe_paths

def p2_optimize_within_channel(safe_paths):
    """P2: Choose the shortest path among safe channels.
       Returns the optimal path (list of coordinates)."""
    if not safe_paths:
        return []
    return min(safe_paths, key=len)

def visualize_path(grid, path, start, goal):
    """Create a copy of grid with the path marked with '*'."""
    vis_grid = [row[:] for row in grid]
    for (x, y) in path[1:-1]:  # Exclude start and goal
        if vis_grid[y][x] not in ('S', 'G'):
            vis_grid[y][x] = '*'
    return vis_grid

def main():
    print("=== Deadband Protocol Visual Demo ===")
    print("Doctrine: P0→P1→P2 (strict priority)")
    
    # Generate maze
    grid, start, goal = generate_maze(10, 10, rock_density=0.25)
    print_grid(grid, "Initial Maze (S=start, G=goal, #=rock, .=empty):")
    
    # P0: Map rocks
    rocks = p0_map_rocks(grid)
    print(f"\n[P0] Mapped negative space: {len(rocks)} rocks identified.")
    
    # P1: Find safe channels
    safe_paths = p1_find_safe_channels(grid, start, goal, rocks)
    print(f"[P1] Found {len(safe_paths)} safe channel(s).")
    
    if not safe_paths:
        print("[FAIL] No safe channels exist. Agent cannot proceed.")
        return
    
    # P2: Optimize within channel
    optimal_path = p2_optimize_within_channel(safe_paths)
    print(f"[P2] Selected optimal path length: {len(optimal_path)} steps.")
    
    # Visualize
    vis_grid = visualize_path(grid, optimal_path, start, goal)
    print_grid(vis_grid, "Optimal Path (*) within Safe Channels:")
    
    # Summary
    print("\n=== Protocol Summary ===")
    print(f"Start: {start}, Goal: {goal}")
    print(f"P0 rocks: {len(rocks)}")
    print(f"P1 safe channels: {len(safe_paths)}")
    print(f"P2 optimal steps: {len(optimal_path)-1}")  # Exclude start
    print("Doctrine followed: P0→P1→P2 (strict).")

if __name__ == "__main__":
    main()
```

**Explanation:**
1.  **P0 (Map Rocks):** The script scans the grid and records all `#` (rock) coordinates into a set. This is the negative space.
2.  **P1 (Find Safe Channels):** Using Breadth-First Search (BFS), it finds *all* possible paths from start (S) to goal (G) that avoid the rocks. These are the safe channels.
3.  **P2 (Optimize Within Channel):** From the list of safe paths, it selects the shortest one. Optimization occurs *only* within the constraint of safe channels identified in P1.
4.  **Visualization:** The final grid is printed with the optimal path marked by `*`.

This script is a direct, runnable implementation of the Deadband Protocol as defined in `DEADBAND-PROTOCOL.md`. It serves as a training tile demonstrating the core fleet doctrine.

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map rocks (negative space)
P1: Find safe channels (deadbands)
P2: Optimize path within safe channels
"""

import random
from typing import List, Tuple, Set

# Simulation parameters from fleet-knowledge
OPTIMAL_PATH_LENGTH = 21
ROCK_COUNT = 64
MAZE_SIZE = 20

class DeadbandMaze:
    def __init__(self, size: int = MAZE_SIZE):
        self.size = size
        self.start = (0, 0)
        self.goal = (size-1, size-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
    def p0_map_negative_space(self) -> None:
        """P0: Identify rocks (what NOT to do)"""
        # Generate random obstacles (rocks)
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, self.size-1)
            y = random.randint(0, self.size-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
        
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Identified {len(self.rocks)} rocks")
        print(f"Rocks cover {len(self.rocks)/(self.size*self.size)*100:.1f}% of maze")
    
    def p1_find_safe_channels(self) -> bool:
        """P1: Find all paths that avoid rocks"""
        # Simple BFS to find safe paths
        visited = set()
        queue = [(self.start, [self.start])]
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.goal:
                self.safe_channels.append(path)
                continue
            
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            
            # Try all 4 directions
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.size and 0 <= ny < self.size and 
                    (nx, ny) not in self.rocks and 
                    (nx, ny) not in visited):
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        print("\n=== P1: IDENTIFY SAFE CHANNELS ===")
        print(f"Found {len(self.safe_channels)} safe channels to goal")
        
        return len(self.safe_channels) > 0
    
    def p2_optimize_within_channel(self) -> None:
        """P2: Find shortest path within safe channels"""
        if not self.safe_channels:
            print("No safe channels found! Cannot optimize.")
            return
        
        # Find shortest safe path
        self.optimal_path = min(self.safe_channels, key=len)
        
        print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
        print(f"Optimal path length: {len(self.optimal_path)}")
        print(f"Theoretical optimal: {OPTIMAL_PATH_LENGTH}")
        print(f"Efficiency: {OPTIMAL_PATH_LENGTH/len(self.optimal_path)*100:.1f}%")
    
    def visualize(self, show_path: bool = False) -> None:
        """ASCII visualization of maze"""
        print("\n" + "="*40)
        print("MAZE VISUALIZATION")
        print("="*40)
        
        for y in range(self.size):
            row = []
            for x in range(self.size):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')
                elif show_path and pos in self.optimal_path:
                    row.append('·')
                else:
                    row.append(' ')
            print(' '.join(row))
        
        legend = [
            "S = Start",
            "G = Goal", 
            "█ = Rock (P0 negative space)",
            "· = Optimal path (P2 within safe channel)" if show_path else ""
        ]
        print("\n".join(filter(None, legend)))
    
    def run_protocol(self) -> None:
        """Execute full Deadband Protocol"""
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*50)
        
        # P0: Map negative space
        self.p0_map_negative_space()
        self.visualize(show_path=False)
        
        # P1: Find safe channels
        if not self.p1_find_safe_channels():
            print("ABORT: No safe channels found. Protocol complete.")
            return
        
        # P2: Optimize within channel
        self.p2_optimize_within_channel()
        self.visualize(show_path=True)
        
        # Protocol validation
        print("\n" + "="*50)
        print("PROTOCOL VALIDATION")
        print("-"*50)
        
        # Check P0 compliance: path avoids all rocks
        hits_rocks = any(pos in self.rocks for pos in self.optimal_path)
        print(f"P0 Compliance: {'✓' if not hits_rocks else '✗'} Path avoids all rocks")
        
        # Check P1 compliance: path is in safe channels
        in_channels = any(self.optimal_path == channel for channel in self.safe_channels)
        print(f"P1 Compliance: {'✓' if in_channels else '✗'} Path uses safe channel")
        
        # Check P2 compliance: path is optimal among safe options
        if self.safe_channels:
            is_optimal = len(self.optimal_path) == min(len(c) for c in self.safe_channels)
            print(f"P2 Compliance: {'✓' if is_optimal else '✗'} Path is shortest safe option")
        
        print("\nProtocol execution complete.")

def main():
    """Run the demonstration"""
    # Initialize with same seed for reproducibility
    random.seed(42)
    
    maze = DeadbandMaze()
    maze.run_protocol()
    
    # Show comparison with unconstrained approach
    print("\n" + "="*50)
    print("COMPARISON: Unconstrained vs Deadband Protocol")
    print("-"*50)
    print("From fleet-knowledge data:")
    print(f"  Unconstrained success: 33/50 (66%)")
    print(f"  Deadband success: 50/50 (100%)")
    print(f"  Unconstrained avg steps: 2320.9")
    print(f"  Deadband avg steps: 21.0")
    print("\nKey insight: P0+P1 mapping enables P2 optimization")
    print("Skipping to P2 (greedy optimization) hits rocks.")

if __name__ == "__main__":
    main()
```

**Key Features:**
1. **P0 Implementation**: Randomly places 64 rocks (negative space) as per simulation data
2. **P1 Implementation**: BFS finds all safe paths avoiding rocks (deadbands)
3. **P2 Implementation**: Selects shortest path among safe options
4. **Visualization**: ASCII maze showing rocks, start, goal, and optimal path
5. **Protocol Validation**: Checks compliance with each protocol step
6. **Data-Driven**: Uses actual fleet simulation parameters (optimal=21, rocks=64)

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 strict priority in a 20x20 maze.
Based on fleet simulation data: optimal=21, rocks=64.
"""

import random
from typing import List, Tuple, Set

# Simulation parameters from fleet-knowledge
MAZE_SIZE = 20
OPTIMAL_PATH_LENGTH = 21  # From constraint_sim_final.json
ROCK_COUNT = 64           # 64 rocks in 20x20 grid

class DeadbandMaze:
    """Implements the Deadband Protocol in a maze environment."""
    
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.best_path: List[Tuple[int, int]] = []
        
    def p0_map_negative_space(self) -> None:
        """P0: Identify rocks (what NOT to do)."""
        # Place rocks randomly (but ensure start/goal are clear)
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[x][y] = '#'
        
        print("P0: Mapped negative space")
        print(f"  - Rocks identified: {len(self.rocks)}")
        print(f"  - Start: {self.start}, Goal: {self.goal}")
    
    def p1_find_safe_channels(self) -> bool:
        """P1: Find all paths that avoid rocks."""
        # Simple BFS to find safe paths (deadbands)
        visited = set()
        queue = [(self.start, [self.start])]
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.goal:
                self.safe_channels.append(path)
                continue
            
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Explore neighbors (up, down, left, right)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in self.rocks:
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        has_channels = len(self.safe_channels) > 0
        print(f"P1: Found safe channels: {len(self.safe_channels)}")
        return has_channels
    
    def p2_optimize_within_channel(self) -> None:
        """P2: Find shortest path within safe channels."""
        if not self.safe_channels:
            print("P2 ERROR: No safe channels mapped (P1 not completed)")
            return
        
        self.best_path = min(self.safe_channels, key=len)
        print(f"P2: Optimized within channels")
        print(f"  - Best path length: {len(self.best_path)}")
        print(f"  - Optimal (from sim): {OPTIMAL_PATH_LENGTH}")
        print(f"  - Efficiency: {len(self.best_path)/OPTIMAL_PATH_LENGTH:.1%}")
    
    def run_protocol(self) -> bool:
        """Execute P0→P1→P2 in strict order."""
        print("=" * 50)
        print("DEADBAND PROTOCOL DEMO - 20x20 Maze")
        print("=" * 50)
        
        # P0: Map negative space (mandatory first step)
        self.p0_map_negative_space()
        if not self.rocks:
            print("ERROR: P0 failed - no rocks identified")
            return False
        
        # P1: Find safe channels (mandatory second step)
        if not self.p1_find_safe_channels():
            print("ERROR: P1 failed - no safe channels found")
            return False
        
        # P2: Optimize within channels (only after P0+P1)
        self.p2_optimize_within_channel()
        
        return True
    
    def visualize(self, show_path: bool = True) -> None:
        """Display maze with optional best path."""
        display_grid = [row[:] for row in self.grid]
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        display_grid[sx][sy] = 'S'
        display_grid[gx][gy] = 'G'
        
        # Mark best path if requested
        if show_path and self.best_path:
            for x, y in self.best_path[1:-1]:  # Skip start and goal
                if display_grid[x][y] == '.':
                    display_grid[x][y] = '*'
        
        print("\nMAZE VISUALIZATION:")
        print("  S = Start, G = Goal, # = Rock, * = Safe Path, . = Open")
        print("-" * (MAZE_SIZE * 2 + 3))
        
        for y in range(MAZE_SIZE):
            row = ' '.join(display_grid[x][y] for x in range(MAZE_SIZE))
            print(f"| {row} |")
        
        print("-" * (MAZE_SIZE * 2 + 3))
    
    def simulate_violation(self) -> None:

        print("\n" + "=" * 50)
        print("PROTOCOL VIOLATION: Greedy P2-only (no P0+P1)")
        print("=" * 50)
        
        # Greedy approach: head toward goal without rock awareness
        path = [self.start]
        x, y = self.start
        visited = set([self.start])
        
        steps = 0
        max_steps = MAZE_SIZE * 2  # Prevent infinite loops
        
        while (x, y) != self.goal and steps < max_steps:
            # Move toward goal (greedy)
            dx = 1 if self.goal[0] > x else -1 if self.goal[0] < x else 0
            dy = 1 if self.goal[1] > y else -1 if self.goal[1] < y else 0
            
            nx, ny = x + dx, y + dy
            
            # Check bounds
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) in self.rocks:
                    print(f"  Step {steps}: HIT ROCK at ({nx}, {ny})")
                    # Try random direction
                    directions = [(1,0), (-1,0), (0,1), (0,-1)]
                    random.shuffle(directions)
                    for dx2, dy2 in directions:
                        nx2, ny2 = x + dx2, y + dy2
                        if (0 <= nx2 < MAZE_SIZE and 0 <= ny2 < MAZE_SIZE and 
                            (nx2, ny2) not in self.rocks and

                            nx, ny = nx2, ny2
                            break
                    else:
                        print("  TRAPPED - No safe move available")
                        break
                
                x, y = nx, ny
                path.append((x, y))

        print(f"\n  Result: {'SUCCESS' if success else 'FAILURE'}")

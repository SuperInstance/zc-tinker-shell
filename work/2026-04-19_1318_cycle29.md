**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on the fleet knowledge, I have the constraint simulation data and the Deadband Protocol definition. I will implement a concrete demonstration using a small grid maze, showing the three-step process.

**Approach (P0→P1→P2):**
1. **P0 (Map negative space):** Identify obstacles ('#') in the maze.
2. **P1 (Find safe channels):** Compute all reachable empty cells ('.') from the start, avoiding obstacles.
3. **P2 (Optimize within channels):** Find the shortest path from start to goal using only safe cells.

**Output:** A Python script that prints each step visually.

**File:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 on a simple ASCII maze.
"""

import sys
from collections import deque

# Example maze (5x7) from constraint theory simulation spirit
# 'S' = start, 'G' = goal, '#' = rock/obstacle, '.' = open
MAZE = [
    ['S', '.', '.', '#', '.', '.', '.'],
    ['.', '#', '.', '#', '.', '#', '.'],
    ['.', '#', '.', '.', '.', '#', 'G'],
    ['.', '#', '#', '#', '.', '#', '.'],
    ['.', '.', '.', '.', '.', '.', '.'],
]

HEIGHT = len(MAZE)
WIDTH = len(MAZE[0])

def find_char(ch):
    """Return (r,c) of first occurrence of ch."""
    for r in range(HEIGHT):
        for c in range(WIDTH):
            if MAZE[r][c] == ch:
                return (r, c)
    return None

def print_maze(title, overlay=None):
    """Print maze with optional overlay markers."""
    print(f"\n=== {title} ===")
    for r in range(HEIGHT):
        row = []
        for c in range(WIDTH):
            if overlay and (r, c) in overlay:
                row.append(overlay[(r, c)])
            else:
                row.append(MAZE[r][c])
        print(' '.join(row))

def p0_map_negative_space():
    """P0: Identify all obstacle cells (rocks)."""
    rocks = []
    for r in range(HEIGHT):
        for c in range(WIDTH):
            if MAZE[r][c] == '#':
                rocks.append((r, c))
    print_maze("P0 - Negative Space (Rocks)", {(r,c): '#' for (r,c) in rocks})
    print(f"Rocks mapped: {len(rocks)} cells")
    return set(rocks)

def p1_find_safe_channels(start, rocks):
    """P1: Breadth-first search to find all reachable empty cells."""
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    safe = set()
    queue = deque([start])
    safe.add(start)
    
    while queue:
        r, c = queue.popleft()
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < HEIGHT and 0 <= nc < WIDTH:
                if (nr, nc) not in rocks and (nr, nc) not in safe:
                    # Treat 'G' as reachable endpoint
                    if MAZE[nr][nc] != '#':
                        safe.add((nr, nc))
                        queue.append((nr, nc))
    
    print_maze("P1 - Safe Channels (Reachable)", {(r,c): 'O' for (r,c) in safe if MAZE[r][c] not in 'SG'})
    print(f"Safe cells found: {len(safe)}")
    return safe

def p2_optimize_within_channels(start, goal, safe):
    """P2: Shortest path BFS constrained to safe cells."""
    if goal not in safe:
        print("Goal not reachable within safe channels.")
        return []
    
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    parent = {}
    queue = deque([start])
    parent[start] = None
    
    while queue:
        r, c = queue.popleft()
        if (r, c) == goal:
            break
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if (nr, nc) in safe and (nr, nc) not in parent:
                parent[(nr, nc)] = (r, c)
                queue.append((nr, nc))
    
    # Reconstruct path
    if goal not in parent:
        return []
    path = []
    cur = goal
    while cur is not None:
        path.append(cur)
        cur = parent[cur]
    path.reverse()
    
    print_maze("P2 - Optimal Path (Within Channels)", {pos: '*' for pos in path if pos not in (start, goal)})
    print(f"Path length: {len(path)-1} steps")
    return path

def main():
    print("Deadband Protocol Maze Demonstration")
    print("Based on fleet-knowledge/DEADBAND-PROTOCOL.md")
    print("=" * 50)
    
    start = find_char('S')
    goal = find_char('G')
    if not start or not goal:
        print("Error: Start or goal not found.")
        sys.exit(1)
    
    print_maze("Initial Maze")
    print(f"Start: {start}, Goal: {goal}")
    
    # P0: Map rocks
    rocks = p0_map_negative_space()
    
    # P1: Find safe channels
    safe = p1_find_safe_channels(start, rocks)
    
    # P2: Optimize within safe channels
    path = p2_optimize_within_channels(start, goal, safe)
    
    # Summary
    print("\n=== Protocol Summary ===")
    print(f"P0 rocks: {len(rocks)}")
    print(f"P1 safe cells: {len(safe)}")
    print(f"P2 path steps: {len(path)-1 if path else 'No path'}")
    print(f"Success: {'Yes' if path else 'No'}")

if __name__ == "__main__":
    main()
```

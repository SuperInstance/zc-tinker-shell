#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path) in a 20x20 ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        """
        Initialize maze with rocks (obstacles) and open cells.
        rock_density ~ 64 rocks / 400 cells = 0.16 (from constraint_sim_final.json)
        """
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self.rocks = set()
        self.generate_rocks(rock_density)
        
        # P1: Safe channels (deadbands) — will be computed
        self.safe_channels = None
        
        # P2: Optimal path within safe channels
        self.optimal_path = None
        
        # Statistics
        self.stats = {
            'rocks_count': len(self.rocks),
            'p0_complete': False,
            'p1_complete': False,
            'p2_complete': False,
            'path_length': 0,
            'violation_attempted': False
        }
    
    def generate_rocks(self, density):
        """Randomly place rocks, ensuring start and goal are clear."""
        cells = [(x, y) for x in range(self.width) for y in range(self.height)]
        cells.remove(self.start)
        cells.remove(self.goal)
        
        num_rocks = int(len(cells) * density)
        self.rocks = set(random.sample(cells, num_rocks))
    
    def p0_map_negative_space(self):
        """P0: Identify all rocks (states where failure probability > threshold)."""
        self.stats['p0_complete'] = True
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all cells reachable from start without hitting rocks."""
        if not self.stats['p0_complete']:
            print("VIOLATION: Attempted P1 before P0!")
            self.stats['violation_attempted'] = True
            return set()
        
        visited = set()
        queue = deque([self.start])
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Check four-directional neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) not in self.rocks and
                    (nx, ny) not in visited):
                    queue.append((nx, ny))
        
        self.safe_channels = visited
        self.stats['p1_complete'] = True
        return visited
    
    def p2_optimize_within_channel(self):
        """P2: Find shortest path from start to goal within safe channels."""
        if not self.stats['p1_complete']:
            print("VIOLATION: Attempted P2 before P1!")
            self.stats['violation_attempted'] = True
            return []
        
        # BFS for shortest path
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                self.optimal_path = path
                self.stats['p2_complete'] = True
                self.stats['path_length'] = len(path) - 1  # steps between cells
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) in self.safe_channels and
                    (nx, ny) not in visited):
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        # No path found
        self.optimal_path = []
        return []
    
    def run_deadband_protocol(self):
        """Execute full P0→P1→P2 sequence."""
        print("=== DEADBAND PROTOCOL DEMO ===")
        print(f"Maze: {self.width}x{self.height}, Rocks: {self.stats['rocks_count']}")
        print()
        
        # P0
        rocks = self.p0_map_negative_space()
        print("P0: Mapped negative space")
        print(f"    Identified {len(rocks)} rocks (failure states)")
        
        # P1
        safe = self.p1_find_safe_channels()
        print("P1: Found safe channels")
        print(f"    {len(safe)} reachable cells without rocks")
        print(f"    Goal reachable: {self.goal in safe}")
        
        # P2
        path = self.p2_optimize_within_channel()
        if path:
            print("P2: Optimized within safe channels")
            print(f"    Optimal path length: {len(path)-1} steps")
        else:
            print("P2: No path to goal within safe channels")
        
        return path
    
    def visualize(self, show_path=True, show_rocks=True):
        """Create ASCII visualization of maze."""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif show_rocks and pos in self.rocks:
                    row.append('#')
                elif show_path and self.optimal_path and pos in self.optimal_path:
                    row.append('·')
                elif self.safe_channels and pos in self.safe_channels:
                    row.append(' ')
                else:
                    row.append('░')
            grid.append(''.join(row))
        return '\n'.join(grid)
    
    def run_greedy_violation(self):
        """Demonstrate what happens when violating P0→P1→P2 (greedy P2-only)."""
        print("\n=== GREEDY VIOLATION (P2-only) ===")
        print("Attempting to find shortest path without mapping rocks first...")
        
        # Greedy BFS ignoring rocks
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                print(f"    Found path of length {len(path)-1} steps")
                print("    BUT path may go through rocks!")
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) not in visited):
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        print("    No path found (unlikely in open maze)")
        return []

def main():
    """Run the demonstration."""
    # Create maze matching constraint_sim_final.json parameters
    maze = DeadbandMaze(width=20, height=20, rock_density=0.16)
    
    # Run proper deadband protocol
    path = maze.run_deadband_protocol()

        print(f"{key}: {value}")

    maze.run_greedy_violation()

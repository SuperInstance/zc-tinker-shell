**Status:** Based on `constraint_sim_final.json` (20×20 maze, 64 rocks, optimal path length 21). Implementing a simulation that shows P0 (map rocks), P1 (find safe channels), P2 (optimize within channels).

**Approach:**
1. Generate a 20×20 maze with 64 random rock positions.
2. Define start (0,0) and goal (19,19).
3. Implement three agent strategies:
   - **Unconstrained (P2-only):** Greedy path toward goal ignoring rocks.
   - **Deadband (P0→P1→P2):** First map rocks (P0), then find safe channels (P1), then A* within safe channels (P2).
4. Visualize each step with ASCII.

**Output:** `deadband_maze_demo.py`

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Simulates P0→P1→P2 in a 20×20 maze with 64 rocks.
"""

import random
import heapq
from collections import deque

# Constants from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21
START = (0, 0)
GOAL = (19, 19)

def generate_maze():
    """Generate 20×20 maze with 64 random rocks."""
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        rocks.add((random.randint(0, MAZE_SIZE-1), random.randint(0, MAZE_SIZE-1)))
    # Ensure start and goal are not rocks
    rocks.discard(START)
    rocks.discard(GOAL)
    return rocks

def print_maze(rocks, path=None, channels=None, title="Maze"):
    """Print ASCII maze."""
    print(f"\n=== {title} ===")
    for y in range(MAZE_SIZE):
        row = []
        for x in range(MAZE_SIZE):
            if (x, y) == START:
                row.append('S')
            elif (x, y) == GOAL:
                row.append('G')
            elif (x, y) in rocks:
                row.append('#')
            elif path and (x, y) in path:
                row.append('*')
            elif channels and (x, y) in channels:
                row.append('.')
            else:
                row.append(' ')
        print(' '.join(row))

def unconstrained_greedy(rocks):
    """P2-only agent: greedy toward goal, ignoring rocks."""
    path = [START]
    x, y = START
    while (x, y) != GOAL:
        # Move toward goal (prefer diagonal)
        dx = 1 if x < GOAL[0] else -1 if x > GOAL[0] else 0
        dy = 1 if y < GOAL[1] else -1 if y > GOAL[1] else 0
        nx, ny = x + dx, y + dy
        # If diagonal blocked, try cardinal
        if (nx, ny) in rocks:
            if dx != 0 and (x + dx, y) not in rocks:
                ny = y
            elif dy != 0 and (x, y + dy) not in rocks:
                nx = x
            else:
                # Stuck
                break
        path.append((nx, ny))
        x, y = nx, ny
        if len(path) > MAZE_SIZE * 2:  # Prevent infinite loops
            break
    return path

def deadband_protocol(rocks):
    """P0→P1→P2 agent: map rocks, find safe channels, optimize."""
    # P0: Map negative space (rocks are already known)
    print_maze(rocks, title="P0: Map Rocks (Negative Space)")
    
    # P1: Find safe channels (connected free spaces)
    visited = set()
    channels = []
    for y in range(MAZE_SIZE):
        for x in range(MAZE_SIZE):
            if (x, y) not in rocks and (x, y) not in visited:
                # BFS to find connected component
                queue = deque([(x, y)])
                component = set()
                while queue:
                    cx, cy = queue.popleft()
                    if (cx, cy) in component:
                        continue
                    component.add((cx, cy))
                    visited.add((cx, cy))
                    for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
                        nx, ny = cx + dx, cy + dy
                        if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                            if (nx, ny) not in rocks and (nx, ny) not in component:
                                queue.append((nx, ny))
                channels.append(component)
    
    # Find channel containing start and goal
    start_channel = None
    goal_channel = None
    for i, ch in enumerate(channels):
        if START in ch:
            start_channel = i
        if GOAL in ch:
            goal_channel = i
    
    safe_cells = set()
    if start_channel is not None and goal_channel is not None and start_channel == goal_channel:
        safe_cells = channels[start_channel]
    
    print_maze(rocks, channels=safe_cells, title="P1: Safe Channels (Deadbands)")
    
    # P2: Optimize within safe channel (A*)
    if not safe_cells:
        return []
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, START))
    came_from = {}
    g_score = {START: 0}
    f_score = {START: heuristic(START, GOAL)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == GOAL:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(START)
            return list(reversed(path))
        
        for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
            neighbor = (current[0] + dx, current[1] + dy)
            if neighbor not in safe_cells:
                continue
            tentative_g = g_score[current] + 1
            if neighbor not in g_score or tentative_g < g_score[neighbor]:
                came_from[neighbor] = current
                g_score[neighbor] = tentative_g
                f_score[neighbor] = tentative_g + heuristic(neighbor, GOAL)
                heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return []

    # Deadband Protocol agent
    print("\n--- Deadband Protocol Agent (P0→P1→P2) ---")

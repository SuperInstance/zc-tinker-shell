#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Tinker - Cocapn Fleet - Cycle 9
Demonstrates P0→P1→P2 priority in pathfinding.
"""

import random
from collections import deque
import heapq

class DeadbandMaze:
    def __init__(self, width=10, height=10, rock_density=0.25):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.rocks = set()
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self._place_rocks(rock_density)
        
        # Mark start and goal
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def _place_rocks(self, density):
        """P0: Identify rocks (negative space)."""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        
        # Exclude start and goal
        available = [(x, y) for x in range(self.width) for y in range(self.height)
                     if (x, y) not in [self.start, self.goal]]
        rock_positions = random.sample(available, min(num_rocks, len(available)))
        
        for x, y in rock_positions:
            self.grid[y][x] = 'X'
            self.rocks.add((x, y))
    
    def print_grid(self, title="Maze"):
        """ASCII visualization."""
        print(f"\n=== {title} ===")
        for row in self.grid:
            print(' '.join(row))
        print(f"Rocks: {len(self.rocks)}")
    
    def unconstrained_path(self):
        """P2-only (greedy) - violates protocol."""
        path = []
        x, y = self.start
        visited = set()
        
        while (x, y) != self.goal:
            path.append((x, y))
            visited.add((x, y))
            
            # Greedy: move toward goal
            dx = 1 if self.goal[0] > x else -1 if self.goal[0] < x else 0
            dy = 1 if self.goal[1] > y else -1 if self.goal[1] < y else 0
            
            # Try horizontal first, then vertical
            next_x, next_y = x + dx, y
            if 0 <= next_x < self.width and (next_x, next_y) not in visited:
                if (next_x, next_y) in self.rocks:
                    # Hit a rock - P2-only fails
                    return path, False
                x, y = next_x
                continue
            
            next_x, next_y = x, y + dy
            if 0 <= next_y < self.height and (next_x, next_y) not in visited:
                if (next_x, next_y) in self.rocks:
                    return path, False
                x, y = next_x, next_y
                continue
            
            # Stuck
            return path, False
        
        path.append(self.goal)
        return path, True
    
    def find_safe_channels(self):
        """P1: Find all safe cells (deadband)."""
        safe = set()
        queue = deque([self.start])
        visited = set([self.start])
        
        while queue:
            x, y = queue.popleft()
            safe.add((x, y))
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) not in visited and (nx, ny) not in self.rocks):
                    visited.add((nx, ny))
                    queue.append((nx, ny))
        
        return safe
    
    def deadband_path(self):
        """P0→P1→P2: Full protocol."""
        # P0 already done (rocks mapped in __init__)
        
        # P1: Find safe channels
        safe = self.find_safe_channels()
        
        # P2: Optimize within safe channels (A*)
        def heuristic(a, b):
            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        open_set = []
        heapq.heappush(open_set, (0, self.start))
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: heuristic(self.start, self.goal)}
        
        while open_set:
            _, current = heapq.heappop(open_set)
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                path.reverse()
                return path, True
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # Constrained to safe channels
                if neighbor not in safe:
                    continue
                
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, self.goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
        
        return [], False  # No path found
    
    def visualize_path(self, path, label):
        """Show path on grid."""
        grid_copy = [row[:] for row in self.grid]
        
        for x, y in path[1:-1]:  # Skip start and goal
            if grid_copy[y][x] == '.':
                grid_copy[y][x] = '*'
        
        print(f"\n=== {label} ===")
        for row in grid_copy:
            print(' '.join(row))
        print(f"Path length: {len(path)-1} steps")

**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on reading `fleet-knowledge/narrow-games/constraint_sim_final.json` and `fleet-knowledge/docs/DEADBAND-PROTOCOL.md`, I will implement a concrete simulation. The data shows constrained-only (P2) fails (0% success), while deadband protocol (P0→P1→P2) achieves 50/50 success at optimal cost (21 steps). I will model this.

**Approach:**
1. **P0 (Map negative space):** Generate a 20x20 maze with random obstacles ("rocks"). Identify all rock coordinates.
2. **P1 (Find safe channels):** From start (0,0) to goal (19,19), compute all paths that avoid rocks using BFS. These are safe channels.
3. **P2 (Optimize within channel):** From the set of safe paths, select the shortest one (optimal path).
4. **Visualization:** Print the maze with start (S), goal (G), rocks (#), and the chosen optimal safe path (.).

**Output:** A Python script `deadband_maze_demo.py` that runs the protocol and prints results.

## File: `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 105)
Simulates P0→P1→P2 in a 20x20 maze based on fleet constraint_sim_final.json data.
"""

import random
from collections import deque

# Constants from fleet-knowledge/narrow-games/constraint_sim_final.json
MAZE_SIZE = 20
OPTIMAL_STEPS = 21
ROCK_COUNT = 64
START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

def generate_maze(size, rocks):
    """P0: Map negative space — generate rocks."""
    maze = [[' ' for _ in range(size)] for _ in range(size)]
    rock_positions = set()
    while len(rock_positions) < rocks:
        x = random.randint(0, size - 1)
        y = random.randint(0, size - 1)
        if (x, y) != START and (x, y) != GOAL:
            rock_positions.add((x, y))
            maze[y][x] = '#'
    maze[START[1]][START[0]] = 'S'
    maze[GOAL[1]][GOAL[0]] = 'G'
    return maze, rock_positions

def find_safe_channels(maze, rocks, start, goal):
    """P1: Find safe channels — BFS to get all paths avoiding rocks."""
    rows = cols = MAZE_SIZE
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    queue = deque([(start, [start])])
    all_safe_paths = []
    visited = {start}
    
    while queue:
        (x, y), path = queue.popleft()
        if (x, y) == goal:
            all_safe_paths.append(path)
            continue
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < cols and 0 <= ny < rows:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    return all_safe_paths

def optimize_within_channel(safe_paths):
    """P2: Optimize within channel — pick shortest safe path."""
    if not safe_paths:
        return None
    return min(safe_paths, key=len)

def visualize_maze(maze, optimal_path):
    """Print maze with path overlay."""
    vis = [row[:] for row in maze]
    for (x, y) in optimal_path[1:-1]:  # exclude start and goal
        if vis[y][x] == ' ':
            vis[y][x] = '.'
    print("Deadband Protocol Maze (20x20)")
    print("S = start, G = goal, # = rock, . = safe optimal path")
    print("-" * (MAZE_SIZE * 2 + 1))
    for row in vis:
        print("|" + " ".join(row) + "|")
    print("-" * (MAZE_SIZE * 2 + 1))

def main():
    print("=== Deadband Protocol Demonstration ===")
    print("P0: Generating maze with 64 rocks (negative space)...")
    maze, rocks = generate_maze(MAZE_SIZE, ROCK_COUNT)
    
    print("P1: Finding safe channels (paths avoiding rocks)...")
    safe_paths = find_safe_channels(maze, rocks, START, GOAL)
    print(f"   Found {len(safe_paths)} safe channel(s).")
    
    print("P2: Optimizing within channel (shortest safe path)...")
    optimal = optimize_within_channel(safe_paths)
    
    if optimal is None:
        print("FAILURE: No safe channel exists — all paths hit rocks.")
        print("This matches constrained-only failure (0% success).")
        return
    
    steps = len(optimal) - 1  # path includes start node
    print(f"   Optimal safe path length: {steps} steps")
    print(f"   Fleet optimal benchmark: {OPTIMAL_STEPS} steps")
    
    visualize_maze(maze, optimal)
    
    # Protocol validation
    if steps >= OPTIMAL_STEPS:
        print("SUCCESS: Deadband protocol followed. Safe path found.")
        print("Matches fleet data: deadband success with optimal/near-optimal cost.")
    else:
        print("NOTE: Path shorter than benchmark — maze less constrained than simulation.")

if __name__ == "__main__":
    random.seed(42)  # reproducible for demo
    main()
```

3. **P2:** Optimize within the safe channel to find the shortest path.

## File: `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 in a 10x10 ASCII maze.
Based on fleet-knowledge/docs/DEADBAND-PROTOCOL.md
"""

import sys
from collections import deque

# Maze definition: '.' = open, '#' = rock (obstacle), 'S' = start, 'G' = goal
MAZE = [
    "S..#......",
    ".#.#.#..#.",
    "....#...#.",
    "##.#.#.#..",
    "...#....#.",
    ".#.##.#..#",
    "....#.....",
    "#..#.#.##.",
    ".#........",
    "......#..G"
]

HEIGHT = len(MAZE)
WIDTH = len(MAZE[0])

def find_char(maze, char):
    """Find coordinates of a character in the maze."""
    for y in range(HEIGHT):
        for x in range(WIDTH):
            if maze[y][x] == char:
                return (x, y)
    return None

def p0_map_rocks(maze):
    """P0: Map negative space (rocks)."""
    rocks = []
    for y in range(HEIGHT):
        for x in range(WIDTH):
            if maze[y][x] == '#':
                rocks.append((x, y))
    print(f"[P0] Mapped {len(rocks)} rocks (negative space).")
    return set(rocks)

def p1_find_safe_channels(maze, start, goal, rocks):
    """P1: Find safe channels (paths avoiding rocks) using BFS."""
    # Directions: up, down, left, right
    directions = [(0, -1), (0, 1), (-1, 0), (1, 0)]
    
    queue = deque([(start, [start])])
    visited = {start}
    safe_paths = []
    
    while queue:
        (x, y), path = queue.popleft()
        
        if (x, y) == goal:
            safe_paths.append(path)
            continue
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < WIDTH and 0 <= ny < HEIGHT:
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    print(f"[P1] Found {len(safe_paths)} safe channels (paths avoiding rocks).")
    return safe_paths

def p2_optimize_within_channel(safe_paths):
    """P2: Optimize within safe channels (shortest path)."""
    if not safe_paths:
        print("[P2] No safe channels available. Cannot optimize.")
        return None
    
    shortest_path = min(safe_paths, key=len)
    print(f"[P2] Selected optimal path length: {len(shortest_path)-1} steps.")
    return shortest_path

def visualize_maze(maze, path=None, rocks=None):
    """Display the maze with optional path overlay."""
    display = [list(row) for row in maze]
    
    if path:
        for (x, y) in path[1:-1]:  # Exclude start and goal
            if display[y][x] == '.':
                display[y][x] = '*'
    
    print("\n=== MAZE VISUALIZATION ===")
    for row in display:
        print(''.join(row))
    if rocks:
        print(f"Rocks: {len(rocks)}")
    if path:
        print(f"Path length: {len(path)-1} steps")
    print("==========================\n")

def main():
    print("Deadband Protocol Maze Demo")
    print("P0 → P1 → P2 strict priority\n")
    
    # Find start and goal
    start = find_char(MAZE, 'S')
    goal = find_char(MAZE, 'G')
    
    if not start or not goal:
        print("Error: Start or goal not found in maze.")
        sys.exit(1)
    
    print(f"Start: {start}, Goal: {goal}")
    visualize_maze(MAZE)
    
    # P0: Map rocks
    rocks = p0_map_rocks(MAZE)
    
    # P1: Find safe channels
    safe_paths = p1_find_safe_channels(MAZE, start, goal, rocks)
    
    if not safe_paths:
        print("No safe path exists. Protocol stops at P1.")
        return
    
    # P2: Optimize within channel
    optimal_path = p2_optimize_within_channel(safe_paths)
    
    # Visualize result
    visualize_maze(MAZE, optimal_path, rocks)
    
    # Protocol summary
    print("=== DEADBAND PROTOCOL SUMMARY ===")
    print("P0: Mapped negative space (rocks) — what NOT to do.")
    print("P1: Identified safe channels — where we CAN move.")
    print("P2: Optimized within safe channel — shortest path.")
    print("=================================")

if __name__ == "__main__":
    main()
```

## Key Takeaways

- The script follows strict P0→P1→P2 priority as per Deadband Protocol.
- **P0** must complete before **P1** (cannot find paths without knowing rocks).
- **P1** must complete before **P2** (cannot optimize without safe channels).
- This mirrors the constraint simulation results where deadband protocol achieved 50/50 success vs. 0/50 for unconstrained optimization.

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Simulates P0→P1→P2 in a 20x20 maze with rocks.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time
from typing import List, Tuple, Set

# Simulation parameters from constraint_sim_final.json
MAZE_SIZE = 20
NUM_ROCKS = 64
OPTIMAL_PATH_LENGTH = 21

# Symbols for visualization
ROCK = '█'
AGENT = 'A'
GOAL = 'G'
PATH = '·'
VISITED = 'x'
EMPTY = ' '

class Maze:
    """20x20 maze with rocks, start at (0,0), goal at (19,19)"""
    
    def __init__(self):
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self._generate_rocks()
        
    def _generate_rocks(self):
        """Place rocks randomly, ensuring start and goal are clear"""
        possible = [(x, y) for x in range(MAZE_SIZE) for y in range(MAZE_SIZE)]
        possible.remove(self.start)
        possible.remove(self.goal)
        
        self.rocks = set(random.sample(possible, NUM_ROCKS))
        for x, y in self.rocks:
            self.grid[y][x] = ROCK
            
        self.grid[self.start[1]][self.start[0]] = AGENT
        self.grid[self.goal[1]][self.goal[0]] = GOAL
    
    def is_rock(self, x: int, y: int) -> bool:
        return (x, y) in self.rocks
    
    def is_valid(self, x: int, y: int) -> bool:
        return 0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE
    
    def display(self, agent_pos=None, visited=None):
        """Display maze with optional agent position and visited markers"""
        display_grid = [row[:] for row in self.grid]
        
        if visited:
            for x, y in visited:
                if (x, y) not in [self.start, self.goal] and not self.is_rock(x, y):
                    display_grid[y][x] = VISITED
        
        if agent_pos:
            x, y = agent_pos
            if (x, y) != self.start:
                display_grid[y][x] = AGENT
        
        print("+" + "-" * MAZE_SIZE + "+")
        for row in display_grid:
            print("|" + "".join(row) + "|")
        print("+" + "-" * MAZE_SIZE + "+")
        print(f"Legend: {AGENT}=Agent, {GOAL}=Goal, {ROCK}=Rock, {VISITED}=Visited")
        print()

class Agent:
    """Base agent with movement capabilities"""
    
    def __init__(self, maze: Maze):
        self.maze = maze
        self.pos = maze.start
        self.path = [maze.start]
        self.visited = set([maze.start])
        self.steps = 0
        self.success = False
        
    def move_toward_goal(self) -> bool:
        """Move one step toward goal using Manhattan distance heuristic"""
        x, y = self.pos
        gx, gy = self.maze.goal
        
        # Possible moves (right, down, left, up)
        moves = [(x+1, y), (x, y+1), (x-1, y), (x, y-1)]
        valid_moves = [(nx, ny) for nx, ny in moves 
                      if self.maze.is_valid(nx, ny) 
                      and not self.maze.is_rock(nx, ny)]
        
        if not valid_moves:
            return False
            
        # Choose move that minimizes distance to goal
        best_move = min(valid_moves, 
                       key=lambda p: abs(p[0]-gx) + abs(p[1]-gy))
        
        self.pos = best_move
        self.path.append(best_move)
        self.visited.add(best_move)
        self.steps += 1
        
        if self.pos == self.maze.goal:
            self.success = True
            return False
            
        return True

class UnconstrainedAgent(Agent):
    """P2-only agent: Greedy optimization without rock mapping"""

        print("Strategy: Greedy move toward goal, no rock awareness")

            if self.steps % 5 == 0:

                self.maze.display(agent_pos=self.pos, visited=self.visited)
                time.sleep(0.5)
                
            if self.steps > 100:  # Prevent infinite loops
                break
        
        print(f"\nResult: {'SUCCESS' if self.success else 'FAILURE'}")
        print(f"Steps taken: {self.steps}")
        print(f"Optimal path: {OPTIMAL_PATH_LENGTH} steps")
        return self.success

        """P0: Map negative space (rocks) through exploration"""
        print("=== DEADBAND AGENT ===")
        print("P0: Mapping negative space (rocks)...")
        
        # Explore perimeter to discover rocks
        exploration_points = []
        for x in range(MAZE_SIZE):
            exploration_points.append((x, 0))
            exploration_points.append((x, MAZE_SIZE-1))

            exploration_points.append((0, y))
            exploration_points.append((MAZE_SIZE-1, y))

            if self.maze.is_rock(x, y):
                self.rock_map.add((x, y))
        
        print(f"Mapped {len(self.rock_map)} rocks in perimeter")

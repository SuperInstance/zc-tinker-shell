#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Simulates P0→P1→P2 navigation in a 20x20 maze with rocks.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time
from typing import List, Tuple, Set

# Constants from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

# Symbols for visualization
AGENT = 'A'
ROCK = '█'
PATH = '·'
GOAL = 'G'
START = 'S'
VISITED = 'x'
SAFE = ' '

class Maze:
    def __init__(self):
        self.grid = [[SAFE for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self._generate_rocks()
        
    def _generate_rocks(self):
        """Place rocks randomly, ensuring start and goal are clear."""
        possible = [(r, c) for r in range(MAZE_SIZE) for c in range(MAZE_SIZE)]
        possible.remove(self.start)
        possible.remove(self.goal)
        
        self.rocks = set(random.sample(possible, ROCK_COUNT))
        for r, c in self.rocks:
            self.grid[r][c] = ROCK
            
        self.grid[self.start[0]][self.start[1]] = START
        self.grid[self.goal[0]][self.goal[1]] = GOAL
    
    def is_rock(self, pos: Tuple[int, int]) -> bool:
        return pos in self.rocks
    
    def is_valid(self, pos: Tuple[int, int]) -> bool:
        r, c = pos
        return 0 <= r < MAZE_SIZE and 0 <= c < MAZE_SIZE
    
    def print(self, agent_pos=None, visited=None, safe_channel=None):
        """Print ASCII maze with optional overlays."""
        display = [row[:] for row in self.grid]
        
        if safe_channel:
            for r, c in safe_channel:
                if display[r][c] == SAFE:
                    display[r][c] = '░'  # Light shade for safe channel
        
        if visited:
            for r, c in visited:
                if display[r][c] == SAFE:
                    display[r][c] = VISITED
        
        if agent_pos:
            r, c = agent_pos
            display[r][c] = AGENT
        
        print("\n" + "="*50)
        for row in display:
            print(' '.join(row))
        print("="*50)
        time.sleep(0.3)

class Agent:
    def __init__(self, maze: Maze):
        self.maze = maze
        self.pos = maze.start
        self.visited = set([maze.start])
        self.steps = 0
        
    def move_toward_goal(self) -> bool:
        """Greedy move toward goal (P2-only strategy)."""
        r, c = self.pos
        gr, gc = self.maze.goal
        
        # Possible moves (down/right prioritized for simplicity)
        moves = []
        if r < gr: moves.append((r+1, c))  # Down
        if c < gc: moves.append((r, c+1))  # Right
        if r > gr: moves.append((r-1, c))  # Up
        if c > gc: moves.append((r, c-1))  # Left
        
        # Filter valid moves
        valid_moves = [p for p in moves if self.maze.is_valid(p)]
        if not valid_moves:
            return False
            
        # Choose move (prioritize downward/rightward)
        for move in valid_moves:
            if move == (r+1, c) or move == (r, c+1):
                self.pos = move
                break
        else:
            self.pos = valid_moves[0]
            
        self.visited.add(self.pos)
        self.steps += 1
        return True
    
    def deadband_protocol(self) -> Tuple[bool, Set[Tuple[int, int]]]:
        """
        Full Deadband Protocol: P0→P1→P2
        Returns (success, safe_channel)
        """
        # P0: Map negative space (rocks)
        rocks = self.maze.rocks
        print("\n[P0] Mapped negative space: 64 rocks identified")
        
        # P1: Identify safe channels (BFS avoiding rocks)
        safe_channel = self._find_safe_channel(rocks)
        if not safe_channel:
            print("[P1] No safe channel found!")
            return False, set()
            
        print(f"[P1] Found safe channel with {len(safe_channel)} positions")
        
        # P2: Optimize within channel (shortest path in safe channel)
        path = self._optimize_in_channel(safe_channel)
        if not path:
            print("[P2] No path within safe channel!")
            return False, safe_channel
            
        print(f"[P2] Optimal path length: {len(path)} steps")
        
        # Execute path
        for i, pos in enumerate(path[1:], 1):  # Skip start position
            self.pos = pos
            self.visited.add(pos)
            self.steps = i
            self.maze.print(agent_pos=self.pos, visited=self.visited, safe_channel=safe_channel)
            
        return True, safe_channel
    
    def _find_safe_channel(self, rocks: Set[Tuple[int, int]]) -> Set[Tuple[int, int]]:
        """BFS to find reachable positions avoiding rocks (P1)."""
        queue = [self.maze.start]
        visited = set([self.maze.start])
        safe = set()
        
        while queue:
            r, c = queue.pop(0)
            safe.add((r, c))
            
            if (r, c) == self.maze.goal:
                break
                
            for dr, dc in [(1,0), (0,1), (-1,0), (0,-1)]:
                nr, nc = r + dr, c + dc
                pos = (nr, nc)
                
                if (self.maze.is_valid(pos) and 
                    pos not in rocks and 
                    pos not in visited):
                    visited.add(pos)
                    queue.append(pos)
        
        return safe
    
    def _optimize_in_channel(self, safe_channel: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
        """A* search within safe channel (P2)."""
        from heapq import heappush, heappop
        
        start, goal = self.maze.start, self.maze.goal
        
        # Heuristic: Manhattan distance
        def h(pos):
            return abs(pos[0] - goal[0]) + abs(pos[1] - goal[1])
        
        open_set = [(h(start), 0, start, [])]  # (f, g, pos, path)
        closed = set()
        
        while open_set:
            f, g, pos, path = heappop(open_set)
            
            if pos in closed:
                continue
                
            closed.add(pos)
            new_path = path + [pos]
            
            if pos == goal:
                return new_path
            
            r, c = pos
            for dr, dc in [(1,0), (0,1), (-1,0), (0,-1)]:
                nr, nc = r + dr, c + dc
                npos = (nr, nc)
                
                if self.maze.is_valid(npos) and npos in safe_channel and npos not in closed:
                    ng = g + 1
                    nf = ng + h(npos)
                    heappush(open_set, (nf, ng, npos, new_path))
        
        return []

#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — Tinker (Cycle 349)
A concrete demonstration of P0→P1→P2 in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time
from typing import List, Tuple, Set

# ============================================================================
# CONFIGURATION (from fleet data)
# ============================================================================
MAZE_SIZE = 20
ROCK_COUNT = 64          # from "rocks": 64
OPTIMAL_PATH_LEN = 21    # from "optimal": 21
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

# ============================================================================
# MAZE GENERATION (P0: Map the negative space)
# ============================================================================
def generate_maze() -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """Create a 20x20 maze with rocks, start, and goal."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    maze[0][0] = 'S'          # start
    maze[MAZE_SIZE-1][MAZE_SIZE-1] = 'G'  # goal
    
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        r = random.randint(0, MAZE_SIZE-1)
        c = random.randint(0, MAZE_SIZE-1)
        if (r, c) != (0, 0) and (r, c) != (MAZE_SIZE-1, MAZE_SIZE-1):
            rocks.add((r, c))
            maze[r][c] = '#'
    
    return maze, rocks

def print_maze(maze: List[List[str]], title: str):
    """Print the maze with a title."""
    print(f"\n{'='*60}")
    print(f"{title}")
    print(f"{'='*60}")
    for row in maze:
        print(' '.join(row))
    print()

# ============================================================================
# P0: IDENTIFY NEGATIVE SPACE (the rocks)
# ============================================================================
def p0_map_negative_space(rocks: Set[Tuple[int, int]]) -> Set[Tuple[int, int]]:
    """P0 step: return the set of states to avoid."""
    print("[P0] Mapping negative space...")
    print(f"    Found {len(rocks)} rock cells (failure states).")
    print("    These are ABSOLUTELY FORBIDDEN.")
    return rocks  # S_neg

# ============================================================================
# P1: FIND SAFE CHANNELS (deadbands)
# ============================================================================
def p1_find_safe_channels(rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
    """P1 step: find all paths from start to goal that avoid rocks."""
    print("\n[P1] Identifying safe channels...")
    
    # Simple BFS to find all safe paths (up/right only for demonstration)
    from collections import deque
    start = (0, 0)
    
    queue = deque()
    queue.append([start])
    safe_paths = []
    
    while queue:
        path = queue.popleft()
        r, c = path[-1]
        
        if (r, c) == GOAL:
            safe_paths.append(path)
            continue
        
        # Try moving right
        if c + 1 < MAZE_SIZE and (r, c+1) not in rocks:
            new_path = path + [(r, c+1)]
            queue.append(new_path)
        
        # Try moving down
        if r + 1 < MAZE_SIZE and (r+1, c) not in rocks:
            new_path = path + [(r+1, c)]
            queue.append(new_path)
    
    print(f"    Found {len(safe_paths)} safe channels (rock-avoiding paths).")
    if safe_paths:
        print(f"    Shortest safe channel: {len(safe_paths[0])} steps")
        print(f"    Longest safe channel: {len(safe_paths[-1])} steps")
    
    return safe_paths

# ============================================================================
# P2: OPTIMIZE WITHIN CHANNEL
# ============================================================================
def p2_optimize_within_channel(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """P2 step: select the shortest path from safe channels."""
    print("\n[P2] Optimizing within safe channels...")
    
    if not safe_paths:
        print("    ERROR: No safe channels found. Cannot optimize.")
        return []
    
    # Select shortest path (minimum cost)
    optimal_path = min(safe_paths, key=len)
    
    print(f"    Selected path with {len(optimal_path)} steps.")
    print(f"    Theoretical optimal: {OPTIMAL_PATH_LEN} steps")
    
    if len(optimal_path) == OPTIMAL_PATH_LEN:
        print("    ✓ Achieved optimal path length!")
    else:
        print(f"    Δ Deviation: {len(optimal_path) - OPTIMAL_PATH_LEN} steps")
    
    return optimal_path

# ============================================================================
# VISUALIZATION
# ============================================================================
def visualize_process(maze: List[List[str]], 
                     rocks: Set[Tuple[int, int]],
                     safe_paths: List[List[Tuple[int, int]]],
                     optimal_path: List[Tuple[int, int]]):
    """Create visualizations for each protocol step."""
    
    # P0 visualization: show rocks
    maze_p0 = [row[:] for row in maze]
    for r, c in rocks:
        maze_p0[r][c] = '█'  # solid block for rocks
    print_maze(maze_p0, "P0: NEGATIVE SPACE (Rocks = █)")
    
    # P1 visualization: show one sample safe channel
    if safe_paths:
        maze_p1 = [row[:] for row in maze]
        sample_path = safe_paths[0]  # first found safe path
        for r, c in sample_path[1:-1]:  # exclude start and goal
            maze_p1[r][c] = '~'  # water symbol for safe channel
        print_maze(maze_p1, "P1: SAFE CHANNEL (Water = ~)")
    
    # P2 visualization: show optimal path
    if optimal_path:
        maze_p2 = [row[:] for row in maze]
        for r, c in optimal_path[1:-1]:  # exclude start and goal
            maze_p2[r][c] = '●'  # solid circle for optimal path
        print_maze(maze_p2, "P2: OPTIMAL PATH WITHIN CHANNEL (Path = ●)")

# ============================================================================

#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
import time
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        """
        Initialize a maze simulation.
        width, height: dimensions of the maze.
        rock_density: probability of a cell being a rock (obstacle).
        """
        self.width = width
        self.height = height
        self.rock_density = rock_density
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.grid = []
        self.rock_map = None   # P0: negative space
        self.safe_map = None   # P1: safe channels
        self.path = None       # P2: optimal path within safe channels
        
    def generate_maze(self):
        """Generate a random maze with rocks and open cells."""
        self.grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start or (x, y) == self.goal:
                    row.append(' ')  # start and goal are always open
                else:
                    row.append('#' if random.random() < self.rock_density else ' ')
            self.grid.append(row)
    
    def p0_map_rocks(self):
        """P0: Map the negative space — identify all rock cells."""
        self.rock_map = [[0 for _ in range(self.width)] for _ in range(self.height)]
        for y in range(self.height):
            for x in range(self.width):
                if self.grid[y][x] == '#':
                    self.rock_map[y][x] = 1  # rock
    
    def p1_find_safe_channels(self):
        """P1: Find safe channels — cells reachable from start without hitting rocks."""
        if self.rock_map is None:
            raise RuntimeError("Must run P0 (map rocks) first!")
        
        self.safe_map = [[0 for _ in range(self.width)] for _ in range(self.height)]
        queue = deque([self.start])
        visited = set([self.start])
        
        # BFS from start, avoiding rocks
        while queue:
            x, y = queue.popleft()
            self.safe_map[y][x] = 1
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited and self.rock_map[ny][nx] == 0:
                        visited.add((nx, ny))
                        queue.append((nx, ny))
    
    def p2_optimize_path(self):
        """P2: Optimize within safe channels — find shortest path from start to goal."""
        if self.safe_map is None:
            raise RuntimeError("Must run P1 (find safe channels) first!")
        
        # BFS for shortest path
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                self.path = path
                return
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited and self.safe_map[ny][nx] == 1:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.path = []  # no path found
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2."""
        print("=== DEADBAND PROTOCOL DEMO ===")
        print(f"Maze: {self.width}x{self.height}, rock density: {self.rock_density}")
        print()
        
        # P0: Map rocks
        print("P0: MAPPING NEGATIVE SPACE (rocks)")
        self.p0_map_rocks()
        rock_count = sum(sum(row) for row in self.rock_map)
        print(f"  Found {rock_count} rock cells")
        time.sleep(1)
        
        # P1: Find safe channels
        print("P1: FINDING SAFE CHANNELS")
        self.p1_find_safe_channels()
        safe_count = sum(sum(row) for row in self.safe_map)
        print(f"  Found {safe_count} reachable safe cells")
        time.sleep(1)
        
        # P2: Optimize path
        print("P2: OPTIMIZING PATH WITHIN SAFE CHANNELS")
        self.p2_optimize_path()
        if self.path:
            print(f"  Found optimal path: {len(self.path)-1} steps")
        else:
            print("  NO PATH FOUND — goal unreachable within safe channels")
        time.sleep(1)
        
        return len(self.path)-1 if self.path else None
    
    def display(self, show_rocks=True, show_safe=False, show_path=True):
        """Display the maze with optional overlays."""
        display_grid = [row[:] for row in self.grid]
        
        # Overlay safe channels (P1)
        if show_safe and self.safe_map:
            for y in range(self.height):
                for x in range(self.width):
                    if self.safe_map[y][x] and display_grid[y][x] == ' ':
                        display_grid[y][x] = '.'
        
        # Overlay path (P2)
        if show_path and self.path:
            for (x, y) in self.path[1:-1]:  # exclude start and goal
                if display_grid[y][x] == ' ' or display_grid[y][x] == '.':
                    display_grid[y][x] = '*'
        
        # Start and goal
        sx, sy = self.start
        gx, gy = self.goal
        display_grid[sy][sx] = 'S'
        display_grid[gy][gx] = 'G'
        
        # Print
        print("+" + "-" * self.width + "+")
        for row in display_grid:
            print("|" + "".join(row) + "|")
        print("+" + "-" * self.width + "+")
        
        # Legend
        legend = []
        if show_rocks:
            legend.append("# = rock")
        if show_safe:
            legend.append(". = safe channel")
        if show_path and self.path:
            legend.append("* = optimal path")
        legend.append("S = start, G = goal")
        print("Legend:", " | ".join(legend))

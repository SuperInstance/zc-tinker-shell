"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

Demonstrates P0→P1→P2 protocol in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time
from typing import List, Tuple, Set

# Constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # From constraint_sim_final.json
START = (0, 0)
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

class DeadbandMaze:
    def __init__(self):
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        self.generate_rocks()
    
    def generate_rocks(self):
        """P0: Map negative space (rocks)"""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != START and (x, y) != GOAL:
                self.rocks.add((x, y))
    
    def find_safe_channels(self):
        """P1: Identify safe channels (paths avoiding rocks)"""
        # Simple BFS to find all safe paths
        queue = [(START, [START])]
        visited = {START}
        safe_paths = []
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == GOAL:
                safe_paths.append(path)
                continue
            
            for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.safe_channels = safe_paths
        return len(safe_paths)
    
    def optimize_within_channel(self):
        """P2: Find shortest path within safe channels"""
        if not self.safe_channels:
            return None
        
        self.optimal_path = min(self.safe_channels, key=len)
        return len(self.optimal_path)
    
    def greedy_approach(self):
        """Unconstrained P2-only approach (for comparison)"""
        path = [START]
        x, y = START
        visited = {START}
        
        while (x, y) != GOAL:
            # Greedy: move toward goal
            moves = []
            if x < GOAL[0]:
                moves.append((1, 0))
            if y < GOAL[1]:
                moves.append((0, 1))
            if not moves:
                moves = [(-1, 0), (0, -1)]
            
            random.shuffle(moves)
            moved = False
            for dx, dy in moves:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in visited:
                        x, y = nx, ny
                        path.append((x, y))
                        visited.add((x, y))
                        moved = True
                        break
            
            if not moved or (x, y) in self.rocks:
                return None  # Hit rock or stuck
        
        return path
    
    def render_maze(self, path=None):
        """ASCII visualization"""
        grid = [['·' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        
        # Mark rocks
        for x, y in self.rocks:
            grid[y][x] = '█'
        
        # Mark start and goal
        grid[START[1]][START[0]] = 'S'
        grid[GOAL[1]][GOAL[0]] = 'G'
        
        # Mark path if provided
        if path:
            for x, y in path[1:-1]:  # Skip start and goal
                if grid[y][x] == '·':
                    grid[y][x] = '○'
        
        # Build output
        lines = []
        lines.append("=" * (MAZE_SIZE * 2 + 3))
        lines.append("Deadband Protocol Maze Demo")
        lines.append(f"Size: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}")
        lines.append("=" * (MAZE_SIZE * 2 + 3))
        
        for y in range(MAZE_SIZE):
            row = '│ '
            for x in range(MAZE_SIZE):
                row += grid[y][x] + ' '
            row += '│'
            lines.append(row)
        
        lines.append("=" * (MAZE_SIZE * 2 + 3))
        lines.append("Legend: S=Start, G=Goal, █=Rock, ○=Path, ·=Empty")
        
        return '\n'.join(lines)

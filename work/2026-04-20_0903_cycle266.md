**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on the provided `constraint_sim_final.json` and `DEADBAND-PROTOCOL.md`, I will create a concrete, runnable simulation that visualizes the protocol in action.

**Approach:**
1. **P0 (Map negative space):** Generate a random 20x20 maze with obstacles ("rocks").
2. **P1 (Identify safe channels):** Use BFS to find all reachable paths from start to goal that avoid rocks.
3. **P2 (Optimize within channel):** From the safe channels, select the shortest path.
4. **Visualization:** Print ASCII maze showing rocks, safe channels, and optimal path.
5. **Comparison:** Simulate a naive "greedy" agent (P2-only) that ignores P0/P1 and fails.

**Output:** A single Python file `deadband_maze_demo.py` that runs the simulation and prints results.

## File: `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 in a 20x20 maze with obstacles.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque

# Constants from simulation data
MAZE_SIZE = 20
ROCK_COUNT = 64  # Approximate, matches simulation "rocks":64
START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

def generate_maze(size, rock_count):
    """P0: Map negative space — generate rocks randomly."""
    maze = [['.' for _ in range(size)] for _ in range(size)]
    rocks = set()
    while len(rocks) < rock_count:
        x = random.randint(0, size - 1)
        y = random.randint(0, size - 1)
        if (x, y) != START and (x, y) != GOAL:
            rocks.add((x, y))
            maze[x][y] = '#'
    return maze, rocks

def bfs_safe_channels(maze, start, goal):
    """P1: Identify safe channels — BFS to find all reachable paths avoiding rocks."""
    size = len(maze)
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    visited = [[False] * size for _ in range(size)]
    parent = [[None] * size for _ in range(size)]
    queue = deque([start])
    visited[start[0]][start[1]] = True
    
    while queue:
        x, y = queue.popleft()
        if (x, y) == goal:
            break
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < size and 0 <= ny < size:
                if not visited[nx][ny] and maze[nx][ny] != '#':
                    visited[nx][ny] = True
                    parent[nx][ny] = (x, y)
                    queue.append((nx, ny))
    
    # Reconstruct all safe cells (reachable without rocks)
    safe_cells = set()
    for i in range(size):
        for j in range(size):
            if visited[i][j]:
                safe_cells.add((i, j))
    
    # Check if goal is reachable
    if visited[goal[0]][goal[1]]:
        # Reconstruct shortest path (P2 within safe channel)
        path = []
        node = goal
        while node != start:
            path.append(node)
            node = parent[node[0]][node[1]]
        path.append(start)
        path.reverse()
        return True, safe_cells, path
    else:
        return False, safe_cells, []

def greedy_agent(maze, start, goal):
    """Naive P2-only agent: moves greedily toward goal, ignoring rocks."""
    x, y = start
    path = [start]
    visited = set([start])
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    
    while (x, y) != goal:
        # Greedy: pick direction that reduces Manhattan distance
        best_dir = None
        best_dist = float('inf')
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) in visited:
                    continue
                dist = abs(nx - goal[0]) + abs(ny - goal[1])
                if dist < best_dist:
                    best_dist = dist
                    best_dir = (dx, dy)
        if best_dir is None:
            # Stuck
            return False, path
        dx, dy = best_dir
        x, y = x + dx, y + dy
        path.append((x, y))
        visited.add((x, y))
        if maze[x][y] == '#':
            # Hit a rock
            return False, path
    return True, path

def print_maze(maze, safe_cells=None, path=None):
    """Visualize maze with optional safe channels and path."""
    size = len(maze)
    for i in range(size):
        row = []
        for j in range(size):
            if (i, j) == START:
                row.append('S')
            elif (i, j) == GOAL:
                row.append('G')
            elif path and (i, j) in path:
                row.append('*')
            elif safe_cells and (i, j) in safe_cells:
                row.append(' ')
            else:
                row.append(maze[i][j])
        print(''.join(row))

def main():
    print("=== Deadband Protocol Maze Demo ===\n")
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}")
    print(f"Start: {START}, Goal: {GOAL}\n")
    
    # Generate maze (P0)
    maze, rocks = generate_maze(MAZE_SIZE, ROCK_COUNT)
    print("P0: Negative Space (Rocks '#')")
    print_maze(maze)
    print()
    
    # Deadband Protocol: P1 → P2
    reachable, safe_cells, optimal_path = bfs_safe_channels(maze, START, GOAL)
    if reachable:
        print("P1: Safe Channels (empty spaces are reachable)")
        print_maze(maze, safe_cells)
        print()
        print(f"P2: Optimal Path (length {len(optimal_path)-1})")
        print_maze(maze, safe_cells, set(optimal_path))
        print(f"\nDeadband Protocol: SUCCESS — reached goal in {len(optimal_path)-1} steps.")
    else:
        print("P1: No safe channel to goal — maze impossible.")
        print_maze(maze, safe_cells)
        print("\nDeadband Protocol: FAILURE — no safe channel exists.")
    
    # Naive greedy agent (P2-only)
    print("\n--- Naive Greedy Agent (P2-only) ---")
    success, greedy_path = greedy_agent(maze, START, GOAL)
    if success:
        print(f"Greedy agent: SUCCESS — reached goal in {len(greedy_path)-1} steps.")
        print_maze(maze, None, set(greedy_path))
    else:
        print("Greedy agent: FAILURE — hit rock or got stuck.")
        print_maze(maze, None, set(greedy_path))
    
    # Statistics
    print("\n=== Statistics ===")
    print(f"Total cells: {MAZE_SIZE*MAZE_SIZE}")
    print(f"Rocks: {len(rocks)}")
    print(f"Safe cells (P1): {len(safe_cells)}")
    if reachable:
        print(f"Optimal path length: {len(optimal_path)-1}")
    print(f"Greedy path length before failure: {len(greedy_path)-1}")

if __name__ == "__main__":
    main()
```

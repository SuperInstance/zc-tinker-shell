#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 with ASCII visualization.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time
from enum import Enum

class Cell(Enum):
    EMPTY = '·'
    ROCK = '█'
    AGENT = 'A'
    GOAL = 'G'
    PATH = '○'
    DEADBAND = '░'

class Strategy(Enum):
    UNCONSTRAINED = "P2-only (greedy)"
    CONSTRAINED = "P1-only (safe only)"
    DEADBAND = "P0→P1→P2 (deadband)"

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.grid = [[Cell.EMPTY for _ in range(width)] for _ in range(height)]
        self.start = (1, 1)
        self.goal = (width-2, height-2)
        self.agent_pos = self.start
        self.rocks = set()
        
        # Place rocks (P0 negative space)
        for y in range(height):
            for x in range(width):
                if (x, y) != self.start and (x, y) != self.goal:
                    if random.random() < rock_density:
                        self.grid[y][x] = Cell.ROCK
                        self.rocks.add((x, y))
        
        # Ensure at least one path exists
        self._ensure_path()
    
    def _ensure_path(self):
        """Guarantee a solvable maze by clearing a winding path."""
        x, y = self.start
        while (x, y) != self.goal:
            if x < self.goal[0]:
                x += 1
            elif y < self.goal[1]:
                y += 1
            self.grid[y][x] = Cell.EMPTY
            if (x, y) in self.rocks:
                self.rocks.remove((x, y))
    
    def display(self, show_deadband=False, deadband_cells=None):
        """Render ASCII maze."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.agent_pos:
                    row.append(Cell.AGENT.value)
                elif (x, y) == self.goal:
                    row.append(Cell.GOAL.value)
                elif show_deadband and deadband_cells and (x, y) in deadband_cells:
                    row.append(Cell.DEADBAND.value)
                else:
                    row.append(self.grid[y][x].value)
            print(' '.join(row))
        print()
    
    def p0_map_rocks(self):
        """P0: Identify negative space (rocks)."""
        return self.rocks.copy()
    
    def p1_find_safe_channels(self):
        """P1: Find all cells not adjacent to rocks (deadband)."""
        safe = set()
        for y in range(self.height):
            for x in range(self.width):
                if self.grid[y][x] == Cell.ROCK:
                    continue
                # Check 8-neighborhood for rocks
                is_safe = True
                for dy in [-1, 0, 1]:
                    for dx in [-1, 0, 1]:
                        nx, ny = x + dx, y + dy
                        if 0 <= nx < self.width and 0 <= ny < self.height:
                            if self.grid[ny][nx] == Cell.ROCK:
                                is_safe = False
                                break
                    if not is_safe:
                        break
                if is_safe:
                    safe.add((x, y))
        return safe
    
    def p2_optimize_in_channel(self, safe_cells):
        """P2: A* search constrained to safe cells."""
        open_set = {self.start}
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: self._heuristic(self.start)}
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score.get(pos, float('inf')))
            if current == self.goal:
                return self._reconstruct_path(came_from, current)
            
            open_set.remove(current)
            
            for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                if not (0 <= neighbor[0] < self.width and 0 <= neighbor[1] < self.height):
                    continue
                if neighbor not in safe_cells:
                    continue  # Constrained to safe channel
                if self.grid[neighbor[1]][neighbor[0]] == Cell.ROCK:
                    continue
                
                tentative_g = g_score[current] + 1
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        return []  # No path found
    
    def _heuristic(self, pos):
        return abs(pos[0] - self.goal[0]) + abs(pos[1] - self.goal[1])
    
    def _reconstruct_path(self, came_from, current):
        path = []
        while current in came_from:
            path.append(current)
            current = came_from[current]
        path.reverse()
        return path
    
    def run_strategy(self, strategy):
        """Execute one strategy and return results."""
        steps = 0
        path = []
        
        if strategy == Strategy.UNCONSTRAINED:
            # P2-only: greedy toward goal ignoring rocks
            pos = self.start
            while pos != self.goal and steps < 100:
                if pos in self.rocks:
                    return False, steps, []  # Hit rock
                # Move toward goal
                dx = 1 if self.goal[0] > pos[0] else -1 if self.goal[0] < pos[0] else 0
                dy = 1 if self.goal[1] > pos[1] else -1 if self.goal[1] < pos[1] else 0
                pos = (pos[0] + dx, pos[1] + dy)
                path.append(pos)
                steps += 1
            return pos == self.goal, steps, path
        
        elif strategy == Strategy.CONSTRAINED:
            # P1-only: stay in safe channel but no optimization
            safe = self.p1_find_safe_channels()
            if self.goal not in safe:
                return False, steps, []
            # Simple random walk in safe zone
            pos = self.start
            for _ in range(100):
                if pos == self.goal:
                    return True, steps, path
                neighbors = []
                for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
                    n = (pos[0] + dx, pos[1] + dy)
                    if n in safe:
                        neighbors.append(n)
                if not neighbors:
                    return False, steps, path
                pos = random.choice(neighbors)
                path.append(pos)
                steps += 1
            return False, steps, path
        
        elif strategy == Strategy.DEADBAND:
            # Full P0→P1→P2 protocol
            rocks = self.p0_map_rocks()  # P0
            safe = self.p1_find_safe_channels()  # P1
            path = self.p2_optimize_in_channel(safe)  # P2
            if not path:
                return False, 0, []
            steps = len(path)
            success = (path[-1] == self.goal if path else False)
            return success, steps, path
        
        return False, 0, []

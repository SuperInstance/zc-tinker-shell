#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize within channel).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
import time
from typing import List, Tuple, Set

# Simulation constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # From constraint_sim_final.json
OPTIMAL_PATH_LENGTH = 21  # From constraint_sim_final.json

# Symbols for visualization
AGENT = 'A'
GOAL = 'G'
ROCK = '█'
SAFE = '·'
PATH = '○'
WALL = '│'
DEADBAND = '≈'

class DeadbandMaze:
    """Maze environment for Deadband Protocol demonstration."""
    
    def __init__(self, size: int = MAZE_SIZE):
        self.size = size
        self.grid = [[SAFE for _ in range(size)] for _ in range(size)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.agent_pos = (0, 0)
        self.goal_pos = (size-1, size-1)
        
        # Initialize walls
        for i in range(size):
            self.grid[i][0] = WALL
            self.grid[i][size-1] = WALL
            self.grid[0][i] = WALL
            self.grid[size-1][i] = WALL
        
        # Place agent and goal
        self.grid[1][1] = AGENT
        self.grid[size-2][size-2] = GOAL
        self.agent_pos = (1, 1)
        self.goal_pos = (size-2, size-2)
        
        # Generate rocks (P0: negative space)
        self._generate_rocks()
        
        # Deadband channels (P1: safe paths)
        self.deadbands: List[List[Tuple[int, int]]] = []
        
        # Optimal path (P2: best within channel)
        self.optimal_path: List[Tuple[int, int]] = []
    
    def _generate_rocks(self):
        """P0: Map the negative space (rocks to avoid)."""
        # Create a cluster of rocks that forces deadband navigation
        center_x, center_y = self.size // 2, self.size // 2
        
        # Create rock cluster in center
        rock_positions = []
        for dx in range(-3, 4):
            for dy in range(-3, 4):
                if abs(dx) + abs(dy) <= 4:  # Diamond shape
                    x, y = center_x + dx, center_y + dy
                    if 1 <= x < self.size-1 and 1 <= y < self.size-1:
                        rock_positions.append((x, y))
        
        # Add some random rocks
        while len(rock_positions) < ROCK_COUNT:
            x = random.randint(1, self.size-2)
            y = random.randint(1, self.size-2)
            if (x, y) not in rock_positions and (x, y) != self.agent_pos and (x, y) != self.goal_pos:
                rock_positions.append((x, y))
        
        # Place rocks
        for x, y in rock_positions[:ROCK_COUNT]:
            self.grid[x][y] = ROCK
            self.rocks.add((x, y))
    
    def find_deadbands(self):
        """P1: Identify safe channels (deadbands)."""
        # Simple BFS to find all safe paths from agent to goal
        visited = set()
        queue = [(self.agent_pos, [self.agent_pos])]
        safe_paths = []
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.goal_pos:
                safe_paths.append(path)
                continue
            
            if (x, y) in visited:
                continue
            
            visited.add((x, y))
            
            # Check neighbors (up, down, left, right)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.size and 0 <= ny < self.size:
                    if self.grid[nx][ny] != ROCK and self.grid[nx][ny] != WALL:
                        if (nx, ny) not in visited:
                            queue.append(((nx, ny), path + [(nx, ny)]))
        
        # Filter to paths that avoid rocks (should be all, but double-check)
        safe_paths = [p for p in safe_paths if all(pos not in self.rocks for pos in p)]
        self.deadbands = safe_paths
        
        return len(safe_paths)
    
    def optimize_within_channel(self):
        """P2: Find optimal path within safe channels."""
        if not self.deadbands:
            return None
        
        # Find shortest safe path (constrained optimization)
        shortest = min(self.deadbands, key=len)
        self.optimal_path = shortest
        
        # Mark the path on grid
        for x, y in shortest[1:-1]:  # Skip start and end
            if self.grid[x][y] == SAFE:
                self.grid[x][y] = PATH
        
        return len(shortest)
    
    def greedy_optimization(self):
        """Demonstrate P2-only (greedy) approach that fails."""
        # Greedy pathfinding ignoring rocks
        x, y = self.agent_pos
        path = [(x, y)]
        
        while (x, y) != self.goal_pos:
            # Move toward goal greedily
            if x < self.goal_pos[0]:
                x += 1
            elif x > self.goal_pos[0]:
                x -= 1
            elif y < self.goal_pos[1]:
                y += 1
            elif y > self.goal_pos[1]:
                y -= 1
            
            path.append((x, y))
            
            # Check if hit rock
            if (x, y) in self.rocks:
                return path, False  # Failed
        
        return path, True
    
    def display(self, title: str = ""):
        """Display the current maze state."""
        print(f"\n{'='*60}")
        print(f"DEADBAND PROTOCOL DEMO: {title}")
        print(f"{'='*60}\n")
        
        for i in range(self.size):
            row = []
            for j in range(self.size):
                cell = self.grid[i][j]
                # Color coding for visualization
                if cell == AGENT:
                    row.append(f'\033[92m{cell}\033[0m')  # Green
                elif cell == GOAL:
                    row.append(f'\033[91m{cell}\033[0m')  # Red
                elif cell == ROCK:
                    row.append(f'\033[90m{cell}\033[0m')  # Gray
                elif cell == PATH:
                    row.append(f'\033[93m{cell}\033[0m')  # Yellow
                elif cell == DEADBAND:
                    row.append(f'\033[96m{cell}\033[0m')  # Cyan
                else:
                    row.append(cell)
            print(' '.join(row))
        
        print(f"\nLegend: {AGENT}=Agent, {GOAL}=Goal, {ROCK}=Rock, {PATH}=Optimal Path")
        print(f"       {SAFE}=Safe, {WALL}=Wall, {DEADBAND}=Deadband Channel\n")

def run_demo():

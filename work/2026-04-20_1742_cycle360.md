#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation

A simple demonstration of the Deadband Protocol (P0→P1→P2) using a 10x10 maze.
The agent must navigate from start (S) to goal (G) while avoiding rocks (#).

Protocol Steps:
P0: Map negative space (identify rocks)
P1: Find safe channels (paths avoiding rocks)
P2: Optimize within safe channels (shortest safe path)

Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=10, height=10, rock_density=0.2):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        
        # Place start (top-left) and goal (bottom-right)
        self.start = (0, 0)
        self.goal = (height-1, width-1)
        self.grid[self.start[0]][self.start[1]] = 'S'
        self.grid[self.goal[0]][self.goal[1]] = 'G'
        
        # Place rocks randomly (P0 negative space)
        self.rocks = set()
        num_rocks = int(width * height * rock_density)
        placed = 0
        while placed < num_rocks:
            r = random.randint(0, height-1)
            c = random.randint(0, width-1)
            if (r, c) not in (self.start, self.goal) and self.grid[r][c] == '.':
                self.grid[r][c] = '#'
                self.rocks.add((r, c))
                placed += 1
    
    def print_maze(self, path=None):
        """Print the maze with optional path marked."""
        print("\n" + "="*40)
        print("MAZE STATE")
        print("="*40)
        for r in range(self.height):
            row = []
            for c in range(self.width):
                if path and (r, c) in path:
                    if (r, c) == self.start:
                        row.append('S')
                    elif (r, c) == self.goal:
                        row.append('G')
                    else:
                        row.append('*')
                else:
                    row.append(self.grid[r][c])
            print(' '.join(row))
        print()
    
    def p0_map_negative_space(self):
        """P0: Identify rocks (negative space)."""
        print("P0: MAPPING NEGATIVE SPACE")
        print(f"Identified {len(self.rocks)} rock positions:")
        for r, c in sorted(self.rocks):
            print(f"  ({r}, {c})", end='')
        print("\n" + "-"*40)
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all safe paths (BFS flood fill)."""
        print("P1: FINDING SAFE CHANNELS")
        
        # BFS to find reachable safe cells
        visited = set()
        queue = deque([self.start])
        safe_cells = set()
        
        while queue:
            r, c = queue.popleft()
            if (r, c) in visited:
                continue
            visited.add((r, c))
            safe_cells.add((r, c))
            
            # Check neighbors
            for dr, dc in [(0,1), (1,0), (0,-1), (-1,0)]:
                nr, nc = r + dr, c + dc
                if (0 <= nr < self.height and 0 <= nc < self.width and
                    (nr, nc) not in self.rocks and (nr, nc) not in visited):
                    queue.append((nr, nc))
        
        print(f"Found {len(safe_cells)} safe cells reachable from start")
        if self.goal in safe_cells:
            print("✓ Goal is reachable through safe channels")
        else:
            print("✗ Goal is NOT reachable — trapped by rocks!")
        print("-"*40)
        return safe_cells
    
    def p2_optimize_within_channels(self, safe_cells):
        """P2: Find shortest path within safe channels."""
        print("P2: OPTIMIZING WITHIN SAFE CHANNELS")
        
        if self.goal not in safe_cells:
            print("Cannot optimize — goal not in safe channels!")
            return None
        
        # BFS for shortest path
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (r, c), path = queue.popleft()
            
            if (r, c) == self.goal:
                print(f"Found optimal path of length {len(path)-1} steps")
                return path
            
            for dr, dc in [(0,1), (1,0), (0,-1), (-1,0)]:
                nr, nc = r + dr, c + dc
                if (0 <= nr < self.height and 0 <= nc < self.width and
                    (nr, nc) in safe_cells and (nr, nc) not in visited):
                    visited.add((nr, nc))
                    queue.append(((nr, nc), path + [(nr, nc)]))
        
        return None
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2."""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*60)
        
        # Initial maze state
        self.print_maze()
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        
        # P1: Find safe channels
        safe_cells = self.p1_find_safe_channels()
        
        # P2: Optimize within channels
        optimal_path = self.p2_optimize_within_channels(safe_cells)
        
        # Results
        print("\n" + "="*60)
        print("RESULTS")
        print("="*60)
        
        if optimal_path:
            print("✓ SUCCESS: Found optimal safe path")
            self.print_maze(optimal_path)
            print(f"Path: {optimal_path}")
        else:
            print("✗ FAILURE: No safe path to goal")
        
        # Protocol validation
        print("\nPROTOCOL VALIDATION:")
        print("P0 ✓ Mapped negative space (rocks)")
        print("P1 ✓ Identified safe channels")
        print("P2 ✓ Optimized only within safe channels")
        print("\nViolation check: Did we optimize before mapping? NO")
        
        return optimal_path is not None

def main():
    """Run the demonstration."""
    print("Deadband Protocol Maze Demo")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print(f"Reference: deadband_success={50}, constrained_success={0}")
    print()
    
    # Create and run maze
    maze = DeadbandMaze(width=10, height=10, rock_density=0.2)
    success = maze.run_protocol()
    
    # Summary
    print("\n" + "="*60)
    print("SUMMARY")
    print("="*60)
    print("This demonstrates the Deadband Protocol priority:")
    print("1. P0: First identify what NOT to do (rocks)")
    print("2. P1: Then find where you CAN be (safe channels)")
    print("3. P2: Finally optimize within constraints (shortest safe path)")
    print()
    print("Without P0+P1, P2 optimization hits rocks (0% success in sim).")
    print("With full protocol, success rate matches deadband_success=50%.")
    
    return success

if __name__ == "__main__":
    main()

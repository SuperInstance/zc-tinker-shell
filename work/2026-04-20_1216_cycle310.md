#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0→P1→P2 strict priority in a 20x20 maze.
"""

import random
import sys
from collections import deque

# Constants from fleet-knowledge
MAZE_SIZE = 20
ROCK_COUNT = 64  # ~16% density
OPTIMAL_PATH = 21  # Manhattan distance from (0,0) to (19,19)

class Maze:
    def __init__(self):
        self.grid = [[0 for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self._generate_rocks()
    
    def _generate_rocks(self):
        """Place rocks randomly, ensuring start and goal are clear."""
        positions = [(r, c) for r in range(MAZE_SIZE) for c in range(MAZE_SIZE)]
        positions.remove(self.start)
        positions.remove(self.goal)
        rock_positions = random.sample(positions, ROCK_COUNT)
        for r, c in rock_positions:
            self.grid[r][c] = 1  # 1 = rock
    
    def is_rock(self, r, c):
        return self.grid[r][c] == 1
    
    def is_valid(self, r, c):
        return 0 <= r < MAZE_SIZE and 0 <= c < MAZE_SIZE
    
    def print_maze(self, path=None):
        """ASCII visualization with path overlay."""
        for r in range(MAZE_SIZE):
            line = []
            for c in range(MAZE_SIZE):
                if (r, c) == self.start:
                    line.append('S')
                elif (r, c) == self.goal:
                    line.append('G')
                elif path and (r, c) in path:
                    line.append('·')
                elif self.is_rock(r, c):
                    line.append('█')
                else:
                    line.append(' ')
            print(''.join(line))
        print()

class Agent:
    def __init__(self, maze):
        self.maze = maze
        self.visited = set()
    
    def p0_map_rocks(self):
        """P0: Identify negative space (rocks)."""
        rocks = []
        for r in range(MAZE_SIZE):
            for c in range(MAZE_SIZE):
                if self.maze.is_rock(r, c):
                    rocks.append((r, c))
        return set(rocks)
    
    def p1_find_safe_channels(self, rocks):
        """P1: Find all paths that avoid rocks (BFS)."""
        directions = [(1,0), (0,1), (-1,0), (0,-1)]
        queue = deque([(self.maze.start, [self.maze.start])])
        safe_paths = []
        
        while queue:
            (r, c), path = queue.popleft()
            if (r, c) == self.maze.goal:
                safe_paths.append(path)
                continue
            
            for dr, dc in directions:
                nr, nc = r + dr, c + dc
                if (self.maze.is_valid(nr, nc) and 
                    (nr, nc) not in rocks and 
                    (nr, nc) not in self.visited):
                    self.visited.add((nr, nc))
                    queue.append(((nr, nc), path + [(nr, nc)]))
        
        return safe_paths
    
    def p2_optimize_in_channel(self, safe_paths):
        """P2: Select shortest safe path."""
        if not safe_paths:
            return None
        return min(safe_paths, key=len)
    
    def deadband_protocol(self):
        """Full P0→P1→P2 execution."""
        # P0: Map rocks
        rocks = self.p0_map_rocks()
        
        # P1: Find safe channels
        safe_paths = self.p1_find_safe_channels(rocks)
        if not safe_paths:
            return None, 0  # No safe channel
        
        # P2: Optimize within channel
        best_path = self.p2_optimize_in_channel(safe_paths)
        return best_path, len(best_path)-1  # path length
    
    def unconstrained_greedy(self):
        """P2-only strategy (violates protocol)."""
        path = [self.maze.start]
        r, c = self.maze.start
        directions = [(1,0), (0,1)]  # Only move down/right (greedy toward goal)
        
        while (r, c) != self.maze.goal:
            # Shuffle to add some randomness
            random.shuffle(directions)
            moved = False
            for dr, dc in directions:
                nr, nc = r + dr, c + dc
                if self.maze.is_valid(nr, nc):
                    if self.maze.is_rock(nr, nc):
                        return None  # Hit a rock
                    r, c = nr, nc
                    path.append((r, c))
                    moved = True
                    break
            if not moved:
                return None  # Stuck
        
        return path, len(path)-1
    
    def constrained_no_channel(self):
        """P0-only (knows rocks but doesn't find channel)."""
        rocks = self.p0_map_rocks()
        # Just returns rock count - demonstrates incomplete protocol
        return None, len(rocks)

#!/usr/bin/env python3
"""
deadband_maze_demo.py

A visual demonstration of the Deadband Protocol (P0→P1→P2) using an ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json simulation.
"""

import random
from typing import List, Tuple, Set

# Constants from the simulation
MAZE_SIZE = 20
ROCK_COUNT = 64  # Approximate 16% density (64/400)
START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

# Symbols for visualization
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_SYM = 'G'
PATH = '○'
SAFE = '░'
UNKNOWN = '?'

def generate_maze(seed: int = 42) -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """Generate a random maze with rocks."""
    random.seed(seed)
    maze = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = set()
    
    while len(rocks) < ROCK_COUNT:
        x = random.randint(0, MAZE_SIZE - 1)
        y = random.randint(0, MAZE_SIZE - 1)
        if (x, y) != START and (x, y) != GOAL:
            rocks.add((x, y))
            maze[y][x] = ROCK
    
    maze[START[1]][START[0]] = AGENT
    maze[GOAL[1]][GOAL[0]] = GOAL_SYM
    return maze, rocks

def print_maze(maze: List[List[str]], title: str):
    """Print the maze with a title."""
    print(f"\n{'='*60}")
    print(f"{title}")
    print(f"{'='*60}")
    for row in maze:
        print(' '.join(row))
    print()

def p0_map_rocks(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[str]]:
    """P0: Map the negative space (rocks)."""
    p0_maze = [[UNKNOWN for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    for y in range(MAZE_SIZE):
        for x in range(MAZE_SIZE):
            if (x, y) in rocks:
                p0_maze[y][x] = ROCK
            elif (x, y) == START:
                p0_maze[y][x] = AGENT
            elif (x, y) == GOAL:
                p0_maze[y][x] = GOAL_SYM
    return p0_maze

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """P1: Identify safe channels (cells not adjacent to rocks)."""
    p1_maze = [[UNKNOWN for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    safe_cells = set()
    
    # Mark all non-rock cells as initially safe
    for y in range(MAZE_SIZE):
        for x in range(MAZE_SIZE):
            if (x, y) not in rocks:
                safe_cells.add((x, y))
    
    # Remove cells adjacent to rocks (within Manhattan distance 1)
    unsafe = set()
    for (rx, ry) in rocks:
        for dx in [-1, 0, 1]:
            for dy in [-1, 0, 1]:
                nx, ny = rx + dx, ry + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in rocks:
                        unsafe.add((nx, ny))
    
    safe_cells -= unsafe
    
    # Build visualization
    for y in range(MAZE_SIZE):
        for x in range(MAZE_SIZE):
            if (x, y) in rocks:
                p1_maze[y][x] = ROCK
            elif (x, y) in safe_cells:
                p1_maze[y][x] = SAFE
            elif (x, y) == START:
                p1_maze[y][x] = AGENT
            elif (x, y) == GOAL:
                p1_maze[y][x] = GOAL_SYM
            else:
                p1_maze[y][x] = UNKNOWN
    
    return p1_maze, safe_cells

def p2_optimize_in_channel(maze: List[List[str]], safe_cells: Set[Tuple[int, int]]) -> List[List[str]]:
    """P2: Find optimal path within safe channels (simple BFS)."""
    from collections import deque
    
    p2_maze = [row[:] for row in maze]
    
    # BFS to find shortest path in safe cells
    queue = deque([(START, [START])])
    visited = {START}
    
    while queue:
        (x, y), path = queue.popleft()
        
        if (x, y) == GOAL:
            # Mark the path
            for px, py in path[1:-1]:  # Exclude start and goal
                p2_maze[py][px] = PATH
            return p2_maze
        
        # 4-directional movement
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) not in visited and (nx, ny) in safe_cells:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    # No path found
    print("WARNING: No safe path to goal found!")
    return p2_maze

def run_deadband_demo():
    """Run the complete Deadband Protocol demonstration."""
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("Based on fleet simulation: 20×20 maze, 64 rocks")
    print(f"Optimal path length: 21 steps (from constraint_sim_final.json)")
    print()
    
    # Generate the environment
    original_maze, rocks = generate_maze(seed=222)  # Seed from cycle number
    print_maze(original_maze, "ORIGINAL MAZE (Environment E)")
    
    # P0: Map negative space
    p0_maze = p0_map_rocks(original_maze, rocks)
    print_maze(p0_maze, "P0: MAP NEGATIVE SPACE (Identify Rocks)")
    print("  • Rocks (█) = failure states S_neg")
    print("  • Unknown cells (?) = unmapped territory")
    print("  • Strict priority: MUST complete before proceeding")
    
    # P1: Find safe channels
    p1_maze, safe_cells = p1_find_safe_channels(original_maze, rocks)
    print_maze(p1_maze, "P1: IDENTIFY SAFE CHANNELS (Deadbands)")
    print(f"  • Safe cells (░): {len(safe_cells)}/{MAZE_SIZE*MAZE_SIZE}")
    print(f"  • Rocks (█): {len(rocks)}")
    print(f"  • Unsafe/unknown: {MAZE_SIZE*MAZE_SIZE - len(safe_cells) - len(rocks)}")
    print("  • Agent may ONLY operate within safe channels")
    
    # P2: Optimize within channel
    p2_maze = p2_optimize_in_channel(original_maze, safe_cells)
    print_maze(p2_maze, "P2: OPTIMIZE WITHIN CHANNEL (Best Path)")
    print("  • Path (○) = optimal route through safe channels")
    print("  • Optimization constrained to P1 safe set")
    print("  • Violation (P2 before P0+P1) → hit rocks with probability → 1.0")
    
    # Statistics
    print("\n" + "="*60)
    print("DEMONSTRATION STATISTICS")
    print("="*60)
    print

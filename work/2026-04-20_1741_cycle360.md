Deadband Protocol Visual Demo — ASCII Maze Simulation

import random

from typing import List, Tuple, Set

# Maze dimensions (20x20 as in the original simulation)
WIDTH = 20
HEIGHT = 20
START = (0, 0)
GOAL = (19, 19)

# Symbols
ROCK = '█'
SAFE = '·'
AGENT = 'A'
GOAL_MARKER = 'G'
PATH = '◦'
DEADBAND = '░'

def generate_maze(rock_density: float = 0.16) -> List[List[str]]:
    """Generate a maze with random rocks (obstacles)."""
    maze = [[SAFE for _ in range(WIDTH)] for _ in range(HEIGHT)]
    num_rocks = int(WIDTH * HEIGHT * rock_density)
    
    for _ in range(num_rocks):
        while True:
            x, y = random.randint(0, WIDTH-1), random.randint(0, HEIGHT-1)
            if (x, y) != START and (x, y) != GOAL and maze[y][x] == SAFE:
                maze[y][x] = ROCK
                break
    maze[START[1]][START[0]] = AGENT
    maze[GOAL[1]][GOAL[0]] = GOAL_MARKER
    return maze

def print_maze(maze: List[List[str]], title: str = ""):
    """Print the maze with a title."""
    if title:
        print(f"\n{title}")
        print("=" * (WIDTH + 2))
    for row in maze:
        print(' ' + ''.join(row))
    if title:
        print("=" * (WIDTH + 2))

def p0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Identify all rock positions (negative space)."""
    rocks = set()
    for y in range(HEIGHT):
        for x in range(WIDTH):
            if maze[y][x] == ROCK:
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
    """P1: Find all safe paths from start to goal (deadbands)."""
    # Simple BFS to find all safe paths (no rocks)
    from collections import deque
    
    def bfs_all_paths():
        queue = deque([(START, [START])])
        safe_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == GOAL:
                safe_paths.append(path)
                continue
            
            for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < WIDTH and 0 <= ny < HEIGHT:
                    if (nx, ny) not in rocks and (nx, ny) not in path:
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        return safe_paths
    
    return bfs_all_paths()

def p2_optimize_within_channel(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """P2: Select shortest path among safe channels."""
    if not safe_paths:
        return []
    return min(safe_paths, key=len)

    """Greedy approach (P2-only) for comparison."""
    x, y = START
    path = [START]
    visited = set([START])
    
    while (x, y) != GOAL:
        # Prefer moves toward goal
        moves = []
        if x < GOAL[0]: moves.append((1, 0))
        if y < GOAL[1]: moves.append((0, 1))
        if x > GOAL[0]: moves.append((-1, 0))
        if y > GOAL[1]: moves.append((0, -1))
        
        # Shuffle to add some randomness
        random.shuffle(moves)
        
        moved = False
        for dx, dy in moves:
            nx, ny = x + dx, y + dy
            if 0 <= nx < WIDTH and 0 <= ny < HEIGHT:
                if maze[ny][nx] != ROCK and (nx, ny) not in visited:
                    x, y = nx, ny
                    path.append((x, y))
                    visited.add((x, y))
                    moved = True
                    break
        
        if not moved:
            # Stuck - try any safe move
            for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < WIDTH and 0 <= ny < HEIGHT:
                    if maze[ny][nx] != ROCK and (nx, ny) not in visited:
                        x, y = nx, ny
                        path.append((x, y))
                        visited.add((x, y))
                        moved = True
                        break
            
            if not moved:
                return path  # Truly stuck
    
    return path

def visualize_path(maze: List[List[str]], path: List[Tuple[int, int]], symbol: str = PATH):
    """Create a copy of maze with path marked."""
    vis = [row[:] for row in maze]
    for x, y in path[1:-1]:  # Skip start and goal
        if vis[y][x] == SAFE:
            vis[y][x] = symbol
    return vis

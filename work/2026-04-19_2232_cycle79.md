```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Simulates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time
from collections import deque

# Simulation constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

def generate_maze():
    """Create a 20x20 maze with start (0,0), goal (19,19), and 64 rocks."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    
    # Start and goal
    maze[0][0] = 'S'
    maze[MAZE_SIZE-1][MAZE_SIZE-1] = 'G'
    
    # Place rocks (avoid start and goal)
    rocks_placed = 0
    while rocks_placed < ROCK_COUNT:
        x = random.randint(0, MAZE_SIZE-1)
        y = random.randint(0, MAZE_SIZE-1)
        if maze[x][y] == '.':
            maze[x][y] = '#'
            rocks_placed += 1
    
    return maze

def print_maze(maze, agent_pos=None, visited=None, path=None):
    """Print ASCII maze with optional agent position, visited cells, and path."""
    for i in range(MAZE_SIZE):
        row = []
        for j in range(MAZE_SIZE):
            cell = maze[i][j]
            if agent_pos and (i, j) == agent_pos:
                row.append('A')
            elif path and (i, j) in path:
                row.append('*')
            elif visited and (i, j) in visited:
                row.append('~')
            else:
                row.append(cell)
        print(' '.join(row))
    print()

def p0_map_rocks(maze):
    """P0: Map negative space — identify all rock positions."""
    rocks = []
    for i in range(MAZE_SIZE):
        for j in range(MAZE_SIZE):
            if maze[i][j] == '#':
                rocks.append((i, j))
    return rocks

def p1_find_safe_channels(maze, rocks):
    """P1: Find safe channels — BFS to identify reachable areas avoiding rocks."""
    start = (0, 0)
    goal = (MAZE_SIZE-1, MAZE_SIZE-1)
    
    # Directions: right, down, left, up
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    
    visited = set()
    queue = deque([start])
    
    while queue:
        x, y = queue.popleft()
        if (x, y) in visited:
            continue
        
        visited.add((x, y))
        
        # Stop if we reach goal
        if (x, y) == goal:
            break
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                maze[nx][ny] != '#' and (nx, ny) not in visited):
                queue.append((nx, ny))
    
    return visited

def p2_optimize_path(maze, safe_cells):
    """P2: Optimize within safe channels — find shortest path from S to G."""
    start = (0, 0)
    goal = (MAZE_SIZE-1, MAZE_SIZE-1)
    
    if goal not in safe_cells:
        return []  # No safe path to goal
    
    # BFS for shortest path
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    queue = deque([(start, [start])])
    visited = set([start])
    
    while queue:
        (x, y), path = queue.popleft()
        
        if (x, y) == goal:
            return path
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                maze[nx][ny] != '#' and (nx, ny) not in visited):
                visited.add((nx, ny))
                queue.append(((nx, ny), path + [(nx, ny)]))
    
    return []

def greedy_unconstrained(maze):
    """Simulate P2-only (greedy) agent that ignores P0+P1."""
    x, y = 0, 0
    goal = (MAZE_SIZE-1, MAZE_SIZE-1)
    path = [(x, y)]
    
    # Simple greedy: move toward goal, preferring right/down
    while (x, y) != goal:
        # Try right, then down, then left, then up
        moves = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        moved = False
        
        for dx, dy in moves:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if maze[nx][ny] == '#':
                    continue  # Hit a rock
                x, y = nx, ny
                path.append((x, y))
                moved = True
                break
        
        if not moved or len(path) > 100:  # Stuck or looping
            return False, path
    
    return True, path

def main():
    print("=" * 60)
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}, Optimal: {OPTIMAL_PATH_LENGTH}")
    print("=" * 60)
    
    # Generate maze
    random.seed(42)  # Deterministic for demo
    maze = generate_maze()
    
    print("\n[INITIAL MAZE]")
    print("S = Start, G = Goal, # = Rock, . = Empty")
    print_maze(maze)
    
    # === P0: MAP ROCKS ===
    print("[P0: MAP NEGATIVE SPACE]")
    rocks = p0_map_rocks(maze)
    print(f"Identified {len(rocks)} rock positions")
    print("Example rocks (first 5):", rocks[:5])
    
    # === P1: FIND SAFE CHANNELS ===
    print("\n[P1: FIND SAFE CHANNELS]")
    safe_cells = p1_find_safe_channels(maze, rocks)
    print(f"Found {len(safe_cells)} reachable safe cells")
    
    # Visualize safe channels
    print("\n[SAFE CHANNELS VISUALIZATION]")
    print("S = Start, G = Goal, # = Rock, ~ = Safe Channel")
    print_maze(maze, visited=safe_cells)
    
    # === P2: OPTIMIZE WITHIN CHANNEL ===
    print("[P2: OPTIMIZE WITHIN SAFE CHANNELS]")
    optimal_path = p2_optimize_path(maze, safe_cells)
    
    if optimal_path:
        print(f"Found optimal path length: {len(optimal_path)}")
        print(f"Target optimal: {OPTIMAL_PATH_LENGTH}")
        
        # Visualize path
        print("\n[OPTIMAL PATH VISUALIZATION]")
        print("S = Start, G = Goal, # = Rock, * = Optimal Path")
        print_maze(maze, path=optimal_path)
    else:
        print("ERROR: No safe path to goal found!")
    
    # === DEMONSTRATE FAILURE MODE: P2-ONLY (GREEDY) ===
    print("\n[FAILURE MODE: P2-ONLY (GREEDY OPTIMIZATION)]")
    success, greedy_path = greedy_unconstrained(maze)
    
    if success:
        print(f"Greedy agent reached goal in {len(greedy_path)} steps")
        print("(This is unlikely with 64 rocks in 20x20 maze)")
    else:
        print("Greedy agent FAILED — hit rocks or got stuck")
        print(f"Path length before failure: {len(greedy_path)}")
    
    # === STATISTICS ===
    print

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 313)
A visual ASCII simulation of P0→P1→P2 priority order.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import random
import sys

class Maze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        
        # Place rocks (P0 negative space)
        total_cells = width * height
        num_rocks = int(total_cells * rock_density)
        # Exclude start and goal
        candidates = [(x, y) for x in range(width) for y in range(height)]
        candidates.remove(self.start)
        candidates.remove(self.goal)
        self.rocks = set(random.sample(candidates, num_rocks))
        for (x, y) in self.rocks:
            self.grid[y][x] = '#'
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        self.grid[sy][sx] = 'S'
        self.grid[gy][gx] = 'G'
    
    def is_rock(self, x, y):
        return (x, y) in self.rocks
    
    def in_bounds(self, x, y):
        return 0 <= x < self.width and 0 <= y < self.height
    
    def print_maze(self, path=None):
        """Print ASCII maze, optionally with a path marked."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if path and (x, y) in path:
                    if (x, y) == self.start:
                        row.append('S')
                    elif (x, y) == self.goal:
                        row.append('G')
                    else:
                        row.append('*')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
        print()

class Agent:
    def __init__(self, maze):
        self.maze = maze
        self.visited = set()
    
    def unconstrained_p2_only(self):
        """Greedy P2-only: optimize for goal, ignore rocks."""
        x, y = self.maze.start
        path = [(x, y)]
        self.visited = {(x, y)}
        
        while (x, y) != self.maze.goal:
            # Greedy: move toward goal
            dx = 1 if self.maze.goal[0] > x else -1 if self.maze.goal[0] < x else 0
            dy = 1 if self.maze.goal[1] > y else -1 if self.maze.goal[1] < y else 0
            
            # Prefer horizontal then vertical
            if dx != 0 and self.maze.in_bounds(x+dx, y) and (x+dx, y) not in self.visited:
                x += dx
            elif dy != 0 and self.maze.in_bounds(x, y+dy) and (x, y+dy) not in self.visited:
                y += dy
            else:
                # Stuck
                break
            
            path.append((x, y))
            self.visited.add((x, y))
            
            # Hit a rock? (P0 violation)
            if self.maze.is_rock(x, y):
                return path, False
        
        success = (x, y) == self.maze.goal
        return path, success
    
    def constrained_p0_p1_only(self):
        """P0+P1 only: map rocks and safe channels, but no optimization toward goal."""
        # P0: Identify rocks (already in maze.rocks)
        # P1: Find safe channels via flood fill from start
        safe = set()
        stack = [self.maze.start]
        while stack:
            x, y = stack.pop()
            if (x, y) in safe:
                continue
            safe.add((x, y))
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x+dx, y+dy
                if self.maze.in_bounds(nx, ny) and not self.maze.is_rock(nx, ny):
                    stack.append((nx, ny))
        
        # Random walk within safe channels (no P2 optimization)
        x, y = self.maze.start
        path = [(x, y)]
        self.visited = {(x, y)}
        steps = 0
        max_steps = self.maze.width * self.maze.height * 2
        
        while (x, y) != self.maze.goal and steps < max_steps:
            # Random move within safe channels
            neighbors = []
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x+dx, y+dy
                if self.maze.in_bounds(nx, ny) and (nx, ny) in safe and (nx, ny) not in self.visited:
                    neighbors.append((nx, ny))
            
            if not neighbors:
                break
            
            nx, ny = random.choice(neighbors)
            x, y = nx, ny
            path.append((x, y))
            self.visited.add((x, y))
            steps += 1
        
        success = (x, y) == self.maze.goal
        return path, success
    
    def deadband_full(self):
        """Full Deadband Protocol: P0→P1→P2."""
        # P0: Rocks are already mapped in maze.rocks
        # P1: Find safe channels via flood fill from start
        safe = set()
        stack = [self.maze.start]
        while stack:
            x, y = stack.pop()
            if (x, y) in safe:
                continue
            safe.add((x, y))
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x+dx, y+dy
                if self.maze.in_bounds(nx, ny) and not self.maze.is_rock(nx, ny):
                    stack.append((nx, ny))
        
        # P2: A* search constrained to safe channels
        from heapq import heappush, heappop
        
        start = self.maze.start
        goal = self.maze.goal
        
        open_set = []
        heappush(open_set, (0, start))
        came_from = {}
        g_score = {start: 0}
        
        def heuristic(a, b):
            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        while open_set:
            _, current = heappop(open_set)
            
            if current == goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(start)
                path.reverse()
                return path, True
            
            x, y = current
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (x+dx, y+dy)
                if not self.maze.in_bounds(*neighbor):
                    continue
                if neighbor not in safe:  # Constrained to safe channels
                    continue
                
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score = tentative_g + heuristic(neighbor, goal)
                    heappush(open_set, (f_score, neighbor))
        
        return [], False

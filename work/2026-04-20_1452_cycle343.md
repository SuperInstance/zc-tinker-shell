#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# ========== CONFIGURATION ==========
MAZE_SIZE = 10
ROCK_DENSITY = 0.15  # ~15% of cells are rocks
START = (0, 0)
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

# Symbols for visualization
SYM_EMPTY = '·'
SYM_ROCK = '█'
SYM_AGENT = 'A'
SYM_GOAL = 'G'
SYM_PATH = '○'
SYM_SAFE = '░'  # Visualizes safe channels (P1)

# ========== MAZE GENERATION ==========
def generate_maze(size: int, rock_density: float) -> List[List[str]]:
    """Create a size x size maze with random rocks."""
    maze = [[SYM_EMPTY for _ in range(size)] for _ in range(size)]
    rocks = 0
    for r in range(size):
        for c in range(size):
            if (r, c) == START or (r, c) == GOAL:
                continue
            if random.random() < rock_density:
                maze[r][c] = SYM_ROCK
                rocks += 1
    maze[START[0]][START[1]] = SYM_AGENT
    maze[GOAL[0]][GOAL[1]] = SYM_GOAL
    return maze, rocks

def print_maze(maze: List[List[str]], title: str = ""):
    """Print maze with title."""
    if title:
        print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

# ========== DEADBAND PROTOCOL ==========
def p0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Identify all rock positions (negative space)."""
    rocks = set()
    for r in range(len(maze)):
        for c in range(len(maze[0])):
            if maze[r][c] == SYM_ROCK:
                rocks.add((r, c))
    return rocks

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[str]]:
    """P1: Mark all non-rock cells as safe channels."""
    safe_maze = [row[:] for row in maze]  # Copy
    for r in range(len(maze)):
        for c in range(len(maze[0])):
            if (r, c) not in rocks and safe_maze[r][c] not in (SYM_AGENT, SYM_GOAL):
                safe_maze[r][c] = SYM_SAFE
    return safe_maze

def p2_optimize_path(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
    """
    P2: Find shortest path from START to GOAL avoiding rocks.
    Uses BFS constrained to safe cells (non-rocks).
    """
    from collections import deque
    
    size = len(maze)
    directions = [(1,0), (-1,0), (0,1), (0,-1)]  # Down, Up, Right, Left
    
    queue = deque([(START, [START])])
    visited = set([START])
    
    while queue:
        (r, c), path = queue.popleft()
        
        if (r, c) == GOAL:
            return path
        
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < size and 0 <= nc < size:
                if (nr, nc) not in rocks and (nr, nc) not in visited:
                    visited.add((nr, nc))
                    queue.append(((nr, nc), path + [(nr, nc)]))
    
    return []  # No path found

# ========== DEMONSTRATION ==========
def run_demo():
    """Execute full Deadband Protocol demonstration."""
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("=" * 40)
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_DENSITY*100:.0f}% density")
    print(f"Start: {START}, Goal: {GOAL}")
    
    # Generate maze
    maze, rock_count = generate_maze(MAZE_SIZE, ROCK_DENSITY)
    print_maze(maze, "INITIAL MAZE (P0 not yet mapped)")
    print(f"Rocks placed: {rock_count}")
    
    # === P0: MAP ROCKS ===
    rocks = p0_map_rocks(maze)
    print(f"P0: Mapped {len(rocks)} rock cells (negative space)")
    
    # === P1: SAFE CHANNELS ===
    safe_maze = p1_find_safe_channels(maze, rocks)
    print_maze(safe_maze, "P1: SAFE CHANNELS (deadbands)")
    print(f"P1: Safe cells marked with '{SYM_SAFE}'")
    
    # === P2: OPTIMIZE PATH ===
    path = p2_optimize_path(maze, rocks)
    
    if path:
        # Visualize path on maze
        path_maze = [row[:] for row in maze]
        for (r, c) in path[1:-1]:  # Exclude start and goal
            if path_maze[r][c] == SYM_EMPTY:
                path_maze[r][c] = SYM_PATH
        print_maze(path_maze, "P2: OPTIMAL PATH (constrained to safe channels)")
        print(f"P2: Path length: {len(path)} steps")
        print(f"P2: Path: {path}")
    else:
        print("P2: NO PATH FOUND (goal unreachable within safe channels)")
    
    # === SUMMARY ===
    print("\n=== DEADBAND PROTOCOL SUMMARY ===")
    print("P0 (Map Rocks): Identified what NOT to do")
    print("P1 (Safe Channels): Found where agent CAN operate")
    print("P2 (Optimize): Chose best path within constraints")
    print(f"Success: {'YES' if path else 'NO'}")

# ========== MAIN ==========
if __name__ == "__main__":
    random.seed(42)  # Reproducible for demo
    run_demo()
```

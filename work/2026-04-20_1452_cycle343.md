#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 strict priority in a simple ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self.agent_pos = self.start
        
        # P0: Map negative space (rocks)
        num_rocks = int(width * height * rock_density)
        for _ in range(num_rocks):
            x = random.randint(0, width-1)
            y = random.randint(0, height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[y][x] = '#'
        
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def print_maze(self, agent_pos=None):
        """Print current maze state with optional agent position."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if agent_pos and (x, y) == agent_pos:
                    row.append('A')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
        print()
    
    def p0_map_negative_space(self):
        """P0: Identify rocks (what NOT to do)."""
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Rocks identified: {len(self.rocks)}")
        print("Positions (x,y):", list(self.rocks)[:10], "..." if len(self.rocks) > 10 else "")
        print()
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find safe paths (deadbands) using BFS that avoids rocks."""
        print("=== P1: FIND SAFE CHANNELS ===")
        
        # BFS to find all reachable positions avoiding rocks
        queue = [self.start]
        visited = {self.start}
        parent = {self.start: None}
        
        while queue:
            x, y = queue.pop(0)
            
            # Check if we reached goal
            if (x, y) == self.goal:
                # Reconstruct path
                path = []
                while (x, y) != self.start:
                    path.append((x, y))
                    x, y = parent[(x, y)]
                path.reverse()
                print(f"Found safe channel to goal (length: {len(path)})")
                return path
            
            # Check neighbors (up, down, left, right)
            for dx, dy in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        parent[(nx, ny)] = (x, y)
                        queue.append((nx, ny))
        
        print("No safe channel found to goal!")
        return None
    
    def p2_optimize_within_channel(self, safe_path):
        """P2: Optimize movement along the safe channel."""
        if not safe_path:
            return None
        
        print("=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        # Simple optimization: move directly along the path
        optimized_path = safe_path  # In a real scenario, we might smooth or shorten
        
        print(f"Following safe channel of {len(optimized_path)} steps")
        print("Path:", optimized_path[:5], "..." if len(optimized_path) > 5 else "")
        
        # Visualize the path
        temp_grid = [row[:] for row in self.grid]
        for x, y in optimized_path:
            if temp_grid[y][x] == '.':
                temp_grid[y][x] = '·'
        
        # Print the channel visualization
        for y in range(self.height):
            row = []
            for x in range(self.width):
                row.append(temp_grid[y][x])
            print(' '.join(row))
        print()
        
        return optimized_path
    
    def run_deadband_protocol(self):
        """Execute full P0→P1→P2 protocol."""
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("=" * 40)
        print(f"Maze: {self.width}x{self.height}, Rocks: {len(self.rocks)}")
        print()
        
        # Initial maze
        print("Initial Maze (S=Start, G=Goal, #=Rock):")
        self.print_maze()
        time.sleep(1)
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        time.sleep(1)
        
        # P1: Find safe channels
        safe_path = self.p1_find_safe_channels()
        time.sleep(1)
        
        if not safe_path:
            print("FAILURE: No safe channel found. Protocol stops at P1.")
            return False
        
        # P2: Optimize within channel
        final_path = self.p2_optimize_within_channel(safe_path)
        time.sleep(1)
        
        # Demonstrate violation (P2-only greedy approach)
        print("\n" + "=" * 40)
        print("VIOLATION DEMO: P2-only (greedy) approach")
        print("=" * 40)
        
        # Greedy approach: move toward goal ignoring rocks
        pos = list(self.start)
        steps = 0
        max_steps = 100
        hit_rock = False
        
        while pos != list(self.goal) and steps < max_steps and not hit_rock:
            # Move toward goal
            dx = 1 if self.goal[0] > pos[0] else -1 if self.goal[0] < pos[0] else 0
            dy = 1 if self.goal[1] > pos[1] else -1 if self.goal[1] < pos[1] else 0
            
            # Try horizontal first, then vertical
            if dx != 0:
                new_pos = (pos[0] + dx, pos[1])
            else:
                new_pos = (pos[0], pos[1] + dy)
            
            # Check for rock
            if new_pos in self.rocks:
                print(f"  Step {steps+1}: Hit rock at {new_pos}! (P2-only fails)")
                hit_rock = True
                break
            
            pos = new_pos
            steps += 1
        
        if hit_rock:
            print("RESULT: P2-only approach fails by hitting rocks (0% success)")
        elif pos == list(self.goal):
            print(f"RESULT: P2-only reached goal in {steps} steps (lucky, not reliable)")
        else:
            print("RESULT: P2-only timed out")
        
        print("\n" + "=" * 40)
        print("PROTOCOL SUMMARY:")
        print(f"P0: Mapped {len(rocks)} rocks")
        print(f"P1: Found safe channel ({len(safe_path)} steps)")
        print(f"P2: Optimized path within safe channel")
        print("Deadband Protocol: Success")
        
        return True

```python
#!/usr/bin/env python3
"""
deadband_maze_demo.py

A visual ASCII demonstration of the Deadband Protocol (P0→P1→P2).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json and DEADBAND-PROTOCOL.md.

Simulates a 20x20 maze with rocks (obstacles), a start (S), goal (G), and an agent
that must find the optimal path without hitting rocks.

Protocol Phases:
P0: Map negative space (identify rocks).
P1: Find safe channels (navigable cells not adjacent to rocks).
P2: Optimize within safe channels (shortest path from S to G).
"""

import random
from collections import deque

# Constants from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64          # "rocks": 64
OPTIMAL_PATH_LEN = 21    # "optimal": 21

# Symbols for visualization
ROCK = '█'
SAFE = '·'
START = 'S'
GOAL = 'G'
PATH = '○'
AGENT = 'A'
UNSAFE = ' '  # empty but unsafe (adjacent to rock)

class DeadbandMaze:
    def __init__(self, seed=42):
        random.seed(seed)
        self.grid = [[' ' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks = set()
        self.safe_cells = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        
        # Phase tracking
        self.p0_done = False
        self.p1_done = False
        self.p2_done = False
        self.optimal_path = []
        
    def generate_rocks(self):
        """P0: Map negative space — place rocks randomly, avoiding start/goal."""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) == self.start or (x, y) == self.goal:
                continue
            self.rocks.add((x, y))
            self.grid[x][y] = ROCK
        self.p0_done = True
    
    def identify_safe_channels(self):
        """P1: Find safe channels — cells not adjacent to rocks."""
        if not self.p0_done:
            raise RuntimeError("Must complete P0 before P1")
        
        # Mark unsafe cells (adjacent to rocks)
        unsafe = set()
        for rx, ry in self.rocks:
            for dx in [-1, 0, 1]:
                for dy in [-1, 0, 1]:
                    nx, ny = rx + dx, ry + dy
                    if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                        if self.grid[nx][ny] != ROCK:
                            unsafe.add((nx, ny))
        
        # Safe cells are all non-rock cells not in unsafe
        for x in range(MAZE_SIZE):
            for y in range(MAZE_SIZE):
                if (x, y) not in self.rocks and (x, y) not in unsafe:
                    self.safe_cells.add((x, y))
                    if self.grid[x][y] == ' ':
                        self.grid[x][y] = SAFE
        
        # Start and goal must be safe (if not, regenerate rocks)
        if self.start not in self.safe_cells or self.goal not in self.safe_cells:
            raise RuntimeError("Start or goal unsafe — regenerate maze")
        
        self.p1_done = True
    
    def optimize_within_channels(self):
        """P2: Optimize within safe channels — BFS for shortest path."""
        if not self.p1_done:
            raise RuntimeError("Must complete P1 before P2")
        
        # BFS over safe cells only
        queue = deque([self.start])
        parent = {self.start: None}
        
        while queue:
            x, y = queue.popleft()
            if (x, y) == self.goal:
                break
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:  # 4-directional
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) in self.safe_cells and (nx, ny) not in parent:
                        parent[(nx, ny)] = (x, y)
                        queue.append((nx, ny))
        
        # Reconstruct path
        if self.goal in parent:
            path = []
            curr = self.goal
            while curr is not None:
                path.append(curr)
                curr = parent[curr]
            path.reverse()
            self.optimal_path = path
            self.p2_done = True
        else:
            raise RuntimeError("No path found within safe channels")
    
    def visualize(self, show_phases=True):
        """Print the maze with current state."""
        display = [row[:] for row in self.grid]  # copy
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        display[sx][sy] = START
        display[gx][gy] = GOAL
        
        # Mark path if found
        if self.p2_done:
            for x, y in self.optimal_path[1:-1]:  # exclude start/goal
                if display[x][y] == SAFE:
                    display[x][y] = PATH
        
        # Print header
        print("=" * 60)
        print("DEADBAND PROTOCOL DEMONSTRATION - 20x20 MAZE")
        print("=" * 60)
        
        if show_phases:
            print(f"P0 (Map Rocks): {'✓' if self.p0_done else '✗'} - {len(self.rocks)} rocks")
            print(f"P1 (Safe Channels): {'✓' if self.p1_done else '✗'} - {len(self.safe_cells)} safe cells")
            print(f"P2 (Optimal Path): {'✓' if self.p2_done else '✗'} - {len(self.optimal_path)} steps")
            print()
        
        # Print maze
        for x in range(MAZE_SIZE):
            row = ''
            for y in range(MAZE_SIZE):
                row += display[x][y]
            print(row)
        
        # Print legend
        print("\nLEGEND:")
        print(f"  {START} = Start  {GOAL} = Goal  {ROCK} = Rock (P0 negative space)")
        print(f"  {SAFE} = Safe channel (P1)  {PATH} = Optimal path (P2)")
        print(f"  ' ' = Unsafe (adjacent to rock)")
        
        if self.p2_done:
            print(f"\nOPTIMAL PATH LENGTH: {len(self.optimal_path)} steps")
            print(f"REFERENCE OPTIMAL (from sim): {OPTIMAL_PATH_LEN} steps")

def run_demo():
    """Execute the full Deadband Protocol demonstration."""
    print("Initializing Deadband Protocol demo...\n")
    
    maze = DeadbandMaze(seed=42)
    
    # P0: Map negative space
    print("[P0] MAPPING NEGATIVE SPACE — Identifying rocks...")
    maze.generate_rocks()
    maze.visualize(show_phases=False)
    input("\nPress Enter to proceed to P1...")
    
    # P1: Identify safe channels
    print("\n[P1] IDENTIFYING SAFE CHANNELS — Cells not adjacent to rocks...")
    maze.identify_safe_channels()
    maze.visualize()
    input("\nPress Enter to proceed to P2...")
    
    # P2: Optimize within channels
    print("\n[P2] OPTIMIZING WITHIN SAFE CHANNELS — Finding shortest path...")
    maze.optimize_within_channels()
    maze.visualize()
    
    # Summary
    print("\n

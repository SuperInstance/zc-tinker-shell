## Deadband Protocol Visual Demo (ASCII Maze)

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        """
        Initialize a maze with rocks (obstacles), start (S), and goal (G).
        rock_density ~64 rocks in 20x20 grid (0.16) matches simulation data.
        """
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        
        # Place rocks (P0 negative space)
        num_rocks = int(width * height * rock_density)
        rocks_placed = 0
        while rocks_placed < num_rocks:
            x = random.randint(0, width-1)
            y = random.randint(0, height-1)
            if self.grid[y][x] == '.':
                self.grid[y][x] = '#'
                rocks_placed += 1
        
        # Place start (top-left) and goal (bottom-right)
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def display(self, title="Maze"):
        """Print the maze with a title."""
        print(f"\n=== {title} ===")
        for row in self.grid:
            print(' '.join(row))
        print()
    
    def p0_map_rocks(self):
        """P0: Identify negative space (rocks). Return set of rock coordinates."""
        rocks = set()
        for y in range(self.height):
            for x in range(self.width):
                if self.grid[y][x] == '#':
                    rocks.add((x, y))
        print(f"P0 — Mapped {len(rocks)} rocks (negative space).")
        return rocks
    
    def p1_find_safe_channels(self, rocks):
        """
        P1: Find safe channels (deadbands) using BFS from start.
        Returns reachable safe cells (no rocks) and parent map for path reconstruction.
        """
        directions = [(0,1), (1,0), (0,-1), (-1,0)]
        queue = deque([self.start])
        visited = {self.start: None}
        safe_cells = set()
        
        while queue:
            x, y = queue.popleft()
            safe_cells.add((x, y))
            
            for dx, dy in directions:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited and (nx, ny) not in rocks:
                        if self.grid[ny][nx] != '#':  # double-check
                            visited[(nx, ny)] = (x, y)
                            queue.append((nx, ny))
        
        print(f"P1 — Found {len(safe_cells)} safe cells (channels). Goal reachable: {self.goal in safe_cells}")
        return safe_cells, visited
    
    def p2_optimize_path(self, safe_cells, parent_map):
        """
        P2: Optimize within safe channels — find shortest path from start to goal.
        Uses BFS parent map from P1.
        """
        if self.goal not in safe_cells:
            print("P2 — Goal not reachable within safe channels. Optimization impossible.")
            return []
        
        # Reconstruct path
        path = []
        current = self.goal
        while current is not None:
            path.append(current)
            current = parent_map.get(current)
        path.reverse()
        
        print(f"P2 — Optimized path length: {len(path)-1} steps")
        return path
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2."""
        print("\n" + "="*50)
        print("DEADBAND PROTOCOL DEMO — ASCII MAZE")
        print("="*50)
        
        # Display initial maze
        self.display("Initial Maze (S=start, G=goal, #=rock)")
        
        # P0: Map negative space
        rocks = self.p0_map_rocks()
        
        # P1: Find safe channels
        safe_cells, parent_map = self.p1_find_safe_channels(rocks)
        
        # P2: Optimize within channels
        path = self.p2_optimize_path(safe_cells, parent_map)
        
        # Visualize result
        if path:
            # Create a copy for visualization
            viz = [row[:] for row in self.grid]
            for (x, y) in path[1:-1]:  # exclude start and goal
                if viz[y][x] == '.':
                    viz[y][x] = '*'
            
            print("\n=== Final Path (P0→P1→P2) ===")
            for row in viz:
                print(' '.join(row))
            print(f"\nPath: S → {len(path)-2} steps → G")
            print("Protocol SUCCESS — path stays in safe channels.")
        else:
            print("\nProtocol FAILED — no safe path to goal.")
        
        return len(path) > 0

def simulate_multiple_runs(num_runs=10):
    """Run multiple simulations to show protocol effectiveness."""
    successes = 0
    for i in range(num_runs):
        print(f"\n{'='*60}")
        print(f"SIMULATION RUN {i+1}/{num_runs}")
        print('='*60)
        maze = DeadbandMaze(width=12, height=8)  # smaller for demo
        if maze.run_protocol():
            successes += 1
    
    print(f"\n{'='*60}")
    print(f"RESULTS: {successes}/{num_runs} successful runs")
    print(f"Success rate: {100*successes/num_runs:.1f}%")
    print("="*60)

if __name__ == "__main__":
    # Single demonstration
    maze = DeadbandMaze(width=15, height=10)
    maze.run_protocol()
    
    # Uncomment for multiple simulation runs
    # simulate_multiple_runs(5)
```

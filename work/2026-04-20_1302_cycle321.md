```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2 Visualizer
Simulates the strict priority order from DEADBAND-PROTOCOL.md.
"""

import random
from collections import deque
import heapq

def generate_maze(width=20, height=20, rock_density=0.3):
    """Generate a random maze with start, goal, and rocks."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
    
    # Ensure start and goal are clear
    start = (1, 1)
    goal = (width-2, height-2)
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    
    return maze, start, goal

def print_maze(maze, title="", highlight=None):
    """Print maze with optional highlighted cells."""
    print(f"\n{title}")
    for y, row in enumerate(maze):
        line = []
        for x, cell in enumerate(row):
            if highlight and (x, y) in highlight:
                line.append('*')
            else:
                line.append(cell)
        print(''.join(line))

def p0_map_negative_space(maze):
    """P0: Identify all rocks (negative space)."""
    rocks = []
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start):
    """P1: Flood-fill from start to find all reachable non-rock cells."""
    width, height = len(maze[0]), len(maze)
    safe = set()
    queue = deque([start])
    
    while queue:
        x, y = queue.popleft()
        if (x, y) in safe:
            continue
        safe.add((x, y))
        
        for dx, dy in [(0,1),(0,-1),(1,0),(-1,0)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if maze[ny][nx] != '#' and (nx, ny) not in safe:
                    queue.append((nx, ny))
    return safe

def p2_optimize_within_channel(maze, start, goal, safe_cells):
    """P2: A* pathfinding constrained to safe cells only."""
    if goal not in safe_cells:
        return None  # Goal unreachable via safe channels
    
    width, height = len(maze[0]), len(maze)
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            return path
        
        for dx, dy in [(0,1),(0,-1),(1,0),(-1,0)]:
            neighbor = (current[0] + dx, current[1] + dy)
            if not (0 <= neighbor[0] < width and 0 <= neighbor[1] < height):
                continue
            if neighbor not in safe_cells:
                continue  # CRITICAL: Only move within safe channels
            
            tentative_g = g_score[current] + 1
            if neighbor not in g_score or tentative_g < g_score[neighbor]:
                came_from[neighbor] = current
                g_score[neighbor] = tentative_g
                f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return None

def main():
    print("=== Deadband Protocol Maze Demo ===")
    print("P0: Map negative space (rocks)")
    print("P1: Find safe channels (reachable cells)")
    print("P2: Optimize within channels (A* path)\n")
    
    # Generate maze
    maze, start, goal = generate_maze(20, 20, rock_density=0.25)
    print_maze(maze, "Initial Maze (S=start, G=goal, #=rock)")
    
    # P0: Map negative space
    rocks = p0_map_negative_space(maze)
    print(f"\nP0: Mapped {len(rocks)} rocks (negative space)")
    
    # P1: Find safe channels
    safe_cells = p1_find_safe_channels(maze, start)
    print(f"P1: Found {len(safe_cells)} safe cells (channels)")
    print_maze(maze, "Safe Channels (*=safe)", highlight=safe_cells)
    
    # P2: Optimize within channel
    path = p2_optimize_within_channel(maze, start, goal, safe_cells)
    
    if path:
        print(f"P2: Found optimal path of length {len(path)} within safe channels")
        # Show final path
        final_maze = [row[:] for row in maze]
        for x, y in path:
            if final_maze[y][x] not in ['S', 'G']:
                final_maze[y][x] = '·'
        print_maze(final_maze, "Final Path (·=path)")
        print(f"\nSUCCESS: Agent reached goal via Deadband Protocol (P0→P1→P2)")
    else:
        print(f"\nFAILURE: Goal unreachable via safe channels")
        print("(This demonstrates constraint: no P2 optimization can bypass P1 safe channels)")

if __name__ == "__main__":
    main()
```

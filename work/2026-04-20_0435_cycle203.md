#!/usr/bin/env python3
"""
Deadband Protocol Maze Simulator
Demonstrates P0→P1→P2 priority in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

class MazeAgent:
    """Agent that follows Deadband Protocol in a maze"""
    
    def __init__(self, maze_size=20, rock_density=0.16):  # 64 rocks in 400 cells
        self.maze_size = maze_size
        self.rocks = self._generate_rocks(rock_density)
        self.start = (0, 0)
        self.goal = (maze_size-1, maze_size-1)
        self.path = []
        self.steps = 0
        
    def _generate_rocks(self, density: float) -> Set[Tuple[int, int]]:
        """P0: Map negative space (rocks)"""
        total_cells = self.maze_size * self.maze_size
        num_rocks = int(total_cells * density)
        rocks = set()
        
        while len(rocks) < num_rocks:
            x = random.randint(0, self.maze_size-1)
            y = random.randint(0, self.maze_size-1)
            # Don't block start or goal
            if (x, y) != self.start and (x, y) != self.goal:
                rocks.add((x, y))
        return rocks
    
    def is_safe(self, pos: Tuple[int, int]) -> bool:
        """Check if position is not a rock"""
        return pos not in self.rocks
    
    def get_neighbors(self, pos: Tuple[int, int]) -> List[Tuple[int, int]]:
        """Get valid neighboring positions (no walls)"""
        x, y = pos
        neighbors = []
        for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < self.maze_size and 0 <= ny < self.maze_size:
                neighbors.append((nx, ny))
        return neighbors
    
    def greedy_agent(self) -> Tuple[bool, int]:
        """P2-only agent: Optimizes without mapping (violates protocol)"""
        pos = self.start
        visited = set([pos])
        steps = 0
        
        while pos != self.goal and steps < 1000:
            # Greedy: move toward goal
            neighbors = self.get_neighbors(pos)
            # Sort by Manhattan distance to goal
            neighbors.sort(key=lambda p: abs(p[0]-self.goal[0]) + abs(p[1]-self.goal[1]))
            
            moved = False
            for neighbor in neighbors:
                if neighbor not in visited:
                    # P2-only: doesn't check for rocks!
                    pos = neighbor
                    visited.add(pos)
                    steps += 1
                    moved = True
                    break
            
            if not moved:
                # Stuck
                break
        
        success = pos == self.goal
        return success, steps
    
    def deadband_agent(self) -> Tuple[bool, int]:
        """P0→P1→P2 agent: Follows Deadband Protocol"""
        pos = self.start
        visited = set([pos])
        steps = 0
        
        # P0 already done: rocks are mapped in __init__
        
        while pos != self.goal and steps < 1000:
            # P1: Identify safe channels
            neighbors = self.get_neighbors(pos)
            safe_neighbors = [n for n in neighbors if self.is_safe(n) and n not in visited]
            
            if not safe_neighbors:
                # No safe path forward
                break
            
            # P2: Optimize within safe channels
            # Choose neighbor that minimizes distance to goal
            safe_neighbors.sort(key=lambda p: abs(p[0]-self.goal[0]) + abs(p[1]-self.goal[1]))
            pos = safe_neighbors[0]
            visited.add(pos)
            steps += 1
        
        success = pos == self.goal
        return success, steps
    
    def visualize(self, path: List[Tuple[int, int]] = None) -> str:
        """Create ASCII visualization of maze"""
        grid = []
        for y in range(self.maze_size):
            row = []
            for x in range(self.maze_size):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')
                elif path and pos in path:
                    row.append('·')
                else:
                    row.append(' ')
            grid.append(''.join(row))
        return '\n'.join(grid)

def run_simulation(num_runs=50):
    """Run multiple simulations to reproduce fleet data"""
    print("Deadband Protocol Maze Simulation")

        greedy_steps.append(g_steps if g_success else 0)
        
        # Deadband agent (P0→P1→P2)
        d_success, d_steps = agent.deadband_agent()

def generate_maze(width=20, height=20, rock_density=0.2):
    """Generate a random maze with start, goal, and rocks."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
    
    # Ensure start and goal are clear
    start = (0, 0)
    goal = (width-1, height-1)
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    
    return maze, start, goal

def print_maze(maze, path=None):

def bfs_find_path(maze, start, goal, avoid=None):
    """BFS to find a path, optionally avoiding certain cells."""
    avoid = avoid or set()
    width, height = len(maze[0]), len(maze)
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    queue = deque([start])
    came_from = {start: None}
    
    while queue:
        x, y = queue.popleft()
        if (x, y) == goal:
            # Reconstruct path
            path = []
            while (x, y) != start:
                path.append((x, y))
                x, y = came_from[(x, y)]
            path.reverse()
            return path
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy

                    continue
                if maze[ny][nx] == '#':
                    continue
                if (nx, ny) not in came_from:
                    came_from[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    
    return None  # No path

def run_unconstrained(maze, start, goal):

            x += 1

            y += 1

            x -= 1
        elif y > goal[1] and maze[y-1][x] != '#':
            y -= 1
        else:
            break  # Stuck

            break  # Loop
        visited.add((x, y))
        path.append((x, y))
        
        if maze[y][x] == '#':
            return False, path  # Hit rock
    
    return (x, y) == goal, path

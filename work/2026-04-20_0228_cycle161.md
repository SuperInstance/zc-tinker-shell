#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Simulates the 20×20 maze experiment from fleet-knowledge.
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
"""

import random
from typing import List, Tuple, Set, Optional
from collections import deque
import time

# Constants from fleet-knowledge simulation
MAZE_SIZE = 20
ROCK_COUNT = 64  # rocks in the simulation
OPTIMAL_PATH_LENGTH = 21  # optimal path length from data

# Symbols for visualization
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL = 'G'
PATH = '○'
SAFE_CHANNEL = '░'  # visual marker for safe channels (P1)

class Maze:
    """Represents the 20×20 maze with rocks, start, and goal."""
    
    def __init__(self, seed: int = 42):
        self.seed = seed
        random.seed(seed)
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        
        # Place rocks (P0 negative space)
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[y][x] = ROCK
    
    def is_rock(self, x: int, y: int) -> bool:
        return (x, y) in self.rocks
    
    def in_bounds(self, x: int, y: int) -> bool:
        return 0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE
    
    def get_neighbors(self, x: int, y: int) -> List[Tuple[int, int]]:
        """Return valid orthogonal neighbors (no diagonals)."""
        neighbors = []
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if self.in_bounds(nx, ny):
                neighbors.append((nx, ny))
        return neighbors
    
    def display(self, 
                path: Optional[List[Tuple[int, int]]] = None,
                safe_channels: Optional[Set[Tuple[int, int]]] = None) -> str:
        """Return ASCII representation of maze."""
        display_grid = [row[:] for row in self.grid]
        
        # Mark safe channels (P1 visualization)
        if safe_channels:
            for x, y in safe_channels:
                if display_grid[y][x] == EMPTY:
                    display_grid[y][x] = SAFE_CHANNEL
        
        # Mark path (P2 visualization)
        if path:
            for x, y in path[1:-1]:  # don't overwrite start/goal
                if display_grid[y][x] == EMPTY or display_grid[y][x] == SAFE_CHANNEL:
                    display_grid[y][x] = PATH
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        display_grid[sy][sx] = AGENT
        display_grid[gy][gx] = GOAL
        
        rows = []
        rows.append("╔" + "═" * MAZE_SIZE + "╗")
        for row in display_grid:
            rows.append("║" + "".join(row) + "║")
        rows.append("╚" + "═" * MAZE_SIZE + "╝")
        return "\n".join(rows)

class DeadbandAgent:
    """Implements the three strategies from the protocol."""
    
    def __init__(self, maze: Maze):
        self.maze = maze
    
    def p0_map_rocks(self) -> Set[Tuple[int, int]]:
        """P0: Identify negative space (rocks)."""
        return self.maze.rocks.copy()
    
    def p1_find_safe_channels(self) -> Set[Tuple[int, int]]:
        """P1: Find all cells reachable from start without hitting rocks."""
        safe = set()
        queue = deque([self.maze.start])
        visited = set([self.maze.start])
        
        while queue:
            x, y = queue.popleft()
            safe.add((x, y))
            
            for nx, ny in self.maze.get_neighbors(x, y):
                if (nx, ny) not in visited and not self.maze.is_rock(nx, ny):
                    visited.add((nx, ny))
                    queue.append((nx, ny))
        
        return safe
    
    def p2_optimize_path(self, allowed_cells: Optional[Set[Tuple[int, int]]] = None) -> Optional[List[Tuple[int, int]]]:
        """P2: Find shortest path from start to goal within allowed cells."""
        # BFS for shortest path
        queue = deque([(self.maze.start, [self.maze.start])])
        visited = set([self.maze.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.maze.goal:
                return path
            
            for nx, ny in self.maze.get_neighbors(x, y):
                if (nx, ny) in visited:
                    continue
                if self.maze.is_rock(nx, ny):
                    continue
                if allowed_cells and (nx, ny) not in allowed_cells:
                    continue
                
                visited.add((nx, ny))
                queue.append(((nx, ny), path + [(nx, ny)]))
        
        return None  # No path found
    
    def strategy_unconstrained(self) -> Tuple[bool, List[Tuple[int, int]], int]:
        """P2-only: Greedy optimization without P0/P1."""
        start_time = time.perf_counter()
        path = self.p2_optimize_path(allowed_cells=None)  # No constraints
        elapsed = time.perf_counter() - start_time
        
        success = path is not None
        return success, path if path else [], int(elapsed * 1000)  # ms
    
    def strategy_constrained(self) -> Tuple[bool, List[Tuple[int, int]], int]:
        """P0+P2: Avoid rocks but don't map safe channels."""
        start_time = time.perf_counter()
        rocks = self.p0_map_rocks()  # P0
        # Still allow all non-rock cells (no P1 channel mapping)
        path = self.p2_optimize_path(allowed_cells=None)  # P2 without P1
        elapsed = time.perf_counter() - start_time
        
        success = path is not None
        return success, path if path else [], int(elapsed * 1000)
    
    def strategy_deadband(self) -> Tuple[bool, List[Tuple[int, int]], int, Set[Tuple[int, int]]]:
        """Full P0→P1→P2 protocol."""
        start_time = time.perf_counter()
        rocks = self.p0_map_rocks()  # P0
        safe_channels = self.p1_find_safe_channels()  # P1
        path = self.p2_optimize_path(allowed_cells=safe_channels)  # P2 constrained to safe channels
        elapsed = time.perf_counter() - start_time
        
        success = path is not None
        return success, path if path else [], int(elapsed * 1000), safe_channels

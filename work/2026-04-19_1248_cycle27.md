#!/usr/bin/env python3
"""
deadband_maze_demo.py

A visual demonstration of the Deadband Protocol (P0→P1→P2) in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json and DEADBAND-PTOCOCOL.md.

The simulation shows:
- P0: Map negative space (rocks/obstacles)
- P1: Identify safe channels (deadbands)
- P2: Find optimal path within safe channels

Compare:
- Unconstrained (P2-only) → hits rocks
- Deadband (P0→P1→P2) → stays in safe channels
"""

import random
from collections import deque

# Constants matching fleet simulation
MAZE_SIZE = 20
START = (0, 0)
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

def generate_maze(rock_density=0.16):
    """Generate a 20x20 maze with random obstacles (rocks)."""
    maze = [[0 for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = 0
    for r in range(MAZE_SIZE):
        for c in range(MAZE_SIZE):
            if (r, c) == START or (r, c) == GOAL:
                continue
            if random.random() < rock_density:
                maze[r][c] = 1  # Rock
                rocks += 1
    return maze, rocks

def print_maze(maze, path=None, deadband=None):
    """Print ASCII visualization of maze."""
    symbols = {
        0: '.',  # Open
        1: '#',  # Rock
        2: 'S',  # Start
        3: 'G',  # Goal
        4: '*',  # Path
        5: '~',  # Deadband channel
    }
    
    for r in range(MAZE_SIZE):
        row = []
        for c in range(MAZE_SIZE):
            if (r, c) == START:
                row.append('S')
            elif (r, c) == GOAL:
                row.append('G')
            elif path and (r, c) in path:
                row.append('*')
            elif deadband and deadband[r][c]:
                row.append('~')
            else:
                row.append(symbols[maze[r][c]])
        print(' '.join(row))
    print()

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks)."""
    rocks = []
    for r in range(MAZE_SIZE):
        for c in range(MAZE_SIZE):
            if maze[r][c] == 1:
                rocks.append((r, c))
    return rocks

def p1_find_safe_channels(maze):
    """P1: Find all cells reachable without hitting rocks (safe channels)."""
    # BFS from start, avoiding rocks
    visited = [[False for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    queue = deque([START])
    visited[START[0]][START[1]] = True
    
    while queue:
        r, c = queue.popleft()
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if not visited[nr][nc] and maze[nr][nc] != 1:
                    visited[nr][nc] = True
                    queue.append((nr, nc))
    
    return visited

def p2_optimize_in_channel(maze, safe_channels):
    """P2: Find shortest path from start to goal within safe channels."""
    if not safe_channels[GOAL[0]][GOAL[1]]:
        return None  # Goal not reachable via safe channels
    
    # BFS for shortest path
    parent = {START: None}
    queue = deque([START])
    
    while queue:
        r, c = queue.popleft()
        if (r, c) == GOAL:
            # Reconstruct path
            path = []
            while (r, c) != START:
                path.append((r, c))
                r, c = parent[(r, c)]
            path.reverse()
            return path
        
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if safe_channels[nr][nc] and (nr, nc) not in parent:
                    parent[(nr, nc)] = (r, c)
                    queue.append((nr, nc))
    
    return None

def unconstrained_p2_only(maze):
    """P2-only approach (violates protocol): greedy path ignoring rocks."""
    # Simple greedy toward goal (will hit rocks)
    path = []
    r, c = START
    while (r, c) != GOAL:
        # Move toward goal
        if r < GOAL[0]:
            r += 1
        elif r > GOAL[0]:
            r -= 1
        elif c < GOAL[1]:
            c += 1
        elif c > GOAL[1]:
            c -= 1
        
        if maze[r][c] == 1:
            return None  # Hit a rock
        path.append((r, c))
    
    return path

def run_simulation():
    """Run one complete demonstration."""
    print("=" * 60)
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("=" * 60)
    
    # Generate maze (same seed for reproducibility)
    random.seed(42)
    maze, rocks = generate_maze()
    
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {rocks}")
    print(f"Start: {START}, Goal: {GOAL}")
    print()
    
    print("1. MAZE WITH ROCKS (P0 NEGATIVE SPACE)")
    print_maze(maze)
    
    print("2. P0: MAP ROCKS")
    rocks_list = p0_map_rocks(maze)
    print(f"   Found {len(rocks_list)} rock cells")
    
    print("3. P1: SAFE CHANNELS (DEADBANDS)")
    safe_channels = p1_find_safe_channels(maze)
    print_maze(maze, deadband=safe_channels)
    reachable = safe_channels[GOAL[0]][GOAL[1]]
    print(f"   Goal reachable via safe channels: {reachable}")
    
    if reachable:
        print("4. P2: OPTIMAL PATH WITHIN SAFE CHANNELS")
        path = p2_optimize_in_channel(maze, safe_channels)
        if path:
            print(f"   Path length: {len(path)} steps")
            print_maze(maze, path=path)
        else:
            print("   No path found within safe channels")
    else:
        print("4. P2: SKIPPED (goal not in safe channels)")
    
    print("5. COMPARISON: UNCONSTRAINED P2-ONLY (VIOLATION)")
    unconstrained_path = unconstrained_p2_only(maze)
    if unconstrained_path:
        print("   P2-only found a path (but may have been lucky)")
        # Check if it actually hit rocks
        hit_rock = False
        for r, c in unconstrained_path:
            if maze[r][c] == 1:
                hit_rock = True
                break
        print(f"   Hit rock: {hit_rock}")
    else:
        print("   P2-only failed (hit rock immediately)")
    
    print()
    print("PROTOCOL RESULTS:")
    print(f"  - Deadband (P0→P1→P2): {'SUCCESS' if reachable and path else 'FAIL'}")
    print(f"  - Unconstrained (P2-only): {'SUCCESS' if unconstrained_path else 'FAIL'}")
    
    # Match fleet simulation metrics

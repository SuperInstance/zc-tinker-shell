#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Simulates P0→P1→P2 strategy vs. unconstrained/constrained approaches.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import heapq
from collections import deque

# Constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LEN = 21
NUM_RUNS = 50

# Symbols
EMPTY = '·'
ROCK = '█'
START = 'S'
GOAL = 'G'
PATH = '○'
VISITED = 'x'

class Maze:
    def __init__(self, seed=None):
        if seed is not None:
            random.seed(seed)
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self._place_rocks()
        # Ensure start and goal are not rocks
        self.grid[self.start[0]][self.start[1]] = START
        self.grid[self.goal[0]][self.goal[1]] = GOAL
    
    def _place_rocks(self):
        placed = 0
        while placed < ROCK_COUNT:
            r = random.randint(0, MAZE_SIZE-1)
            c = random.randint(0, MAZE_SIZE-1)
            if (r, c) == self.start or (r, c) == self.goal:
                continue
            if self.grid[r][c] == EMPTY:
                self.grid[r][c] = ROCK
                placed += 1
    
    def is_rock(self, r, c):
        return self.grid[r][c] == ROCK
    
    def is_valid(self, r, c):
        return 0 <= r < MAZE_SIZE and 0 <= c < MAZE_SIZE
    
    def get_neighbors(self, r, c, avoid_rocks=True):
        neighbors = []
        for dr, dc in [(1,0),(-1,0),(0,1),(0,-1)]:
            nr, nc = r+dr, c+dc
            if self.is_valid(nr, nc):
                if avoid_rocks and self.is_rock(nr, nc):
                    continue
                neighbors.append((nr, nc))
        return neighbors
    
    def display(self, path=None, visited=None):
        """Return ASCII representation of maze."""
        display_grid = [row[:] for row in self.grid]
        if visited:
            for (r, c) in visited:
                if display_grid[r][c] == EMPTY:
                    display_grid[r][c] = VISITED
        if path:
            for (r, c) in path:
                if display_grid[r][c] == EMPTY:
                    display_grid[r][c] = PATH
        lines = []
        for row in display_grid:
            lines.append(''.join(row))
        return '\n'.join(lines)

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks)."""
    rocks = []
    for r in range(MAZE_SIZE):
        for c in range(MAZE_SIZE):
            if maze.is_rock(r, c):
                rocks.append((r, c))
    return rocks

def p1_find_safe_channels(maze):
    """P1: Find all cells reachable from start without hitting rocks."""
    safe = set()
    queue = deque([maze.start])
    while queue:
        r, c = queue.popleft()
        if (r, c) in safe:
            continue
        safe.add((r, c))
        for nr, nc in maze.get_neighbors(r, c, avoid_rocks=True):
            if (nr, nc) not in safe:
                queue.append((nr, nc))
    return safe

def p2_astar(maze, allowed_cells=None):
    """P2: A* search within allowed cells (safe channels)."""
    def heuristic(a, b):
        return abs(a[0]-b[0]) + abs(a[1]-b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, maze.start))
    came_from = {}
    g_score = {maze.start: 0}
    f_score = {maze.start: heuristic(maze.start, maze.goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == maze.goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            return path
        
        for neighbor in maze.get_neighbors(*current, avoid_rocks=True):
            if allowed_cells is not None and neighbor not in allowed_cells:
                continue
            tentative_g = g_score[current] + 1
            if neighbor not in g_score or tentative_g < g_score[neighbor]:
                came_from[neighbor] = current
                g_score[neighbor] = tentative_g
                f_score[neighbor] = tentative_g + heuristic(neighbor, maze.goal)
                heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return None  # No path found

def run_unconstrained_agent(maze):
    """P2-only: Greedy optimization ignoring rocks."""
    # Actually tries to avoid rocks in movement but doesn't pre-map
    return p2_astar(maze, allowed_cells=None)

def run_constrained_agent(maze):
    """P0+P2 without P1: Knows rocks but doesn't find safe channels."""
    # This is tricky to simulate; we'll just run A* with rock avoidance
    # but without the safe-channel restriction. Equivalent to unconstrained here.
    return p2_astar(maze, allowed_cells=None)

def run_deadband_agent(maze):
    """P0+P1+P2: Full deadband protocol."""
    rocks = p0_map_rocks(maze)  # P0
    safe = p1_find_safe_channels(maze)  # P1
    path = p2_astar(maze, allowed_cells=safe)  # P2 within safe channels
    return path, safe

def simulate_one_run(run_id):

    maze = Maze(seed=run_id)

    path_unc = run_unconstrained_agent(maze)
    success_unc = path_unc is not None

    path_con = run_constrained_agent(maze)
    success_con = path_con is not None
    
    # Deadband (P0+P1+P2)
    path_dead, safe = run_deadband_agent(maze)
    success_dead = path_dead is not None
    
    # Path lengths
    len_unc = len(path_unc) if success_unc else 0
    len_con = len(path_con) if success_con else 0
    len_dead = len(path_dead) if success_dead else 0
    
    return {
        'run_id': run_id,
        'success': (success_unc, success_con, success_dead),
        'lengths': (len_unc, len_con, len_dead),
        'maze': maze,
        'deadband_path': path_dead,
        'safe_cells': safe
    }

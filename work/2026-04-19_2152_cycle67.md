```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 67)
Demonstrates P0→P1→P2 priority in a 20×20 maze with 64 rocks.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_count: int = 64):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self.rocks: Set[Tuple[int, int]] = set()
        self._generate_rocks(rock_count)
        
        # P1: Safe channels (deadband paths)
        self.safe_channels: List[List[Tuple[int, int]]] = []
        
        # P2: Optimal path within channels
        self.optimal_path: List[Tuple[int, int]] = []
        
    def _generate_rocks(self, count: int):
        """P0: Generate rocks (negative space)"""
        while len(self.rocks) < count:
            x = random.randint(0, self.width-1)
            y = random.randint(0, self.height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def find_safe_channels(self):
        """P1: Find all safe channels (paths avoiding rocks)"""
        # Simple BFS to find all safe paths
        from collections import deque
        
        visited = set()
        queue = deque([(self.start, [self.start])])
        safe_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            
            # Check all four directions
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) not in self.rocks and
                    (nx, ny) not in visited):
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.safe_channels = safe_paths
        return len(safe_paths)
    
    def optimize_within_channels(self):
        """P2: Find shortest path within safe channels"""
        if not self.safe_channels:
            return None
        
        shortest = min(self.safe_channels, key=len)
        self.optimal_path = shortest
        return len(shortest)
    
    def run_greedy_unconstrained(self):
        """Simulate P2-only (greedy) approach that ignores P0+P1"""
        # Greedy Manhattan distance approach
        path = [self.start]
        x, y = self.start
        visited = set([self.start])
        
        while (x, y) != self.goal:
            # Try to move toward goal
            moves = []
            if x < self.goal[0]:
                moves.append((1, 0))
            if y < self.goal[1]:
                moves.append((0, 1))
            if x > self.goal[0]:
                moves.append((-1, 0))
            if y > self.goal[1]:
                moves.append((0, -1))
            
            # Shuffle to add some randomness
            random.shuffle(moves)
            
            moved = False
            for dx, dy in moves:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) not in visited):
                    
                    # P2-only agent doesn't check rocks!
                    if (nx, ny) in self.rocks:
                        # Hit a rock - greedy fails
                        return None
                    
                    path.append((nx, ny))
                    visited.add((nx, ny))
                    x, y = nx, ny
                    moved = True
                    break
            
            if not moved:
                # Stuck
                return None
        
        return path
    
    def visualize(self, show_path: bool = False):
        """Create ASCII visualization of maze"""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif (x, y) in self.rocks:
                    row.append('█')
                elif show_path and (x, y) in self.optimal_path:
                    row.append('·')
                else:
                    row.append('.')
            grid.append(' '.join(row))
        
        return '\n'.join(grid)

def main():
    print("=== Deadband Protocol Maze Demo ===")
    print("Based on constraint_sim_final.json: 20×20, 64 rocks, optimal=21")
    print()
    
    # Create maze matching simulation parameters
    maze = DeadbandMaze(width=20, height=20, rock_count=64)
    
    print("P0: Negative Space (Rocks)")
    print(f"  Mapped {len(maze.rocks)} rocks")
    print()
    
    print("P1: Safe Channels")
    safe_count = maze.find_safe_channels()
    if safe_count == 0:
        print("  WARNING: No safe channels found! Agent trapped.")
        return
    print(f"  Found {safe_count} safe channels")
    print()
    
    print("P2: Optimize Within Channels")
    optimal_length = maze.optimize_within_channels()
    if optimal_length:
        print(f"  Optimal path length: {optimal_length}")
        print(f"  Matches simulation optimal: {optimal_length == 21}")
    else:
        print("  No optimal path found")
    print()
    
    # Demonstrate P2-only failure
    print("P2-Only (Greedy) Simulation")
    greedy_path = maze.run_greedy_unconstrained()
    if greedy_path is None:
        print("  Result: Failed (hit rocks)")
        print("  Matches simulation: 0/50 success")
    else:
        print(f"  Result: Found path length {len(greedy_path)}")
    print()
    
    # Show visualization
    print("Maze Visualization:")
    print("S = Start, G = Goal, █ = Rock, · = Optimal Path")
    print(maze.visualize(show_path=True))
    
    # Statistics
    print()
    print("=== Statistics ===")
    print(f"Grid size: {maze.width}×{maze.height}")
    print(f"Rocks: {len(maze.rocks)} ({(len(maze.rocks)/(maze.width*maze.height))*100:.1f}% of cells)")
    print(f"Safe channels found: {safe_count}")
    if optimal_length:
        print(f"Optimal path length: {optimal_length}")
        efficiency = (optimal_length / (maze.width + maze.height - 2)) * 100
        print(f"Path efficiency: {efficiency:.1f}% of Manhattan distance")

if __name__ == "__main__":
    main()
```

5. **Visualization**: ASCII grid showing start (S), goal (G), rocks (█), and optimal path (·)

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 with visual ASCII output.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 10, height: int = 8, rock_density: float = 0.25):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (P0 negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        for y in range(height):
            for x in range(width):
                if (x, y) not in [self.start, self.goal] and random.random() < rock_density:
                    self.rocks.add((x, y))
        
        # Initialize state
        self.safe_channels: Set[Tuple[int, int]] = set()
        self.optimal_path: List[Tuple[int, int]] = []
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify rocks (what NOT to do)."""
        return self.rocks.copy()
    
    def p1_find_safe_channels(self) -> Set[Tuple[int, int]]:
        """P1: Find safe channels (where you CAN be)."""
        # Simple flood fill from start, avoiding rocks
        visited = set()
        stack = [self.start]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Check neighbors
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks:
                        stack.append((nx, ny))
        
        self.safe_channels = visited
        return visited
    
    def p2_optimize_within_channel(self) -> List[Tuple[int, int]]:
        """P2: Find shortest path within safe channels."""
        if not self.safe_channels:
            return []
        
        # BFS for shortest path
        queue = [(self.start, [self.start])]
        visited = set()
        
        while queue:
            (x, y), path = queue.pop(0)
            if (x, y) == self.goal:
                self.optimal_path = path
                return path
            
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Only move through safe channels
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if (nx, ny) in self.safe_channels and (nx, ny) not in visited:
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        return []  # No path found
    
    def render(self, show_p0: bool = True, show_p1: bool = False, show_p2: bool = False) -> str:
        """Render maze with current state."""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif show_p0 and pos in self.rocks:
                    row.append('█')  # Rock
                elif show_p2 and pos in self.optimal_path:
                    row.append('·')  # Optimal path
                elif show_p1 and pos in self.safe_channels:
                    row.append('░')  # Safe channel
                else:
                    row.append(' ')
            grid.append(''.join(row))
        return '\n'.join(grid)

def main():
    print("=== Deadband Protocol Maze Demo ===\n")
    
    # Create maze with seed for reproducibility
    random.seed(42)
    maze = DeadbandMaze(width=12, height=8, rock_density=0.28)
    
    print("Initial Maze (Start=S, Goal=G):")
    print(maze.render(show_p0=False))
    print()
    
    # P0: Map negative space
    rocks = maze.p0_map_negative_space()
    print("P0: MAP NEGATIVE SPACE (Rocks = █)")
    print("Identify what NOT to do")
    print(f"Found {len(rocks)} rocks")
    print(maze.render(show_p0=True))
    print()
    
    # P1: Find safe channels
    safe = maze.p1_find_safe_channels()
    print("P1: FIND SAFE CHANNELS (Safe = ░)")
    print("Identify where you CAN be")
    print(f"Found {len(safe)} safe positions")
    print(maze.render(show_p0=True, show_p1=True))
    print()
    
    # P2: Optimize within channel
    path = maze.p2_optimize_within_channel()
    if path:
        print("P2: OPTIMIZE WITHIN CHANNEL (Path = ·)")
        print("Find best path through safe water")
        print(f"Path length: {len(path)} steps")
        print(maze.render(show_p0=True, show_p1=False, show_p2=True))
        print()
        
        # Show combined view
        print("Final Result (All phases combined):")
        combined = []
        for y in range(maze.height):
            row = []
            for x in range(maze.width):
                pos = (x, y)
                if pos == maze.start:
                    row.append('S')
                elif pos == maze.goal:
                    row.append('G')
                elif pos in maze.rocks:
                    row.append('█')
                elif pos in maze.optimal_path:
                    row.append('·')
                elif pos in maze.safe_channels:
                    row.append('░')
                else:
                    row.append(' ')
            combined.append(''.join(row))
        print('\n'.join(combined))
    else:
        print("P2: No path found! Goal unreachable within safe channels.")
    
    print("\n" + "="*40)
    print("Protocol Summary:")
    print("  P0: Map rocks → Avoid failure states")
    print("  P1: Find safe channels → Constrain search space")
    print("  P2: Optimize path → Minimize cost within safety")
    print("\nBased on fleet simulation data:")
    print("  - Unconstrained (P2-only): 0% success")
    print("  - Deadband (P0→P1→P2): 100% success")

if __name__ == "__main__":
    main()
```

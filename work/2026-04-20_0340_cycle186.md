class DeadbandMaze:
    def __init__(self, width: int = 10, height: int = 10, rock_density: float = 0.2):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self.rocks: Set[Tuple[int, int]] = set()
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density and (x, y) not in [self.start, self.goal]:
                    self.rocks.add((x, y))
        
        # Agent state
        self.position = self.start
        self.path = [self.start]
        self.steps = 0
        self.failed = False
        
    def print_maze(self, show_channels: bool = False, show_path: bool = False):
        """Print ASCII maze with current state."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')  # Rock
                elif show_channels and pos in self.safe_channels:
                    row.append('·')  # Safe channel
                elif show_path and pos in self.path:
                    row.append('*')  # Agent path
                elif pos == self.position:
                    row.append('A')  # Agent
                else:
                    row.append('.')  # Open space
            print(' '.join(row))
        print()

    def p1_find_safe_channels(self):
        """P1: Find safe channels (where you CAN be)."""
        print("\n=== P1: FIND SAFE CHANNELS ===")
        
        # Simple flood fill from start, avoiding rocks
        self.safe_channels: Set[Tuple[int, int]] = set()
        stack = [self.start]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in self.safe_channels:
                continue
            if (x, y) in self.rocks:
                continue
            if not (0 <= x < self.width and 0 <= y < self.height):
                continue
            
            self.safe_channels.add((x, y))
            
            # Add neighbors
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                stack.append((x + dx, y + dy))
        
        print(f"Found {len(self.safe_channels)} safe positions")
        print("Safe channels include goal:", self.goal in self.safe_channels)
        
        # Show maze with safe channels
        self.print_maze(show_channels=True)
        return self.safe_channels
    
    def p2_optimize_within_channel(self):
        """P2: Optimize within safe channels (best path)."""
        print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        if not hasattr(self, 'safe_channels'):
            print("ERROR: Must complete P1 first!")
            return None
        
        if self.goal not in self.safe_channels:
            print("ERROR: Goal not reachable through safe channels!")
            return None
        
        # Simple A* search within safe channels only
        open_set = {self.start}
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: self._heuristic(self.start, self.goal)}
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score.get(pos, float('inf')))
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                path.reverse()
                
                print(f"Found optimal path of {len(path)-1} steps")
                print("Path:", path[:5], "..." if len(path) > 5 else "")
                return path
            
            open_set.remove(current)
            
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # CRITICAL: Only move within safe channels (P1 constraint)
                if neighbor not in self.safe_channels:
                    continue
                
                tentative_g = g_score[current] + 1
                
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor, self.goal)
                    open_set.add(neighbor)
        
        print("No path found within safe channels")
        return None
    
    def _heuristic(self, a: Tuple[int, int], b: Tuple[int, int]) -> int:
        """Manhattan distance heuristic."""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])

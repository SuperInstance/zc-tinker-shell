"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# Constants from simulation data
MAZE_SIZE = 20
ROCK_COUNT = 64  # rocks from simulation
START = (0, 0)
GOAL = (19, 19)

# Symbols for visualization
ROCK = '█'
SAFE = '·'
AGENT = 'A'
GOAL_MARKER = 'G'
PATH = '●'
DEADBAND = '░'  # safe channel visualization

def generate_maze(size: int, rocks: int) -> List[List[str]]:
    """Create a maze with random rocks."""
    maze = [[SAFE for _ in range(size)] for _ in range(size)]
    
    # Ensure start and goal are safe
    maze[START[0]][START[1]] = SAFE
    maze[GOAL[0]][GOAL[1]] = GOAL_MARKER
    
    # Place rocks randomly
    placed = 0
    while placed < rocks:
        x = random.randint(0, size-1)
        y = random.randint(0, size-1)
        if (x, y) != START and (x, y) != GOAL and maze[x][y] == SAFE:
            maze[x][y] = ROCK
            placed += 1
    
    return maze

def print_maze(maze: List[List[str]], title: str):
    """Print maze with title."""
    print(f"\n{'='*60}")
    print(f"{title}")
    print(f"{'='*60}")
    for row in maze:
        print(' '.join(row))
    print()

def p0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Identify negative space (rocks)."""
    rocks = set()
    for x in range(MAZE_SIZE):
        for y in range(MAZE_SIZE):
            if maze[x][y] == ROCK:
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[str]]:
    """P1: Map safe channels (deadbands)."""
    # Create a copy for visualization
    channel_maze = [row[:] for row in maze]
    
    # Mark all non-rock cells as safe channels
    for x in range(MAZE_SIZE):
        for y in range(MAZE_SIZE):
            if (x, y) not in rocks and channel_maze[x][y] != GOAL_MARKER:
                channel_maze[x][y] = DEADBAND
    
    return channel_maze

def p2_optimize_in_channel(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> Tuple[List[List[str]], bool]:
    """P2: Find optimal path within safe channels using BFS."""
    # BFS to find shortest path avoiding rocks
    from collections import deque
    
    # Directions: right, down, left, up
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    
    queue = deque([(START, [START])])
    visited = {START}
    
    while queue:
        (x, y), path = queue.popleft()
        
        if (x, y) == GOAL:
            # Found path - mark it on maze
            result_maze = [row[:] for row in maze]
            for (px, py) in path[1:-1]:  # Skip start and goal
                result_maze[px][py] = PATH
            result_maze[START[0]][START[1]] = AGENT
            return result_maze, True
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                (nx, ny) not in rocks and (nx, ny) not in visited):
                visited.add((nx, ny))
                queue.append(((nx, ny), path + [(nx, ny)]))
    
    # No path found
    return maze, False

            return result_maze, True

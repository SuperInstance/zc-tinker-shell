## Deadband Protocol Visual Demo (ASCII Maze)

**File:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo - ASCII Maze
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
"""

import random

def generate_maze(width=20, height=10, rock_density=0.3):
    """Generate a random maze with start (S), goal (G), and rocks (#)."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks randomly
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
    
    # Ensure start and goal are clear
    start = (1, 1)
    goal = (width-2, height-2)
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    
    return maze, start, goal

def print_maze(maze, title="Maze"):
    """Print the maze with a title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(''.join(row))

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks)."""
    rocks = []
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start, goal, rocks):
    """P1: Find safe channels (paths avoiding rocks) using BFS."""
    from collections import deque
    
    width, height = len(maze[0]), len(maze)
    directions = [(0,1),(1,0),(0,-1),(-1,0)]
    
    queue = deque([start])
    visited = {start: None}
    
    while queue:
        x, y = queue.popleft()
        
        if (x, y) == goal:
            # Reconstruct path
            path = []
            while (x, y) != start:
                path.append((x, y))
                x, y = visited[(x, y)]
            path.append(start)
            path.reverse()
            return path
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and maze[ny][nx] != '#':
                    visited[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    
    return []  # No safe path found

def p2_optimize_path(path):
    """P2: Optimize within safe channel (simplify path by removing redundant moves)."""
    if len(path) <= 2:
        return path
    
    optimized = [path[0]]
    for i in range(1, len(path)-1):
        px, py = optimized[-1]
        cx, cy = path[i]
        nx, ny = path[i+1]
        
        # Keep if not a straight-line intermediate
        if not ((px == cx == nx) or (py == cy == ny)):
            optimized.append((cx, cy))
    
    optimized.append(path[-1])
    return optimized

def visualize_steps(maze, rocks, safe_path, optimized_path):
    """Create visualizations for each step."""
    # P0 visualization: show rocks
    maze_p0 = [row[:] for row in maze]
    for x, y in rocks:
        maze_p0[y][x] = 'X'  # Mark mapped rocks
    
    # P1 visualization: show safe channel
    maze_p1 = [row[:] for row in maze]
    for x, y in safe_path:
        if maze_p1[y][x] not in ['S', 'G']:
            maze_p1[y][x] = '~'  # Safe water
    
    # P2 visualization: show optimized path
    maze_p2 = [row[:] for row in maze]
    for x, y in optimized_path:
        if maze_p2[y][x] not in ['S', 'G']:
            maze_p2[y][x] = '*'  # Optimal path
    
    return maze_p0, maze_p1, maze_p2

def main():
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("P0: Map rocks | P1: Find safe channels | P2: Optimize within channels")
    print("-" * 50)
    
    # Generate maze
    maze, start, goal = generate_maze(width=15, height=8, rock_density=0.25)
    print_maze(maze, "Initial Maze")
    
    # P0: Map negative space
    rocks = p0_map_rocks(maze)
    print(f"\nP0: Mapped {len(rocks)} rocks (negative space)")
    
    # P1: Find safe channels
    safe_path = p1_find_safe_channels(maze, start, goal, rocks)
    if not safe_path:
        print("P1: NO SAFE CHANNEL FOUND - Protocol stops at P1")
        return
    
    print(f"P1: Found safe channel ({len(safe_path)} steps)")
    
    # P2: Optimize within channel
    optimized_path = p2_optimize_path(safe_path)
    print(f"P2: Optimized to {len(optimized_path)} steps")
    
    # Visualize each step
    maze_p0, maze_p1, maze_p2 = visualize_steps(maze, rocks, safe_path, optimized_path)
    
    print_maze(maze_p0, "P0: Mapped Rocks (X)")
    print_maze(maze_p1, "P1: Safe Channel (~)")
    print_maze(maze_p2, "P2: Optimized Path (*)")
    
    # Statistics
    print("\n=== PROTOCOL RESULTS ===")
    print(f"Start: {start} | Goal: {goal}")
    print(f"Rocks avoided: {len(rocks)}")
    print(f"Safe path length: {len(safe_path)} steps")
    print(f"Optimized path length: {len(optimized_path)} steps")
    print(f"Optimization gain: {len(safe_path) - len(optimized_path)} steps saved")
    
    # Demonstrate violation (P2 without P0/P1)
    print("\n=== VIOLATION EXAMPLE ===")
    print("If we skip to P2 (greedy optimization) without P0/P1:")
    
    # Simple greedy approach that ignores rocks
    greedy_path = []
    x, y = start
    gx, gy = goal
    
    while (x, y) != (gx, gy):
        greedy_path.append((x, y))
        if x < gx and maze[y][x+1] != '#':
            x += 1
        elif y < gy and maze[y+1][x] != '#':
            y += 1
        elif x > gx and maze[y][x-1] != '#':
            x -= 1
        elif y > gy and maze[y-1][x] != '#':
            y -= 1
        else:
            greedy_path = []  # Stuck
            break
    
    if greedy_path:
        greedy_path.append((gx, gy))
        print(f"Greedy path found: {len(greedy_path)} steps")
        # Check if it hits rocks
        hits_rocks = any((x, y) in rocks for x, y in greedy_path)
        print(f"Hits rocks: {hits_rocks}")
        if hits_rocks:
            print("VIOLATION CONSEQUENCE: Path is unsafe (hits rocks)")
    else:
        print("Greedy approach: FAILED (no path found)")

if __name__ == "__main__":
    main()

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Simulates the constraint theory results from fleet-knowledge.
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    """
    A 20x20 maze simulation of the Deadband Protocol.
    Rocks (P0) are obstacles.
    Safe channels (P1) are clear paths.
    Optimal path (P2) is shortest route within safe channels.
    """
    
    def __init__(self, seed: int = 42):
        self.width = 20
        self.height = 20
        self.start = (0, 0)
        self.goal = (19, 19)
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: Set[Tuple[int, int]] = set()
        self.optimal_path: List[Tuple[int, int]] = []
        
        random.seed(seed)
        self._generate_rocks()
    
    def _generate_rocks(self):
        """P0: Map negative space (64 rocks as in simulation)."""
        # Generate 64 random rock positions
        while len(self.rocks) < 64:
            x = random.randint(0, self.width - 1)
            y = random.randint(0, self.height - 1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def find_safe_channels(self):
        """P1: Identify all positions not in negative space."""
        for x in range(self.width):
            for y in range(self.height):
                if (x, y) not in self.rocks:
                    self.safe_channels.add((x, y))
    
    def find_optimal_path(self):
        """P2: Find shortest path within safe channels using BFS."""
        from collections import deque
        
        # BFS setup
        queue = deque([self.start])
        visited = {self.start: None}
        
        # Directions: right, down, left, up
        directions = [(1, 0), (0, 1), (-1, 0), (0, -1)]
        
        while queue:
            current = queue.popleft()
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current is not None:
                    path.append(current)
                    current = visited[current]
                self.optimal_path = list(reversed(path))
                return True
            
            for dx, dy in directions:
                nx, ny = current[0] + dx, current[1] + dy
                neighbor = (nx, ny)
                
                # Check bounds and safety (P1 constraint)
                if (0 <= nx < self.width and 0 <= ny < self.height and 
                    neighbor in self.safe_channels and 
                    neighbor not in visited):
                    
                    visited[neighbor] = current
                    queue.append(neighbor)
        
        return False  # No path found
    
    def run_protocol(self) -> bool:
        """Execute full P0→P1→P2 protocol."""
        print("=== DEADBAND PROTOCOL DEMO ===")
        print(f"Maze: {self.width}x{self.height}")
        print(f"Rocks: {len(self.rocks)} (P0 negative space)")
        print()
        
        # P0: Already done in init
        print("P0: Negative space mapped ✓")
        
        # P1: Find safe channels
        self.find_safe_channels()
        safe_count = len(self.safe_channels)
        total_cells = self.width * self.height
        print(f"P1: Safe channels identified: {safe_count}/{total_cells} cells ✓")
        
        # P2: Optimize within channels
        success = self.find_optimal_path()
        if success:
            print(f"P2: Optimal path found: {len(self.optimal_path)} steps ✓")
            print(f"    Path cost: {len(self.optimal_path) - 1}")
        else:
            print("P2: No safe path to goal ✗")
        
        return success
    
    def visualize(self, show_path: bool = True):
        """ASCII visualization of the maze."""
        print("\nMAZE VISUALIZATION:")
        print("  S = Start, G = Goal, # = Rock, · = Safe, * = Path")
        print()
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('#')
                elif show_path and pos in self.optimal_path:
                    row.append('*')
                elif pos in self.safe_channels:
                    row.append('·')
                else:
                    row.append(' ')
            print(' '.join(row))
    
    def simulate_greedy_failure(self):
        """Demonstrate P2-only (greedy) failure mode."""
        print("\n=== GREEDY (P2-ONLY) FAILURE SIMULATION ===")
        print("Attempting to optimize without P0+P1...")
        
        # Greedy approach: try to go right/down without checking rocks
        path = [self.start]
        current = self.start
        steps = 0
        
        while current != self.goal and steps < 100:
            # Greedy move: prefer right, then down
            next_pos = None
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = current[0] + dx, current[1] + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    test_pos = (nx, ny)
                    if test_pos not in path:  # Simple avoidance
                        next_pos = test_pos
                        break
            
            if next_pos is None or next_pos in self.rocks:
                print(f"  Hit rock at step {steps} ✗")
                return False
            
            path.append(next_pos)
            current = next_pos
            steps += 1
        
        if current == self.goal:
            print(f"  Reached goal in {steps} steps (lucky!)")
            return True
        else:
            print(f"  Failed after {steps} steps")
            return False


def main():
    """Run the demonstration."""
    # Create maze matching simulation parameters
    maze = DeadbandMaze(seed=219)  # Using cycle number as seed
    
    # Run full protocol
    print("=" * 50)
    success = maze.run_protocol()
    
    if success:
        # Show visualization
        maze.visualize(show_path=True)
        
        # Compare with simulation results
        print("\n" + "=" * 50)
        print("SIMULATION VALIDATION:")
        print(f"Optimal path length: {len(maze.optimal_path) - 1}")
        print("Expected optimal (from constraint_sim_final.json): 21")
        print("Deadband success rate in simulation: 50/50")
        
        # Demonstrate greedy failure
        maze.simulate_greedy_failure()
        print("\nGreedy (unconstrained) success in simulation: 0/50")
    else:
        maze.visualize(show_path=False)
        print("\nNo path found - rare case where rocks block all paths")


if __name__ == "__main__":
    main()
```

**Key Features:**
1. **P0 Implementation**: Generates 64 rocks (negative space) matching the simulation data
2. **P1 Implementation**: Identifies all safe cells (not rocks)
3. **P2 Implementation**: BFS search constrained to safe channels only
4. **Visualization**: ASCII grid showing rocks, safe cells, and optimal path
5. **Greedy Failure Demo**: Shows what happens when skipping P0+P1
6. **Validation**: Compares results with actual fleet simulation data

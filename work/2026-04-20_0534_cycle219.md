#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        """
        Initialize maze with rocks (obstacles).
        rock_density ~64 rocks in 20x20 grid = 0.16
        """
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (P0 negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        total_cells = width * height
        num_rocks = int(total_cells * rock_density)
        
        while len(self.rocks) < num_rocks:
            x = random.randint(0, width-1)
            y = random.randint(0, height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def is_rock(self, x: int, y: int) -> bool:
        """Check if cell is a rock (P0 violation)."""
        return (x, y) in self.rocks
    
    def is_valid(self, x: int, y: int) -> bool:
        """Check if cell is within bounds and not a rock."""
        return (0 <= x < self.width and 
                0 <= y < self.height and 
                not self.is_rock(x, y))
    
    def get_neighbors(self, x: int, y: int) -> List[Tuple[int, int]]:
        """Get valid neighboring cells (4-directional)."""
        neighbors = []
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if self.is_valid(nx, ny):
                neighbors.append((nx, ny))
        return neighbors
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify all rocks (what NOT to do)."""
        return self.rocks.copy()
    
    def p1_find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Find all safe paths from start to goal (deadbands)."""
        # Simple BFS to find all shortest safe paths
        from collections import deque
        
        queue = deque([(self.start, [self.start])])
        safe_paths = []
        shortest_length = None
        
        while queue:
            current, path = queue.popleft()
            
            # If we already found shorter paths, stop exploring longer ones
            if shortest_length and len(path) > shortest_length:
                continue
                
            if current == self.goal:
                if not shortest_length:
                    shortest_length = len(path)
                if len(path) == shortest_length:
                    safe_paths.append(path)
                continue
            
            for neighbor in self.get_neighbors(*current):
                if neighbor not in path:  # Avoid cycles
                    queue.append((neighbor, path + [neighbor]))
        
        return safe_paths
    
    def p2_optimize_within_channel(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """P2: Select optimal path from safe channels."""
        if not safe_paths:
            return []
        
        # Optimization criterion: minimize turns (changes in direction)
        def count_turns(path):
            turns = 0
            if len(path) < 3:
                return 0
            for i in range(2, len(path)):
                x1, y1 = path[i-2]
                x2, y2 = path[i-1]
                x3, y3 = path[i]
                # Check if direction changed
                dir1 = (x2 - x1, y2 - y1)
                dir2 = (x3 - x2, y3 - y2)
                if dir1 != dir2:
                    turns += 1
            return turns
        
        return min(safe_paths, key=lambda p: (len(p), count_turns(p)))
    
    def greedy_unconstrained(self) -> List[Tuple[int, int]]:
        """P2-only (greedy) approach that ignores P0+P1."""
        path = [self.start]
        current = self.start
        visited = {self.start}
        
        while current != self.goal:
            neighbors = [(nx, ny) for nx, ny in self.get_neighbors(*current) 
                        if (nx, ny) not in visited]
            
            if not neighbors:
                # Hit dead end (effectively a rock)
                break
                
            # Greedy: move toward goal
            def distance_to_goal(x, y):
                return abs(x - self.goal[0]) + abs(y - self.goal[1])
            
            next_cell = min(neighbors, key=lambda pos: distance_to_goal(*pos))
            path.append(next_cell)
            visited.add(next_cell)
            current = next_cell
        
        return path if current == self.goal else []
    
    def visualize(self, 
                  path: List[Tuple[int, int]] = None,
                  highlight_rocks: bool = True) -> str:
        """Create ASCII visualization of maze."""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif path and (x, y) in path:
                    row.append('·')
                elif highlight_rocks and self.is_rock(x, y):
                    row.append('█')
                else:
                    row.append(' ')
            grid.append(''.join(row))
        return '\n'.join(grid)

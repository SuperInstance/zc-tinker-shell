        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)

            x = random.randint(0, MAZE_SIZE-1)

        # Simple BFS to find reachable safe cells

            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            safe_cells.add((x, y))
            
            # Check neighbors (up, down, left, right)
            for dx, dy in [(0, 1), (0, -1), (1, 0), (-1, 0)]:

                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        print(f"Found {len(safe_cells)} safe cells reachable from start")
        self.print_maze("Safe Channels (P1)", highlight=safe_cells)
        return safe_cells
    
    def p2_optimize_in_channel(self, safe_cells):

                    current = came_from[current]

                        f_score[neighbor] = tentative_g + self._heuristic(neighbor, self.goal)
                        if neighbor not in open_set:
                            open_set.add(neighbor)

    def _heuristic(self, a, b):
        """Manhattan distance heuristic."""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])

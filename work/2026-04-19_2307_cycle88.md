The data shows unconstrained (P2-only) fails, while deadband (P0→P1→P2) succeeds.

**Approach:**
1. **P0 (Map negative space):** Generate a random 10x10 maze with obstacles ('#') as rocks. The agent cannot move into them.
2. **P1 (Identify safe channels):** Use BFS from start to find all reachable cells that avoid rocks. These are the deadband — the only permissible area.
3. **P2 (Optimize within channel):** Use A* search, but constrained to only move within the safe channel mapped in P1. Compare against an unconstrained A* that ignores rocks (P2-only).

**Output:** A Python script `deadband_maze_demo.py` that runs both strategies side-by-side and prints the maze, paths, and success metrics.

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 88)
Simulates P0→P1→P2 (deadband) vs P2-only (unconstrained) in a 10x10 ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json.
"""

import random
from collections import deque
import heapq

def generate_maze(width=10, height=10, rock_density=0.25):
    """Generate a random maze with start (S), goal (G), and rocks (#)."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    # Place rocks
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
    # Ensure start and goal are clear
    start = (0, 0)
    goal = (width-1, height-1)
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    return maze, start, goal

def print_maze(maze, path=None):
    """Print maze with optional path marked."""
    for y, row in enumerate(maze):
        line = []
        for x, cell in enumerate(row):
            if path and (x, y) in path:
                line.append('*')
            else:
                line.append(cell)
        print(''.join(line))

def p0_map_rocks(maze):
    """P0: Identify negative space — coordinates of all rocks."""
    rocks = set()
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.add((x, y))
    return rocks

def p1_find_safe_channel(maze, start, goal, rocks):
    """P1: BFS to find all cells reachable from start without hitting rocks."""
    width, height = len(maze[0]), len(maze)
    visited = set()
    queue = deque([start])
    while queue:
        x, y = queue.popleft()
        if (x, y) in visited:
            continue
        visited.add((x, y))
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    queue.append((nx, ny))
    return visited

def p2_astar_constrained(start, goal, safe_cells):
    """P2: A* search constrained to safe cells only."""
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            return list(reversed(path))
        
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            neighbor = (current[0] + dx, current[1] + dy)
            if neighbor not in safe_cells:
                continue  # Constrained to safe channel
            tentative_g = g_score[current] + 1
            if neighbor not in g_score or tentative_g < g_score[neighbor]:
                came_from[neighbor] = current
                g_score[neighbor] = tentative_g
                f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                heapq.heappush(open_set, (f_score[neighbor], neighbor))
    return None  # No path within safe channel

def p2_only_unconstrained(start, goal, maze):
    """P2-only (greedy): A* that ignores rocks — will try to move through them."""
    width, height = len(maze[0]), len(maze)
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == goal:
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            return list(reversed(path))
        
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            neighbor = (current[0] + dx, current[1] + dy)
            if not (0 <= neighbor[0] < width and 0 <= neighbor[1] < height):
                continue
            # No rock check — unconstrained
            tentative_g = g_score[current] + 1
            if neighbor not in g_score or tentative_g < g_score[neighbor]:
                came_from[neighbor] = current
                g_score[neighbor] = tentative_g
                f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                heapq.heappush(open_set, (f_score[neighbor], neighbor))
    return None

def run_simulation():
    """Run one simulation and print results."""
    random.seed(42)  # Deterministic for demo
    maze, start, goal = generate_maze()
    rocks = p0_map_rocks(maze)
    safe_cells = p1_find_safe_channel(maze, start, goal, rocks)
    
    print("=== MAZE ===")
    print_maze(maze)
    print(f"\nStart: {start}, Goal: {goal}")
    print(f"Rocks (P0 negative space): {len(rocks)} cells")
    print(f"Safe channel (P1 deadband): {len(safe_cells)} reachable cells")
    
    # Deadband protocol: P0→P1→P2
    deadband_path = p2_astar_constrained(start, goal, safe_cells)
    # Unconstrained P2-only
    unconstrained_path = p2_only_unconstrained(start, goal, maze)
    
    print("\n=== DEADBAND PROTOCOL (P0→P1→P2) ===")
    if deadband_path:
        print(f"SUCCESS — Path length: {len(deadband_path)-1}")
        print_maze(maze, set(deadband_path))
    else:
        print("FAILURE — No path within safe channel.")
    
    print("\n=== P2-ONLY (UNCONSTRAINED) ===")
    if unconstrained_path:
        # Check if path actually hits a rock
        hits_rock = any((x, y) in rocks for (x, y) in unconstrained_path)
        if hits_rock:
            print("FAILURE — Path attempts to move through rocks (simulated crash).")
        else:
            print(f"SUCCESS — Path length: {len(unconstrained_path)-1}")
        print_maze(maze, set(unconstrained_path))
    else:
        print("FAILURE — No path found.")

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Based on fleet simulation data: optimal path length = 21, rocks = 64
"""

import random
from typing import List, Tuple, Set

# Simulation parameters from fleet-knowledge
OPTIMAL_PATH_LEN = 21
ROCK_COUNT = 64
GRID_SIZE = 20  # 20x20 maze from the simulation

def generate_maze() -> Tuple[List[List[str]], Tuple[int, int], Tuple[int, int]]:
    """Generate a 20x20 maze with rocks, start, and goal."""
    maze = [['.' for _ in range(GRID_SIZE)] for _ in range(GRID_SIZE)]
    
    # Place rocks (P0 negative space)
    rocks_placed = 0
    while rocks_placed < ROCK_COUNT:
        x, y = random.randint(0, GRID_SIZE-1), random.randint(0, GRID_SIZE-1)
        if maze[y][x] == '.':
            maze[y][x] = '#'
            rocks_placed += 1
    
    # Place start (top-left) and goal (bottom-right)
    start = (0, 0)
    goal = (GRID_SIZE-1, GRID_SIZE-1)
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    
    return maze, start, goal

def p0_map_negative_space(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Identify all rocks (states to avoid)."""
    rocks = set()
    for y in range(GRID_SIZE):
        for x in range(GRID_SIZE):
            if maze[y][x] == '#':
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(maze: List[List[str]], start: Tuple[int, int], 
                         goal: Tuple[int, int], rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
    """P1: Find all safe paths (deadbands) avoiding rocks."""
    def dfs(pos: Tuple[int, int], path: List[Tuple[int, int]], 
            visited: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
        if pos == goal:
            return [path.copy()]
        
        x, y = pos
        safe_paths = []
        
        # Check four possible moves (right, down, left, up)
        for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < GRID_SIZE and 0 <= ny < GRID_SIZE:
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited.add((nx, ny))
                    path.append((nx, ny))
                    safe_paths.extend(dfs((nx, ny), path, visited))
                    path.pop()
                    visited.remove((nx, ny))
        
        return safe_paths
    
    return dfs(start, [start], {start})

def p2_optimize_within_channel(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """P2: Select shortest path from safe channels."""
    if not safe_paths:
        return []
    return min(safe_paths, key=len)

def visualize_maze(maze: List[List[str]], path: List[Tuple[int, int]] = None) -> str:
    """Create ASCII visualization."""
    vis = []
    path_set = set(path) if path else set()
    
    for y in range(GRID_SIZE):
        row = []
        for x in range(GRID_SIZE):
            cell = maze[y][x]
            if (x, y) in path_set and cell not in ['S', 'G']:
                row.append('*')  # Path marker
            else:
                row.append(cell)
        vis.append(' '.join(row))
    
    return '\n'.join(vis)

def run_deadband_demo():
    """Execute full P0→P1→P2 protocol."""
    print("=" * 60)
    print("DEADBAND PROTOCOL DEMO - MAZE NAVIGATION")
    print("=" * 60)
    
    # Generate environment
    maze, start, goal = generate_maze()
    print("\n1. MAZE ENVIRONMENT (20×20):")
    print(f"   Start: {start}, Goal: {goal}")
    print(f"   Rocks: {ROCK_COUNT} (P0 negative space)")
    print(visualize_maze(maze))
    
    # P0: Map negative space
    print("\n2. P0: MAP NEGATIVE SPACE")
    rocks = p0_map_negative_space(maze)
    print(f"   Identified {len(rocks)} rock positions")
    print(f"   Agent knows what NOT to do")
    
    # P1: Find safe channels
    print("\n3. P1: FIND SAFE CHANNELS")
    safe_paths = p1_find_safe_channels(maze, start, goal, rocks)
    print(f"   Found {len(safe_paths)} safe paths (deadbands)")
    print(f"   Agent knows where it CAN operate")
    
    if not safe_paths:
        print("   WARNING: No safe paths found! Protocol fails at P1.")
        return
    
    # P2: Optimize within channel
    print("\n4. P2: OPTIMIZE WITHIN CHANNEL")
    optimal_path = p2_optimize_within_channel(safe_paths)
    path_length = len(optimal_path) - 1  # Exclude start
    
    print(f"   Selected path length: {path_length}")
    print(f"   Fleet optimal: {OPTIMAL_PATH_LEN}")
    print(f"   Difference: {abs(path_length - OPTIMAL_PATH_LEN)}")
    
    # Visualization
    print("\n5. FINAL PATH (P0→P1→P2):")
    print("   S = Start, G = Goal, # = Rock, * = Safe Path")
    print(visualize_maze(maze, optimal_path))
    
    # Protocol validation
    print("\n6. PROTOCOL VALIDATION:")
    print(f"   P0 completed: ✓ ({len(rocks)} rocks mapped)")
    print(f"   P1 completed: ✓ ({len(safe_paths)} safe channels)")
    print(f"   P2 completed: ✓ (path length {path_length})")
    
    # Check for violations
    path_positions = set(optimal_path)
    rock_collisions = path_positions.intersection(rocks)
    
    if rock_collisions:
        print(f"   VIOLATION: Path intersects rocks at {len(rock_collisions)} positions!")
    else:
        print(f"   SUCCESS: Path avoids all rocks (Deadband Protocol followed)")
    
    print("\n" + "=" * 60)

if __name__ == "__main__":
    # Run multiple trials to show consistency
    for trial in range(3):
        print(f"\n{' TRIAL ' + str(trial+1) + ' ':=^60}")
        run_deadband_demo()
```

**Key Features:**
1. **P0 Implementation**: `p0_map_negative_space()` identifies all rock positions
2. **P1 Implementation**: `p1_find_safe_channels()` finds all paths avoiding rocks
3. **P2 Implementation**: `p2_optimize_within_channel()` selects shortest safe path
4. **Visualization**: ASCII maze showing S (start), G (goal), # (rocks), * (path)
5. **Protocol Validation**: Checks for P0→P1→P2 compliance and rock collisions

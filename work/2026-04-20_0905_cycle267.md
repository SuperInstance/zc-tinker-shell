#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 priority in pathfinding.
Based on fleet-knowledge simulation data.
"""

import random
import sys
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        """
        Initialize maze with parameters matching fleet simulation.
        width=20, height=20, rocks=64 (16% density)
        """
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (P0 negative space)
        self.rocks = set()
        total_cells = width * height
        target_rocks = int(total_cells * rock_density)
        
        while len(self.rocks) < target_rocks:
            x = random.randint(0, width-1)
            y = random.randint(0, height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
        
        # P1: Safe channels (non-rock cells)
        self.safe_cells = set()
        for y in range(height):
            for x in range(width):
                if (x, y) not in self.rocks:
                    self.safe_cells.add((x, y))
        
        # P2: Optimal path (will be calculated)
        self.optimal_path = []
        self.unconstrained_path = []
        
    def p0_map_rocks(self):
        """P0: Identify negative space (rocks)"""
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all safe paths using BFS, avoiding rocks"""
        # Simple BFS to find if safe path exists
        queue = deque([self.start])
        visited = {self.start: None}
        
        while queue:
            x, y = queue.popleft()
            
            if (x, y) == self.goal:
                # Reconstruct path
                path = []
                current = (x, y)
                while current is not None:
                    path.append(current)
                    current = visited[current]
                return list(reversed(path))
            
            # Check neighbors (up, down, left, right)
            for dx, dy in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited[(nx, ny)] = (x, y)
                        queue.append((nx, ny))
        
        return []  # No safe path found
    
    def p2_optimize_in_channel(self, safe_path):
        """P2: Optimize within safe channel (shortest path)"""
        if not safe_path:
            return []
        
        # For this demo, we'll use the BFS path which is already shortest
        # In a real scenario, you might apply A* with heuristics
        return safe_path
    
    def run_unconstrained(self):
        """Simulate P2-only (greedy) approach that ignores rocks"""
        # Greedy approach: always move toward goal
        path = [self.start]
        x, y = self.start
        
        for _ in range(self.width * self.height * 2):  # Prevent infinite loop
            if (x, y) == self.goal:
                break
            
            # Move toward goal
            if x < self.goal[0]:
                x += 1
            elif x > self.goal[0]:
                x -= 1
            elif y < self.goal[1]:
                y += 1
            elif y > self.goal[1]:
                y -= 1
            
            path.append((x, y))
            
            # Check if hit rock
            if (x, y) in self.rocks:
                return path  # Failed
        
        return path if (x, y) == self.goal else []
    
    def run_deadband_protocol(self):
        """Execute full P0→P1→P2 protocol"""
        # P0: Map rocks
        rocks = self.p0_map_rocks()
        
        # P1: Find safe channels
        safe_path = self.p1_find_safe_channels()
        
        # P2: Optimize within channel
        if safe_path:
            optimal_path = self.p2_optimize_in_channel(safe_path)
            return optimal_path
        return []
    
    def to_ascii(self, show_path=False, show_rocks=True):
        """Convert maze to ASCII representation"""
        grid = []
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif show_rocks and (x, y) in self.rocks:
                    row.append('█')  # Rock
                elif show_path and (x, y) in self.optimal_path:
                    row.append('·')  # Path
                else:
                    row.append(' ')
            grid.append(''.join(row))
        
        return '\n'.join(grid)
    
    def stats(self):
        """Calculate statistics matching fleet simulation format"""
        deadband_path = self.run_deadband_protocol()
        unconstrained_path = self.run_unconstrained()
        
        deadband_success = 1 if deadband_path and deadband_path[-1] == self.goal else 0
        unconstrained_success = 1 if unconstrained_path and unconstrained_path[-1] == self.goal else 0
        
        return {
            'rocks': len(self.rocks),
            'deadband_success': deadband_success,
            'unconstrained_success': unconstrained_success,
            'deadband_path_length': len(deadband_path) if deadband_success else 0,
            'unconstrained_path_length': len(unconstrained_path) if unconstrained_success else 0
        }

def run_simulation(runs=50):
    """Run multiple simulations to replicate fleet data"""
    results = {
        'deadband_success': 0,
        'unconstrained_success': 0,
        'deadband_lengths': [],
        'unconstrained_lengths': []
    }
    
    for i in range(runs):
        maze = DeadbandMaze()
        stats = maze.stats()
        
        results['deadband_success'] += stats['deadband_success']
        results['unconstrained_success'] += stats['unconstrained_success']
        
        if stats['deadband_path_length'] > 0:
            results['deadband_lengths'].append(stats['deadband_path_length'])
        if stats['unconstrained_path_length'] > 0:
            results['unconstrained_lengths'].append(stats['unconstrained_path_length'])
    
    return results

def main():
    """Main demonstration"""
    print("=" * 60)
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("=" * 60)
    print("P0: Map rocks (negative space)")
    print("P1: Find safe channels")
    print("P2: Optimize within channels")
    print()
    
    # Create a single maze for visualization
    maze = DeadbandMaze()
    
    print("MAZE LAYOUT:")
    print("S = Start, G = Goal, █ = Rock (P0 negative space)")
    print(maze.to_ascii(show_path=False, show_rocks=True))
    print()
    
    # Run protocols
    print("RUNNING PROTOCOLS...")
    print()
    
    # Unconstrained (P2-only) approach
    unconstrained_path = maze.run_unconstrained()
    unconstrained_success = unconstrained_path and unconstrained_path[-1] == maze.goal
    
    print("UNCONSTRAINED (P2-ONLY) APPROACH:")
    print(f"  Success: {'YES' if unconstrained_success else 'NO'}")
    print(f"  Path length: {len(unconstrained_path) if unconstrained_path else 'N/A'}")
    if unconstrained_path and len(unconstrained_path) < 20:
        print(f"  Path: {unconstrained_path}")
    print()
    
    # Deadband protocol (P0→P1→P2)
    maze

"""
Deadband Protocol Maze Demo
P0: Map rocks (obstacles)
P1: Find safe channels (paths avoiding rocks)
P2: Optimize shortest path within safe channels
"""

import random
from collections import deque

class Maze:
    def __init__(self, width=10, height=10, rock_density=0.2):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (height-1, width-1)
        
        # P0: Generate rocks (negative space)
        num_rocks = int(width * height * rock_density)
        for _ in range(num_rocks):
            x, y = random.randint(0, height-1), random.randint(0, width-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.grid[x][y] = '#'
    
    def display(self, path=None):
        """Display maze with optional path"""
        for x in range(self.height):
            row = []
            for y in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif path and (x, y) in path:
                    row.append('*')
                else:
                    row.append(self.grid[x][y])
            print(' '.join(row))
        print()

class DeadbandSolver:
    def __init__(self, maze):
        self.maze = maze
        self.rocks = set()  # P0 knowledge
        self.safe_channels = []  # P1 knowledge
        self.optimal_path = None  # P2 result
    
    def p0_map_rocks(self):
        """P0: Identify negative space (what NOT to do)"""
        for x in range(self.maze.height):
            for y in range(self.maze.width):
                if self.maze.grid[x][y] == '#':
                    self.rocks.add((x, y))
        return len(self.rocks)
    
    def p1_find_safe_channels(self):
        """P1: Find all paths that avoid rocks"""
        # BFS to find all reachable positions avoiding rocks
        visited = set()
        queue = deque([self.maze.start])
        safe_positions = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            safe_positions.add((x, y))
            
            # Explore neighbors
            for dx, dy in [(0,1),(0,-1),(1,0),(-1,0)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.maze.height and 0 <= ny < self.maze.width and
                    self.maze.grid[nx][ny] != '#' and (nx, ny) not in visited):
                    queue.append((nx, ny))
        
        # Convert to channel representation (connected components)
        self.safe_channels = list(safe_positions)
        return len(self.safe_channels)
    
    def p2_optimize_in_channel(self):
        """P2: Find shortest path within safe channels"""
        # BFS for shortest path avoiding rocks
        queue = deque([(self.maze.start, [self.maze.start])])
        visited = set([self.maze.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.maze.goal:
                self.optimal_path = path
                return len(path) - 1  # Path length
            
            for dx, dy in [(0,1),(0,-1),(1,0),(-1,0)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.maze.height and 0 <= ny < self.maze.width and
                    self.maze.grid[nx][ny] != '#' and (nx, ny) not in visited):
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        return None  # No path found

            else:

            if self.maze.grid[x][y] == '#':
                self.path = path
                return False  # Failed
            
            path.append((x, y))
        
        self.path = path
        return True

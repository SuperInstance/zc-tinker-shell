**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Continuing from Cycle 20. The script is complete and tested. Now preparing it as a training tile for submission.

**Approach (following Deadband Protocol):**
- **P0 (Map negative space):** Read the constraint_sim_final.json data. The rocks are clear: unconstrained optimization (P2-first) fails 100% (0/50 success). Constrained optimization without deadband mapping also fails. The negative space is any path that hits obstacles.
- **P1 (Find safe channels):** In a maze, safe channels are the open corridors. The script must first map all walkable cells before planning.
- **P2 (Optimize within channels):** Once safe cells are identified, use A* search to find the shortest path.

**Output:** The script below implements the protocol. It generates a random maze, maps obstacles (P0), identifies walkable cells (P1), then finds the shortest path from start to goal (P2). Each step is visualized with ASCII.

```python
#!/usr/bin/env python3
"""
deadband_maze_demo.py
Demonstrates the Deadband Protocol (P0→P1→P2) in a maze navigation problem.
Tinker - Cycle 22 - Cocapn Fleet
"""

import random
from collections import deque

def generate_maze(width=20, height=20, obstacle_density=0.3):
    """Generate a random maze with obstacles."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    for y in range(height):
        for x in range(width):
            if random.random() < obstacle_density:
                maze[y][x] = '#'
    # Ensure start (0,0) and goal (width-1, height-1) are clear
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    return maze

def print_maze(maze, title="Maze"):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))

def p0_map_negative_space(maze):
    """P0: Identify obstacles (rocks). Return set of (x,y) that are blocked."""
    obstacles = set()
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                obstacles.add((x, y))
    return obstacles

def p1_find_safe_channels(maze, obstacles):
    """P1: Find all walkable cells (safe channels) via BFS from start."""
    height, width = len(maze), len(maze[0])
    start = (0, 0)
    safe = set()
    queue = deque([start])
    visited = set([start])
    
    while queue:
        x, y = queue.popleft()
        safe.add((x, y))
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and (nx, ny) not in obstacles:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    return safe

def p2_optimize_within_channel(maze, safe, start, goal):
    """P2: Find shortest path from start to goal using A* (constrained to safe cells)."""
    height, width = len(maze), len(maze[0])
    
    # Heuristic: Manhattan distance
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    from queue import PriorityQueue
    open_set = PriorityQueue()
    open_set.put((0, start))
    came_from = {}
    g_score = {start: 0}
    
    while not open_set.empty():
        _, current = open_set.get()
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            path.reverse()
            return path
        
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            neighbor = (x + dx, y + dy)
            nx, ny = neighbor
            if 0 <= nx < width and 0 <= ny < height:
                if neighbor not in safe:
                    continue  # Not in safe channel → skip
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score = tentative_g + heuristic(neighbor, goal)
                    open_set.put((f_score, neighbor))
    return None  # No path found

def visualize_steps(maze, obstacles, safe, path):
    """Create visualizations for each protocol step."""
    height, width = len(maze), len(maze[0])
    
    # Step P0 visualization: show obstacles
    p0_viz = [row[:] for row in maze]
    for x, y in obstacles:
        p0_viz[y][x] = '#'  # Already marked, but ensure
    print_maze(p0_viz, "P0 - Negative Space (Obstacles)")
    print(f"Found {len(obstacles)} rocks.")
    
    # Step P1 visualization: show safe channels
    p1_viz = [row[:] for row in maze]
    for x, y in safe:
        if p1_viz[y][x] not in ['S', 'G']:
            p1_viz[y][x] = ' '
    print_maze(p1_viz, "P1 - Safe Channels (Walkable)")
    print(f"Found {len(safe)} safe cells.")
    
    # Step P2 visualization: show optimal path
    if path:
        p2_viz = [row[:] for row in maze]
        for x, y in path[1:-1]:  # Exclude start and goal
            p2_viz[y][x] = '*'
        print_maze(p2_viz, "P2 - Optimal Path (Within Safe Channels)")
        print(f"Path length: {len(path)-1} steps")
    else:
        print("No path found within safe channels.")

def main():
    print("Deadband Protocol Maze Demo")
    print("=" * 40)
    
    # Generate maze
    maze = generate_maze(width=15, height=10, obstacle_density=0.25)
    print_maze(maze, "Initial Maze")
    
    # P0: Map negative space
    obstacles = p0_map_negative_space(maze)
    
    # P1: Find safe channels
    safe = p1_find_safe_channels(maze, obstacles)
    
    # P2: Optimize within channels
    start = (0, 0)
    goal = (len(maze[0])-1, len(maze)-1)
    path = p2_optimize_within_channel(maze, safe, start, goal)
    
    # Visualize each step
    visualize_steps(maze, obstacles, safe, path)
    
    # Summary
    print("\n=== Protocol Summary ===")
    print("P0: Mapped", len(obstacles), "obstacles (rocks to avoid)")
    print("P1: Found", len(safe), "safe cells (channels)")
    print("P2:", "Found optimal path" if path else "No path found", "within safe channels")
    
    # Demonstrate violation: what if we skip P0+P1?
    print("\n=== Violation Demo (Skip P0+P1) ===")
    print("If we run P2 (A*) without P0+P1, the agent might:")
    print("- Try to move through walls (impossible)")
    print("- Get stuck in dead ends")
    print("- Fail to reach goal (as shown in constraint_sim_final.json)")

if __name__ == "__main__":
    main()
```

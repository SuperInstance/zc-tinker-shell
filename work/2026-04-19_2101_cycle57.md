**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Implementing based on fleet knowledge. Using the simulation results from `constraint_sim_final.json` and the formal rule from `DEADBAND-PROTOCOL.md`. Building a concrete, runnable example.

**Approach:**
1. **P0 (Map negative space):** Generate a 20x20 maze with 64 rocks (obstacles) placed randomly, matching the simulation data.
2. **P1 (Identify safe channels):** Use BFS to find all reachable cells from the start that avoid rocks — these are the deadbands.
3. **P2 (Optimize within channel):** Find the shortest path from start to goal using only safe cells.
4. **Visualization:** Print ASCII maze showing rocks, safe channels, and the optimal path.

**Output:** A single Python file `deadband_maze_demo.py` that runs the protocol and prints results.

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Based on fleet-knowledge simulation: 20x20 maze, 64 rocks, optimal path length 21.
"""

import random
from collections import deque

# Constants from fleet-knowledge
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LEN = 21

START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

def generate_maze():
    """P0: Map negative space — place rocks randomly."""
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        x = random.randint(0, MAZE_SIZE - 1)
        y = random.randint(0, MAZE_SIZE - 1)
        if (x, y) != START and (x, y) != GOAL:
            rocks.add((x, y))
    return rocks

def find_safe_channels(rocks):
    """P1: Identify safe channels — all cells reachable from start without hitting rocks."""
    safe = set()
    queue = deque([START])
    visited = set([START])
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        x, y = queue.popleft()
        safe.add((x, y))
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    return safe

def optimize_within_channel(safe, rocks):
    """P2: Optimize within channel — BFS shortest path from start to goal using only safe cells."""
    if GOAL not in safe:
        return None  # No safe channel to goal
    
    queue = deque([(START, [START])])
    visited = set([START])
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        (x, y), path = queue.popleft()
        if (x, y) == GOAL:
            return path
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) in safe and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    return None

def visualize(maze_id, rocks, safe, path):
    """ASCII visualization of the maze."""
    print(f"\n=== Maze {maze_id} ===")
    for y in range(MAZE_SIZE):
        row = []
        for x in range(MAZE_SIZE):
            if (x, y) == START:
                row.append('S')
            elif (x, y) == GOAL:
                row.append('G')
            elif (x, y) in rocks:
                row.append('#')  # Rock (P0 negative space)
            elif (x, y) in path:
                row.append('*')  # Optimal path (P2)
            elif (x, y) in safe:
                row.append('.')  # Safe channel (P1)
            else:
                row.append(' ')  # Unreachable/unsafe
        print(''.join(row))
    print()

def run_protocol():
    """Execute full Deadband Protocol P0→P1→P2."""
    print("Deadband Protocol Maze Demo")
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}, Optimal: {OPTIMAL_PATH_LEN} steps")
    print("-" * 50)
    
    # P0: Map negative space
    rocks = generate_maze()
    print(f"P0: Mapped {len(rocks)} rocks (negative space)")
    
    # P1: Identify safe channels
    safe = find_safe_channels(rocks)
    print(f"P1: Found {len(safe)} safe cells (channels)")
    
    # P2: Optimize within channel
    path = optimize_within_channel(safe, rocks)
    if path:
        print(f"P2: Optimal path length = {len(path)-1} steps")
        if len(path)-1 == OPTIMAL_PATH_LEN:
            print("   ✓ Matches fleet optimal (21)")
        else:
            print(f"   ⚠ Fleet optimal was {OPTIMAL_PATH_LEN}")
    else:
        print("P2: No safe path to goal (channel blocked)")
    
    # Visualization
    visualize(1, rocks, safe, path if path else [])
    
    # Statistics
    reachable_goal = GOAL in safe
    print("Summary:")
    print(f"  Goal reachable: {reachable_goal}")
    print(f"  Safe coverage: {len(safe)}/{MAZE_SIZE*MAZE_SIZE} cells")
    print(f"  Rocks in safe zone: {len(rocks & safe)}")
    
    return rocks, safe, path

if __name__ == "__main__":
    random.seed(42)  # Deterministic for demo
    rocks, safe, path = run_protocol()
```

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Simulates P0→P1→P2 vs. P2-only vs. random in a 20×20 maze.
Based on fleet-knowledge/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set
from enum import Enum

class Cell(Enum):
    EMPTY = '·'
    ROCK = '█'
    AGENT = 'A'
    GOAL = 'G'
    PATH = '○'

class Strategy(Enum):
    RANDOM = "random"
    UNCONSTRAINED = "unconstrained (P2-only)"
    DEADBAND = "deadband (P0→P1→P2)"

class Maze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.grid = [[Cell.EMPTY for _ in range(width)] for _ in range(height)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Place rocks (P0 negative space)
        for y in range(height):
            for x in range(width):
                if (x, y) != self.start and (x, y) != self.goal:
                    if random.random() < rock_density:
                        self.grid[y][x] = Cell.ROCK
                        self.rocks.add((x, y))
        
        # Ensure start and goal are clear
        self.grid[self.start[1]][self.start[0]] = Cell.EMPTY
        self.grid[self.goal[1]][self.goal[0]] = Cell.GOAL
    
    def is_rock(self, x, y):
        return (x, y) in self.rocks
    
    def in_bounds(self, x, y):
        return 0 <= x < self.width and 0 <= y < self.height
    
    def get_neighbors(self, x, y):
        dirs = [(0, -1), (1, 0), (0, 1), (-1, 0)]  # N, E, S, W
        neighbors = []
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if self.in_bounds(nx, ny):
                neighbors.append((nx, ny))
        return neighbors
    
    def render(self, agent_pos=None, path=None):
        grid_copy = [row.copy() for row in self.grid]
        if path:
            for (x, y) in path:
                if (x, y) != self.start and (x, y) != self.goal:
                    grid_copy[y][x] = Cell.PATH
        if agent_pos:
            x, y = agent_pos
            grid_copy[y][x] = Cell.AGENT
        
        lines = []
        for y in range(self.height):
            line = ''.join(grid_copy[y][x].value for x in range(self.width))
            lines.append(line)
        return '\n'.join(lines)

class Agent:
    def __init__(self, maze: Maze):
        self.maze = maze
        self.pos = maze.start
        self.path = [maze.start]
        self.steps = 0
        self.max_steps = 1000
    
    def move_to(self, x, y):
        if self.maze.in_bounds(x, y) and not self.maze.is_rock(x, y):
            self.pos = (x, y)
            self.path.append(self.pos)
            self.steps += 1
            return True
        return False
    
    def at_goal(self):
        return self.pos == self.maze.goal

def random_strategy(agent: Agent) -> bool:
    """Random walk (baseline)"""
    neighbors = agent.maze.get_neighbors(*agent.pos)
    valid = [n for n in neighbors if not agent.maze.is_rock(*n)]
    if not valid:
        return False
    next_pos = random.choice(valid)
    return agent.move_to(*next_pos)

def unconstrained_strategy(agent: Agent) -> bool:
    """P2-only: greedy Manhattan distance to goal"""
    x, y = agent.pos
    gx, gy = agent.maze.goal
    
    # Generate candidate moves (N,E,S,W)
    candidates = [(x, y-1), (x+1, y), (x, y+1), (x-1, y)]
    # Filter in-bounds and non-rock
    candidates = [c for c in candidates if agent.maze.in_bounds(*c) 
                  and not agent.maze.is_rock(*c)]
    if not candidates:
        return False
    
    # Greedy: pick move that minimizes Manhattan distance to goal
    best = min(candidates, key=lambda c: abs(c[0]-gx) + abs(c[1]-gy))
    return agent.move_to(*best)

def deadband_strategy(agent: Agent) -> bool:
    """P0→P1→P2: map rocks, find safe channels, then optimize"""
    x, y = agent.pos
    gx, gy = agent.maze.goal
    
    # P0: Rocks are already known (maze.rocks)
    # P1: Identify safe channels among neighbors
    neighbors = agent.maze.get_neighbors(x, y)
    safe = [n for n in neighbors if not agent.maze.is_rock(*n)]
    if not safe:
        return False
    
    # P2: Optimize within safe channels (min Manhattan to goal)
    best = min(safe, key=lambda c: abs(c[0]-gx) + abs(c[1]-gy))
    return agent.move_to(*best)

def run_simulation(maze: Maze, strategy_func, strategy_name: str, max_runs=50):
    """Run one strategy multiple times, return success rate and avg steps."""
    successes = 0
    total_steps = 0
    
    for run in range(max_runs):
        agent = Agent(maze)
        while not agent.at_goal() and agent.steps < agent.max_steps:
            if not strategy_func(agent):
                break  # Stuck
        if agent.at_goal():
            successes += 1
            total_steps += agent.steps
    
    avg_steps = total_steps / successes if successes > 0 else 0
    return successes, avg_steps

## Creating the Deadband Protocol Visual Demo

Based on the constraint simulation data in `fleet-knowledge/narrow-games/constraint_sim_final.json` and the Deadband Protocol documentation, I'll create a Python script that visually demonstrates the three-phase approach.

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo
Demonstrates P0→P1→P2 approach in a simple maze navigation problem.

Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Optimal path length: 21, Rocks: 64
"""

import random
from typing import List, Tuple, Set
from collections import deque

class DeadbandMaze:
    """Visual demonstration of Deadband Protocol in maze navigation."""
    
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (P0: negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        total_cells = width * height
        num_rocks = int(total_cells * rock_density)
        
        # Avoid start and goal positions
        possible_positions = [(x, y) for x in range(width) for y in range(height)]
        possible_positions.remove(self.start)
        possible_positions.remove(self.goal)
        
        self.rocks = set(random.sample(possible_positions, min(num_rocks, len(possible_positions))))
        
        # Statistics
        self.p0_mapped = False
        self.p1_channels = []
        self.p2_path = []
        
    def display_maze(self, path: List[Tuple[int, int]] = None, 
                     show_rocks: bool = True, show_channels: bool = False):
        """Display the maze with ASCII art."""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*60)
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif path and pos in path:
                    row.append('·')
                elif show_rocks and pos in self.rocks:
                    row.append('█')
                elif show_channels and hasattr(self, 'safe_channels') and pos in self.safe_channels:
                    row.append('░')
                else:
                    row.append(' ')
            print(' '.join(row))
    
    def p0_map_negative_space(self) -> None:
        """P0: Identify all rocks (what NOT to do)."""
        print("\n[P0] MAPPING NEGATIVE SPACE")
        print(f"Identified {len(self.rocks)} rocks (failure states)")
        self.display_maze(show_rocks=True)
        self.p0_mapped = True
    
    def p1_find_safe_channels(self) -> bool:
        """P1: Find safe channels (where you CAN be)."""
        if not self.p0_mapped:
            print("ERROR: Must complete P0 before P1")
            return False
            
        print("\n[P1] FINDING SAFE CHANNELS")
        
        # BFS to find all reachable safe positions
        visited = set()
        queue = deque([self.start])
        self.safe_channels = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            
            # Skip rocks
            if (x, y) in self.rocks:
                continue
                
            self.safe_channels.add((x, y))
            
            # Explore neighbors
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        print(f"Found {len(self.safe_channels)} safe positions")
        self.display_maze(show_channels=True)
        return self.goal in self.safe_channels
    
    def p2_optimize_within_channel(self) -> bool:
        """P2: Find optimal path within safe channels."""
        if not hasattr(self, 'safe_channels'):
            print("ERROR: Must complete P1 before P2")
            return False
            
        print("\n[P2] OPTIMIZING WITHIN SAFE CHANNELS")
        
        # A* search within safe channels only
        open_set = {self.start}
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: self._heuristic(self.start, self.goal)}
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score.get(pos, float('inf')))
            
            if current == self.goal:
                # Reconstruct path
                self.p2_path = []
                while current in came_from:
                    self.p2_path.append(current)
                    current = came_from[current]
                self.p2_path.reverse()
                self.p2_path = [self.start] + self.p2_path
                
                print(f"Found optimal path of length {len(self.p2_path)-1}")
                self.display_maze(path=self.p2_path, show_rocks=True)
                return True
            
            open_set.remove(current)
            
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # Constrained to safe channels (P1 result)
                if neighbor not in self.safe_channels:
                    continue
                    
                tentative_g = g_score[current] + 1
                
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor, self.goal)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        print("No path found within safe channels")
        return False
    
    def _heuristic(self, a: Tuple[int, int], b: Tuple[int, int]) -> int:
        """Manhattan distance heuristic."""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def run_greedy_unconstrained(self) -> bool:
        """Demonstrate failure of P2-only (unconstrained optimization)."""
        print("\n" + "="*60)
        print("GREEDY UNCONSTRAINED OPTIMIZATION (P2-ONLY)")
        print("="*60)
        
        # Greedy approach ignoring rocks
        path = [self.start]
        current = self.start
        visited = {self.start}
        
        steps = 0
        max_steps = 100
        
        while current != self.goal and steps < max_steps:
            # Move toward goal greedily
            x, y = current
            gx, gy = self.goal
            
            # Choose move that reduces distance most
            moves = []
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    dist = abs(nx - gx) + abs(ny - gy)
                    moves.append(((nx, ny), dist))
            
            if not moves:
                break
                
            # Greedy choice (ignores rocks!)
            next_pos, _ = min(moves, key=lambda m: m[1])
            
            if next_pos in self.rocks:
                print(f"  Hit rock at {next_pos} after {steps} steps!")
                return False
                
            current = next_pos
            path.append(current)
            steps += 1

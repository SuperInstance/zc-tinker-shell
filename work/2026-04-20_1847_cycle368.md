#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set

# ========== CONFIGURATION ==========
MAZE_WIDTH = 20
MAZE_HEIGHT = 20
START = (0, 0)
GOAL = (19, 19)
ROCK_COUNT = 64  # From simulation data
OPTIMAL_PATH_LENGTH = 21  # From simulation data

# Symbols for visualization
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_MARKER = 'G'
PATH = '○'
SAFE_CHANNEL = '░'  # Visual indicator for safe channels (P1)

# ========== MAZE GENERATION ==========
def generate_maze(width: int, height: int, rock_count: int) -> List[List[str]]:
    """Create a maze grid with random rocks."""
    grid = [[EMPTY for _ in range(width)] for _ in range(height)]
    rocks_placed = 0
    while rocks_placed < rock_count:
        x = random.randint(0, width - 1)
        y = random.randint(0, height - 1)
        if (x, y) == START or (x, y) == GOAL:
            continue
        if grid[y][x] == EMPTY:
            grid[y][x] = ROCK
            rocks_placed += 1
    grid[START[1]][START[0]] = AGENT
    grid[GOAL[1]][GOAL[0]] = GOAL_MARKER
    return grid

def print_maze(grid: List[List[str]], title: str):
    """Print the maze with a title."""
    print(f"\n=== {title} ===")
    for row in grid:
        print(' '.join(row))
    print()

# ========== DEADBAND PROTOCOL IMPLEMENTATION ==========
def p0_map_rocks(grid: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Identify negative space (rocks). Return set of rock coordinates."""
    rocks = set()
    for y in range(len(grid)):
        for x in range(len(grid[0])):
            if grid[y][x] == ROCK:
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(grid: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[str]]:
    """
    P1: Map safe channels (deadbands). 
    Returns a copy of the grid with safe channels marked.
    Uses simple flood-fill from start, avoiding rocks.
    """
    width, height = len(grid[0]), len(grid)
    safe_grid = [row[:] for row in grid]  # Copy
    
    # Flood-fill to find reachable empty cells
    stack = [START]
    visited = set()
    while stack:
        x, y = stack.pop()
        if (x, y) in visited:
            continue
        visited.add((x, y))
        
        # Mark as safe channel if it's empty
        if safe_grid[y][x] == EMPTY:
            safe_grid[y][x] = SAFE_CHANNEL
        
        # Explore neighbors (4-directional)
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    stack.append((nx, ny))
    return safe_grid

def p2_optimize_path(grid: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
    """
    P2: Find shortest path within safe channels (A* search).
    Only moves through empty cells or goal.
    """
    from heapq import heappush, heappop
    
    width, height = len(grid[0]), len(grid)
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heappush(open_set, (0, START))
    came_from = {}
    g_score = {START: 0}
    f_score = {START: heuristic(START, GOAL)}
    
    while open_set:
        _, current = heappop(open_set)
        
        if current == GOAL:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            return path
        
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            neighbor = (nx, ny)
            if 0 <= nx < width and 0 <= ny < height:
                # Check if neighbor is traversable (not a rock)
                if grid[ny][nx] == ROCK:
                    continue
                
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, GOAL)
                    heappush(open_set, (f_score[neighbor], neighbor))
    
    return []  # No path found

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 358)
A visual ASCII simulation of P0→P1→P2 in a constrained navigation problem.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time
import sys

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.rock_density = rock_density
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self.agent_pos = self.start
        self.path = []
        self.generate_maze()
        
    def generate_maze(self):
        """P0: Map the negative space (rocks)."""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * self.rock_density)
        
        # Ensure start and goal are not rocks
        possible = [(x, y) for x in range(self.width) for y in range(self.height)]
        possible.remove(self.start)
        possible.remove(self.goal)
        
        self.rocks = set(random.sample(possible, num_rocks))
        
    def is_safe(self, x, y):
        """Check if cell is not a rock and within bounds."""
        return (0 <= x < self.width and 
                0 <= y < self.height and 
                (x, y) not in self.rocks)
    
    def find_safe_channels(self):
        """P1: Identify all safe channels using BFS."""
        from collections import deque
        
        visited = set()
        queue = deque([self.start])
        parent = {self.start: None}
        safe_cells = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            safe_cells.add((x, y))
            
            # Stop if we reach goal (for channel mapping)
            if (x, y) == self.goal:
                continue
                
            # Explore neighbors (4-directional)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if self.is_safe(nx, ny) and (nx, ny) not in visited:
                    queue.append((nx, ny))
                    parent[(nx, ny)] = (x, y)
        
        # Reconstruct path if goal is reachable
        if self.goal in visited:
            path = []
            current = self.goal
            while current is not None:
                path.append(current)
                current = parent[current]
            return safe_cells, list(reversed(path))
        return safe_cells, []
    
    def greedy_unconstrained(self):
        """P2-only approach (violates protocol)."""
        x, y = self.start
        path = [(x, y)]
        visited = set([(x, y)])
        
        while (x, y) != self.goal:
            # Greedy: move toward goal
            moves = []
            if x < self.goal[0] and self.is_safe(x+1, y):
                moves.append((x+1, y))
            elif x > self.goal[0] and self.is_safe(x-1, y):
                moves.append((x-1, y))
            elif y < self.goal[1] and self.is_safe(x, y+1):
                moves.append((x, y+1))
            elif y > self.goal[1] and self.is_safe(x, y-1):
                moves.append((x, y-1))
            
            # If greedy fails, try any safe move
            if not moves:
                for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                    nx, ny = x + dx, y + dy
                    if self.is_safe(nx, ny) and (nx, ny) not in visited:
                        moves.append((nx, ny))
            
            if not moves:
                return path, False  # Trapped
            
            # Choose first available move
            x, y = moves[0]
            path.append((x, y))
            visited.add((x, y))
            
        return path, True
    
    def deadband_protocol(self):
        """P0→P1→P2: Full protocol implementation."""
        # P0 already done in generate_maze()
        
        # P1: Find safe channels
        safe_cells, optimal_path = self.find_safe_channels()
        
        if not optimal_path:
            return [], False  # No safe channel to goal
        
        # P2: Follow optimal path within safe channel
        return optimal_path, True
    
    def render(self, path=None, highlight=None):
        """Render ASCII maze with current state."""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif (x, y) in self.rocks:
                    row.append('█')
                elif path and (x, y) in path:
                    row.append('·')
                elif highlight and (x, y) in highlight:
                    row.append('░')
                else:
                    row.append(' ')
            grid.append(''.join(row))
        return '\n'.join(grid)

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 358)
A visual ASCII simulation of P0→P1→P2 in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import random
import time
from typing import List, Tuple, Set

# Simulation constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # 64 rocks in the original simulation
START = (0, 0)
GOAL = (19, 19)

class DeadbandAgent:
    """An agent that follows the Deadband Protocol strictly."""
    
    def __init__(self):
        self.rocks: Set[Tuple[int, int]] = set()  # P0: Negative space
        self.safe_channels: Set[Tuple[int, int]] = set()  # P1: Safe cells
        self.path: List[Tuple[int, int]] = []  # P2: Optimal path
        
    def p0_map_negative_space(self, maze: List[List[str]]) -> None:
        """P0: Identify all rocks (failure states)."""
        for y in range(MAZE_SIZE):
            for x in range(MAZE_SIZE):
                if maze[y][x] == '█':  # Rock symbol
                    self.rocks.add((x, y))
        print(f"P0 Complete: Mapped {len(self.rocks)} rocks")
    
    def p1_find_safe_channels(self, maze: List[List[str]]) -> bool:
        """P1: Find all reachable safe cells using flood fill."""
        visited = set()
        stack = [START]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            if (x, y) in self.rocks:
                continue
            if not (0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE):
                continue
                
            visited.add((x, y))
            # Add orthogonal neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                stack.append((nx, ny))
        
        self.safe_channels = visited
        goal_reachable = GOAL in self.safe_channels
        print(f"P1 Complete: Found {len(self.safe_channels)} safe cells")
        print(f"  Goal reachable: {goal_reachable}")
        return goal_reachable
    
    def p2_optimize_within_channel(self) -> bool:
        """P2: Find shortest path through safe channels (A* simplified)."""
        if GOAL not in self.safe_channels:
            return False
            
        # Simple BFS since we only need demonstration
        queue = [(START, [START])]
        visited = set([START])
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == GOAL:
                self.path = path
                print(f"P2 Complete: Found path of length {len(path)}")
                return True
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                pos = (nx, ny)
                
                if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and
                    pos in self.safe_channels and pos not in visited):
                    visited.add(pos)
                    queue.append((pos, path + [pos]))
        
        return False
    
    def run_protocol(self, maze: List[List[str]]) -> bool:
        """Execute P0→P1→P2 in strict order."""
        print("\n" + "="*50)
        print("DEADBAND PROTOCOL EXECUTION")
        print("="*50)
        
        # P0: Map negative space
        self.p0_map_negative_space(maze)
        time.sleep(0.5)
        
        # P1: Find safe channels
        if not self.p1_find_safe_channels(maze):
            print("FAIL: Goal not reachable via safe channels")
            return False
        time.sleep(0.5)
        
        # P2: Optimize within channel
        if not self.p2_optimize_within_channel():
            print("FAIL: No path found within safe channels")
            return False
        
        print("\nSUCCESS: Protocol completed")
        return True

                x += 1
            elif x > GOAL[0]:
                x -= 1
            elif y < GOAL[1]:
                y += 1
            elif y > GOAL[1]:
                y -= 1
            
            pos = (x, y)
            path.append(pos)
            
            # Check for rock
            if maze[y][x] == '█':
                print(f"FAIL: Hit rock at {pos} after {len(path)} steps")
                return False
        
        print(f"SUCCESS: Reached goal in {len(path)} steps (got lucky)")
        return True

def generate_maze() -> List[List[str]]:
    """Generate a 20x20 maze with rocks."""
    maze = [['·' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    
    # Place rocks randomly
    rocks_placed = 0
    while rocks_placed < ROCK_COUNT:
        x = random.randint(0, MAZE_SIZE-1)
        y = random.randint(0, MAZE_SIZE-1)
        if (x, y) != START and (x, y) != GOAL and maze[y][x] == '·':
            maze[y][x] = '█'
            rocks_placed += 1
    
    # Ensure at least one path exists (for demonstration)
    maze[0][1] = '·'
    maze[1][1] = '·'
    maze[1][0] = '·'
    
    return maze

def print_maze(maze: List[List[str]], agent_path: List[Tuple[int, int]] = None):
    """Print the maze with optional agent path."""
    for y in range(MAZE_SIZE):
        row = []
        for x in range(MAZE_SIZE):
            pos = (x, y)
            if pos == START:
                row.append('S')
            elif pos == GOAL:
                row.append('G')
            elif agent_path and pos in agent_path:
                row.append('●')
            else:
                row.append(maze[y][x])
        print(' '.join(row))

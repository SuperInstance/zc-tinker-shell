#!/usr/bin/env python3
"""
deadband_maze_demo.py

A visual demonstration of the Deadband Protocol (P0→P1→P2) in a maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set
from enum import Enum

class Cell(Enum):
    EMPTY = ' '
    ROCK = '█'
    START = 'S'
    GOAL = 'G'
    PATH = '·'
    DEADBAND = '░'
    VISITED = 'x'

class DeadbandProtocol:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.grid = [[Cell.EMPTY for _ in range(width)] for _ in range(height)]
        self.start = (1, 1)
        self.goal = (width-2, height-2)
        self.rocks: Set[Tuple[int, int]] = set()
        
        # Initialize maze
        self.grid[self.start[1]][self.start[0]] = Cell.START
        self.grid[self.goal[1]][self.goal[0]] = Cell.GOAL
        
        # P0: Map negative space (rocks)
        self._place_rocks(rock_density)
        
    def _place_rocks(self, density):
        """P0: Identify rocks (negative space)"""
        total_cells = (self.width - 2) * (self.height - 2)  # Exclude borders
        num_rocks = int(total_cells * density)
        
        for _ in range(num_rocks):
            while True:
                x = random.randint(1, self.width-2)
                y = random.randint(1, self.height-2)
                if (x, y) != self.start and (x, y) != self.goal:
                    self.grid[y][x] = Cell.ROCK
                    self.rocks.add((x, y))
                    break
    
    def _is_safe(self, x, y, visited):
        """Check if cell is within safe channel"""
        if not (0 <= x < self.width and 0 <= y < self.height):
            return False
        if self.grid[y][x] == Cell.ROCK:
            return False
        if (x, y) in visited:
            return False
        return True
    
    def p1_find_safe_channels(self) -> List[Set[Tuple[int, int]]]:
        """P1: Find all safe channels using BFS from start"""
        channels = []
        visited_global = set()
        
        # Try multiple BFS to find different channels
        for attempt in range(5):
            queue = [(self.start, [self.start])]
            visited = set([self.start])
            found_channels = []
            
            while queue:
                (x, y), path = queue.pop(0)
                
                # If we reach goal, this is a complete channel
                if (x, y) == self.goal:
                    found_channels.append(set(path))
                    continue
                
                # Explore neighbors (up, down, left, right)
                for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                    nx, ny = x + dx, y + dy
                    if self._is_safe(nx, ny, visited):
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
            
            if found_channels:
                channels.extend(found_channels)
                visited_global.update(visited)
            
            # Early exit if we found channels
            if channels and len(channels) >= 3:
                break
        
        return channels
    
    def p2_optimize_in_channel(self, channel: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
        """P2: Find shortest path within a safe channel using A*"""
        # Simple BFS since we're constrained to channel cells
        queue = [(self.start, [self.start])]
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.goal:
                return path
            
            for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                nx, ny = x + dx, y + dy
                if (nx, ny) in channel and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        return []  # No path found in this channel
    
    def run_protocol(self):
        """Execute full P0→P1→P2 protocol"""
        print("=" * 60)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("=" * 60)
        
        # Display initial maze with rocks (P0)
        print("\nP0: MAP NEGATIVE SPACE (Rocks)")
        print(f"Rocks identified: {len(self.rocks)}")
        self.display_grid()
        
        # P1: Find safe channels
        print("\nP1: IDENTIFY SAFE CHANNELS")
        channels = self.p1_find_safe_channels()
        print(f"Found {len(channels)} safe channels")
        
        if not channels:
            print("ERROR: No safe channels found!")
            return
        
        # Display first channel
        channel_grid = [row[:] for row in self.grid]
        for x, y in list(channels[0])[:20]:  # Show first 20 cells
            if channel_grid[y][x] == Cell.EMPTY:
                channel_grid[y][x] = Cell.DEADBAND
        
        print("\nFirst safe channel (partial):")
        self.display_grid(channel_grid)
        
        # P2: Optimize within best channel
        print("\nP2: OPTIMIZE WITHIN SAFE CHANNEL")
        
        # Find shortest path in first channel
        best_path = self.p2_optimize_in_channel(channels[0])
        
        if not best_path:
            print("No path found in first channel, trying others...")
            for i, channel in enumerate(channels[1:], 1):
                best_path = self.p2_optimize_in_channel(channel)
                if best_path:
                    print(f"Found path in channel {i+1}")
                    break
        
        if best_path:
            # Display final path
            final_grid = [row[:] for row in self.grid]
            for x, y in best_path[1:-1]:  # Exclude start and goal
                if final_grid[y][x] == Cell.EMPTY:
                    final_grid[y][x] = Cell.PATH
            
            print(f"\nOptimal path found: {len(best_path)} steps")
            print(f"Optimal (from simulation): 21 steps")
            self.display_grid(final_grid)
            
            # Statistics
            print("\n" + "=" * 60)
            print("SIMULATION RESULTS vs FLEET DATA")
            print("-" * 60)
            print(f"Your path length:     {len(best_path)} steps")
            print(f"Fleet optimal:        21 steps")
            print(f"Rocks in maze:        {len(self.rocks)}")
            print(f"Fleet rocks:          64")
            print(f"Success rate:         {'YES' if best_path else 'NO'}")
            print(f"Fleet deadband success: 50/50 (100%)")
            print("=" * 60)
        else:
            print("ERROR: Could not find path in any channel!")
    
    def display_grid(self, grid=None):
        """Display the maze in ASCII"""
        if grid is None:
            grid = self.grid
        
        print("+" + "-" * self.width + "+")
        for y in range(self.height):
            row = "|"
            for x in range(self.width):
                cell = grid[y][x]
                row += cell.value
            row += "|"
            print(row)
        print("+" + "-" * self.width + "+")
        
        # Legend
        print("\nLegend: S=Start, G=Goal, █=Rock, ·=Path, ░=Deadband")

    """Run the demonstration"""
    print("Initializing Deadband Protocol demonstration...")

    # Create protocol instance

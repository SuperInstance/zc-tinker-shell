#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map rocks (obstacles)
P1: Find safe channels (deadbands)
P2: Optimize path within safe channels
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks based on simulation data (64 rocks in 20x20 grid)
        self.rocks: Set[Tuple[int, int]] = set()
        self._generate_rocks(rock_density)
        
        # State tracking
        self.visited = set()
        self.safe_channels = []
        self.optimal_path = []
    
    def _generate_rocks(self, density: float):
        """P0: Map the negative space (rocks)"""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        
        # Use deterministic seed for reproducibility
        random.seed(42)
        while len(self.rocks) < num_rocks:
            x = random.randint(0, self.width-1)
            y = random.randint(0, self.height-1)
            # Don't place rocks on start or goal
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def is_rock(self, x: int, y: int) -> bool:
        """Check if cell is a rock (failure state)"""
        return (x, y) in self.rocks
    
    def is_valid(self, x: int, y: int) -> bool:
        """Check if cell is within bounds and not a rock"""
        return (0 <= x < self.width and 
                0 <= y < self.height and 
                not self.is_rock(x, y))
    
    def find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Identify all safe channels (deadbands) from start to goal"""
        channels = []
        
        def dfs(x: int, y: int, path: List[Tuple[int, int]]):
            if (x, y) == self.goal:
                channels.append(path.copy())
                return
            
            if (x, y) in path:  # Avoid cycles
                return
            
            # Only move to valid (non-rock) cells
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if self.is_valid(nx, ny):
                    dfs(nx, ny, path + [(nx, ny)])
        
        dfs(self.start[0], self.start[1], [self.start])
        self.safe_channels = channels
        return channels
    
    def optimize_within_channels(self) -> List[Tuple[int, int]]:
        """P2: Find shortest path within safe channels"""
        if not self.safe_channels:
            self.find_safe_channels()
        
        if not self.safe_channels:
            return []  # No safe path exists
        
        # Find shortest path among safe channels
        shortest = min(self.safe_channels, key=len)
        self.optimal_path = shortest
        return shortest
    
    def run_protocol(self) -> dict:
        """Execute full Deadband Protocol P0→P1→P2"""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===\n")
        
        # P0: Map negative space
        print("P0: MAP NEGATIVE SPACE")
        print(f"  • Grid: {self.width}x{self.height}")
        print(f"  • Rocks: {len(self.rocks)} cells marked as failure states")
        print(f"  • Start: {self.start}, Goal: {self.goal}")
        
        # P1: Find safe channels
        print("\nP1: IDENTIFY SAFE CHANNELS")
        channels = self.find_safe_channels()
        print(f"  • Found {len(channels)} safe channels (deadbands)")
        if channels:
            print(f"  • Channel lengths: {[len(c) for c in channels[:3]]}...")
        
        # P2: Optimize within channels
        print("\nP2: OPTIMIZE WITHIN CHANNELS")
        if channels:
            optimal = self.optimize_within_channels()
            print(f"  • Optimal path length: {len(optimal)} steps")
            print(f"  • Path: {optimal[:3]}...{optimal[-3:] if len(optimal) > 6 else ''}")
        else:
            print("  • NO SAFE CHANNELS FOUND (protocol prevents failure)")
        
        return {
            "rocks_count": len(self.rocks),
            "safe_channels": len(channels),
            "optimal_length": len(self.optimal_path) if self.optimal_path else 0,
            "success": len(channels) > 0
        }
    
    def visualize(self, show_path: bool = True):
        """Create ASCII visualization of maze and path"""
        print("\n" + "="*50)
        print("MAZE VISUALIZATION")
        print("  . = empty cell")
        print("  # = rock (failure)")
        print("  S = start, G = goal")
        if show_path and self.optimal_path:
            print("  * = optimal path")
        print()
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif self.is_rock(x, y):
                    row.append('#')
                elif show_path and (x, y) in self.optimal_path:
                    row.append('*')
                else:
                    row.append('.')
            print(' '.join(row))
        
        # Show protocol violation example
        print("\n" + "="*50)
        print("PROTOCOL VIOLATION EXAMPLE")
        print("If we skip P0+P1 and go straight to P2 (greedy optimization):")
        
        # Simulate greedy agent that doesn't check for rocks
        greedy_path = []
        x, y = self.start
        steps = 0
        max_steps = self.width * self.height
        
        while (x, y) != self.goal and steps < max_steps:
            steps += 1
            greedy_path.append((x, y))
            
            # Greedy move toward goal (no rock checking)
            if x < self.goal[0]:
                x += 1
            elif y < self.goal[1]:
                y += 1
            elif x > self.goal[0]:
                x -= 1
            elif y > self.goal[1]:
                y -= 1
            
            # Check if hit rock
            if self.is_rock(x, y):
                print(f"  • Hit rock at ({x}, {y}) after {steps} steps")
                print(f"  • FAILURE: Greedy optimization without P0 map")
                break
        
        if (x, y) == self.goal:
            print(f"  • Reached goal in {steps} steps (lucky, but unreliable)")
        
        print(f"  • Deadband Protocol success rate: {len(self.safe_channels) > 0}")

def main():
    """Run the demonstration"""
    # Create maze matching simulation parameters
    maze = DeadbandMaze(width=20, height=20, rock_density=0.16)
    
    # Execute full protocol
    results = maze.run_protocol()
    
    # Visualize
    maze.visualize()
    
    # Compare with simulation data
    print("\n" + "="*50)
    print("COMPARISON WITH FLEET SIMULATION DATA")
    print("From constraint_sim_final.json:")
    print("  • Rocks: 64 (16% of 400 cells)")
    print("  • Unconstrained success: 33/50 (66%)")
    print("  • Deadband success: 50/50 (100%)")
    print("  • Deadband optimal path: 21 steps")
    
    print("\nThis demo shows:")
    print("  1. P0 prevents entering failure states")
    print("  2. P1 ensures we only consider safe paths")
    print("  3. P2 finds optimal path within constraints")
    print("\nViolating P0→P1→P2 order leads to failure as complexity increases.")

if __name__ == "__main__":
    main()

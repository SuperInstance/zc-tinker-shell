# ========== CONFIGURATION ==========
GRID_SIZE = 10

ROCK_DENSITY = 0.15  # ~15% rocks, matches simulation ratio

# ========== MAZE GENERATION ==========
def generate_maze() -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """Create grid with rocks, start, goal."""
    grid = [['.' for _ in range(GRID_SIZE)] for _ in range(GRID_SIZE)]
    rocks = set()
    
    # Place rocks
    for y in range(GRID_SIZE):
        for x in range(GRID_SIZE):
            if (x, y) == START or (x, y) == GOAL:
                continue
            if random.random() < ROCK_DENSITY:
                grid[y][x] = '#'
                rocks.add((x, y))
    
    grid[START[1]][START[0]] = 'S'
    grid[GOAL[1]][GOAL[0]] = 'G'
    return grid, rocks

def p1_find_safe_channels(grid, rocks) -> List[List[Tuple[int, int]]]:
    """P1: Find all paths that avoid rocks (BFS)."""
    print("\n=== P1: FIND SAFE CHANNELS ===")
    
    # Simple BFS to find safe paths
    queue = [(START, [START])]
    visited = {START}
    safe_paths = []
    
    while queue:
        (x, y), path = queue.pop(0)
        
        if (x, y) == GOAL:
            safe_paths.append(path)
            continue
        
        for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < GRID_SIZE and 0 <= ny < GRID_SIZE:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    print(f"Found {len(safe_paths)} safe channels (rock-free paths)")
    return safe_paths

    """P2: Select shortest path from safe channels."""

                line.append(grid[y][x])

"""
Deadband Protocol Maze Demo
Based on fleet-knowledge simulation data:
- 64 rocks (negative space)
- Optimal path length: 21
- Demonstrates P0→P1→P2 strict priority
"""

import json
from typing import List, Tuple, Set
from collections import deque

# Simulation parameters from constraint_sim_final.json
MAZE_SIZE = 20  # 20x20 maze
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21
START = (0, 0)
GOAL = (19, 19)

def generate_rocks() -> Set[Tuple[int, int]]:
    """P0: Map negative space (rocks) - deterministic from seed."""
    rocks = set()
    # Use a deterministic seed based on rock count
    # In real simulation, this would come from actual rock positions
    # For demo, generate consistent rock pattern
    seed = 42
    for i in range(ROCK_COUNT):
        x = (i * 13 + seed) % MAZE_SIZE
        y = (i * 7 + seed) % MAZE_SIZE
        # Ensure start and goal are not rocks
        if (x, y) != START and (x, y) != GOAL:
            rocks.add((x, y))
    return rocks

def find_safe_channels(rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
    """P1: Identify all safe paths from START to GOAL avoiding rocks."""
    # BFS to find all shortest safe paths
    queue = deque([(START, [START])])
    visited = {START: 0}
    shortest_paths = []
    shortest_length = None
    
    while queue:
        (x, y), path = queue.popleft()
        
        # Check if we found the goal
        if (x, y) == GOAL:
            if shortest_length is None:
                shortest_length = len(path)
            if len(path) == shortest_length:
                shortest_paths.append(path)
            continue
        
        # Generate valid moves (up, down, left, right)
        for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) not in rocks:
                    new_path = path + [(nx, ny)]
                    # Only explore if this is shortest or equal shortest path
                    if (nx, ny) not in visited or len(new_path) <= visited[(nx, ny)]:
                        visited[(nx, ny)] = len(new_path)
                        queue.append(((nx, ny), new_path))
    
    return shortest_paths

def optimize_within_channel(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """P2: Select optimal path from safe channels."""
    if not safe_paths:

    # For this demo, we'll select the first shortest path
    # In a real scenario, you might optimize based on additional criteria

    """Create ASCII visualization of the maze."""

        row += " │"
        print(row)
    print(" " + "─" * (MAZE_SIZE * 2 + 1))
    
    print("\nLEGEND:")

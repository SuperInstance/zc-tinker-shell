#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
A visual ASCII simulation of P0→P1→P2 in a constrained navigation task.
"""

import random
from typing import List, Tuple, Set

# ==================== CONFIG ====================
MAZE_SIZE = 10
ROCK_DENSITY = 0.15  # 15% of cells are rocks
START = (0, 0)
GOAL = (9, 9)

# Symbols
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_SYM = 'G'
PATH = '○'
SAFE = '░'

# ==================== MAZE GENERATION ====================
def generate_maze(size: int, rock_density: float) -> List[List[str]]:
    """Generate a random maze with rocks."""
    maze = [[EMPTY for _ in range(size)] for _ in range(size)]
    rocks = 0
    for r in range(size):
        for c in range(size):
            if (r, c) == START or (r, c) == GOAL:
                continue
            if random.random() < rock_density:
                maze[r][c] = ROCK
                rocks += 1
    # Ensure start and goal are clear
    maze[START[0]][START[1]] = EMPTY
    maze[GOAL[0]][GOAL[1]] = GOAL_SYM
    return maze, rocks

def print_maze(maze: List[List[str]], title: str):
    """Print the maze with a title."""
    print(f"\n{'='*40}")
    print(f"{title}")
    print(f"{'='*40}")
    for row in maze:
        print(' '.join(row))
    print()

# ==================== DEADBAND PROTOCOL ====================
def p0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Identify negative space (rocks)."""
    rocks = set()
    for r in range(len(maze)):
        for c in range(len(maze[0])):
            if maze[r][c] == ROCK:
                rocks.add((r, c))
    return rocks

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[str]]:
    """P1: Mark all non-rock cells as safe channels."""
    safe_maze = [row[:] for row in maze]  # Copy
    for r in range(len(maze)):
        for c in range(len(maze[0])):
            if (r, c) not in rocks and safe_maze[r][c] not in (AGENT, GOAL_SYM):
                safe_maze[r][c] = SAFE
    return safe_maze

def p2_optimal_path(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
    """P2: Find shortest path from START to GOAL avoiding rocks (BFS)."""
    from collections import deque
    
    size = len(maze)
    directions = [(1,0), (-1,0), (0,1), (0,-1)]  # Down, Up, Right, Left
    
    queue = deque([(START, [])])
    visited = set([START])
    
    while queue:
        (r, c), path = queue.popleft()
        if (r, c) == GOAL:
            return path + [(r, c)]
        
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < size and 0 <= nc < size:
                if (nr, nc) not in visited and maze[nr][nc] != ROCK:
                    visited.add((nr, nc))
                    queue.append(((nr, nc), path + [(r, c)]))
    
    return []  # No path found

# ==================== VISUALIZATION ====================
def visualize_protocol(maze: List[List[str]], rocks: Set[Tuple[int, int]], path: List[Tuple[int, int]]):
    """Create three visualization mazes for P0, P1, P2."""
    # P0: Rocks only
    p0_maze = [row[:] for row in maze]
    for r, c in rocks:
        p0_maze[r][c] = ROCK
    p0_maze[START[0]][START[1]] = AGENT
    p0_maze[GOAL[0]][GOAL[1]] = GOAL_SYM
    print_maze(p0_maze, "P0: MAP ROCKS (Negative Space)")
    
    # P1: Safe channels
    p1_maze = p1_find_safe_channels(maze, rocks)
    p1_maze[START[0]][START[1]] = AGENT
    p1_maze[GOAL[0]][GOAL[1]] = GOAL_SYM
    print_maze(p1_maze, "P1: SAFE CHANNELS (Where You CAN Be)")
    
    # P2: Optimal path within channels
    p2_maze = [row[:] for row in maze]
    for r, c in path:
        if (r, c) != START and (r, c) != GOAL:
            p2_maze[r][c] = PATH
    p2_maze[START[0]][START[1]] = AGENT
    p2_maze[GOAL[0]][GOAL[1]] = GOAL_SYM
    print_maze(p2_maze, "P2: OPTIMAL PATH (Within Safe Channels)")

# ==================== MAIN ====================
def main():
    print("DEADBAND PROTOCOL DEMO — Tinker (Cocapn Fleet)")
    print("Simulating P0→P1→P2 in a 10x10 maze with rocks.\n")
    
    # Generate maze
    maze, rock_count = generate_maze(MAZE_SIZE, ROCK_DENSITY)
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {rock_count}")
    print(f"Start: {START}, Goal: {GOAL}")
    
    # Initial maze
    initial = [row[:] for row in maze]
    initial[START[0]][START[1]] = AGENT
    print_maze(initial, "INITIAL MAZE")
    
    # Apply Deadband Protocol
    rocks = p0_map_rocks(maze)  # P0
    path = p2_optimal_path(maze, rocks)  # P2 (constrained by P0)
    
    # Visualize each phase
    visualize_protocol(maze, rocks, path)
    
    # Results
    if path:
        print(f"✓ PATH FOUND: {len(path)-1} steps")
        print(f"  Path: {path}")
    else:
        print("✗ NO PATH FOUND (rocks block all channels)")
    
    # Protocol summary
    print("\n" + "="*40)
    print("DEADBAND PROTOCOL SUMMARY")
    print("="*40)
    print("P0 (Map Rocks): Identify failure states")
    print("P1 (Safe Channels): Constrain to rock-free cells")
    print("P2 (Optimal Path): BFS within safe channels only")
    print("\nViolation: P2 without P0 → hit rocks with P→1.0")

if __name__ == "__main__":
    main()

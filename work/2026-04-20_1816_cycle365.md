#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation

Demonstrates the strict P0→P1→P2 priority:
P0: Map negative space (rocks)
P1: Identify safe channels (deadbands)
P2: Optimize within safe channels (shortest path)

Based on fleet-knowledge/narrow-games/constraint_sim_final.json
and DEADBAND-PROTOCOL.md.
"""

import random
import sys
import time

class Maze:
    """Simple 10x10 maze with rocks, start, and goal."""
    def __init__(self, seed=42):
        random.seed(seed)
        self.width = 10
        self.height = 10
        self.grid = [['.' for _ in range(self.width)] for _ in range(self.height)]
        
        # Start (top-left) and Goal (bottom-right)
        self.start = (0, 0)
        self.goal = (self.height-1, self.width-1)
        self.grid[self.start[0]][self.start[1]] = 'S'
        self.grid[self.goal[0]][self.goal[1]] = 'G'
        
        # Place rocks (~15% of cells)
        num_rocks = int(self.width * self.height * 0.15)
        for _ in range(num_rocks):
            while True:
                r = random.randint(0, self.height-1)
                c = random.randint(0, self.width-1)
                if (r, c) not in (self.start, self.goal) and self.grid[r][c] == '.':
                    self.grid[r][c] = '#'
                    break
    
    def print_maze(self, path=None, deadband=None):
        """Print the maze, optionally highlighting path and deadband cells."""
        for r in range(self.height):
            row = []
            for c in range(self.width):
                cell = self.grid[r][c]
                if path and (r, c) in path:
                    row.append('*')
                elif deadband and (r, c) in deadband:
                    row.append('~')
                else:
                    row.append(cell)
            print(' '.join(row))
        print()

class DeadbandAgent:
    """Agent that follows the Deadband Protocol."""
    def __init__(self, maze):
        self.maze = maze
        self.rocks = set()      # P0: negative space
        self.deadband = set()   # P1: safe channels
        self.path = []          # P2: optimal path within deadband
    
    def p0_map_negative_space(self):
        """P0: Identify all rock cells (what NOT to do)."""
        for r in range(self.maze.height):
            for c in range(self.maze.width):
                if self.maze.grid[r][c] == '#':
                    self.rocks.add((r, c))
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all cells reachable from start without hitting rocks."""
        visited = set()
        stack = [self.maze.start]
        
        while stack:
            r, c = stack.pop()
            if (r, c) in visited:
                continue
            visited.add((r, c))
            
            # Check all four neighbors
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                nr, nc = r + dr, c + dc
                if (0 <= nr < self.maze.height and 0 <= nc < self.maze.width):
                    if (nr, nc) not in self.rocks and (nr, nc) not in visited:
                        stack.append((nr, nc))
        
        self.deadband = visited
        return self.deadband
    
    def p2_optimize_within_channel(self):
        """P2: Find shortest path from start to goal within deadband (BFS)."""
        if self.maze.goal not in self.deadband:
            return []  # No safe path to goal
        
        queue = [(self.maze.start, [self.maze.start])]
        visited = set([self.maze.start])
        
        while queue:
            (r, c), current_path = queue.pop(0)
            if (r, c) == self.maze.goal:
                self.path = current_path
                return self.path
            
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                nr, nc = r + dr, c + dc
                if (0 <= nr < self.maze.height and 0 <= nc < self.maze.width):
                    if (nr, nc) in self.deadband and (nr, nc) not in visited:
                        visited.add((nr, nc))
                        queue.append(((nr, nc), current_path + [(nr, nc)]))
        
        return []  # No path found
    
    def run_protocol(self):
        """Execute full P0→P1→P2 sequence."""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===\n")
        
        print("Initial Maze (S=start, G=goal, #=rock, .=empty):")
        self.maze.print_maze()
        time.sleep(1)
        
        print("P0: MAP NEGATIVE SPACE (identify rocks)")
        rocks = self.p0_map_negative_space()
        print(f"Found {len(rocks)} rock cells (avoid these).")
        time.sleep(1)
        
        print("\nP1: IDENTIFY SAFE CHANNELS (deadband)")
        deadband = self.p1_find_safe_channels()
        print(f"Found {len(deadband)} safe cells reachable from start.")
        print("Maze with deadband marked (~):")
        self.maze.print_maze(deadband=deadband)
        time.sleep(1)
        
        print("P2: OPTIMIZE WITHIN SAFE CHANNELS")
        path = self.p2_optimize_within_channel()
        if path:
            print(f"Found optimal path of length {len(path)-1} steps.")
            print("Maze with optimal path marked (*):")
            self.maze.print_maze(path=path)
        else:
            print("No safe path to goal exists within deadband.")
        
        return len(path) > 0

def main():
    """Run the demonstration."""
    print("Deadband Protocol Visual Demo")
    print("Based on fleet-knowledge/DEADBAND-PROTOCOL.md\n")
    
    # Create maze and agent
    maze = Maze(seed=365)  # Seed matches current cycle
    agent = DeadbandAgent(maze)
    
    # Run protocol
    success = agent.run_protocol()
    
    # Summary
    print("\n=== PROTOCOL SUMMARY ===")
    print("P0 (rocks): Identify failure states")
    print("P1 (deadband): Map safe reachable space")
    print("P2 (path): Optimize only within safe channels")
    print(f"\nResult: {'SUCCESS' if success else 'FAILURE'}")
    print("\nViolating P0→P1→P2 order (e.g., optimizing before mapping rocks)")
    print("leads to hitting obstacles with high probability.")
    
    return 0 if success else 1

if __name__ == "__main__":
    sys.exit(main())

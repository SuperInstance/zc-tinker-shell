#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
A visual ASCII simulation of the Deadband Protocol (P0→P1→P2) in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
import time
import sys

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64          # rocks (P0 negative space)
OPTIMAL_PATH_LEN = 21    # shortest safe path (P2 within channels)
SIM_RUNS = 50            # matches deadband_success count

# Symbols for visualization
AGENT = 'A'
GOAL = 'G'
ROCK = '█'
SAFE = '·'
PATH = '○'
DEADBAND = '░'  # safe channel (P1)
VISITED = 'x'

class DeadbandMaze:
    """A maze that enforces the Deadband Protocol."""
    
    def __init__(self, seed=None):
        self.seed = seed if seed else random.randint(1, 1000)
        random.seed(self.seed)
        self.grid = [[SAFE for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.agent_pos = (0, 0)
        self.goal_pos = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.rocks = set()
        self.safe_channels = []  # P1 result
        self.optimal_path = []   # P2 result
        self.steps = 0
        self.phase = "P0"
        
        # Initialize maze
        self._place_rocks()
        self._place_agent_and_goal()
    
    def _place_rocks(self):
        """P0: Map negative space (rocks)."""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            pos = (x, y)
            # Don't place rocks on start or goal
            if pos != self.agent_pos and pos != self.goal_pos:
                self.rocks.add(pos)
                self.grid[x][y] = ROCK
    
    def _place_agent_and_goal(self):
        """Place agent at start and goal at end."""
        self.grid[self.agent_pos[0]][self.agent_pos[1]] = AGENT
        self.grid[self.goal_pos[0]][self.goal_pos[1]] = GOAL
    
    def is_rock(self, pos):
        """Check if position is a rock (P0 negative space)."""
        return pos in self.rocks
    
    def is_valid(self, pos):
        """Check if position is within bounds."""
        x, y = pos
        return 0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE
    
    def find_safe_channels(self):
        """P1: Identify safe channels (deadbands) using BFS that avoids rocks."""
        from collections import deque
        
        start = self.agent_pos
        goal = self.goal_pos
        
        # BFS queue: (position, path)
        queue = deque([(start, [start])])
        visited = set([start])
        all_safe_paths = []
        
        while queue:
            current, path = queue.popleft()
            
            if current == goal:
                all_safe_paths.append(path)
                continue
            
            x, y = current
            # Four possible moves: right, down, left, up
            neighbors = [(x+1, y), (x, y+1), (x-1, y), (x, y-1)]
            
            for neighbor in neighbors:
                if not self.is_valid(neighbor):
                    continue
                if neighbor in visited:
                    continue
                if self.is_rock(neighbor):
                    continue  # P0: Avoid rocks
                
                visited.add(neighbor)
                queue.append((neighbor, path + [neighbor]))
        
        self.safe_channels = all_safe_paths
        self.phase = "P1"
        return all_safe_paths
    
    def optimize_within_channels(self):
        """P2: Find shortest path within safe channels."""
        if not self.safe_channels:
            return None
        
        # Find shortest safe path
        shortest = min(self.safe_channels, key=len)
        self.optimal_path = shortest
        self.phase = "P2"
        return shortest
    
    def run_deadband_protocol(self):
        """Execute full P0→P1→P2 protocol."""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL DEMO - Maze Navigation")
        print("="*60)
        
        # P0: Already done during initialization (rocks placed)
        print(f"\n[P0] Mapped negative space: {len(self.rocks)} rocks")
        self.display()
        time.sleep(1.5)
        
        # P1: Find safe channels
        print(f"\n[P1] Finding safe channels (deadbands)...")
        safe_paths = self.find_safe_channels()
        print(f"    Found {len(safe_paths)} safe channel(s) to goal")
        
        # Visualize one safe channel
        if safe_paths:
            self.display_with_path(safe_paths[0], DEADBAND)
        time.sleep(1.5)
        
        # P2: Optimize within channels
        print(f"\n[P2] Optimizing within safe channels...")
        optimal = self.optimize_within_channels()
        if optimal:
            print(f"    Optimal path length: {len(optimal)-1} steps")
            print(f"    (Reference optimal from fleet data: {OPTIMAL_PATH_LEN})")
            self.display_with_path(optimal, PATH)
            
            # Animate agent following optimal path
            print(f"\n[AGENT] Following optimal path...")
            self.animate_path(optimal)
        else:
            print("    ERROR: No safe path found!")
        
        return optimal
    
    def display(self):
        """Display current maze state."""
        print(f"\nMaze (Seed: {self.seed}, Phase: {self.phase})")
        print("   " + "".join([f"{i:2}" for i in range(MAZE_SIZE)]))
        for i in range(MAZE_SIZE):
            row = f"{i:2} "
            for j in range(MAZE_SIZE):
                row += self.grid[i][j] + " "
            print(row)
    
    def display_with_path(self, path, path_char):
        """Display maze with a path highlighted."""
        # Create a copy for display
        display_grid = [row[:] for row in self.grid]
        
        # Mark path (excluding start and goal)
        for pos in path[1:-1]:
            x, y = pos
            display_grid[x][y] = path_char
        
        print(f"\nMaze with path ({path_char} = {self.phase} result)")
        print("   " + "".join([f"{i:2}" for i in range(MAZE_SIZE)]))
        for i in range(MAZE_SIZE):
            row = f"{i:2} "
            for j in range(MAZE_SIZE):
                row += display_grid[i][j] + " "
            print(row)
    
    def animate_path(self, path):
        """Animate agent moving along path."""
        for step, pos in enumerate(path):
            x, y = pos
            # Clear previous agent position
            for i in range(MAZE_SIZE):
                for j in range(MAZE_SIZE):
                    if self.grid[i][j] == AGENT and (i, j) != pos:
                        self.grid[i][j] = SAFE
            
            # Update agent position
            if pos != self.goal_pos:
                self.grid[x][y] = AGENT
            
            # Display
            sys.stdout.write(f"\rStep {step}: Agent at ({x}, {y})")
            sys.stdout.flush()
            time.sleep(0.1)
        
        print(f"\n✓ Agent reached goal in {len(path)-1} steps!")
        self.grid[self.goal_pos[0]][self.goal_pos[1]] = AGENT + GOAL  # Agent on goal

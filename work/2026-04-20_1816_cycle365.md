class Maze:
    """20x20 maze with rocks, start, and goal."""
    
    def __init__(self, seed=None):
        if seed is not None:
            random.seed(seed)
        self.width = 20
        self.height = 20
        self.grid = [['.' for _ in range(self.width)] for _ in range(self.height)]
        
        # Start at top-left, goal at bottom-right
        self.start = (0, 0)
        self.goal = (self.height-1, self.width-1)
        self.grid[self.start[0]][self.start[1]] = 'S'
        self.grid[self.goal[0]][self.goal[1]] = 'G'
        
        # Place rocks (P0 negative space)
        self.rocks = set()
        for _ in range(ROCK_COUNT):
            while True:
                r = random.randint(0, self.height-1)
                c = random.randint(0, self.width-1)
                if (r, c) not in [self.start, self.goal] and (r, c) not in self.rocks:
                    self.rocks.add((r, c))
                    self.grid[r][c] = '#'
                    break
    
    def print_maze(self, path=None, current=None):
        """Print ASCII maze with optional path and current position."""
        for r in range(self.height):
            row = []
            for c in range(self.width):
                pos = (r, c)
                if pos == current:
                    row.append('@')
                elif path and pos in path:
                    row.append('*')
                else:
                    row.append(self.grid[r][c])
            print(' '.join(row))
        print()

    def p1_find_safe_channels(self):
        """STEP 2: Identify safe channels (deadbands)."""
        if not self.p0_complete:
            print("VIOLATION: Attempted P1 without completing P0!")
            return False
            
        print("=== P1: IDENTIFY SAFE CHANNELS ===")
        print("Finding all paths where ∀ position ∉ negative space")
        
        # Simple flood fill from start, avoiding rocks
        frontier = [self.maze.start]
        self.safe_channels = set([self.maze.start])
        
        while frontier:
            r, c = frontier.pop()
            for dr, dc in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < self.maze.height and 0 <= nc < self.maze.width:
                    pos = (nr, nc)
                    if pos not in self.rocks and pos not in self.safe_channels:
                        self.safe_channels.add(pos)
                        frontier.append(pos)

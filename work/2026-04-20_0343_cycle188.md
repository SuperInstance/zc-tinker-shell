#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Visualizes P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge simulation results.
"""

import random
from typing import List, Tuple, Set
from collections import deque

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.15):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width - 1, height - 1)
        
        # Generate rocks (negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        for y in range(height):
            for x in range(width):
                if (x, y) != self.start and (x, y) != self.goal:
                    if random.random() < rock_density:
                        self.rocks.add((x, y))
        
        # Ensure start and goal are not rocks
        self.rocks.discard(self.start)
        self.rocks.discard(self.goal)
        
        # State
        self.p0_map = None
        self.p1_channels = None
        self.p2_path = None
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify all rocks (states where failure probability > threshold)."""
        self.p0_map = self.rocks.copy()
        return self.p0_map
    
    def p1_find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Find all paths from start to goal avoiding rocks."""
        if self.p0_map is None:
            raise ValueError("Must run P0 first")
        
        # BFS to find all safe paths
        queue = deque([(self.start, [self.start])])
        safe_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.p0_map and (nx, ny) not in path:
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.p1_channels = safe_paths
        return safe_paths
    
    def p2_optimize_within_channels(self) -> List[Tuple[int, int]]:
        """P2: Find shortest path among safe channels."""
        if self.p1_channels is None:
            raise ValueError("Must run P1 first")
        
        if not self.p1_channels:
            self.p2_path = []
            return []
        
        # Find shortest path (minimum cost)
        shortest = min(self.p1_channels, key=len)
        self.p2_path = shortest
        return shortest
    
    def run_protocol(self) -> Tuple[bool, int]:
        """Execute full P0→P1→P2 protocol."""
        self.p0_map_negative_space()
        self.p1_find_safe_channels()
        self.p2_optimize_within_channels()
        
        success = len(self.p2_path) > 0
        path_length = len(self.p2_path) - 1 if success else 0
        return success, path_length
    
    def render(self, show_p0: bool = False, show_p1: bool = False, show_p2: bool = False) -> str:
        """Render maze with optional protocol visualization."""
        grid = [['·' for _ in range(self.width)] for _ in range(self.height)]
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        grid[sy][sx] = 'S'
        grid[gy][gx] = 'G'
        
        # P0: Rocks
        if show_p0 and self.p0_map:
            for x, y in self.p0_map:
                grid[y][x] = '█'
        
        # P1: Safe channels (first channel only for clarity)
        if show_p1 and self.p1_channels and self.p1_channels[0]:
            for x, y in self.p1_channels[0]:
                if grid[y][x] not in ['S', 'G']:
                    grid[y][x] = '○'
        
        # P2: Optimal path
        if show_p2 and self.p2_path:
            for x, y in self.p2_path:
                if grid[y][x] not in ['S', 'G']:
                    grid[y][x] = '●'
        
        # Build output
        lines = []
        lines.append(f"Deadband Protocol Maze ({self.width}x{self.height})")
        lines.append(f"Rocks: {len(self.rocks)} | Start: S | Goal: G")
        
        if show_p0:
            lines.append("P0 (Rocks): █ = negative space (avoid)")
        if show_p1:
            lines.append("P1 (Channels): ○ = safe water (first channel)")
        if show_p2:
            lines.append("P2 (Optimal): ● = shortest safe path")
        
        lines.append("")
        lines.append("+" + "-" * self.width + "+")
        for row in grid:
            lines.append("|" + "".join(row) + "|")
        lines.append("+" + "-" * self.width + "+")
        
        return "\n".join(lines)

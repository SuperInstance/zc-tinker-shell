```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

Simulation results reference:
- Optimal path length: 21 steps
- Rocks (obstacles): 64
- Deadband protocol success: 50/50 runs
- Unconstrained greedy success: 33/50 runs
- Constrained (no P0) success: 0/50 runs
"""

import random
from typing import List, Tuple, Set
from enum import Enum

class Cell(Enum):
    EMPTY = " "
    ROCK = "█"
    AGENT = "A"
    GOAL = "G"
    PATH = "·"
    DEADBAND = "░"

class DeadbandProtocol:
    """Implements P0→P1→P2 protocol for maze navigation"""
    
    def __init__(self, width: int = 20, height: int = 20):
        self.width = width
        self.height = height
        self.grid = [[Cell.EMPTY for _ in range(width)] for _ in range(height)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.deadbands: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Initialize based on simulation data
        self._place_rocks(64)  # 64 rocks from simulation
        self.grid[self.start[1]][self.start[0]] = Cell.AGENT
        self.grid[self.goal[1]][self.goal[0]] = Cell.GOAL
    
    def _place_rocks(self, count: int):
        """P0: Map negative space (rocks)"""
        placed = 0
        while placed < count:
            x = random.randint(0, self.width-1)
            y = random.randint(0, self.height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[y][x] = Cell.ROCK
                placed += 1
    
    def find_deadbands(self):
        """P1: Identify safe channels (deadbands)"""
        # Simple flood fill from start, avoiding rocks
        visited = set()
        queue = [self.start]
        
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            self.deadbands.add((x, y))
            
            # Mark as deadband (safe channel) if not start/goal
            if (x, y) != self.start and (x, y) != self.goal:
                self.grid[y][x] = Cell.DEADBAND
            
            # Check neighbors
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) not in self.rocks and
                    (nx, ny) not in visited):
                    queue.append((nx, ny))
    
    def find_optimal_path(self) -> List[Tuple[int, int]]:
        """P2: Optimize within safe channels (A* search through deadbands)"""
        if not self.deadbands:
            self.find_deadbands()
        
        # A* search only through deadbands (safe channels)
        open_set = {self.start}
        came_from = {}
        
        g_score = {self.start: 0}
        f_score = {self.start: self._heuristic(self.start, self.goal)}
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score.get(pos, float('inf')))
            
            if current == self.goal:
                return self._reconstruct_path(came_from, current)
            
            open_set.remove(current)
            
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # P1 constraint: Only move through deadbands (or start/goal)
                if (neighbor not in self.deadbands and 
                    neighbor != self.start and 
                    neighbor != self.goal):
                    continue
                
                # P0 constraint: Never move through rocks
                if neighbor in self.rocks:
                    continue
                
                if not (0 <= neighbor[0] < self.width and 0 <= neighbor[1] < self.height):
                    continue
                
                tentative_g = g_score[current] + 1
                
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor, self.goal)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        return []  # No path found
    
    def _heuristic(self, a: Tuple[int, int], b: Tuple[int, int]) -> int:
        """Manhattan distance heuristic"""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def _reconstruct_path(self, came_from: dict, current: Tuple[int, int]) -> List[Tuple[int, int]]:
        """Reconstruct path from start to goal"""
        path = [current]
        while current in came_from:
            current = came_from[current]
            path.append(current)
        path.reverse()
        return path
    
    def display(self, path: List[Tuple[int, int]] = None):
        """Display the maze with protocol visualization"""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*60)
        
        print(f"\nMaze: {self.width}x{self.height}")
        print(f"Rocks (P0 negative space): {len(self.rocks)}")
        print(f"Safe cells (P1 deadbands): {len(self.deadbands)}")
        
        print("\nLEGEND:")
        print(f"  {Cell.AGENT.value} = Agent (start)")
        print(f"  {Cell.GOAL.value} = Goal")
        print(f"  {Cell.ROCK.value} = Rock (avoid!)")
        print(f"  {Cell.DEADBAND.value} = Safe channel (deadband)")
        print(f"  {Cell.PATH.value} = Optimal path (P2)")
        print(f"  {Cell.EMPTY.value} = Unreachable/unsafe")
        
        print("\nPROTOCOL STEPS:")
        print("P0 (RED): Identify rocks - what NOT to do")
        print("P1 (BLUE): Map safe channels - where you CAN be")
        print("P2 (GREEN): Find optimal path - best through safe channels")
        
        # Create display grid
        display_grid = [row.copy() for row in self.grid]
        
        # Mark path if provided
        if path:
            for x, y in path[1:-1]:  # Skip start and goal
                if display_grid[y][x] == Cell.DEADBAND:
                    display_grid[y][x] = Cell.PATH
        
        print("\n" + "MAZE:" + " "*(self.width-4) + "P0→P1→P2")
        print("┌" + "─"*self.width + "┐")
        
        for y in range(self.height):
            row_str = "│"
            for x in range(self.width):
                cell = display_grid[y][x]
                
                # Color coding for protocol phases
                if cell == Cell.ROCK:
                    row_str += f"\033[91m{cell.value}\033[0m"  # Red for P0
                elif cell == Cell.DEADBAND:
                    row_str += f"\033[94m{cell.value}\033[0m"  # Blue for P1
                elif cell == Cell.PATH:
                    row_str += f"\033[92m{cell.value}\033[0m"  # Green for P2
                elif cell == Cell.AGENT:
                    row_str += f"\033[93m{cell.value}\033[0m"  # Yellow for agent
                elif cell == Cell.GOAL:
                    row_str += f"\033[95m{cell.value}\033[0m"  # Magenta for goal
                else:
                    row_str += cell.value
            row_str += "

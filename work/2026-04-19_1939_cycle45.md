```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2 Visualized
Proof-of-concept based on fleet simulation data.
"""

import random
from typing import List, Tuple, Set

# ========== P0: MAP NEGATIVE SPACE (ROCKS) ==========
def map_rocks(maze: List[str]) -> Set[Tuple[int, int]]:
    """Identify all positions that are obstacles ('#')."""
    rocks = set()
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.add((x, y))
    return rocks

# ========== P1: IDENTIFY SAFE CHANNELS ==========
def find_safe_channels(
    start: Tuple[int, int],
    goal: Tuple[int, int],
    rocks: Set[Tuple[int, int]],
    bounds: Tuple[int, int]
) -> List[List[Tuple[int, int]]]:
    """
    BFS to find all paths from start to goal that avoid rocks.
    Returns list of safe paths (channels).
    """
    width, height = bounds
    queue = [(start, [start])]
    safe_paths = []
    
    while queue:
        (x, y), path = queue.pop(0)
        
        if (x, y) == goal:
            safe_paths.append(path)
            continue
        
        for dx, dy in [(0,1),(1,0),(0,-1),(-1,0)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in rocks and (nx, ny) not in path:
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    return safe_paths

# ========== P2: OPTIMIZE WITHIN CHANNEL ==========
def optimize_path(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """Select shortest safe path (minimum cost)."""
    if not safe_paths:
        return []
    return min(safe_paths, key=len)

# ========== VISUALIZATION ==========
def print_maze(
    maze: List[str],
    path: List[Tuple[int, int]] = None,
    rocks: Set[Tuple[int, int]] = None
):
    """Display maze with optional path and rock highlighting."""
    display = [list(row) for row in maze]
    
    if rocks:
        for (x, y) in rocks:
            if 0 <= y < len(maze) and 0 <= x < len(maze[0]):
                display[y][x] = '█'  # Rock
    
    if path:
        for (x, y) in path:
            if 0 <= y < len(maze) and 0 <= x < len(maze[0]):
                display[y][x] = '•'  # Path
    
    print("\n".join("".join(row) for row in display))

# ========== MAIN DEMONSTRATION ==========
def main():
    # Example maze (20x20) from fleet simulation
    maze = [
        "####################",
        "#S                 #",
        "#  ###  #####  ##  #",
        "#  #           #   #",
        "#  #  #####  # #   #",
        "#  #  #      # #   #",
        "#  #  #  #######   #",
        "#  #  #            #",
        "#  #  ##########   #",
        "#                  #",
        "#  ######  #####   #",
        "#        #         #",
        "#  ######  # ##### #",
        "#          #       #",
        "#  #############   #",
        "#             #    #",
        "#  #####  #   #    #",
        "#      #  #   #    #",
        "#                 G#",
        "####################"
    ]
    
    # Find start (S) and goal (G)
    start = goal = None
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == 'S':
                start = (x, y)
            elif cell == 'G':
                goal = (x, y)
    
    if not start or not goal:
        print("Error: Start or goal not found in maze.")
        return
    
    width, height = len(maze[0]), len(maze)
    
    print("=== DEADBAND PROTOCOL DEMONSTRATION ===\n")
    print("Original Maze:")
    print_maze(maze)
    
    # === P0: MAP ROCKS ===
    rocks = map_rocks(maze)
    print(f"\nP0: Mapped {len(rocks)} rocks (negative space).")
    
    # === P1: FIND SAFE CHANNELS ===
    safe_paths = find_safe_channels(start, goal, rocks, (width, height))
    print(f"P1: Found {len(safe_paths)} safe channels.")
    
    if not safe_paths:
        print("No safe channels exist. Agent cannot proceed.")
        return
    
    # === P2: OPTIMIZE WITHIN CHANNEL ===
    optimal_path = optimize_path(safe_paths)
    print(f"P2: Selected optimal path (length={len(optimal_path)}).")
    
    print("\nOptimal Path (P0→P1→P2):")
    print_maze(maze, path=optimal_path, rocks=rocks)
    
    # === UNCONSTRAINED (P2-ONLY) FAILURE DEMO ===
    print("\n--- UNCONSTRAINED (P2-ONLY) FAILURE ---")
    print("If we skip P0+P1 and greedily move toward goal...")
    
    # Simple greedy move (ignores rocks)
    x, y = start
    greedy_path = [start]
    while (x, y) != goal and len(greedy_path) < 100:
        if x < goal[0]:
            x += 1
        elif x > goal[0]:
            x -= 1
        elif y < goal[1]:
            y += 1
        elif y > goal[1]:
            y -= 1
        
        pos = (x, y)
        greedy_path.append(pos)
        
        if pos in rocks:
            print(f"  Hit rock at {pos} after {len(greedy_path)} steps.")
            break
    
    if (x, y) == goal:
        print("  (Unlikely — but in this maze greedy might get lucky.)")
    else:
        print("  Result: Failure (as per fleet simulation: 0/50 success).")

if __name__ == "__main__":
    main()
```

- P0 scans the maze and identifies all `#` as rocks (negative space).
- P1 performs BFS to find all possible paths from start (S) to goal (G) that avoid rocks—these are the safe channels.
- P2 selects the shortest safe path (optimal within constraints).
- It also demonstrates the failure mode of unconstrained P2-only optimization (greedy move) which often hits rocks.
- The maze is a 20x20 example similar to the fleet simulation that produced the proof data.

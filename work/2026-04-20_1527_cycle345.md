```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 protocol with ASCII visualization.
Based on DEADBAND-PROTOCOL.md and constraint_sim_final.json data.
"""

import random

def generate_maze(width=20, height=20, rock_density=0.16):
    """Generate a maze with rocks (obstacles) and open cells."""
    # Based on simulation data: 64 rocks in 20x20 grid ≈ 0.16 density
    maze = [['.' for _ in range(width)] for _ in range(height)]
    rocks = 0
    while rocks < int(width * height * rock_density):
        x = random.randint(0, width-1)
        y = random.randint(0, height-1)
        if maze[y][x] == '.':
            maze[y][x] = '█'  # rock
            rocks += 1
    # Set start (bottom-left) and goal (top-right)
    maze[height-1][0] = 'S'
    maze[0][width-1] = 'G'
    return maze

def p0_map_rocks(maze):
    """P0: Map negative space — identify all rock positions."""
    rocks = []
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == '█':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start, goal, rocks):
    """P1: Find safe channels — BFS avoiding rocks."""
    width, height = len(maze[0]), len(maze)
    directions = [(0,1),(1,0),(0,-1),(-1,0)]
    queue = [(start, [start])]
    visited = set([start])
    safe_paths = []
    
    while queue:
        (x, y), path = queue.pop(0)
        if (x, y) == goal:
            safe_paths.append(path)
            continue
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    return safe_paths

def p2_optimize_path(safe_paths):
    """P2: Optimize within safe channels — pick shortest path."""
    if not safe_paths:
        return None
    return min(safe_paths, key=len)

def visualize(maze, rocks, safe_paths, optimal_path):
    """Print ASCII visualization of the protocol steps."""
    width, height = len(maze[0]), len(maze)
    print("=== DEADBAND PROTOCOL DEMO ===\n")
    
    # Step 0: Original maze
    print("Original Maze (S=start, G=goal, █=rock):")
    for row in maze:
        print(' '.join(row))
    print()
    
    # Step P0: Rocks (negative space)
    print("P0 — Map Negative Space (rocks):")
    p0_grid = [['.' for _ in range(width)] for _ in range(height)]
    for (x, y) in rocks:
        p0_grid[y][x] = '█'
    p0_grid[height-1][0] = 'S'
    p0_grid[0][width-1] = 'G'
    for row in p0_grid:
        print(' '.join(row))
    print(f"Identified {len(rocks)} rocks.\n")
    
    # Step P1: Safe channels (first found path)
    print("P1 — Safe Channels (example path, ·=safe):")
    p1_grid = [['.' for _ in range(width)] for _ in range(height)]
    for (x, y) in rocks:
        p1_grid[y][x] = '█'
    if safe_paths:
        example_path = safe_paths[0]
        for (x, y) in example_path[1:-1]:  # exclude start/goal
            p1_grid[y][x] = '·'
    p1_grid[height-1][0] = 'S'
    p1_grid[0][width-1] = 'G'
    for row in p1_grid:
        print(' '.join(row))
    print(f"Found {len(safe_paths)} safe channel(s).\n")
    
    # Step P2: Optimal path within channels
    print("P2 — Optimal Path Within Safe Channels (*=optimal):")
    p2_grid = [['.' for _ in range(width)] for _ in range(height)]
    for (x, y) in rocks:
        p2_grid[y][x] = '█'
    if optimal_path:
        for (x, y) in optimal_path[1:-1]:
            p2_grid[y][x] = '*'
    p2_grid[height-1][0] = 'S'
    p2_grid[0][width-1] = 'G'
    for row in p2_grid:
        print(' '.join(row))
    if optimal_path:
        print(f"Optimal path length: {len(optimal_path)-1} steps.")
    else:
        print("No safe path found (trapped).")
    print()

def main():
    # Configuration matching fleet simulation (20x20, ~64 rocks)
    WIDTH, HEIGHT = 20, 20
    START = (0, HEIGHT-1)
    GOAL = (WIDTH-1, 0)
    
    # Generate maze
    maze = generate_maze(WIDTH, HEIGHT)
    
    # Apply Deadband Protocol
    rocks = p0_map_rocks(maze)                     # P0
    safe_paths = p1_find_safe_channels(maze, START, GOAL, rocks)  # P1
    optimal_path = p2_optimize_path(safe_paths)    # P2
    
    # Results
    visualize(maze, rocks, safe_paths, optimal_path)
    
    # Protocol summary
    print("=== PROTOCOL SUMMARY ===")
    print("P0 (rocks): Identify what NOT to do → avoid failure states.")
    print("P1 (channels): Find where you CAN be → safe operational space.")
    print("P2 (optimize): Best path within safe channels → constrained optimization.")
    print("\nViolation (skipping P0/P1):")
    print("  Greedy P2-only would hit rocks with high probability.")
    print("  Reference: constraint_sim_final.json shows 0/50 success for constrained-only.")

if __name__ == "__main__":
    main()
```

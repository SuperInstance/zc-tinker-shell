#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
import time
from typing import List, Tuple, Set

# Simulation constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

# Symbols for visualization
AGENT = 'A'
GOAL = 'G'
ROCK = '█'
SAFE = '·'
PATH = '○'
WALL = '│'

class DeadbandMaze:
    """Maze environment implementing Deadband Protocol."""
    
    def __init__(self, seed: int = 42):
        random.seed(seed)
        self.grid = [[SAFE for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.agent_pos = (0, 0)
        self.goal_pos = (MAZE_SIZE-1, MAZE_SIZE-1)
        
        # Place rocks (P0: negative space)
        self._place_rocks()
        
        # Initialize display
        self._update_display()
    
    def _place_rocks(self):
        """P0: Map the negative space (rocks to avoid)."""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            # Don't place rocks on start or goal
            if (x, y) != self.agent_pos and (x, y) != self.goal_pos:
                self.rocks.add((x, y))
                self.grid[y][x] = ROCK
    
    def _update_display(self):
        """Update grid with current positions."""
        display_grid = [row[:] for row in self.grid]
        gx, gy = self.goal_pos
        display_grid[gy][gx] = GOAL
        ax, ay = self.agent_pos
        display_grid[ay][ax] = AGENT
        return display_grid
    
    def print_maze(self, title: str = ""):
        """Print ASCII maze with title."""
        display = self._update_display()
        print(f"\n{'='*60}")
        print(f"DEADBAND PROTOCOL DEMO: {title}")
        print(f"{'='*60}")
        for y in range(MAZE_SIZE):
            row = []
            for x in range(MAZE_SIZE):
                row.append(display[y][x])
            print(' '.join(row))
        print(f"{'='*60}")
    
    def find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Identify all safe channels (paths avoiding rocks)."""
        # Simple BFS to find all safe paths
        start = self.agent_pos
        goal = self.goal_pos
        queue = [(start, [start])]
        safe_paths = []
        
        while queue:
            (x, y), path = queue.pop(0)
            
            # Check if reached goal
            if (x, y) == goal:
                safe_paths.append(path)
                continue
            
            # Explore neighbors (right and down only for simplicity)
            for dx, dy in [(1, 0), (0, 1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in self.rocks and (nx, ny) not in path:
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        return safe_paths
    
    def optimize_within_channel(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """P2: Select optimal path from safe channels."""
        if not safe_paths:
            return []
        
        # Find shortest path (minimum cost)
        optimal_path = min(safe_paths, key=len)
        return optimal_path
    
    def run_unconstrained_greedy(self) -> Tuple[bool, int]:
        """Simulate P2-only (greedy) approach for comparison."""
        path = [self.agent_pos]
        steps = 0
        x, y = self.agent_pos
        
        while (x, y) != self.goal_pos and steps < 100:
            # Greedy: move toward goal
            if x < MAZE_SIZE-1:
                x += 1
            elif y < MAZE_SIZE-1:
                y += 1
            
            new_pos = (x, y)
            
            # Check for rock
            if new_pos in self.rocks:
                return False, steps
            
            path.append(new_pos)
            steps += 1
        
        success = (x, y) == self.goal_pos
        return success, steps

        print("\n[P1] Identifying safe channels...")
        safe_paths = self.find_safe_channels()

import random
from typing import List, Tuple, Set
from collections import deque

class Maze:
    def __init__(self, width=20, height=20, num_rocks=64):
        self.width = width
        self.height = height
        self.num_rocks = num_rocks
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_maze()
        
    def generate_maze(self):
        """Generate random rocks, ensuring start and goal are clear"""
        all_cells = [(x, y) for x in range(self.width) for y in range(self.height)]
        all_cells.remove(self.start)
        all_cells.remove(self.goal)
        
        self.rocks = set(random.sample(all_cells, self.num_rocks))
    
    def is_valid(self, x: int, y: int) -> bool:
        """Check if cell is within bounds and not a rock"""
        return (0 <= x < self.width and 
                0 <= y < self.height and 
                (x, y) not in self.rocks)
    
    def get_neighbors(self, x: int, y: int) -> List[Tuple[int, int]]:
        """Get valid neighboring cells (4-direction movement)"""
        neighbors = []
        for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
            nx, ny = x + dx, y + dy
            if self.is_valid(nx, ny):
                neighbors.append((nx, ny))
        return neighbors
    
    def render(self, path=None, visited=None) -> str:
        """Render maze as ASCII with optional path and visited cells"""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('#')
                elif path and pos in path:
                    row.append('*')
                elif visited and pos in visited:
                    row.append('.')
                else:
                    row.append(' ')
            grid.append(''.join(row))
        return '\n'.join(grid)

import random
from typing import List, Tuple, Set

# Constants from the simulation data
MAZE_SIZE = 20
ROCK_COUNT = 64          # rocks (negative space)
OPTIMAL_PATH_LEN = 21    # shortest possible path
NUM_RUNS = 50            # simulation runs

# Symbols for visualization
AGENT = 'A'
GOAL = 'G'
ROCK = '█'
PATH = '·'
EMPTY = ' '
VISITED = 'x'
SAFE_CHANNEL = '░'

class Maze:
    """20×20 maze with rocks, start at (0,0), goal at (19,19)."""
    
    def __init__(self, seed=None):
        if seed is not None:
            random.seed(seed)
        
        # Initialize empty maze
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        
        # Place rocks (negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            pos = (x, y)
            if pos != self.start and pos != self.goal:
                self.rocks.add(pos)
                self.grid[x][y] = ROCK
    
    def is_valid(self, x: int, y: int) -> bool:
        """Check if position is within bounds and not a rock."""
        return (0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE 
                and (x, y) not in self.rocks)
    
    def get_neighbors(self, x: int, y: int) -> List[Tuple[int, int]]:
        """Get valid neighboring positions (right, down, left, up)."""
        neighbors = []
        for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
            nx, ny = x + dx, y + dy
            if self.is_valid(nx, ny):
                neighbors.append((nx, ny))
        return neighbors
    
    def visualize(self, path=None, visited=None, channels=None):
        """Return ASCII representation of maze."""
        lines = []
        # Create a copy for visualization
        viz_grid = [row[:] for row in self.grid]
        
        # Mark safe channels if provided
        if channels:
            for x, y in channels:
                if viz_grid[x][y] == EMPTY:
                    viz_grid[x][y] = SAFE_CHANNEL
        
        # Mark visited positions
        if visited:
            for x, y in visited:
                if viz_grid[x][y] == EMPTY:
                    viz_grid[x][y] = VISITED
        
        # Mark path
        if path:
            for x, y in path:
                if (x, y) != self.start and (x, y) != self.goal:
                    viz_grid[x][y] = PATH
        
        # Mark start and goal

        gx, gy = self.goal
        viz_grid[sx][sy] = AGENT
        viz_grid[gx][gy] = GOAL
        
        # Build output

        lines.append("=" * 42)
        for y in range(MAZE_SIZE):
            row = []

                row.append(viz_grid[x][y])
            lines.append("│" + "".join(row) + "│")
        lines.append("=" * 42)
        return "\n".join(lines)

def unconstrained_greedy(maze: Maze) -> Tuple[bool, List[Tuple[int, int]], int]:
    """
    P2-only strategy: Greedy optimization toward goal.
    Always moves toward goal without considering rocks.
    """
    path = [maze.start]
    visited = set([maze.start])
    steps = 0
    max_steps = 1000  # Prevent infinite loops
    
    current = maze.start
    while current != maze.goal and steps < max_steps:
        cx, cy = current
        gx, gy = maze.goal
        
        # Greedy: move in direction that reduces Manhattan distance
        best_dist = float('inf')
        best_pos = current
        
        for nx, ny in [(cx+1, cy), (cx, cy+1), (cx-1, cy), (cx, cy-1)]:
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                dist = abs(nx - gx) + abs(ny - gy)
                if dist < best_dist:
                    best_dist = dist
                    best_pos = (nx, ny)
        
        # Move to best position (even if it's a rock!)
        current = best_pos
        path.append(current)
        visited.add(current)
        steps += 1
        
        # Check if hit rock
        if current in maze.rocks:
            return False, path, steps
    
    return current == maze.goal, path, steps

def deadband_protocol(maze: Maze) -> Tuple[bool, List[Tuple[int, int]], int, Set[Tuple[int, int]]]:
    """
    Full P0→P1→P2 strategy:
    P0: Map rocks (negative space)
    P1: Find safe channels (BFS avoiding rocks)
    P2: Optimize within channels (shortest path in safe graph)
    """
    # P0: Rocks are already mapped in maze.rocks
    
    # P1: Find all reachable safe positions (safe channels)
    safe_channels = set()
    queue = [maze.start]
    visited_bfs = set([maze.start])
    
    while queue:
        x, y = queue.pop(0)
        safe_channels.add((x, y))
        
        for nx, ny in maze.get_neighbors(x, y):
            if (nx, ny) not in visited_bfs:
                visited_bfs.add((nx, ny))
                queue.append((nx, ny))
    
    # Check if goal is reachable
    if maze.goal not in safe_channels:
        return False, [], 0, safe_channels
    
    # P2: Find shortest path within safe channels (BFS)
    parent = {maze.start: None}
    queue = [maze.start]
    
    while queue:
        current = queue.pop(0)
        if current == maze.goal:
            break
        
        x, y = current
        for neighbor in maze.get_neighbors(x, y):
            if neighbor not in parent:
                parent[neighbor] = current
                queue.append(neighbor)
    
    # Reconstruct path
    if maze.goal not in parent:
        return False, [], 0, safe_channels
    
    path = []
    current = maze.goal
    while current is not None:
        path.append(current)
        current = parent[current]
    path.reverse()
    
    return True, path, len(path)-1, safe_channels

            total_steps += steps
            paths.append(path)
    
    avg_steps = total_steps / successes if successes > 0 else 0
    return successes, avg_steps, paths

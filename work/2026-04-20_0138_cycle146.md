#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 priority in a 20x20 maze with rocks.
Based on fleet-knowledge simulation data.
"""

import random
from typing import List, Tuple, Set
from collections import deque

# Simulation parameters from fleet-knowledge
MAZE_SIZE = 20
ROCK_COUNT = 64  # From constraint_sim_final.json: "rocks": 64
OPTIMAL_PATH_LENGTH = 21  # "optimal": 21
SIM_RUNS = 50  # Matches the proof's 50 runs

# Maze symbols
EMPTY = '·'
ROCK = '█'
START = 'S'
GOAL = 'G'
PATH = '○'
DEADBAND_PATH = '◎'
FAILED_PATH = '×'

class Maze:
    def __init__(self):
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self._generate_rocks()
        
    def _generate_rocks(self):
        """Place rocks randomly, avoiding start and goal."""
        possible = [(r, c) for r in range(MAZE_SIZE) for c in range(MAZE_SIZE)]
        possible.remove(self.start)
        possible.remove(self.goal)
        self.rocks = set(random.sample(possible, ROCK_COUNT))
        for r, c in self.rocks:
            self.grid[r][c] = ROCK
        self.grid[self.start[0]][self.start[1]] = START
        self.grid[self.goal[0]][self.goal[1]] = GOAL
    
    def is_rock(self, pos: Tuple[int, int]) -> bool:
        return pos in self.rocks
    
    def in_bounds(self, pos: Tuple[int, int]) -> bool:
        r, c = pos
        return 0 <= r < MAZE_SIZE and 0 <= c < MAZE_SIZE
    
    def get_neighbors(self, pos: Tuple[int, int]) -> List[Tuple[int, int]]:
        r, c = pos
        candidates = [(r-1, c), (r+1, c), (r, c-1), (r, c+1)]
        return [p for p in candidates if self.in_bounds(p)]
    
    def display(self, path=None, path_symbol=PATH):
        """Print maze with optional path overlay."""
        display_grid = [row[:] for row in self.grid]
        if path:
            for r, c in path:
                if (r, c) not in (self.start, self.goal):
                    display_grid[r][c] = path_symbol
        for row in display_grid:
            print(''.join(row))
        print()

class Agent:
    def __init__(self, maze: Maze):
        self.maze = maze
        self.steps = 0
        
    def unconstrained_greedy(self) -> Tuple[List[Tuple[int, int]], bool]:
        """P2-only: Greedy Manhattan distance, ignores rocks."""
        path = [self.maze.start]
        visited = {self.maze.start}
        current = self.maze.start
        
        while current != self.maze.goal and self.steps < 1000:
            neighbors = self.maze.get_neighbors(current)
            # Greedy: pick neighbor closest to goal
            neighbors.sort(key=lambda p: abs(p[0]-self.maze.goal[0]) + abs(p[1]-self.maze.goal[1]))
            for neighbor in neighbors:
                if neighbor not in visited:
                    current = neighbor
                    path.append(current)
                    visited.add(current)
                    self.steps += 1
                    break
            else:
                # No unvisited neighbors
                break
        
        success = (current == self.maze.goal)
        return path, success
    
    def constrained_random_avoid(self) -> Tuple[List[Tuple[int, int]], bool]:
        """P0+P2 (flawed): Knows rocks but doesn't map channels properly."""
        path = [self.maze.start]
        visited = {self.maze.start}
        current = self.maze.start
        
        while current != self.maze.goal and self.steps < 1000:
            neighbors = [n for n in self.maze.get_neighbors(current) if not self.maze.is_rock(n)]
            if not neighbors:
                break
            # Random choice (simulates poor channel mapping)
            current = random.choice(neighbors)
            if current in visited:
                continue
            path.append(current)
            visited.add(current)
            self.steps += 1
        
        success = (current == self.maze.goal)
        return path, success
    
    def deadband_protocol(self) -> Tuple[List[Tuple[int, int]], bool]:
        """P0→P1→P2: Full deadband protocol."""
        # P0: Map negative space (rocks)
        rocks = self.maze.rocks  # Already known
        
        # P1: Identify safe channels (BFS to find all reachable non-rock positions)
        safe = set()
        queue = deque([self.maze.start])
        while queue:
            pos = queue.popleft()
            if pos in safe:
                continue
            safe.add(pos)
            for neighbor in self.maze.get_neighbors(pos):
                if not self.maze.is_rock(neighbor) and neighbor not in safe:
                    queue.append(neighbor)
        
        # If goal not in safe, no channel exists
        if self.maze.goal not in safe:
            return [self.maze.start], False
        
        # P2: Optimize within safe channel (BFS shortest path)
        parent = {self.maze.start: None}
        queue = deque([self.maze.start])
        while queue:
            pos = queue.popleft()
            if pos == self.maze.goal:
                break
            for neighbor in self.maze.get_neighbors(pos):
                if neighbor in safe and neighbor not in parent:
                    parent[neighbor] = pos
                    queue.append(neighbor)
        
        # Reconstruct path
        if self.maze.goal not in parent:
            return [self.maze.start], False
        
        path = []
        current = self.maze.goal
        while current is not None:
            path.append(current)
            current = parent[current]
        path.reverse()
        self.steps = len(path) - 1
        return path, True

    stats = {

        # Constrained (P0+P2 flawed)

        if success:

            stats['constrained']['total_steps'] += agent.steps
        
        # Deadband Protocol (P0→P1→P2)
        agent = Agent(maze)
        path, success = agent.deadband_protocol()

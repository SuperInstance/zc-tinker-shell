**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on reading `fleet-knowledge/narrow-games/constraint_sim_final.json` and `fleet-knowledge/docs/DEADBAND-PROTOCOL.md`, I will implement a concrete demonstration. The simulation data shows: unconstrained greedy (P2-only) fails (0% success), while deadband protocol (P0→P1→P2) achieves 100% success at optimal cost (avg 21 vs 2320.9). I will model this.

**Approach:**
1. **P0 (Map negative space):** Generate a maze with rocks ('#') and open cells ('.'). Identify all rock positions.
2. **P1 (Find safe channels):** From start to goal, find all paths that avoid rocks. Use BFS to enumerate safe paths.
3. **P2 (Optimize within channels):** From the set of safe paths, select the shortest one.
4. **Visualization:** Print the maze with the chosen path marked.

**Output:** A Python script `deadband_maze_demo.py` that runs the protocol and prints each step.

## File: `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 in a 10x10 ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque

def generate_maze(rows=10, cols=10, rock_density=0.2):
    """Generate a random maze with rocks (#) and open cells (.)."""
    maze = []
    for r in range(rows):
        row = []
        for c in range(cols):
            if random.random() < rock_density:
                row.append('#')  # rock
            else:
                row.append('.')  # open
        maze.append(row)
    # Ensure start (0,0) and goal (rows-1, cols-1) are open
    maze[0][0] = '.'
    maze[rows-1][cols-1] = '.'
    return maze

def print_maze(maze, path=None):
    """Print maze, optionally highlighting path with *."""
    for r in range(len(maze)):
        line = []
        for c in range(len(maze[0])):
            if path and (r, c) in path:
                line.append('*')
            else:
                line.append(maze[r][c])
        print(' '.join(line))

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks). Return set of rock positions."""
    rocks = set()
    for r in range(len(maze)):
        for c in range(len(maze[0])):
            if maze[r][c] == '#':
                rocks.add((r, c))
    return rocks

def p1_find_safe_channels(maze, rocks, start, goal):
    """P1: Find all safe paths (avoiding rocks) from start to goal using BFS."""
    rows, cols = len(maze), len(maze[0])
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    # BFS queue: (position, path_so_far)
    queue = deque([(start, [start])])
    safe_paths = []
    
    while queue:
        (r, c), path = queue.popleft()
        
        if (r, c) == goal:
            safe_paths.append(path)
            continue
        
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < rows and 0 <= nc < cols:
                if (nr, nc) not in rocks and (nr, nc) not in path:
                    queue.append(((nr, nc), path + [(nr, nc)]))
    
    return safe_paths

def p2_optimize_within_channels(safe_paths):
    """P2: Select shortest path from safe channels."""
    if not safe_paths:
        return None
    return min(safe_paths, key=len)

def run_deadband_protocol():
    """Execute P0→P1→P2 and display results."""
    print("=== Deadband Protocol Maze Demo ===\n")
    
    # Generate maze
    maze = generate_maze(rows=10, cols=10, rock_density=0.2)
    print("1. Generated Maze ('.' = open, '#' = rock):")
    print_maze(maze)
    print()
    
    # Define start and goal
    start = (0, 0)
    goal = (9, 9)
    print(f"Start: {start}, Goal: {goal}\n")
    
    # P0: Map negative space
    rocks = p0_map_rocks(maze)
    print(f"2. P0 — Map Negative Space: Found {len(rocks)} rocks.")
    
    # P1: Find safe channels
    safe_paths = p1_find_safe_channels(maze, rocks, start, goal)
    print(f"3. P1 — Find Safe Channels: Found {len(safe_paths)} safe paths.")
    
    if not safe_paths:
        print("   No safe path exists. Protocol stops at P1.")
        return
    
    # P2: Optimize within channels
    optimal_path = p2_optimize_within_channels(safe_paths)
    print(f"4. P2 — Optimize Within Channels: Selected shortest safe path (length {len(optimal_path)-1}).\n")
    
    # Visualization
    print("5. Final Maze with Optimal Path (*):")
    print_maze(maze, set(optimal_path))
    print(f"\nPath: {optimal_path}")
    
    # Simulate unconstrained greedy (P2-only) for comparison
    print("\n--- Comparison: Unconstrained Greedy (P2-only) ---")
    # Simple greedy: move right/down toward goal, ignoring rocks
    greedy_path = []
    r, c = start
    while (r, c) != goal:
        greedy_path.append((r, c))
        if r < goal[0] and maze[r+1][c] != '#':
            r += 1
        elif c < goal[1] and maze[r][c+1] != '#':
            c += 1
        else:
            break
    greedy_path.append((r, c))
    
    if greedy_path[-1] == goal:
        print(f"Greedy path found (length {len(greedy_path)-1}): {greedy_path}")
    else:
        print(f"Greedy failed — hit a rock or dead end at {greedy_path[-1]}.")
    
    # Summary
    print("\n=== Protocol Summary ===")
    print(f"Deadband Protocol (P0→P1→P2): Success = {'Yes' if optimal_path else 'No'}")
    print(f"Unconstrained Greedy (P2-only): Success = {'Yes' if greedy_path[-1]==goal else 'No'}")
    print("Matches simulation trend: deadband succeeds, greedy often fails.")

if __name__ == "__main__":
    run_deadband_protocol()
```

This demonstrates the fleet doctrine: **P0 (don't hit rocks) → P1 (find safe channels) → P2 (optimize within channels)**.

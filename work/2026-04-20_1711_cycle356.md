#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_rocks(rock_density)
        self.safe_channels: List[Set[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
    def generate_rocks(self, density: float):
        """P0: Generate rocks (negative space)"""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        # Avoid start and goal
        possible = [(x, y) for x in range(self.width) for y in range(self.height)
                   if (x, y) not in [self.start, self.goal]]
        self.rocks = set(random.sample(possible, num_rocks))
    
    def is_safe(self, x: int, y: int) -> bool:
        """Check if cell is not a rock"""
        return (x, y) not in self.rocks and 0 <= x < self.width and 0 <= y < self.height
    
    def find_safe_channels(self):
        """P1: Find all safe channels (connected safe cells) using BFS"""
        visited = set()
        for x in range(self.width):
            for y in range(self.height):
                if (x, y) not in visited and self.is_safe(x, y):
                    # BFS to find connected safe region
                    stack = [(x, y)]
                    channel = set()
                    while stack:
                        cx, cy = stack.pop()
                        if (cx, cy) in visited:
                            continue
                        visited.add((cx, cy))
                        channel.add((cx, cy))
                        # Check 4-connected neighbors
                        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                            nx, ny = cx + dx, cy + dy
                            if self.is_safe(nx, ny) and (nx, ny) not in visited:
                                stack.append((nx, ny))
                    if channel:
                        self.safe_channels.append(channel)
    
    def find_optimal_path(self):
        """P2: Find shortest path within safe channels (A* within largest channel)"""
        if not self.safe_channels:
            return []
        
        # Find channel containing start and goal
        start_channel = None
        goal_channel = None
        for channel in self.safe_channels:
            if self.start in channel:
                start_channel = channel
            if self.goal in channel:
                goal_channel = channel
        
        # Need same channel for both
        if not start_channel or start_channel != goal_channel:
            return []
        
        # A* search within safe channel
        open_set = {self.start}
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: self.heuristic(self.start)}
        
        while open_set:
            current = min(open_set, key=lambda p: f_score.get(p, float('inf')))
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                return list(reversed(path))
            
            open_set.remove(current)
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                if neighbor not in start_channel:
                    continue
                
                tentative_g = g_score[current] + 1
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self.heuristic(neighbor)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        return []
    
    def heuristic(self, pos: Tuple[int, int]) -> int:
        """Manhattan distance to goal"""
        return abs(pos[0] - self.goal[0]) + abs(pos[1] - self.goal[1])
    
    def render(self, show_path: bool = False) -> str:
        """Render ASCII maze"""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif (x, y) in self.rocks:
                    row.append('#')
                elif show_path and (x, y) in self.optimal_path:
                    row.append('·')
                elif self.is_safe(x, y):
                    row.append('.')
                else:
                    row.append(' ')
            grid.append(''.join(row))
        return '\n'.join(grid)

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        """
        Initialize a maze with rocks (obstacles), start (S), goal (G).
        rock_density ~ 64 rocks / 400 cells = 0.16 from simulation.
        """
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        
        # Place rocks (P0 negative space)
        num_rocks = int(width * height * rock_density)
        rocks_placed = 0
        while rocks_placed < num_rocks:
            x = random.randint(0, width-1)
            y = random.randint(0, height-1)
            if self.grid[y][x] == '.':
                self.grid[y][x] = '#'
                rocks_placed += 1
        
        # Start (top-left) and Goal (bottom-right)
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
        
        # For visualization
        self.rock_map = None
        self.safe_channels = None
        self.optimal_path = None
    
    def print_maze(self, title="Initial Maze"):
        """Print the current maze state."""
        print(f"\n=== {title} ===")
        for row in self.grid:
            print(' '.join(row))
        print(f"Start: {self.start}, Goal: {self.goal}")
    
    def p0_map_rocks(self):
        """P0: Map negative space (rocks)."""
        rock_map = [[False for _ in range(self.width)] for _ in range(self.height)]
        for y in range(self.height):
            for x in range(self.width):
                if self.grid[y][x] == '#':
                    rock_map[y][x] = True
        self.rock_map = rock_map
        print("\n[P0] Rock map complete — identified", 
              sum(sum(row) for row in rock_map), "rocks.")
        return rock_map
    
    def p1_find_safe_channels(self):
        """P1: Find safe channels (cells reachable without hitting rocks)."""
        if self.rock_map is None:
            self.p0_map_rocks()
        
        # BFS from start, avoiding rocks
        visited = [[False for _ in range(self.width)] for _ in range(self.height)]
        queue = deque([self.start])
        visited[self.start[1]][self.start[0]] = True
        
        directions = [(1,0), (-1,0), (0,1), (0,-1)]  # 4-connected
        
        while queue:
            x, y = queue.popleft()
            for dx, dy in directions:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if not visited[ny][nx] and not self.rock_map[ny][nx]:
                        visited[ny][nx] = True
                        queue.append((nx, ny))
        
        self.safe_channels = visited
        safe_cells = sum(sum(row) for row in visited)
        print(f"[P1] Safe channels mapped — {safe_cells} reachable cells.")
        return visited
    
    def p2_optimize_within_channel(self):
        """P2: Find shortest path within safe channels (deadband)."""
        if self.safe_channels is None:
            self.p1_find_safe_channels()
        
        # BFS for shortest path
        parent = {self.start: None}
        queue = deque([self.start])
        directions = [(1,0), (-1,0), (0,1), (0,-1)]
        
        while queue:
            x, y = queue.popleft()
            if (x, y) == self.goal:
                break
            for dx, dy in directions:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if self.safe_channels[ny][nx] and (nx, ny) not in parent:
                        parent[(nx, ny)] = (x, y)
                        queue.append((nx, ny))
        
        # Reconstruct path
        if self.goal in parent:
            path = []
            current = self.goal
            while current is not None:
                path.append(current)
                current = parent[current]
            path.reverse()
            self.optimal_path = path
            print(f"[P2] Optimal path found — length {len(path)-1} steps.")
            return path
        else:
            print("[P2] No safe path to goal within channels.")
            return None
    
    def visualize_protocol(self):
        """Show the three steps with ASCII art."""
        # Step P0
        print("\n" + "="*50)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*50)
        self.print_maze("Initial State")
        
        # P0 visualization
        print("\n--- P0: MAP ROCKS (Negative Space) ---")
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if self.grid[y][x] in ['S', 'G']:
                    row.append(self.grid[y][x])
                elif self.rock_map[y][x]:
                    row.append('#')
                else:
                    row.append('.')
            print(' '.join(row))
        
        # P1 visualization
        print("\n--- P1: SAFE CHANNELS (Deadband) ---")
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if self.grid[y][x] in ['S', 'G']:
                    row.append(self.grid[y][x])
                elif self.safe_channels[y][x]:
                    row.append(' ')
                else:
                    row.append('#')
            print(' '.join(row))
        
        # P2 visualization
        print("\n--- P2: OPTIMAL PATH (Within Channels) ---")
        display_grid = [row[:] for row in self.grid]
        if self.optimal_path:
            for (x, y) in self.optimal_path[1:-1]:  # Exclude start and goal
                if display_grid[y][x] == '.':
                    display_grid[y][x] = '*'
        
        for row in display_grid:
            print(' '.join(row))
        
        # Summary
        print("\n=== PROTOCOL SUMMARY ===")
        print(f"Rocks (P0): {sum(sum(row) for row in self.rock_map)}")
        print(f"Safe cells (P1): {sum(sum(row) for row in self.safe_channels)}")
        if self.optimal_path:
            print(f"Path length (P2): {len(self.optimal_path)-1}")
            print("Success: Agent reaches goal via deadband protocol.")
        else:
            print("Failure: No safe path exists.")
        
        # Compare with unconstrained greedy (simulated failure)
        print("\n--- CONTRAST: Unconstrained Greedy (P2-only) ---")
        print("A greedy agent moving right/down toward goal would:")
        x, y = self.start
        steps = 0
        while (x, y) != self.goal and steps < 100:
            if x < self.goal[0] and self.grid[y][x+1] != '#':
                x += 1
            elif y < self.goal[1] and self.grid[y+1][x] != '#':
                y += 1
            else:
                break
            steps += 1
            if self.grid[y][x] == '#':
                print(f"  Step {steps}: Hit rock at ({x},{y}) — FAILURE")
                break
        if (x, y) == self.goal:
            print(f"  Reached goal in {steps} steps (lucky, no rocks in path).")
        else:
            print("  Trapped in local minimum.")

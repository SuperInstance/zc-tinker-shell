#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Don't hit rocks (walls)
P1: Find safe channels (paths)
P2: Optimize (shortest path)
"""

import random
import time

class Maze:
    def __init__(self, width=20, height=10):
        self.width = width
        self.height = height
        self.grid = [['#' for _ in range(width)] for _ in range(height)]
        self.start = (1, 1)
        self.goal = (width-2, height-2)
        self.generate()
    
    def generate(self):
        # Create simple maze with open path
        for y in range(1, self.height-1):
            for x in range(1, self.width-1):
                if random.random() > 0.3:  # 70% open space
                    self.grid[y][x] = '.'
        
        # Ensure start and goal are open
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
        
        # Create a guaranteed path
        x, y = self.start
        while (x, y) != self.goal:
            if x < self.goal[0]:
                x += 1
            elif y < self.goal[1]:
                y += 1
            self.grid[y][x] = '.'
    
    def display(self, agent_pos=None):
        for y in range(self.height):
            row = ''
            for x in range(self.width):
                if agent_pos and (x, y) == agent_pos:
                    row += 'A'
                else:
                    row += self.grid[y][x]
            print(row)
        print()

class DeadbandAgent:
    def __init__(self, maze):
        self.maze = maze
        self.pos = maze.start
        self.path = []
        self.state = "P0"
    
    def run_protocol(self):
        print("=== DEADBAND PROTOCOL DEMO ===")
        print("P0: Don't hit rocks (walls)")
        print("P1: Find safe channels (paths)")
        print("P2: Optimize (shortest path)")
        print()
        
        # P0: Basic collision avoidance
        print("[P0] Avoiding walls...")
        self.avoid_walls()
        self.maze.display(self.pos)
        time.sleep(1)
        
        # P1: Path finding
        print("[P1] Finding safe channels...")
        self.find_path()
        self.maze.display(self.pos)
        time.sleep(1)
        
        # P2: Optimization
        print("[P2] Optimizing path...")
        self.optimize_path()
        self.maze.display(self.pos)
        print("Goal reached! Protocol complete.")
    
    def avoid_walls(self):
        # Simple wall avoidance - move toward goal but don't hit walls
        x, y = self.pos
        gx, gy = self.maze.goal
        
        # Try moving right/down if open
        if x < gx and self.maze.grid[y][x+1] != '#':
            x += 1
        elif y < gy and self.maze.grid[y+1][x] != '#':
            y += 1
        
        self.pos = (x, y)
        self.path.append(self.pos)
    
    def find_path(self):
        # Simple pathfinding - continue toward goal
        x, y = self.pos
        gx, gy = self.maze.goal
        
        while (x, y) != (gx, gy):
            # Move toward goal
            if x < gx and self.maze.grid[y][x+1] != '#':
                x += 1
            elif y < gy and self.maze.grid[y+1][x] != '#':
                y += 1
            elif x > gx and self.maze.grid[y][x-1] != '#':
                x -= 1
            elif y > gy and self.maze.grid[y-1][x] != '#':
                y -= 1
            
            self.pos = (x, y)
            self.path.append(self.pos)
            
            # Show progress
            if len(self.path) % 5 == 0:
                self.maze.display(self.pos)
                time.sleep(0.3)
    
    def optimize_path(self):
        # Simple optimization - remove unnecessary steps
        if len(self.path) > 2:
            # Keep only key turning points
            optimized = [self.path[0]]
            for i in range(1, len(self.path)-1):
                prev = optimized[-1]
                curr = self.path[i]
                next_pos = self.path[i+1]
                
                # Keep if direction changes
                if (curr[0] - prev[0]) != (next_pos[0] - curr[0]) or \
                   (curr[1] - prev[1]) != (next_pos[1] - curr[1]):
                    optimized.append(curr)
            
            optimized.append(self.path[-1])
            self.path = optimized
        
        print(f"Path optimized: {len(self.path)} steps")
        print(f"Path: {self.path}")

def main():
    print("Initializing maze...")
    maze = Maze(30, 15)
    maze.display()
    
    agent = DeadbandAgent(maze)
    agent.run_protocol()

if __name__ == "__main__":
    main()

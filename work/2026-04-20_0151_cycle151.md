#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Simulates the P0→P1→P2 priority rule in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# Constants from fleet-knowledge
MAZE_SIZE = 20
OPTIMAL_PATH_LENGTH = 21  # from constraint_sim_final.json
ROCK_COUNT = 64           # from constraint_sim_final.json

class DeadbandMaze:
    """Implements the Deadband Protocol in a maze environment."""
    
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
    def p0_map_negative_space(self) -> None:
        """P0: Identify rocks (what NOT to do)."""
        # Generate random rocks (obstacles)
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[x][y] = '█'
        
        print("P0: Mapped negative space (rocks)")
        print(f"   Found {len(self.rocks)} rocks (failure states)")
    
    def p1_find_safe_channels(self) -> bool:
        """P1: Find all paths that avoid rocks (safe channels)."""
        # Simple BFS to find safe paths
        visited = set()
        queue = [(self.start, [self.start])]
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.goal:
                self.safe_channels.append(path)
                continue
            
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            
            # Check adjacent cells (up, down, left, right)
            for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        if self.safe_channels:
            print(f"P1: Found {len(self.safe_channels)} safe channels")
            return True
        else:
            print("P1: No safe channels found (impossible maze)")
            return False
    
    def p2_optimize_within_channel(self) -> None:
        """P2: Find shortest path within safe channels."""
        if not self.safe_channels:
            print("P2 ERROR: Cannot optimize without safe channels (P1 not completed)")
            return
        
        # Find shortest safe path
        self.optimal_path = min(self.safe_channels, key=len)
        
        # Mark optimal path on grid (excluding start/goal for clarity)
        for (x, y) in self.optimal_path[1:-1]:
            if self.grid[x][y] == '.':
                self.grid[x][y] = '○'
        
        print(f"P2: Optimal path length = {len(self.optimal_path)} steps")
        print(f"   (Theoretical optimal = {OPTIMAL_PATH_LENGTH})")
    
    def run_protocol(self) -> bool:
        """Execute strict P0→P1→P2 priority order."""
        print("\n" + "="*50)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*50)
        
        # P0: Map negative space (MANDATORY FIRST STEP)
        self.p0_map_negative_space()
        
        # P1: Find safe channels (MANDATORY SECOND STEP)
        if not self.p1_find_safe_channels():
            return False
        
        # P2: Optimize within channels (ONLY AFTER P0+P1)
        self.p2_optimize_within_channel()
        
        return True
    
    def display_maze(self, show_legend: bool = True) -> None:
        """Display the maze with current state."""
        print("\n" + "="*50)
        print("MAZE STATE")
        print("="*50)
        
        # Add start and goal markers
        sx, sy = self.start
        gx, gy = self.goal
        self.grid[sx][sy] = 'S'
        self.grid[gx][gy] = 'G'
        
        # Print grid
        for y in range(MAZE_SIZE):
            row = []
            for x in range(MAZE_SIZE):
                row.append(self.grid[x][y])
            print(' '.join(row))
        
        if show_legend:
            print("\nLEGEND:")
            print("  S = Start (0,0)")
            print("  G = Goal (19,19)")
            print("  █ = Rock (P0 negative space)")
            print("  ○ = Optimal path (P2 within safe channel)")
            print("  . = Unexplored safe space")
    
    def simulate_violation(self) -> None:
        """Demonstrate what happens when violating P0→P1→P2 order."""
        print("\n" + "="*50)
        print("PROTOCOL VIOLATION: P2 BEFORE P0+P1")
        print("="*50)
        
        # Reset grid
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks.clear()
        self.safe_channels.clear()
        self.optimal_path.clear()
        
        # Generate rocks but don't map them (skip P0)
        for _ in range(ROCK_COUNT):
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[x][y] = '█'
        
        # Try to optimize directly (P2 without P0+P1)
        print("Attempting direct optimization (P2 only)...")
        
        # Greedy pathfinder that ignores rocks
        path = [self.start]
        x, y = self.start
        steps = 0
        max_steps = 100
        
        while (x, y) != self.goal and steps < max_steps:
            # Move toward goal (greedy)
            if x < self.goal[0]:
                x += 1
            elif y < self.goal[1]:
                y += 1
            elif x > self.goal[0]:
                x -= 1
            elif y > self.goal[1]:
                y -= 1
            
            path.append((x, y))
            steps += 1
            
            # Check for rock collision
            if (x, y) in self.rocks:
                print(f"  HIT ROCK at ({x},{y}) after {steps} steps!")
                print("  Result: FAILURE (as predicted by Deadband Protocol)")
                return
        
        if (x, y) == self.goal:
            print(f"  Reached goal in {steps} steps (lucky, but unreliable)")
        else:
            print(f"  Failed to reach goal in {steps} steps")

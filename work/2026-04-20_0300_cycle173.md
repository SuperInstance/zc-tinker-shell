#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 priority in a simple ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
and DEADBAND-PROTOCOL.md.
"""

import random
from typing import List, Tuple, Set

class Maze:
    """A 10x10 maze with rocks, start, goal, and safe channels."""
    def __init__(self, seed=42):
        random.seed(seed)
        self.width = 10
        self.height = 10
        self.start = (0, 0)
        self.goal = (9, 9)
        
        # P0: Rocks (negative space) — positions to avoid
        self.rocks: Set[Tuple[int, int]] = set()
        num_rocks = 15
        while len(self.rocks) < num_rocks:
            x = random.randint(0, self.width-1)
            y = random.randint(0, self.height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
        
        # P1: Safe channels — all non‑rock cells
        self.safe_cells = set()
        for x in range(self.width):
            for y in range(self.height):
                if (x, y) not in self.rocks:
                    self.safe_cells.add((x, y))
    
    def is_rock(self, x, y):
        return (x, y) in self.rocks
    
    def is_safe(self, x, y):
        return (x, y) in self.safe_cells
    
    def in_bounds(self, x, y):
        return 0 <= x < self.width and 0 <= y < self.height
    
    def render(self, path=None):
        """Return ASCII representation of the maze."""
        lines = []
        for y in range(self.height):
            line = []
            for x in range(self.width):
                if (x, y) == self.start:
                    line.append('S')
                elif (x, y) == self.goal:
                    line.append('G')
                elif path and (x, y) in path:
                    line.append('*')
                elif self.is_rock(x, y):
                    line.append('#')
                else:
                    line.append('.')
            lines.append(' '.join(line))
        return '\n'.join(lines)

class Agent:
    """Base agent with movement logic."""
    def __init__(self, maze):
        self.maze = maze
        self.path = []
        self.steps = 0
    
    def move_toward_goal(self, x, y):
        """Greedy move: reduce Manhattan distance to goal."""
        gx, gy = self.maze.goal
        dx = gx - x
        dy = gy - y
        moves = []
        if dx > 0 and self.maze.in_bounds(x+1, y):
            moves.append((x+1, y))
        elif dx < 0 and self.maze.in_bounds(x-1, y):
            moves.append((x-1, y))
        if dy > 0 and self.maze.in_bounds(x, y+1):
            moves.append((x, y+1))
        elif dy < 0 and self.maze.in_bounds(x, y-1):
            moves.append((x, y-1))
        if moves:
            return random.choice(moves)
        return (x, y)  # stay if no progress possible

class UnconstrainedAgent(Agent):
    """P2‑only agent: optimizes for goal without mapping rocks (violates protocol)."""
    def run(self):
        x, y = self.maze.start
        self.path = [(x, y)]
        visited = set(self.path)
        while (x, y) != self.maze.goal and self.steps < 100:
            nx, ny = self.move_toward_goal(x, y)
            if (nx, ny) in visited:
                # random walk to avoid loops
                candidates = [(nx+1, ny), (nx-1, ny), (nx, ny+1), (nx, ny-1)]
                candidates = [c for c in candidates if self.maze.in_bounds(*c)]
                if candidates:
                    nx, ny = random.choice(candidates)
            if self.maze.is_rock(nx, ny):
                # hit a rock → failure
                self.path.append((nx, ny))
                self.steps += 1
                return False
            x, y = nx, ny
            self.path.append((x, y))
            visited.add((x, y))
            self.steps += 1
        return (x, y) == self.maze.goal

class DeadbandAgent(Agent):
    """P0→P1→P2 agent: follows the Deadband Protocol."""
    def run(self):
        # P0: Map negative space (rocks)
        rocks = self.maze.rocks  # known from environment
        
        # P1: Identify safe channels (non‑rock cells)
        safe = self.maze.safe_cells
        
        # P2: Optimize within safe channels
        x, y = self.maze.start
        self.path = [(x, y)]
        visited = set(self.path)
        while (x, y) != self.maze.goal and self.steps < 100:
            # candidate moves (4‑connected)
            candidates = [(x+1, y), (x-1, y), (x, y+1), (x, y-1)]
            candidates = [c for c in candidates if self.maze.in_bounds(*c)]
            # filter to safe channels only (P1 constraint)
            safe_moves = [c for c in candidates if c in safe]
            if not safe_moves:
                # no safe move → failure (trapped)
                return False
            # choose move that reduces distance to goal (P2 optimization)
            best = None
            best_dist = float('inf')
            gx, gy = self.maze.goal
            for cx, cy in safe_moves:
                dist = abs(cx - gx) + abs(cy - gy)
                if dist < best_dist and (cx, cy) not in visited:
                    best_dist = dist
                    best = (cx, cy)
            if best is None:
                # all safe moves visited, pick any safe
                best = random.choice(safe_moves)
            x, y = best
            self.path.append((x, y))
            visited.add((x, y))
            self.steps += 1
        return (x, y) == self.maze.goal

    print(f"  Path length: {len(ua.path)}")
    if ua_success:
        print("  Path visualization:")
        print(maze.render(set(ua.path)))
    else:
        print("  Failed — hit a rock or trapped.")
    print()
    
    # Run deadband agent (P0→P1→P2)
    da = DeadbandAgent(maze)

    print(f"Deadband Agent (P0→P1→P2):")
    print(f"  Success: {da_success}")
    print(f"  Steps: {da.steps}")
    print(f"  Path length: {len(da.path)}")
    if da_success:
        print("  Path visualization (* = agent path):")
        print(maze.render(set(da.path)))

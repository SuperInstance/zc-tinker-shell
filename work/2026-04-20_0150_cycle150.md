**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Implementing based on fleet-knowledge data. Reading `constraint_sim_final.json` and `DEADBAND-PROTOCOL.md` to ensure the simulation matches the fleet's proven results.

**Approach:**
1. **P0 (Map negative space):** Generate a 20x20 maze with random obstacles ("rocks"). The agent cannot pass through rocks.
2. **P1 (Identify safe channels):** Use BFS to find all reachable cells from the start that avoid rocks. These form the safe deadband.
3. **P2 (Optimize within channel):** Use A* search constrained to the safe deadband to find the shortest path to the goal.
4. **Visualization:** Print ASCII maze showing start (S), goal (G), rocks (#), safe path (.).

**Output:** A Python script `deadband_maze_demo.py` that runs the simulation and prints results matching the fleet's data pattern.

## File: `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 in a 20x20 maze.
Based on fleet-knowledge/constraint_sim_final.json and DEADBAND-PROTOCOL.md.
"""

import random
from collections import deque
import heapq

# Constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # matches "rocks": 64 in constraint_sim_final.json
START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

def generate_maze():
    """P0: Generate maze with rocks (negative space)."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        r = random.randint(0, MAZE_SIZE - 1)
        c = random.randint(0, MAZE_SIZE - 1)
        if (r, c) != START and (r, c) != GOAL:
            rocks.add((r, c))
            maze[r][c] = '#'
    maze[START[0]][START[1]] = 'S'
    maze[GOAL[0]][GOAL[1]] = 'G'
    return maze, rocks

def p1_find_safe_channels(maze, rocks):
    """P1: BFS to find all cells reachable without hitting rocks."""
    safe = set()
    queue = deque([START])
    visited = set([START])
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        r, c = queue.popleft()
        safe.add((r, c))
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if (nr, nc) not in visited and (nr, nc) not in rocks:
                    visited.add((nr, nc))
                    queue.append((nr, nc))
    return safe

def p2_optimize_in_channel(maze, safe):
    """P2: A* search constrained to safe cells."""
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, START))
    came_from = {}
    g_score = {START: 0}
    f_score = {START: heuristic(START, GOAL)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == GOAL:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            return path
        
        for dr, dc in dirs:
            nr, nc = current[0] + dr, current[1] + dc
            neighbor = (nr, nc)
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if neighbor not in safe:
                    continue  # Not in safe channel → skip
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, GOAL)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    return None  # No path found

def run_simulation():
    """Run one simulation cycle and return results."""
    maze, rocks = generate_maze()
    safe = p1_find_safe_channels(maze, rocks)
    path = p2_optimize_in_channel(maze, safe)
    
    # Mark path on maze
    maze_display = [row[:] for row in maze]
    if path:
        for r, c in path:
            if (r, c) != START and (r, c) != GOAL:
                maze_display[r][c] = '.'
    
    success = path is not None
    path_length = len(path) if path else 0
    
    return maze_display, success, path_length, len(safe)

def print_maze(maze):
    """Print ASCII maze."""
    for row in maze:
        print(' '.join(row))
    print()

def main():
    print("=== Deadband Protocol Maze Demo ===\n")
    print("P0: Generate maze with 64 rocks (negative space)")
    print("P1: Find safe channels (reachable cells avoiding rocks)")
    print("P2: Optimize path within safe channels (A*)\n")
    
    maze_display, success, path_len, safe_cells = run_simulation()
    
    print("Maze (20x20):")
    print("S = Start, G = Goal, # = Rock, . = Safe/Path")
    print_maze(maze_display)
    
    print("Results:")
    print(f"  Success (reached goal): {success}")
    print(f"  Path length: {path_len}")
    print(f"  Safe cells found (deadband size): {safe_cells}")
    print(f"  Rocks: {ROCK_COUNT}")
    
    # Compare with fleet data
    print("\nComparison with fleet-knowledge data:")
    print("  Constrained success (P2-only): 0/50 (0%)")
    print("  Deadband success (P0+P1+P2): 50/50 (100%)")
    print("  Optimal path length: 21 (Manhattan distance)")
    print(f"  This run path length: {path_len}")

if __name__ == "__main__":
    main()
```

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 35)
Demonstrates P0→P1→P2 on a 20×20 maze with rocks.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# Constants from simulation data
MAZE_SIZE = 20
ROCK_COUNT = 64          # P0: negative space
OPTIMAL_PATH_LEN = 21    # P2: best possible path

class DeadbandMaze:
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.best_path: List[Tuple[int, int]] = []
        
    def p0_map_negative_space(self):
        """P0: Identify rocks (what NOT to do)"""
        # Place rocks randomly (but not on start/goal)
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[x][y] = '█'
        
        print("P0: Mapped negative space")
        print(f"  - Rocks identified: {len(self.rocks)}")
        print(f"  - Start: {self.start}, Goal: {self.goal}")
    
    def p1_find_safe_channels(self):
        """P1: Find paths that avoid all rocks"""
        # Simple BFS to find all rock-avoiding paths (max 3 for demo)
        from collections import deque
        
        paths = []
        queue = deque([([self.start], set([self.start]))])
        
        while queue and len(paths) < 3:
            path, visited = queue.popleft()
            x, y = path[-1]
            
            if (x, y) == self.goal:
                paths.append(path)
                continue
            
            # Try neighbors (right, down, left, up)
            for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        new_path = path + [(nx, ny)]
                        new_visited = visited.copy()
                        new_visited.add((nx, ny))
                        queue.append((new_path, new_visited))
        
        self.safe_channels = paths
        print(f"\nP1: Found {len(self.safe_channels)} safe channels")
        for i, channel in enumerate(self.safe_channels):
            print(f"  Channel {i+1}: {len(channel)-1} steps")
    
    def p2_optimize_within_channels(self):
        """P2: Pick shortest path from safe channels"""
        if not self.safe_channels:
            print("ERROR: Cannot optimize (P2) without safe channels (P1)")
            return
        
        self.best_path = min(self.safe_channels, key=len)
        print(f"\nP2: Optimized within safe channels")
        print(f"  Selected path: {len(self.best_path)-1} steps")
        print(f"  Optimal possible: {OPTIMAL_PATH_LEN} steps")
        
        # Mark path on grid
        for x, y in self.best_path[1:-1]:  # Skip start and goal
            if self.grid[x][y] == '.':
                self.grid[x][y] = '·'
    
    def run_protocol(self):
        """Execute full P0→P1→P2 sequence"""
        print("=" * 50)
        print("DEADBAND PROTOCOL DEMO")
        print("=" * 50)
        
        self.p0_map_negative_space()
        self.p1_find_safe_channels()
        self.p2_optimize_within_channels()
        
        print("\n" + "=" * 50)
        print("FINAL MAZE (ASCII Visualization)")
        print("=" * 50)
        self.display_maze()
        
        # Protocol validation
        print("\n" + "=" * 50)
        print("PROTOCOL VALIDATION")
        print("=" * 50)
        if not self.rocks:
            print("❌ P0 FAILED: No rocks mapped")
        elif not self.safe_channels:
            print("❌ P1 FAILED: No safe channels found")
        elif not self.best_path:
            print("❌ P2 FAILED: No path selected")
        else:
            # Check if path hits any rocks
            hits_rock = any(pos in self.rocks for pos in self.best_path)
            if hits_rock:
                print("❌ PROTOCOL VIOLATION: Path hits rocks!")
            else:
                print("✅ Protocol followed correctly")
                print(f"   Path length: {len(self.best_path)-1} steps")
                print(f"   Rocks avoided: {len(self.rocks)}")
    
    def display_maze(self):
        """ASCII visualization"""
        # Create display grid
        display = [row.copy() for row in self.grid]
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        display[sx][sy] = 'S'
        display[gx][gy] = 'G'
        
        # Print with borders
        print("   " + "".join(f"{i:2}" for i in range(MAZE_SIZE)))
        for i in range(MAZE_SIZE):
            row_str = "".join(display[i])
            print(f"{i:2} {row_str}")
        
        print("\nLegend:")
        print("  S = Start, G = Goal")
        print("  █ = Rock (P0 negative space)")
        print("  · = Safe path (P2 optimized)")
        print("  . = Open space")

def main():
    """Run the demo"""
    print("Fleet: Cocapn | Agent: Tinker | Cycle: 35")
    print("Task: Deadband Protocol Maze Demo")
    print("Based on constraint_sim_final.json data\n")
    
    # Run multiple trials to show protocol effectiveness
    successes = 0
    trials = 5
    
    for trial in range(trials):
        print(f"\n{'='*60}")
        print(f"TRIAL {trial+1}/{trials}")
        print(f"{'='*60}")
        
        maze = DeadbandMaze()
        maze.run_protocol()
        
        # Check if protocol was successful
        if maze.best_path and not any(pos in maze.rocks for pos in maze.best_path):
            successes += 1
    
    print(f"\n{'='*60}")
    print("SUMMARY")
    print(f"{'='*60}")
    print(f"Trials: {trials}")
    print(f"Successful protocols: {successes}")
    print(f"Success rate: {successes/trials*100:.0f}%")
    print(f"\nReference from fleet data:")
    print(f"  Deadband success: 50/50 (100%)")
    print(f"  Unconstrained: 33/50 (66%)")
    print(f"  Constrained (P2-only): 0/50 (0%)")

if __name__ == "__main__":
    main()
```

**Key Implementation Notes:**

1. **P0 Implementation**: Randomly places 64 rocks (from simulation data) on 20×20 grid, avoiding start/goal.

2. **P1 Implementation**: Uses BFS to find up to 3 rock-avoiding paths from start to goal.

3. **P2 Implementation**: Selects shortest path from P1's safe channels.

4. **Protocol Validation**: Checks that each step was completed and that final path doesn't hit rocks.

5. **Visualization**: ASCII grid showing rocks (█), safe path (·), start (S), goal (G).

6. **Multiple Trials**: Runs 5 trials to demonstrate protocol reliability.

**Factual Basis:**
- Maze size 20×20 and rock count 64 from `constraint_sim_final.json`
- Protocol steps from `DEADBAND-PROTOCOL.md`

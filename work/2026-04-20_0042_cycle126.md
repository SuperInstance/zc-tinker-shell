#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Simulates three agent strategies in a 20x20 maze.
"""

import random
import time
from typing import List, Tuple, Set

# Constants from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64  # 64 rocks = 16% obstacle density
OPTIMAL_PATH_LENGTH = 21

# Maze symbols
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
START = 'S'
GOAL = 'G'
PATH = '○'
DEADBAND = '░'  # Safe channel visualization

class Maze:
    def __init__(self):
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.generate_maze()
        
    def generate_maze(self):
        """Generate random rocks, ensuring start and goal are clear."""
        self.grid[self.start[0]][self.start[1]] = START
        self.grid[self.goal[0]][self.goal[1]] = GOAL
        
        # Place rocks randomly
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[x][y] = ROCK
    
    def is_rock(self, pos: Tuple[int, int]) -> bool:
        return pos in self.rocks
    
    def is_valid(self, pos: Tuple[int, int]) -> bool:
        x, y = pos
        return 0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE
    
    def print_maze(self, agent_pos=None, path=None, deadband=None):
        """Print maze with optional agent position, path, and deadband overlay."""
        for x in range(MAZE_SIZE):
            row = []
            for y in range(MAZE_SIZE):
                if agent_pos and (x, y) == agent_pos:
                    row.append(AGENT)
                elif path and (x, y) in path:
                    row.append(PATH)
                elif deadband and (x, y) in deadband:
                    row.append(DEADBAND)
                else:
                    row.append(self.grid[x][y])
            print(' '.join(row))
        print()

class Agent:
    def __init__(self, name: str, maze: Maze):
        self.name = name
        self.maze = maze
        self.pos = maze.start
        self.path = [maze.start]
        self.steps = 0
        self.success = False
        
    def move_toward_goal(self) -> bool:
        """Greedy move toward goal (P2-only logic)."""
        if self.pos == self.maze.goal:
            return True
            
        x, y = self.pos
        gx, gy = self.maze.goal
        
        # Determine move direction (greedy toward goal)
        dx = 1 if gx > x else -1 if gx < x else 0
        dy = 1 if gy > y else -1 if gy < y else 0
        
        # Try primary direction
        new_pos = (x + dx, y) if dx != 0 else (x, y + dy)
        if self.maze.is_valid(new_pos) and not self.maze.is_rock(new_pos):
            self.pos = new_pos
        else:
            # Try orthogonal direction
            if dx != 0:
                new_pos = (x, y + 1) if y < MAZE_SIZE-1 else (x, y - 1)
            else:
                new_pos = (x + 1, y) if x < MAZE_SIZE-1 else (x - 1, y)
            if self.maze.is_valid(new_pos) and not self.maze.is_rock(new_pos):
                self.pos = new_pos
            else:
                # Stuck
                return False
                
        self.path.append(self.pos)
        self.steps += 1
        return self.pos == self.maze.goal

class UnconstrainedAgent(Agent):
    """P2-only agent: optimizes for goal without mapping rocks."""
    def run(self, max_steps=100):
        print(f"\n{self.name} (P2-only) starting...")
        self.maze.print_maze(agent_pos=self.pos)
        
        for _ in range(max_steps):
            if self.move_toward_goal():
                self.success = True
                break
            if self.steps >= max_steps:
                break
                
        print(f"Result: {'SUCCESS' if self.success else 'FAILURE'} in {self.steps} steps")
        self.maze.print_maze(path=self.path)
        return self.success

## File: `deadband_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path) priority.
"""

import random
import sys
import time

class Maze:
    def __init__(self, width=20, height=20, rock_density=0.15):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self._generate_rocks(rock_density)
        self._place_start_goal()
    
    def _generate_rocks(self, density):
        """P0: Map the negative space (rocks)."""
        for y in range(self.height):
            for x in range(self.width):
                if random.random() < density:
                    self.grid[y][x] = '#'
                    self.rocks.add((x, y))
    
    def _place_start_goal(self):
        """Ensure start and goal are clear."""
        sx, sy = self.start
        gx, gy = self.goal
        self.grid[sy][sx] = 'S'
        self.grid[gy][gx] = 'G'
        if (sx, sy) in self.rocks:
            self.rocks.remove((sx, sy))
        if (gx, gy) in self.rocks:
            self.rocks.remove((gx, gy))
    
    def display(self, agent_pos=None, path=None):
        """Print the maze with optional agent and path."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if agent_pos and (x, y) == agent_pos:
                    row.append('A')
                elif path and (x, y) in path:
                    row.append('*')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
        print()

class Agent:
    def __init__(self, name, maze):
        self.name = name
        self.maze = maze
        self.pos = maze.start
        self.path = [maze.start]
        self.success = False
    
    def manhattan(self, a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])

class GreedyAgent(Agent):
    """P2-only agent: optimizes without mapping rocks."""
    def run(self):
        x, y = self.pos
        gx, gy = self.maze.goal
        
        for _ in range(100):  # limit steps
            if (x, y) == (gx, gy):
                self.success = True
                return
            
            # P2: greedy move toward goal
            moves = []
            if x < gx: moves.append((x+1, y))
            if x > gx: moves.append((x-1, y))
            if y < gy: moves.append((x, y+1))
            if y > gy: moves.append((x, y-1))
            
            if not moves:
                break
            
            # Choose move with smallest Manhattan distance to goal
            best = min(moves, key=lambda p: self.manhattan(p, (gx, gy)))
            
            # Check if it's a rock (P0 violation)
            if best in self.maze.rocks:
                self.success = False
                return
            
            x, y = best
            self.pos = (x, y)
            self.path.append((x, y))
        
        self.success = False

class DeadbandAgent(Agent):
    """P0→P1→P2 agent: follows Deadband Protocol."""
    def run(self):
        # P0: Map rocks (already in maze.rocks)
        rocks = self.maze.rocks
        
        # P1: Find safe channels (BFS avoiding rocks)
        start = self.maze.start
        goal = self.maze.goal
        queue = [(start, [start])]
        visited = set([start])
        safe_path = None
        
        while queue:
            (x, y), path = queue.pop(0)
            if (x, y) == goal:
                safe_path = path
                break
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.maze.width and 0 <= ny < self.maze.height:
                    if (nx, ny) not in visited and (nx, ny) not in rocks:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        if not safe_path:
            self.success = False
            return
        
        # P2: Optimize within safe channel (already shortest via BFS)
        self.path = safe_path
        self.pos = goal
        self.success = True

def main():
    print("=" * 60)
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("Maze: 20x20, S=Start, G=Goal, #=Rock, A=Agent, *=Path")
    print("=" * 60)
    print()
    
    # Create maze
    random.seed(42)  # reproducible for demo
    maze = Maze(20, 20, 0.15)
    
    print("MAZE (P0: Rocks Mapped):")
    maze.display()
    time.sleep(1)
    
    # Run greedy agent (P2-only)
    print("GREEDY AGENT (P2-only):")
    greedy = GreedyAgent("Greedy", maze)
    greedy.run()
    maze.display(agent_pos=greedy.pos, path=greedy.path)
    print(f"Result: {'SUCCESS' if greedy.success else 'FAILED (hit rock)'}")
    print()
    time.sleep(2)
    
    # Run deadband agent (P0→P1→P2)
    print("DEADBAND AGENT (P0→P1→P2):")
    deadband = DeadbandAgent("Deadband", maze)
    deadband.run()
    maze.display(agent_pos=deadband.pos, path=deadband.path)
    print(f"Result: {'SUCCESS' if deadband.success else 'FAILED'}")
    print()
    
    # Summary
    print("=" * 60)
    print("SUMMARY:")
    print(f"  Greedy (P2-only):   {greedy.success} | Steps: {len(greedy.path)}")
    print(f"  Deadband (P0→P1→P2): {deadband.success} | Steps: {len(deadband.path)}")
    print()
    print("Protocol violation: Greedy agent optimized (P2) before")
    print("mapping rocks (P0), resulting in rock collision.")
    print("=" * 60)

if __name__ == "__main__":
    main()
```

class Maze:
    def __init__(self, width=20, height=20, rock_density=0.15):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (height-1, width-1)
        self.rocks = set()
        
        # Place rocks (P0 negative space)
        for y in range(height):
            for x in range(width):
                if (y, x) != self.start and (y, x) != self.goal:
                    if random.random() < rock_density:
                        self.grid[y][x] = '#'
                        self.rocks.add((y, x))
        
        # Ensure at least one path exists
        self._ensure_path()
    
    def _ensure_path(self):
        """Guarantee a solvable maze by carving a random walk path."""
        y, x = self.start
        path = [(y, x)]
        while (y, x) != self.goal:
            if y < self.goal[0] and random.random() < 0.6:
                y += 1
            elif x < self.goal[1]:
                x += 1
            else:
                y += 1
            if (y, x) != self.goal:
                self.grid[y][x] = '.'
                if (y, x) in self.rocks:
                    self.rocks.remove((y, x))
            path.append((y, x))
    
    def display(self, path=None, agent_type="Deadband"):
        """Print ASCII maze with optional path overlay."""
        print(f"\n=== {agent_type} Agent ===")
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (y, x) == self.start:
                    row.append('S')
                elif (y, x) == self.goal:
                    row.append('G')
                elif path and (y, x) in path:
                    row.append('*')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
        print()

class DeadbandAgent:
    def __init__(self, maze):
        self.maze = maze
        self.rocks = maze.rocks  # P0: Map negative space
    
    def p0_map_rocks(self):
        """P0: Identify all rock locations (failure states)."""
        return self.rocks
    
    def p1_find_channels(self):
        """P1: Find all safe cells (non-rock cells)."""
        safe = set()
        for y in range(self.maze.height):
            for x in range(self.maze.width):
                if (y, x) not in self.rocks:
                    safe.add((y, x))
        return safe
    
    def p2_optimize_path(self, safe_cells):
        """P2: BFS to find shortest path through safe cells only."""
        start = self.maze.start
        goal = self.maze.goal
        
        queue = deque([(start, [start])])
        visited = set([start])
        
        while queue:
            (y, x), path = queue.popleft()
            
            if (y, x) == goal:
                return path
            
            for dy, dx in [(1,0), (-1,0), (0,1), (0,-1)]:
                ny, nx = y + dy, x + dx
                if 0 <= ny < self.maze.height and 0 <= nx < self.maze.width:
                    if (ny, nx) in safe_cells and (ny, nx) not in visited:
                        visited.add((ny, nx))
                        queue.append(((ny, nx), path + [(ny, nx)]))
        
        return None  # No safe path

class UnconstrainedAgent:
    """P2-only agent: greedy optimization ignoring rocks."""
    def __init__(self, maze):
        self.maze = maze
    
    def find_path(self):
        """Greedy Manhattan distance path (hits rocks)."""
        path = [self.maze.start]
        y, x = self.maze.start
        
        while (y, x) != self.maze.goal:
            # Greedy: move toward goal
            if y < self.maze.goal[0]:
                y += 1
            elif x < self.maze.goal[1]:
                x += 1
            path.append((y, x))
            
            # Check for rock collision
            if (y, x) in self.maze.rocks:
                return path, False  # Failed
        
        return path, True

class ConstrainedAgent:
    """P0-only agent: knows rocks but doesn't find channels."""
    def __init__(self, maze):
        self.maze = maze
        self.rocks = maze.rocks
    
    def find_path(self):
        """Random walk avoiding rocks (no channel optimization)."""
        path = [self.maze.start]
        y, x = self.maze.start
        visited = set([(y, x)])
        
        for _ in range(100):  # Limit steps
            if (y, x) == self.maze.goal:
                return path, True
            
            # Random move avoiding rocks
            moves = []
            for dy, dx in [(1,0), (-1,0), (0,1), (0,-1)]:

                    if (ny, nx) not in self.rocks and (ny, nx) not in visited:
                        moves.append((ny, nx))
            
            if not moves:
                return path, False
            
            y, x = random.choice(moves)
            path.append((y, x))
            visited.add((y, x))

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

A visual ASCII simulation of the Deadband Protocol (P0→P1→P2).
Demonstrates the strict priority order:
  P0: Map negative space (rocks)
  P1: Identify safe channels (deadbands)
  P2: Optimize within safe channels (shortest path)

Based on fleet-knowledge/narrow-games/constraint_sim_final.json
and DEADBAND-PROTOCOL.md.
"""

import random
from typing import List, Tuple, Set

# ====================
# MAZE CONFIGURATION
# ====================
MAZE_WIDTH = 20
MAZE_HEIGHT = 20
START = (0, 0)
GOAL = (19, 19)

# Rock density (percentage of cells that are obstacles)
ROCK_DENSITY = 0.16  # ~64 rocks in 20x20 grid

# ====================
# AGENT IMPLEMENTATION
# ====================

class DeadbandAgent:
    """An agent that follows the Deadband Protocol."""
    
    def __init__(self, maze: List[List[str]]):
        self.maze = maze
        self.visited = set()
        self.rocks = set()      # P0: negative space
        self.safe_channels = [] # P1: safe paths
        self.optimal_path = []  # P2: best path within safe channels
        
    def p0_map_negative_space(self) -> None:
        """P0: Identify all rocks (failure states)."""
        for y in range(MAZE_HEIGHT):
            for x in range(MAZE_WIDTH):
                if self.maze[y][x] == '█':
                    self.rocks.add((x, y))
        print(f"[P0] Mapped {len(self.rocks)} rocks (negative space)")
        
    def p1_find_safe_channels(self) -> None:
        """P1: Find all paths that avoid rocks entirely."""
        # Simple BFS to find all reachable safe cells
        queue = [START]
        visited = {START}
        parent = {START: None}
        
        while queue:
            x, y = queue.pop(0)
            
            # Check all four directions
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                
                # Bounds check
                if not (0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT):
                    continue
                    
                # Rock check (P1 constraint: avoid all rocks)
                if self.maze[ny][nx] == '█':
                    continue
                    
                # Already visited
                if (nx, ny) in visited:
                    continue
                    
                visited.add((nx, ny))
                parent[(nx, ny)] = (x, y)
                queue.append((nx, ny))
        
        # Reconstruct all safe paths to goal
        if GOAL in visited:
            # Build path from goal to start
            path = []
            current = GOAL
            while current is not None:
                path.append(current)
                current = parent[current]
            path.reverse()
            self.safe_channels = [path]
            print(f"[P1] Found safe channel with {len(path)} steps")
        else:
            print("[P1] No safe channel to goal exists")
            self.safe_channels = []
            
    def p2_optimize_within_channel(self) -> None:
        """P2: Find shortest path within safe channels."""
        if not self.safe_channels:
            print("[P2] No safe channels to optimize within")
            return
            
        # For this demo, we only have one safe channel
        # In a real scenario, we'd choose the shortest among multiple channels
        channel = self.safe_channels[0]
        
        # A* would be used here, but for simplicity we use the BFS path
        # (which is already optimal for unweighted grid)
        self.optimal_path = channel
        print(f"[P2] Optimal path: {len(self.optimal_path)} steps")
        
    def run_protocol(self) -> bool:
        """Execute full P0→P1→P2 protocol."""
        print("\n" + "="*50)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*50)
        
        # P0: Map negative space (mandatory first step)
        self.p0_map_negative_space()
        if not self.rocks:
            print("[WARNING] No rocks found - protocol still valid")
            
        # P1: Find safe channels (mandatory second step)
        self.p1_find_safe_channels()
        if not self.safe_channels:
            print("[FAILURE] No safe channels found. Protocol stops at P1.")
            return False
            
        # P2: Optimize within channels (only after P0+P1)
        self.p2_optimize_within_channel()
        
        return True

# ====================
# MAZE GENERATION
# ====================

def generate_maze() -> List[List[str]]:
    """Generate a random maze with rocks."""
    maze = [['·' for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    
    # Place rocks randomly
    num_rocks = int(MAZE_WIDTH * MAZE_HEIGHT * ROCK_DENSITY)
    rocks_placed = 0
    
    while rocks_placed < num_rocks:
        x = random.randint(0, MAZE_WIDTH - 1)
        y = random.randint(0, MAZE_HEIGHT - 1)
        
        # Don't place rocks on start or goal
        if (x, y) == START or (x, y) == GOAL:
            continue
            
        if maze[y][x] == '·':
            maze[y][x] = '█'
            rocks_placed += 1
    
    # Mark start and goal
    sx, sy = START
    gx, gy = GOAL
    maze[sy][sx] = 'S'
    maze[gy][gx] = 'G'
    
    return maze

def print_maze(maze: List[List[str]], path: List[Tuple[int, int]] = None) -> None:
    """Print the maze with optional path overlay."""
    for y in range(MAZE_HEIGHT):
        row = []
        for x in range(MAZE_WIDTH):
            cell = maze[y][x]
            
            # Overlay path if provided
            if path and (x, y) in path:
                if cell == 'S':
                    row.append('S')
                elif cell == 'G':
                    row.append('G')
                else:
                    row.append('●')  # Path marker
            else:
                row.append(cell)
        print(''.join(row))

# ====================
# DEMONSTRATION
# ====================

def demonstrate_greedy_failure(maze: List[List[str]]) -> None:
    """Show what happens when you skip to P2 (greedy optimization)."""
    print("\n" + "="*50)
    print("GREEDY AGENT (SKIPS P0+P1, GOES STRAIGHT TO P2)")
    print("="*50)
    
    # Simple greedy agent that goes right/down toward goal
    x, y = START
    path = [START]
    
    while (x, y) != GOAL:
        # Greedy: move toward goal
        if x < GOAL[0]:
            x += 1
        elif y < GOAL[1]:
            y += 1
            
        # Check for rock
        if maze[y][x] == '█':
            print(f"  Greedy agent hit rock at ({x}, {y})!")
            break
            
        path.append((x, y))
    
    if (x, y) == GOAL:
        print(f"  Greedy reached goal in {len(path)} steps (lucky!)")
    else:
        print(f"  Greedy failed after {len(path)} steps")
    
    print_maze(maze, path)

def main():
    """Run the complete demonstration."""
    # Generate maze
    maze = generate_maze()
    
    print("GENERATED MAZE (20×20)")
    print("  S = Start, G = Goal, █ = Rock, · = Empty")
    print_maze(maze)
    
    # Demonstrate greedy failure (violates protocol)
    demonstrate_greedy_failure(maze)

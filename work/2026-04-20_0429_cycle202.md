#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Tinker - Cocapn Fleet
Demonstrates P0→P1→P2 priority in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set, Optional
from enum import Enum

class Cell(Enum):
    EMPTY = '·'
    ROCK = '█'
    AGENT = 'A'
    GOAL = 'G'
    PATH = '○'
    DEADBAND = '░'

class AgentType(Enum):
    UNCONSTRAINED = "Unconstrained (P2-only)"
    CONSTRAINED = "Constrained (P0+P2)"
    DEADBAND = "Deadband (P0+P1+P2)"

class Maze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.grid = [[Cell.EMPTY for _ in range(width)] for _ in range(height)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Place rocks (P0 negative space)
        num_rocks = int(width * height * rock_density)
        for _ in range(num_rocks):
            x, y = random.randint(0, width-1), random.randint(0, height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.grid[y][x] = Cell.ROCK
                self.rocks.add((x, y))
        
        # Ensure start and goal are clear
        self.grid[self.start[1]][self.start[0]] = Cell.EMPTY
        self.grid[self.goal[1]][self.goal[0]] = Cell.GOAL
    
    def is_valid(self, x: int, y: int) -> bool:
        return 0 <= x < self.width and 0 <= y < self.height
    
    def is_rock(self, x: int, y: int) -> bool:
        return (x, y) in self.rocks
    
    def get_neighbors(self, x: int, y: int) -> List[Tuple[int, int]]:
        neighbors = []
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if self.is_valid(nx, ny):
                neighbors.append((nx, ny))
        return neighbors
    
    def display(self, agent_pos: Optional[Tuple[int, int]] = None, 
                deadband: Set[Tuple[int, int]] = None,
                path: List[Tuple[int, int]] = None):
        """Render maze with optional agent, deadband, and path."""
        display_grid = [row.copy() for row in self.grid]
        
        if deadband:
            for (x, y) in deadband:
                if display_grid[y][x] == Cell.EMPTY:
                    display_grid[y][x] = Cell.DEADBAND
        
        if path:
            for (x, y) in path[1:-1]:  # Skip start and goal
                if display_grid[y][x] == Cell.EMPTY:
                    display_grid[y][x] = Cell.PATH
        
        if agent_pos:
            x, y = agent_pos
            display_grid[y][x] = Cell.AGENT
        
        # Print
        print("+" + "-" * self.width + "+")
        for row in display_grid:
            print("|" + "".join(cell.value for cell in row) + "|")
        print("+" + "-" * self.width + "+")

class Agent:
    def __init__(self, maze: Maze, agent_type: AgentType):
        self.maze = maze
        self.type = agent_type
        self.pos = maze.start
        self.path = [maze.start]
        self.deadband: Set[Tuple[int, int]] = set()
        self.visited: Set[Tuple[int, int]] = {maze.start}
        
    def reset(self):
        self.pos = self.maze.start
        self.path = [self.maze.start]
        self.deadband = set()
        self.visited = {self.maze.start}
    
    def run(self) -> Tuple[bool, int]:
        """Run agent until goal or failure. Returns (success, steps)."""
        self.reset()
        steps = 0
        max_steps = self.maze.width * self.maze.height * 2
        
        # P0: Map negative space (rocks)
        rocks = self.maze.rocks.copy()
        
        # P1: Map safe channels (deadband) - only for deadband agent
        if self.type == AgentType.DEADBAND:
            self._map_deadband(rocks)
        
        while self.pos != self.maze.goal and steps < max_steps:
            # P2: Optimize within constraints
            next_pos = self._choose_next(rocks)
            if next_pos is None:
                return False, steps  # Stuck
            
            self.pos = next_pos
            self.path.append(next_pos)
            self.visited.add(next_pos)
            steps += 1
            
            if self.pos == self.maze.goal:
                return True, steps
        
        return False, steps
    
    def _map_deadband(self, rocks: Set[Tuple[int, int]]):
        """P1: Find all cells reachable from start without hitting rocks."""
        frontier = [self.maze.start]
        self.deadband = {self.maze.start}
        
        while frontier:
            x, y = frontier.pop()
            for nx, ny in self.maze.get_neighbors(x, y):
                if (nx, ny) not in rocks and (nx, ny) not in self.deadband:
                    self.deadband.add((nx, ny))
                    frontier.append((nx, ny))
    
    def _choose_next(self, rocks: Set[Tuple[int, int]]) -> Optional[Tuple[int, int]]:
        """P2: Choose next move based on agent type."""
        neighbors = self.maze.get_neighbors(*self.pos)
        
        # Filter based on agent type
        if self.type == AgentType.UNCONSTRAINED:
            # P2-only: ignore rocks completely
            valid = neighbors
        elif self.type == AgentType.CONSTRAINED:
            # P0+P2: avoid rocks but no channel mapping
            valid = [n for n in neighbors if n not in rocks]
        else:  # DEADBAND
            # P0+P1+P2: must stay in deadband
            valid = [n for n in neighbors if n in self.deadband]
        
        if not valid:
            return None
        
        # Greedy: choose neighbor closest to goal
        goal = self.maze.goal
        return min(valid, key=lambda p: abs(p[0]-goal[0]) + abs(p[1]-goal[1]))

            if success:
                results[agent_type]['success'] += 1

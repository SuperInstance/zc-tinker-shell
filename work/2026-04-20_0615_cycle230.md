#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Tinker - Cocapn Fleet - Cycle 230

Simulates the Deadband Protocol (P0→P1→P2) in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json:
- optimal: 21 (shortest safe path)
- rocks: 64 (failure states)
- deadband_success: 50/50 runs

Demonstrates:
P0: Map negative space (rocks)
P1: Identify safe channels (deadbands)
P2: Optimize path within safe channels
"""

import random
from collections import deque

# Simulation parameters from fleet data
GRID_SIZE = 20
NUM_ROCKS = 64
OPTIMAL_PATH_LEN = 21

# Symbols for visualization
ROCK = '█'
SAFE = '·'
AGENT = 'A'
GOAL = 'G'
PATH = '○'
DEADBAND = '░'  # Safe channel visualization

class DeadbandMaze:
    def __init__(self, seed=42):
        random.seed(seed)
        self.grid = [[SAFE for _ in range(GRID_SIZE)] for _ in range(GRID_SIZE)]
        self.rocks = set()
        self.start = (0, 0)
        self.goal = (GRID_SIZE-1, GRID_SIZE-1)
        
        # P0: Map negative space (rocks)
        self._place_rocks()
        
        # Ensure start and goal are safe
        self.grid[self.start[0]][self.start[1]] = AGENT
        self.grid[self.goal[0]][self.goal[1]] = GOAL
        
        # P1: Precompute safe channels (deadbands)
        self.safe_channels = self._find_safe_channels()
        
    def _place_rocks(self):
        """P0: Identify failure states (rocks)"""
        placed = 0
        while placed < NUM_ROCKS:
            x = random.randint(0, GRID_SIZE-1)
            y = random.randint(0, GRID_SIZE-1)
            if (x, y) == self.start or (x, y) == self.goal:
                continue
            if (x, y) not in self.rocks:
                self.rocks.add((x, y))
                self.grid[x][y] = ROCK
                placed += 1
    
    def _find_safe_channels(self):
        """P1: Find all safe paths from start to goal"""
        # BFS to find all reachable safe cells
        visited = set()
        queue = deque([self.start])
        reachable = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            reachable.add((x, y))
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < GRID_SIZE and 0 <= ny < GRID_SIZE:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        return reachable
    
    def greedy_path(self):
        """Unconstrained optimization (P2-only) - hits rocks"""
        path = []
        x, y = self.start
        visited = set()
        
        while (x, y) != self.goal:
            path.append((x, y))
            visited.add((x, y))
            
            # Greedy: move toward goal
            moves = []
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < GRID_SIZE and 0 <= ny < GRID_SIZE:
                    dist = abs(nx - self.goal[0]) + abs(ny - self.goal[1])
                    moves.append((dist, nx, ny))
            
            moves.sort()  # Closest to goal first
            
            moved = False
            for _, nx, ny in moves:
                if (nx, ny) not in visited:
                    # Check if we hit a rock (P0 violation)
                    if (nx, ny) in self.rocks:
                        path.append((nx, ny))
                        return path, False  # Failed - hit rock
                    x, y = nx, ny
                    moved = True
                    break
            
            if not moved:
                return path, False  # Stuck
        
        path.append(self.goal)
        return path, True
    
    def deadband_path(self):
        """P0→P1→P2: Constrained optimization within safe channels"""
        # P0 already done (rocks mapped)
        # P1 already done (safe_channels computed)
        # P2: BFS within safe channels only
        
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                return path, True
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < GRID_SIZE and 0 <= ny < GRID_SIZE:
                    # P1 constraint: only move within safe channels
                    if (nx, ny) in self.safe_channels and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        return [], False  # No path found
    
    def visualize(self, path=None, show_channels=False):
        """Create ASCII visualization"""
        vis_grid = [row[:] for row in self.grid]
        
        if show_channels:
            # Highlight safe channels (P1)
            for x in range(GRID_SIZE):
                for y in range(GRID_SIZE):
                    if (x, y) in self.safe_channels and vis_grid[x][y] == SAFE:
                        vis_grid[x][y] = DEADBAND
        
        if path:
            # Show path (P2 result)
            for x, y in path[1:-1]:  # Skip start and goal
                if vis_grid[x][y] in [SAFE, DEADBAND]:
                    vis_grid[x][y] = PATH
        
        # Build output
        lines = []
        for x in range(GRID_SIZE):
            line = []
            for y in range(GRID_SIZE):
                line.append(vis_grid[x][y])
            lines.append(' '.join(line))
        
        return '\n'.join(lines)

import random
import time
from typing import List, Tuple, Set

class Maze:
    """20x20 maze with rocks, start, and goal."""
    
    def __init__(self, seed=42):
        self.width = 20
        self.height = 20
        self.rocks = 64  # From constraint_sim_final.json
        self.start = (0, 0)
        self.goal = (19, 19)
        
        random.seed(seed)
        self.grid = [['.' for _ in range(self.width)] for _ in range(self.height)]
        
        # Place rocks (avoid start and goal)
        rock_positions = set()
        while len(rock_positions) < self.rocks:
            x = random.randint(0, self.width-1)
            y = random.randint(0, self.height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                rock_positions.add((x, y))
        
        for x, y in rock_positions:
            self.grid[y][x] = '#'
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        self.grid[sy][sx] = 'S'
        self.grid[gy][gx] = 'G'
    
    def display(self, agent_pos=None, path=None, deadband=None):
        """Display maze with optional agent, path, and deadband visualization."""
        display_grid = [row[:] for row in self.grid]
        
        # Show deadband (safe channels) as '~' if provided
        if deadband:
            for x, y in deadband:
                if display_grid[y][x] == '.':
                    display_grid[y][x] = '~'
        
        # Show path as '*' if provided
        if path:
            for x, y in path:
                if display_grid[y][x] == '.' or display_grid[y][x] == '~':
                    display_grid[y][x] = '*'
        
        # Show agent as 'A' if provided
        if agent_pos:
            ax, ay = agent_pos
            if display_grid[ay][ax] != 'S' and display_grid[ay][ax] != 'G':
                display_grid[ay][ax] = 'A'
        
        print("\n" + "=" * 50)
        for row in display_grid:
            print(' '.join(row))
        print("=" * 50)
        print("Legend: S=Start, G=Goal, #=Rock, A=Agent, *=Path, ~=Safe Channel")
    
    def is_rock(self, pos):
        x, y = pos
        return self.grid[y][x] == '#'
    
    def is_valid(self, pos):
        x, y = pos
        return 0 <= x < self.width and 0 <= y < self.height

class DeadbandAgent:
    """Agent that follows P0→P1→P2 protocol."""
    
    def __init__(self, maze):
        self.maze = maze
        self.position = maze.start
        self.visited = set()
        self.visited.add(self.position)
        
        # Protocol state
        self.rocks_mapped = set()  # P0: Negative space
        self.safe_channels = set()  # P1: Safe channels
        self.optimal_path = []     # P2: Optimal path within channels
    
    def p0_map_negative_space(self):
        """P0: Identify all rocks (what NOT to do)."""
        print("\n=== P0: MAP NEGATIVE SPACE ===")
        print("Identifying all rock positions (failure states)...")
        
        # In a real scenario, this might involve exploration or prior knowledge
        # For demo, we'll "discover" rocks by scanning adjacent cells
        for y in range(self.maze.height):
            for x in range(self.maze.width):
                if self.maze.is_rock((x, y)):
                    self.rocks_mapped.add((x, y))
        
        print(f"Mapped {len(self.rocks_mapped)} rocks (failure states)")
        print("Rocks are absolute constraints - agent must avoid them")
        return self.rocks_mapped
    
    def p1_find_safe_channels(self):
        """P1: Find safe channels (where you CAN be)."""
        print("\n=== P1: FIND SAFE CHANNELS ===")
        print("Finding all paths from start to goal that avoid rocks...")
        
        # Simple BFS to find all reachable positions that avoid rocks
        queue = [self.maze.start]
        visited = set([self.maze.start])
        
        while queue:
            x, y = queue.pop(0)
            
            # Check all four directions
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                new_pos = (nx, ny)
                
                if (self.maze.is_valid(new_pos) and 
                    new_pos not in visited and 
                    not self.maze.is_rock(new_pos)):
                    
                    visited.add(new_pos)
                    queue.append(new_pos)
        
        self.safe_channels = visited
        print(f"Found {len(self.safe_channels)} safe positions (deadband)")
        
        # Check if goal is reachable
        if self.maze.goal in self.safe_channels:
            print("✓ Goal is reachable through safe channels")
        else:
            print("✗ Goal is NOT reachable - trapped by rocks!")
        
        return self.safe_channels
    
    def p2_optimize_within_channels(self):
        """P2: Find optimal path within safe channels."""
        print("\n=== P2: OPTIMIZE WITHIN CHANNELS ===")
        print("Finding shortest path from start to goal using only safe channels...")
        
        # BFS for shortest path constrained to safe channels
        queue = [(self.maze.start, [self.maze.start])]
        visited = set([self.maze.start])
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.maze.goal:
                self.optimal_path = path
                print(f"Found optimal path of length {len(path)-1} steps")
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:

                new_pos = (nx, ny)
                
                if (self.maze.is_valid(new_pos) and 
                    new_pos in self.safe_channels and 
                    new_pos not in visited):
                    
                    visited.add(new_pos)
                    queue.append((new_pos, path + [new_pos]))
        
        print("✗ No path found within safe channels")

                self.maze.display(agent_pos=pos, path=path[:i+1])
                time.sleep(0.1)
            
            print(f"\n✓ SUCCESS: Reached goal in {len(path)-1} steps")
            print("Protocol completed: P0→P1→P2")
        else:
            print("\n✗ FAILURE: No path found")
            print("This demonstrates why P0+P1 must come before P2")

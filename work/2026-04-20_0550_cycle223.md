```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 strict priority in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# Simulation parameters from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64  # rocks = failure states
OPTIMAL_PATH_LENGTH = 21

class DeadbandAgent:
    """Agent implementing strict P0→P1→P2 protocol."""
    
    def __init__(self, maze_size: int, rock_count: int):
        self.size = maze_size
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (maze_size-1, maze_size-1)
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.best_path: List[Tuple[int, int]] = []
        
        # Generate random rocks (negative space)
        self._generate_rocks(rock_count)
    
    def _generate_rocks(self, count: int):
        """P0: Map negative space (what NOT to do)."""
        while len(self.rocks) < count:
            x = random.randint(0, self.size-1)
            y = random.randint(0, self.size-1)
            # Don't block start or goal
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def _is_safe(self, x: int, y: int) -> bool:
        """Check if cell is safe (not a rock)."""
        return (x, y) not in self.rocks
    
    def _find_safe_channels(self):
        """P1: Find safe channels (where you CAN be)."""
        # Use BFS to find all safe paths
        queue = [(self.start, [self.start])]
        visited = set([self.start])
        all_safe_paths = []
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.goal:
                all_safe_paths.append(path)
                continue
            
            # Explore neighbors (right, down, left, up)
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.size and 0 <= ny < self.size:
                    if self._is_safe(nx, ny) and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.safe_channels = all_safe_paths
    
    def _optimize_within_channels(self):
        """P2: Optimize within safe channels (best path)."""
        if not self.safe_channels:
            return
        
        # Find shortest safe path
        self.best_path = min(self.safe_channels, key=len)
    
    def run_protocol(self) -> bool:
        """Execute strict P0→P1→P2 priority."""
        print("=== DEADBAND PROTOCOL DEMO ===")
        print(f"Maze: {self.size}x{self.size}, Rocks: {len(self.rocks)}")
        print(f"Start: {self.start}, Goal: {self.goal}")
        print()
        
        # P0: Already done in __init__ (rocks generated)
        print("P0: Negative space mapped")
        print(f"   - Identified {len(self.rocks)} rocks (failure states)")
        
        # P1: Find safe channels
        print("\nP1: Finding safe channels...")
        self._find_safe_channels()
        print(f"   - Found {len(self.safe_channels)} safe channels")
        
        if not self.safe_channels:
            print("   ✗ No safe channels to goal!")
            return False
        
        # P2: Optimize within channels
        print("\nP2: Optimizing within safe channels...")
        self._optimize_within_channels()
        print(f"   - Optimal path length: {len(self.best_path)}")
        print(f"   - Theoretical optimum: {OPTIMAL_PATH_LENGTH}")
        
        return True
    
    def visualize(self, show_path: bool = True):
        """ASCII visualization of maze and protocol results."""
        print("\n" + "="*50)
        print("MAZE VISUALIZATION")
        print("="*50)
        
        for y in range(self.size):
            row = []
            for x in range(self.size):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')  # Rock
                elif show_path and pos in self.best_path:
                    row.append('·')  # Path
                else:
                    row.append('.')  # Open space
            print(' '.join(row))
        
        print("\nLEGEND: S=Start, G=Goal, █=Rock, ·=Safe Path")
        
        if self.best_path:
            print(f"\nOPTIMAL PATH ({len(self.best_path)} steps):")
            for i, (x, y) in enumerate(self.best_path):
                if i % 5 == 0:
                    print()
                print(f"({x},{y})", end=" → " if i < len(self.best_path)-1 else "\n")
    
    def print_stats(self):
        """Print statistics matching constraint_sim_final.json format."""
        print("\n" + "="*50)
        print("SIMULATION STATISTICS")
        print("="*50)
        
        success = len(self.best_path) > 0
        path_len = len(self.best_path) if success else 0
        
        print(f"Success: {'YES' if success else 'NO'}")
        print(f"Path length: {path_len}")
        print(f"Rocks avoided: {len(self.rocks)}")
        print(f"Safe channels found: {len(self.safe_channels)}")
        
        # Compare to constraint_sim_final.json results
        print("\nCOMPARISON TO FLEET DATA:")
        print(f"  Deadband success rate: 50/50 (100%)")
        print(f"  Deadband avg path: 21.0")
        print(f"  This run: {path_len} steps")
        
        if success and path_len <= OPTIMAL_PATH_LENGTH * 1.5:
            print("  ✓ Within acceptable bounds")
        elif not success:
            print("  ✗ Failed (hit rocks)")
        else:
            print("  ⚠ Suboptimal but safe")

def main():
    """Run the deadband protocol demonstration."""
    # Initialize agent with parameters from constraint_sim_final.json
    agent = DeadbandAgent(MAZE_SIZE, ROCK_COUNT)
    
    # Execute strict P0→P1→P2 protocol
    success = agent.run_protocol()
    
    if success:
        # Visualize results
        agent.visualize(show_path=True)
        agent.print_stats()
        
        # Demonstrate protocol violation (what NOT to do)
        print("\n" + "="*50)
        print("PROTOCOL VIOLATION EXAMPLE")
        print("="*50)
        print("If we skip P0+P1 and go straight to P2 (greedy optimization):")
        print("  - 0/50 success in fleet simulations")
        print("  - Average path: 2320.9 (vs 21.0 with protocol)")
        print("  - Always hits rocks as complexity increases")
    else:
        print("\nProtocol failed: no safe path exists.")
        agent.visualize(show_path=False)

if __name__ == "__main__":
    main()
```

**Key Features:**
1. **Strict P0→P1→P2 Priority**: Agent cannot optimize (P2) without first mapping rocks (P0) and finding safe channels (P1)
2. **Real Parameters**: Uses actual data from `constraint_sim_final.json` (20x20 maze, 64 rocks, optimal 21 steps)
3. **Visualization**: ASCII maze showing rocks (█), start (S), goal (G), and safe path (·)
4. **Statistics**: Compares run to fleet simulation results
5. **Protocol Violation Warning**: Shows what happens when skipping P0+P1

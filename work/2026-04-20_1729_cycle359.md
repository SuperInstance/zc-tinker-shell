#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 359)
A visual ASCII simulation of P0→P1→P2 in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# Simulation constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64          # rocks from constraint_sim_final.json
OPTIMAL_PATH_LEN = 21    # optimal path length from data

# Symbols for visualization
AGENT = 'A'
GOAL = 'G'
ROCK = '█'
SAFE = '·'
PATH = '○'
WALL = '│'
FLOOR = '─'

class DeadbandMaze:
    """Implements the Deadband Protocol in a grid maze."""
    
    def __init__(self, seed=42):
        random.seed(seed)
        self.grid = [[SAFE for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        
        # P0: Map negative space (rocks)
        self._place_rocks()
        
        # P1: Identify safe channels
        self.safe_channels = self._find_safe_channels()
        
        # P2: Optimize within channels
        self.optimal_path = self._find_optimal_path()
    
    def _place_rocks(self):
        """P0: Randomly place rocks (negative space)."""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[y][x] = ROCK
    
    def _find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Find all safe paths from start to goal (DFS)."""
        def dfs(pos, visited, path):
            if pos == self.goal:
                return [path.copy()]
            
            paths = []
            x, y = pos
            
            # 4-direction movement
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        result = dfs((nx, ny), visited, path + [(nx, ny)])
                        if result:
                            paths.extend(result)
                        visited.remove((nx, ny))
            return paths
        
        all_paths = dfs(self.start, {self.start}, [self.start])
        return all_paths
    
    def _find_optimal_path(self) -> List[Tuple[int, int]]:
        """P2: Find shortest safe path (BFS within safe channels)."""
        if not self.safe_channels:
            return []
        
        # BFS for shortest path
        queue = [(self.start, [self.start])]
        visited = {self.start}
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.goal:
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        return []
    
    def visualize(self, show_path=True, show_channels=False):
        """Create ASCII visualization of the maze."""
        # Create display grid
        display = [[self.grid[y][x] for x in range(MAZE_SIZE)] for y in range(MAZE_SIZE)]
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        display[sy][sx] = AGENT
        display[gy][gx] = GOAL
        
        # Show safe channels (P1)
        if show_channels and self.safe_channels:
            for channel in self.safe_channels[:3]:  # Show first 3 channels
                for x, y in channel[1:-1]:  # Exclude start and goal
                    if display[y][x] == SAFE:
                        display[y][x] = '≈'
        
        # Show optimal path (P2)
        if show_path and self.optimal_path:
            for x, y in self.optimal_path[1:-1]:  # Exclude start and goal
                if display[y][x] == SAFE or display[y][x] == '≈':
                    display[y][x] = PATH
        
        # Build border
        top_border = '┌' + '─' * (MAZE_SIZE * 2 - 1) + '┐'
        bottom_border = '└' + '─' * (MAZE_SIZE * 2 - 1) + '┘'
        
        # Assemble output
        lines = [top_border]
        for y in range(MAZE_SIZE):
            row = '│' + ' '.join(display[y]) + '│'
            lines.append(row)
        lines.append(bottom_border)
        
        return '\n'.join(lines)
    
    def print_metrics(self):
        """Print simulation metrics matching fleet data."""
        print("\n" + "="*50)
        print("DEADBAND PROTOCOL SIMULATION METRICS")
        print("="*50)
        print(f"Maze Size: {MAZE_SIZE}x{MAZE_SIZE}")
        print(f"Rocks (P0 negative space): {len(self.rocks)}")
        print(f"Safe channels found (P1): {len(self.safe_channels)}")
        
        if self.optimal_path:
            print(f"Optimal path length (P2): {len(self.optimal_path)-1}")
            success = "SUCCESS" if len(self.optimal_path)-1 <= OPTIMAL_PATH_LEN else "SUBOPTIMAL"
            print(f"Result: {success}")
        else:
            print("Optimal path: NO PATH FOUND (all channels blocked)")
        
        # Compare with fleet data
        print("\n" + "-"*50)
        print("COMPARISON WITH FLEET DATA")
        print("-"*50)
        print("From constraint_sim_final.json:")
        print(f"  Optimal path length: {OPTIMAL_PATH_LEN}")
        print(f"  Rocks: {ROCK_COUNT}")
        print(f"  Deadband success rate: 50/50 (100%)")
        print(f"  Deadband avg path length: 21.0")
        
        if self.optimal_path:
            our_len = len(self.optimal_path)-1
            diff = abs(our_len - OPTIMAL_PATH_LEN)
            print(f"\nOur simulation: {our_len} steps ({diff} from optimal)")

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 priority in a 20x20 ASCII maze.
Based on simulation data from fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import sys
import time

class DeadbandMaze:
    """
    A simple maze simulation that enforces the Deadband Protocol.
    P0: Map rocks (obstacles)
    P1: Find safe channels (paths avoiding rocks)
    P2: Optimize within safe channels (shortest path)
    """
    
    def __init__(self, width=20, height=20, rock_density=0.15):
        self.width = width
        self.height = height
        self.rock_density = rock_density
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self.safe_channels = []
        self.optimal_path = []
        
        # Generate maze
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self._place_rocks()
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def _place_rocks(self):
        """P0: Identify negative space (rocks)"""
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) == self.start or (x, y) == self.goal:
                    continue
                if random.random() < self.rock_density:
                    self.rocks.add((x, y))
                    self.grid[y][x] = '#'
    
    def _find_safe_channels(self):
        """P1: Find all paths that avoid rocks (safe channels)"""
        # Simple BFS to find all reachable positions avoiding rocks
        visited = set()
        queue = [(self.start, [self.start])]
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.goal:
                self.safe_channels.append(path)
                continue
            
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Check four directions
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks:
                        queue.append(((nx, ny), path + [(nx, ny)]))
    
    def _optimize_within_channels(self):
        """P2: Find shortest path within safe channels"""
        if not self.safe_channels:
            return []
        
        # Find channel with shortest path
        self.optimal_path = min(self.safe_channels, key=len)
        return self.optimal_path
    
    def run_protocol(self):
        """Execute full P0→P1→P2 sequence"""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===\n")
        
        # P0: Map negative space
        print("P0: MAP NEGATIVE SPACE")
        print(f"Identified {len(self.rocks)} rocks (obstacles)")
        self.print_maze()
        print()
        time.sleep(1)
        
        # P1: Find safe channels
        print("P1: IDENTIFY SAFE CHANNELS")
        self._find_safe_channels()
        print(f"Found {len(self.safe_channels)} safe channels to goal")
        if self.safe_channels:
            print(f"Channel lengths: {[len(c) for c in self.safe_channels[:5]]}...")
        print()
        time.sleep(1)
        
        # P2: Optimize within channels
        print("P2: OPTIMIZE WITHIN SAFE CHANNELS")
        self._optimize_within_channels()
        if self.optimal_path:
            print(f"Optimal path length: {len(self.optimal_path)-1} steps")
            self.print_maze_with_path()
        else:
            print("No safe path found (maze impossible)")
        print()
        
        # Results summary
        print("=== RESULTS ===")
        print(f"Rocks (P0): {len(self.rocks)}")
        print(f"Safe channels (P1): {len(self.safe_channels)}")
        print(f"Optimal path steps (P2): {len(self.optimal_path)-1 if self.optimal_path else 'N/A'}")
        
        # Compare with protocol violation (P2-only greedy)
        greedy_result = self._simulate_greedy_violation()
        print(f"\nProtocol VIOLATION (P2-only greedy):")
        print(f"  Success: {'Yes' if greedy_result else 'No'}")
        print(f"  Path length: {greedy_result if greedy_result else 'N/A (stuck)'}")
    
    def _simulate_greedy_violation(self):
        """Simulate what happens when skipping P0+P1 (protocol violation)"""
        # Greedy approach: always move toward goal without checking rocks first
        x, y = self.start
        path = [(x, y)]
        visited = set([(x, y)])
        
        for _ in range(100):  # Limit steps
            if (x, y) == self.goal:
                return len(path) - 1
            
            # Find direction that moves toward goal
            moves = []
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited:
                        # Greedy heuristic: Manhattan distance to goal
                        dist = abs(nx - self.goal[0]) + abs(ny - self.goal[1])
                        moves.append((dist, (nx, ny)))
            
            if not moves:
                break
            
            # Choose move that minimizes distance to goal
            moves.sort()
            _, (nx, ny) = moves[0]
            
            # Check if we hit a rock (P0 violation!)
            if (nx, ny) in self.rocks:
                return False  # Failed
            
            x, y = nx, ny
            path.append((x, y))
            visited.add((x, y))
        
        return False
    
    def print_maze(self):
        """Print the maze with rocks, start, and goal"""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                row.append(self.grid[y][x])
            print(' '.join(row))
    
    def print_maze_with_path(self):
        """Print maze with optimal path marked"""
        # Create copy of grid
        display = [row[:] for row in self.grid]
        
        # Mark path (avoid start and goal)
        for x, y in self.optimal_path[1:-1]:
            display[y][x] = '○'
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                row.append(display[y][x])
            print(' '.join(row))

def main():
    """Run the demonstration"""
    print("Deadband Protocol Maze Demo")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print(f"Optimal success rate with protocol: 50/50")
    print(f"Success rate without protocol: 0/50")
    print("-" * 40)
    
    # Create and run maze
    maze = DeadbandMaze(width=20, height=20, rock_density=0.15)
    maze.run_protocol()
    
    print("\n" + "=" * 40)
    print("KEY INSIGHTS:")
    print("1. P0 (map rocks) must come first")
    print("2. P1 (find safe channels) constrains search space")
    print("3. P2 (optimize) only works within safe channels")
    print("4. Violating P0→P1→P2 order leads to failure")

if __name__ == "__main__":
    main()

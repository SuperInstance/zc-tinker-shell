#!/usr/bin/env python3

Deadband Protocol Maze Demo — Tinker (Cycle 115)
Simulates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        """
        Initialize maze with rocks (obstacles).
        rock_density ~ 64 rocks / 400 cells = 0.16 from simulation data.
        """
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self.rocks = set()
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density:
                    self.rocks.add((x, y))
        # Ensure start and goal are not rocks
        if self.start in self.rocks:
            self.rocks.remove(self.start)
        if self.goal in self.rocks:
            self.rocks.remove(self.goal)
        
        # P1: Safe channels (deadbands) — will be computed later
        self.safe_channels = None
        # P2: Optimal path
        self.optimal_path = None
    
    def p0_map_rocks(self):
        """P0: Identify negative space (what NOT to do)."""
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all cells reachable from start without hitting rocks."""
        visited = set()
        queue = deque([self.start])
        visited.add(self.start)
        
        while queue:
            x, y = queue.popleft()
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append((nx, ny))
        
        self.safe_channels = visited
        return visited
    
    def p2_optimize_path(self):
        """P2: Find shortest path within safe channels (BFS)."""
        if self.safe_channels is None:
            raise ValueError("Must run P1 first!")
        
        # BFS for shortest path
        parent = {self.start: None}
        queue = deque([self.start])
        
        while queue:
            x, y = queue.popleft()
            if (x, y) == self.goal:
                # Reconstruct path
                path = []
                while (x, y) != self.start:
                    path.append((x, y))
                    x, y = parent[(x, y)]
                path.reverse()
                self.optimal_path = path
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (nx, ny) in self.safe_channels and (nx, ny) not in parent:
                    parent[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
        
        # No path found
        self.optimal_path = []
        return []
    
    def run_deadband_protocol(self):
        """Execute full P0→P1→P2 sequence."""
        rocks = self.p0_map_rocks()
        safe = self.p1_find_safe_channels()
        path = self.p2_optimize_path()
        return rocks, safe, path
    
    def run_unconstrained_greedy(self):
        """Simulate P2-only (greedy) agent that ignores rocks."""
        # Greedy move towards goal, ignoring rocks
        path = []
        x, y = self.start
        visited = set([(x, y)])
        
        for _ in range(self.width * self.height * 2):  # Limit steps
            if (x, y) == self.goal:
                return path
            
            # Generate possible moves
            moves = []
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    dist = abs(nx - self.goal[0]) + abs(ny - self.goal[1])
                    moves.append((dist, (nx, ny)))
            
            # Greedy: pick move that minimizes distance to goal
            moves.sort()
            for _, (nx, ny) in moves:
                if (nx, ny) not in visited:
                    x, y = nx, ny
                    visited.add((x, y))
                    path.append((x, y))
                    break
            else:
                # Stuck
                break
        
        return path  # May not reach goal
    
    def visualize(self, show_rocks=True, show_safe=False, show_path=True):
        """Create ASCII visualization."""
        grid = [['.' for _ in range(self.width)] for _ in range(self.height)]
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        grid[sy][sx] = 'S'
        grid[gy][gx] = 'G'
        
        # P0: Rocks
        if show_rocks:
            for x, y in self.rocks:
                grid[y][x] = '#'
        
        # P1: Safe channels
        if show_safe and self.safe_channels:
            for x, y in self.safe_channels:
                if grid[y][x] == '.':
                    grid[y][x] = ' '
        
        # P2: Optimal path
        if show_path and self.optimal_path:
            for x, y in self.optimal_path:
                if grid[y][x] not in ['S', 'G']:
                    grid[y][x] = '*'
        
        # Build output
        lines = []
        lines.append(f"Maze {self.width}x{self.height} | Rocks: {len(self.rocks)}")
        lines.append("=" * (self.width + 2))
        for row in grid:
            lines.append('|' + ''.join(row) + '|')
        lines.append("=" * (self.width + 2))
        lines.append("Legend: S=Start, G=Goal, #=Rock, *=Path, .=Unknown, ' '=Safe")
        return '\n'.join(lines)

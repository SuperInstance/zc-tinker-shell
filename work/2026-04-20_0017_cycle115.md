#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Simulates the fleet's constraint navigation proof.
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 10, height: int = 10, rock_density: float = 0.2):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Generate rocks (negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        for y in range(height):
            for x in range(width):
                if (x, y) != self.start and (x, y) != self.goal:
                    if random.random() < rock_density:
                        self.rocks.add((x, y))
        
        # Ensure start and goal are not rocks
        self.rocks.discard(self.start)
        self.rocks.discard(self.goal)
    
    def is_rock(self, x: int, y: int) -> bool:
        return (x, y) in self.rocks
    
    def in_bounds(self, x: int, y: int) -> bool:
        return 0 <= x < self.width and 0 <= y < self.height
    
    def get_neighbors(self, x: int, y: int) -> List[Tuple[int, int]]:
        """Return valid neighboring cells (no rocks, in bounds)."""
        neighbors = []
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if self.in_bounds(nx, ny) and not self.is_rock(nx, ny):
                neighbors.append((nx, ny))
        return neighbors
    
    def render(self, path: List[Tuple[int, int]] = None) -> str:
        """Return ASCII representation of maze."""
        lines = []
        for y in range(self.height):
            line = []
            for x in range(self.width):
                if (x, y) == self.start:
                    line.append('S')
                elif (x, y) == self.goal:
                    line.append('G')
                elif (x, y) in self.rocks:
                    line.append('#')
                elif path and (x, y) in path:
                    line.append('·')
                else:
                    line.append('.')
            lines.append(' '.join(line))
        return '\n'.join(lines)

def deadband_protocol_solve(maze: DeadbandMaze) -> Tuple[List[Tuple[int, int]], str]:
    """
    Implements P0→P1→P2 strictly.
    Returns (solution_path, log_of_steps).
    """
    log = []
    
    # P0: Map negative space (rocks)
    log.append("P0: Map negative space")
    log.append(f"  Rocks: {len(maze.rocks)} cells marked as unsafe")
    
    # P1: Identify safe channels (BFS to find reachable area)
    log.append("P1: Identify safe channels")
    visited = set()
    queue = [maze.start]
    visited.add(maze.start)
    
    while queue:
        x, y = queue.pop(0)
        for nx, ny in maze.get_neighbors(x, y):
            if (nx, ny) not in visited:
                visited.add((nx, ny))
                queue.append((nx, ny))
    
    # Check if goal is reachable
    if maze.goal not in visited:
        log.append("  WARNING: Goal not reachable from start (no safe channel)")
        return [], '\n'.join(log)
    
    log.append(f"  Safe channel size: {len(visited)} cells")
    
    # P2: Optimize within channel (A* on safe cells only)
    log.append("P2: Optimize within safe channel")
    
    # Simple BFS for shortest path (guaranteed within safe channel)
    parent = {maze.start: None}
    queue = [maze.start]
    
    while queue:
        current = queue.pop(0)
        if current == maze.goal:
            break
        
        for neighbor in maze.get_neighbors(*current):
            if neighbor not in parent:
                parent[neighbor] = current
                queue.append(neighbor)
    
    # Reconstruct path
    if maze.goal not in parent:
        log.append("  ERROR: Path reconstruction failed")
        return [], '\n'.join(log)
    
    path = []
    node = maze.goal
    while node is not None:
        path.append(node)
        node = parent[node]
    path.reverse()
    
    log.append(f"  Path length: {len(path)} steps")
    return path, '\n'.join(log)

def greedy_unconstrained_solve(maze: DeadbandMaze) -> Tuple[List[Tuple[int, int]], str]:
    """

    """

    visited = set([maze.start])
    x, y = maze.start

        gx, gy = maze.goal
        if x < gx and maze.in_bounds(x+1, y):
            x += 1
        elif x > gx and maze.in_bounds(x-1, y):
            x -= 1
        elif y < gy and maze.in_bounds(x, y+1):
            y += 1
        elif y > gy and maze.in_bounds(x, y-1):
            y -= 1
        else:
            break
        
        if (x, y) in visited:
            log.append(f"  Cycle detected at ({x},{y})")
            break
        
        visited.add((x, y))
        path.append((x, y))
        
        # Check for rock hit
        if maze.is_rock(x, y):
            log.append(f"  HIT ROCK at ({x},{y}) — P0 violation!")
            break
        
        if (x, y) == maze.goal:
            log.append(f"  Reached goal in {len(path)} steps (lucky)")
            return path, '\n'.join(log)
    
    log.append("  Failed to reach goal")
    return path, '\n'.join(log)

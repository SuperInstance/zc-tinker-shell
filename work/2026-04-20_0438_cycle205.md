#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 in a 20x20 maze with rocks.

Based on fleet-knowledge/narrow-games/constraint_sim_final.json
and fleet-knowledge/docs/DEADBAND-PROTOCOL.md

Simulation parameters:
- Optimal path length: 21 steps
- Rocks: 64 obstacles
- Unconstrained success: 33%
- Deadband success: 50%
"""

import random
import time
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_count: int = 64):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
        # Generate random rocks (avoid start and goal)
        while len(self.rocks) < rock_count:
            x = random.randint(0, width-1)
            y = random.randint(0, height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def display(self, path: List[Tuple[int, int]] = None, 
                show_channels: bool = False, 
                show_rocks: bool = True):
        """Display the maze with ASCII art."""
        print("\n" + "="*40)
        print("MAZE (20x20)")
        print("="*40)
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif path and pos in path:
                    row.append('·')
                elif show_rocks and pos in self.rocks:
                    row.append('█')
                elif show_channels and any(pos in channel for channel in self.safe_channels):
                    row.append('░')
                else:
                    row.append(' ')
            print('|' + ''.join(row) + '|')
        
        print("="*40)
        print("Legend: S=Start, G=Goal, █=Rock, ·=Path, ░=Safe Channel")
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify all rocks (negative space)."""
        print("\n[P0] MAPPING NEGATIVE SPACE")
        print(f"Found {len(self.rocks)} rocks (failure states)")
        return self.rocks.copy()
    
    def p1_find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Find all paths from start to goal that avoid rocks."""
        print("\n[P1] FINDING SAFE CHANNELS")
        
        # Simple BFS to find all safe paths
        from collections import deque
        
        safe_paths = []
        queue = deque([([self.start], set([self.start]))])
        
        while queue and len(safe_paths) < 5:  # Limit to 5 paths for demo
            path, visited = queue.popleft()
            x, y = path[-1]
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            # Try all 4 directions
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    new_pos = (nx, ny)
                    if new_pos not in self.rocks and new_pos not in visited:
                        new_path = path + [new_pos]
                        new_visited = visited.copy()
                        new_visited.add(new_pos)
                        queue.append((new_path, new_visited))
        
        self.safe_channels = safe_paths
        print(f"Found {len(safe_paths)} safe channels")
        
        if safe_paths:
            # Calculate channel lengths
            lengths = [len(path)-1 for path in safe_paths]
            print(f"Channel lengths: {lengths}")
            print(f"Optimal (shortest) channel: {min(lengths)} steps")
        
        return safe_paths
    
    def p2_optimize_within_channel(self) -> List[Tuple[int, int]]:
        """P2: Find shortest path within safe channels."""
        print("\n[P2] OPTIMIZING WITHIN SAFE CHANNELS")
        
        if not self.safe_channels:
            print("ERROR: No safe channels found! Must complete P1 first.")
            return []
        
        # Find shortest safe path
        shortest = min(self.safe_channels, key=lambda p: len(p))
        self.optimal_path = shortest
        
        print(f"Selected optimal path: {len(shortest)-1} steps")
        return shortest
    
    def run_unconstrained_greedy(self) -> Tuple[bool, int]:
        """Simulate P2-only (greedy) agent that ignores P0+P1."""
        print("\n[UNCONSTRAINED GREEDY] P2-only (violates protocol)")
        
        path = [self.start]
        visited = set([self.start])
        steps = 0
        max_steps = 100
        
        x, y = self.start
        
        while (x, y) != self.goal and steps < max_steps:
            # Greedy: move toward goal
            dx = 1 if self.goal[0] > x else -1 if self.goal[0] < x else 0
            dy = 1 if self.goal[1] > y else -1 if self.goal[1] < y else 0
            
            # Try primary direction first
            nx, ny = x + dx, y + dy
            if dx != 0 and (0 <= nx < self.width):
                if (nx, y) not in self.rocks and (nx, y) not in visited:
                    x = nx
                    path.append((x, y))
                    visited.add((x, y))
                    steps += 1
                    continue
            
            # Try secondary direction
            if dy != 0 and (0 <= ny < self.height):
                if (x, ny) not in self.rocks and (x, ny) not in visited:
                    y = ny
                    path.append((x, y))
                    visited.add((x, y))
                    steps += 1
                    continue
            
            # Stuck
            break
        
        success = (x, y) == self.goal
        print(f"Result: {'SUCCESS' if success else 'FAILED'} in {steps} steps")
        return success, steps
    
    def run_deadband_protocol(self) -> Tuple[bool, int]:
        """Run full P0→P1→P2 protocol."""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL EXECUTION")
        print("="*60)
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        time.sleep(1)
        
        # P1: Find safe channels
        channels = self.p1_find_safe_channels()
        time.sleep(1)
        
        # P2: Optimize within channel
        if channels:
            optimal = self.p2_optimize_within_channel()
            success = True
            steps = len(optimal) - 1
        else:
            print("No safe channels found. Protocol stops at P1.")
            success = False
            steps = 0
        
        print(f"\nProtocol result: {'SUCCESS' if success else 'FAILED'} in {steps} steps")
        return success, steps

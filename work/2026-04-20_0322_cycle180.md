**Status:** Based on the fleet knowledge, specifically the `constraint_sim_final.json` results and the DEADBAND-PROTOCOL.md, I will implement a concrete simulation. The data shows: unconstrained (P2-only) success 33%, deadband (P0→P1→P2) success 50%. I will model this.

**Approach:**
1. **P0 (Map negative space):** Generate a random 10x10 maze with obstacles ('#') and open cells ('.'). The agent's starting position is (0,0), goal is (9,9). Rocks are obstacles.
2. **P1 (Identify safe channels):** Use BFS from start to find all reachable cells that avoid obstacles. These are the safe channels.
3. **P2 (Optimize within channel):** Use A* pathfinding, but only allowed to move through cells marked as safe in P1.
4. **Visualization:** Print the maze with the path overlaid.

**Output:** A Python script `deadband_maze_demo.py` that runs the protocol and prints each step.

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map rocks (obstacles)
P1: Find safe channels (reachable cells avoiding rocks)
P2: Find shortest path within safe channels
Based on fleet-knowledge/constraint_sim_final.json results.
"""

import random
from collections import deque
import heapq

def generate_maze(width=10, height=10, obstacle_density=0.2):
    """Generate a random maze with obstacles."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    for y in range(height):
        for x in range(width):
            if (x, y) != (0, 0) and (x, y) != (width-1, height-1):
                if random.random() < obstacle_density:
                    maze[y][x] = '#'
    return maze

def print_maze(maze, path=None):
    """Print maze with optional path overlay."""
    for y, row in enumerate(maze):
        line = []
        for x, cell in enumerate(row):
            if path and (x, y) in path:
                line.append('*')
            else:
                line.append(cell)
        print(' '.join(line))

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks). Return set of obstacle coordinates."""
    rocks = set()
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(maze, start, goal, rocks):
    """P1: BFS to find all cells reachable from start while avoiding rocks."""
    width, height = len(maze[0]), len(maze)
    safe = set()
    queue = deque([start])
    visited = set([start])
    
    while queue:
        x, y = queue.popleft()
        safe.add((x, y))
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    return safe

def p2_optimize_path(start, goal, safe_cells):
    """P2: A* search constrained to safe cells only."""
    if start not in safe_cells or goal not in safe_cells:
        return None  # No safe path
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            path.reverse()
            return path
        
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            neighbor = (x + dx, y + dy)
            if neighbor in safe_cells:
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    return None

def run_deadband_simulation():
    """Run one complete Deadband Protocol demonstration."""
    random.seed(42)  # Deterministic for demo
    maze = generate_maze()
    start = (0, 0)
    goal = (9, 9)
    
    print("=== DEADBAND PROTOCOL MAZE DEMO ===")
    print("\n1. Initial Maze ('.' = open, '#' = rock):")
    print_maze(maze)
    
    # P0: Map rocks
    rocks = p0_map_rocks(maze)
    print(f"\n2. P0: Mapped {len(rocks)} rocks (negative space)")
    
    # P1: Find safe channels
    safe_cells = p1_find_safe_channels(maze, start, goal, rocks)
    print(f"   P1: Found {len(safe_cells)} safe cells (channels)")
    
    # P2: Optimize path within safe channels
    path = p2_optimize_path(start, goal, safe_cells)
    
    if path:
        print(f"   P2: Found optimal path of length {len(path)-1} steps")
        print("\n3. Final Maze with Deadband Path ('*' = path):")
        print_maze(maze, set(path))
        print(f"\nResult: SUCCESS - Path from {start} to {goal} within safe channels")
        return True
    else:
        print("\n3. No safe path exists (goal unreachable within safe channels)")
        print_maze(maze)
        print(f"\nResult: FAILURE - Goal {goal} not reachable from {start} avoiding rocks")
        return False

def compare_strategies(runs=50):
    """Compare Deadband vs Greedy (P2-only) strategies."""
    deadband_success = 0
    greedy_success = 0
    
    for i in range(runs):
        random.seed(i)
        maze = generate_maze()
        start = (0, 0)
        goal = (9, 9)
        
        # Deadband Protocol
        rocks = p0_map_rocks(maze)
        safe_cells = p1_find_safe_channels(maze, start, goal, rocks)
        deadband_path = p2_optimize_path(start, goal, safe_cells)
        if deadband_path:
            deadband_success += 1
        
        # Greedy (P2-only - ignores rocks, just tries A*)
        def greedy_search():
            return p2_optimize_path(start, goal, set([(x, y) for y in range(10) for x in range(10)]))
        greedy_path = greedy_search()
        if greedy_path:
            greedy_success += 1
    
    print(f"\n=== STRATEGY COMPARISON ({runs} runs) ===")
    print(f"Deadband (P0→P1→P2) success: {deadband_success}/{runs} ({deadband_success/runs*100:.1f}%)")
    print(f"Greedy (P2-only) success: {greedy_success}/{runs} ({greedy_success/runs*100:.1f}%)")
    print("\nNote: Greedy may 'succeed' by pathing through rocks (impossible in reality).")

if __name__ == "__main__":
    # Single demonstration
    success = run_deadband_simulation()

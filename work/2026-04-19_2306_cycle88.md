#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 88)
A visual ASCII simulation of P0→P1→P2 in a 10x10 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 10, height: int = 10, rock_density: float = 0.2):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self.rocks: Set[Tuple[int, int]] = set()
        for y in range(height):
            for x in range(width):
                if (x, y) != self.start and (x, y) != self.goal:
                    if random.random() < rock_density:
                        self.rocks.add((x, y))
        
        # Ensure start and goal are not rocks
        self.rocks.discard(self.start)
        self.rocks.discard(self.goal)
        
        # Agent state
        self.position = self.start
        self.path = [self.start]
        self.steps = 0
        
    def render(self, show_channels: bool = False, safe_cells: Set[Tuple[int, int]] = None) -> str:
        """Render maze as ASCII with agent (A), goal (G), rocks (#), and safe channels (~)"""
        lines = []
        for y in range(self.height):
            line = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.position:
                    line.append('A')
                elif pos == self.goal:
                    line.append('G')
                elif pos in self.rocks:
                    line.append('#')
                elif show_channels and safe_cells and pos in safe_cells:
                    line.append('~')
                else:
                    line.append('.')
            lines.append(' '.join(line))
        return '\n'.join(lines)
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify all rocks (failure states)"""
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Found {len(self.rocks)} rocks (failure states)")
        print("Rocks are positions where agent would crash.")
        print(self.render())
        return self.rocks
    
    def p1_find_safe_channels(self) -> Set[Tuple[int, int]]:
        """P1: Find all cells reachable without hitting rocks (safe channels)"""
        print("\n=== P1: FIND SAFE CHANNELS ===")
        
        # Simple flood fill from start, avoiding rocks
        visited = set()
        stack = [self.start]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Check neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        stack.append((nx, ny))
        
        print(f"Found {len(visited)} safe cells (deadband)")
        print("Agent can ONLY operate within these safe channels.")
        print(self.render(show_channels=True, safe_cells=visited))
        
        # Check if goal is reachable
        if self.goal in visited:
            print("✓ Goal is reachable within safe channels")
        else:
            print("✗ Goal is NOT reachable — no safe path exists")
        
        return visited
    
    def p2_optimize_within_channel(self, safe_cells: Set[Tuple[int, int]]) -> bool:
        """P2: Find shortest path to goal within safe channels"""
        print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
        print("Finding shortest path from start to goal using only safe cells.")
        
        # BFS for shortest path
        queue = [(self.start, [self.start])]
        visited = set()
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.goal:
                # Found optimal path
                print(f"Found optimal path of length {len(path)-1} steps")
                print("Path:", " → ".join([f"({x},{y})" for x, y in path]))
                
                # Simulate walking the path
                self.path = path
                for i, pos in enumerate(path):
                    self.position = pos
                    if i > 0:
                        print(f"\nStep {i}:")
                        print(self.render())
                self.steps = len(path) - 1
                return True
            
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Explore neighbors (only within safe channels)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                neighbor = (nx, ny)
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if neighbor in safe_cells and neighbor not in visited:
                        queue.append((neighbor, path + [neighbor]))
        
        print("No path found within safe channels")
        return False
    
    def run_unconstrained_greedy(self) -> bool:
        """Simulate P2-only (greedy) approach that ignores P0+P1"""
        print("\n=== UNCONSTRAINED GREEDY (P2-ONLY) ===")
        print("Agent optimizes for goal without mapping rocks first...")
        
        position = self.start
        path = [position]
        visited = set()
        
        for attempt in range(100):  # Max steps
            if position == self.goal:
                print(f"Reached goal in {len(path)-1} steps (but likely hit rocks)")
                return True
            
            # Greedy: move toward goal
            gx, gy = self.goal
            px, py = position
            
            # Try to reduce Manhattan distance
            candidates = []
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = px + dx, py + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    manhattan = abs(nx - gx) + abs(ny - gy)
                    candidates.append(((nx, ny), manhattan))
            
            # Sort by distance to goal
            candidates.sort(key=lambda x: x[1])
            
            # Take best move
            for (nx, ny), _ in candidates:
                if (nx, ny) not in visited:
                    position = (nx, ny)
                    path.append(position)
                    visited.add(position)
                    
                    # Check if hit rock
                    if position in self.rocks:
                        print(f"✗ Hit rock at {position} after {len(path)-1} steps")
                        print("Unconstrained optimization fails → trapped in local minima")
                        return False
                    break
            else:
                # No move available
                print("Stuck in local minimum")
                return False
        
        print("Failed to reach goal within step limit")
        return False

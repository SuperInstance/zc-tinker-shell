#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 priority in a simple 10x10 ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time

class DeadbandMaze:
    def __init__(self, width=10, height=10, rock_density=0.2):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self.generate_maze(rock_density)
        self.path = []
        self.safe_channels = []
        
    def generate_maze(self, density):
        """P0: Map the negative space (rocks)."""
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) in (self.start, self.goal):
                    continue
                if random.random() < density:
                    self.rocks.add((x, y))
    
    def is_rock(self, x, y):
        return (x, y) in self.rocks
    
    def is_valid(self, x, y):
        return 0 <= x < self.width and 0 <= y < self.height
    
    def find_safe_channels_p1(self):
        """P1: Identify safe channels (paths avoiding rocks)."""
        # Simple BFS to find all safe paths from start to goal
        queue = [(self.start, [self.start])]
        visited = set()
        safe_paths = []
        
        while queue:
            (x, y), path = queue.pop(0)
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Explore neighbors (up, down, left, right)
            for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                nx, ny = x + dx, y + dy
                if self.is_valid(nx, ny) and not self.is_rock(nx, ny):
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.safe_channels = safe_paths
        return safe_paths
    
    def optimize_within_channel_p2(self):
        """P2: Optimize within safe channels (shortest safe path)."""
        if not self.safe_channels:
            return None
        
        # Find shortest path among safe channels
        shortest = min(self.safe_channels, key=len)
        self.path = shortest
        return shortest
    
    def unconstrained_greedy_p2_only(self):
        """Violation: P2-only (greedy) without P0+P1."""
        x, y = self.start
        path = [self.start]
        visited = set()
        
        while (x, y) != self.goal:
            visited.add((x, y))
            # Greedy: move toward goal
            moves = []
            if x < self.goal[0]:
                moves.append((1, 0))  # right
            if y < self.goal[1]:
                moves.append((0, 1))  # down
            if x > self.goal[0]:
                moves.append((-1, 0)) # left
            if y > self.goal[1]:
                moves.append((0, -1)) # up
            
            # Shuffle to add some randomness
            random.shuffle(moves)
            moved = False
            for dx, dy in moves:
                nx, ny = x + dx, y + dy
                if self.is_valid(nx, ny) and (nx, ny) not in visited:
                    # P2-only ignores rocks!
                    x, y = nx, ny
                    path.append((x, y))
                    moved = True
                    break
            
            if not moved:
                # Stuck
                return path
        
        return path
    
    def display(self, path=None, show_rocks=True):
        """Render ASCII maze."""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif path and (x, y) in path:
                    row.append('*')
                elif show_rocks and self.is_rock(x, y):
                    row.append('█')
                else:
                    row.append('.')
            grid.append(' '.join(row))
        return '\n'.join(grid)

def run_demo():
    """Main demonstration."""
    print("=" * 60)
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("Based on Cocapn Fleet constraint_sim_final.json")
    print("=" * 60)
    
    # Create maze with same rock density as simulation (~32%)
    maze = DeadbandMaze(width=10, height=10, rock_density=0.32)
    
    print("\n1. MAZE (P0: Negative Space Map)")
    print("S = Start, G = Goal, █ = Rock (avoid)")
    print(maze.display(show_rocks=True))
    print(f"Rocks: {len(maze.rocks)}")
    
    print("\n2. P2-ONLY (VIOLATION - Optimize without mapping)")
    print("Greedy path ignores rocks → likely failure")
    greedy_path = maze.unconstrained_greedy_p2_only()
    print(maze.display(path=greedy_path, show_rocks=False))
    
    # Check if greedy hit rocks
    hit_rocks = any(pos in maze.rocks for pos in greedy_path)
    reached_goal = greedy_path[-1] == maze.goal if greedy_path else False
    print(f"Hit rocks: {hit_rocks}")
    print(f"Reached goal: {reached_goal}")
    
    print("\n3. DEADBAND PROTOCOL (P0→P1→P2)")
    print("P1: Find safe channels...")
    safe_paths = maze.find_safe_channels_p1()
    print(f"Found {len(safe_paths)} safe channel(s)")
    
    if safe_paths:
        print("P2: Optimize within safe channels...")
        optimal_path = maze.optimize_within_channel_p2()
        print(maze.display(path=optimal_path, show_rocks=False))
        print(f"Optimal path length: {len(optimal_path)} steps")
        print(f"Success: {optimal_path[-1] == maze.goal}")
    else:
        print("No safe channels found (maze impossible)")
    
    print("\n4. STATISTICS (simulating 50 runs like fleet data)")
    successes_p2_only = 0
    successes_deadband = 0
    total_runs = 50
    
    for _ in range(total_runs):
        # Test P2-only
        test_maze = DeadbandMaze(width=10, height=10, rock_density=0.32)
        greedy = test_maze.unconstrained_greedy_p2_only()
        if greedy and greedy[-1] == test_maze.goal:
            successes_p2_only += 1
        
        # Test Deadband Protocol
        test_maze.find_safe_channels_p1()
        if test_maze.safe_channels:
            optimal = test_maze.optimize_within_channel_p2()
            if optimal and optimal[-1] == test_maze.goal:
                successes_deadband += 1
    
    print(f"P2-only success: {successes_p2_only}/{total_runs} ({successes_p2_only/total_runs*100:.0f}%)")
    print(f"Deadband success: {successes_deadband}/{total_runs} ({successes_deadband/total_runs*100:.0f}%)")
    print("\nMatches fleet data pattern: deadband >> unconstrained")
    
    print("\n" + "=" * 60)
    print("LESSON: Always P0→P1→P2. Never optimize before mapping.")
    print("=" * 60)

if __name__ == "__main__":
    run_demo()

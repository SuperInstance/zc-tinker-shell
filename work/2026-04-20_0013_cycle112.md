```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 112)
Simulates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json.
"""

import random
from typing import List, Tuple, Set

# ========== MAZE CONFIG (from narrow‑games simulation) ==========
MAZE_SIZE = 20
ROCK_COUNT = 64          # matches "rocks": 64 in JSON
OPTIMAL_PATH_LEN = 21    # matches "optimal": 21
START = (0, 0)
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

# ========== DEADBAND PROTOCOL IMPLEMENTATION ==========
class DeadbandAgent:
    def __init__(self, maze: List[List[str]]):
        self.maze = maze
        self.rocks: Set[Tuple[int, int]] = set()      # P0 result
        self.safe_channels: List[List[Tuple[int, int]]] = []  # P1 result
        self.final_path: List[Tuple[int, int]] = []   # P2 result

    # --- P0: MAP THE NEGATIVE SPACE ---
    def map_negative_space(self):
        """Identify all rock cells (states where failure probability > threshold)."""
        for y in range(MAZE_SIZE):
            for x in range(MAZE_SIZE):
                if self.maze[y][x] == '█':
                    self.rocks.add((x, y))
        print(f"[P0] Mapped {len(self.rocks)} rocks (negative space).")

    # --- P1: IDENTIFY SAFE CHANNELS ---
    def find_safe_channels(self):
        """BFS from start to goal, avoiding rocks, collecting all safe paths."""
        from collections import deque
        queue = deque([(START, [START])])
        visited = {START}
        while queue:
            (x, y), path = queue.popleft()
            if (x, y) == GOAL:
                self.safe_channels.append(path)
                continue
            for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in visited and (nx, ny) not in self.rocks:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        print(f"[P1] Found {len(self.safe_channels)} safe channels (deadbands).")

    # --- P2: OPTIMIZE WITHIN CHANNEL ---
    def optimize_within_channel(self):
        """Select shortest safe path (constrained optimization)."""
        if not self.safe_channels:
            print("[P2] No safe channels — optimization impossible.")
            return
        self.final_path = min(self.safe_channels, key=len)
        print(f"[P2] Optimal path length: {len(self.final_path)} (theoretical optimum: {OPTIMAL_PATH_LEN})")

    def run_protocol(self):
        """Strict P0→P1→P2 sequence."""
        self.map_negative_space()
        self.find_safe_channels()
        self.optimize_within_channel()
        return self.final_path

# ========== UNCONSTRAINED GREEDY AGENT (FAILS) ==========
class GreedyAgent:
    """P2‑only agent that optimizes without P0/P1 — hits rocks."""
    def __init__(self, maze):
        self.maze = maze

    def run(self):
        """Greedy move towards goal, ignoring rocks."""
        path = [START]
        x, y = START
        while (x, y) != GOAL:
            # prefer moving right or down
            if x < GOAL[0] and self.maze[y][x+1] != '█':
                x += 1
            elif y < GOAL[1] and self.maze[y+1][x] != '█':
                y += 1
            elif x > GOAL[0] and self.maze[y][x-1] != '█':
                x -= 1
            elif y > GOAL[1] and self.maze[y-1][x] != '█':
                y -= 1
            else:
                break  # stuck
            path.append((x, y))
        return path

# ========== MAZE GENERATION ==========
def generate_maze() -> List[List[str]]:
    """Create a 20x20 maze with 64 rocks placed randomly."""
    maze = [['·' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks_set = set()
    while len(rocks_set) < ROCK_COUNT:
        x = random.randint(0, MAZE_SIZE-1)
        y = random.randint(0, MAZE_SIZE-1)
        if (x, y) != START and (x, y) != GOAL:
            rocks_set.add((x, y))
    for x, y in rocks_set:
        maze[y][x] = '█'
    maze[START[1]][START[0]] = 'S'
    maze[GOAL[1]][GOAL[0]] = 'G'
    return maze

def print_maze(maze, path=None):
    """Visualize maze with optional path."""
    for y in range(MAZE_SIZE):
        row = []
        for x in range(MAZE_SIZE):
            if path and (x, y) in path:
                row.append('○')
            else:
                row.append(maze[y][x])
        print(''.join(row))

# ========== MAIN DEMONSTRATION ==========
def main():
    print("=" * 60)
    print("DEADBAND PROTOCOL MAZE DEMO — Tinker (Fleet Cocapn)")
    print("=" * 60)
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}, Optimal: {OPTIMAL_PATH_LEN}")
    print()

    # Generate a reproducible maze for demo
    random.seed(42)
    maze = generate_maze()

    print("[Initial Maze]")
    print_maze(maze)
    print()

    # === UNCONSTRAINED GREEDY (P2‑only) ===
    print("\n--- UNCONSTRAINED GREEDY AGENT (P2‑only) ---")
    greedy = GreedyAgent(maze)
    greedy_path = greedy.run()
    print(f"Greedy path length: {len(greedy_path)}")
    if greedy_path[-1] == GOAL:
        print("Result: Reached goal (unlikely in complex maze).")
    else:
        print("Result: Stuck — hit rocks or dead‑end.")
    print_maze(maze, greedy_path)
    print()

    # === DEADBAND PROTOCOL (P0→P1→P2) ===
    print("\n--- DEADBAND PROTOCOL AGENT (P0→P1→P2) ---")
    deadband = DeadbandAgent(maze)
    final_path = deadband.run_protocol()
    if final_path:
        print(f"Deadband path length: {len(final_path)}")
        print_maze(maze, final_path)
        # Compare to optimal
        if len(final_path) == OPTIMAL_PATH_LEN:
            print("✓ Achieved theoretical optimum.")
        else:
            print(f"Note: Optimal is {OPTIMAL_PATH_LEN}; this is a safe approximation.")
    else:
        print("✗ No safe path found — protocol respected, but environment impossible.")

    print("\n" + "=" * 60)
    print("Summary:")
    print("  - Greedy (P2‑only) often fails (simulation: 0% success).")
    print("  - Deadband (P0→P1→P2) finds safe, near‑optimal path.")
    print("  - Strict priority prevents hitting rocks.")
    print("=" * 60)

if __name__ == "__main__":
    main()
```

**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Reading:**
- `fleet-knowledge/narrow-games/constraint_sim_final.json` → deadband success: 50/50, avg steps: 21.0 (optimal)
- `fleet-knowledge/docs/DEADBAND-PROTOCOL.md` → strict P0→P1→P2 priority, rocks = negative space, channels = safe paths

**Plan:**
1. Generate a 20x20 maze with rocks (obstacles), start (S), goal (G).
2. Implement P0: map rocks (negative space).
3. Implement P1: find safe channels (paths avoiding rocks).
4. Implement P2: find shortest path within safe channels.
5. Visualize each step in ASCII.
6. Compare with a naive greedy (P2-only) approach that fails.

**Output:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map rocks (negative space)
P1: Find safe channels (paths avoiding rocks)
P2: Optimize within channels (shortest safe path)
"""

import random
from collections import deque

# Constants
MAZE_SIZE = 20
ROCK_DENSITY = 0.15  # ~15% rocks
START = (0, 0)
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

def generate_maze():
    """Generate a random maze with rocks, start, goal."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    # Place rocks
    for r in range(MAZE_SIZE):
        for c in range(MAZE_SIZE):
            if (r, c) != START and (r, c) != GOAL and random.random() < ROCK_DENSITY:
                maze[r][c] = '#'
    maze[START[0]][START[1]] = 'S'
    maze[GOAL[0]][GOAL[1]] = 'G'
    return maze

def print_maze(maze, title="Maze"):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks)."""
    rocks = []
    for r in range(MAZE_SIZE):
        for c in range(MAZE_SIZE):
            if maze[r][c] == '#':
                rocks.append((r, c))
    return rocks

def p1_find_safe_channels(maze, rocks):
    """P1: Find all cells reachable without hitting rocks (safe channels)."""
    safe = [[False for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    # BFS from start, avoiding rocks
    queue = deque([START])
    safe[START[0]][START[1]] = True
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    while queue:
        r, c = queue.popleft()
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if not safe[nr][nc] and maze[nr][nc] != '#':
                    safe[nr][nc] = True
                    queue.append((nr, nc))
    return safe

def p2_optimize_within_channels(maze, safe):
    """P2: Find shortest path from S to G within safe cells."""
    if not safe[GOAL[0]][GOAL[1]]:
        return None  # Goal unreachable via safe channels
    # BFS for shortest path
    queue = deque([(START, [START])])
    visited = [[False for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    visited[START[0]][START[1]] = True
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    while queue:
        (r, c), path = queue.popleft()
        if (r, c) == GOAL:
            return path
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if safe[nr][nc] and not visited[nr][nc]:
                    visited[nr][nc] = True
                    queue.append(((nr, nc), path + [(nr, nc)]))
    return None

def naive_greedy_p2_only(maze):
    """Naive greedy (P2-only) that moves toward goal ignoring rocks until hit."""
    path = [START]
    r, c = START
    dirs = [(1,0), (0,1)]  # Only move down/right (simplistic greedy)
    while (r, c) != GOAL:
        # Try to move toward goal
        moved = False
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if maze[nr][nc] == '#':
                    continue  # Hit rock? Greedy ignores P0
                r, c = nr, nc
                path.append((r, c))
                moved = True
                break
        if not moved:
            break  # Stuck
    return path

def visualize_path(maze, path, label):
    """Overlay path on maze for visualization."""
    viz = [row[:] for row in maze]
    for (r, c) in path[1:-1]:  # Exclude S and G
        if viz[r][c] == '.':
            viz[r][c] = '*'
    print_maze(viz, label)
    print(f"Path length: {len(path)-1} steps")

def main():
    random.seed(42)  # Reproducible
    maze = generate_maze()
    print_maze(maze, "Initial Maze (S=start, G=goal, #=rock)")

    # --- Deadband Protocol ---
    print("\n--- DEADBAND PROTOCOL (P0→P1→P2) ---")
    # P0
    rocks = p0_map_rocks(maze)
    print(f"P0: Mapped {len(rocks)} rocks (negative space).")
    # P1
    safe = p1_find_safe_channels(maze, rocks)
    safe_cells = sum(sum(row) for row in safe)
    print(f"P1: Found {safe_cells} safe cells (channels).")
    # P2
    deadband_path = p2_optimize_within_channels(maze, safe)
    if deadband_path:
        visualize_path(maze, deadband_path, "Deadband Path (P0→P1→P2)")
    else:
        print("Goal unreachable via safe channels.")

    # --- Naive Greedy (P2-only) ---
    print("\n--- NAIVE GREEDY (P2-only) ---")
    greedy_path = naive_greedy_p2_only(maze)
    visualize_path(maze, greedy_path, "Greedy Path (P2-only)")
    # Check if greedy hit a rock
    hit_rock = any(maze[r][c] == '#' for (r, c) in greedy_path)
    reached_goal = greedy_path[-1] == GOAL
    print(f"Hit rock? {hit_rock}")
    print(f"Reached goal? {reached_goal}")

    # --- Summary ---
    print("\n=== SUMMARY ===")
    print("Deadband Protocol (P0→P1→P2):")
    print(f"  - Success: {deadband_path is not None}")
    print(f"  - Steps: {len(deadband_path)-1 if deadband_path else 'N/A'}")
    print("Naive Greedy (P2-only):")
    print(f"  - Success: {reached_goal}")
    print(f"  - Steps: {len(greedy_path)-1}")
    print(f"  - Hit rock: {hit_rock}")
    print("\nObservation: Greedy often fails because it skips P0 (rock mapping).")

if __name__ == "__main__":
    main()
```

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 46)
Visualizes P0→P1→P2 using a simple ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
import sys

def generate_maze(width=20, height=20, rock_density=0.2):
    """Generate a random maze with rocks (X), start (S), goal (G)."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks
    rocks = int(width * height * rock_density)
    for _ in range(rocks):
        x, y = random.randint(0, width-1), random.randint(0, height-1)
        maze[y][x] = 'X'
    
    # Place start (top-left) and goal (bottom-right)
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    
    return maze

def print_maze(maze, title="Maze"):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks). Return set of rock coordinates."""
    rocks = set()
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == 'X':
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(maze, rocks):
    """P1: Find safe channels (paths avoiding rocks) using BFS from start to goal."""
    from collections import deque
    
    width, height = len(maze[0]), len(maze)
    start = (0, 0)
    goal = (width-1, height-1)
    
    # Directions: right, down, left, up
    dirs = [(1, 0), (0, 1), (-1, 0), (0, -1)]
    
    queue = deque([start])
    visited = {start: None}
    
    while queue:
        x, y = queue.popleft()
        if (x, y) == goal:
            # Reconstruct path
            path = []
            while (x, y) != start:
                path.append((x, y))
                x, y = visited[(x, y)]
            path.append(start)
            path.reverse()
            return set(path)  # Safe channel coordinates
        
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    
    return set()  # No safe path

def p2_optimize_within_channel(maze, safe_channel):
    """P2: Optimize path within safe channel (shortest path)."""
    # In this simple case, safe_channel is already the shortest path from BFS.
    # We'll mark it with 'O' for visualization.
    optimized_maze = [row[:] for row in maze]
    for x, y in safe_channel:
        if optimized_maze[y][x] not in ['S', 'G']:
            optimized_maze[y][x] = 'O'
    return optimized_maze

def simulate_unconstrained_greedy(maze):
    """Simulate P2-only (greedy) agent that ignores rocks."""
    width, height = len(maze[0]), len(maze)
    x, y = 0, 0
    path = [(0, 0)]
    steps = 0
    max_steps = width * height * 2  # Prevent infinite loop
    
    while (x, y) != (width-1, height-1) and steps < max_steps:
        # Greedy: move right if possible, else down
        if x + 1 < width and maze[y][x+1] != 'X':
            x += 1
        elif y + 1 < height and maze[y+1][x] != 'X':
            y += 1
        else:
            # Stuck
            break
        path.append((x, y))
        steps += 1
    
    # Mark path
    greedy_maze = [row[:] for row in maze]
    for px, py in path:
        if greedy_maze[py][px] not in ['S', 'G']:
            greedy_maze[py][px] = '*'
    
    success = (x, y) == (width-1, height-1)
    return greedy_maze, success, len(path)-1

def main():
    random.seed(42)  # Deterministic for demo
    maze = generate_maze(width=12, height=8, rock_density=0.15)
    
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print("Optimal cost: 21, Deadband success: 50/50, Unconstrained success: 0/50")
    print()
    
    # Step 0: Initial maze
    print_maze(maze, "Initial Maze (S=start, G=goal, X=rock, .=empty)")
    
    # P0: Map rocks
    rocks = p0_map_rocks(maze)
    print(f"P0 — Map Negative Space: Found {len(rocks)} rocks.")
    
    # P1: Find safe channels
    safe_channel = p1_find_safe_channels(maze, rocks)
    if not safe_channel:
        print("P1 — No safe channel exists! Agent cannot proceed.")
        return
    
    safe_maze = [row[:] for row in maze]
    for x, y in safe_channel:
        if safe_maze[y][x] not in ['S', 'G']:
            safe_maze[y][x] = '~'
    print_maze(safe_maze, "P1 — Safe Channel (~ = safe cells)")
    print(f"P1 — Safe channel length: {len(safe_channel)} cells")
    
    # P2: Optimize within channel
    optimized_maze = p2_optimize_within_channel(maze, safe_channel)
    print_maze(optimized_maze, "P2 — Optimized Path (O = optimal path within safe channel)")
    print(f"P2 — Optimal path cost: {len(safe_channel)-1} steps")
    
    # Contrast with unconstrained greedy (P2-only)
    print("\n--- CONTRAST: Unconstrained Greedy (P2-only) ---")
    greedy_maze, success, steps = simulate_unconstrained_greedy(maze)
    print_maze(greedy_maze, f"Greedy Path (* = greedy moves, success={success})")
    print(f"Greedy steps: {steps}, Success: {success}")
    
    # Summary
    print("\n=== PROTOCOL SUMMARY ===")
    print("P0 (rocks): Identify failure states — what NOT to do.")
    print("P1 (channel): Find safe paths avoiding rocks — where you CAN be.")
    print("P2 (optimize): Choose shortest path within safe channel.")
    print()
    print("Violating priority (P2 before P0+P1) leads to rock collisions.")
    print("Data: deadband 50/50 success vs unconstrained 0/50 success.")

if __name__ == "__main__":
    main()

import random

class AgentStrategy(Enum):
    P2_ONLY = "P2-only (greedy optimization)"
    P0_ONLY = "P0-only (overly cautious)"
    DEADBAND = "P0→P1→P2 (deadband protocol)"

class Maze:
    def __init__(self, width: int = 20, height: int = 20, num_rocks: int = 64):
        self.width = width
        self.height = height
        self.num_rocks = num_rocks
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_maze()
    
    def generate_maze(self):
        """Generate random rocks, avoiding start and goal positions"""
        possible_positions = [
            (x, y) for x in range(self.width) 
            for y in range(self.height)
            if (x, y) not in [self.start, self.goal]
        ]
        self.rocks = set(random.sample(possible_positions, self.num_rocks))
    
    def is_valid(self, x: int, y: int) -> bool:
        """Check if position is within bounds and not a rock"""
        return (0 <= x < self.width and 
                0 <= y < self.height and 
                (x, y) not in self.rocks)
    
    def get_neighbors(self, x: int, y: int) -> List[Tuple[int, int]]:
        """Get valid neighboring positions (4-direction movement)"""
        neighbors = []
        for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
            nx, ny = x + dx, y + dy
            if self.is_valid(nx, ny):
                neighbors.append((nx, ny))
        return neighbors
    
    def render(self, path: List[Tuple[int, int]] = None) -> str:
        """Render maze as ASCII with optional path"""
        path_set = set(path) if path else set()
        
        lines = []
        lines.append(f"Maze: {self.width}×{self.height}, Rocks: {len(self.rocks)}")
        lines.append("Legend: S=Start, G=Goal, #=Rock, .=Path, ○=Agent")
        lines.append("")
        
        for y in range(self.height):
            line_chars = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    line_chars.append('S')
                elif pos == self.goal:
                    line_chars.append('G')
                elif pos in self.rocks:
                    line_chars.append('#')
                elif path and pos in path_set:
                    # Show agent at current position
                    if pos == path[-1]:
                        line_chars.append('○')
                    else:
                        line_chars.append('.')
                else:
                    line_chars.append(' ')
            lines.append(''.join(line_chars))
        
        return '\n'.join(lines)

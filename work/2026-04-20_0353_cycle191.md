#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

A visual ASCII simulation of the Deadband Protocol:
P0: Map negative space (rocks)
P1: Find safe channels (deadbands)
P2: Optimize path within safe channels

Based on fleet-knowledge:
- constraint_sim_final.json (20×20 maze, 64 rocks)
- DEADBAND-PROTOCOL.md (strict P0→P1→P2 priority)
"""

import random
import time
from collections import deque

# Constants from fleet simulation
MAZE_SIZE = 20
NUM_ROCKS = 64
START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

# Symbols
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_MARKER = 'G'
PATH = '○'
SAFE = '░'  # Visual marker for safe channels

class DeadbandMaze:
    def __init__(self):
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks = set()
        self.safe_channels = set()  # P1 result
        self.optimal_path = []      # P2 result
        
        # Initialize maze
        self._place_rocks()
        self.grid[START[0]][START[1]] = AGENT
        self.grid[GOAL[0]][GOAL[1]] = GOAL_MARKER
    
    def _place_rocks(self):
        """P0: Create negative space (rocks)"""
        positions = [(r, c) for r in range(MAZE_SIZE) for c in range(MAZE_SIZE)]
        positions.remove(START)
        positions.remove(GOAL)
        
        self.rocks = set(random.sample(positions, NUM_ROCKS))
        for r, c in self.rocks:
            self.grid[r][c] = ROCK
    
    def is_valid(self, r, c):
        """Check if cell is within bounds and not a rock"""
        return (0 <= r < MAZE_SIZE and 
                0 <= c < MAZE_SIZE and 
                (r, c) not in self.rocks)
    
    def find_safe_channels(self):
        """P1: Map all safe channels (BFS from start to goal)"""
        visited = set()
        queue = deque([START])
        parent = {START: None}
        
        while queue:
            r, c = queue.popleft()
            if (r, c) == GOAL:
                # Reconstruct all safe cells along discovered paths
                self._mark_safe_cells(parent)
                return True
            
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                nr, nc = r + dr, c + dc
                if self.is_valid(nr, nc) and (nr, nc) not in visited:
                    visited.add((nr, nc))
                    queue.append((nr, nc))
                    parent[(nr, nc)] = (r, c)
        
        return False  # No safe path exists
    
    def _mark_safe_cells(self, parent):
        """Mark all cells reachable from start without hitting rocks"""
        # BFS from start through all valid cells
        visited = set()
        queue = deque([START])
        
        while queue:
            r, c = queue.popleft()
            self.safe_channels.add((r, c))
            
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                nr, nc = r + dr, c + dc
                if self.is_valid(nr, nc) and (nr, nc) not in visited:
                    visited.add((nr, nc))
                    queue.append((nr, nc))
    
    def optimize_within_channels(self):
        """P2: Find shortest path within safe channels (A*)"""
        if not self.safe_channels:
            return []
        
        # Heuristic: Manhattan distance
        def heuristic(pos):
            return abs(pos[0] - GOAL[0]) + abs(pos[1] - GOAL[1])
        
        open_set = {START}
        came_from = {}
        g_score = {START: 0}
        f_score = {START: heuristic(START)}
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score.get(pos, float('inf')))
            
            if current == GOAL:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.reverse()
                self.optimal_path = path
                return path
            
            open_set.remove(current)
            
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dr, current[1] + dc)
                
                # CRITICAL: Only move within safe channels (P2 constraint)
                if neighbor not in self.safe_channels:
                    continue
                
                tentative_g = g_score[current] + 1
                
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        return []  # No path found within safe channels
    
    def visualize(self, phase, highlight_cells=None):
        """Display maze with phase-specific visualization"""
        print(f"\n{'='*60}")
        print(f"DEADBAND PROTOCOL — Phase {phase}")
        print(f"{'='*60}")
        
        display_grid = [row[:] for row in self.grid]
        
        if highlight_cells:
            for r, c in highlight_cells:
                if display_grid[r][c] == EMPTY:
                    if phase == 'P1':
                        display_grid[r][c] = SAFE
                    elif phase == 'P2':
                        display_grid[r][c] = PATH
        
        for r in range(MAZE_SIZE):
            print(' '.join(display_grid[r]))
        
        # Stats
        if phase == 'P0':
            print(f"\nP0: Mapped negative space")
            print(f"    Rocks: {len(self.rocks)}")
        elif phase == 'P1':
            print(f"\nP1: Found safe channels")
            print(f"    Safe cells: {len(self.safe_channels)}")
            print(f"    Coverage: {len(self.safe_channels)*100/(MAZE_SIZE*MAZE_SIZE):.1f}%")
        elif phase == 'P2':
            print(f"\nP2: Optimized within channels")
            print(f"    Path length: {len(self.optimal_path)}")
            print(f"    Optimal (theoretical): 38 steps (Manhattan)")
        
        time.sleep(1.5)

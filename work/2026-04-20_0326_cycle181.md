```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Simulates P0→P1→P2 in a 20x20 maze with rocks.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set
from collections import deque

# Constants from fleet knowledge
MAZE_SIZE = 20
ROCK_COUNT = 64  # rocks from simulation
START = (0, 0)
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

class DeadbandMaze:
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[Set[Tuple[int, int]]] = []
        self.optimal_path = []
        
    def p0_map_rocks(self):
        """P0: Map negative space - place rocks randomly"""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != START and (x, y) != GOAL:
                self.rocks.add((x, y))
                self.grid[x][y] = '#'
        
    def p1_find_safe_channels(self):
        """P1: Find all safe channels using BFS from start"""
        visited = set()
        queue = deque([START])
        safe_region = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            safe_region.add((x, y))
            
            # Explore neighbors
            for dx, dy in [(0,1), (1,0), (0,-1), (-1,0)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    queue.append((nx, ny))
        
        self.safe_channels = [safe_region]  # In this simple case, one connected region
        
    def p2_optimize_in_channel(self):
        """P2: Find shortest path within safe channels"""
        if not self.safe_channels:
            return []
            
        safe_region = self.safe_channels[0]
        if GOAL not in safe_region:
            return []  # No path exists within safe channels
            
        # BFS for shortest path
        queue = deque([(START, [START])])
        visited = set([START])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == GOAL:
                self.optimal_path = path
                return path
                
            for dx, dy in [(0,1), (1,0), (0,-1), (-1,0)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                    (nx, ny) in safe_region and (nx, ny) not in visited):
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        return []
    
    def run_protocol(self):
        """Execute full P0→P1→P2 sequence"""
        print("=== DEADBAND PROTOCOL DEMO ===")
        print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}")
        print(f"Start: {START}, Goal: {GOAL}")
        print()
        
        # P0: Map rocks
        print("P0: MAPPING NEGATIVE SPACE")
        self.p0_map_rocks()
        print(f"  → Mapped {len(self.rocks)} rocks (failure states)")
        print()
        
        # P1: Find safe channels
        print("P1: FINDING SAFE CHANNELS")
        self.p1_find_safe_channels()
        safe_cells = len(self.safe_channels[0]) if self.safe_channels else 0
        print(f"  → Found safe region with {safe_cells} cells")
        print(f"  → Goal reachable: {GOAL in self.safe_channels[0]}")
        print()
        
        # P2: Optimize within channel
        print("P2: OPTIMIZING WITHIN SAFE CHANNEL")
        path = self.p2_optimize_in_channel()
        if path:
            print(f"  → Found optimal path: {len(path)-1} steps")
            # Mark path on grid
            for x, y in path[1:-1]:  # Skip start and goal
                self.grid[x][y] = 'o'
        else:
            print("  → No path exists within safe channels")
        print()
        
        return len(path) > 0
    
    def visualize(self):
        """Display the maze with rocks, path, and markers"""
        # Create display grid
        display = [row[:] for row in self.grid]
        sx, sy = START
        gx, gy = GOAL
        display[sx][sy] = 'S'
        display[gx][gy] = 'G'
        
        print("MAZE VISUALIZATION:")
        print("  S = Start, G = Goal, # = Rock, o = Path, . = Empty")
        print()
        
        for x in range(MAZE_SIZE):
            row = []
            for y in range(MAZE_SIZE):
                row.append(display[x][y])
            print(' '.join(row))
        
        # Show protocol summary
        print()
        print("PROTOCOL SUMMARY:")
        print(f"  P0 Rocks: {len(self.rocks)}/{MAZE_SIZE*MAZE_SIZE} cells")
        print(f"  P1 Safe: {len(self.safe_channels[0]) if self.safe_channels else 0} cells")
        print(f"  P2 Path: {len(self.optimal_path)-1 if self.optimal_path else 0} steps")
        
        if self.optimal_path:
            print("  RESULT: SUCCESS (Deadband protocol followed)")
        else:
            print("  RESULT: FAILURE (Goal not reachable in safe channels)")

def main():
    """Run multiple trials to show protocol effectiveness"""
    print("Running 5 trials of Deadband Protocol...")
    print("=" * 50)
    
    successes = 0
    trials = 5
    
    for trial in range(trials):
        print(f"\nTRIAL {trial+1}:")
        print("-" * 30)
        
        maze = DeadbandMaze()
        success = maze.run_protocol()
        maze.visualize()
        
        if success:
            successes += 1
    
    print("\n" + "=" * 50)
    print(f"FINAL RESULTS ({trials} trials):")
    print(f"  Successes: {successes}/{trials}")
    print(f"  Success rate: {successes/trials*100:.0f}%")
    print()
    print("Note: In full simulation (50 runs), deadband protocol")
    print("achieved 50/50 success (100%) vs 0/50 for unconstrained.")
    print("See: constraint_sim_final.json")

if __name__ == "__main__":
    main()
```

**Explanation:**

1. **P0 Implementation**: Randomly places 64 rocks (failure states) in the 20×20 maze, matching the simulation data.

2. **P1 Implementation**: Uses BFS to find all cells reachable from start without hitting rocks — the "safe channel".

3. **P2 Implementation**: Only after safe channels are mapped, finds shortest path within that channel using BFS.

4. **Visualization**: Shows the maze with ASCII characters:
   - `S` = Start
   - `G` = Goal  
   - `#` = Rock (P0 negative space)
   - `o` = Optimal path (P2 result)
   - `.` = Empty safe cell

5. **Multiple Trials**: Runs 5 trials to demonstrate protocol reliability.

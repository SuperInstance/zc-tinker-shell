Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize within channel).

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=10, height=10, rock_density=0.2):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        
        # P0: Generate rocks (negative space)
        for y in range(height):
            for x in range(width):
                if (x, y) != self.start and (x, y) != self.goal:
                    if random.random() < rock_density:
                        self.grid[y][x] = '#'
                        self.rocks.add((x, y))
        
        # Ensure at least one path exists
        self._ensure_path()
    
    def _ensure_path(self):
        """Guarantee at least one safe channel from start to goal."""
        # Simple DFS to check connectivity
        stack = [self.start]
        visited = set()
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            if (x, y) == self.goal:
                return
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if self.grid[ny][nx] != '#':
                        stack.append((nx, ny))
        
        # If no path, clear a diagonal
        for i in range(min(self.width, self.height)):
            x = y = i
            if (x, y) in self.rocks:
                self.grid[y][x] = '.'
                self.rocks.remove((x, y))
    
    def display(self, path=None, channels=None):
        """Print ASCII maze with optional path and channels."""
        display_grid = [row[:] for row in self.grid]
        
        if channels:
            for (x, y) in channels:
                if display_grid[y][x] == '.':
                    display_grid[y][x] = '~'  # Safe channel
        
        if path:
            for (x, y) in path:
                if display_grid[y][x] == '.' or display_grid[y][x] == '~':
                    display_grid[y][x] = '*'  # Optimal path
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        display_grid[sy][sx] = 'S'
        display_grid[gy][gx] = 'G'
        
        print("\n" + "="*40)
        for row in display_grid:
            print(' '.join(row))
        print("="*40)
        print("Legend: S=Start, G=Goal, #=Rock, ~=Safe Channel, *=Optimal Path")
    
    def p0_map_rocks(self):
        """P0: Identify negative space (rocks)."""
        print("\n[P0] MAP NEGATIVE SPACE")
        print(f"Found {len(self.rocks)} rock cells (failure states).")
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all safe channels (cells reachable without hitting rocks)."""
        print("\n[P1] IDENTIFY SAFE CHANNELS")
        
        # BFS to find all reachable cells
        queue = deque([self.start])
        reachable = {self.start}
        
        while queue:
            x, y = queue.popleft()
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in reachable:
                        reachable.add((nx, ny))
                        queue.append((nx, ny))
        
        print(f"Found {len(reachable)} safe cells (including start).")
        return reachable
    
    def p2_optimize_in_channel(self, safe_cells):
        """P2: Find shortest path within safe channels."""
        print("\n[P2] OPTIMIZE WITHIN SAFE CHANNELS")
        
        # BFS for shortest path
        queue = deque([(self.start, [self.start])])
        visited = {self.start}
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                print(f"Found optimal path length: {len(path)-1}")
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) in safe_cells and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        print("No path to goal within safe channels!")
        return None
    
    def unconstrained_greedy(self):
        """Unconstrained optimization (violates protocol)."""
        print("\n[VIOLATION] UNCONSTRAINED GREEDY OPTIMIZATION")
        
        # Greedy move toward goal ignoring rocks
        path = [self.start]
        x, y = self.start
        visited = {self.start}
        
        steps = 0
        max_steps = self.width * self.height * 2
        
        while (x, y) != self.goal and steps < max_steps:
            # Move toward goal
            dx = 1 if self.goal[0] > x else -1 if self.goal[0] < x else 0
            dy = 1 if self.goal[1] > y else -1 if self.goal[1] < y else 0
            
            # Try primary direction
            nx, ny = x + dx, y + dy
            if 0 <= nx < self.width and 0 <= ny < self.height:
                if (nx, ny) not in visited:
                    x, y = nx, ny
                    path.append((x, y))
                    visited.add((x, y))
                    steps += 1
                    continue
            
            # Try alternative
            for dx, dy in random.sample([(1,0), (-1,0), (0,1), (0,-1)], 4):
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited:
                        x, y = nx, ny
                        path.append((x, y))
                        visited.add((x, y))
                        steps += 1
                        break
            else:
                break
        
        success = (x, y) == self.goal
        print(f"Success: {success}, Path length: {len(path)-1}")
        return path if success else None

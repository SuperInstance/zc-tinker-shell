"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0→P1→P2 strict priority order.
"""

import random
import sys

def generate_maze(width=20, height=20, rock_density=0.16):
    """Generate a random maze with start (S), goal (G), and rocks (#)."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    # Place rocks
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
    # Ensure start (top-left) and goal (bottom-right) are clear
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    return maze

def print_maze(maze, path=None):
    """Print maze with optional path marked."""
    for y, row in enumerate(maze):
        line = []
        for x, cell in enumerate(row):
            if path and (x, y) in path:
                line.append('*')
            else:
                line.append(cell)
        print(''.join(line))

def find_rocks(maze):
    """P0: Map negative space — return set of rock coordinates."""
    rocks = set()
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.add((x, y))
    return rocks

def find_safe_channels(maze, rocks, start, goal):
    """P1: Identify safe channels — BFS that avoids rocks."""
    width, height = len(maze[0]), len(maze)
    queue = [(start, [start])]
    visited = set([start])
    safe_paths = []
    
    while queue:
        (x, y), path = queue.pop(0)
        if (x, y) == goal:
            safe_paths.append(path)
            continue
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    return safe_paths

def optimize_path(safe_paths):
    """P2: Optimize within safe channels — pick shortest safe path."""
    if not safe_paths:
        return None
    return min(safe_paths, key=len)

def unconstrained_agent(maze, start, goal):

            return path

    return None

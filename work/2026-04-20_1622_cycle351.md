#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 351)

A visual ASCII simulation of the Deadband Protocol (P0→P1→P2) in a 10x10 maze.
Demonstrates the strict priority: map rocks (P0), find safe channels (P1),
then optimize path (P2) within safe channels only.

Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set

# ====================
# MAZE CONFIGURATION
# ====================
MAZE_WIDTH = 10
MAZE_HEIGHT = 10
START = (0, 0)
GOAL = (9, 9)
ROCK_COUNT = 15  # ~15% rocks, similar to constraint_sim_final.json density

# Symbols for visualization
SYM_EMPTY = '·'
SYM_ROCK = '█'
SYM_START = 'S'
SYM_GOAL = 'G'
SYM_PATH = '○'
SYM_SAFE = ' '  # inside deadband channel

# ====================
# DEADBAND PROTOCOL IMPLEMENTATION
# ====================

class DeadbandMaze:
    def __init__(self):
        self.grid = [[SYM_EMPTY for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
        # Initialize maze
        self.grid[START[1]][START[0]] = SYM_START
        self.grid[GOAL[1]][GOAL[0]] = SYM_GOAL
        self._place_rocks()
    
    def _place_rocks(self):
        """P0: Generate rocks (negative space) randomly, avoiding start/goal."""
        possible = [(x, y) for x in range(MAZE_WIDTH) for y in range(MAZE_HEIGHT)]
        possible.remove(START)
        possible.remove(GOAL)
        
        self.rocks = set(random.sample(possible, ROCK_COUNT))
        for x, y in self.rocks:
            self.grid[y][x] = SYM_ROCK
    
    def _find_safe_channels(self):
        """P1: Find all paths from start to goal that avoid rocks (safe channels)."""
        def dfs(pos: Tuple[int, int], visited: Set[Tuple[int, int]], path: List[Tuple[int, int]]):
            if pos == GOAL:
                self.safe_channels.append(path.copy())
                return
            
            x, y = pos
            # 4-directional movement
            neighbors = [(x+1, y), (x-1, y), (x, y+1), (x, y-1)]
            for nx, ny in neighbors:
                if (0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT and
                    (nx, ny) not in self.rocks and
                    (nx, ny) not in visited):
                    visited.add((nx, ny))
                    path.append((nx, ny))
                    dfs((nx, ny), visited, path)
                    path.pop()
                    visited.remove((nx, ny))
        
        dfs(START, {START}, [START])
    
    def _optimize_within_channels(self):
        """P2: Select shortest path among safe channels."""
        if not self.safe_channels:
            return
        
        self.optimal_path = min(self.safe_channels, key=len)
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2."""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===\n")
        
        # P0: Map negative space (already done in init)
        print("P0: MAP NEGATIVE SPACE (ROCKS)")
        print(f"  Identified {len(self.rocks)} rock positions")
        self.print_maze()
        print()
        
        # P1: Find safe channels
        print("P1: FIND SAFE CHANNELS")
        self._find_safe_channels()
        print(f"  Found {len(self.safe_channels)} safe channels from S to G")
        if self.safe_channels:
            print(f"  Channel lengths: {[len(c)-1 for c in self.safe_channels]}")
        print()
        
        # P2: Optimize within channels
        print("P2: OPTIMIZE WITHIN SAFE CHANNELS")
        self._optimize_within_channels()
        if self.optimal_path:
            print(f"  Selected optimal path length: {len(self.optimal_path)-1} steps")
            # Mark path on grid for visualization
            for x, y in self.optimal_path[1:-1]:  # exclude start/goal
                if self.grid[y][x] == SYM_EMPTY:
                    self.grid[y][x] = SYM_PATH
        else:
            print("  WARNING: No safe channels found!")
        print()
    
    def print_maze(self):
        """Print current maze state."""
        print("   " + " ".join(str(i) for i in range(MAZE_WIDTH)))
        for y in range(MAZE_HEIGHT):
            row = f"{y:2} "
            for x in range(MAZE_WIDTH):
                row += self.grid[y][x] + " "
            print(row)
    
    def print_statistics(self):
        """Print protocol performance stats."""
        print("=== PROTOCOL STATISTICS ===")
        print(f"Rocks (P0 negative space): {len(self.rocks)}")
        print(f"Safe channels found (P1): {len(self.safe_channels)}")
        if self.optimal_path:
            print(f"Optimal path length (P2): {len(self.optimal_path)-1} steps")
            success = "SUCCESS" if self.optimal_path else "FAILURE"
            print(f"Result: {success}")
        else:
            print("Result: FAILURE (no safe channels)")
        
        # Compare with unconstrained greedy approach (hypothetical)
        print("\nHypothetical unconstrained greedy (P2-only):")
        print("  Would hit rocks with probability → 1.0")
        print("  (as shown in constraint_sim_final.json: 0/50 success)")

# ====================
# DEMONSTRATION
# ====================

def main():
    print("Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json\n")
    
    # Set seed for reproducible demonstration
    random.seed(42)
    
    # Create and run maze
    maze = DeadbandMaze()
    maze.run_protocol()
    
    print("FINAL MAZE WITH OPTIMAL PATH:")
    maze.print_maze()
    print()
    
    maze.print_statistics()
    
    print("\n" + "="*50)
    print("PROTOCOL VALIDATION:")
    print("- P0 completed: Rocks mapped before any pathfinding")
    print("- P1 completed: All safe channels enumerated")
    print("- P2 completed: Optimization constrained to safe channels only")
    print("- Strict P0→P1→P2 priority maintained")
    print("="*50)

if __name__ == "__main__":
    main()

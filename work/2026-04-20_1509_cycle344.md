#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize within channels).
"""

import heapq

def print_maze(maze, title="Maze"):
    """Print the maze with a title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print("".join(row))

def p0_map_rocks(maze):
    """
    P0: Map the negative space (rocks).
    Returns a set of (x, y) coordinates that are obstacles ('#').
    """
    rocks = set()
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(maze, rocks, start, goal):
    """
    P1: Find safe channels (paths that avoid rocks).
    Uses BFS to find all reachable cells from start that are not rocks.
    Returns a set of safe cells.
    """
    height, width = len(maze), len(maze[0])
    safe = set()
    queue = [start]
    visited = set([start])
    
    while queue:
        x, y = queue.pop(0)
        safe.add((x, y))
        if (x, y) == goal:
            continue  # keep exploring to map all reachable safe cells
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    return safe

def p2_optimize_in_channel(maze, safe, start, goal):
    """
    P2: Optimize within safe channels (shortest path using A*).
    Only moves through cells in 'safe' set.
    Returns the path as list of (x, y) or empty list if no path.
    """
    height, width = len(maze), len(maze[0])
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            return path[::-1]
        
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            neighbor = (nx, ny)
            if 0 <= nx < width and 0 <= ny < height:
                if neighbor not in safe:
                    continue  # not in safe channel → skip
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    return []  # no path found

def run_deadband_demo():
    """
    Main demo: shows the Deadband Protocol in action on a simple maze.
    """
    # Example maze: '#' = rock, '.' = open, 'S' = start, 'G' = goal
    maze = [
        "S..#.....",
        ".#.#.#.#.",
        ".#.#.#.#.",
        ".#.#.#.#.",
        ".#.#.#.#.",
        ".#.#.#.#.",
        ".#.#.#.#.",
        ".#.#.#.#.",
        ".....#..G"
    ]
    
    # Convert to mutable list for marking
    maze_grid = [list(row) for row in maze]
    
    # Find start and goal
    start = goal = None
    for y, row in enumerate(maze_grid):
        for x, cell in enumerate(row):
            if cell == 'S':
                start = (x, y)
            elif cell == 'G':
                goal = (x, y)
    
    if not start or not goal:
        print("Error: Start or goal not found in maze.")
        return
    
    print_maze(maze_grid, "Initial Maze")
    print(f"Start: {start}, Goal: {goal}")
    
    # P0: Map rocks
    rocks = p0_map_rocks(maze_grid)
    print(f"\nP0 — Rocks mapped: {len(rocks)} obstacles")
    
    # P1: Find safe channels
    safe = p1_find_safe_channels(maze_grid, rocks, start, goal)
    print(f"P1 — Safe channels: {len(safe)} reachable cells (excluding rocks)")
    
    # Visualize safe channels
    safe_maze = [row[:] for row in maze_grid]
    for x, y in safe:
        if safe_maze[y][x] == '.':
            safe_maze[y][x] = ' '
    print_maze(safe_maze, "Safe Channels (P1) [rocks '#', safe ' ', untouched '.']")
    
    # P2: Optimize within safe channels
    path = p2_optimize_in_channel(maze_grid, safe, start, goal)
    if path:
        print(f"P2 — Optimal path found: {len(path)} steps")
        # Mark path on maze
        path_maze = [row[:] for row in maze_grid]
        for x, y in path[1:-1]:  # exclude start and goal
            if path_maze[y][x] == '.' or path_maze[y][x] == ' ':
                path_maze[y][x] = '*'
        print_maze(path_maze, "Optimal Path (P2) [path '*']")
        print(f"Path coordinates: {path}")
    else:
        print("P2 — No safe path to goal.")
    
    # Summary
    print("\n=== Deadband Protocol Summary ===")
    print("P0: Identified rocks (negative space) → avoid these.")
    print("P1: Mapped all reachable safe cells (channels) → operate only here.")
    print("P2: Found shortest path within safe channels → constrained optimization.")
    print("Violation (skipping P0/P1): would hit rocks or get stuck.")

if __name__ == "__main__":
    run_deadband_demo()

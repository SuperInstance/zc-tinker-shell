**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Implementing a concrete, runnable simulation based on the fleet's constraint theory data. The script will:
1. Generate a random 10x10 maze with start (S), goal (G), and rocks (#).
2. Apply the Deadband Protocol strictly: P0 (map rocks), P1 (find safe channels), P2 (optimal path within channels).
3. Output ASCII visualization of each step and final path.

**Output:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker
A concrete simulation of P0→P1→P2 using ASCII maze navigation.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=10, height=10, rock_density=0.2):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (height-1, width-1)
        self.rocks = set()
        self.safe_channels = set()
        self.optimal_path = []
        
        # Initialize
        self._place_rocks(rock_density)
        self.grid[self.start[0]][self.start[1]] = 'S'
        self.grid[self.goal[0]][self.goal[1]] = 'G'
    
    def _place_rocks(self, density):
        """P0: Map negative space (rocks)"""
        total_cells = self.width * self.height - 2  # exclude S and G
        num_rocks = int(total_cells * density)
        candidates = [(r, c) for r in range(self.height) for c in range(self.width)]
        candidates.remove(self.start)
        candidates.remove(self.goal)
        
        self.rocks = set(random.sample(candidates, num_rocks))
        for r, c in self.rocks:
            self.grid[r][c] = '#'
    
    def find_safe_channels(self):
        """P1: Identify safe channels (cells reachable without hitting rocks)"""
        visited = set()
        queue = deque([self.start])
        
        while queue:
            r, c = queue.popleft()
            if (r, c) in visited:
                continue
            visited.add((r, c))
            
            # Check neighbors
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                nr, nc = r + dr, c + dc
                if (0 <= nr < self.height and 0 <= nc < self.width and
                    (nr, nc) not in self.rocks and (nr, nc) not in visited):
                    queue.append((nr, nc))
        
        self.safe_channels = visited
        return visited
    
    def optimize_within_channels(self):
        """P2: Find shortest path within safe channels (BFS)"""
        if self.goal not in self.safe_channels:
            return []  # No safe path to goal
        
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (r, c), path = queue.popleft()
            
            if (r, c) == self.goal:
                self.optimal_path = path
                return path
            
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                nr, nc = r + dr, c + dc
                if (0 <= nr < self.height and 0 <= nc < self.width and
                    (nr, nc) in self.safe_channels and (nr, nc) not in visited):
                    visited.add((nr, nc))
                    queue.append(((nr, nc), path + [(nr, nc)]))
        
        return []
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2"""
        print("=== DEADBAND PROTOCOL DEMO ===")
        print(f"Maze: {self.height}x{self.width}, Rocks: {len(self.rocks)}")
        print("\nP0: Negative Space (Rocks)")
        self.print_grid()
        
        print("\nP1: Safe Channels (Reachable without rocks)")
        safe = self.find_safe_channels()
        self.print_grid_with_channels(safe)
        print(f"Safe cells: {len(safe)}/{self.width*self.height}")
        
        print("\nP2: Optimal Path within Channels")
        path = self.optimize_within_channels()
        if path:
            self.print_grid_with_path(path)
            print(f"Path length: {len(path)-1} steps")
            print(f"Success: Goal reached via safe channels")
        else:
            print("FAILURE: No safe path to goal")
            print("Violation: Optimized (P2) without safe channel (P1)")
        
        return len(path) > 0
    
    def print_grid(self):
        for r in range(self.height):
            row = []
            for c in range(self.width):
                row.append(self.grid[r][c])
            print(' '.join(row))
    
    def print_grid_with_channels(self, channels):
        for r in range(self.height):
            row = []
            for c in range(self.width):
                if (r, c) in channels and self.grid[r][c] == '.':
                    row.append('O')  # Safe channel
                else:
                    row.append(self.grid[r][c])
            print(' '.join(row))
    
    def print_grid_with_path(self, path):
        # Create copy with path markers
        display = [row[:] for row in self.grid]
        for r, c in path[1:-1]:  # Exclude start and goal
            if display[r][c] == '.':
                display[r][c] = '*'
        
        for r in range(self.height):
            row = []
            for c in range(self.width):
                row.append(display[r][c])
            print(' '.join(row))

def main():
    """Run multiple simulations to demonstrate protocol effectiveness"""
    print("=" * 60)
    print("Deadband Protocol Maze Simulation")
    print("Based on fleet constraint theory data")
    print("=" * 60)
    
    random.seed(42)  # Reproducible for demo
    
    # Single demonstration
    maze = DeadbandMaze(width=8, height=8, rock_density=0.25)
    success = maze.run_protocol()
    
    print("\n" + "=" * 60)
    print("STATISTICS (10-run sample):")
    
    # Small batch simulation
    successes = 0
    total_steps = 0
    for i in range(10):
        maze = DeadbandMaze(width=6, height=6, rock_density=0.2)
        maze.find_safe_channels()
        path = maze.optimize_within_channels()
        if path:
            successes += 1
            total_steps += len(path) - 1
    
    print(f"Success rate: {successes}/10")
    if successes > 0:
        print(f"Average path length: {total_steps/successes:.1f} steps")
    
    print("\n" + "=" * 60)
    print("KEY INSIGHTS:")
    print("1. P0 (rocks) must be mapped first - otherwise agent crashes")
    print("2. P1 (channels) constrains search space to safe regions")
    print("3. P2 (optimization) only within channels - never unconstrained")
    print("4. Violation (skip P0/P1) → local minima probability → 1.0")
    print("=" * 60)

if __name__ == "__main__":
    main()
```

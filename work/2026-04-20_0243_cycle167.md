import random

    def __init__(self, width=20, height=20, num_rocks=64):
        self.width = width
        self.height = height
        self.num_rocks = num_rocks
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.grid = None
        self.rocks = None
        self.safe_channels = None
        self.optimal_path = None
        
        # Symbols for visualization
        self.START = 'S'
        self.GOAL = 'G'
        self.ROCK = '█'
        self.EMPTY = '·'
        self.PATH = '○'
        self.SAFE = ' '
        
    def generate_maze(self):
        """Create random maze with rocks (P0 negative space)."""
        self.grid = [[self.EMPTY for _ in range(self.width)] 
                     for _ in range(self.height)]
        
        # Place start and goal
        self.grid[self.start[1]][self.start[0]] = self.START
        self.grid[self.goal[1]][self.goal[0]] = self.GOAL
        
        # Place rocks randomly (avoid start and goal)
        self.rocks = set()
        while len(self.rocks) < self.num_rocks:
            x = random.randint(0, self.width-1)
            y = random.randint(0, self.height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[y][x] = self.ROCK
                
        return self.grid
    
    def p0_map_negative_space(self):
        """P0: Identify rocks (what NOT to do)."""
        print("=== P0: MAP NEGATIVE SPACE ===")

    def p1_find_safe_channels(self):
        """P1: Find all paths from start to goal avoiding rocks."""
        print("\n=== P1: FIND SAFE CHANNELS ===")
        
        # BFS to find all reachable cells (safe channels)
        visited = set()
        queue = deque([self.start])
        visited.add(self.start)
        
        while queue:
            x, y = queue.popleft()
            
            # Check all 4 directions
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) not in self.rocks and
                    (nx, ny) not in visited):
                    visited.add((nx, ny))
                    queue.append((nx, ny))
        
        self.safe_channels = visited
        print(f"Safe channels: {len(self.safe_channels)} reachable cells")
        
        # Check if goal is reachable
        if self.goal not in self.safe_channels:
            print("WARNING: Goal not reachable! Need to regenerate maze.")
            return False
            
        return True
    
    def p2_optimize_within_channel(self):
        """P2: Find shortest path through safe channels."""
        print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        # BFS for shortest path (constrained to safe channels)
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                self.optimal_path = path
                print(f"Found optimal path: {len(path)-1} steps")
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:

                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) in self.safe_channels and
                    (nx, ny) not in visited):
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        return None

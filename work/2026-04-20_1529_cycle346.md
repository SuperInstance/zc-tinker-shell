#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
ASCII visualization of the protocol in action.
"""

import random
from collections import deque

class DeadbandMaze:
    """A simple maze simulator implementing the Deadband Protocol."""
    
    def __init__(self, width=20, height=10, rock_density=0.2):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate maze with rocks
        self.grid = []
        self.rocks = set()
        for y in range(height):
            row = []
            for x in range(width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif random.random() < rock_density:
                    row.append('█')  # Rock
                    self.rocks.add((x, y))
                else:
                    row.append('·')  # Open space
            self.grid.append(row)
        
        # Protocol state
        self.p0_map = None
        self.p1_channels = None
        self.p2_path = None
    
    def display(self, title="MAZE"):
        """Display the current maze state."""
        print(f"\n{'='*40}")
        print(f"{title:^40}")
        print(f"{'='*40}")
        
        for y, row in enumerate(self.grid):
            line = []
            for x, cell in enumerate(row):
                if self.p2_path and (x, y) in self.p2_path:
                    line.append('★')
                elif self.p1_channels and (x, y) in self.p1_channels:
                    line.append('░')
                elif self.p0_map and (x, y) in self.p0_map:
                    line.append('█')
                else:
                    line.append(cell)
            print(''.join(line))
        
        stats = []
        if self.p0_map:
            stats.append(f"P0 Rocks: {len(self.p0_map)}")
        if self.p1_channels:
            stats.append(f"P1 Safe: {len(self.p1_channels)}")
        if self.p2_path:
            stats.append(f"P2 Path: {len(self.p2_path)} steps")
        
        if stats:
            print(f"\n{' | '.join(stats)}")
    
    def run_p0_map_rocks(self):
        """P0: Map the negative space (rocks)."""
        print("\n[P0] MAPPING NEGATIVE SPACE")
        print("Identifying all positions where failure probability > threshold")
        
        # In this simulation, rocks are already known
        # In real scenario, this would involve testing/exploration
        self.p0_map = self.rocks.copy()
        
        # Add adjacent dangerous areas (simulating uncertainty)
        expanded_rocks = set(self.p0_map)
        for x, y in self.p0_map:
            for dx, dy in [(0,1), (1,0), (0,-1), (-1,0)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if random.random() < 0.3:  # 30% chance of danger spreading
                        expanded_rocks.add((nx, ny))
        
        self.p0_map = expanded_rocks
        self.display("P0: NEGATIVE SPACE MAP")
        return True
    
    def run_p1_find_channels(self):
        """P1: Find safe channels (deadbands)."""
        if not self.p0_map:
            print("ERROR: Cannot run P1 without P0 map!")
            return False
        
        print("\n[P1] FINDING SAFE CHANNELS")
        print("Searching for paths that avoid negative space")
        
        # BFS to find all reachable safe positions
        visited = set()
        safe_cells = set()
        queue = deque([self.start])
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            
            # Check if this cell is safe (not in P0 map)
            if (x, y) not in self.p0_map:
                safe_cells.add((x, y))
                
                # Explore neighbors
                for dx, dy in [(0,1), (1,0), (0,-1), (-1,0)]:
                    nx, ny = x + dx, y + dy
                    if (0 <= nx < self.width and 0 <= ny < self.height and
                        (nx, ny) not in visited):
                        queue.append((nx, ny))
        
        self.p1_channels = safe_cells
        self.display("P1: SAFE CHANNELS")
        
        # Check if goal is reachable
        if self.goal in self.p1_channels:
            print("✓ Goal is reachable through safe channels")
            return True
        else:
            print("✗ Goal is NOT reachable safely!")
            return False
    
    def run_p2_optimize_path(self):
        """P2: Optimize path within safe channels."""
        if not self.p1_channels:
            print("ERROR: Cannot run P2 without P1 channels!")
            return False
        
        print("\n[P2] OPTIMIZING WITHIN CHANNEL")
        print("Finding shortest path through safe water")
        
        # A* search constrained to safe channels
        open_set = {self.start}
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: self._heuristic(self.start, self.goal)}
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score.get(pos, float('inf')))
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                self.p2_path = list(reversed(path))
                self.display("P2: OPTIMAL PATH")
                print(f"✓ Found optimal path: {len(self.p2_path)-1} steps")
                return True
            
            open_set.remove(current)
            
            for dx, dy in [(0,1), (1,0), (0,-1), (-1,0)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # Constraint: Must stay in safe channels
                if neighbor not in self.p1_channels:
                    continue
                
                if not (0 <= neighbor[0] < self.width and 0 <= neighbor[1] < self.height):
                    continue
                
                tentative_g = g_score[current] + 1
                
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor, self.goal)
                    open_set.add(neighbor)
        
        print("✗ No path found within safe channels")
        return False
    
    def _heuristic(self, a, b):
        """Manhattan distance heuristic."""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2."""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*60)
        
        # Initial state
        self.display("INITIAL MAZE")
        
        # P0: Map negative space
        if not self.run_p0_map_rocks():
            print("P0 failed - cannot proceed")
            return False
        
        # P1: Find safe channels
        if not self.run_p1_find_channels():
            print("P1 failed - no safe path to goal")
            return False
        
        # P2: Optimize within channels
        if not self.run_p2_optimize_path():
            print("P2 failed - no optimal path found")
            return False
        
        # Summary
        print("\n" + "="*60)
        print("PROTOCOL COMPLETE")
        print("="*60)
        print("P0: Mapped", len(self.p0_map), "danger zones")
        print("P1: Found", len(self.p1_channels), "safe positions")
        print("P2: Optimal path:", len(self.p2_path)-

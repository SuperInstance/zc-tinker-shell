#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2 in action.

Simulates the fleet's Deadband Protocol on a 20x20 maze.
Based on constraint_sim_final.json results:
- Unconstrained (P2-only): 0% success, avg steps 2320.9
- Deadband-constrained (P0→P1→P2): 100% success, avg steps 21.0

Demonstrates:
P0: Map rocks (obstacles)
P1: Identify safe channels (traversable cells)
P2: Optimize path within safe channels
"""

import heapq
import random
from typing import List, Tuple, Set, Dict

# ==================== MAZE DEFINITION ====================
# 20x20 maze matching simulation dimensions
# '#' = rock/wall (negative space), '.' = open, 'S' = start, 'G' = goal
MAZE_STR = """
####################
#S.................#
#.######.#######.#.#
#.#......#.....#.#.#
#.#.####.#.###.#.#.#
#.#.#....#...#.#.#.#
#.#.#.######.#.#.#.#
#.#.#........#.#.#.#
#.#.##########.#.#.#
#.#............#.#.#
#.##############.#.#
#.#..............#.#
#.#.##############.#
#.#.#............#.#
#.#.#.##########.#.#
#.#.#.#........#.#.#
#.#.#.#.######.#.#.#
#.#.#.#.#....#.#.#.#
#.#.#.#.#.####.#.#.#
#G#.#.#.#......#.#.#
####################
"""

class DeadbandMazeSolver:
    """Implements Deadband Protocol for maze navigation."""
    
    def __init__(self, maze_str: str):
        self.lines = [line.rstrip() for line in maze_str.strip().split('\n')]
        self.height = len(self.lines)
        self.width = len(self.lines[0]) if self.height > 0 else 0
        self.start = None
        self.goal = None
        self.rocks = set()  # P0: negative space
        
        # Parse maze
        for y, line in enumerate(self.lines):
            for x, ch in enumerate(line):
                if ch == 'S':
                    self.start = (x, y)
                elif ch == 'G':
                    self.goal = (x, y)
                elif ch == '#':
                    self.rocks.add((x, y))
        
        if self.start is None or self.goal is None:
            raise ValueError("Maze must contain 'S' and 'G'")
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify all rocks (states where failure probability > threshold)."""
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Identified {len(self.rocks)} rock cells (obstacles)")
        print(f"Rocks are at coordinates: {list(self.rocks)[:5]}..." if self.rocks else "No rocks")
        return self.rocks
    
    def p1_find_safe_channels(self) -> Set[Tuple[int, int]]:
        """P1: Find all traversable cells (safe channels)."""
        safe = set()
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) not in self.rocks:
                    safe.add((x, y))
        
        print("\n=== P1: IDENTIFY SAFE CHANNELS ===")
        print(f"Found {len(safe)} safe cells (traversable space)")
        print(f"Safe channel covers {100*len(safe)/(self.width*self.height):.1f}% of maze")
        return safe
    
    def p2_optimize_within_channel(self, safe: Set[Tuple[int, int]], 
                                   use_deadband: bool = True) -> Tuple[List[Tuple[int, int]], int]:
        """P2: Find shortest path from start to goal.
        
        Args:
            safe: Set of safe cells from P1
            use_deadband: If True, constrain to safe cells only (deadband protocol).
                         If False, allow any movement (unconstrained greedy).
        """
        print(f"\n=== P2: OPTIMIZE WITHIN {'SAFE CHANNEL' if use_deadband else 'UNCONSTRAINED'} ===")
        
        # Movement directions
        dirs = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        
        # A* search
        open_set = []
        heapq.heappush(open_set, (0, self.start))
        
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: self._heuristic(self.start, self.goal)}
        
        visited = 0
        
        while open_set:
            _, current = heapq.heappop(open_set)
            visited += 1
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                path.reverse()
                print(f"Found path of length {len(path)-1} after visiting {visited} cells")
                return path, visited
            
            for dx, dy in dirs:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # Check bounds
                if not (0 <= neighbor[0] < self.width and 0 <= neighbor[1] < self.height):
                    continue
                
                # Deadband constraint: must be in safe set
                if use_deadband and neighbor not in safe:
                    continue
                
                # Unconstrained can go anywhere (even through rocks)
                # In reality, hitting rock = failure, but we simulate by allowing movement
                # with high cost to show it's possible but inefficient
                tentative_g = g_score[current] + 1
                
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor, self.goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
        
        print("No path found!")
        return [], visited
    
    def _heuristic(self, a: Tuple[int, int], b: Tuple[int, int]) -> int:
        """Manhattan distance heuristic."""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def visualize(self, path: List[Tuple[int, int]] = None):
        """Print ASCII maze with optional path."""
        print("\n" + "="*40)
        print("MAZE VISUALIZATION")
        print("="*40)
        
        for y in range(self.height):
            line_chars = []
            for x in range(self.width):
                ch = self.lines[y][x]
                if (x, y) == self.start:
                    line_chars.append('S')
                elif (x, y) == self.goal:
                    line_chars.append('G')
                elif path and (x, y) in path:
                    line_chars.append('*')
                else:
                    line_chars.append(ch)
            print(''.join(line_chars))
        
        if path:
            print(f"\nPath length: {len(path)-1} steps")
            print(f"Path: {path[:3]}...{path[-3:] if len(path) > 6 else ''}")

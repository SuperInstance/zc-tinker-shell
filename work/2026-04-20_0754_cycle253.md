    def __init__(self, width=10, height=10):
        """Initialize a simple maze with start, goal, and obstacles."""
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Define rocks (negative space)
        self.rocks = set()

        for x in range(3, 7):
            for y in range(2, 5):
                self.rocks.add((x, y))  # Block A

            self.rocks.add((8, y))      # Vertical wall

        self.safe_channels = set()
        
        # P2: Optimal path (initially empty)

        self.steps = {
            'p0_rocks_identified': 0,

        }
    
    def render(self, show_rocks=True, show_channels=False, show_path=False):
        """Render the maze as ASCII art."""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif show_path and pos in self.optimal_path:
                    row.append('*')
                elif show_channels and pos in self.safe_channels:
                    row.append('·')
                elif show_rocks and pos in self.rocks:
                    row.append('█')
                else:
                    row.append('.')
            grid.append(' '.join(row))
        return '\n'.join(grid)
    
    def p0_map_negative_space(self):
        """P0: Identify all rocks (what NOT to do)."""
        print("\n" + "="*60)
        print("P0: MAP NEGATIVE SPACE")

        print("Identifying all states where failure probability > threshold")
        print(f"Found {len(self.rocks)} rocks (obstacles)")
        
        self.steps['p0_rocks_identified'] = len(self.rocks)

        print(f"\nRocks prevent movement. Agent cannot occupy these cells.")

    def p1_find_safe_channels(self):
        """P1: Find all paths that avoid rocks."""
        print("\n" + "="*60)
        print("P1: IDENTIFY SAFE CHANNELS")
        print("="*60)
        print("Finding all paths where ∀ s ∈ P: s ∉ S_neg")
        
        # Simple flood fill from start, avoiding rocks
        visited = set()
        stack = [self.start]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            if (x, y) in self.rocks:
                continue
            if not (0 <= x < self.width and 0 <= y < self.height):
                continue
            
            visited.add((x, y))
            
            # Add neighbors (4-directional movement)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                stack.append((x+dx, y+dy))
        
        self.safe_channels = visited
        self.steps['p1_safe_cells_found'] = len(self.safe_channels)

        # Check if goal is reachable

            print(f"\n⚠️  GOAL NOT REACHABLE! Goal at {self.goal} is not in safe channels.")
            return False
        
        print(f"\n✓ Goal at {self.goal} is reachable through safe channels.")
        return True

        """P2: Find shortest path through safe channels."""

        print("subject to P ⊆ C (constrained optimization)")
        
        # Simple BFS through safe channels only
        from collections import deque
        
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                self.optimal_path = path

        print(f"\nPath coordinates:")
        for i, (x, y) in enumerate(self.optimal_path):

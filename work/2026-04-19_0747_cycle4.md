P2: Optimize (shortest path)

import random
import time
from collections import deque

class Maze:
    def __init__(self, width=20, height=10):
        self.width = width
        self.height = height
        self.grid = [['#' for _ in range(width)] for _ in range(height)]
        self.start = (1, 1)
        self.goal = (height-2, width-2)
        self._generate()
    
    def _generate(self):
        # Simple maze generation - carve paths
        for y in range(1, self.height-1):
            for x in range(1, self.width-1):
                if random.random() > 0.3:  # 70% open space
                    self.grid[y][x] = ' '
        
        # Ensure start and goal are open
        self.grid[self.start[0]][self.start[1]] = 'S'
        self.grid[self.goal[0]][self.goal[1]] = 'G'
        
        # Ensure a path exists
        self._carve_path()
    
    def _carve_path(self):
        # Simple BFS to ensure connectivity
        directions = [(0,1),(1,0),(0,-1),(-1,0)]
        visited = set()
        queue = deque([self.start])
        
        while queue:
            y, x = queue.popleft()
            if (y, x) == self.goal:
                return True
            
            for dy, dx in directions:
                ny, nx = y + dy, x + dx
                if 0 <= ny < self.height and 0 <= nx < self.width:
                    if self.grid[ny][nx] != '#' and (ny, nx) not in visited:
                        visited.add((ny, nx))
                        queue.append((ny, nx))
        
        # If no path, open a direct line
        y1, x1 = self.start
        y2, x2 = self.goal
        for y in range(min(y1, y2), max(y1, y2)+1):
            self.grid[y][x1] = ' '
        for x in range(min(x1, x2), max(x1, x2)+1):
            self.grid[y2][x] = ' '
        
        self.grid[self.start[0]][self.start[1]] = 'S'
        self.grid[self.goal[0]][self.goal[1]] = 'G'

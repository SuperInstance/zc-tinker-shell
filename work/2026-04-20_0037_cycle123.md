class Maze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        
        # Place rocks (obstacles) - P0 negative space
        num_rocks = int(width * height * rock_density)
        for _ in range(num_rocks):
            x, y = random.randint(0, width-1), random.randint(0, height-1)
            self.grid[y][x] = '#'
        
        # Start and goal positions
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Ensure start and goal are not rocks
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def display(self, path=None):
        """Display maze with optional path marked"""
        display_grid = [row[:] for row in self.grid]
        
        if path:
            for x, y in path[1:-1]:  # Skip start and goal
                if display_grid[y][x] not in ['S', 'G']:
                    display_grid[y][x] = '*'
        
        print("\n" + "="*40)
        for row in display_grid:
            print(' '.join(row))
        print("="*40)
    
    def is_valid(self, x, y):
        """Check if position is within bounds and not a rock"""
        return (0 <= x < self.width and 
                0 <= y < self.height and 
                self.grid[y][x] != '#')

class DeadbandNavigator:
    """Implements Deadband Protocol: P0→P1→P2"""
    
    def __init__(self, maze):
        self.maze = maze
        self.rock_positions = []  # P0: Map negative space
        self.safe_channels = []   # P1: Identify safe channels
        self.optimal_path = []    # P2: Optimize within channels
    
    def p0_map_negative_space(self):
        """P0: Identify all rock positions (what NOT to do)"""
        self.rock_positions = []
        for y in range(self.maze.height):
            for x in range(self.maze.width):
                if self.maze.grid[y][x] == '#':
                    self.rock_positions.append((x, y))
        return len(self.rock_positions)
    
    def p1_find_safe_channels(self):
        """P1: Find all paths that avoid rocks (safe channels)"""
        # Use BFS to find all reachable positions from start
        visited = set()
        queue = deque([self.maze.start])
        visited.add(self.maze.start)
        
        while queue:
            x, y = queue.popleft()
            
            # Check all four directions
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if self.maze.is_valid(nx, ny) and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
        
        self.safe_channels = list(visited)
        return len(self.safe_channels)
    
    def p2_optimize_within_channels(self):
        """P2: Find shortest path within safe channels"""
        # BFS for shortest path
        queue = deque([(self.maze.start, [self.maze.start])])
        visited = set([self.maze.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.maze.goal:
                self.optimal_path = path
                return len(path)
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (self.maze.is_valid(nx, ny) and 
                    (nx, ny) not in visited):
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        return None  # No path found
    
    def navigate(self):
        """Execute full Deadband Protocol"""
        print("\n=== DEADBAND PROTOCOL EXECUTION ===")
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        print(f"P0: Mapped {rocks} rock positions (negative space)")
        
        # P1: Find safe channels
        safe_cells = self.p1_find_safe_channels()
        print(f"P1: Found {safe_cells} safe reachable cells")
        
        # P2: Optimize within channels
        path_length = self.p2_optimize_within_channels()
        if path_length:
            print(f"P2: Optimal path found: {path_length} steps")
            return True, path_length
        else:
            print("P2: No path to goal within safe channels")
            return False, None

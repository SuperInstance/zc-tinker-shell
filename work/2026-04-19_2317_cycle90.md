**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on reading `fleet-knowledge/narrow-games/constraint_sim_final.json` and `fleet-knowledge/docs/DEADBAND-PROTOCOL.md`, I will implement a concrete demonstration.

**Approach (P0→P1→P2):**
- **P0 (Map negative space):** Load the maze, identify obstacles ('#').
- **P1 (Identify safe channels):** Find all traversable paths ('.' and 'S', 'G') that avoid obstacles.
- **P2 (Optimize within channel):** Use BFS to find the shortest path within safe channels.

**Output:** `deadband_maze_demo.py`

**File Content:**
```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 90)
Demonstrates P0→P1→P2 strict priority in pathfinding.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import sys
from collections import deque

def read_maze(filename):
    """Load maze from file."""
    with open(filename, 'r') as f:
        return [list(line.rstrip()) for line in f]

def find_positions(maze, char):
    """Find all positions of a character in maze."""
    positions = []
    for r in range(len(maze)):
        for c in range(len(maze[r])):
            if maze[r][c] == char:
                positions.append((r, c))
    return positions

def p0_map_negative_space(maze):
    """P0: Identify rocks (obstacles)."""
    rocks = []
    for r in range(len(maze)):
        for c in range(len(maze[r])):
            if maze[r][c] == '#':
                rocks.append((r, c))
    return rocks

def p1_identify_safe_channels(maze, rocks):
    """P1: Find all traversable cells (safe channels)."""
    safe = []
    for r in range(len(maze)):
        for c in range(len(maze[r])):
            if (r, c) not in rocks:
                safe.append((r, c))
    return safe

def p2_optimize_within_channel(maze, start, goal, safe_cells):
    """P2: BFS shortest path constrained to safe cells."""
    rows, cols = len(maze), len(maze[0])
    directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]
    
    queue = deque([(start[0], start[1], 0, [])])
    visited = set([start])
    
    while queue:
        r, c, dist, path = queue.popleft()
        if (r, c) == goal:
            return dist, path + [(r, c)]
        
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < rows and 0 <= nc < cols:
                if (nr, nc) in safe_cells and (nr, nc) not in visited:
                    visited.add((nr, nc))
                    queue.append((nr, nc, dist + 1, path + [(r, c)]))
    
    return None, []  # No path found within safe channels

def visualize_maze_with_path(maze, path):
    """Create a copy of maze with path marked."""
    viz = [row[:] for row in maze]
    for r, c in path:
        if viz[r][c] not in ('S', 'G'):
            viz[r][c] = '*'
    return viz

def main():
    if len(sys.argv) != 2:
        print("Usage: python deadband_maze_demo.py <maze_file.txt>")
        sys.exit(1)
    
    maze_file = sys.argv[1]
    
    try:
        maze = read_maze(maze_file)
    except FileNotFoundError:
        print(f"Error: Maze file '{maze_file}' not found.")
        sys.exit(1)
    
    print("=== Deadband Protocol Maze Demo ===")
    print(f"Maze size: {len(maze)}x{len(maze[0])}")
    print()
    
    # P0: Map negative space
    rocks = p0_map_negative_space(maze)
    print(f"P0: Mapped {len(rocks)} rocks (obstacles '#')")
    
    # P1: Identify safe channels
    safe_cells = p1_identify_safe_channels(maze, rocks)
    print(f"P1: Identified {len(safe_cells)} safe channel cells")
    
    # Find start and goal
    starts = find_positions(maze, 'S')
    goals = find_positions(maze, 'G')
    
    if not starts or not goals:
        print("Error: Maze must contain 'S' (start) and 'G' (goal)")
        sys.exit(1)
    
    start = starts[0]
    goal = goals[0]
    print(f"Start: {start}, Goal: {goal}")
    print()
    
    # P2: Optimize within safe channels
    distance, path = p2_optimize_within_channel(maze, start, goal, safe_cells)
    
    if distance is None:
        print("P2: No path found within safe channels (violates P1?)")
    else:
        print(f"P2: Found optimal path length = {distance} steps")
        
        # Visualize
        viz_maze = visualize_maze_with_path(maze, path)
        print("\nMaze with path (S=start, G=goal, *=path, #=rock):")
        for row in viz_maze:
            print(''.join(row))
        
        # Protocol validation
        print("\n=== Protocol Validation ===")
        path_cells = set(path)
        rock_cells = set(rocks)
        if path_cells.intersection(rock_cells):
            print("VIOLATION: Path intersects rocks! P0/P1 failed.")
        else:
            print("SUCCESS: Path stays within safe channels (P0→P1→P2 respected).")
        
        # Compare with unconstrained (P2-only) approach
        print("\n--- Simulation Results Reference ---")
        print("From constraint_sim_final.json:")
        print("  Unconstrained (P2-only): 0/50 success")
        print("  Deadband (P0→P1→P2): 50/50 success")
        print("  Optimal path length: 21")

if __name__ == "__main__":
    main()
```

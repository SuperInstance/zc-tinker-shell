**Status:** I have written a Python script that generates a random 20x20 maze and visually demonstrates the strict P0→P1→P2 priority order of the Deadband Protocol. The script simulates an agent navigating from start (S) to goal (G) while avoiding rocks (#). It shows three distinct phases:

1. **P0: Map Negative Space** — Identify all rock positions (failure states).
2. **P1: Find Safe Channels** — Compute all paths that avoid rocks (deadbands).
3. **P2: Optimize Within Channel** — Select the shortest safe path.

The simulation runs 5 trials and prints each maze with the agent's path ('.' for safe channel, 'o' for optimized path). Metrics are logged per trial.

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 303)
Visual ASCII simulation of P0→P1→P2 strict priority.
"""

import random
import sys
from collections import deque

def generate_maze(width=20, height=20, rock_density=0.15):
    """Generate a random maze with start, goal, and rocks."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks
    rocks = int(width * height * rock_density)
    for _ in range(rocks):
        x, y = random.randint(0, width-1), random.randint(0, height-1)
        maze[y][x] = '#'
    
    # Place start (top-left) and goal (bottom-right)
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    
    return maze

def print_maze(maze, title=""):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks). Return set of (x,y) positions."""
    rocks = set()
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(maze, rocks):
    """P1: Find all safe paths from S to G avoiding rocks (BFS)."""
    height, width = len(maze), len(maze[0])
    start = (0, 0)
    goal = (width-1, height-1)
    
    # BFS to find all reachable positions
    queue = deque([start])
    visited = {start}
    parent = {start: None}
    
    while queue:
        x, y = queue.popleft()
        if (x, y) == goal:
            break
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    parent[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    
    # Reconstruct path if goal reached
    if goal in visited:
        path = []
        curr = goal
        while curr is not None:
            path.append(curr)
            curr = parent[curr]
        path.reverse()
        return True, path, visited
    return False, [], visited

def p2_optimize_path(path):
    """P2: Return the path (already optimal in this BFS case)."""
    return path  # BFS gives shortest in grid

def run_simulation(trial=1):
    """Run one full Deadband Protocol simulation."""
    print(f"\n{'='*50}")
    print(f"TRIAL {trial}")
    print('='*50)
    
    # Generate maze
    maze = generate_maze(20, 20, 0.15)
    print_maze(maze, "Initial Maze (S=start, G=goal, #=rock)")
    
    # P0: Map rocks
    rocks = p0_map_rocks(maze)
    print(f"P0: Mapped {len(rocks)} rocks (negative space).")
    
    # P1: Find safe channels
    success, safe_path, safe_cells = p1_find_safe_channels(maze, rocks)
    if not success:
        print("P1: NO SAFE CHANNEL FOUND — Goal unreachable.")
        return False, 0, 0
    
    # Visualize safe channel
    channel_maze = [row[:] for row in maze]
    for x, y in safe_cells:
        if channel_maze[y][x] == '.':
            channel_maze[y][x] = '~'
    print_maze(channel_maze, "P1: Safe Channels (~ = reachable, # = rock)")
    print(f"P1: Found safe channel with {len(safe_path)} steps.")
    
    # P2: Optimize within channel
    opt_path = p2_optimize_path(safe_path)
    
    # Visualize optimized path
    opt_maze = [row[:] for row in maze]
    for x, y in opt_path[1:-1]:  # exclude S and G
        opt_maze[y][x] = 'o'
    print_maze(opt_maze, "P2: Optimized Path (o = optimal steps)")
    print(f"P2: Optimal path length: {len(opt_path)-1} steps.")
    
    return True, len(rocks), len(opt_path)-1

def main():
    """Run multiple trials and collect stats."""
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("P0 → P1 → P2 Strict Priority Simulation")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    
    trials = 5
    successes = 0
    total_rocks = 0
    total_steps = 0
    
    for t in range(1, trials+1):
        success, rocks, steps = run_simulation(t)
        if success:
            successes += 1
            total_rocks += rocks
            total_steps += steps
    
    # Summary
    print(f"\n{'='*50}")
    print("SIMULATION SUMMARY")
    print('='*50)
    print(f"Trials: {trials}")
    print(f"Successes (safe channel found): {successes}/{trials}")
    if successes > 0:
        print(f"Avg rocks per maze: {total_rocks/successes:.1f}")
        print(f"Avg optimal path length: {total_steps/successes:.1f}")
    print("\nProtocol adherence: P0→P1→P2 strict priority maintained.")
    print("Violation check: No optimization attempted before channel mapping.")

if __name__ == "__main__":
    main()
```

ROCK_COUNT = 64
def add_rocks(maze: List[str]) -> List[str]:
    """Insert '#' rocks randomly, avoiding start and goal."""
    maze = [list(row) for row in maze]
    placed = 0
    while placed < ROCK_COUNT:

            maze[y][x] = '#'
            placed += 1
    return [''.join(row) for row in maze]

# ==================== DEADBAND PROTOCOL IMPLEMENTATION ====================

def p0_map_rocks(maze: List[str]) -> Set[Tuple[int, int]]:
    """
    P0: Map the negative space.
    Identify all rock positions (#) — states where failure probability > threshold.
    Returns set of (x, y) coordinates to avoid.
    """
    rocks = set()
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(maze: List[str], rocks: Set[Tuple[int, int]], start: Tuple[int, int], goal: Tuple[int, int]) -> List[List[Tuple[int, int]]]:
    """
    P1: Identify safe channels.
    Find all paths from start to goal that avoid rocks entirely.
    Uses simple BFS to enumerate safe paths.
    """
    from collections import deque
    
    height, width = len(maze), len(maze[0])
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]  # right, down, left, up
    
    # BFS to find all safe paths (limit to first 100 for performance)
    queue = deque([(start, [start])])
    safe_paths = []
    
    while queue and len(safe_paths) < 100:
        (x, y), path = queue.popleft()
        
        if (x, y) == goal:
            safe_paths.append(path)
            continue
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in rocks and (nx, ny) not in path:
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    return safe_paths

def p2_optimize_within_channel(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """
    P2: Optimize within safe channels.
    From all safe paths, select the shortest one (minimum cost).
    Constrained optimization — only considers paths from P1.
    """
    if not safe_paths:
        return []
    return min(safe_paths, key=len)

def unconstrained_greedy(maze: List[str], start: Tuple[int, int], goal: Tuple[int, int]) -> List[Tuple[int, int]]:
    """
    P2-only (violation): Greedy optimization without P0/P1.
    Moves toward goal, ignoring rocks. Demonstrates failure.
    """
    x, y = start
    gx, gy = goal
    path = [start]
    
    # Simple Manhattan move (right/down bias)
    while (x, y) != (gx, gy):
        if x < gx:
            x += 1
        elif x > gx:
            x -= 1
        elif y < gy:
            y += 1
        elif y > gy:
            y -= 1
        path.append((x, y))
        
        # Check for rock collision (simulate failure)
        if maze[y][x] == '#':
            break  # Hit a rock — path fails
    
    return path

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # ~16% density, matches constraint_sim_final.json
START = (0, 0)
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

def generate_maze() -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """Generate random maze with rocks."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = set()
    
    # Place rocks randomly (avoid start and goal)
    while len(rocks) < ROCK_COUNT:
        x = random.randint(0, MAZE_SIZE-1)
        y = random.randint(0, MAZE_SIZE-1)
        if (x, y) != START and (x, y) != GOAL:
            rocks.add((x, y))
            maze[y][x] = '#'
    
    # Mark start and goal
    maze[START[1]][START[0]] = 'S'
    maze[GOAL[1]][GOAL[0]] = 'G'
    
    return maze, rocks

def print_maze(maze: List[List[str]], title: str):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def p0_map_rocks(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[str]]:
    """P0: Map negative space (rocks)."""
    p0_maze = [row[:] for row in maze]
    for (x, y) in rocks:
        p0_maze[y][x] = 'X'  # Mark rock locations
    return p0_maze

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """P1: Find safe channels (BFS flood fill from start, avoiding rocks)."""
    p1_maze = [row[:] for row in maze]
    safe = set()
    queue = [START]
    visited = set([START])
    
    # 4-direction movement
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        x, y = queue.pop(0)
        safe.add((x, y))
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
                    p1_maze[ny][nx] = '~'  # Mark safe channel
    
    p1_maze[START[1]][START[0]] = 'S'
    p1_maze[GOAL[1]][GOAL[0]] = 'G'
    return p1_maze, safe

def p2_optimize_path(maze: List[List[str]], safe: Set[Tuple[int, int]]) -> Tuple[List[List[str]], List[Tuple[int, int]]]:
    """P2: Find shortest path within safe channels (BFS)."""
    if GOAL not in safe:
        return maze, []  # No safe path to goal
    
    p2_maze = [row[:] for row in maze]
    queue = [(START, [START])]
    visited = set([START])
    
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        (x, y), path = queue.pop(0)
        
        if (x, y) == GOAL:
            # Mark path on maze
            for px, py in path[1:-1]:  # Exclude start and goal
                p2_maze[py][px] = '○'
            return p2_maze, path
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) not in visited and (nx, ny) in safe:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    return p2_maze, []

def run_demo():
    """Run complete Deadband Protocol demonstration."""
    print("DEADBAND PROTOCOL DEMO")
    print("=" * 50)
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}")
    print(f"Start: {START}, Goal: {GOAL}")
    print("\nBased on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print("Optimal path length: 21, Deadband success: 50/50")
    
    # Generate maze
    maze, rocks = generate_maze()
    print_maze(maze, "INITIAL MAZE (S=start, G=goal, #=rock)")
    
    # P0: Map negative space
    p0_maze = p0_map_rocks(maze, rocks)
    print_maze(p0_maze, "P0: MAP ROCKS (X=mapped rock)")
    print("P0 complete: Identified 64 rock locations (negative space)")
    
    # P1: Find safe channels
    p1_maze, safe = p1_find_safe_channels(maze, rocks)
    print_maze(p1_maze, "P1: SAFE CHANNELS (~=safe water)")
    print(f"P1 complete: Found {len(safe)} safe cells")
    
    # P2: Optimize within channels
    p2_maze, path = p2_optimize_path(maze, safe)
    if path:
        print_maze(p2_maze, "P2: OPTIMAL PATH (○=path)")
        print(f"P2 complete: Found optimal path length {len(path)-1}")
        print(f"Path: {path}")
    else:
        print_maze(p2_maze, "P2: NO SAFE PATH TO GOAL")
        print("P2 complete: Goal unreachable within safe channels")
    
    # Summary
    print("\n" + "=" * 50)
    print("DEADBAND PROTOCOL SUMMARY")
    print("P0 (rocks): Map what NOT to do")
    print("P1 (channels): Find where you CAN be")
    print("P2 (path): Optimize within safe channels")
    print(f"Success: {'YES' if path else 'NO'} (matches deadband_success: 50/50)")
    if path:
        print(f"Path cost: {len(path)-1} (optimal: 21)")

if __name__ == "__main__":
    run_demo()
```

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (🔧)
A visual ASCII simulation of the strict P0→P1→P2 priority order.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time
import sys

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.rock_density = rock_density
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self.safe_channels = set()
        self.optimal_path = []
        self.generate_maze()
    
    def generate_maze(self):
        """Generate random rocks (negative space P0)"""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * self.rock_density)
        
        # Ensure start and goal are not rocks
        possible = [(x, y) for x in range(self.width) for y in range(self.height)]
        possible.remove(self.start)
        possible.remove(self.goal)
        
        self.rocks = set(random.sample(possible, num_rocks))
    
    def map_negative_space(self):
        """P0: Identify rocks (what NOT to do)"""
        return self.rocks
    
    def find_safe_channels(self):
        """P1: Find all paths from start to goal avoiding rocks"""
        # Simple BFS to find all reachable cells (safe channels)
        visited = set()
        queue = [self.start]
        
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Check neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) not in self.rocks and
                    (nx, ny) not in visited):
                    queue.append((nx, ny))
        
        self.safe_channels = visited
        return visited
    
    def optimize_within_channel(self):
        """P2: Find shortest path through safe channels"""
        # BFS for shortest path
        queue = [(self.start, [self.start])]
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.goal:
                self.optimal_path = path
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) in self.safe_channels and
                    (nx, ny) not in visited):
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        return []  # No path found
    
    def run_protocol(self):
        """Execute strict P0→P1→P2 priority"""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*60)
        
        # P0: Map negative space
        print("\n[P0] MAPPING NEGATIVE SPACE")
        print(f"Identified {len(self.rocks)} rocks (failure states)")
        time.sleep(1)
        
        # P1: Find safe channels
        print("\n[P1] FINDING SAFE CHANNELS")
        safe = self.find_safe_channels()
        print(f"Found {len(safe)} reachable safe cells")
        time.sleep(1)
        
        # P2: Optimize within channel
        print("\n[P2] OPTIMIZING WITHIN SAFE CHANNELS")
        path = self.optimize_within_channel()
        if path:
            print(f"Found optimal path: {len(path)} steps")
        else:
            print("NO PATH FOUND - Goal unreachable through safe channels")
        time.sleep(1)
        
        return path
    
    def display_maze(self, show_p0=False, show_p1=False, show_p2=False):
        """Display ASCII maze with current protocol phase"""
        print("\n" + "-"*40)
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif show_p0 and pos in self.rocks:
                    row.append('█')  # Rocks (P0)
                elif show_p1 and pos in self.safe_channels:
                    row.append('·')  # Safe channels (P1)
                elif show_p2 and pos in self.optimal_path:
                    row.append('○')  # Optimal path (P2)
                elif pos in self.rocks:
                    row.append('█')
                else:
                    row.append(' ')
            
            print(' '.join(row))
        
        legend = []
        if show_p0: legend.append("█ = Rocks (P0)")
        if show_p1: legend.append("· = Safe (P1)")
        if show_p2: legend.append("○ = Path (P2)")
        if legend:
            print("\n" + " | ".join(legend))

def main():
    """Run the demonstration"""
    print("Initializing Deadband Protocol Maze Demo...")
    
    # Create maze matching simulation parameters
    maze = DeadbandMaze(width=20, height=20, rock_density=0.16)
    
    # Phase 0: Show initial maze with rocks
    print("\nINITIAL MAZE (Hidden Rocks)")
    maze.display_maze()
    
    # Run protocol step by step
    path = maze.run_protocol()
    
    # Visual demonstration of each phase
    print("\n" + "="*60)
    print("VISUAL DEMONSTRATION")
    print("="*60)
    
    print("\n1. P0 ONLY: Rocks identified (what NOT to do)")
    maze.display_maze(show_p0=True)
    time.sleep(2)
    
    print("\n2. P0+P1: Safe channels mapped (where you CAN be)")
    maze.display_maze(show_p0=True, show_p1=True)
    time.sleep(2)
    
    print("\n3. P0+P1+P2: Optimal path through safe channels")
    maze.display_maze(show_p0=True, show_p1=True, show_p2=True)
    
    # Results summary
    print("\n" + "="*60)
    print("RESULTS SUMMARY")
    print("="*60)
    print(f"Rocks (P0): {len(maze.rocks)}")
    print(f"Safe cells (P1): {len(maze.safe_channels)}")
    print(f"Path length (P2): {len(path) if path else 'N/A'}")
    print(f"Goal reachable: {'YES' if path else 'NO'}")
    
    # Compare with simulation data
    print("\n" + "-"*60)
    print("COMPARISON TO FLEET SIMULATION DATA")
    print("-"*60)
    print("From constraint_sim_final.json:")
    print("  Deadband success rate: 50/50 (100%)")
    print("  Deadband avg steps: 21.0")
    print("  Constrained success: 0/50 (0%)")
    print("\nThis demo shows WHY deadband works:")
    print("  P0→P1→P2 strict priority prevents hitting rocks")

if __name__ == "__main__":
    main()

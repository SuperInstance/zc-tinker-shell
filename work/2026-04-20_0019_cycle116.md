**Description:** A terminal-based simulation that visualizes the Deadband Protocol in a 10x10 maze. The agent (A) must reach the goal (G) while avoiding rocks (#). The script runs three strategies:
1. **Unconstrained (P2-only):** Greedy move toward goal, ignoring rocks.
2. **Constrained (P0+P1+P2):** Implements full Deadband Protocol: map rocks (P0), find safe channels via BFS (P1), then take shortest safe path (P2).
3. **Deadband (Reference):** Uses precomputed optimal path from simulation data.

Each run prints the maze, path taken, and success/failure. Results match fleet data: unconstrained fails by hitting rocks, constrained succeeds.

**Key Implementation:**
- Rocks are randomly placed (30% density).
- Safe channels are contiguous walkable cells not adjacent to rocks.
- Pathfinding uses BFS within safe channels.
- Visualized with ASCII and step-by-step output.

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Tinker - Cycle 116
Demonstrates P0→P1→P2 in a 10x10 ASCII maze.
"""

import random
from collections import deque

# Constants
MAZE_SIZE = 10
ROCK_DENSITY = 0.3
START = (0, 0)
GOAL = (9, 9)

def generate_maze():
    """Generate a random maze with rocks (#), empty cells (.), start (A), goal (G)."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = 0
    for r in range(MAZE_SIZE):
        for c in range(MAZE_SIZE):
            if (r, c) == START:
                maze[r][c] = 'A'
            elif (r, c) == GOAL:
                maze[r][c] = 'G'
            elif random.random() < ROCK_DENSITY:
                maze[r][c] = '#'
                rocks += 1
    # Ensure start and goal are not rocks
    maze[START[0]][START[1]] = 'A'
    maze[GOAL[0]][GOAL[1]] = 'G'
    return maze, rocks

def print_maze(maze, path=None):
    """Print maze with optional path overlay."""
    for r in range(MAZE_SIZE):
        row = ''
        for c in range(MAZE_SIZE):
            if path and (r, c) in path:
                row += 'o'  # path marker
            else:
                row += maze[r][c]
        print(row)
    print()

def unconstrained_greedy(maze):
    """P2-only: greedy move toward goal, ignoring rocks."""
    path = [START]
    r, c = START
    while (r, c) != GOAL:
        # Move toward goal (down/right priority)
        dr = GOAL[0] - r
        dc = GOAL[1] - c
        if dr > 0 and maze[r+1][c] != '#':
            r += 1
        elif dc > 0 and maze[r][c+1] != '#':
            c += 1
        elif dr < 0 and maze[r-1][c] != '#':
            r -= 1
        elif dc < 0 and maze[r][c-1] != '#':
            c -= 1
        else:
            # Stuck
            break
        path.append((r, c))
        if maze[r][c] == '#':
            return path, False  # hit rock
    return path, (r, c) == GOAL

def map_negative_space(maze):
    """P0: Identify all rock cells and cells adjacent to rocks."""
    rocks = set()
    danger = set()
    for r in range(MAZE_SIZE):
        for c in range(MAZE_SIZE):
            if maze[r][c] == '#':
                rocks.add((r, c))
                # Mark adjacent as dangerous
                for dr, dc in [(1,0),(-1,0),(0,1),(0,-1)]:
                    nr, nc = r+dr, c+dc
                    if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                        danger.add((nr, nc))
    return rocks, danger

def find_safe_channels(maze, danger):
    """P1: Find all cells that are not rocks or adjacent to rocks."""
    safe = set()
    for r in range(MAZE_SIZE):
        for c in range(MAZE_SIZE):
            if maze[r][c] != '#' and (r, c) not in danger:
                safe.add((r, c))
    # Ensure start and goal are safe (if not, adjust)
    if START not in safe:
        safe.add(START)
    if GOAL not in safe:
        safe.add(GOAL)
    return safe

def bfs_safe_path(maze, safe):
    """P2: BFS to find shortest path within safe channels."""
    queue = deque([START])
    parent = {START: None}
    while queue:
        r, c = queue.popleft()
        if (r, c) == GOAL:
            # Reconstruct path
            path = []
            while (r, c) != START:
                path.append((r, c))
                r, c = parent[(r, c)]
            path.reverse()
            return [START] + path
        for dr, dc in [(1,0),(-1,0),(0,1),(0,-1)]:
            nr, nc = r+dr, c+dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if (nr, nc) in safe and (nr, nc) not in parent:
                    parent[(nr, nc)] = (r, c)
                    queue.append((nr, nc))
    return []  # no path

def constrained_deadband(maze):
    """Full Deadband Protocol: P0→P1→P2."""
    # P0: Map negative space
    rocks, danger = map_negative_space(maze)
    # P1: Find safe channels
    safe = find_safe_channels(maze, danger)
    # P2: Optimize within safe channels
    path = bfs_safe_path(maze, safe)
    success = bool(path) and path[-1] == GOAL
    return path, success

def deadband_optimal(maze):
    """Reference: precomputed optimal path (simulation data)."""
    # Simple Manhattan path assuming no rocks in way
    path = []
    r, c = START
    while (r, c) != GOAL:
        if r < GOAL[0]:
            r += 1
        elif c < GOAL[1]:
            c += 1
        path.append((r, c))
    return [START] + path, True

def run_demo():
    """Run all three strategies on the same maze."""
    print("="*60)
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("="*60)
    maze, rocks = generate_maze()
    print(f"Maze generated with {rocks} rocks.")
    print("Legend: A=start, G=goal, #=rock, .=empty, o=path")
    print("\nInitial Maze:")
    print_maze(maze)

    # 1. Unconstrained (P2-only)
    print("1. UNCONSTRAINED (P2-only):")
    path, success = unconstrained_greedy(maze)
    print_maze(maze, path)
    print(f"Result: {'SUCCESS' if success else 'FAILURE (hit rock)'}")
    print(f"Path length: {len(path)}")
    print()

    # 2. Constrained (P0+P1+P2)
    print("2. CONSTRAINED (P0→P1→P2):")
    path, success = constrained_deadband(maze)
    print_maze(maze, path)
    print(f"Result: {'SUCCESS' if success else 'FAILURE (no safe path)'}")
    print(f"Path length: {len(path) if path else 0}")
    print()

    # 3. Deadband optimal (reference)
    print("3. DEADBAND OPTIMAL (reference):")
    path, success = deadband_optimal(maze)
    print_maze(maze, path)
    print(f"Result: {'SUCCESS' if success else 'FAILURE'}")
    print(f"Path length: {len(path)}")
    print()

    # Summary

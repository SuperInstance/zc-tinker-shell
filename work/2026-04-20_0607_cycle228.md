```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map rocks (negative space)
P1: Find safe channels (deadbands)
P2: Optimize path within safe channels
"""

import random
from collections import deque

# Simulation parameters from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64
START = (0, 0)
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

def generate_maze():
    """Create a maze with rocks (obstacles)."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        r = random.randint(0, MAZE_SIZE-1)
        c = random.randint(0, MAZE_SIZE-1)
        if (r, c) != START and (r, c) != GOAL:
            rocks.add((r, c))
            maze[r][c] = '#'
    maze[START[0]][START[1]] = 'S'
    maze[GOAL[0]][GOAL[1]] = 'G'
    return maze, rocks

def print_maze(maze, title=""):
    """Display maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def p0_map_rocks(maze, rocks):
    """P0: Identify negative space (rocks)."""
    print("P0: MAP NEGATIVE SPACE")
    print(f"   Found {len(rocks)} rocks (obstacles)")
    print("   Rocks are marked '#'")
    return list(rocks)

def p1_find_safe_channels(maze, rocks):
    """P1: Find all cells reachable without hitting rocks."""
    safe = [[True for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    for r, c in rocks:
        safe[r][c] = False
    
    # BFS from start to find reachable safe cells
    visited = [[False for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    queue = deque([START])
    visited[START[0]][START[1]] = True
    reachable_safe = [START]
    
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    while queue:
        r, c = queue.popleft()
        for dr, dc in dirs:
            nr, nc = r+dr, c+dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if safe[nr][nc] and not visited[nr][nc]:
                    visited[nr][nc] = True
                    queue.append((nr, nc))
                    reachable_safe.append((nr, nc))
    
    # Mark safe channels on maze copy
    maze_p1 = [row[:] for row in maze]
    for r, c in reachable_safe:
        if maze_p1[r][c] == '.':
            maze_p1[r][c] = '~'  # safe water
    
    print("P1: IDENTIFY SAFE CHANNELS")
    print(f"   Found {len(reachable_safe)} reachable safe cells")
    print("   Safe channels marked '~'")
    return maze_p1, reachable_safe, visited

def p2_optimize_in_channel(maze, safe_visited):
    """P2: Find shortest path within safe channels."""
    # BFS for shortest path
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    queue = deque([(START, [START])])
    visited = [[False for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    visited[START[0]][START[1]] = True
    
    while queue:
        (r, c), path = queue.popleft()
        if (r, c) == GOAL:
            # Mark path on maze copy
            maze_p2 = [row[:] for row in maze]
            for pr, pc in path:
                if maze_p2[pr][pc] == '.' or maze_p2[pr][pc] == '~':
                    maze_p2[pr][pc] = '*'
            print("P2: OPTIMIZE WITHIN CHANNEL")
            print(f"   Found optimal path length: {len(path)-1} steps")
            print("   Optimal path marked '*'")
            return maze_p2, path
        
        for dr, dc in dirs:
            nr, nc = r+dr, c+dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if safe_visited[nr][nc] and not visited[nr][nc]:
                    visited[nr][nc] = True
                    queue.append(((nr, nc), path + [(nr, nc)]))
    
    print("P2: OPTIMIZE WITHIN CHANNEL")
    print("   No path to goal within safe channels")
    return maze, []

def run_demo():
    """Execute the full Deadband Protocol demonstration."""
    print("\n" + "="*60)
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("Based on constraint_sim_final.json (20x20 maze, 64 rocks)")
    print("="*60)
    
    # Generate environment
    maze, rocks = generate_maze()
    print_maze(maze, "INITIAL MAZE (S=start, G=goal, #=rock)")
    
    # P0: Map negative space
    rocks_list = p0_map_rocks(maze, rocks)
    
    # P1: Find safe channels
    maze_p1, safe_cells, safe_visited = p1_find_safe_channels(maze, rocks)
    print_maze(maze_p1, "SAFE CHANNELS MAPPED (P1)")
    
    # P2: Optimize within channels
    maze_p2, optimal_path = p2_optimize_in_channel(maze, safe_visited)
    print_maze(maze_p2, "OPTIMAL PATH IN SAFE CHANNELS (P2)")
    
    # Summary
    print("\n" + "="*60)
    print("PROTOCOL SUMMARY")
    print(f"P0: Mapped {len(rocks_list)} rocks")
    print(f"P1: Found {len(safe_cells)} safe reachable cells")
    if optimal_path:
        print(f"P2: Optimal path length: {len(optimal_path)-1} steps")
        success = True
    else:
        print("P2: No path found (goal unreachable in safe channels)")
        success = False
    print("="*60)
    
    return success

if __name__ == "__main__":
    random.seed(42)  # Deterministic for demonstration
    success = run_demo()
    
    # Reference simulation results
    print("\nREFERENCE: constraint_sim_final.json results:")
    print("  Unconstrained (P2-only): 0/50 success")
    print("  Deadband (P0→P1→P2): 50/50 success")
    print(f"\nThis demo run: {'SUCCESS' if success else 'NO PATH'} (one sample)")
```

**Explanation:**
1. **P0 (Map negative space)**: Identifies all rock positions ('#').
2. **P1 (Find safe channels)**: Uses BFS to find all reachable cells that aren't rocks, marks them as safe water ('~').
3. **P2 (Optimize within channel)**: Finds the shortest path ('*') through safe channels only.

**Key demonstration:** The protocol strictly follows P0→P1→P2. Optimization (P2) only operates within the safe channels identified in P1, never considering rock cells.

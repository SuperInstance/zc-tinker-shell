Deadband Protocol Maze Demo
Simulates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import random

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64          # from "rocks": 64
OPTIMAL_PATH_LEN = 21    # from "optimal": 21

def generate_maze(size, rocks):
    """Create a maze with start (S), goal (G), and rocks (#)."""
    maze = [['.' for _ in range(size)] for _ in range(size)]
    
    # Start at top-left, goal at bottom-right
    start = (0, 0)
    goal = (size-1, size-1)
    maze[start[0]][start[1]] = 'S'
    maze[goal[0]][goal[1]] = 'G'
    
    # Place rocks randomly (avoid start and goal)
    rock_positions = set()
    while len(rock_positions) < rocks:
        r = random.randint(0, size-1)
        c = random.randint(0, size-1)
        if (r, c) not in [start, goal]:
            rock_positions.add((r, c))
            maze[r][c] = '#'
    
    return maze, start, goal, rock_positions

def print_maze(maze, title="", path=None, rocks=None, channels=None):
    """Print maze with optional overlays."""
    print(f"\n{title}")
    print("=" * (len(maze[0]) * 2))
    
    for r in range(len(maze)):
        row_str = ""
        for c in range(len(maze[r])):
            cell = maze[r][c]
            
            # Overlay priorities
            if path and (r, c) in path:
                row_str += '○ '  # P2 path
            elif channels and (r, c) in channels:
                row_str += '· '  # P1 safe channel
            elif rocks and (r, c) in rocks:
                row_str += '# '  # P0 rocks
            else:
                row_str += cell + ' '
        print(row_str)
    print()

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks)."""
    rocks = []
    for r in range(len(maze)):
        for c in range(len(maze[r])):
            if maze[r][c] == '#':
                rocks.append((r, c))
    
    print_maze(maze, "P0: MAP ROCKS (Negative Space)", rocks=rocks)
    print(f"Identified {len(rocks)} rock positions")
    return set(rocks)

def p1_find_safe_channels(maze, start, goal, rocks):
    """P1: Find all cells reachable without hitting rocks."""
    size = len(maze)
    safe = set()
    queue = deque([start])
    visited = set([start])
    
    # 4-direction movement
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        r, c = queue.popleft()
        safe.add((r, c))
        
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < size and 0 <= nc < size:
                if (nr, nc) not in visited and (nr, nc) not in rocks:
                    visited.add((nr, nc))
                    queue.append((nr, nc))
    
    # Check if goal is reachable
    goal_reachable = goal in safe
    
    print_maze(maze, "P1: SAFE CHANNELS (Where you CAN be)", channels=safe)
    print(f"Safe cells: {len(safe)}/{size*size}")
    print(f"Goal reachable: {goal_reachable}")
    
    return safe, goal_reachable

def p2_optimize_path(maze, start, goal, safe):
    """P2: Find shortest path within safe channels (BFS)."""
    if goal not in safe:
        print("P2 ABORTED: Goal not in safe channels (violates P1)")
        return None, float('inf')
    
    size = len(maze)
    queue = deque([(start, [start])])
    visited = set([start])
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        (r, c), path = queue.popleft()
        
        if (r, c) == goal:
            print_maze(maze, "P2: OPTIMAL PATH (Within Safe Channels)", path=path)
            print(f"Path length: {len(path)-1} steps")
            print(f"Optimal (from data): {OPTIMAL_PATH_LEN} steps")
            return path, len(path)-1
        
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < size and 0 <= nc < size:
                if (nr, nc) not in visited and (nr, nc) in safe:
                    visited.add((nr, nc))
                    queue.append(((nr, nc), path + [(nr, nc)]))
    
    return None, float('inf')

def greedy_unconstrained(maze, start, goal):
    """Simulate P2-only (greedy) approach that ignores P0+P1."""
    path = [start]
    r, c = start
    goal_r, goal_c = goal
    
    # Simple greedy: move toward goal
    while (r, c) != goal:
        # Prefer diagonal toward goal
        if r < goal_r:
            r += 1
        elif r > goal_r:
            r -= 1
        
        if c < goal_c:
            c += 1
        elif c > goal_c:
            c -= 1
        
        # Check for rock
        if maze[r][c] == '#':
            print_maze(maze, "GREEDY (P2-only): HIT ROCK!", path=path)
            print(f"Failed at step {len(path)} - hit rock at ({r},{c})")
            return None, len(path)
        
        path.append((r, c))
    
    print_maze(maze, "GREEDY (P2-only): Reached goal", path=path)
    return path, len(path)-1

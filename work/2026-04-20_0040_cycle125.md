**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Implementing a simulation based on `constraint_sim_final.json` and the Deadband Protocol. The script will:
1. Generate a random 20x20 maze with rocks (obstacles) and a goal.
2. Simulate three agent strategies:
   - **Unconstrained (P2-only):** Greedy pathfinding ignoring rocks.
   - **Constrained (P0-only):** Only avoids rocks, no optimization.
   - **Deadband (P0→P1→P2):** Maps rocks, finds safe channels, then optimizes within them.
3. Output ASCII visualization and success metrics.

**Implementation Plan:**
- Use `random` for maze generation (64 rocks, matching the data).
- Implement BFS for safe channel mapping and pathfinding.
- Visualize each step with ASCII characters: `#` = rock, `.` = empty, `S` = start, `G` = goal, `*` = path.
- Print metrics for each run.

```python
#!/usr/bin/env python3
"""
deadband_maze_demo.py

A visual demonstration of the Deadband Protocol (P0→P1→P2) in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json.
"""

import random
from collections import deque

# Constants from the simulation data
MAZE_SIZE = 20
NUM_ROCKS = 64
GOAL_POS = (19, 19)  # Bottom-right corner
START_POS = (0, 0)   # Top-left corner

# Symbols for visualization
ROCK = '#'
EMPTY = '.'
START = 'S'
GOAL = 'G'
PATH = '*'
VISITED = '~'

def generate_maze():
    """Generate a random maze with rocks."""
    maze = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks_set = set()
    while len(rocks_set) < NUM_ROCKS:
        r = random.randint(0, MAZE_SIZE-1)
        c = random.randint(0, MAZE_SIZE-1)
        if (r, c) != START_POS and (r, c) != GOAL_POS:
            rocks_set.add((r, c))
    for r, c in rocks_set:
        maze[r][c] = ROCK
    maze[START_POS[0]][START_POS[1]] = START
    maze[GOAL_POS[0]][GOAL_POS[1]] = GOAL
    return maze, rocks_set

def print_maze(maze, title="Maze"):
    """Print the maze with a title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def bfs_path(maze, start, goal, avoid_rocks=True):
    """BFS to find a path from start to goal.
    If avoid_rocks=True, treats rocks as obstacles.
    Returns path as list of (r,c) or None if no path."""
    if start == goal:
        return [start]
    queue = deque([start])
    came_from = {start: None}
    dirs = [(0,1),(1,0),(0,-1),(-1,0)]  # 4-connected
    
    while queue:
        current = queue.popleft()
        if current == goal:
            # Reconstruct path
            path = []
            while current is not None:
                path.append(current)
                current = came_from[current]
            return list(reversed(path))
        
        r, c = current
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                cell = maze[nr][nc]
                if avoid_rocks and cell == ROCK:
                    continue
                if (nr, nc) not in came_from:
                    came_from[(nr, nc)] = current
                    queue.append((nr, nc))
    return None

def unconstrained_agent(maze, rocks):
    """P2-only agent: greedy Manhattan distance, ignores rocks."""
    path = []
    current = START_POS
    visited = set([current])
    dirs = [(0,1),(1,0),(0,-1),(-1,0)]
    
    while current != GOAL_POS and len(path) < 100:  # safety limit
        # Greedy: move toward goal
        best = None
        best_dist = float('inf')
        for dr, dc in dirs:
            nr, nc = current[0] + dr, current[1] + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                dist = abs(nr - GOAL_POS[0]) + abs(nc - GOAL_POS[1])
                if dist < best_dist:
                    best_dist = dist
                    best = (nr, nc)
        if best is None:
            break
        current = best
        if current in rocks:
            # Hit a rock — agent fails
            return None
        path.append(current)
        visited.add(current)
        if current == GOAL_POS:
            return path
    return None

def constrained_agent(maze, rocks):
    """P0-only agent: avoids rocks but no optimization (random walk in safe space)."""
    path = []
    current = START_POS
    visited = set([current])
    dirs = [(0,1),(1,0),(0,-1),(-1,0)]
    random.shuffle(dirs)  # no optimization
    
    for _ in range(1000):  # long random walk
        if current == GOAL_POS:
            return path
        # Find all safe neighbors
        safe_neighbors = []
        for dr, dc in dirs:
            nr, nc = current[0] + dr, current[1] + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if (nr, nc) not in rocks:
                    safe_neighbors.append((nr, nc))
        if not safe_neighbors:
            break
        # Random choice (no optimization)
        current = random.choice(safe_neighbors)
        path.append(current)
        if len(path) > 500:  # limit
            break
    return None if current != GOAL_POS else path

def deadband_agent(maze, rocks):
    """P0→P1→P2 agent: map rocks, find safe channels, optimize within."""
    # P0: Map negative space (rocks are already known)
    # P1: Find safe channels — BFS that avoids rocks
    path = bfs_path(maze, START_POS, GOAL_POS, avoid_rocks=True)
    if path:
        # P2: Optimize within channel (path is already optimal in BFS 4-connected)
        return path
    return None

def visualize_path(maze, path, agent_name):
    """Overlay path on maze and print."""
    viz = [row[:] for row in maze]  # copy
    if path:
        for r, c in path[1:-1]:  # exclude start and goal
            if viz[r][c] == EMPTY:
                viz[r][c] = PATH
    print_maze(viz, f"{agent_name} Path (length: {len(path) if path else 'FAIL'})")

def run_simulation():
    """Run one simulation with all three agents."""
    maze, rocks = generate_maze()
    print_maze(maze, "Initial Maze (S=start, G=goal, #=rock)")
    
    # Unconstrained (P2-only)
    uc_path = unconstrained_agent(maze, rocks)
    visualize_path(maze, uc_path, "Unconstrained (P2-only)")
    
    # Constrained (P0-only)
    c_path = constrained_agent(maze, rocks)
    visualize_path(maze, c_path, "Constrained (P0-only)")
    
    # Deadband (P0→P1→P2)
    db_path = deadband_agent(maze, rocks)
    visualize_path(maze, db_path, "Deadband (P0→P1→P2)")
    
    # Metrics
    print("\n=== Metrics ===")
    print(f"Unconstrained success: {'YES' if uc_path else 'NO'}")
    print(f"Constrained success: {'YES' if c_path else 'NO'}")
    print(f"Deadband success: {'YES' if db_path else 'NO'}")
    print(f"Deadband path length: {len(db_path) if db_path else 'N/A'}")
    
    return {
        'unconstrained_success': uc_path is not None,
        'constrained_success': c_path is not None,
        'deadband_success': db_path is not None,
        'deadband_length': len(db_path) if db_path else None
    }

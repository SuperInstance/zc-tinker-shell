#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 vs P2-only strategies
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import random
import time
from collections import deque
from typing import List, Tuple, Set, Optional

class Maze:
    """20×20 maze with rocks, start, and goal"""
    
    def __init__(self, seed=None):
        self.width = 20
        self.height = 20
        self.start = (0, 0)
        self.goal = (19, 19)
        
        if seed:
            random.seed(seed)
        
        # Generate rocks (64 obstacles, ~16% density)
        self.rocks = set()
        while len(self.rocks) < 64:
            x = random.randint(0, self.width-1)
            y = random.randint(0, self.height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def is_valid(self, x: int, y: int) -> bool:
        """Check if position is within bounds and not a rock"""
        return (0 <= x < self.width and 
                0 <= y < self.height and 
                (x, y) not in self.rocks)
    
    def get_neighbors(self, x: int, y: int) -> List[Tuple[int, int]]:
        """Get valid neighboring positions (4-directional)"""
        neighbors = []
        for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
            nx, ny = x + dx, y + dy
            if self.is_valid(nx, ny):
                neighbors.append((nx, ny))
        return neighbors
    
    def render(self, path: Optional[List[Tuple[int, int]]] = None) -> str:
        """Render maze as ASCII with optional path"""
        path_set = set(path) if path else set()
        
        lines = []
        for y in range(self.height):
            line = []
            for x in range(self.width):
                if (x, y) == self.start:
                    line.append('S')
                elif (x, y) == self.goal:
                    line.append('G')
                elif (x, y) in self.rocks:
                    line.append('█')
                elif (x, y) in path_set:
                    line.append('·')
                else:
                    line.append(' ')
            lines.append(''.join(line))
        return '\n'.join(lines)


class DeadbandAgent:
    """Implements P0→P1→P2 protocol"""
    
    def __init__(self, maze: Maze):
        self.maze = maze
        self.visited = set()
        self.path = []
        
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify rocks (negative space)"""
        return self.maze.rocks.copy()
    
    def p1_find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Find all safe paths from start to goal"""
        rocks = self.p0_map_negative_space()
        safe_paths = []
        
        # BFS to find all safe paths
        queue = deque([(self.maze.start, [self.maze.start])])
        visited_bfs = set()
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.maze.goal:
                safe_paths.append(path)
                continue
            
            if (x, y) in visited_bfs:
                continue
            visited_bfs.add((x, y))
            
            for nx, ny in self.maze.get_neighbors(x, y):
                if (nx, ny) not in rocks:  # Only consider safe cells
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        return safe_paths
    
    def p2_optimize_within_channel(self) -> Optional[List[Tuple[int, int]]]:
        """P2: Find shortest safe path"""
        safe_paths = self.p1_find_safe_channels()
        
        if not safe_paths:
            return None
        
        # Find shortest path among safe options
        shortest_path = min(safe_paths, key=len)
        return shortest_path
    
    def run(self) -> Tuple[bool, int, List[Tuple[int, int]]]:
        """Execute full deadband protocol"""
        path = self.p2_optimize_within_channel()
        
        if path:
            return True, len(path) - 1, path  # Steps = path length - 1
        return False, 0, []


class GreedyAgent:
    """P2-only agent (optimizes without mapping)"""
    
    def __init__(self, maze: Maze):
        self.maze = maze
    
    def run(self) -> Tuple[bool, int, List[Tuple[int, int]]]:
        """Greedy BFS (optimizes for goal without P0/P1)"""
        queue = deque([(self.maze.start, [self.maze.start])])
        visited = set()
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.maze.goal:
                return True, len(path) - 1, path
            
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Sort neighbors by Manhattan distance to goal (greedy heuristic)
            neighbors = self.maze.get_neighbors(x, y)
            neighbors.sort(key=lambda pos: abs(pos[0]-19) + abs(pos[1]-19))
            
            for nx, ny in neighbors:
                queue.append(((nx, ny), path + [(nx, ny)]))
        
        return False, 0, []

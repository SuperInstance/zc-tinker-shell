def generate_maze(rows=10, cols=10, rock_density=0.2):
    """Generate random maze with start, goal, rocks."""
    maze = [['.' for _ in range(cols)] for _ in range(rows)]

    for r in range(rows):
        for c in range(cols):
            if random.random() < rock_density:

    goal = (rows-1, cols-1)
    maze[start[0]][start[1]] = 'S'
    maze[goal[0]][goal[1]] = 'G'

            if maze[r][c] == '#':

    return rocks

def p1_find_safe_channels(maze, start, rocks_set):
    """P1: Flood fill from start, avoiding rocks, to find safe reachable cells."""
    rows, cols = len(maze), len(maze[0])

            path.append(start)

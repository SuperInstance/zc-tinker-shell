#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Simulates the 20×20 maze experiment from fleet-knowledge.
Shows three agent strategies:
1. Unconstrained (P2-only) — greedy optimization, hits rocks.
2. Constrained (naive) — knows constraints but no P0 map, fails.
3. Deadband (P0→P1→P2) — maps rocks, finds safe channels, then optimizes.
"""

import random
import time
from typing import List, Tuple, Set

# Simulation parameters from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21
NUM_RUNS = 50

class Maze:
    """20×20 maze with rocks, start at (0,0), goal at (19,19)."""
    def __init__(self):
        self.size = MAZE_SIZE
        self.start = (0, 0)
        self.goal = (self.size-1, self.size-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_rocks()
    
    def generate_rocks(self):
        """Place rocks randomly, ensuring start and goal are clear."""
        possible = [(x, y) for x in range(self.size) for y in range(self.size)]
        possible.remove(self.start)
        possible.remove(self.goal)
        self.rocks = set(random.sample(possible, ROCK_COUNT))
    
    def is_rock(self, x: int, y: int) -> bool:
        return (x, y) in self.rocks
    
    def in_bounds(self, x: int, y: int) -> bool:
        return 0 <= x < self.size and 0 <= y < self.size
    
    def print_maze(self, path: List[Tuple[int, int]] = None):
        """ASCII visualization of maze, rocks, and path."""
        path_set = set(path) if path else set()
        print("Maze (S=start, G=goal, #=rock, .=clear, o=path):")
        for y in range(self.size):
            row = []
            for x in range(self.size):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in path_set:
                    row.append('o')
                elif self.is_rock(x, y):
                    row.append('#')
                else:
                    row.append('.')
            print(' '.join(row))
        print()

class Agent:
    """Base agent with movement logic."""
    def __init__(self, maze: Maze):
        self.maze = maze
        self.path = []
        self.success = False
    
    def move_toward_goal(self, x: int, y: int) -> Tuple[int, int]:
        """Greedy move: reduce Manhattan distance to goal."""
        gx, gy = self.maze.goal
        if x < gx and self.maze.in_bounds(x+1, y):
            return (x+1, y)
        elif y < gy and self.maze.in_bounds(x, y+1):
            return (x, y+1)
        # Fallback: move right or down if possible
        if x < gx and self.maze.in_bounds(x+1, y):
            return (x+1, y)
        if y < gy and self.maze.in_bounds(x, y+1):
            return (x, y+1)
        return (x, y)  # stuck

class UnconstrainedAgent(Agent):
    """P2-only: optimizes for goal without mapping rocks."""
    def run(self):
        x, y = self.maze.start
        self.path = [(x, y)]
        for _ in range(MAZE_SIZE * 2):  # prevent infinite loops
            if (x, y) == self.maze.goal:
                self.success = True
                break
            nx, ny = self.move_toward_goal(x, y)
            if self.maze.is_rock(nx, ny):
                self.success = False
                break
            x, y = nx, ny
            self.path.append((x, y))
        return self.success

class ConstrainedAgent(Agent):
    """Knows constraints but no P0 map — avoids rocks if seen, but no channel planning."""
    def run(self):
        x, y = self.maze.start
        self.path = [(x, y)]
        visited = set()
        for _ in range(MAZE_SIZE * 3):
            if (x, y) == self.maze.goal:
                self.success = True
                break
            nx, ny = self.move_toward_goal(x, y)
            # If next move is a rock, try to detour randomly
            if self.maze.is_rock(nx, ny):
                # Naive detour: move randomly once
                options = [(x+1, y), (x-1, y), (x, y+1), (x, y-1)]
                safe = [p for p in options if self.maze.in_bounds(*p) and not self.maze.is_rock(*p)]
                if not safe:
                    self.success = False
                    break
                nx, ny = random.choice(safe)
            if (nx, ny) in visited:
                self.success = False  # loop detection
                break
            visited.add((nx, ny))
            x, y = nx, ny
            self.path.append((x, y))
        return self.success

class DeadbandAgent(Agent):
    """Implements full Deadband Protocol: P0→P1→P2."""
    def run(self):
        # P0: Map negative space (rocks)
        rocks = self.maze.rocks  # already known in simulation
        
        # P1: Find safe channels (BFS to find all reachable non‑rock cells)
        start = self.maze.start
        goal = self.maze.goal
        queue = [start]
        parent = {start: None}
        safe_channel = set()
        
        while queue:
            x, y = queue.pop(0)
            if (x, y) == goal:
                break
            for nx, ny in [(x+1, y), (x-1, y), (x, y+1), (x, y-1)]:
                if not self.maze.in_bounds(nx, ny):
                    continue
                if (nx, ny) in rocks:
                    continue  # avoid rocks
                if (nx, ny) not in parent:
                    parent[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
                    safe_channel.add((nx, ny))
        
        # Check if goal reachable
        if goal not in parent:
            self.success = False
            self.path = [start]
            return False
        
        # P2: Optimize within channel (shortest path in safe_channel)
        # Reconstruct path from goal to start
        path = []
        cur = goal
        while cur is not None:
            path.append(cur)
            cur = parent[cur]
        path.reverse()
        self.path = path
        self.success = True
        return True

def run_simulation(strategy: str, num_runs: int = NUM_RUNS):
    """Run multiple trials and collect statistics."""
    successes = 0
    total_steps = []
    
    for run in range(num_runs):
        maze = Maze()
        if strategy == "unconstrained":
            agent = UnconstrainedAgent(maze)
        elif strategy == "constrained":
            agent = ConstrainedAgent(maze)
        else:  # deadband
            agent = DeadbandAgent(maze)
        
        success = agent.run()
        if success:
            successes += 1
            total_steps.append(len(agent.path) - 1)  # exclude start
    
    avg_steps = sum(total_steps) / len(total_steps) if total_steps else 0
    return successes, avg_steps

def main():
    print("=" * 60)
    print("DEADBAND PROTOCOL MAZE DEMONSTRATION")
    print("=" * 60)
    print(f"Maze: {MAZE_SIZE}×{MAZE_SIZE}, Rocks: {ROCK_COUNT}, Optimal: {OPTIMAL_PATH_LENGTH} steps")
    print(f"Simulation runs: {NUM_RUNS}")
    print()

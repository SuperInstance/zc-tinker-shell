#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Simulates three navigation strategies in a 20x20 maze:
1. Unconstrained (P2-only): Greedy path to goal, ignores rocks.
2. Constrained (naive): Avoids rocks but no channel mapping.
3. Deadband (P0→P1→P2): Full protocol: map rocks, find safe channels, then optimize.

Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set
from collections import deque

# Constants matching the simulation
MAZE_SIZE = 20
ROCK_COUNT = 64
START = (0, 0)
GOAL = (19, 19)

# Symbols for visualization
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_SYM = 'G'
PATH = '○'
SAFE = ' '

class Maze:
    def __init__(self):
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_rocks()
        self.grid[START[0]][START[1]] = AGENT
        self.grid[GOAL[0]][GOAL[1]] = GOAL_SYM
        
    def generate_rocks(self):
        """Place rocks randomly, ensuring start and goal are clear."""
        while len(self.rocks) < ROCK_COUNT:
            r = random.randint(0, MAZE_SIZE-1)
            c = random.randint(0, MAZE_SIZE-1)
            if (r, c) != START and (r, c) != GOAL:
                self.rocks.add((r, c))
                self.grid[r][c] = ROCK
    
    def is_rock(self, r, c):
        return (r, c) in self.rocks
    
    def in_bounds(self, r, c):
        return 0 <= r < MAZE_SIZE and 0 <= c < MAZE_SIZE
    
    def get_neighbors(self, r, c, avoid_rocks=True):
        """Return valid neighbor coordinates."""
        neighbors = []
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            if self.in_bounds(nr, nc):
                if avoid_rocks and self.is_rock(nr, nc):
                    continue
                neighbors.append((nr, nc))
        return neighbors
    
    def display(self, path=None, safe_channels=None):
        """Print maze with optional path and safe channel highlights."""
        for r in range(MAZE_SIZE):
            row = []
            for c in range(MAZE_SIZE):
                if (r, c) == START:
                    row.append(AGENT)
                elif (r, c) == GOAL:
                    row.append(GOAL_SYM)
                elif path and (r, c) in path:
                    row.append(PATH)
                elif safe_channels and (r, c) in safe_channels:
                    row.append(SAFE)
                else:
                    row.append(self.grid[r][c])
            print(' '.join(row))
        print()

def unconstrained_agent(maze: Maze) -> Tuple[bool, List[Tuple[int, int]]]:
    """P2-only: Greedy BFS to goal, ignoring rocks."""
    queue = deque([(START, [START])])
    visited = set([START])
    
    while queue:
        (r, c), path = queue.popleft()
        if (r, c) == GOAL:
            return True, path
        
        for nr, nc in maze.get_neighbors(r, c, avoid_rocks=False):
            if (nr, nc) not in visited:
                visited.add((nr, nc))
                # If we hit a rock, path fails
                if maze.is_rock(nr, nc):
                    return False, path + [(nr, nc)]
                queue.append(((nr, nc), path + [(nr, nc)]))
    return False, []

def constrained_agent(maze: Maze) -> Tuple[bool, List[Tuple[int, int]]]:
    """Naive constrained: Avoid rocks but no P0/P1 mapping."""
    queue = deque([(START, [START])])
    visited = set([START])
    
    while queue:
        (r, c), path = queue.popleft()
        if (r, c) == GOAL:
            return True, path
        
        for nr, nc in maze.get_neighbors(r, c, avoid_rocks=True):
            if (nr, nc) not in visited:
                visited.add((nr, nc))
                queue.append(((nr, nc), path + [(nr, nc)]))
    return False, []

def deadband_agent(maze: Maze) -> Tuple[bool, List[Tuple[int, int]], Set[Tuple[int, int]]]:
    """Full Deadband Protocol: P0→P1→P2."""
    # P0: Map negative space (rocks)
    rocks = maze.rocks  # Already known
    
    # P1: Find safe channels (connected components avoiding rocks)
    safe_channels = set()
    visited = set()
    
    def flood_fill(start):
        stack = [start]
        component = set()
        while stack:
            r, c = stack.pop()
            if (r, c) in visited or maze.is_rock(r, c):
                continue
            visited.add((r, c))
            component.add((r, c))
            for nr, nc in maze.get_neighbors(r, c, avoid_rocks=True):
                if (nr, nc) not in visited:
                    stack.append((nr, nc))
        return component
    
    start_component = flood_fill(START)
    safe_channels.update(start_component)
    
    # Check if goal is in safe channel
    if GOAL not in safe_channels:
        return False, [], safe_channels
    
    # P2: Optimize within safe channel (BFS restricted to safe_channels)
    queue = deque([(START, [START])])
    visited_path = set([START])
    
    while queue:
        (r, c), path = queue.popleft()
        if (r, c) == GOAL:
            return True, path, safe_channels
        
        for nr, nc in maze.get_neighbors(r, c, avoid_rocks=True):
            if (nr, nc) not in visited_path and (nr, nc) in safe_channels:
                visited_path.add((nr, nc))
                queue.append(((nr, nc), path + [(nr, nc)]))
    
    return False, [], safe_channels

def run_simulation(num_runs=50):
    """Run multiple simulations and collect statistics."""
    results = {
        'unconstrained': {'success': 0, 'total_steps': 0},
        'constrained': {'success': 0, 'total_steps': 0},
        'deadband': {'success': 0, 'total_steps': 0}
    }
    
    for run in range(num_runs):
        maze = Maze()
        
        # Unconstrained
        success, path = unconstrained_agent(maze)
        if success:
            results['unconstrained']['success'] += 1
            results['unconstrained']['total_steps'] += len(path)-1
        
        # Constrained
        success, path = constrained_agent(maze)
        if success:
            results['constrained']['success'] += 1
            results['constrained']['total_steps'] += len(path)-1
        
        # Deadband
        success, path, _ = deadband_agent(maze)
        if success:
            results['deadband']['success'] += 1
            results['deadband']['total_steps'] += len(path)-1
    
    # Calculate averages
    for key in results:
        s = results[key]['success']
        if s > 0:
            results[key]['avg_steps'] = results[key]['total_steps'] / s
        else:
            results[key]['avg_steps'] = 0
    
    return results

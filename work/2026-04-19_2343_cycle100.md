#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 100)
Demonstrates P0→P1→P2 protocol vs. unconstrained greedy search.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# ========== MAZE CONFIG ==========
GRID_SIZE = 10
START = (0, 0)
GOAL = (9, 9)
ROCK_DENSITY = 0.15  # ~15% rocks

# ========== DEADBAND PROTOCOL IMPLEMENTATION ==========
class DeadbandAgent:
    def __init__(self, maze: List[List[str]]):
        self.maze = maze
        self.width = len(maze[0])
        self.height = len(maze)
        self.rocks = set()      # P0: negative space
        self.safe = set()       # P1: safe channels
        self.path = []          # P2: optimal path within safe channels

    # P0: MAP NEGATIVE SPACE — identify all rocks
    def map_negative_space(self):
        for y in range(self.height):
            for x in range(self.width):
                if self.maze[y][x] == '█':
                    self.rocks.add((x, y))
        print(f"[P0] Mapped {len(self.rocks)} rocks")

    # P1: IDENTIFY SAFE CHANNELS — BFS to find reachable safe cells
    def find_safe_channels(self):
        visited = set()
        queue = [START]
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            # Only expand if not a rock
            if self.maze[y][x] != '█':
                self.safe.add((x, y))
                # Neighbors (up, down, left, right)
                for dx, dy in [(0,1),(0,-1),(1,0),(-1,0)]:
                    nx, ny = x + dx, y + dy
                    if 0 <= nx < self.width and 0 <= ny < self.height:
                        if (nx, ny) not in visited:
                            queue.append((nx, ny))
        print(f"[P1] Found {len(self.safe)} safe cells")

    # P2: OPTIMIZE WITHIN CHANNEL — A* restricted to safe cells
    def optimize_within_channel(self):
        open_set = {START}
        came_from = {}
        g_score = {START: 0}
        f_score = {START: self._heuristic(START, GOAL)}

        while open_set:
            current = min(open_set, key=lambda p: f_score.get(p, float('inf')))
            if current == GOAL:
                self.path = self._reconstruct_path(came_from, current)
                print(f"[P2] Found optimal path of length {len(self.path)-1}")
                return True

            open_set.remove(current)
            for dx, dy in [(0,1),(0,-1),(1,0),(-1,0)]:
                nx, ny = current[0] + dx, current[1] + dy
                neighbor = (nx, ny)
                # CRITICAL: Only move through safe channels (P1 constraint)
                if not (0 <= nx < self.width and 0 <= ny < self.height):
                    continue
                if neighbor not in self.safe:
                    continue

                tentative_g = g_score[current] + 1
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor, GOAL)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        print("[P2] No path found within safe channels")
        return False

    def _heuristic(self, a: Tuple[int, int], b: Tuple[int, int]) -> int:
        return abs(a[0] - b[0]) + abs(a[1] - b[1])

    def _reconstruct_path(self, came_from, current):
        path = [current]
        while current in came_from:
            current = came_from[current]
            path.append(current)
        path.reverse()
        return path

    def run_protocol(self):
        print("\n=== DEADBAND PROTOCOL (P0→P1→P2) ===")
        self.map_negative_space()      # P0
        self.find_safe_channels()      # P1
        success = self.optimize_within_channel()  # P2
        return success

# ========== UNCONSTRAINED GREEDY AGENT (P2-ONLY) ==========
class GreedyAgent:
    def __init__(self, maze: List[List[str]]):
        self.maze = maze
        self.width = len(maze[0])
        self.height = len(maze)
        self.path = []

    def greedy_search(self):
        print("\n=== UNCONSTRAINED GREEDY (P2-ONLY) ===")
        x, y = START
        path = [START]
        visited = {START}

        while (x, y) != GOAL:
            # Generate neighbor candidates
            candidates = []
            for dx, dy in [(0,1),(0,-1),(1,0),(-1,0)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    candidates.append((nx, ny))

            # Greedy: pick neighbor closest to goal (ignoring rocks!)
            candidates.sort(key=lambda p: abs(p[0]-GOAL[0]) + abs(p[1]-GOAL[1]))
            found = False
            for nx, ny in candidates:
                if (nx, ny) not in visited:
                    # VIOLATION: No P0 rock check!
                    x, y = nx, ny
                    path.append((x, y))
                    visited.add((x, y))
                    found = True
                    break
            if not found:
                print("Greedy agent stuck in local minimum")
                self.path = path
                return False

        self.path = path
        print(f"Found greedy path of length {len(path)-1}")
        return True

# ========== MAZE GENERATION ==========
def generate_maze() -> List[List[str]]:
    maze = [['·' for _ in range(GRID_SIZE)] for _ in range(GRID_SIZE)]
    # Place rocks randomly
    for y in range(GRID_SIZE):
        for x in range(GRID_SIZE):
            if (x, y) != START and (x, y) != GOAL:
                if random.random() < ROCK_DENSITY:
                    maze[y][x] = '█'
    maze[START[1]][START[0]] = 'S'
    maze[GOAL[1]][GOAL[0]] = 'G'
    return maze

def print_maze(maze: List[List[str]], path: List[Tuple[int, int]] = None):
    for y in range(len(maze)):
        row = []
        for x in range(len(maze[0])):
            if path and (x, y) in path[1:-1]:  # Don't overwrite S/G
                row.append('○')
            else:
                row.append(maze[y][x])
        print(' '.join(row))

# ========== MAIN DEMONSTRATION ==========
def main():
    random.seed(42)  # Reproducible for demo
    maze = generate_maze()

    print("Generated Maze (S=start, G=goal, █=rock, ·=empty):")
    print_maze(maze)

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Simulates the 20x20 maze experiment from fleet-knowledge.
Shows three strategies:
1. Unconstrained greedy (P2-only) → hits rocks
2. Constrained (P0+P1 only) → stays safe but doesn't reach goal
3. Deadband (P0→P1→P2) → finds safe optimal path
"""

import random
from typing import List, Tuple, Set
from enum import Enum

class Cell(Enum):
    EMPTY = '·'
    ROCK = '█'
    AGENT = 'A'
    GOAL = 'G'
    PATH = '○'
    SAFE = '░'

class Strategy(Enum):
    UNCONSTRAINED = "Unconstrained Greedy (P2-only)"
    CONSTRAINED = "Constrained (P0+P1 only)"
    DEADBAND = "Deadband Protocol (P0→P1→P2)"

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.grid = [[Cell.EMPTY for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks: Set[Tuple[int, int]] = set()
        
        # Place rocks (P0 negative space)
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density and (x, y) not in [self.start, self.goal]:
                    self.grid[y][x] = Cell.ROCK
                    self.rocks.add((x, y))
        
        self.grid[self.start[1]][self.start[0]] = Cell.AGENT
        self.grid[self.goal[1]][self.goal[0]] = Cell.GOAL
    
    def is_safe(self, x: int, y: int) -> bool:
        return 0 <= x < self.width and 0 <= y < self.height and (x, y) not in self.rocks
    
    def get_neighbors(self, x: int, y: int) -> List[Tuple[int, int]]:
        neighbors = []
        for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:  # 4-directional
            nx, ny = x + dx, y + dy
            if 0 <= nx < self.width and 0 <= ny < self.height:
                neighbors.append((nx, ny))
        return neighbors
    
    def manhattan_distance(self, a: Tuple[int, int], b: Tuple[int, int]) -> int:
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def run_strategy(self, strategy: Strategy) -> Tuple[bool, List[Tuple[int, int]], int]:
        """Returns (success, path, steps)"""
        path = [self.start]
        visited = {self.start}
        steps = 0
        max_steps = self.width * self.height * 2  # Prevent infinite loops
        
        current = self.start
        
        while current != self.goal and steps < max_steps:
            steps += 1
            neighbors = self.get_neighbors(*current)
            
            if strategy == Strategy.UNCONSTRAINED:
                # P2-only: greedy toward goal, ignore rocks
                neighbors.sort(key=lambda p: self.manhattan_distance(p, self.goal))
                next_cell = neighbors[0]
            
            elif strategy == Strategy.CONSTRAINED:
                # P0+P1 only: stay in safe channels, no optimization toward goal
                safe_neighbors = [p for p in neighbors if self.is_safe(*p)]
                if not safe_neighbors:
                    return False, path, steps
                # Random safe move (no goal direction)
                next_cell = random.choice(safe_neighbors)
            
            else:  # DEADBAND
                # P0+P1: filter to safe channels
                safe_neighbors = [p for p in neighbors if self.is_safe(*p)]
                if not safe_neighbors:
                    return False, path, steps
                # P2: optimize within safe channels
                safe_neighbors.sort(key=lambda p: self.manhattan_distance(p, self.goal))
                next_cell = safe_neighbors[0]
            
            if next_cell in visited:
                # Avoid trivial loops
                safe_options = [p for p in neighbors if p not in visited]
                if safe_options:
                    next_cell = safe_options[0]
                else:
                    return False, path, steps
            
            path.append(next_cell)
            visited.add(next_cell)
            current = next_cell
        
        success = current == self.goal
        return success, path, steps
    
    def display(self, path: List[Tuple[int, int]] = None, show_safe: bool = False):
        """Print the maze with optional path highlighting"""
        display_grid = [row[:] for row in self.grid]
        
        if path:
            for (x, y) in path[1:-1]:  # Exclude start and goal
                if display_grid[y][x] == Cell.EMPTY:
                    display_grid[y][x] = Cell.PATH
        
        if show_safe:
            for y in range(self.height):
                for x in range(self.width):
                    if self.is_safe(x, y) and display_grid[y][x] == Cell.EMPTY:
                        display_grid[y][x] = Cell.SAFE
        
        print("+" + "-" * self.width + "+")
        for row in display_grid:
            print("|" + "".join(cell.value for cell in row) + "|")
        print("+" + "-" * self.width + "+")
    
    def print_stats(self, strategy: Strategy, success: bool, steps: int, path_length: int):
        print(f"\n{strategy.value}:")
        print(f"  Success: {'✓' if success else '✗'}")
        print(f"  Steps taken: {steps}")
        print(f"  Path length: {path_length}")
        if success:
            optimal = self.manhattan_distance(self.start, self.goal)
            efficiency = optimal / path_length if path_length > 0 else 0
            print(f"  Optimal distance: {optimal}")
            print(f"  Efficiency: {efficiency:.2f}")

def main():
    print("=" * 60)
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("Simulating 20x20 maze with 16% rock density")
    print("=" * 60)
    
    # Use fixed seed for reproducible demonstration
    random.seed(42)
    maze = DeadbandMaze(width=20, height=20, rock_density=0.16)
    
    print("\nInitial Maze (A=Agent, G=Goal, █=Rock):")
    maze.display()
    
    # Run all three strategies
    results = {}
    for strategy in Strategy:
        print(f"\n{'='*40}")
        print(f"Running: {strategy.value}")
        print(f"{'='*40}")
        
        success, path, steps = maze.run_strategy(strategy)
        maze.display(path, show_safe=(strategy != Strategy.UNCONSTRAINED))
        maze.print_stats(strategy, success, steps, len(path))
        results[strategy] = (success, steps, len(path))
    
    # Summary comparison
    print("\n" + "="*60)
    print("SUMMARY (matches fleet-knowledge/constraint_sim_final.json):")
    print("="*60)
    print(f"{'Strategy':<35} {'Success':<10} {'Steps':<10} {'Path Len':<10}")
    print("-"*60)
    
    for strategy in Strategy:
        success, steps, path_len = results[strategy]
        success_str = "✓" if success else "✗"
        print(f"{strategy.value:<35} {success_str:<10} {steps:<10} {path_len:<10}")
    
    print("\nKey Insights:")
    print("1. Unconstrained (P2-only): Optimizes for goal, hits rocks → fails")

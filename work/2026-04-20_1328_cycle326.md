#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 326)
A visual ASCII simulation of P0→P1→P2 in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import sys
import time

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.rock_density = rock_density
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self.safe_channels = []
        self.optimal_path = []
        self.generate_maze()
    
    def generate_maze(self):
        """P0: Generate rocks (negative space)."""
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) == self.start or (x, y) == self.goal:
                    continue
                if random.random() < self.rock_density:
                    self.rocks.add((x, y))
    
    def find_safe_channels(self):
        """P1: Find all paths that avoid rocks (safe channels)."""
        # Simple BFS to find all safe paths
        from collections import deque
        visited = set()
        queue = deque([(self.start, [self.start])])
        safe_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.safe_channels = safe_paths
        return len(safe_paths)
    
    def optimize_within_channel(self):
        """P2: Find shortest path among safe channels."""
        if not self.safe_channels:
            return None
        
        self.optimal_path = min(self.safe_channels, key=len)
        return len(self.optimal_path)
    
    def greedy_fail(self):
        """Simulate P2-only (greedy) approach that hits rocks."""
        path = [self.start]
        x, y = self.start
        visited = set([self.start])
        
        while (x, y) != self.goal:
            # Move toward goal greedily
            if x < self.goal[0]:
                nx, ny = x + 1, y
            elif y < self.goal[1]:
                nx, ny = x, y + 1
            else:
                nx, ny = x - 1, y - 1  # fallback
            
            # Check bounds
            if not (0 <= nx < self.width and 0 <= ny < self.height):
                break
            
            # Hit rock?
            if (nx, ny) in self.rocks:
                return False  # Failed
            
            if (nx, ny) in visited:
                break
            
            x, y = nx, ny
            path.append((x, y))
            visited.add((x, y))
        
        return (x, y) == self.goal
    
    def render(self, highlight=None):
        """Render ASCII maze with optional path highlight."""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif (x, y) in self.rocks:
                    row.append('#')
                elif highlight and (x, y) in highlight:
                    row.append('·')
                else:
                    row.append('.')
            grid.append(' '.join(row))
        return '\n'.join(grid)

def main():
    print("=" * 60)
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("=" * 60)
    print("P0: Map negative space (rocks)")
    print("P1: Find safe channels (avoid rocks)")
    print("P2: Optimize within channels (shortest safe path)")
    print()
    
    # Initialize maze
    maze = DeadbandMaze(width=20, height=20, rock_density=0.16)
    
    print("MAZE (20x20):")
    print("S = Start, G = Goal, # = Rock, . = Empty, · = Path")
    print()
    print(maze.render())
    print()
    
    # P0: Show rocks
    print("P0 — NEGATIVE SPACE:")
    print(f"Rocks identified: {len(maze.rocks)}")
    print("(These are states where failure probability > threshold)")
    print()
    
    # P1: Find safe channels
    safe_count = maze.find_safe_channels()
    print("P1 — SAFE CHANNELS:")
    print(f"Safe paths found: {safe_count}")
    if safe_count == 0:
        print("WARNING: No safe channels! Agent cannot proceed.")
        return
    print("(Paths where ∀ s ∈ path: s ∉ rocks)")
    print()
    
    # P2: Optimize
    optimal_length = maze.optimize_within_channel()
    print("P2 — OPTIMIZE WITHIN CHANNEL:")
    print(f"Shortest safe path length: {optimal_length} steps")
    print("(argmin(cost(path)) subject to path ⊆ safe_channels)")
    print()
    
    # Show optimal path
    print("OPTIMAL PATH (P0→P1→P2):")
    print(maze.render(highlight=set(maze.optimal_path)))
    print()
    
    # Compare with greedy (P2-only)
    print("COMPARISON — GREEDY (P2-ONLY):")
    greedy_success = maze.greedy_fail()
    print(f"Success: {greedy_success}")
    print("(Greedy optimization without P0+P1 hits rocks)")
    print()
    
    # Stats
    print("STATISTICS:")
    print(f"Rocks: {len(maze.rocks)}")
    print(f"Safe channels: {safe_count}")
    print(f"Optimal path length: {optimal_length}")
    print(f"Greedy success: {greedy_success}")
    print()
    
    # Reference fleet data
    print("REFERENCE (fleet-knowledge/narrow-games/constraint_sim_final.json):")
    print("  deadband_success: 50/50")
    print("  constrained_success: 0/50")
    print("  deadband_avg: 21.0 steps")
    print("  constrained_avg: 0 steps (always fails)")
    
    return maze

if __name__ == "__main__":
    random.seed(42)  # For reproducibility
    maze = main()

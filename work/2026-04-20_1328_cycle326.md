**Output:** Below is the final Python script. It includes:

2. P0: Map negative space (rocks).
3. P1: Find safe channels (deadbands) using BFS to identify reachable safe cells.
4. P2: Optimize within channel using A* search constrained to safe cells.
5. Visual ASCII output showing each step.

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo
P0: Map negative space (rocks)
P1: Find safe channels (deadbands)
P2: Optimize within channel
"""

import random
from collections import deque
import heapq

# Constants
WIDTH, HEIGHT = 20, 20
ROCK_DENSITY = 0.15  # ~15% rocks
START = (0, 0)
GOAL = (WIDTH-1, HEIGHT-1)

def generate_environment():
    """Create a random 20x20 grid with rocks, start, and goal."""
    grid = [['.' for _ in range(WIDTH)] for _ in range(HEIGHT)]
    
    # Place rocks
    rocks = []
    for y in range(HEIGHT):
        for x in range(WIDTH):
            if (x, y) != START and (x, y) != GOAL and random.random() < ROCK_DENSITY:
                grid[y][x] = '#'
                rocks.append((x, y))
    
    # Place start and goal
    grid[START[1]][START[0]] = 'S'
    grid[GOAL[1]][GOAL[0]] = 'G'
    
    return grid, rocks

def print_grid(grid, title="Environment"):
    """Print ASCII grid with title."""
    print(f"\n=== {title} ===")
    for row in grid:
        print(' '.join(row))
    print()

def p0_map_negative_space(grid, rocks):
    """P0: Identify rocks (negative space)."""
    print("P0: MAPPING NEGATIVE SPACE (ROCKS)")
    print(f"Found {len(rocks)} rock cells.")
    # Visualize rocks
    rock_grid = [row[:] for row in grid]
    for x, y in rocks:
        rock_grid[y][x] = '#'
    print_grid(rock_grid, "P0 - Rocks (Negative Space)")
    return set(rocks)

def p1_find_safe_channels(grid, rocks_set):
    """P1: Find all cells reachable from start without hitting rocks."""
    print("P1: FINDING SAFE CHANNELS (DEADBANDS)")
    
    safe_cells = set()
    queue = deque([START])
    visited = set([START])
    
    # 4-direction movements
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        x, y = queue.popleft()
        safe_cells.add((x, y))
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < WIDTH and 0 <= ny < HEIGHT:
                if (nx, ny) not in visited and (nx, ny) not in rocks_set:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    
    # Create visualization
    channel_grid = [row[:] for row in grid]
    for y in range(HEIGHT):
        for x in range(WIDTH):
            if (x, y) in safe_cells and channel_grid[y][x] == '.':
                channel_grid[y][x] = '~'  # Safe channel marker
    
    print_grid(channel_grid, "P1 - Safe Channels (Deadbands)")
    print(f"Found {len(safe_cells)} safe cells reachable from start.")
    
    return safe_cells

def p2_optimize_within_channel(grid, safe_cells):
    """P2: Find shortest path from start to goal within safe channels using A*."""
    print("P2: OPTIMIZING WITHIN CHANNEL (A* on Safe Cells)")
    
    # Heuristic: Manhattan distance
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    # A* search
    open_set = []
    heapq.heappush(open_set, (0, START))
    came_from = {}
    g_score = {START: 0}
    f_score = {START: heuristic(START, GOAL)}
    
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while open_set:
        current_f, current = heapq.heappop(open_set)
        
        if current == GOAL:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(START)
            path.reverse()
            return path
        
        for dx, dy in directions:
            neighbor = (current[0] + dx, current[1] + dy)
            if 0 <= neighbor[0] < WIDTH and 0 <= neighbor[1] < HEIGHT:
                if neighbor not in safe_cells:
                    continue  # Skip unsafe cells (P1 constraint)
                
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, GOAL)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return None  # No path found

def visualize_path(grid, path):
    """Overlay path on grid for visualization."""
    path_grid = [row[:] for row in grid]
    for x, y in path[1:-1]:  # Exclude start and goal
        if path_grid[y][x] == '.':
            path_grid[y][x] = '*'
    print_grid(path_grid, "P2 - Optimal Path Within Channel")
    print(f"Path length: {len(path)-1} steps")

def main():
    print("DEADBAND PROTOCOL DEMO")
    print("=" * 40)
    
    # Generate environment
    grid, rocks = generate_environment()
    print_grid(grid, "Initial Environment")
    
    # P0: Map negative space
    rocks_set = p0_map_negative_space(grid, rocks)
    
    # P1: Find safe channels
    safe_cells = p1_find_safe_channels(grid, rocks_set)
    
    # Check if goal is reachable
    if GOAL not in safe_cells:
        print("ERROR: Goal is not in safe channels. Protocol violation avoided.")
        return
    
    # P2: Optimize within channel
    path = p2_optimize_within_channel(grid, safe_cells)
    
    if path:
        visualize_path(grid, path)
        print("SUCCESS: Deadband Protocol completed.")
        print(f"P0→P1→P2: {len(rocks)} rocks → {len(safe_cells)} safe cells → {len(path)-1} step path")
    else:
        print("FAILURE: No path found within safe channels.")

if __name__ == "__main__":
    main()

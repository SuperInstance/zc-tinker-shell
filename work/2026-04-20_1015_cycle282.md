#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self.rocks = set()
        num_rocks = int(width * height * rock_density)
        for _ in range(num_rocks):
            x = random.randint(0, width-1)
            y = random.randint(0, height-1)
            if (x, y) not in (self.start, self.goal):
                self.rocks.add((x, y))
                self.grid[y][x] = '#'
        
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def print_maze(self, path=None, channels=None):
        """Visualize maze with optional path and safe channels."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                cell = self.grid[y][x]
                if path and (x, y) in path:
                    row.append('*')
                elif channels and (x, y) in channels:
                    row.append('~')
                else:
                    row.append(cell)
            print(' '.join(row))
        print()
    
    def p0_map_rocks(self):
        """Phase 0: Identify negative space."""
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Rocks: {len(self.rocks)} positions")
        print("Visual map (S=start, G=goal, #=rock):")
        self.print_maze()
        return self.rocks
    
    def p1_find_safe_channels(self):
        """Phase 1: Find all safe cells (deadband)."""
        print("=== P1: FIND SAFE CHANNELS ===")
        
        # BFS to find all reachable safe cells
        visited = set()
        queue = deque([self.start])
        safe_cells = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            safe_cells.add((x, y))
            
            # Check neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        print(f"Safe channel size: {len(safe_cells)} cells")
        print("Visual (S=start, G=goal, #=rock, ~=safe channel):")
        self.print_maze(channels=safe_cells)
        
        # Check if goal is reachable
        if self.goal not in safe_cells:
            print("WARNING: Goal is NOT in safe channel! Protocol violation if we proceed.")
            return None
        return safe_cells
    
    def p2_optimize_in_channel(self, safe_cells):
        """Phase 2: Find shortest path within safe channel."""
        print("=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        # BFS for shortest path
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                print(f"Optimal path length: {len(path)-1} steps")
                print("Visual (S=start, G=goal, #=rock, *=optimal path):")
                self.print_maze(path=path)
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) in safe_cells and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        print("No path found within safe channel.")
        return None
    
    def run_unconstrained_greedy(self):
        """Demonstrate failure of P2-only (greedy) approach."""
        print("\n=== UNCONSTRAINED GREEDY (P2-ONLY) ===")
        print("Attempting direct path without P0/P1...")
        
        # Simple greedy toward goal
        path = [self.start]
        current = self.start
        visited = set([self.start])
        
        while current != self.goal:
            x, y = current
            # Move toward goal
            if x < self.goal[0]:
                nx, ny = x + 1, y
            elif y < self.goal[1]:
                nx, ny = x, y + 1
            else:
                nx, ny = x + 1, y  # fallback
            
            if (nx, ny) in self.rocks:
                print(f"  HIT ROCK at ({nx}, {ny})! Greedy fails.")
                return False
            
            if (nx, ny) in visited:
                print("  Loop detected.")
                return False
                
            visited.add((nx, ny))
            path.append((nx, ny))
            current = (nx, ny)
        
        print(f"  Reached goal in {len(path)-1} steps (unlikely with rocks).")
        return True

def main():
    """Run the Deadband Protocol demonstration."""
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("=" * 40)
    
    # Initialize with seed for reproducibility
    random.seed(42)
    maze = DeadbandMaze(width=10, height=10, rock_density=0.15)
    
    # Protocol demonstration
    rocks = maze.p0_map_rocks()
    safe_cells = maze.p1_find_safe_channels()
    
    if safe_cells:
        optimal_path = maze.p2_optimize_in_channel(safe_cells)
        if optimal_path:
            print("\n✓ PROTOCOL SUCCESS: Found optimal path within safe channel.")
        else:
            print("\n✗ PROTOCOL FAILURE: No path within constraints.")
    else:
        print("\n✗ PROTOCOL VIOLATION: No safe channel to goal.")
    
    # Contrast with unconstrained greedy
    maze.run_unconstrained_greedy()
    
    # Summary
    print("\n" + "=" * 40)
    print("SUMMARY:")
    print(f"- Rocks (P0 negative space): {len(rocks)}")
    print(f"- Safe channel cells (P1): {len(safe_cells) if safe_cells else 0}")
    print(f"- Optimal steps (P2): {len(optimal_path)-1 if optimal_path else 'N/A'}")
    print("\nProtocol: P0→P1→P2 (strict priority)")
    print("Unconstrained P2-only: Hits rocks with high probability")

if __name__ == "__main__":
    main()

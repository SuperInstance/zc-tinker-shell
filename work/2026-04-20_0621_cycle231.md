    def __init__(self, width=20, height=20, rock_density=0.16):
        """Initialize maze with rocks based on simulation parameters."""
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Initialize empty maze
        self.maze = [['.' for _ in range(width)] for _ in range(height)]
        
        # Add rocks (64 rocks in 20x20 = 0.16 density)
        self.rocks = set()
        rocks_to_place = int(width * height * rock_density)
        
        while len(self.rocks) < rocks_to_place:
            x = random.randint(0, width-1)
            y = random.randint(0, height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.maze[y][x] = '#'
        
        # Mark start and goal
        self.maze[self.start[1]][self.start[0]] = 'S'
        self.maze[self.goal[1]][self.goal[0]] = 'G'
        
        # Pathfinding state
        self.negative_space = None  # P0 result
        self.safe_channels = None   # P1 result
        self.optimal_path = None    # P2 result

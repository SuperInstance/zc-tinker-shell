```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Simulates the constraint theory results from fleet-knowledge.
"""

import random
from collections import deque

# Simulation constants from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

def generate_maze(size, rocks):
    """Create a maze with start (S), goal (G), and rocks (#)."""
    maze = [['.' for _ in range(size)] for _ in range(size)]
    
    # Start at top-left, goal at bottom-right
    maze[0][0] = 'S'
    maze[size-1][size-1] = 'G'
    
    # Place rocks randomly (but not on start/goal)
    placed = 0
    while placed < rocks:
        x = random.randint(0, size-1)
        y = random.randint(0, size-1)
        if maze[x][y] == '.':
            maze[x][y] = '#'
            placed += 1
    
    return maze

def print_maze(maze, title="Maze"):
    """Print ASCII maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def p0_map_rocks(maze):
    """P0: Map negative space — identify all rock coordinates."""
    rocks = []
    for i in range(len(maze)):
        for j in range(len(maze[0])):
            if maze[i][j] == '#':
                rocks.append((i, j))
    return rocks

def p1_find_safe_channels(maze, rocks):
    """P1: Find safe channels — BFS to identify reachable areas avoiding rocks."""
    size = len(maze)
    visited = [[False for _ in range(size)] for _ in range(size)]
    directions = [(0,1), (1,0), (0,-1), (-1,0)]
    
    # Mark rocks as visited (blocked)
    for x, y in rocks:
        visited[x][y] = True
    
    # BFS from start
    queue = deque([(0, 0)])
    visited[0][0] = True
    safe_cells = [(0, 0)]
    
    while queue:
        x, y = queue.popleft()
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < size and 0 <= ny < size and not visited[nx][ny]:
                visited[nx][ny] = True
                safe_cells.append((nx, ny))
                queue.append((nx, ny))
    
    return safe_cells

def p2_optimize_in_channel(maze, safe_cells):
    """P2: Optimize within safe channels — BFS shortest path using only safe cells."""
    size = len(maze)
    directions = [(0,1), (1,0), (0,-1), (-1,0)]
    
    # Convert safe cells to set for O(1) lookup
    safe_set = set(safe_cells)
    
    # BFS for shortest path
    queue = deque([(0, 0, 0, [])])  # (x, y, dist, path)
    visited = [[False for _ in range(size)] for _ in range(size)]
    visited[0][0] = True
    
    while queue:
        x, y, dist, path = queue.popleft()
        
        if (x, y) == (size-1, size-1):
            return dist, path + [(x, y)]
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < size and 0 <= ny < size:
                if (nx, ny) in safe_set and not visited[nx][ny]:
                    visited[nx][ny] = True
                    queue.append((nx, ny, dist + 1, path + [(x, y)]))
    
    return None, []  # No path found

def visualize_protocol_steps(maze, rocks, safe_cells, path):
    """Create visualizations for each protocol step."""
    size = len(maze)
    
    # P0 visualization: show rocks
    p0_viz = [row[:] for row in maze]
    for x, y in rocks:
        p0_viz[x][y] = '#'
    print_maze(p0_viz, "P0: Map Rocks (Negative Space)")
    
    # P1 visualization: show safe channels
    p1_viz = [row[:] for row in maze]
    for x, y in safe_cells:
        if p1_viz[x][y] == '.':
            p1_viz[x][y] = '·'
    print_maze(p1_viz, "P1: Safe Channels (Deadbands)")
    
    # P2 visualization: show optimal path
    p2_viz = [row[:] for row in maze]
    for x, y in path:
        if p2_viz[x][y] not in ['S', 'G']:
            p2_viz[x][y] = '○'
    print_maze(p2_viz, "P2: Optimal Path Within Channels")

def main():
    """Run the Deadband Protocol demonstration."""
    print("Deadband Protocol Maze Demo")
    print("=" * 40)
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}")
    print(f"Optimal path length from simulation: {OPTIMAL_PATH_LENGTH}")
    print()
    
    # Generate maze matching simulation parameters
    random.seed(42)  # Deterministic for demo
    maze = generate_maze(MAZE_SIZE, ROCK_COUNT)
    print_maze(maze, "Initial Maze (S=start, G=goal, #=rock)")
    
    # P0: Map rocks
    rocks = p0_map_rocks(maze)
    print(f"P0: Mapped {len(rocks)} rocks (negative space)")
    
    # P1: Find safe channels
    safe_cells = p1_find_safe_channels(maze, rocks)
    print(f"P1: Found {len(safe_cells)} safe cells (channels)")
    
    # P2: Optimize within channels
    path_length, path = p2_optimize_in_channel(maze, safe_cells)
    
    if path_length is None:
        print("P2: No path found — maze is blocked!")
        return
    
    print(f"P2: Optimal path length: {path_length}")
    
    # Compare with simulation optimal
    if path_length == OPTIMAL_PATH_LENGTH:
        print("✓ Matches simulation optimal path length!")
    else:
        print(f"Δ Difference from simulation: {path_length - OPTIMAL_PATH_LENGTH}")
    
    # Visualize each protocol step
    visualize_protocol_steps(maze, rocks, safe_cells, path)
    
    # Show protocol summary
    print("\n=== Deadband Protocol Summary ===")
    print("P0: Identified what NOT to do (rocks)")
    print("P1: Found where we CAN be (safe channels)")
    print("P2: Optimized path within safe channels")
    print(f"Result: Path of length {path_length} vs unconstrained failure")

if __name__ == "__main__":
    main()
```

3. Path length calculation that should approximate the optimal 21 steps
4. Clear demonstration of the strict P0→P1→P2 priority order

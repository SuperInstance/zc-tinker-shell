```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 252)
Simulates P0→P1→P2 in a 20x20 maze with 64 rocks.
Matches parameters from constraint_sim_final.json.
"""

import random
import sys
from typing import List, Tuple, Set

# Simulation parameters from fleet-knowledge
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LEN = 21
TRIALS = 50

class Maze:
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_rocks()
        
    def generate_rocks(self):
        """Place ROCK_COUNT rocks randomly, avoiding start and goal."""
        candidates = [(r, c) for r in range(MAZE_SIZE) for c in range(MAZE_SIZE)]
        candidates.remove(self.start)
        candidates.remove(self.goal)
        self.rocks = set(random.sample(candidates, ROCK_COUNT))
        for r, c in self.rocks:
            self.grid[r][c] = '#'
    
    def is_rock(self, pos: Tuple[int, int]) -> bool:
        return pos in self.rocks
    
    def in_bounds(self, pos: Tuple[int, int]) -> bool:
        r, c = pos
        return 0 <= r < MAZE_SIZE and 0 <= c < MAZE_SIZE
    
    def print(self, path=None):
        """Print ASCII maze, optionally highlighting a path."""
        for r in range(MAZE_SIZE):
            row = []
            for c in range(MAZE_SIZE):
                pos = (r, c)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif path and pos in path:
                    row.append('*')
                elif self.is_rock(pos):
                    row.append('#')
                else:
                    row.append('.')
            print(' '.join(row))
        print()

def greedy_path(maze: Maze) -> List[Tuple[int, int]]:
    """P2-only: greedy optimization toward goal, ignoring rocks."""
    path = [maze.start]
    current = maze.start
    while current != maze.goal:
        r, c = current
        dr = maze.goal[0] - r
        dc = maze.goal[1] - c
        # Move toward goal (prefer diagonal if possible)
        next_pos = (r + (1 if dr > 0 else -1 if dr < 0 else 0),
                    c + (1 if dc > 0 else -1 if dc < 0 else 0))
        if not maze.in_bounds(next_pos):
            break
        path.append(next_pos)
        current = next_pos
        if len(path) > MAZE_SIZE * 2:  # prevent infinite loops
            break
    return path

def deadband_path(maze: Maze) -> List[Tuple[int, int]]:
    """P0→P1→P2: map rocks, find safe channels, then optimize."""
    # P0: Map negative space (rocks)
    rocks = maze.rocks
    
    # P1: Identify safe channels (BFS avoiding rocks)
    from collections import deque
    queue = deque([(maze.start, [maze.start])])
    visited = {maze.start}
    safe_paths = []
    
    while queue:
        pos, path = queue.popleft()
        if pos == maze.goal:
            safe_paths.append(path)
            continue
        r, c = pos
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            npos = (nr, nc)
            if maze.in_bounds(npos) and npos not in rocks and npos not in visited:
                visited.add(npos)
                queue.append((npos, path + [npos]))
    
    # P2: Optimize within safe channels (shortest safe path)
    if safe_paths:
        return min(safe_paths, key=len)
    return []

def run_trial(strategy: str) -> Tuple[bool, int]:
    """Run one maze trial with given strategy."""
    maze = Maze()
    if strategy == 'unconstrained':
        path = greedy_path(maze)
    elif strategy == 'deadband':
        path = deadband_path(maze)
    else:
        path = []
    
    success = False
    if path and path[-1] == maze.goal:
        # Check if path hits any rocks
        hit_rock = any(maze.is_rock(pos) for pos in path)
        if not hit_rock:
            success = True
    
    return success, len(path) if success else 0

def main():
    print("Deadband Protocol Maze Demo")
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}, Optimal: {OPTIMAL_PATH_LEN} steps")
    print(f"Trials: {TRIALS}")
    print()
    
    strategies = ['unconstrained', 'deadband']
    results = {s: {'success': 0, 'steps': []} for s in strategies}
    
    for trial in range(TRIALS):
        for strat in strategies:
            success, steps = run_trial(strat)
            if success:
                results[strat]['success'] += 1
                results[strat]['steps'].append(steps)
    
    # Print results
    for strat in strategies:
        succ = results[strat]['success']
        avg = sum(results[strat]['steps']) / max(1, len(results[strat]['steps']))
        print(f"{strat}: {succ}/{TRIALS} success, avg steps: {avg:.1f}")
    
    print()
    
    # Show one example maze
    print("Example maze (one trial):")
    maze = Maze()
    print("Maze layout (S=start, G=goal, #=rock):")
    maze.print()
    
    print("Greedy (P2-only) path:")
    greedy = greedy_path(maze)
    maze.print(greedy)
    hit = any(maze.is_rock(p) for p in greedy)
    print(f"Hit rock: {hit}, Reached goal: {greedy[-1]==maze.goal}")
    
    print("\nDeadband (P0→P1→P2) path:")
    deadband = deadband_path(maze)
    maze.print(deadband)
    hit = any(maze.is_rock(p) for p in deadband)
    print(f"Hit rock: {hit}, Reached goal: {deadband[-1]==maze.goal if deadband else False}")
    
    # Compare to fleet data
    print("\n--- Comparison to fleet data ---")
    print("From constraint_sim_final.json:")
    print("  unconstrained_success: 33/50")
    print("  deadband_success: 50/50")
    print("  deadband_avg: 21.0")

if __name__ == "__main__":
    main()
```

- The `deadband_path` function implements strict P0→P1→P2: first identifies rocks (negative space), then BFS for safe channels, then picks the shortest safe path.
- The `greedy_path` function implements P2-only optimization, moving directly toward the goal without rock avoidance.
- Running 50 trials shows a clear difference in success rates, though random rock placement leads to variance.
- One example maze is printed with both paths visualized in ASCII.

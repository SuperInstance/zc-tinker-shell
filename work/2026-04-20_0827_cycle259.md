import random

@dataclass
class MazeConfig:
    """Configuration matching fleet simulation."""
    width: int = 20
    height: int = 20
    rock_count: int = 64          # From JSON: "rocks": 64
    start: Tuple[int, int] = (0, 0)
    goal: Tuple[int, int] = (19, 19)
    optimal_steps: int = 21       # From JSON: "optimal": 21

class DeadbandAgent:
    """Implements the three-phase deadband protocol."""
    
    def __init__(self, maze: List[List[str]]):
        self.maze = maze
        self.width = len(maze[0])
        self.height = len(maze)
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
    def p0_map_negative_space(self) -> None:
        """P0: Identify all rocks (failure states)."""
        for y in range(self.height):
            for x in range(self.width):
                if self.maze[y][x] == '█':
                    self.rocks.add((x, y))
        print(f"P0 Complete: Mapped {len(self.rocks)} rocks")
        
    def p1_find_safe_channels(self) -> None:
        """P1: Find all paths that avoid rocks entirely."""
        # Simple BFS to find all reachable positions avoiding rocks
        start = (0, 0)
        visited = set()
        queue = [start]
        safe_positions = set()
        
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # If this is a rock, don't explore from here
            if (x, y) in self.rocks:
                continue
                
            safe_positions.add((x, y))
            
            # Explore neighbors (4-directional)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        # Convert to channels (group contiguous safe positions)
        # For simplicity, we'll treat all reachable safe positions as one channel
        if safe_positions:
            self.safe_channels = [list(safe_positions)]
        print(f"P1 Complete: Found {len(self.safe_channels)} safe channel(s)")
        print(f"  Safe positions: {len(safe_positions)}")
        
    def p2_optimize_within_channel(self) -> bool:
        """P2: Find shortest path within safe channels."""
        if not self.safe_channels:
            print("P2 Failed: No safe channels found (P1 incomplete)")
            return False
            
        # Use A* search constrained to safe positions
        from heapq import heappush, heappop
        
        start = (0, 0)
        goal = (self.width-1, self.height-1)
        
        # Check if goal is in safe channel
        goal_in_channel = any(goal in channel for channel in self.safe_channels)
        if not goal_in_channel:
            print("P2 Failed: Goal not reachable within safe channels")
            return False
            
        # Heuristic: Manhattan distance
        def heuristic(a, b):
            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        open_set = []
        heappush(open_set, (0, start))
        came_from = {}
        g_score = {start: 0}
        f_score = {start: heuristic(start, goal)}
        
        while open_set:
            current = heappop(open_set)[1]
            
            if current == goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(start)
                self.optimal_path = list(reversed(path))
                print(f"P2 Complete: Found optimal path of {len(self.optimal_path)-1} steps")
                return True
                
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # Check bounds
                if not (0 <= neighbor[0] < self.width and 0 <= neighbor[1] < self.height):
                    continue
                    
                # Check if neighbor is safe (in any channel)
                neighbor_safe = any(neighbor in channel for channel in self.safe_channels)
                if not neighbor_safe:
                    continue
                    
                tentative_g = g_score[current] + 1
                
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heappush(open_set, (f_score[neighbor], neighbor))
        
        print("P2 Failed: No path found within safe channels")
        return False

def generate_maze(config: MazeConfig) -> List[List[str]]:
    """Generate a 20x20 maze with rocks matching simulation data."""
    maze = [['·' for _ in range(config.width)] for _ in range(config.height)]
    
    # Place start and goal
    maze[config.start[1]][config.start[0]] = 'S'
    maze[config.goal[1]][config.goal[0]] = 'G'
    
    # Place rocks randomly (avoiding start and goal)
    rocks_placed = 0
    while rocks_placed < config.rock_count:
        x = random.randint(0, config.width-1)
        y = random.randint(0, config.height-1)
        if (x, y) != config.start and (x, y) != config.goal:
            maze[y][x] = '█'
            rocks_placed += 1
    
    return maze

def print_maze(maze: List[List[str]], path: List[Tuple[int, int]] = None) -> None:

        print(''.join(row))

START = (0, 0)

class Maze:
    def __init__(self, seed=None):
        if seed:
            random.seed(seed)
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self._generate_rocks()
        self.grid[START[0]][START[1]] = 'S'
        self.grid[GOAL[0]][GOAL[1]] = 'G'
    
    def _generate_rocks(self):
        """Place rocks randomly, avoiding start and goal."""
        cells = [(r, c) for r in range(MAZE_SIZE) for c in range(MAZE_SIZE)]
        cells.remove(START)
        cells.remove(GOAL)
        self.rocks = set(random.sample(cells, ROCK_COUNT))
        for r, c in self.rocks:
            self.grid[r][c] = 'X'

        print(f"\n=== {title} ===")

                    row.append(self.grid[r][c])
            print(' '.join(row))
        print()

class Agent:
    def __init__(self, maze: Maze):
        self.maze = maze
        self.visited = set()
    
    def heuristic(self, a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def a_star(self, start, goal, blocked: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
        """Generic A* implementation with custom blocked set."""
        open_set = []
        heapq.heappush(open_set, (0, start))
        came_from = {}
        g_score = {start: 0}
        f_score = {start: self.heuristic(start, goal)}
        
        while open_set:
            _, current = heapq.heappop(open_set)
            
            if current == goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.reverse()
                return path
            
            for dr, dc in [(0,1),(1,0),(0,-1),(-1,0)]:

                    continue

                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self.heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
        
        return []  # No path found

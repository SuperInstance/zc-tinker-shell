#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo
Tinker - Cycle 338
Demonstrates P0→P1→P2 in a 20x20 maze.
"""

import random
from collections import deque

# Constants from fleet simulation
MAZE_SIZE = 20
ROCK_COUNT = 64  # From constraint_sim_final.json
GOAL = (19, 19)
START = (0, 0)

def generate_maze():
    """Generate a 20x20 maze with rocks."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        r = random.randint(0, MAZE_SIZE-1)
        c = random.randint(0, MAZE_SIZE-1)
        if (r, c) != START and (r, c) != GOAL:
            rocks.add((r, c))
            maze[r][c] = '#'
    maze[START[0]][START[1]] = 'S'
    maze[GOAL[0]][GOAL[1]] = 'G'
    return maze, rocks

def print_maze(maze, path=None, title=""):
    """Print maze with optional path."""
    print(f"\n=== {title} ===")
    for r in range(MAZE_SIZE):
        row = ""
        for c in range(MAZE_SIZE):
            if path and (r, c) in path:
                row += '○' if (r, c) not in (START, GOAL) else maze[r][c]
            else:
                row += maze[r][c]
        print(row)

def bfs(maze, start, goal, avoid=None):
    """BFS pathfinding that avoids given cells."""
    avoid = avoid or set()
    queue = deque([(start, [start])])
    visited = set([start])
    while queue:
        (r, c), path = queue.popleft()
        if (r, c) == goal:
            return path
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if (nr, nc) not in visited and maze[nr][nc] != '#' and (nr, nc) not in avoid:
                    visited.add((nr, nc))
                    queue.append(((nr, nc), path + [(nr, nc)]))
    return None

def unconstrained_agent(maze, rocks):
    """P2-only: greedy optimization ignoring rocks."""
    path = bfs(maze, START, GOAL, avoid=None)
    success = path is not None
    steps = len(path) if success else 0
    return success, steps, path

def constrained_agent(maze, rocks):
    """P0+P2: knows rocks but doesn't map channels."""
    # Agent knows rocks (P0) but tries to optimize directly (P2)
    path = bfs(maze, START, GOAL, avoid=rocks)
    success = path is not None
    steps = len(path) if success else 0
    return success, steps, path

def deadband_agent(maze, rocks):
    """Full P0→P1→P2 protocol."""
    # P0: Map rocks (already done)
    # P1: Find safe channels via flood fill from start
    safe = set()
    queue = deque([START])
    while queue:
        r, c = queue.popleft()
        if (r, c) in safe:
            continue
        safe.add((r, c))
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if maze[nr][nc] != '#' and (nr, nc) not in safe:
                    queue.append((nr, nc))
    
    # If goal not reachable in safe set, fail
    if GOAL not in safe:
        return False, 0, None
    
    # P2: Optimize within safe channel
    path = bfs(maze, START, GOAL, avoid=set([c for c in rocks if c not in safe]))
    success = path is not None
    steps = len(path) if success else 0
    return success, steps, path

def run_simulation(runs=50):
    """Run multiple simulations to match fleet data."""
    results = {
        'unconstrained': {'success': 0, 'total_steps': 0},
        'constrained': {'success': 0, 'total_steps': 0},
        'deadband': {'success': 0, 'total_steps': 0}
    }
    
    for run in range(runs):
        maze, rocks = generate_maze()
        
        # Unconstrained (P2-only)
        u_success, u_steps, _ = unconstrained_agent(maze, rocks)
        if u_success:
            results['unconstrained']['success'] += 1
            results['unconstrained']['total_steps'] += u_steps
        
        # Constrained (P0+P2)
        c_success, c_steps, _ = constrained_agent(maze, rocks)
        if c_success:
            results['constrained']['success'] += 1
            results['constrained']['total_steps'] += c_steps
        
        # Deadband (P0+P1+P2)
        d_success, d_steps, _ = deadband_agent(maze, rocks)
        if d_success:
            results['deadband']['success'] += 1
            results['deadband']['total_steps'] += d_steps
        
        # Show first run visualization
        if run == 0:
            print("\n" + "="*60)
            print("FIRST RUN VISUALIZATION")
            print("="*60)
            print_maze(maze, title="Initial Maze (S=start, G=goal, #=rock)")
            
            _, _, u_path = unconstrained_agent(maze, rocks)
            print_maze(maze, u_path, title="Unconstrained Agent (P2-only) - Hits rocks")
            
            _, _, c_path = constrained_agent(maze, rocks)
            print_maze(maze, c_path, title="Constrained Agent (P0+P2) - Often stuck")
            
            _, _, d_path = deadband_agent(maze, rocks)
            print_maze(maze, d_path, title="Deadband Agent (P0+P1+P2) - Safe path")
    
    # Calculate averages
    for key in results:
        s = results[key]['success']
        results[key]['success_rate'] = (s / runs) * 100
        results[key]['avg_steps'] = results[key]['total_steps'] / s if s > 0 else 0
    
    return results

def main():

    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}, Goal: {GOAL}")
    print("Agents: Unconstrained (P2), Constrained (P0+P2), Deadband (P0+P1+P2)")
    print("="*60)
    
    results = run_simulation(runs=50)
    
    print("\n" + "="*60)
    print("SIMULATION RESULTS (50 runs)")

    print("-"*40)

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Demonstrates strict priority: map rocks, find safe channels, then optimize.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        """
        Initialize maze with rocks (obstacles).
        rock_density ~64/400 from simulation data (64 rocks in 20x20)
        """
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self.rocks: Set[Tuple[int, int]] = set()
        self._generate_rocks(rock_density)
        
        # P1: Safe channels (deadbands)
        self.safe_channels: List[List[Tuple[int, int]]] = []
        
        # P2: Optimal path
        self.optimal_path: List[Tuple[int, int]] = []
        
        # Statistics
        self.stats = {
            'rocks_count': len(self.rocks),
            'channels_found': 0,
            'path_length': 0,
            'p0_time': 0,
            'p1_time': 0,
            'p2_time': 0
        }
    
    def _generate_rocks(self, density: float):
        """Generate random obstacles (negative space)"""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        
        # Ensure start and goal are not rocks
        available = [(x, y) for x in range(self.width) for y in range(self.height)
                    if (x, y) not in [self.start, self.goal]]
        
        self.rocks = set(random.sample(available, num_rocks))
    
    def run_protocol(self):
        """Execute Deadband Protocol P0→P1→P2"""
        print("=== DEADBAND PROTOCOL DEMO ===")
        print(f"Maze: {self.width}x{self.height}, Rocks: {len(self.rocks)}")
        print()
        
        # P0: Map negative space (what NOT to do)
        print("P0: MAP NEGATIVE SPACE")
        print(f"  Identified {len(self.rocks)} rocks (failure states)")
        print("  Rocks marked with '█'")
        print()
        
        # P1: Find safe channels (where you CAN be)
        print("P1: FIND SAFE CHANNELS")
        self._find_safe_channels()
        print(f"  Found {len(self.safe_channels)} safe channels")
        if self.safe_channels:
            print(f"  Channel lengths: {[len(c) for c in self.safe_channels[:3]]}...")
        print()
        
        # P2: Optimize within safe channels
        print("P2: OPTIMIZE WITHIN CHANNELS")
        if self.safe_channels:
            self._optimize_path()
            print(f"  Optimal path length: {len(self.optimal_path)}")
            print(f"  Theoretical optimal: 21 (Manhattan distance)")
        else:
            print("  WARNING: No safe channels found! Cannot proceed to P2.")
        print()
        
        self._print_comparison()
    
    def _find_safe_channels(self):
        """Find all paths from start to goal avoiding rocks (BFS)"""
        from collections import deque
        
        # BFS to find all shortest safe paths
        queue = deque([(self.start, [self.start])])
        visited = {self.start: 1}
        paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                paths.append(path)
                continue
            
            # Explore neighbors (right, down, left, up)
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks:
                        if (nx, ny) not in visited or len(path) + 1 <= visited[(nx, ny)]:
                            visited[(nx, ny)] = len(path) + 1
                            queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.safe_channels = paths
        self.stats['channels_found'] = len(paths)
    
    def _optimize_path(self):
        """Select shortest path from safe channels"""
        if not self.safe_channels:
            return
        
        self.optimal_path = min(self.safe_channels, key=len)
        self.stats['path_length'] = len(self.optimal_path)
    
    def display_maze(self, show_path: bool = False):
        """Display ASCII maze with rocks, path, start, goal"""
        print("MAZE VISUALIZATION:")
        print("  S = Start, G = Goal, █ = Rock, · = Safe, ○ = Path")
        print()
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')
                elif show_path and pos in self.optimal_path:
                    row.append('○')
                else:
                    row.append('·')
            print('  ' + ' '.join(row))
        print()
    
    def _print_comparison(self):
        """Compare with simulation results"""
        print("=== SIMULATION VALIDATION ===")
        print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json:")
        print(f"  Optimal path length: 21")
        print(f"  Rocks in simulation: 64")
        print(f"  Unconstrained success: 33/50 (greedy P2-only)")
        print(f"  Deadband success: 50/50 (P0→P1→P2)")
        print()
        print("Our demo results:")
        print(f"  Rocks: {self.stats['rocks_count']}")
        print(f"  Safe channels found: {self.stats['channels_found']}")
        print(f"  Path length: {self.stats['path_length']}")
        
        if self.stats['channels_found'] > 0:
            print("✓ Protocol successful: Found safe channels → can optimize")
        else:
            print("✗ Protocol failed: No safe channels (environment too constrained)")
    
    def run_greedy_comparison(self):
        """Demonstrate what happens with P2-only (greedy) approach"""
        print("\n=== GREEDY P2-ONLY (FOR COMPARISON) ===")
        print("Attempting to optimize without P0/P1...")
        
        # Greedy approach: always move toward goal, ignore rocks
        path = [self.start]
        x, y = self.start
        
        while (x, y) != self.goal:
            # Move right if possible, else down
            if x < self.goal[0] and (x+1, y) not in self.rocks:
                x += 1
            elif y < self.goal[1] and (x, y+1) not in self.rocks:
                y += 1
            else:
                # Hit a rock - greedy fails
                print(f"  Greedy path stuck at ({x}, {y}) after {len(path)} steps")
                print("  ✗ P2-only fails without P0 rock mapping")
                return False
            
            path.append((x, y))
        
        print(f"  Greedy path length: {len(path)}")
        return True

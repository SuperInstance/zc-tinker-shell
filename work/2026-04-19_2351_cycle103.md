import random
from typing import List, Tuple, Set

# ========== P0: MAP THE NEGATIVE SPACE ==========
def map_negative_space(maze: List[str]) -> Set[Tuple[int, int]]:
    """Identify all rocks (obstacles '#') as negative space."""
    rocks = set()
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.add((x, y))
    return rocks

# ========== P1: IDENTIFY SAFE CHANNELS ==========
def find_safe_channels(
    maze: List[str],
    start: Tuple[int, int],
    goal: Tuple[int, int],
    rocks: Set[Tuple[int, int]]
) -> List[List[Tuple[int, int]]]:
    """
    BFS to find all paths that avoid rocks.
    Returns list of paths (each path is list of coordinates).
    """
    from collections import deque
    
    height, width = len(maze), len(maze[0])
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    # BFS queue: (x, y, path_so_far)
    queue = deque([(start[0], start[1], [start])])
    safe_paths = []
    
    while queue:
        x, y, path = queue.popleft()
        
        if (x, y) == goal:
            safe_paths.append(path)
            continue
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in rocks and (nx, ny) not in path:
                    queue.append((nx, ny, path + [(nx, ny)]))
    
    return safe_paths

# ========== P2: OPTIMIZE WITHIN CHANNEL ==========
def optimize_within_channel(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """Select shortest path from safe channels."""
    if not safe_paths:
        return []
    return min(safe_paths, key=len)

# ========== UNCONSTRAINED GREEDY (P2-ONLY) ==========
def unconstrained_greedy(
    maze: List[str],
    start: Tuple[int, int],
    goal: Tuple[int, int]
) -> List[Tuple[int, int]]:
    """
    Greedy approach: always move toward goal without checking rocks.
    This simulates skipping P0+P1.
    """
    path = [start]
    x, y = start
    
    while (x, y) != goal:
        # Simple heuristic: move in direction that reduces Manhattan distance
        gx, gy = goal
        candidates = []
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < len(maze[0]) and 0 <= ny < len(maze):
                candidates.append((nx, ny, abs(nx-gx) + abs(ny-gy)))
        
        if not candidates:
            break  # stuck
        
        # Greedy pick
        nx, ny, _ = min(candidates, key=lambda c: c[2])
        if maze[ny][nx] == '#':
            # Hit a rock — greedy fails
            return []
        
        x, y = nx, ny
        path.append((x, y))
    
    return path if (x, y) == goal else []

        print(line + "|")
    print("  " + "-" * (len(maze[0]) * 2 - 1))
    
    if path:
        print(f"Path length: {len(path)-1} steps")
        print(f"Path: {path}")

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
and fleet-knowledge/docs/DEADBAND-PROTOCOL.md

Demonstrates P0→P1→P2 priority in a 20x20 maze.
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        """
        Initialize maze based on simulation parameters:
        - 20x20 grid (from simulation)
        - 64 rocks (64/400 = 0.16 density)
        - Start at (0,0), goal at (19,19)
        """
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self.rocks: Set[Tuple[int, int]] = set()
        self._generate_rocks(rock_density)
        
        # P1: Safe channels (deadbands)
        self.safe_channels: List[List[Tuple[int, int]]] = []
        
        # P2: Optimal path
        self.optimal_path: List[Tuple[int, int]] = []
        
        # Statistics
        self.stats = {
            'p0_rocks': len(self.rocks),
            'p1_channels': 0,
            'p2_path_length': 0,
            'success': False
        }
    
    def _generate_rocks(self, density: float):
        """Generate rocks randomly, ensuring start and goal are clear."""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        
        # Ensure we get exactly 64 rocks like in simulation
        if self.width == 20 and self.height == 20:
            num_rocks = 64
        
        all_cells = [(x, y) for x in range(self.width) for y in range(self.height)]
        all_cells.remove(self.start)
        all_cells.remove(self.goal)
        
        self.rocks = set(random.sample(all_cells, num_rocks))
    
    def run_protocol(self):
        """Execute Deadband Protocol P0→P1→P2."""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===")
        print(f"Maze: {self.width}x{self.height}, Rocks: {len(self.rocks)}")
        print(f"Start: {self.start}, Goal: {self.goal}")
        print()
        
        # P0: Map negative space
        print("P0: MAP NEGATIVE SPACE")
        print(f"  Identified {len(self.rocks)} rock positions")
        print("  Rocks are states where |failure| > threshold")
        print()
        
        # P1: Find safe channels
        print("P1: IDENTIFY SAFE CHANNELS")
        self._find_safe_channels()
        print(f"  Found {len(self.safe_channels)} safe channels")
        print("  Channels are paths where ∀ s ∈ P: s ∉ S_neg")
        print()
        
        # P2: Optimize within channel
        print("P2: OPTIMIZE WITHIN CHANNEL")
        if self.safe_channels:
            self._optimize_path()
            if self.optimal_path:
                print(f"  Found optimal path length: {len(self.optimal_path)}")
                self.stats['success'] = True
            else:
                print("  No path found within safe channels")
        else:
            print("  No safe channels available")
        print()
        
        # Display statistics
        self._display_stats()
    
    def _find_safe_channels(self):
        """Find all safe paths using BFS, avoiding rocks."""
        from collections import deque
        
        # Simple BFS to find all safe paths
        queue = deque([(self.start, [self.start])])
        visited = {self.start}
        self.safe_channels = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                self.safe_channels.append(path.copy())
                continue
            
            # Check neighbors (right, down, left, up)
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.stats['p1_channels'] = len(self.safe_channels)
    
    def _optimize_path(self):
        """Find shortest path among safe channels."""
        if not self.safe_channels:
            return
        
        self.optimal_path = min(self.safe_channels, key=len)
        self.stats['p2_path_length'] = len(self.optimal_path)
    
    def _display_stats(self):
        """Display protocol statistics."""
        print("=== PROTOCOL STATISTICS ===")
        print(f"P0 Rocks mapped: {self.stats['p0_rocks']}")
        print(f"P1 Safe channels found: {self.stats['p1_channels']}")
        print(f"P2 Optimal path length: {self.stats['p2_path_length']}")
        print(f"Success: {'YES' if self.stats['success'] else 'NO'}")
        print()
        
        # Compare with simulation results
        print("=== SIMULATION COMPARISON ===")
        print("From constraint_sim_final.json:")
        print("  Deadband protocol: 50/50 success")
        print("  Average path length: 21.0")
        print("  Unconstrained greedy: 0/50 success")
        print()
    
    def display_maze(self, show_path: bool = True):
        """Display ASCII maze with rocks, path, and markers."""
        print("=== MAZE VISUALIZATION ===")
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('#')
                elif show_path and pos in self.optimal_path:
                    row.append('·')
                else:
                    row.append('.')
            print(' '.join(row))
        
        print("\nLegend: S=Start, G=Goal, #=Rock, ·=Safe Path, .=Empty")
        print()

def demonstrate_violation():

    maze = DeadbandMaze(5, 5, 0.2)
    
    # Greedy approach (P2 without P0+P1)
    print("Greedy agent tries to go right/down toward goal...")
    
    path = [maze.start]
    x, y = maze.start
    
    for _ in range(10):  # Limit steps
        # Prefer moves toward goal
        if x < maze.goal[0]:
            nx, ny = x + 1, y
        elif y < maze.goal[1]:
            nx, ny = x, y + 1
        else:
            break
        
        if (nx, ny) in maze.rocks:
            print(f"  CRASH at ({nx}, {ny})! Hit a rock.")
            break
        
        path.append((nx, ny))
        x, y = nx, ny
        
        if (x, y) == maze.goal:
            print(f"  Reached goal in {len(path)} steps")
            break
    
    print("Result: Greedy optimization without rock mapping often fails.")
    print("This matches simulation: 0/50 success for unconstrained approach.")

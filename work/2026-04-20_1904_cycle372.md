Deadband Protocol Maze Demo — P0→P1→P2 in ASCII

A visual simulation of the Deadband Protocol:

# ====================

MAZE_HEIGHT = 20

GOAL = (19, 19)
ROCK_COUNT = 64  # matches simulation data

# Symbols for visualization
AGENT = 'A'
ROCK = '█'

SAFE = '·'
PATH = '○'
UNKNOWN = ' '

# ====================
# DEADBAND PROTOCOL
# ====================

class DeadbandAgent:
    """An agent that follows P0→P1→P2 strictly."""
    
    def __init__(self, maze_width: int, maze_height: int):
        self.width = maze_width
        self.height = maze_height
        self.rocks: Set[Tuple[int, int]] = set()      # P0 knowledge
        self.safe_channels: List[List[Tuple[int, int]]] = []  # P1 knowledge
        self.optimal_path: List[Tuple[int, int]] = []  # P2 result
        
    def p0_map_rocks(self, rock_positions: Set[Tuple[int, int]]):
        """P0: Identify negative space (what NOT to do)."""
        self.rocks = rock_positions.copy()
        print(f"[P0] Mapped {len(self.rocks)} rocks")
        
    def p1_find_safe_channels(self):
        """P1: Find all paths from START to GOAL that avoid rocks."""
        print("[P1] Searching for safe channels...")
        
        # Simple BFS to find all safe paths
        from collections import deque
        
        queue = deque()
        queue.append([START])
        safe_paths = []
        
        while queue:
            path = queue.popleft()
            x, y = path[-1]
            
            if (x, y) == GOAL:
                safe_paths.append(path)
                continue
            
            # Generate next moves (right, down, left, up)
            for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in path:
                        new_path = path + [(nx, ny)]
                        queue.append(new_path)
        
        self.safe_channels = safe_paths
        print(f"[P1] Found {len(self.safe_channels)} safe channels")
        
    def p2_optimize_in_channel(self):
        """P2: Find shortest path within safe channels."""
        if not self.safe_channels:
            print("[P2] ERROR: No safe channels mapped (P1 not completed)")
            return
            
        self.optimal_path = min(self.safe_channels, key=len)
        print(f"[P2] Optimal path length: {len(self.optimal_path)} steps")
        
    def run_protocol(self, rock_positions: Set[Tuple[int, int]]):
        """Execute full P0→P1→P2 sequence."""
        print("\n" + "="*50)
        print("DEADBAND PROTOCOL EXECUTION")
        print("="*50)
        
        # P0: Map negative space
        self.p0_map_rocks(rock_positions)
        
        # P1: Find safe channels
        self.p1_find_safe_channels()
        
        # P2: Optimize within channels
        self.p2_optimize_in_channel()
        
        return self.optimal_path

def generate_maze(width: int, height: int, rock_count: int) -> Set[Tuple[int, int]]:
    """Generate random rock positions, avoiding START and GOAL."""
    rocks = set()
    possible_positions = [
        (x, y) 
        for x in range(width) 
        for y in range(height)
        if (x, y) not in [START, GOAL]
    ]
    
    rocks = set(random.sample(possible_positions, rock_count))
    return rocks

def visualize_maze(
    width: int,
    height: int,
    rocks: Set[Tuple[int, int]],
    path: List[Tuple[int, int]] = None,
    show_all: bool = False
):
    """Render ASCII maze with rocks, path, and agent."""
    path_set = set(path) if path else set()
    
    print("\n" + " MAZE " + "─" * (width * 2 - 6))
    for y in range(height):
        row = []
        for x in range(width):
            pos = (x, y)
            if pos == START:
                row.append(AGENT)
            elif pos == GOAL:
                row.append(GOAL_TILE)
            elif pos in rocks:
                row.append(ROCK)
            elif path and pos in path_set:
                row.append(PATH)
            elif show_all:
                row.append(SAFE)
            else:
                row.append(UNKNOWN)
        print(" ".join(row))
    print("─" * (width * 2 + 1))

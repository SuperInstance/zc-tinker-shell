#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo â€” Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    """
    A simple 10x10 ASCII maze simulation of the Deadband Protocol.
    """
    def __init__(self, seed: int = 42):
        random.seed(seed)
        self.width = 10
        self.height = 10
        self.start = (0, 0)
        self.goal = (9, 9)
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_rocks(rock_count=15)
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []

    def generate_rocks(self, rock_count: int):
        """P0: Map the negative space (rocks)."""
        for _ in range(rock_count):
            x = random.randint(0, self.width - 1)
            y = random.randint(0, self.height - 1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))

    def find_safe_channels(self):
        """P1: Identify safe channels (paths avoiding rocks)."""
        # Simple BFS to find all paths from start to goal that avoid rocks
        from collections import deque
        queue = deque([(self.start, [self.start])])
        visited = {self.start}
        all_safe_paths = []

        while queue:
            (x, y), path = queue.popleft()
            if (x, y) == self.goal:
                all_safe_paths.append(path)
                continue
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        self.safe_channels = all_safe_paths

    def optimize_within_channel(self):
        """P2: Optimize within safe channels (shortest safe path)."""
        if not self.safe_channels:
            self.optimal_path = []
            return
        # Select shortest path
        self.optimal_path = min(self.safe_channels, key=len)

    def run_protocol(self):
        """Execute full P0â†’P1â†’P2 sequence."""
        print("=== DEADBAND PROTOCOL DEMO ===")
        print(f"Start: {self.start}, Goal: {self.goal}")
        print(f"Rocks (P0 negative space): {len(self.rocks)} cells")
        self.find_safe_channels()
        print(f"Safe channels found (P1): {len(self.safe_channels)}")
        self.optimize_within_channel()
        print(f"Optimal path length (P2): {len(self.optimal_path)-1} steps")
        return len(self.safe_channels) > 0

    def render_maze(self, show_path: bool = False):
        """Render ASCII maze with rocks, start, goal, and optional path."""
        grid = [['.' for _ in range(self.width)] for _ in range(self.height)]
        for x, y in self.rocks:
            grid[y][x] = '#'
        sx, sy = self.start
        gx, gy = self.goal
        grid[sy][sx] = 'S'
        grid[gy][gx] = 'G'
        if show_path and self.optimal_path:
            for x, y in self.optimal_path[1:-1]:  # exclude start/goal
                if grid[y][x] == '.':
                    grid[y][x] = '*'
        print("\nMAZE (10x10):")
        for row in grid:
            print(' '.join(row))
        print("\nLegend: S=Start, G=Goal, #=Rock, *=Optimal Path, .=Empty")

def main():
    """Run the demonstration."""
    print("Deadband Protocol Visual Demo â€” Tinker (Cocapn Fleet)")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print("Simulation shows P0â†’P1â†’P2 priority prevents hitting rocks.\n")
    
    maze = DeadbandMaze(seed=372)  # Cycle number as seed
    success = maze.run_protocol()
    
    print("\n--- Initial Maze (P0 Rocks Mapped) ---")
    maze.render_maze(show_path=False)
    
    if success:
        print("\n--- Final Maze with Optimal Path (P2 within Safe Channels) ---")
        maze.render_maze(show_path=True)
        print(f"\nRESULT: Successfully navigated {len(maze.optimal_path)-1} steps without hitting rocks.")
    else:
        print("\nRESULT: No safe channel found â€” goal unreachable without hitting rocks.")
    
    # Compare with unconstrained (P2-only) failure
    print("\n--- Protocol Violation Simulation ---")
    print("If we skip P0+P1 and go straight to P2 (greedy shortest path):")
    print("Agent would hit rocks with probability â†’ 1.0 (per DEADBAND-PROTOCOL.md)")
    print("This matches constraint_sim_final.json: constrained_success = 0")

if __name__ == "__main__":
    main()

## Deadband Protocol ASCII Maze Demo

**File:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    """
    Simulates the Deadband Protocol in a 10x10 ASCII maze.
    - 'S' = Start
    - 'G' = Goal
    - '#' = Rock (failure state)
    - '.' = Open cell
    - '*' = Safe channel (deadband)
    - 'o' = Optimal path (within channel)
    """
    def __init__(self, seed=42):
        random.seed(seed)
        self.width = 10
        self.height = 10
        self.start = (0, 0)
        self.goal = (9, 9)
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_rocks(rock_count=15)
        self.grid = self.init_grid()
        
    def generate_rocks(self, rock_count: int):
        """P0: Map negative space — generate rock obstacles."""
        for _ in range(rock_count):
            x = random.randint(0, self.width-1)
            y = random.randint(0, self.height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def init_grid(self) -> List[List[str]]:
        """Initialize grid with start, goal, rocks."""
        grid = [['.' for _ in range(self.width)] for _ in range(self.height)]
        grid[self.start[1]][self.start[0]] = 'S'
        grid[self.goal[1]][self.goal[0]] = 'G'
        for (x, y) in self.rocks:
            grid[y][x] = '#'
        return grid
    
    def find_safe_channels(self) -> Set[Tuple[int, int]]:
        """
        P1: Identify safe channels (deadbands) using BFS.
        Returns all cells reachable from start without hitting rocks.
        """
        visited = set()
        queue = [self.start]
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        queue.append((nx, ny))
        return visited
    
    def optimize_in_channel(self, safe_cells: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
        """
        P2: Optimize within safe channel — find shortest path from start to goal
        using only safe cells (A* with Manhattan distance).
        """
        from heapq import heappush, heappop
        
        def heuristic(a, b):
            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        open_set = []
        heappush(open_set, (0, self.start))
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: heuristic(self.start, self.goal)}
        
        while open_set:
            _, current = heappop(open_set)
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.reverse()
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                if neighbor not in safe_cells:
                    continue
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, self.goal)
                    heappush(open_set, (f_score[neighbor], neighbor))
        return []  # No path found within safe channel
    
    def run_protocol(self) -> dict:
        """Execute full P0→P1→P2 Deadband Protocol."""
        # P0 already done during init (rocks mapped)
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Rocks identified: {len(self.rocks)}")
        self.print_grid()
        
        # P1: Find safe channels
        print("\n=== P1: IDENTIFY SAFE CHANNELS ===")
        safe_cells = self.find_safe_channels()
        print(f"Safe cells reachable from start: {len(safe_cells)}")
        channel_grid = self.grid_with_channels(safe_cells)
        self.print_grid(channel_grid)
        
        # P2: Optimize within channel
        print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
        optimal_path = self.optimize_in_channel(safe_cells)
        if optimal_path:
            print(f"Optimal path length: {len(optimal_path)} steps")
            final_grid = self.grid_with_path(channel_grid, optimal_path)
            self.print_grid(final_grid)
            success = True
        else:
            print("No safe path to goal found within channels.")
            final_grid = channel_grid
            success = False
        
        return {
            "rocks": len(self.rocks),
            "safe_cells": len(safe_cells),
            "path_length": len(optimal_path) if optimal_path else None,
            "success": success,
            "final_grid": final_grid
        }
    
    def grid_with_channels(self, safe_cells: Set[Tuple[int, int]]) -> List[List[str]]:
        """Return grid with safe channels marked '*'."""
        grid = [row[:] for row in self.grid]
        for (x, y) in safe_cells:
            if grid[y][x] == '.':
                grid[y][x] = '*'
        return grid
    
    def grid_with_path(self, base_grid: List[List[str]], path: List[Tuple[int, int]]) -> List[List[str]]:
        """Return grid with optimal path marked 'o'."""
        grid = [row[:] for row in base_grid]
        for (x, y) in path:
            if grid[y][x] == '*':
                grid[y][x] = 'o'
        return grid
    
    def print_grid(self, grid=None):
        """Print ASCII grid."""
        if grid is None:
            grid = self.grid
        for row in grid:
            print(' '.join(row))
        print()

def main():
    """Run the Deadband Protocol demonstration."""
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print("Optimal: 21 steps, Deadband success: 50/50\n")
    
    maze = DeadbandMaze(seed=42)
    results = maze.run_protocol()
    
    print("\n=== RESULTS ===")
    print(f"Rocks (P0 negatives): {results['rocks']}")
    print(f"Safe channel cells (P1 deadbands): {results['safe_cells']}")
    print(f"Optimal path length (P2): {results['path_length']}")
    print(f"Success: {results['success']}")
    
    # Protocol validation
    if results['success']:
        print("\n✓ Protocol followed: P0→P1→P2 → SUCCESS")
    else:
        print("\n✗ Protocol followed but no safe path exists (environment constraints)")

if __name__ == "__main__":
    main()
```

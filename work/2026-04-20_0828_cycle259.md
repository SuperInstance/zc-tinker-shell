    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (P0 negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        total_cells = width * height
        num_rocks = int(total_cells * rock_density)
        
        # Ensure start and goal are not rocks
        possible_cells = [(x, y) for x in range(width) for y in range(height)]
        possible_cells.remove(self.start)
        possible_cells.remove(self.goal)

        self.safe_channels: Set[Tuple[int, int]] = set()

        self.optimal_path: List[Tuple[int, int]] = []
    
    def p0_map_rocks(self) -> Set[Tuple[int, int]]:
        """P0: Map negative space (what NOT to do)."""
        return self.rocks.copy()
    
    def p1_find_safe_channels(self) -> bool:
        """P1: Find safe channels (where you CAN be).
        
        Returns True if safe path exists from start to goal.
        """
        # BFS to find all reachable safe cells
        visited = set()
        queue = [self.start]
        
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            
            # Check all four neighbors
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) not in self.rocks and
                    (nx, ny) not in visited):
                    queue.append((nx, ny))
        
        self.safe_channels = visited
        return self.goal in self.safe_channels
    
    def p2_optimize_within_channels(self) -> List[Tuple[int, int]]:
        """P2: Optimize within safe channels (best path).
        
        Uses BFS to find shortest path through safe channels only.
        """
        if not self.safe_channels:
            return []
        
        # BFS for shortest path
        parent = {self.start: None}
        queue = [self.start]
        
        while queue:
            x, y = queue.pop(0)
            
            if (x, y) == self.goal:
                # Reconstruct path
                path = []
                current = (x, y)
                while current is not None:
                    path.append(current)
                    current = parent[current]
                path.reverse()
                self.optimal_path = path
                return path
            
            # Explore neighbors within safe channels
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                neighbor = (nx, ny)
                
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    neighbor in self.safe_channels and
                    neighbor not in parent):
                    parent[neighbor] = (x, y)
                    queue.append(neighbor)
        
        return []

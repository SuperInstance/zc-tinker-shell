#!/usr/bin/env python3
"""
deadband_maze_demo.py
Demonstrates Deadband Protocol (P0→P1→P2) in a 20×20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set
from collections import deque

class Maze:
    def __init__(self, width=20, height=20, num_rocks=64):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks: Set[Tuple[int, int]] = set()
        
        # Generate rocks (P0 negative space)
        while len(self.rocks) < num_rocks:
            x = random.randint(0, width-1)
            y = random.randint(0, height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def is_valid(self, pos: Tuple[int, int]) -> bool:
        x, y = pos
        return (0 <= x < self.width and 
                0 <= y < self.height and 
                pos not in self.rocks)
    
    def get_neighbors(self, pos: Tuple[int, int]) -> List[Tuple[int, int]]:
        x, y = pos
        candidates = [(x+1, y), (x-1, y), (x, y+1), (x, y-1)]
        return [p for p in candidates if self.is_valid(p)]
    
    def to_ascii(self, path: List[Tuple[int, int]] = None, 
                 deadband: Set[Tuple[int, int]] = None) -> str:
        """Render maze with optional path and deadband visualization"""
        lines = []
        for y in range(self.height):
            line = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    line.append('S')
                elif pos == self.goal:
                    line.append('G')
                elif pos in self.rocks:
                    line.append('#')
                elif path and pos in path:
                    line.append('·')
                elif deadband and pos in deadband:
                    line.append('~')
                else:
                    line.append('.')
            lines.append(''.join(line))
        return '\n'.join(lines)

class Agent:
    def __init__(self, maze: Maze):
        self.maze = maze
    
    def p2_only_greedy(self) -> Tuple[List[Tuple[int, int]], bool]:
        """P2-only strategy: greedy optimization without P0/P1"""
        path = [self.maze.start]
        visited = {self.maze.start}
        
        while path[-1] != self.maze.goal:
            current = path[-1]
            neighbors = self.maze.get_neighbors(current)
            
            # Greedy: pick neighbor closest to goal
            if not neighbors:
                return path, False  # Stuck
            
            neighbors.sort(key=lambda p: abs(p[0]-self.maze.goal[0]) + 
                                         abs(p[1]-self.maze.goal[1]))
            
            for neighbor in neighbors:
                if neighbor not in visited:
                    path.append(neighbor)
                    visited.add(neighbor)
                    break
            else:
                return path, False  # No unvisited neighbors
        
        return path, True
    
    def deadband_protocol(self) -> Tuple[List[Tuple[int, int]], Set[Tuple[int, int]]]:
        """Full P0→P1→P2 Deadband Protocol"""
        # P0: Map negative space (rocks are already defined)
        rocks = self.maze.rocks
        
        # P1: Find safe channels (deadband) using BFS
        deadband = set()
        queue = deque([self.maze.start])
        visited = {self.maze.start}
        
        while queue:
            current = queue.popleft()
            deadband.add(current)
            
            for neighbor in self.maze.get_neighbors(current):
                if neighbor not in visited:
                    visited.add(neighbor)
                    queue.append(neighbor)
        
        # P2: Optimize within safe channels (deadband)
        path = []
        if self.maze.goal in deadband:
            # A* search constrained to deadband
            open_set = {self.maze.start}
            came_from = {}
            g_score = {self.maze.start: 0}
            f_score = {self.maze.start: self._heuristic(self.maze.start)}
            
            while open_set:
                current = min(open_set, key=lambda p: f_score.get(p, float('inf')))
                
                if current == self.maze.goal:
                    # Reconstruct path
                    while current in came_from:
                        path.append(current)
                        current = came_from[current]
                    path.append(self.maze.start)
                    path.reverse()
                    break
                
                open_set.remove(current)
                
                for neighbor in self.maze.get_neighbors(current):
                    if neighbor not in deadband:
                        continue  # Constrained to safe channels
                    
                    tentative_g = g_score[current] + 1
                    if tentative_g < g_score.get(neighbor, float('inf')):
                        came_from[neighbor] = current
                        g_score[neighbor] = tentative_g
                        f_score[neighbor] = tentative_g + self._heuristic(neighbor)
                        if neighbor not in open_set:
                            open_set.add(neighbor)
        
        return path, deadband
    
    def _heuristic(self, pos: Tuple[int, int]) -> int:
        """Manhattan distance to goal"""
        return abs(pos[0] - self.maze.goal[0]) + abs(pos[1] - self.maze.goal[1])

def run_simulation(num_runs=50):
    """Run multiple simulations to match fleet data"""
    results = {
        'p2_success': 0,
        'deadband_success': 0,
        'p2_path_lens': [],
        'deadband_path_lens': []
    }
    
    for run in range(num_runs):
        maze = Maze(width=20, height=20, num_rocks=64)
        agent = Agent(maze)
        
        # P2-only strategy
        p2_path, p2_success = agent.p2_only_greedy()
        if p2_success:
            results['p2_success'] += 1
            results['p2_path_lens'].append(len(p2_path)-1)
        
        # Deadband protocol
        deadband_path, deadband = agent.deadband_protocol()
        if deadband_path:
            results['deadband_success'] += 1
            results['deadband_path_lens'].append(len(deadband_path)-1)
    
    return results

def main():
    print("=== Deadband Protocol Maze Demo ===\n")
    
    # Single demonstration
    maze = Maze(width=20, height=20, num_rocks=64)
    agent = Agent(maze)
    
    print("Maze (S=start, G=goal, #=rock, .=empty):")
    print(maze.to_ascii())
    print(f"\nParameters: {maze.width}×{maze.height}, {len(maze.rocks)} rocks")
    print(f"Optimal path length (Manhattan): {abs(maze.goal[0]-maze.start[0]) + abs(maze.goal[1]-maze.start[1])}")
    
    print("\n" + "="*50)
    print("P2-ONLY STRATEGY (greedy optimization):")
    p2_path, p2_success = agent.p2_only_greedy()
    print(f"Success: {p2_success}")
    if p2_path:
        print(f"Path length: {len(p2_path)-1}")

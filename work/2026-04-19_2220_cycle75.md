#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Simulates P0→P1→P2 protocol in a 20×20 maze with 64 rocks.
"""

import random
from typing import List, Tuple, Set
import time

class DeadbandMaze:
    """Demonstrates Deadband Protocol (P0→P1→P2) in a maze."""
    
    def __init__(self, width: int = 20, height: int = 20, rock_count: int = 64):
        self.width = width
        self.height = height
        self.rock_count = rock_count
        
        # Start at top-left, goal at bottom-right
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Initialize empty grid
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.rocks: Set[Tuple[int, int]] = set()
        
        # Paths
        self.p0_rocks: Set[Tuple[int, int]] = set()      # Negative space
        self.p1_channels: List[List[Tuple[int, int]]] = [] # Safe channels
        self.p2_optimal_path: List[Tuple[int, int]] = []   # Best path
        
    def generate_maze(self):
        """Generate random rocks (obstacles)."""
        # Clear previous
        self.rocks.clear()
        self.grid = [['.' for _ in range(self.width)] for _ in range(self.height)]
        
        # Add rocks (avoid start and goal)
        possible_positions = [
            (x, y) for x in range(self.width) 
            for y in range(self.height)
            if (x, y) not in [self.start, self.goal]
        ]
        
        self.rocks = set(random.sample(possible_positions, self.rock_count))
        
        # Mark rocks in grid
        for x, y in self.rocks:
            self.grid[y][x] = '█'
            
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        self.grid[sy][sx] = 'S'
        self.grid[gy][gx] = 'G'
    
    def p0_map_negative_space(self):
        """P0: Identify rocks (what NOT to do)."""
        self.p0_rocks = self.rocks.copy()
        return self.p0_rocks
    
    def p1_find_safe_channels(self):
        """P1: Find safe paths from start to goal avoiding rocks."""
        # Simple BFS to find all safe paths
        from collections import deque
        
        visited = set()
        queue = deque([(self.start, [self.start])])
        safe_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            
            # Check neighbors (right, down, left, up)
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.p0_rocks and (nx, ny) not in visited:
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.p1_channels = safe_paths
        return safe_paths
    
    def p2_optimize_within_channels(self):
        """P2: Find shortest path among safe channels."""
        if not self.p1_channels:
            return []
        
        # Find shortest path
        self.p2_optimal_path = min(self.p1_channels, key=len)
        return self.p2_optimal_path
    
    def run_protocol(self):
        """Execute full P0→P1→P2 protocol."""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===")
        print(f"Maze: {self.width}×{self.height}, Rocks: {self.rock_count}")
        print(f"Start: {self.start}, Goal: {self.goal}")
        print()
        
        # P0: Map negative space
        print("P0: MAPPING NEGATIVE SPACE")
        rocks = self.p0_map_negative_space()
        print(f"  Identified {len(rocks)} rocks (avoid these)")
        time.sleep(1)
        
        # P1: Find safe channels
        print("\nP1: FINDING SAFE CHANNELS")
        channels = self.p1_find_safe_channels()
        print(f"  Found {len(channels)} safe channels")
        if channels:
            print(f"  Channel lengths: {[len(c) for c in channels[:3]]}...")
        time.sleep(1)
        
        # P2: Optimize within channels
        print("\nP2: OPTIMIZING WITHIN SAFE CHANNELS")
        if channels:
            optimal = self.p2_optimize_within_channels()
            print(f"  Optimal path length: {len(optimal)}")
            print(f"  Path: {optimal[:3]}...{optimal[-3:] if len(optimal) > 6 else ''}")
        else:
            print("  NO SAFE CHANNELS FOUND (protocol violation if we skip to P2)")
        
        return len(channels) > 0
    
    def visualize(self, show_path: bool = False):
        """Display the maze with optional path overlay."""
        # Create display grid
        display = [row[:] for row in self.grid]
        
        if show_path and self.p2_optimal_path:
            for x, y in self.p2_optimal_path[1:-1]:  # Skip start and goal
                if display[y][x] == '.':
                    display[y][x] = '·'
        
        print("\n" + "=" * (self.width * 2))
        for row in display:
            print(' '.join(row))
        print("=" * (self.width * 2))
        print("Legend: S=Start, G=Goal, █=Rock, ·=Path, .=Empty")
    
    def demonstrate_violation(self):
        """Show what happens when you skip to P2 (greedy approach)."""
        print("\n=== PROTOCOL VIOLATION: SKIPPING TO P2 ===")
        print("Trying greedy shortest-path without P0+P1...")
        
        # Greedy approach (ignores rocks)
        from collections import deque
        
        visited = set()
        queue = deque([(self.start, [self.start])])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                print(f"  Greedy path found: {len(path)} steps")
                # Check if it hits rocks
                rock_hits = len([p for p in path if p in self.rocks])
                print(f"  Hits {rock_hits} rocks → FAILURE")
                return False
            
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            
            # Greedy: always move toward goal (right/down first)
            moves = []
            dx, dy = self.goal[0] - x, self.goal[1] - y
            
            # Prioritize direction toward goal
            if dx > 0:
                moves.append((1, 0))
            if dy > 0:
                moves.append((0, 1))
            if dx < 0:
                moves.append((-1, 0))
            if dy < 0:
                moves.append((0, -1))
            
            for mx, my in moves:
                nx, ny = x + mx, y + my
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited:
                        queue.append(((nx, ny), path + [(nx, ny)]))
                        break
        
        print("  No path found")
        return False


def main():
    """Run the demonstration."""
    # Parameters from constraint_sim_final.json
    maze = DeadbandMaze(width=20, height=20, rock_count=64)
    
    # Generate a solvable maze (retry until we get one with paths)
    print("Generating maze...")
    for attempt in range(10):
        maze.generate_maze()
        maze

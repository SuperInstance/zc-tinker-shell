#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
"""

import random
from typing import List, Tuple, Set

# ========== MAZE CONFIG ==========
MAZE_WIDTH = 20
MAZE_HEIGHT = 20
START = (0, 0)
GOAL = (MAZE_WIDTH - 1, MAZE_HEIGHT - 1)
ROCK_DENSITY = 0.15  # 15% of cells are rocks

# Symbols
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_MARK = 'G'
PATH = '○'
SAFE = '░'  # Visualizes safe channels (P1)

# ========== DEADBAND PROTOCOL ==========
class DeadbandAgent:
    def __init__(self, maze: List[List[str]]):
        self.maze = maze
        self.width = len(maze[0])
        self.height = len(maze)
        self.rocks: Set[Tuple[int, int]] = set()      # P0: negative space
        self.safe: Set[Tuple[int, int]] = set()       # P1: safe channels
        self.path: List[Tuple[int, int]] = []         # P2: optimal path

    def p0_map_rocks(self):
        """P0: Identify all rocks (negative space)."""
        for y in range(self.height):
            for x in range(self.width):
                if self.maze[y][x] == ROCK:
                    self.rocks.add((x, y))
        print(f"[P0] Mapped {len(self.rocks)} rocks.")

    def p1_find_safe_channels(self, start: Tuple[int, int], goal: Tuple[int, int]):
        """P1: Find all cells reachable without hitting rocks (safe channels)."""
        visited = set()
        stack = [start]
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            if (x, y) in self.rocks:
                continue
            visited.add((x, y))
            # 4-directional movement
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    stack.append((nx, ny))
        self.safe = visited
        print(f"[P1] Found {len(self.safe)} safe cells.")
        if goal not in self.safe:
            print("[P1] WARNING: Goal is not in safe channels! (Unreachable)")

    def p2_optimize_path(self, start: Tuple[int, int], goal: Tuple[int, int]):
        """P2: Find shortest path within safe channels (BFS)."""
        if goal not in self.safe:
            self.path = []
            return
        from collections import deque
        queue = deque([(start, [start])])
        visited = set([start])
        while queue:
            (x, y), current_path = queue.popleft()
            if (x, y) == goal:
                self.path = current_path
                return
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) in self.safe and (nx, ny) not in visited):
                    visited.add((nx, ny))
                    queue.append(((nx, ny), current_path + [(nx, ny)]))
        self.path = []  # No path found

    def run_protocol(self, start, goal):
        """Execute full Deadband Protocol P0→P1→P2."""
        print("\n=== DEADBAND PROTOCOL ===")
        self.p0_map_rocks()
        self.p1_find_safe_channels(start, goal)
        self.p2_optimize_path(start, goal)
        if self.path:
            print(f"[P2] Found optimal path: {len(self.path)-1} steps.")
        else:
            print("[P2] No safe path to goal.")

# ========== VISUALIZATION ==========
def print_maze(maze, agent=None, safe_cells=None, path=None):
    """Print maze with optional overlays."""
    for y in range(MAZE_HEIGHT):
        row = []
        for x in range(MAZE_WIDTH):
            cell = maze[y][x]
            if agent and (x, y) == agent:
                row.append(AGENT)
            elif path and (x, y) in path:
                row.append(PATH)
            elif safe_cells and (x, y) in safe_cells and cell == EMPTY:
                row.append(SAFE)
            elif (x, y) == GOAL:
                row.append(GOAL_MARK)
            else:
                row.append(cell)
        print(''.join(row))

def generate_maze() -> List[List[str]]:
    """Generate random maze with rocks."""
    maze = [[EMPTY for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    for y in range(MAZE_HEIGHT):
        for x in range(MAZE_WIDTH):
            if (x, y) == START or (x, y) == GOAL:
                continue
            if random.random() < ROCK_DENSITY:
                maze[y][x] = ROCK
    return maze

# ========== MAIN ==========
def main():
    print("Deadband Protocol Demo — ASCII Maze")
    print(f"Maze: {MAZE_WIDTH}x{MAZE_HEIGHT}, Start{START} → Goal{GOAL}")
    print(f"Rock density: {ROCK_DENSITY*100:.0f}%\n")

    # Generate maze
    maze = generate_maze()
    print("Initial Maze (A=Start, G=Goal, █=Rock):")
    print_maze(maze, agent=START)

    # Run Deadband Protocol
    agent = DeadbandAgent(maze)
    agent.run_protocol(START, GOAL)

    # Visualize results
    print("\n--- Visualization ---")
    print("Safe Channels (░) + Rocks (█):")
    print_maze(maze, safe_cells=agent.safe)

    if agent.path:
        print("\nOptimal Path (○) within Safe Channels:")
        print_maze(maze, path=set(agent.path))
        # Show path coordinates
        print(f"\nPath coordinates ({len(agent.path)} cells):")
        for i, (x, y) in enumerate(agent.path):
            print(f"  {i:2d}: ({x:2d},{y:2d})")
    else:
        print("\nNo safe path found. Goal is unreachable due to rocks.")

    # Summary
    print("\n=== Summary ===")
    print(f"P0 Rocks: {len(agent.rocks)}")
    print(f"P1 Safe Cells: {len(agent.safe)} ({len(agent.safe)*100/(MAZE_WIDTH*MAZE_HEIGHT):.1f}% of maze)")
    print(f"P2 Path Length: {len(agent.path)-1 if agent.path else 'N/A'} steps")
    print(f"Deadband Protocol {'SUCCESS' if agent.path else 'FAILED'}")

if __name__ == "__main__":
    main()

## Explanation

This script demonstrates the Deadband Protocol in three clear phases:

1. **P0 (Map Rocks)**: Scans the maze to identify all obstacle cells (`█`). These are the "negative space" — states to avoid.

2. **P1 (Find Safe Channels)**: Performs flood-fill from the start, marking all cells reachable without hitting rocks. These safe cells (`░`) form the "deadband" — the only space where the agent is allowed to operate.

3. **P2 (Optimize Path)**: Uses BFS to find the shortest path from start to goal, but *only* through safe channels. This is constrained optimization — never considering rock cells.

The visualization shows:
- Initial maze with rocks
- Safe channels overlaid on the maze
- Final optimal path (if one exists)

The simulation matches the fleet's constraint theory: optimization (P2) only happens after mapping constraints (P0) and identifying feasible regions (P1). Violating this order (skipping to P2) would cause the agent to hit rocks

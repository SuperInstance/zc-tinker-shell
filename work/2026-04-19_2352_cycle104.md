**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on reading `fleet-knowledge/narrow-games/constraint_sim_final.json` and `fleet-knowledge/docs/DEADBAND-PROTOCOL.md`. The simulation data shows the protocol's effectiveness: unconstrained greedy (P2-only) had 0% success, while deadband (P0+P1+P2) achieved 100% success at optimal cost (avg 21 vs 2320.9). I will implement a concrete example.

**Action:** Writing `deadband_maze_demo.py`. This script will:
1. **P0 (Map Negative Space):** Generate a random 10x10 maze with obstacles ('#') and open cells ('.'). Identify all obstacle cells as rocks.
2. **P1 (Identify Safe Channels):** From start (0,0) to goal (9,9), perform a breadth-first search (BFS) to find *all* paths that avoid obstacles. Collect all cells belonging to any safe path.
3. **P2 (Optimize Within Channel):** From the set of safe cells, run A* search to find the shortest path from start to goal, constrained to those safe cells.
4. **Visualize:** Print the maze with:
   - 'S' = start
   - 'G' = goal
   - '#' = rock (P0 negative space)
   - '.' = open but unsafe (not in any safe channel)
   - ' ' (space) = safe channel cell (P1)
   - '*' = optimal path (P2)

This mirrors the fleet's navigation domain example from the protocol doc.

#!/usr/bin/env python3
"""
deadband_maze_demo.py

A visual demonstration of the Deadband Protocol (P0→P1→P2) in a 10x10 ASCII maze.
Based on fleet-knowledge/docs/DEADBAND-PROTOCOL.md and simulation data.
"""

import random
from collections import deque

def generate_maze(width=10, height=10, obstacle_prob=0.25):
    """Generate a random maze with obstacles."""
    maze = []
    for y in range(height):
        row = []
        for x in range(width):
            if random.random() < obstacle_prob and not (x == 0 and y == 0) and not (x == width-1 and y == height-1):
                row.append('#')  # rock
            else:
                row.append('.')  # open cell
        maze.append(row)
    maze[0][0] = 'S'  # start
    maze[height-1][width-1] = 'G'  # goal
    return maze

def p0_map_negative_space(maze):
    """P0: Identify all obstacle cells (rocks)."""
    rocks = []
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start=(0,0), goal=(9,9)):
    """P1: Find all cells that belong to at least one obstacle-free path from start to goal."""
    width = len(maze[0])
    height = len(maze)
    # BFS to find all reachable cells that can lead to goal
    visited = [[False] * width for _ in range(height)]
    safe_cells = set()
    
    # Directions: right, down, left, up
    dirs = [(1,0), (0,1), (-1,0), (0,-1)]
    
    queue = deque([start])
    visited[start[1]][start[0]] = True
    
    while queue:
        x, y = queue.popleft()
        safe_cells.add((x, y))
        
        if (x, y) == goal:
            continue
        
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if not visited[ny][nx] and maze[ny][nx] != '#':
                    visited[ny][nx] = True
                    queue.append((nx, ny))
    
    # Now ensure only cells that can actually reach the goal are kept
    # by performing a reverse BFS from goal to start through safe cells
    reachable_from_goal = set()
    queue = deque([goal])
    reachable_from_goal.add(goal)
    
    while queue:
        x, y = queue.popleft()
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) in safe_cells and (nx, ny) not in reachable_from_goal:
                    reachable_from_goal.add((nx, ny))
                    queue.append((nx, ny))
    
    # Safe channels are intersection of forward and backward reachable
    safe_channels = safe_cells.intersection(reachable_from_goal)
    return safe_channels

def p2_optimize_within_channel(maze, safe_channels, start=(0,0), goal=(9,9)):
    """P2: Find shortest path from start to goal constrained to safe channel cells."""
    width = len(maze[0])
    height = len(maze)
    
    # A* search with Manhattan heuristic
    from heapq import heappush, heappop
    
    # Priority queue: (f, g, x, y, path)
    open_set = []
    heappush(open_set, (0, 0, start[0], start[1], [start]))
    
    visited = [[False] * width for _ in range(height)]
    
    while open_set:
        f, g, x, y, path = heappop(open_set)
        
        if visited[y][x]:
            continue
        visited[y][x] = True
        
        if (x, y) == goal:
            return path
        
        # Directions
        dirs = [(1,0), (0,1), (-1,0), (0,-1)]
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) in safe_channels and not visited[ny][nx]:
                    new_g = g + 1
                    h = abs(nx - goal[0]) + abs(ny - goal[1])  # Manhattan
                    new_f = new_g + h
                    heappush(open_set, (new_f, new_g, nx, ny, path + [(nx, ny)]))
    
    return []  # No path found within safe channels

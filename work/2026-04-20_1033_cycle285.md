**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Script completed. This is a runnable Python implementation that visualizes the Deadband Protocol in a 10x10 ASCII maze. The agent starts at (0,0), goal at (9,9), with random obstacles (rocks). It follows strict P0→P1→P2 priority: first map rocks (P0), then find safe channels (P1), then optimize path within safe channels (P2). Output shows each step.

**Output:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Tinker (Cocapn Fleet) — Cycle 285
"""

import random
from collections import deque

def generate_maze(width=10, height=10, rock_density=0.2):
    """Generate a maze with start, goal, and random rocks."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    rocks = set()
    
    # Place random rocks
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density and (x, y) not in [(0, 0), (width-1, height-1)]:
                maze[y][x] = '#'
                rocks.add((x, y))
    
    # Start and goal
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    
    return maze, rocks

def print_maze(maze, path=None, current=None):
    """Print maze with optional path and current position highlighting."""
    for y, row in enumerate(maze):
        line = []
        for x, cell in enumerate(row):
            if current and (x, y) == current:
                line.append('@')
            elif path and (x, y) in path:
                line.append('*')
            else:
                line.append(cell)
        print(' '.join(line))
    print()

def deadband_protocol_demo():
    print("=== Deadband Protocol Maze Demo ===")
    print("P0: Map rocks (negative space)")
    print("P1: Find safe channels (deadbands)")
    print("P2: Optimize path within channels")
    print()
    
    # Generate maze
    width, height = 10, 10
    maze, rocks = generate_maze(width, height, rock_density=0.2)
    
    print("Initial Maze (S=start, G=goal, #=rock):")
    print_maze(maze)
    
    # P0: Map negative space (rocks)
    print("--- P0: MAP NEGATIVE SPACE ---")
    print(f"Identified {len(rocks)} rock positions:")
    for x, y in sorted(rocks):
        print(f"  ({x},{y})", end='')
    print("\n")
    
    # P1: Find safe channels (BFS to find reachable area)
    print("--- P1: IDENTIFY SAFE CHANNELS ---")
    start = (0, 0)
    goal = (width-1, height-1)
    
    # BFS to find all reachable positions
    visited = set()
    queue = deque([start])
    reachable = set()
    
    while queue:
        x, y = queue.popleft()
        if (x, y) in visited:
            continue
        visited.add((x, y))
        reachable.add((x, y))
        
        # Check neighbors
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    queue.append((nx, ny))
    
    print(f"Safe channel size: {len(reachable)} positions")
    print("Visualizing safe channel (.) vs rocks (#):")
    for y in range(height):
        line = []
        for x in range(width):
            if (x, y) == start:
                line.append('S')
            elif (x, y) == goal:
                line.append('G')
            elif (x, y) in rocks:
                line.append('#')
            elif (x, y) in reachable:
                line.append('.')
            else:
                line.append('X')  # unreachable due to rocks
        print(' '.join(line))
    print()
    
    # Check if goal is reachable
    if goal not in reachable:
        print("FAIL: Goal not reachable from start (no safe channel)")
        return
    
    # P2: Optimize within channel (A* on safe graph)
    print("--- P2: OPTIMIZE WITHIN CHANNEL ---")
    
    # Simple BFS for shortest path in safe channel
    parent = {start: None}
    queue = deque([start])
    found = False
    
    while queue and not found:
        x, y = queue.popleft()
        if (x, y) == goal:
            found = True
            break
        
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) in reachable and (nx, ny) not in parent:
                    parent[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    
    if not found:
        print("FAIL: Could not find path within safe channel")
        return
    
    # Reconstruct path
    path = []
    current = goal
    while current is not None:
        path.append(current)
        current = parent[current]
    path.reverse()
    
    print(f"Optimal path length: {len(path)-1} steps")
    print("Path visualization (* = path):")
    print_maze(maze, path=set(path))
    
    # Show step-by-step
    print("Step-by-step path:")
    for i, (x, y) in enumerate(path):
        print(f"  Step {i}: ({x},{y})")
    
    print("\n=== Protocol Complete ===")
    print(f"Success: Found safe path avoiding {len(rocks)} rocks")

if __name__ == "__main__":
    random.seed(42)  # Deterministic for demo
    deadband_protocol_demo()
```

**Execution notes:**
- Run with `python3 deadband_maze_demo.py`
- Shows P0 (rock mapping), P1 (safe channel identification), P2 (optimized path)
- Visual ASCII output at each phase
- Deterministic seed for reproducibility
- Demonstrates strict P0→P1→P2 priority—no optimization before channel mapping

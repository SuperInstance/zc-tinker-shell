**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on the fleet knowledge, I have the constraint simulation results showing the Deadband Protocol's effectiveness. I will implement a visual demonstration using ASCII to map rocks (P0), find safe channels (P1), and then optimize a path (P2).

**Approach:**
1. **P0 (Map negative space):** Generate a random maze with obstacles ('#') as rocks.
2. **P1 (Identify safe channels):** Use BFS to find all reachable empty cells ('.') from the start, avoiding rocks.
3. **P2 (Optimize within channel):** Use A* search within the safe channels to find the shortest path to the goal.

**Output:** A Python script that prints each step visually.

**deadband_maze_demo.py**

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map rocks (obstacles)
P1: Find safe channels (reachable areas)
P2: Optimize path within safe channels
"""

import random
from collections import deque
import heapq

def generate_maze(width, height, rock_density=0.3):
    """Generate a random maze with rocks as obstacles."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    rocks = 0
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
                rocks += 1
    # Ensure start (0,0) and goal (width-1, height-1) are clear
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    return maze, rocks

def print_maze(maze, title="Maze"):
    """Print the maze with a title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))

def p0_map_rocks(maze):
    """P0: Identify all rock positions (negative space)."""
    rocks = []
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start=(0,0)):
    """P1: BFS to find all cells reachable without hitting rocks."""
    width, height = len(maze[0]), len(maze)
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    safe = set()
    queue = deque([start])
    safe.add(start)
    
    while queue:
        x, y = queue.popleft()
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in safe and maze[ny][nx] != '#':
                    safe.add((nx, ny))
                    queue.append((nx, ny))
    return safe

def p2_optimize_path(maze, safe_channels, start=(0,0), goal=None):
    """P2: A* search within safe channels for shortest path."""
    if goal is None:
        goal = (len(maze[0])-1, len(maze)-1)
    if goal not in safe_channels:
        return []  # Goal unreachable within safe channels
    
    width, height = len(maze[0]), len(maze)
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            return path
        
        x, y = current
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            neighbor = (nx, ny)
            if 0 <= nx < width and 0 <= ny < height:
                if neighbor not in safe_channels or maze[ny][nx] == '#':
                    continue  # Not a safe channel
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    return []  # No path found

def visualize_path(maze, path):
    """Mark the path on the maze with '*'."""
    maze_copy = [row[:] for row in maze]
    for x, y in path:
        if maze_copy[y][x] not in ('S', 'G'):
            maze_copy[y][x] = '*'
    return maze_copy

def main():
    # Configuration
    WIDTH, HEIGHT = 10, 8
    ROCK_DENSITY = 0.25
    
    print("DEADBAND PROTOCOL MAZE DEMONSTRATION")
    print("=" * 40)
    
    # Generate maze
    maze, total_rocks = generate_maze(WIDTH, HEIGHT, ROCK_DENSITY)
    print_maze(maze, "Initial Maze (S=start, G=goal, #=rock)")
    print(f"Total rocks: {total_rocks}")
    
    # P0: Map negative space
    rocks = p0_map_rocks(maze)
    print(f"\nP0: Mapped {len(rocks)} rock positions (negative space)")
    
    # P1: Find safe channels
    safe_channels = p1_find_safe_channels(maze)
    print(f"P1: Found {len(safe_channels)} safe cells (channels)")
    
    # Visualize safe channels
    safe_maze = [row[:] for row in maze]
    for y in range(HEIGHT):
        for x in range(WIDTH):
            if (x, y) in safe_channels and safe_maze[y][x] == '.':
                safe_maze[y][x] = 'O'  # Mark safe channel
    print_maze(safe_maze, "Safe Channels (O=reachable, #=rock)")
    
    # P2: Optimize path within safe channels
    path = p2_optimize_path(maze, safe_channels)
    if path:
        print(f"P2: Found optimal path with {len(path)} steps")
        final_maze = visualize_path(maze, path)
        print_maze(final_maze, "Optimal Path (*=path)")
        
        # Statistics
        goal = (WIDTH-1, HEIGHT-1)
        reachable = goal in safe_channels
        print(f"\n=== RESULTS ===")
        print(f"Rocks (P0): {len(rocks)}")
        print(f"Safe cells (P1): {len(safe_channels)}")
        print(f"Goal reachable: {reachable}")
        print(f"Path length: {len(path)}")
        if reachable:
            print("STATUS: SUCCESS - Deadband protocol followed")
        else:
            print("STATUS: FAILURE - Goal not in safe channels")
    else:
        print("P2: No path to goal within safe channels")
        print("STATUS: FAILURE - Optimization impossible without violating P0/P1")

if __name__ == "__main__":
    main()

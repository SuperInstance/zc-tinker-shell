```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

Simulates P0→P1→P2 in a 20×20 maze with rocks.
Shows ASCII visualization of each step.
"""

import random
from typing import List, Tuple, Set

# Constants from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

# Symbols for visualization
AGENT = 'A'
GOAL = 'G'
ROCK = '█'
SAFE = '·'
PATH = '○'
DEADBAND = '░'
EMPTY = ' '

class DeadbandMaze:
    """Demonstrates Deadband Protocol in a maze."""
    
    def __init__(self, seed=42):
        random.seed(seed)
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        
        # Initialize
        self._place_rocks()
        self.grid[self.start[0]][self.start[1]] = AGENT
        self.grid[self.goal[0]][self.goal[1]] = GOAL
    
    def _place_rocks(self):
        """P0: Map negative space (rocks)."""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            pos = (x, y)
            if pos != self.start and pos != self.goal:
                self.rocks.add(pos)
                self.grid[x][y] = ROCK
    
    def find_safe_channels(self) -> List[Set[Tuple[int, int]]]:
        """P1: Identify safe channels (deadbands)."""
        # Simple flood fill to find reachable safe cells
        visited = set()
        safe_region = set()
        stack = [self.start]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Skip if it's a rock
            if (x, y) in self.rocks:
                continue
            
            safe_region.add((x, y))
            
            # Check neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    stack.append((nx, ny))
        
        # Mark deadband cells (safe but not optimal path yet)
        for x, y in safe_region:
            if self.grid[x][y] == EMPTY:
                self.grid[x][y] = DEADBAND
        
        return [safe_region]  # Return as single channel for demo
    
    def optimize_within_channel(self, safe_channel: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
        """P2: Find optimal path within safe channel."""
        # Simple BFS within safe channel
        from collections import deque
        
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                # Mark path on grid
                for px, py in path[1:-1]:  # Skip start and goal
                    self.grid[px][py] = PATH
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                pos = (nx, ny)
                
                if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                    pos in safe_channel and pos not in visited):
                    visited.add(pos)
                    queue.append((pos, path + [pos]))
        
        return []  # No path found
    
    def run_protocol(self):
        """Execute full P0→P1→P2 protocol."""
        print("=" * 60)
        print("DEADBAND PROTOCOL DEMO - 20×20 MAZE")
        print(f"Rocks: {ROCK_COUNT}, Optimal path: {OPTIMAL_PATH_LENGTH} steps")
        print("=" * 60)
        
        # P0: Already done in init (rocks placed)
        print("\n[P0] NEGATIVE SPACE MAPPED")
        print("Rocks (█) represent failure states.")
        self.print_grid()
        input("\nPress Enter for P1...")
        
        # P1: Find safe channels
        print("\n[P1] SAFE CHANNELS IDENTIFIED")
        safe_channels = self.find_safe_channels()
        safe_cells = len(safe_channels[0]) if safe_channels else 0
        print(f"Safe channel size: {safe_cells} cells")
        print("Deadband (░) shows where agent CAN operate.")
        self.print_grid()
        input("\nPress Enter for P2...")
        
        # P2: Optimize within channel
        print("\n[P2] OPTIMIZE WITHIN SAFE CHANNEL")
        if safe_channels:
            path = self.optimize_within_channel(safe_channels[0])
            if path:
                print(f"Path found! Length: {len(path)-1} steps")
                print("Path (○) shows optimal route through safe water.")
            else:
                print("No path found within safe channel.")
        else:
            print("No safe channels found.")
        
        self.print_grid()
        
        # Statistics
        print("\n" + "=" * 60)
        print("PROTOCOL COMPLETE")
        if safe_channels and path:
            print(f"✓ Successfully navigated {len(path)-1} steps")
            print(f"✓ Stayed within safe channel ({safe_cells} cells)")
            print(f"✓ Avoided all {ROCK_COUNT} rocks")
        print("=" * 60)
    
    def print_grid(self):
        """Print ASCII visualization."""
        print("\n" + " " * 3 + "".join(str(i%10) for i in range(MAZE_SIZE)))
        print("   " + "-" * MAZE_SIZE)
        
        for i in range(MAZE_SIZE):
            row_str = f"{i:2}|"
            for j in range(MAZE_SIZE):
                row_str += self.grid[i][j]
            print(row_str + f"|{i:2}")
        
        print("   " + "-" * MAZE_SIZE)
        print(" " * 3 + "".join(str(i%10) for i in range(MAZE_SIZE)))
        
        # Legend
        print("\nLegend:")
        print(f"  {AGENT} = Agent start")
        print(f"  {GOAL} = Goal")
        print(f"  {ROCK} = Rock (failure)")
        print(f"  {DEADBAND} = Safe channel (deadband)")
        print(f"  {PATH} = Optimal path")
        print(f"  {EMPTY} = Unreachable/unsafe")

def main():
    """Run the demonstration."""
    print("Fleet Knowledge: Deadband Protocol Maze Demo")
    print("Based on constraint_sim_final.json data")
    print("Simulating 20×20 maze with 64 rocks\n")
    
    # Create and run maze
    maze = DeadbandMaze(seed=71)  # Cycle 71
    maze.run_protocol()
    
    # Show comparison to unconstrained approach
    print("\n" + "=" * 60)
    print("COMPARISON TO UNCONSTRAINED APPROACH")
    print("=" * 60)
    print("From constraint_sim_final.json:")
    print(f"  Unconstrained success: 33/50 runs")
    print(f"  Unconstrained avg steps: 2320.9")
    print(f"  Deadband success: 50/50 runs")
    print(f"  Deadband avg steps: 21.0 (optimal)")
    print("\nKey insight: P0+P1 first prevents hitting rocks,")
    print("then P2 finds optimal path through safe space.")

if __name__ == "__main__":
    main()
```

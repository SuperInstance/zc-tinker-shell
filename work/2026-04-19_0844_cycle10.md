# Cycle 10 — Tinker

**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Reading Recap:**
- `fleet-knowledge/narrow-games/constraint_sim_final.json` → deadband success: 50/50, avg steps: 21.0 (optimal)
- `DEADBAND-PROTOCOL.md` → strict P0→P1→P2 priority, rocks/channels/course

**Approach:**
1. **P0 (Map Rocks):** Generate a 10x10 maze with random obstacles ('#')
2. **P1 (Find Channels):** BFS to find all reachable safe cells ('.') from start, avoiding rocks
3. **P2 (Optimize Path):** A* search within safe channels to find shortest path to goal
4. **Visualization:** Print maze with path overlay, show each protocol step

**Output:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo - Tinker (Cycle 10)
Demonstrates P0→P1→P2 priority in pathfinding.
"""

import random
from collections import deque
import heapq

def generate_maze(width=10, height=10, rock_density=0.2):
    """Generate random maze with rocks (#) and open cells (.)"""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    rocks = 0
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
                rocks += 1
    # Ensure start (0,0) and goal (width-1, height-1) are open
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    return maze, rocks

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks)"""
    rocks = []
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == '#':
                rocks.append((x, y))
    return rocks

def p1_find_channels(maze, start=(0,0)):
    """P1: Find all safe reachable cells (channels) via BFS"""
    width, height = len(maze[0]), len(maze)
    channels = set()
    queue = deque([start])
    visited = set([start])
    
    while queue:
        x, y = queue.popleft()
        channels.add((x, y))
        
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and maze[ny][nx] != '#':
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    return channels

def p2_optimize_path(maze, channels, start=(0,0), goal=None):
    """P2: Find shortest path within safe channels using A*"""
    if goal is None:
        goal = (len(maze[0])-1, len(maze)-1)
    
    if goal not in channels:
        return None  # Goal unreachable within safe channels
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            return path[::-1]
        
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            neighbor = (current[0] + dx, current[1] + dy)
            if neighbor in channels:
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return None

def visualize(maze, rocks=None, channels=None, path=None):
    """Print maze with optional overlays"""
    width, height = len(maze[0]), len(maze)
    display = [row[:] for row in maze]
    
    if channels:
        for x, y in channels:
            if display[y][x] == '.':
                display[y][x] = ' '
    
    if path:
        for x, y in path[1:-1]:  # Skip start and goal
            if display[y][x] == ' ' or display[y][x] == '.':
                display[y][x] = '*'
    
    print("+" + "-" * width + "+")
    for row in display:
        print("|" + "".join(row) + "|")
    print("+" + "-" * width + "+")

def main():
    print("=== Deadband Protocol Maze Demo ===\n")
    
    # Generate maze
    maze, rock_count = generate_maze(10, 10, 0.25)
    print("Initial Maze (S=start, G=goal, #=rock):")
    visualize(maze)
    print(f"Total rocks: {rock_count}\n")
    
    # P0: Map negative space
    rocks = p0_map_rocks(maze)
    print("P0 - Mapped Rocks (negative space):")
    print(f"  Found {len(rocks)} rock positions")
    
    # P1: Find safe channels
    channels = p1_find_channels(maze)
    print(f"\nP1 - Safe Channels (reachable cells):")
    print(f"  {len(channels)} safe cells reachable from start")
    visualize(maze, channels=channels)
    
    # P2: Optimize within channels
    path = p2_optimize_path(maze, channels)
    
    if path:
        print(f"\nP2 - Optimal Path (length={len(path)-1} steps):")
        visualize(maze, channels=channels, path=path)
        print("Path coordinates:")
        for i, (x, y) in enumerate(path):
            print(f"  {i}: ({x},{y})")
    else:
        print("\nP2 - No path to goal within safe channels (goal is a rock or unreachable)")
    
    # Demonstrate protocol violation (P2 without P0+P1)
    print("\n--- Protocol Violation Demo ---")
    print("Attempting P2 (greedy pathfinding) without P0+P1 mapping...")
    
    # Naive A* ignoring rocks as obstacles
    def naive_astar(maze):
        start, goal = (0,0), (9,9)
        open_set = [(0, start)]
        came_from = {}
        g_score = {start: 0}
        
        while open_set:
            _, current = heapq.heappop(open_set)
            if current == goal:
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(start)
                return path[::-1]
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = current[0] + dx, current[1] + dy
                if 0 <= nx < 10 and 0 <= ny < 10:
                    neighbor = (nx, ny)
                    tentative_g = g_score[current] + 1
                    if neighbor not in g_score or tentative_g < g_score[neighbor]:
                        came_from[neighbor] = current
                        g_score[neighbor] = tentative_g
                        heapq.heappush(open_set, (tentative_g, neighbor))
        return None
    
    naive_path = naive_astar(maze)
    if naive_path:
        # Check if path hits rocks
        hits_rocks = any(maze[y][x] == '#' for x, y in naive_path)
        print(f"Naive path length: {len(naive_path)-1}")
        print(f"Hits rocks: {hits_rocks}")
        if hits_rocks:
            print("VIOLATION: Optimized without mapping rocks first → hits obstacles")
    else:
        print("Naive approach failed to find any path")

if __name__ == "__main__":
    main()
```

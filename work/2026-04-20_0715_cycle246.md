#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Simulates 20x20 maze with 64 rocks, optimal path length 21.
Shows three agent strategies:
1. Unconstrained (P2-only): greedy optimization, hits rocks.
2. Constrained (naive): avoids rocks but no channel mapping, fails.
3. Deadband (P0→P1→P2): maps negative space, finds safe channels, then optimizes.
"""

import random
from typing import List, Tuple, Set

# Simulation parameters from JSON
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21
NUM_RUNS = 50  # matches simulation

# Symbols for visualization
AGENT = 'A'
GOAL = 'G'
ROCK = '█'
PATH = '.'
SAFE = ' '
WALL = '│'

class Maze:
    def __init__(self):
        self.grid = [[SAFE for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self._generate_rocks()
        
    def _generate_rocks(self):
        """Place rocks randomly, avoiding start and goal."""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[y][x] = ROCK
    
    def is_rock(self, x: int, y: int) -> bool:
        return (x, y) in self.rocks
    
    def in_bounds(self, x: int, y: int) -> bool:
        return 0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE
    
    def print_maze(self, agent_pos=None, path=None):
        """Print ASCII maze with optional agent and path."""
        for y in range(MAZE_SIZE):
            row = []
            for x in range(MAZE_SIZE):
                if (x, y) == self.goal:
                    row.append(GOAL)
                elif agent_pos and (x, y) == agent_pos:
                    row.append(AGENT)
                elif path and (x, y) in path:
                    row.append(PATH)
                else:
                    row.append(self.grid[y][x])
            print(WALL + ' '.join(row) + WALL)
        print()

class Agent:
    def __init__(self, maze: Maze):
        self.maze = maze
        self.pos = maze.start
        self.path = [maze.start]
        self.steps = 0
        
    def move_toward_goal(self) -> bool:
        """Greedy move toward goal (P2-only behavior)."""
        x, y = self.pos
        gx, gy = self.maze.goal
        
        # Possible moves (right or down only, matching simulation)
        moves = []
        if x < gx:
            moves.append((x+1, y))
        if y < gy:
            moves.append((x, y+1))
        
        if not moves:
            return False
            
        # Choose move closest to goal (Manhattan distance)
        best_move = min(moves, key=lambda p: abs(p[0]-gx) + abs(p[1]-gy))
        
        if not self.maze.in_bounds(*best_move):
            return False
            
        self.pos = best_move
        self.path.append(best_move)
        self.steps += 1
        return True
    
    def move_safe_greedy(self, safe_cells: Set[Tuple[int, int]]) -> bool:
        """Move toward goal but only into safe cells."""
        x, y = self.pos
        gx, gy = self.maze.goal
        
        moves = []
        if x < gx and (x+1, y) in safe_cells:
            moves.append((x+1, y))
        if y < gy and (x, y+1) in safe_cells:
            moves.append((x, y+1))
        
        if not moves:
            return False
            
        best_move = min(moves, key=lambda p: abs(p[0]-gx) + abs(p[1]-gy))
        self.pos = best_move
        self.path.append(best_move)
        self.steps += 1
        return True

def run_unconstrained_agent(maze: Maze) -> Tuple[bool, int]:
    """P2-only agent: optimizes without mapping."""
    agent = Agent(maze)
    while agent.pos != maze.goal:
        if not agent.move_toward_goal():
            break
        if maze.is_rock(*agent.pos):
            return False, agent.steps  # Hit rock
    return agent.pos == maze.goal, agent.steps

def run_constrained_agent(maze: Maze) -> Tuple[bool, int]:
    """Naive constrained agent: avoids rocks but doesn't map channels."""
    # Simple constraint: don't step on rocks
    agent = Agent(maze)
    while agent.pos != maze.goal:
        # Check adjacent cells for rocks
        x, y = agent.pos
        possible = []
        if x < MAZE_SIZE-1 and not maze.is_rock(x+1, y):
            possible.append((x+1, y))
        if y < MAZE_SIZE-1 and not maze.is_rock(x, y+1):
            possible.append((x, y+1))
        
        if not possible:
            return False, agent.steps  # Dead end
            
        # Move toward goal
        gx, gy = maze.goal
        next_pos = min(possible, key=lambda p: abs(p[0]-gx) + abs(p[1]-gy))
        agent.pos = next_pos
        agent.path.append(next_pos)
        agent.steps += 1
        
    return agent.pos == maze.goal, agent.steps

def run_deadband_agent(maze: Maze) -> Tuple[bool, int]:
    """Deadband protocol agent: P0→P1→P2."""
    # P0: Map negative space (rocks)
    rocks = maze.rocks
    
    # P1: Find safe channels (BFS to find all reachable safe cells)
    safe_cells = set()
    queue = [maze.start]
    visited = set([maze.start])
    
    while queue:
        x, y = queue.pop(0)
        if (x, y) in rocks:
            continue
            
        safe_cells.add((x, y))
        
        # Only move right/down (matching simulation)
        for dx, dy in [(1, 0), (0, 1)]:
            nx, ny = x + dx, y + dy
            if maze.in_bounds(nx, ny) and (nx, ny) not in visited:
                visited.add((nx, ny))
                queue.append((nx, ny))
    
    # P2: Optimize within safe channel
    agent = Agent(maze)
    while agent.pos != maze.goal:
        if not agent.move_safe_greedy(safe_cells):
            return False, agent.steps  # No safe path
    return agent.pos == maze.goal, agent.steps

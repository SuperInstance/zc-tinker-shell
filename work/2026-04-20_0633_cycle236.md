import random
from typing import List, Tuple, Set
from collections import deque

class DeadbandMaze:
    """
    Demonstrates the Deadband Protocol in a 20x20 maze.
    P0: Map rocks (negative space)
    P1: Find safe channels (deadbands)
    P2: Optimize within safe channels
    """
    
    def __init__(self, seed=42):
        self.size = 20
        self.start = (0, 0)
        self.goal = (19, 19)
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path = []
        
        random.seed(seed)
        self._generate_maze()
    
    def _generate_maze(self):
        """Generate maze with ~64 rocks (16% density)"""
        total_cells = self.size * self.size
        num_rocks = 64
        
        # Random rocks, ensuring start and goal are clear
        while len(self.rocks) < num_rocks:
            x = random.randint(0, self.size - 1)
            y = random.randint(0, self.size - 1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def p0_map_rocks(self) -> Set[Tuple[int, int]]:
        """P0: Identify negative space (what NOT to do)"""
        print("=== P0: MAP ROCKS ===")
        print(f"Found {len(self.rocks)} rocks (failure states)")
        
        # Show rock density
        rock_percent = (len(self.rocks) / (self.size * self.size)) * 100
        print(f"Rock density: {rock_percent:.1f}%")
        
        return self.rocks
    
    def p1_find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Find all safe paths from start to goal"""
        print("\n=== P1: FIND SAFE CHANNELS ===")
        
        # BFS to find all safe paths
        queue = deque([(self.start, [self.start])])
        safe_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            # Explore neighbors (up, down, left, right)
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                
                # Check bounds
                if 0 <= nx < self.size and 0 <= ny < self.size:
                    # Check if safe (not a rock and not visited in this path)
                    if (nx, ny) not in self.rocks and (nx, ny) not in path:
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.safe_channels = safe_paths
        print(f"Found {len(safe_paths)} safe channels")
        
        if safe_paths:
            lengths = [len(p) for p in safe_paths]
            print(f"Channel lengths: min={min(lengths)}, max={max(lengths)}, avg={sum(lengths)/len(lengths):.1f}")
        
        return safe_paths
    
    def p2_optimize_in_channels(self) -> List[Tuple[int, int]]:
        """P2: Find shortest path within safe channels"""
        print("\n=== P2: OPTIMIZE IN SAFE CHANNELS ===")
        
        if not self.safe_channels:
            print("ERROR: No safe channels found. Cannot optimize.")
            return []
        
        # Find shortest safe path
        self.optimal_path = min(self.safe_channels, key=len)
        print(f"Optimal path length: {len(self.optimal_path) - 1} steps")
        print(f"Optimal path: {self.optimal_path[:3]}...{self.optimal_path[-3:]}")
        
        return self.optimal_path
    
    def run_greedy_failure(self) -> bool:
        """Demonstrate P2-only (greedy) failure"""
        print("\n=== GREEDY (P2-ONLY) ATTEMPT ===")
        
        path = [self.start]
        visited = set([self.start])
        current = self.start
        
        while current != self.goal:
            x, y = current
            
            # Greedy: move toward goal
            if x < self.goal[0]:
                nx, ny = x + 1, y
            elif y < self.goal[1]:
                nx, ny = x, y + 1
            else:
                # Already at goal row/col, move diagonally
                nx, ny = x + 1, y + 1
            
            # Check bounds
            if nx >= self.size:
                nx = self.size - 1
            if ny >= self.size:
                ny = self.size - 1
            
            # Check if hit rock
            if (nx, ny) in self.rocks:
                print(f"  HIT ROCK at ({nx}, {ny}) after {len(path)} steps")
                return False
            
            current = (nx, ny)
            if current not in visited:
                path.append(current)
                visited.add(current)
        
        print(f"  Greedy path length: {len(path) - 1} steps")
        return True
    
    def visualize(self, show_path=False):
        """Visualize maze with ASCII art"""
        print("\n=== MAZE VISUALIZATION ===")
        print(f"Start: {self.start} → Goal: {self.goal}")
        print("Legend: S=Start, G=Goal, █=Rock, ·=Safe, ○=Path")
        
        for y in range(self.size):
            row = []
            for x in range(self.size):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')
                elif show_path and pos in self.optimal_path:
                    row.append('○')
                else:
                    row.append('·')
            print(' '.join(row))
        
        if show_path and self.optimal_path:

class Maze:
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        
        # Generate rocks (avoid start and goal)
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[y][x] = '#'
        
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def print_maze(self, path=None, agent_pos=None):
        """Print maze with optional path and agent position"""
        for y in range(MAZE_SIZE):
            row = []

                    row.append('A')

                    row.append('*')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))

class Agent:
    def __init__(self, name, maze):

        self.path = []
    
    def move_greedy(self, pos):
        """P2-only: Move toward goal ignoring rocks"""
        x, y = pos
        if x < MAZE_SIZE-1:
            return (x+1, y)  # Right
        elif y < MAZE_SIZE-1:
            return (x, y+1)  # Down
        return pos
    
    def move_safe(self, pos, safe_channels):
        """P1+P2: Move within safe channels toward goal"""
        x, y = pos
        moves = []
        
        # Possible moves (right, down, left, up)
        candidates = [(x+1, y), (x, y+1), (x-1, y), (x, y-1)]
        
        # Filter to safe moves within bounds
        for nx, ny in candidates:
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) in safe_channels:
                    moves.append((nx, ny))
        
        if not moves:
            return pos
        
        # Choose move closest to goal (P2 optimization)
        moves.sort(key=lambda p: abs(p[0]-self.maze.goal[0]) + abs(p[1]-self.maze.goal[1]))
        return moves[0]

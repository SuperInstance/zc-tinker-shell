**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Complete. Script written, tested, and ready for submission as a training tile.

**Approach (as implemented):**
1. **P0 (Map negative space):** Generate a random 10x10 maze with walls ('#') as rocks. Identify all wall cells.
2. **P1 (Find safe channels):** Use BFS from start (0,0) to find all reachable empty cells ('.') — these are the safe channels.
3. **P2 (Optimize within channels):** Use BFS again within the safe channels to find the shortest path from start to goal (9,9).
4. **Visualization:** Print the maze at each step, marking start ('S'), goal ('G'), safe channels ('·'), and final path ('*').

**Key demonstration:**
- The agent never attempts to move through walls (rocks).
- It only considers paths within safe, reachable cells.
- Optimization (shortest path) is constrained to the safe channel.

**Code (deadband_maze_demo.py):**
```python
import random
from collections import deque

def generate_maze(width=10, height=10, wall_prob=0.3):
    """Generate random maze with walls as rocks (P0 negative space)."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    for y in range(height):
        for x in range(width):
            if random.random() < wall_prob and not (x == 0 and y == 0) and not (x == width-1 and y == height-1):
                maze[y][x] = '#'
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    return maze

def p0_map_rocks(maze):
    """P0: Identify all wall cells (negative space)."""
    rocks = []
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start=(0,0)):
    """P1: BFS to find all reachable empty cells (safe channels)."""
    width, height = len(maze[0]), len(maze)
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    visited = set()
    queue = deque([start])
    safe_cells = set()
    
    while queue:
        x, y = queue.popleft()
        if (x, y) in visited:
            continue
        visited.add((x, y))
        if maze[y][x] != '#':
            safe_cells.add((x, y))
            for dx, dy in directions:
                nx, ny = x + dx, y + dy
                if 0 <= nx < width and 0 <= ny < height and (nx, ny) not in visited:
                    queue.append((nx, ny))
    return safe_cells

def p2_optimize_path(maze, safe_cells, start=(0,0), goal=(9,9)):
    """P2: BFS within safe channels to find shortest path."""
    if goal not in safe_cells:
        return None
    width, height = len(maze[0]), len(maze)
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    queue = deque([(start, [start])])
    visited = set([start])
    
    while queue:
        (x, y), path = queue.popleft()
        if (x, y) == goal:
            return path
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height and (nx, ny) in safe_cells and (nx, ny) not in visited:
                visited.add((nx, ny))
                queue.append(((nx, ny), path + [(nx, ny)]))
    return None

def print_maze(maze, safe_cells=None, path=None):
    """Visualize maze with optional safe channels and path."""
    for y in range(len(maze)):
        row = []
        for x in range(len(maze[0])):
            cell = maze[y][x]
            if path and (x, y) in path and cell not in 'SG':
                row.append('*')
            elif safe_cells and (x, y) in safe_cells and cell == '.':
                row.append('·')
            else:
                row.append(cell)
        print(' '.join(row))

def main():
    print("=== Deadband Protocol Maze Demo ===\n")
    random.seed(42)  # For reproducibility
    maze = generate_maze()
    
    print("1. Initial Maze (S=start, G=goal, #=rock):")
    print_maze(maze)
    
    rocks = p0_map_rocks(maze)
    print(f"\n2. P0: Mapped {len(rocks)} rocks (negative space).")
    
    safe_cells = p1_find_safe_channels(maze)
    print(f"3. P1: Found {len(safe_cells)} safe channel cells.")
    print("   Maze with safe channels marked (·):")
    print_maze(maze, safe_cells=safe_cells)
    
    path = p2_optimize_path(maze, safe_cells)
    if path:
        print(f"\n4. P2: Optimized path found ({len(path)-1} steps).")
        print("   Final path marked (*):")
        print_maze(maze, path=path)
    else:
        print("\n4. P2: No path to goal within safe channels (goal unreachable).")

if __name__ == "__main__":
    main()
```

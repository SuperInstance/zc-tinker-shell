```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 40)
A visual ASCII demonstration of P0→P1→P2 using a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time

# Constants from simulation data
MAZE_SIZE = 20
ROCK_COUNT = 64  # ~16% density, matching sim
START = (0, 0)
GOAL = (19, 19)

def generate_maze():
    """Create a 20x20 maze with rocks ('#'), open cells ('.'), start ('S'), goal ('G')."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    maze[START[0]][START[1]] = 'S'
    maze[GOAL[0]][GOAL[1]] = 'G'
    
    # Place rocks randomly, avoiding start and goal
    rocks_placed = 0
    while rocks_placed < ROCK_COUNT:
        x = random.randint(0, MAZE_SIZE-1)
        y = random.randint(0, MAZE_SIZE-1)
        if (x, y) == START or (x, y) == GOAL:
            continue
        if maze[x][y] == '.':
            maze[x][y] = '#'
            rocks_placed += 1
    return maze

def print_maze(maze, agent_pos=None):
    """Print the maze with optional agent position."""
    for i in range(MAZE_SIZE):
        row = ''
        for j in range(MAZE_SIZE):
            if agent_pos and (i, j) == agent_pos:
                row += 'A '
            else:
                row += maze[i][j] + ' '
        print(row)
    print()

def p0_map_negative_space(maze):
    """P0: Identify all rock positions (what NOT to do)."""
    rocks = []
    for i in range(MAZE_SIZE):
        for j in range(MAZE_SIZE):
            if maze[i][j] == '#':
                rocks.append((i, j))
    print(f"[P0] Mapped negative space: {len(rocks)} rocks")
    return set(rocks)

def p1_find_safe_channels(maze, rocks):
    """P1: Find all cells that are not rocks (safe channels)."""
    safe = []
    for i in range(MAZE_SIZE):
        for j in range(MAZE_SIZE):
            if (i, j) not in rocks:
                safe.append((i, j))
    print(f"[P1] Identified safe channels: {len(safe)} cells")
    return set(safe)

def p2_optimize_within_channel(maze, safe, start, goal):
    """P2: Greedy pathfinding constrained to safe cells only."""
    # Simple greedy: move right/down toward goal, avoiding unsafe cells
    path = [start]
    current = start
    
    while current != goal:
        x, y = current
        # Prefer moves that reduce Manhattan distance
        candidates = []
        if x < goal[0] and (x+1, y) in safe:
            candidates.append((x+1, y))
        if y < goal[1] and (x, y+1) in safe:
            candidates.append((x, y+1))
        if not candidates:
            # Fallback: allow up/left if needed
            if x > 0 and (x-1, y) in safe:
                candidates.append((x-1, y))
            if y > 0 and (x, y-1) in safe:
                candidates.append((x, y-1))
        
        if not candidates:
            print("[P2] No safe path found — trapped!")
            return None
        
        current = candidates[0]  # Take first viable move
        path.append(current)
    
    print(f"[P2] Optimized path found: {len(path)} steps")
    return path

def run_unconstrained_greedy(maze, start, goal):
    """Naive P2-only approach (violates protocol)."""
    print("\n=== VIOLATION: Unconstrained Greedy (P2-only) ===")
    path = [start]
    current = start
    
    while current != goal:
        x, y = current
        # Move toward goal ignoring rocks
        if x < goal[0]:
            next_cell = (x+1, y)
        elif y < goal[1]:
            next_cell = (x, y+1)
        else:
            next_cell = (x-1, y)  # fallback
        
        if maze[next_cell[0]][next_cell[1]] == '#':
            print(f"  HIT ROCK at {next_cell}! Path failed.")
            return None
        
        current = next_cell
        path.append(current)
    
    print(f"  Reached goal in {len(path)} steps (lucky)")
    return path

def main():
    print("=" * 50)
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print("Optimal: 21 steps | Rocks: 64 | Deadband success: 50/50")
    print("=" * 50)
    
    # Generate maze
    random.seed(42)  # Deterministic for demo
    maze = generate_maze()
    
    print("\nInitial Maze:")
    print_maze(maze)
    
    # === DEADBAND PROTOCOL (P0→P1→P2) ===
    print("=== DEADBAND PROTOCOL EXECUTION ===")
    
    # P0: Map negative space
    rocks = p0_map_negative_space(maze)
    time.sleep(1)
    
    # P1: Identify safe channels
    safe = p1_find_safe_channels(maze, rocks)
    time.sleep(1)
    
    # P2: Optimize within safe channels
    path = p2_optimize_within_channel(maze, safe, START, GOAL)
    
    if path:
        print("\nSafe Path Visualization:")
        for step in path:
            print_maze(maze, step)
            time.sleep(0.1)
        print(f"✓ Successfully reached goal in {len(path)-1} moves")
    else:
        print("✗ Failed to find safe path")
    
    # === UNCONSTRAINED GREEDY (for contrast) ===
    run_unconstrained_greedy(maze, START, GOAL)
    
    print("\n" + "=" * 50)
    print("Protocol Summary:")
    print("  P0 (rocks): Know what to avoid")
    print("  P1 (safe): Stay in viable space")  
    print("  P2 (path): Optimize only within safe channels")
    print("Violating P0→P1 order → high probability of failure")
    print("=" * 50)

if __name__ == "__main__":
    main()
```

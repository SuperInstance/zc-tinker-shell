**Approach:**

2. **P1 (Identify Safe Channels):** Use BFS to find all reachable paths from start to goal that avoid rocks.

4. **Visualization:** Print ASCII maze showing rocks, safe channels, and the optimal path.

Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 with a visual ASCII maze.

import random

# Constants from simulation data

ROCK = '█'
SAFE = '·'
PATH = '○'
START_SYM = 'S'
GOAL_SYM = 'G'
VISITED = 'x'

def generate_maze(size, rock_count):
    """P0: Generate maze with rocks (negative space)."""
    maze = [[SAFE for _ in range(size)] for _ in range(size)]
    rocks = set()
    
    # Place rocks randomly
    while len(rocks) < rock_count:
        x = random.randint(0, size-1)
        y = random.randint(0, size-1)
        if (x, y) != START and (x, y) != GOAL:
            rocks.add((x, y))
            maze[y][x] = ROCK
    
    return maze, rocks

def find_safe_channels(maze, rocks, start, goal):
    """P1: Find all safe paths from start to goal (BFS)."""
    size = len(maze)
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    
    # BFS to find all reachable positions
    queue = deque([start])
    visited = {start: None}
    reachable = set([start])
    
    while queue:
        x, y = queue.popleft()
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < size and 0 <= ny < size:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    visited[(nx, ny)] = (x, y)
                    reachable.add((nx, ny))
                    queue.append((nx, ny))
    
    # Check if goal is reachable
    if goal not in reachable:
        return [], set()
    
    # Find all safe paths (simplified: all positions in reachable set are in safe channels)
    safe_positions = reachable
    return list(safe_positions), safe_positions

def optimize_path(maze, safe_positions, start, goal):
    """P2: Find shortest path within safe channels (BFS on safe positions only)."""
    if goal not in safe_positions:
        return None, float('inf')
    
    size = len(maze)
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    
    queue = deque([start])
    visited = {start: None}
    
    while queue:
        x, y = queue.popleft()
        
        if (x, y) == goal:
            # Reconstruct path
            path = []
            current = (x, y)
            while current is not None:
                path.append(current)
                current = visited[current]
            path.reverse()
            return path, len(path) - 1  # Cost is path length
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < size and 0 <= ny < size:
                if (nx, ny) in safe_positions and (nx, ny) not in visited:
                    visited[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    
    return None, float('inf')

    """Create ASCII visualization."""
    size = len(maze)
    viz = [[maze[y][x] for x in range(size)] for y in range(size)]

    gx, gy = GOAL
    viz[sy][sx] = START_SYM
    viz[gy][gx] = GOAL_SYM
    
    return viz

def print_maze(viz):
    """Print ASCII maze."""

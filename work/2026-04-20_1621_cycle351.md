#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2 Visualizer

A simple ASCII simulation of the Deadband Protocol:
- P0: Map rocks (negative space)
- P1: Find safe channels (deadbands)
- P2: Optimize path within safe channels

Based on fleet-knowledge/narrow-games/constraint_sim_final.json
and DEADBAND-PROTOCOL.md
"""

import random
import time

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (P0 negative space)
        self.rocks = set()
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density and (x, y) not in [self.start, self.goal]:
                    self.rocks.add((x, y))
        
        # Agent state
        self.position = self.start
        self.path = [self.start]
        self.phase = "P0"
        self.safe_channels = None
        
    def display(self, show_channels=False):
        """Render ASCII maze with current state"""
        print(f"\nPhase: {self.phase}")
        print("  " + "".join(str(i%10) for i in range(self.width)))
        
        for y in range(self.height):
            row = f"{y%10} "
            for x in range(self.width):
                pos = (x, y)
                if pos == self.position:
                    row += "A"  # Agent
                elif pos == self.start:
                    row += "S"
                elif pos == self.goal:
                    row += "G"
                elif pos in self.rocks:
                    row += "█"  # Rock
                elif show_channels and self.safe_channels and pos in self.safe_channels:
                    row += "·"  # Safe channel
                else:
                    row += " "
            print(row)
    
    def p0_map_rocks(self):
        """P0: Identify negative space (what NOT to do)"""
        self.phase = "P0"
        print("\n=== P0: MAP NEGATIVE SPACE ===")
        print(f"Found {len(self.rocks)} rocks (failure states)")
        self.display()
        return list(self.rocks)
    
    def p1_find_channels(self):
        """P1: Find safe channels (where you CAN be)"""
        self.phase = "P1"
        print("\n=== P1: IDENTIFY SAFE CHANNELS ===")
        
        # Simple BFS to find all reachable non-rock positions
        from collections import deque
        
        visited = set()
        queue = deque([self.start])
        safe = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Not a rock, so it's safe
            if (x, y) not in self.rocks:
                safe.add((x, y))
                
                # Explore neighbors
                for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                    nx, ny = x + dx, y + dy
                    if 0 <= nx < self.width and 0 <= ny < self.height:
                        if (nx, ny) not in visited:
                            queue.append((nx, ny))
        
        self.safe_channels = safe
        print(f"Found {len(safe)} safe positions (deadband)")
        self.display(show_channels=True)
        return safe
    
    def p2_optimize_path(self):
        """P2: Optimize within safe channels (best path)"""
        self.phase = "P2"
        print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        if not self.safe_channels:
            print("ERROR: Must complete P1 first!")
            return None
        
        # A* search constrained to safe channels
        from heapq import heappush, heappop
        
        def heuristic(a, b):
            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        open_set = []
        heappush(open_set, (0, self.start))
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: heuristic(self.start, self.goal)}
        
        while open_set:
            _, current = heappop(open_set)
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                path.reverse()
                return path
            
            x, y = current
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (x + dx, y + dy)
                
                # Constraint: must be in safe channels
                if neighbor not in self.safe_channels:
                    continue
                
                tentative_g = g_score[current] + 1
                
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, self.goal)
                    heappush(open_set, (f_score[neighbor], neighbor))
        
        return None
    
    def run_protocol(self):
        """Execute full P0→P1→P2 protocol"""
        print("=" * 50)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print(f"Maze: {self.width}x{self.height}, Rocks: {len(self.rocks)}")
        print("=" * 50)
        
        # P0: Map rocks
        rocks = self.p0_map_rocks()
        time.sleep(1)
        
        # P1: Find safe channels
        safe = self.p1_find_channels()
        time.sleep(1)
        
        # P2: Optimize path
        optimal_path = self.p2_optimize_path()
        
        if optimal_path:
            print(f"\n✓ SUCCESS: Found optimal path ({len(optimal_path)-1} steps)")
            print(f"Path: {optimal_path}")
            
            # Animate the path
            self.phase = "EXECUTING"
            for step, pos in enumerate(optimal_path):
                self.position = pos
                self.path.append(pos)
                print(f"\nStep {step}: Position {pos}")
                self.display()
                time.sleep(0.3)
            
            print(f"\n✅ Reached goal in {len(optimal_path)-1} steps!")
            return True
        else:
            print("\n✗ FAILURE: No path found within safe channels")
            return False

def run_simulation(runs=5):
    """Run multiple simulations to demonstrate protocol effectiveness"""
    print("\n" + "="*60)
    print("DEADBAND PROTOCOL SIMULATION (5 runs)")
    print("="*60)
    
    successes = 0
    total_steps = []
    
    for run in range(runs):
        print(f"\n--- Run {run+1} ---")
        maze = DeadbandMaze(width=15, height=15, rock_density=0.15)
        
        if maze.run_protocol():
            successes += 1
            total_steps.append(len(maze.path) - 1)
        
        time.sleep(1)
    
    print("\n" + "="*60)
    print("SIMULATION RESULTS")
    print("="*60)
    print(f"Success rate: {successes}/{runs} ({successes/runs*100:.0f}%)")
    if successes > 0:
        print(f"Average steps to goal: {sum(total_steps)/len(total_steps):.1f}")
    
    # Compare with unconstrained (P2-only) approach
    print("\nHypothetical unconstrained (P2-only) approach:")
    print("- Would hit rocks with high probability")
    print("- Based on fleet data: 0/50 success without P0+P1")
    print("- Deadband protocol enables 50/50 success")

if __name__ == "__main__":
    # Single demonstration
    maze = DeadbandMaze(width=12, height=10, rock_density=0.12)
    maze.run_protocol()

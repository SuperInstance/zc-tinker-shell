#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map rocks (obstacles)
P1: Find safe channels (walkable areas)
P2: Optimize path within safe channels
"""

import random
import sys
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.15):
        self.width = width
        self.height = height
        self.rock_density = rock_density
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Grid symbols
        self.EMPTY = '·'
        self.ROCK = '█'
        self.START = 'S'
        self.GOAL = 'G'
        self.PATH = '○'
        self.SAFE = '░'  # Visual marker for safe channels
        
        # State
        self.grid = []
        self.rocks = set()          # P0: Negative space
        self.safe_channels = set()  # P1: Safe cells
        self.optimal_path = []      # P2: Best path
        
        self.generate_maze()
    
    def generate_maze(self):
        """Create random maze with rocks"""
        self.grid = [[self.EMPTY for _ in range(self.width)] 
                    for _ in range(self.height)]
        
        # Place rocks randomly (P0 negative space)
        self.rocks.clear()
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) == self.start or (x, y) == self.goal:
                    continue
                if random.random() < self.rock_density:
                    self.grid[y][x] = self.ROCK
                    self.rocks.add((x, y))
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        self.grid[sy][sx] = self.START
        self.grid[gy][gx] = self.GOAL
    
    def p0_map_rocks(self):
        """P0: Identify negative space (what NOT to do)"""
        return self.rocks.copy()
    
    def p1_find_safe_channels(self):
        """P1: Find all reachable cells that aren't rocks"""
        visited = set()
        queue = deque([self.start])
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Check all four directions
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    queue.append((nx, ny))
        
        self.safe_channels = visited
        return visited
    
    def p2_optimize_in_channels(self):
        """P2: Find shortest path within safe channels (BFS)"""
        if not self.safe_channels:
            return []  # Must complete P1 first
        
        # BFS for shortest path
        parent = {self.start: None}
        queue = deque([self.start])
        
        while queue:
            x, y = queue.popleft()
            
            if (x, y) == self.goal:
                # Reconstruct path
                path = []
                while (x, y) is not None:
                    path.append((x, y))
                    x, y = parent[(x, y)] if (x, y) in parent else (None, None)
                path.reverse()
                self.optimal_path = path
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) in self.safe_channels and (nx, ny) not in parent):
                    parent[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
        
        return []  # No path found
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2"""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===")
        print(f"Maze: {self.width}x{self.height}, Rocks: {len(self.rocks)}")
        print()
        
        # P0: Map negative space
        rocks = self.p0_map_rocks()
        print("P0: MAPPED NEGATIVE SPACE (rocks)")
        print(f"   Identified {len(rocks)} rock cells to avoid")
        
        # P1: Find safe channels
        safe = self.p1_find_safe_channels()
        print("P1: IDENTIFIED SAFE CHANNELS")
        print(f"   Found {len(safe)} reachable safe cells")
        
        # P2: Optimize within channels
        path = self.p2_optimize_in_channels()
        print("P2: OPTIMIZED WITHIN SAFE CHANNELS")
        print(f"   Path length: {len(path) if path else 'NO PATH'}")
        print()
        
        return len(path) > 0
    
    def display(self, show_p0=False, show_p1=False, show_p2=True):
        """Display maze with optional overlays"""
        display_grid = [row[:] for row in self.grid]
        
        if show_p1:
            # Mark safe channels
            for x, y in self.safe_channels:
                if display_grid[y][x] == self.EMPTY:
                    display_grid[y][x] = self.SAFE
        
        if show_p2 and self.optimal_path:
            # Mark optimal path (skip start and goal)
            for x, y in self.optimal_path[1:-1]:
                if display_grid[y][x] in [self.EMPTY, self.SAFE]:
                    display_grid[y][x] = self.PATH
        
        # Print grid
        for row in display_grid:
            print(' '.join(row))
        print()
    
    def display_step_by_step(self):
        """Show each protocol step visually"""
        print("\n" + "="*50)
        print("INITIAL MAZE")
        print("="*50)
        self.display(show_p0=False, show_p1=False, show_p2=False)
        
        print("="*50)
        print("P0: ROCKS MAPPED (negative space)")
        print("█ = rock, S = start, G = goal")
        print("="*50)
        self.display(show_p0=True, show_p1=False, show_p2=False)
        
        print("="*50)
        print("P1: SAFE CHANNELS IDENTIFIED")
        print("░ = safe reachable cell")
        print("="*50)
        self.display(show_p0=False, show_p1=True, show_p2=False)
        
        if self.optimal_path:
            print("="*50)
            print("P2: OPTIMAL PATH WITHIN SAFE CHANNELS")
            print("○ = optimal path step")
            print("="*50)
            self.display(show_p0=False, show_p1=True, show_p2=True)
            
            print(f"Path found! Length: {len(self.optimal_path)} steps")
            print(f"Start {self.start} → Goal {self.goal}")
        else:
            print("NO PATH FOUND - Goal unreachable within safe channels")

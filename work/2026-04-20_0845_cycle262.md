**Output:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Simulates the three‑phase protocol in a 10x10 ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import sys
from collections import deque

# ----------------------------------------------------------------------
# 1. Maze generation & constants
# ----------------------------------------------------------------------
WIDTH, HEIGHT = 10, 10
START = (0, 0)
GOAL = (WIDTH-1, HEIGHT-1)

ROCK = '█'
EMPTY = '·'
AGENT = 'A'
GOAL_MARK = 'G'
PATH = '○'
SAFE = '░'   # visual marker for safe channels

def generate_maze(rock_density=0.25):
    """Create a random maze with rocks (obstacles)."""
    maze = [[EMPTY for _ in range(WIDTH)] for _ in range(HEIGHT)]
    for y in range(HEIGHT):
        for x in range(WIDTH):
            if (x, y) == START or (x, y) == GOAL:
                continue
            if random.random() < rock_density:
                maze[y][x] = ROCK
    return maze

def print_maze(maze, title="", highlight=None):
    """Print the maze with optional highlighted cells."""
    print(f"\n{title}")
    for y in range(HEIGHT):
        row = []
        for x in range(WIDTH):
            if highlight and (x, y) in highlight:
                row.append(SAFE)
            else:
                row.append(maze[y][x])
        print(' '.join(row))
    print()

# ----------------------------------------------------------------------
# 2. Deadband Protocol implementation
# ----------------------------------------------------------------------
def p0_map_rocks(maze):
    """P0: Identify negative space — all rock cells."""
    rocks = []
    for y in range(HEIGHT):
        for x in range(WIDTH):
            if maze[y][x] == ROCK:
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, rocks):
    """P1: Find all cells that are NOT rocks and are reachable from start."""
    safe = set()
    queue = deque([START])
    visited = set([START])
    
    while queue:
        x, y = queue.popleft()
        safe.add((x, y))
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < WIDTH and 0 <= ny < HEIGHT:
                if (nx, ny) not in visited and maze[ny][nx] != ROCK:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    return safe

def p2_optimize_in_channel(maze, safe_cells):
    """P2: BFS to find shortest path from start to goal staying in safe cells."""
    if GOAL not in safe_cells:
        return None  # no safe channel reaches goal
    
    queue = deque([(START, [])])
    visited = set([START])
    
    while queue:
        (x, y), path = queue.popleft()
        if (x, y) == GOAL:
            return path + [(x, y)]
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < WIDTH and 0 <= ny < HEIGHT:
                if (nx, ny) not in visited and (nx, ny) in safe_cells:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(x, y)]))
    return None

# ----------------------------------------------------------------------
# 3. Visualization helpers
# ----------------------------------------------------------------------
def visualize_phase(maze, phase, data):
    """Show maze with phase‑specific overlay."""
    if phase == "P0":
        print_maze(maze, title="[P0] Initial Maze — Rocks are '█'")
        print(f"  → Identified {len(data)} rock cells (negative space)")
    elif phase == "P1":
        print_maze(maze, title="[P1] Safe Channels — Safe cells marked '░'", highlight=data)
        print(f"  → Mapped {len(data)} reachable safe cells")
    elif phase == "P2":
        if data:
            marked = maze.copy()
            for (x, y) in data:
                marked[y][x] = PATH
            marked[START[1]][START[0]] = AGENT
            marked[GOAL[1]][GOAL[0]] = GOAL_MARK
            print_maze(marked, title="[P2] Optimal Path — Path cells '○', Agent 'A', Goal 'G'")
            print(f"  → Found optimal path of length {len(data)} steps")
        else:
            print("  → No safe path to goal (constrained failure)")

# ----------------------------------------------------------------------
# 4. Main simulation
# ----------------------------------------------------------------------
def run_deadband_simulation(seed=None):
    """Run one complete Deadband Protocol demonstration."""
    if seed is not None:
        random.seed(seed)
    
    print("="*60)
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("P0 → P1 → P2 in a 10x10 maze")
    print("="*60)
    
    # Generate environment
    maze = generate_maze(rock_density=0.25)
    maze[START[1]][START[0]] = AGENT
    maze[GOAL[1]][GOAL[0]] = GOAL_MARK
    
    # P0: Map negative space
    rocks = p0_map_rocks(maze)
    visualize_phase(maze, "P0", rocks)
    
    # P1: Identify safe channels
    safe = p1_find_safe_channels(maze, rocks)
    visualize_phase(maze, "P1", safe)
    
    # P2: Optimize within safe channels
    path = p2_optimize_in_channel(maze, safe)
    visualize_phase(maze, "P2", path)
    
    # Summary
    print("\n" + "="*60)
    print("SIMULATION SUMMARY")
    print(f"  Rocks (P0): {len(rocks)} cells")
    print(f"  Safe channel cells (P1): {len(safe)} cells")
    if path:
        print(f"  Optimal path length (P2): {len(path)-1} steps")
        print("  RESULT: SUCCESS — Goal reached via deadband protocol")
    else:
        print("  RESULT: FAILURE — No safe channel reaches goal")
    print("="*60)
    
    return path is not None

# ----------------------------------------------------------------------
# 5. Batch run to mirror fleet data
# ----------------------------------------------------------------------
def batch_run(n_runs=50):
    """Run multiple simulations to collect statistics."""
    print("\n" + "="*60)
    print(f"BATCH RUN — {n_runs} iterations")
    print("="*60)
    
    successes = 0
    for i in range(n_runs):
        print(f"\nRun {i+1}/{n_runs}:")
        success = run_deadband_simulation(seed=i+1000)  # deterministic seeds
        if success:
            successes += 1
    
    print("\n" + "="*60)
    print("BATCH RESULTS")
    print(f"  Successes: {successes}/{n_runs}")
    print(f"  Success rate: {100*successes/n_runs:.1f}%")
    print("="*60)

# ----------------------------------------------------------------------
if __name__ == "__main__":
    # Single demo with a clear seed for reproducibility
    if len(sys.argv) > 1 and sys.argv[1] == "--batch":
        batch_run(10)  # smaller batch for demo
    else:
        run_deadband_simulation(seed=42)  # classic answer
```

**Execution notes:**
1. Run `python deadband_maze_demo.py` for a single clear demonstration.
2. Run `python deadband_maze_demo.py --batch` for 10‑run statistics.
3. The script mirrors the fleet's constraint‑sim data: P0→P1→P2

#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — Tinker (Cycle 355)
A simple ASCII simulation of P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# Simulation constants from fleet data (optimal path length = 21, rocks = 64)
GRID_SIZE = 10
START = (0, 0)
GOAL = (9, 9)
NUM_ROCKS = 15  # scaled down for 10x10 grid

def generate_grid() -> List[List[str]]:
    """Create a grid with rocks, start, and goal."""
    grid = [['.' for _ in range(GRID_SIZE)] for _ in range(GRID_SIZE)]
    grid[START[0]][START[1]] = 'S'
    grid[GOAL[0]][GOAL[1]] = 'G'
    
    rocks = set()
    while len(rocks) < NUM_ROCKS:
        r, c = random.randint(0, GRID_SIZE-1), random.randint(0, GRID_SIZE-1)
        if (r, c) not in (START, GOAL) and (r, c) not in rocks:
            rocks.add((r, c))
            grid[r][c] = '#'
    return grid, rocks

def print_grid(grid: List[List[str]], title: str):
    """Print the grid with a title."""
    print(f"\n=== {title} ===")
    for row in grid:
        print(' '.join(row))
    print()

def p0_map_rocks(grid: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[str]]:
    """P0: Identify negative space (rocks)."""
    p0_grid = [row[:] for row in grid]
    for r, c in rocks:
        p0_grid[r][c] = 'X'  # marked rock
    return p0_grid

def p1_find_safe_channels(grid: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[str]]:
    """P1: Find safe channels (deadbands) using BFS from start, avoiding rocks."""
    from collections import deque
    
    p1_grid = [row[:] for row in grid]
    visited = [[False]*GRID_SIZE for _ in range(GRID_SIZE)]
    queue = deque([START])
    visited[START[0]][START[1]] = True
    
    # BFS to find all reachable cells not blocked by rocks
    while queue:
        r, c = queue.popleft()
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r+dr, c+dc
            if 0 <= nr < GRID_SIZE and 0 <= nc < GRID_SIZE:
                if not visited[nr][nc] and (nr, nc) not in rocks:
                    visited[nr][nc] = True
                    queue.append((nr, nc))
    
    # Mark safe channels
    for r in range(GRID_SIZE):
        for c in range(GRID_SIZE):
            if visited[r][c] and p1_grid[r][c] == '.':
                p1_grid[r][c] = '~'  # safe water
    return p1_grid, visited

def p2_optimize_path(grid: List[List[str]], visited: List[List[bool]], rocks: Set[Tuple[int, int]]) -> List[List[str]]:
    """P2: Find shortest path within safe channels (deadbands)."""
    from collections import deque
    
    p2_grid = [row[:] for row in grid]
    parent = [[None]*GRID_SIZE for _ in range(GRID_SIZE)]
    queue = deque([START])
    
    # BFS for shortest path
    while queue:
        r, c = queue.popleft()
        if (r, c) == GOAL:
            break
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r+dr, c+dc
            if 0 <= nr < GRID_SIZE and 0 <= nc < GRID_SIZE:
                if visited[nr][nc] and (nr, nc) not in rocks and parent[nr][nc] is None:
                    parent[nr][nc] = (r, c)
                    queue.append((nr, nc))
    
    # Reconstruct and mark path
    if parent[GOAL[0]][GOAL[1]] is not None:
        path = []
        curr = GOAL
        while curr != START:
            path.append(curr)
            curr = parent[curr[0]][curr[1]]
        path.append(START)
        for r, c in path:
            if p2_grid[r][c] not in ('S', 'G'):
                p2_grid[r][c] = '*'
    return p2_grid

def run_simulation():
    """Run one full demonstration of the Deadband Protocol."""
    print("DEADBAND PROTOCOL DEMO — Tinker (Cocapn Fleet)")
    print("=" * 50)
    
    # Generate environment
    grid, rocks = generate_grid()
    print_grid(grid, "Initial Grid (S=start, G=goal, #=rock)")
    
    # P0: Map negative space
    p0_grid = p0_map_rocks(grid, rocks)
    print_grid(p0_grid, "P0: Map Rocks (X=rock, avoid these)")
    
    # P1: Find safe channels
    p1_grid, visited = p1_find_safe_channels(grid, rocks)
    print_grid(p1_grid, "P1: Safe Channels (~=safe water)")
    
    # P2: Optimize within channels
    p2_grid = p2_optimize_path(grid, visited, rocks)
    print_grid(p2_grid, "P2: Optimal Path (*=path, within safe channels)")
    
    # Summary
    print("PROTOCOL SUMMARY:")
    print(f"  P0 Rocks identified: {len(rocks)}")
    safe_cells = sum(1 for r in range(GRID_SIZE) for c in range(GRID_SIZE) if visited[r][c])
    print(f"  P1 Safe cells found: {safe_cells}")
    path_len = sum(1 for row in p2_grid for cell in row if cell == '*')
    print(f"  P2 Path length: {path_len} steps")
    print("\n" + "=" * 50)
    print("Reference fleet data (20x20 maze, 50 runs):")
    print("  Unconstrained (P2-only): 0/50 success")
    print("  Deadband (P0→P1→P2): 50/50 success, avg steps 21.0")

if __name__ == "__main__":
    run_simulation()
```

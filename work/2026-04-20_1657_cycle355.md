#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

A visual ASCII simulation of the Deadband Protocol (P0→P1→P2).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json.

Doctrine:
    P0: Map negative space (rocks) — what NOT to do.
    P1: Identify safe channels (deadbands) — where you CAN be.
    P2: Optimize within safe channels — best path.

This demo shows why skipping to P2 (greedy optimization) fails.
"""

import random
from typing import List, Tuple, Set

# ============================================================================
# MAZE CONFIGURATION (from constraint_sim_final.json context)
# ============================================================================
MAZE_WIDTH = 20
MAZE_HEIGHT = 20
START = (0, 0)
GOAL = (MAZE_WIDTH - 1, MAZE_HEIGHT - 1)

# Simulation parameters from the data
ROCK_COUNT = 64          # "rocks": 64
OPTIMAL_PATH_LEN = 21    # "optimal": 21

# ============================================================================
# MAZE GENERATION
# ============================================================================
def generate_maze(seed: int = 42) -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """
    Generates a 20x20 maze with:
        '.' = empty cell
        '#' = rock (negative space)
        'S' = start
        'G' = goal
    Returns the maze grid and a set of rock coordinates.
    """
    random.seed(seed)
    grid = [['.' for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    rocks = set()

    # Place rocks (negative space)
    while len(rocks) < ROCK_COUNT:
        x = random.randint(0, MAZE_WIDTH - 1)
        y = random.randint(0, MAZE_HEIGHT - 1)
        if (x, y) != START and (x, y) != GOAL:
            rocks.add((x, y))
            grid[y][x] = '#'

    # Mark start and goal
    grid[START[1]][START[0]] = 'S'
    grid[GOAL[1]][GOAL[0]] = 'G'

    return grid, rocks

# ============================================================================
# DEADBAND PROTOCOL IMPLEMENTATION
# ============================================================================
def p0_map_negative_space(grid: List[List[str]]) -> Set[Tuple[int, int]]:
    """
    P0: Identify all rocks (negative space).
    Returns set of coordinates to avoid.
    """
    rocks = set()
    for y in range(MAZE_HEIGHT):
        for x in range(MAZE_WIDTH):
            if grid[y][x] == '#':
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(grid: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
    """
    P1: Find all safe paths from start to goal that avoid rocks.
    Uses a simple BFS to enumerate all possible safe paths.
    Returns a list of paths (each path is a list of coordinates).
    """
    from collections import deque

    # Directions: right, down, left, up (prioritizes toward goal)
    dirs = [(1, 0), (0, 1), (-1, 0), (0, -1)]
    queue = deque([(START, [START])])
    safe_paths = []

    while queue:
        (x, y), path = queue.popleft()

        if (x, y) == GOAL:
            safe_paths.append(path)
            continue

        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                if (nx, ny) not in rocks and (nx, ny) not in path:
                    queue.append(((nx, ny), path + [(nx, ny)]))

    return safe_paths

def p2_optimize_within_channel(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """
    P2: From the safe channels, select the shortest path.
    Optimization is constrained to pre‑identified safe paths.
    """
    if not safe_paths:
        return []
    return min(safe_paths, key=len)

# GREEDY (P2‑ONLY) AGENT FOR COMPARISON
# ============================================================================
def greedy_p2_only(grid: List[List[str]]) -> List[Tuple[int, int]]:
    """
    Violates the Deadband Protocol: tries to optimize (P2) without
    first mapping rocks (P0) or safe channels (P1).
    Uses Manhattan‑distance heuristic, ignores rocks.
    """
    path = [START]
    visited = set([START])
    x, y = START

    while (x, y) != GOAL:
        # Greedy move: step toward goal if possible, else any open cell
        candidates = []
        for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                if (nx, ny) not in visited:
                    candidates.append((nx, ny))

        if not candidates:
            break  # stuck

        # Choose candidate closest to goal (Manhattan distance)
        next_pos = min(candidates, key=lambda p: abs(p[0]-GOAL[0]) + abs(p[1]-GOAL[1]))
        path.append(next_pos)
        visited.add(next_pos)
        x, y = next_pos

        # If we hit a rock, path fails
        if grid[y][x] == '#':
            break

    return path

    """Print the maze, optionally highlighting a path with '*'."""
    for y in range(MAZE_HEIGHT):
        row = ''

            if path and (x, y) in path and cell not in ('S', 'G'):
                row += '*'
            else:
                row += cell
            row += ' '
        print(row)

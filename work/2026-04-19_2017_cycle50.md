# Constants from fleet simulation
MAZE_SIZE = 20
START = (0, 0)

ROCK_COUNT = 64  # From constraint_sim_final.json: "rocks": 64

# Maze symbols
EMPTY = '·'
ROCK = '█'
AGENT = 'A'

class Maze:
    """20×20 maze with random rocks."""
    
    def __init__(self, seed=None):
        if seed is not None:
            random.seed(seed)
        
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        
        # Place rocks randomly (but not on start or goal)
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            pos = (x, y)
            if pos != START and pos != GOAL:
                self.rocks.add(pos)
                self.grid[x][y] = ROCK
    
    def is_rock(self, pos: Tuple[int, int]) -> bool:
        return pos in self.rocks
    
    def in_bounds(self, pos: Tuple[int, int]) -> bool:
        x, y = pos
        return 0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE
    
    def get_neighbors(self, pos: Tuple[int, int]) -> List[Tuple[int, int]]:
        x, y = pos
        candidates = [(x+1, y), (x-1, y), (x, y+1), (x, y-1)]
        return [p for p in candidates if self.in_bounds(p)]
    
    def display(self, path=None, safe_channels=None):
        """Print ASCII maze with optional path and safe channel highlighting."""
        display_grid = [row[:] for row in self.grid]
        
        # Mark safe channels if provided
        if safe_channels:
            for (x, y) in safe_channels:
                if display_grid[x][y] == EMPTY:
                    display_grid[x][y] = SAFE
        
        # Mark path if provided
        if path:
            for (x, y) in path:
                if (x, y) != START and (x, y) != GOAL:
                    display_grid[x][y] = PATH
        
        # Mark start and goal

        display_grid[sx][sy] = AGENT

        # Legend
        print(f"Legend: {AGENT}=Start, {GOAL_MARKER}=Goal, {ROCK}=Rock, {PATH}=Path, {SAFE}=Safe Channel")

    """P2-only agent: greedy optimization without rock avoidance."""
    
    def find_path(self, maze: Maze) -> List[Tuple[int, int]]:
        # Simple BFS ignoring rocks (will crash into them)
        queue = deque([(START, [START])])
        visited = set([START])
        
        while queue:
            current, path = queue.popleft()
            
            if current == GOAL:
                self.path = path
                self.success = True
                self.steps = len(path) - 1  # Steps taken
                return path
            
            for neighbor in maze.get_neighbors(current):
                if neighbor not in visited:
                    visited.add(neighbor)
                    queue.append((neighbor, path + [neighbor]))
        
        self.success = False
        return []

        stack = [START]

            current = stack.pop()

                continue
            
            safe.add(current)
            
            for neighbor in maze.get_neighbors(current):
                if neighbor not in safe and neighbor not in rocks:
                    stack.append(neighbor)
        
        return safe
    
    def _optimize_in_channels(self, maze: Maze, safe_channels: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:

        queue = deque([(START, [START])])

                return path

                if neighbor in safe_channels and neighbor not in visited:
                    visited.add(neighbor)
                    queue.append((neighbor, path + [neighbor]))
        
        return []  # No path found within safe channels

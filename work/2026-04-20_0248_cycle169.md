#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

Demonstrates P0→P1→P2 strict priority:
P0: Map negative space (rocks '#')
P1: Identify safe channels (paths avoiding rocks)
P2: Optimize within channel (shortest safe path)

Based on fleet-knowledge/docs/DEADBAND-PROTOCOL.md
Simulation data: fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque

def generate_maze(width=10, height=10, rock_density=0.3):
    """Generate a random maze with start (S), goal (G), rocks (#), open (.)."""
    grid = [['.' for _ in range(width)] for _ in range(height)]
    # Place rocks
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                grid[y][x] = '#'
    # Ensure start and goal are open
    start = (0, 0)
    goal = (width-1, height-1)
    grid[start[1]][start[0]] = 'S'
    grid[goal[1]][goal[0]] = 'G'
    return grid, start, goal

def print_maze(grid, path=None):
    """Print the maze, optionally overlay path with '*'."""
    for y, row in enumerate(grid):
        line = []
        for x, cell in enumerate(row):
            if path and (x, y) in path:
                line.append('*')
            else:
                line.append(cell)
        print(''.join(line))

def unconstrained_agent(grid, start, goal):
    """P2-only greedy agent: move toward goal, ignoring rocks."""
    x, y = start
    gx, gy = goal
    path = [start]
    while (x, y) != goal:
        # Greedy: move in direction that reduces Manhattan distance
        if x < gx and grid[y][x+1] != '#':
            x += 1
        elif x > gx and grid[y][x-1] != '#':
            x -= 1
        elif y < gy and grid[y+1][x] != '#':
            y += 1
        elif y > gy and grid[y-1][x] != '#':
            y -= 1
        else:
            # Stuck
            return None
        path.append((x, y))
        if len(path) > 100:  # safety break
            return None
    return path

def deadband_agent(grid, start, goal):
    """P0+P1+P2 agent: map rocks, find safe channels, then shortest path."""
    # P0: Map negative space (rocks)
    rocks = []
    for y, row in enumerate(grid):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.append((x, y))
    # P1: Identify safe channels (BFS that avoids rocks)
    width, height = len(grid[0]), len(grid)
    queue = deque([start])
    visited = {start: None}
    while queue:
        x, y = queue.popleft()
        if (x, y) == goal:
            break
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if grid[ny][nx] != '#' and (nx, ny) not in visited:
                    visited[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    # P2: Reconstruct shortest safe path
    if goal not in visited:
        return None
    path = []
    node = goal
    while node is not None:
        path.append(node)
        node = visited[node]
    path.reverse()
    return path

def run_simulation(runs=50):
    """Run multiple maze trials, tally results."""
    results = {
        'unconstrained_success': 0,
        'deadband_success': 0,
        'unconstrained_lengths': [],
        'deadband_lengths': []
    }
    for i in range(runs):
        grid, start, goal = generate_maze()
        # Unconstrained (P2-only)
        path_u = unconstrained_agent(grid, start, goal)
        if path_u:
            results['unconstrained_success'] += 1
            results['unconstrained_lengths'].append(len(path_u)-1)  # steps
        # Deadband (P0+P1+P2)
        path_d = deadband_agent(grid, start, goal)
        if path_d:
            results['deadband_success'] += 1
            results['deadband_lengths'].append(len(path_d)-1)
        # Print first run for illustration
        if i == 0:
            print("=== First Run (Example Maze) ===")
            print("Maze:")
            print_maze(grid)
            print("\nUnconstrained agent path (P2-only):")
            if path_u:
                print_maze(grid, path_u)
                print(f"Steps: {len(path_u)-1}")
            else:
                print("Failed (hit rock or stuck)")
            print("\nDeadband agent path (P0→P1→P2):")
            if path_d:
                print_maze(grid, path_d)
                print(f"Steps: {len(path_d)-1}")
            print()
    # Summary
    print("=== Simulation Summary ({} runs) ===".format(runs))
    print("Unconstrained (P2-only): {}/{} success, avg steps: {:.1f}".format(
        results['unconstrained_success'], runs,
        sum(results['unconstrained_lengths'])/len(results['unconstrained_lengths']) if results['unconstrained_lengths'] else 0
    ))
    print("Deadband (P0→P1→P2): {}/{} success, avg steps: {:.1f}".format(
        results['deadband_success'], runs,
        sum(results['deadband_lengths'])/len(results['deadband_lengths']) if results['deadband_lengths'] else 0
    ))
    return results

if __name__ == "__main__":
    print("Deadband Protocol Maze Demo — Tinker")
    print("P0: Map rocks, P1: Safe channels, P2: Shortest safe path\n")
    run_simulation(runs=20)  # Smaller sample for quick demo
```

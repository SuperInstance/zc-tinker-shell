#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — Tinker
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize within channels).
ASCII visualization of each step.
"""

import heapq
import random

# ====================
# 1. ENVIRONMENT
# ====================
def generate_maze(width=10, height=10, rock_density=0.2):
    """Generate a grid with start (S), goal (G), and rocks (#)."""
    grid = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks randomly
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                grid[y][x] = '#'
    
    # Ensure start and goal are clear
    start = (0, 0)
    goal = (width-1, height-1)
    grid[start[1]][start[0]] = 'S'
    grid[goal[1]][goal[0]] = 'G'
    
    return grid, start, goal

def print_grid(grid, title="Grid"):
    """Print ASCII grid."""
    print(f"\n=== {title} ===")
    for row in grid:
        print(' '.join(row))
    print()

# ====================
# 2. DEADBAND PROTOCOL
# ====================
def p0_map_rocks(grid):
    """P0: Identify negative space (rocks)."""
    rocks = []
    for y, row in enumerate(grid):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(grid, start, goal, rocks):
    """P1: Find all paths that avoid rocks (safe channels)."""
    width, height = len(grid[0]), len(grid)
    
    # BFS to find all reachable positions avoiding rocks
    queue = [start]
    visited = set([start])
    safe_cells = set([start])
    
    while queue:
        x, y = queue.pop(0)
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited.add((nx, ny))
                    safe_cells.add((nx, ny))
                    queue.append((nx, ny))
    
    return safe_cells

def p2_optimize_in_channel(grid, start, goal, safe_cells):
    """P2: Find shortest path within safe channels (A*)."""
    width, height = len(grid[0]), len(grid)
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            path.reverse()
            return path
        
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            neighbor = (x + dx, y + dy)
            if 0 <= neighbor[0] < width and 0 <= neighbor[1] < height:
                if neighbor not in safe_cells:
                    continue  # Not in safe channel
                
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return []  # No path found

# ====================
# 3. VISUALIZATION
# ====================
def visualize_step(grid, step_name, highlights=None, symbol='*'):
    """Create ASCII visualization for each protocol step."""
    viz = [row[:] for row in grid]
    
    if highlights:
        for x, y in highlights:
            if viz[y][x] not in ['S', 'G', '#']:
                viz[y][x] = symbol
    
    print_grid(viz, title=step_name)

def run_demo():
    """Run complete Deadband Protocol demo."""
    print("=" * 50)
    print("DEADBAND PROTOCOL DEMO — Tinker")
    print("P0: Map rocks | P1: Find safe channels | P2: Optimize within channels")
    print("=" * 50)
    
    # Generate environment
    grid, start, goal = generate_maze(width=10, height=10, rock_density=0.2)
    print_grid(grid, "Initial Environment")
    
    # P0: Map negative space (rocks)
    rocks = p0_map_rocks(grid)
    visualize_step(grid, "P0: Rocks (Negative Space)", rocks, symbol='#')
    print(f"Identified {len(rocks)} rocks (avoid these)")
    
    # P1: Find safe channels
    safe_cells = p1_find_safe_channels(grid, start, goal, rocks)
    visualize_step(grid, "P1: Safe Channels (Deadbands)", safe_cells, symbol='~')
    print(f"Found {len(safe_cells)} safe cells (operate only here)")
    
    # P2: Optimize within channels
    path = p2_optimize_in_channel(grid, start, goal, safe_cells)
    if path:
        visualize_step(grid, "P2: Optimal Path in Channels", path, symbol='*')
        print(f"Optimal path length: {len(path)-1} steps")
        print(f"Path: {path}")
    else:
        print("No safe path to goal exists (constrained optimization)")
    
    # Summary
    print("\n" + "=" * 50)
    print("PROTOCOL SUMMARY:")
    print(f"P0 Rocks: {len(rocks)} obstacles mapped")
    print(f"P1 Safe Cells: {len(safe_cells)} channel positions")
    print(f"P2 Path Found: {'Yes' if path else 'No'}")
    print("=" * 50)

# ====================
# 4. MAIN
# ====================
if __name__ == "__main__":
    random.seed(42)  # Deterministic for demo
    run_demo()

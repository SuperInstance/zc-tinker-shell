**Status:** Based on `constraint_sim_final.json` (20×20 maze, 64 rocks, optimal path length 21). Implementing a simulation that shows P0 (map rocks), P1 (find safe channels), P2 (optimize path) with ASCII visualization.

**Output:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map rocks (negative space)
P1: Find safe channels (deadbands)
P2: Optimize path within channels
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_count=64):
        self.width = width
        self.height = height
        self.rock_count = rock_count
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Initialize empty grid
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.rocks = set()
        self.safe_channels = []
        self.optimal_path = []
        
    def p0_map_rocks(self):
        """P0: Identify negative space (what NOT to do)"""
        # Place rocks randomly, avoiding start and goal
        possible = [(x, y) for x in range(self.width) 
                          for y in range(self.height)
                          if (x, y) not in [self.start, self.goal]]
        
        self.rocks = set(random.sample(possible, self.rock_count))
        for x, y in self.rocks:
            self.grid[y][x] = '█'
        
        print("P0: Mapped negative space (rocks)")
        print(f"   - Rocks: {len(self.rocks)}")
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find safe channels (where you CAN be)"""
        # BFS to find all reachable positions avoiding rocks
        visited = set()
        queue = deque([self.start])
        safe_positions = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            safe_positions.add((x, y))
            
            # Check neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    queue.append((nx, ny))
        
        # Identify connected components as channels
        components = []
        remaining = safe_positions.copy()
        
        while remaining:
            start = next(iter(remaining))
            component = set()
            stack = [start]
            
            while stack:
                x, y = stack.pop()
                if (x, y) not in component:
                    component.add((x, y))
                    remaining.discard((x, y))
                    
                    for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                        nx, ny = x + dx, y + dy
                        if (nx, ny) in safe_positions and (nx, ny) not in component:
                            stack.append((nx, ny))
            
            components.append(component)
        
        self.safe_channels = [comp for comp in components if self.goal in comp]
        print(f"\nP1: Found safe channels")
        print(f"   - Safe positions: {len(safe_positions)}")
        print(f"   - Channels containing goal: {len(self.safe_channels)}")
        return self.safe_channels
    
    def p2_optimize_path(self):
        """P2: Optimize within safe channels (best path)"""
        if not self.safe_channels:
            print("P2: No safe channels found (cannot proceed)")
            return []
        
        # Find shortest path within safe channel using BFS
        channel = next(iter(self.safe_channels))
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                self.optimal_path = path
                print(f"\nP2: Optimized path found")
                print(f"   - Path length: {len(path)}")
                print(f"   - Optimal from data: 21")
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if ((nx, ny) in channel and (nx, ny) not in visited):
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        return []
    
    def display(self, show_path=False):
        """Display the maze with ASCII art"""
        display_grid = [row[:] for row in self.grid]
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        display_grid[sy][sx] = 'S'
        display_grid[gy][gx] = 'G'
        
        # Mark path if requested
        if show_path and self.optimal_path:
            for x, y in self.optimal_path[1:-1]:  # Skip start and goal
                if display_grid[y][x] == '.':
                    display_grid[y][x] = '·'
        
        print("\n" + "=" * (self.width * 2 + 3))
        print("MAZE (20×20):")
        print("  S = Start, G = Goal, █ = Rock, · = Path, . = Empty")
        print("=" * (self.width * 2 + 3))
        
        for y in range(self.height):
            row = ' '.join(display_grid[y])
            print(f"{y:2d} {row}")
        
        print("=" * (self.width * 2 + 3))
    
    def run_protocol(self):
        """Execute full Deadband Protocol"""
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("=" * 40)
        
        # P0: Map negative space
        self.p0_map_rocks()
        self.display(show_path=False)
        
        # P1: Find safe channels
        channels = self.p1_find_safe_channels()
        if not channels:
            print("FAILURE: Goal unreachable (no safe channels)")
            return False
        
        # P2: Optimize within channel
        path = self.p2_optimize_path()
        if not path:
            print("FAILURE: No path found within safe channel")
            return False
        
        self.display(show_path=True)
        return True

def main():
    """Run the demonstration"""
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print(f"Parameters: 20×20 maze, 64 rocks, optimal path: 21 steps")
    print("-" * 50)
    
    # Run multiple trials to show statistics
    successes = 0
    trials = 5
    
    for i in range(trials):
        print(f"\n\nTRIAL {i+1}/{trials}")
        print("-" * 30)
        
        maze = DeadbandMaze(width=20, height=20, rock_count=64)
        if maze.run_protocol():
            successes += 1
    
    print(f"\n\nSUMMARY")
    print("=" * 30)
    print(f"Trials: {trials}")
    print(f"Successes: {successes}")
    print(f"Success rate: {successes/trials*100:.1f}%")
    print(f"\nNote: Constrained success in data: 0/50")
    print(f"      Deadband success in data: 50/50")

if __name__ == "__main__":
    main()
```

**Key Features:**
1. **P0 Implementation**: Randomly places 64 rocks (█) avoiding start (S) and goal (G)
2. **P1 Implementation**: Uses BFS to find all reachable positions, identifies connected components as "safe channels"
3. **P2 Implementation**: Finds shortest path within safe channel using BFS
4. **ASCII Visualization**: Shows maze with rocks, path markers (·), start, and goal
5. **Statistics**: Runs multiple trials to demonstrate success rate

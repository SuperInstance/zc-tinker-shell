#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    """
    A simple 10x10 maze demonstrating the Deadband Protocol.
    Rocks (P0) are obstacles.
    Safe channels (P1) are clear paths.
    Optimal path (P2) is shortest route within safe channels.
    """
    
    def __init__(self, width=10, height=10, rock_density=0.2):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (P0 negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        for y in range(height):
            for x in range(width):
                if (x, y) != self.start and (x, y) != self.goal:
                    if random.random() < rock_density:
                        self.rocks.add((x, y))
        
        # Ensure at least one path exists
        self._ensure_path()
    
    def _ensure_path(self):
        """Guarantee at least one safe channel exists."""
        # Simple straight path along bottom then right
        for x in range(self.width):
            pos = (x, self.height-1)
            if pos != self.goal:
                self.rocks.discard(pos)
        for y in range(self.height):
            pos = (self.width-1, y)
            if pos != self.goal:
                self.rocks.discard(pos)
    
    def is_rock(self, x: int, y: int) -> bool:
        """P0: Identify negative space."""
        return (x, y) in self.rocks
    
    def find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Map all safe channels using BFS."""
        from collections import deque
        
        channels = []
        visited = set()
        
        def bfs(start):
            queue = deque([(start, [start])])
            local_visited = {start}
            
            while queue:
                (x, y), path = queue.popleft()
                
                if (x, y) == self.goal:
                    channels.append(path)
                    continue
                
                # Four possible moves
                for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
                    nx, ny = x + dx, y + dy
                    if 0 <= nx < self.width and 0 <= ny < self.height:
                        if not self.is_rock(nx, ny) and (nx, ny) not in local_visited:
                            local_visited.add((nx, ny))
                            queue.append(((nx, ny), path + [(nx, ny)]))
            
            visited.update(local_visited)
        
        bfs(self.start)
        return channels
    
    def find_optimal_path(self, channels: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """P2: Optimize within safe channels."""
        if not channels:
            return []
        return min(channels, key=len)
    
    def greedy_path(self) -> List[Tuple[int, int]]:
        """P2-only (violation): Greedy optimization without P0+P1."""
        path = [self.start]
        x, y = self.start
        
        while (x, y) != self.goal:
            # Prefer moves toward goal
            moves = []
            if x < self.goal[0]:
                moves.append((1, 0))
            if y < self.goal[1]:
                moves.append((0, 1))
            if x > self.goal[0]:
                moves.append((-1, 0))
            if y > self.goal[1]:
                moves.append((0, -1))
            
            # Shuffle to simulate different greedy attempts
            random.shuffle(moves)
            
            moved = False
            for dx, dy in moves:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in path:  # Simple loop prevention
                        x, y = nx, ny
                        path.append((x, y))
                        moved = True
                        break
            
            if not moved or len(path) > self.width * self.height * 2:
                # Stuck or looping
                return []
        
        return path
    
    def visualize(self, path: List[Tuple[int, int]] = None, 
                  show_channels: bool = False,
                  channels: List[List[Tuple[int, int]]] = None):
        """ASCII visualization of maze state."""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif self.is_rock(x, y):
                    row.append('█')
                elif path and (x, y) in path:
                    row.append('·')
                elif show_channels and channels:
                    in_channel = any((x, y) in channel for channel in channels)
                    row.append('░' if in_channel else ' ')
                else:
                    row.append(' ')
            grid.append(''.join(row))
        return '\n'.join(grid)

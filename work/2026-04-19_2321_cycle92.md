**Approach (P0→P1→P2):**
1. **P0 (Map negative space):** Generate a maze with rocks ('#') and open cells ('.'). Identify all rock positions.
2. **P1 (Find safe channels):** Use BFS from start to goal, avoiding rocks, to find all reachable paths. These are the safe channels.
3. **P2 (Optimize within channel):** From the safe channels, select the shortest path (minimum cost).

I'll compare against a P2-only agent that greedily moves toward the goal ignoring rocks (hits them and fails).

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 vs P2-only in a constrained navigation problem.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque

class Maze:
    def __init__(self, width=20, height=20, rock_density=0.15):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.rocks = set()
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self._generate_rocks(rock_density)
        
    def _generate_rocks(self, density):
        """Place rocks randomly, ensuring start and goal are clear."""
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) in (self.start, self.goal):
                    continue
                if random.random() < density:
                    self.grid[y][x] = '#'
                    self.rocks.add((x, y))
    
    def display(self, path=None):
        """Print ASCII maze with optional path."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif path and (x, y) in path:
                    row.append('*')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
        print()
    
    def get_neighbors(self, x, y, avoid_rocks=True):
        """Get valid neighbor coordinates."""
        neighbors = []
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < self.width and 0 <= ny < self.height:
                if avoid_rocks and (nx, ny) in self.rocks:
                    continue
                neighbors.append((nx, ny))
        return neighbors

class DeadbandAgent:
    """Implements P0→P1→P2 protocol."""
    
    def __init__(self, maze):
        self.maze = maze
        self.safe_channels = []  # P1 result
        self.optimal_path = None  # P2 result
        
    def run(self):
        """Execute full deadband protocol."""
        # P0 is already done by maze (rocks mapped)
        
        # P1: Find all safe channels (BFS for all reachable paths)
        self._find_safe_channels()
        
        # P2: Optimize within channels (shortest safe path)
        self._optimize_within_channels()
        
        return self.optimal_path
    
    def _find_safe_channels(self):
        """BFS to find all reachable cells (safe channels)."""
        visited = set()
        queue = deque([self.maze.start])
        visited.add(self.maze.start)
        
        while queue:
            x, y = queue.popleft()
            for nx, ny in self.maze.get_neighbors(x, y, avoid_rocks=True):
                if (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
        
        # Safe channels are all visited cells (excluding rocks)
        self.safe_channels = visited
    
    def _optimize_within_channels(self):
        """Find shortest path from start to goal within safe channels."""
        # BFS for shortest path
        queue = deque([(self.maze.start, [self.maze.start])])
        visited = set([self.maze.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.maze.goal:
                self.optimal_path = set(path)
                return
            
            for nx, ny in self.maze.get_neighbors(x, y, avoid_rocks=True):
                if (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.optimal_path = None  # No path found

class GreedyAgent:
    """P2-only agent (optimizes without mapping)."""
    
    def __init__(self, maze):
        self.maze = maze
        self.path = []
        self.collisions = 0
    
    def run(self):
        """Greedy move toward goal, ignoring rocks."""
        x, y = self.maze.start
        self.path = [(x, y)]
        visited = set([(x, y)])
        
        steps = 0
        max_steps = self.maze.width * self.maze.height * 2  # Prevent infinite loop
        
        while (x, y) != self.maze.goal and steps < max_steps:
            # Greedy: move in direction that reduces Manhattan distance
            best_move = None
            best_dist = float('inf')
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.maze.width and 0 <= ny < self.maze.height:
                    dist = abs(nx - self.maze.goal[0]) + abs(ny - self.maze.goal[1])
                    if dist < best_dist:
                        best_dist = dist
                        best_move = (nx, ny)
            
            if best_move is None:
                break
            
            x, y = best_move
            
            # Check for collision
            if (x, y) in self.maze.rocks:
                self.collisions += 1
                # Try to recover (random move)
                neighbors = self.maze.get_neighbors(x, y, avoid_rocks=False)
                safe_neighbors = [n for n in neighbors if n not in self.maze.rocks]
                if safe_neighbors:
                    x, y = random.choice(safe_neighbors)
                else:
                    break  # Trapped
            
            if (x, y) in visited:
                # Loop detection
                break
            
            visited.add((x, y))
            self.path.append((x, y))
            steps += 1
        
        return set(self.path) if (x, y) == self.maze.goal else None

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path) in a 20x20 maze.
Based on fleet simulation data: optimal path length = 21, rocks = 64.
"""

import random
from typing import List, Tuple, Set

# Constants from fleet knowledge
MAZE_SIZE = 20
OPTIMAL_PATH_LEN = 21  # From constraint_sim_final.json
ROCK_COUNT = 64        # From constraint_sim_final.json

class DeadbandMaze:
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
    def generate_rocks(self):
        """P0: Map negative space (rocks)."""
        # Generate random obstacles, ensuring start and goal are clear
        possible = [(x, y) for x in range(MAZE_SIZE) for y in range(MAZE_SIZE)]
        possible.remove(self.start)
        possible.remove(self.goal)
        
        self.rocks = set(random.sample(possible, ROCK_COUNT))
        for x, y in self.rocks:
            self.grid[x][y] = '#'
            
    def find_safe_channels(self):
        """P1: Identify safe channels (deadbands)."""
        # Simple BFS to find all reachable positions avoiding rocks
        visited = set()
        queue = [self.start]
        safe_positions = set()
        
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            safe_positions.add((x, y))
            
            # Check neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    queue.append((nx, ny))
        
        # Convert to channels (simple linear paths for demo)
        # In real implementation, this would find all possible safe paths
        self.safe_channels = [list(safe_positions)]
        return len(safe_positions)
    
    def optimize_path(self):
        """P2: Find optimal path within safe channels."""
        # A* search constrained to safe positions
        from heapq import heappush, heappop
        
        safe_set = set()
        for channel in self.safe_channels:
            safe_set.update(channel)
        
        # Heuristic: Manhattan distance
        def heuristic(pos):
            return abs(pos[0] - self.goal[0]) + abs(pos[1] - self.goal[1])
        
        open_set = []
        heappush(open_set, (heuristic(self.start), 0, self.start, None))
        
        came_from = {}
        g_score = {self.start: 0}
        
        while open_set:
            _, g, current, parent = heappop(open_set)
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current is not None:
                    path.append(current)
                    current = came_from.get(current)
                self.optimal_path = list(reversed(path))
                return len(self.optimal_path) - 1  # Path length
            
            x, y = current
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (x + dx, y + dy)
                if (0 <= neighbor[0] < MAZE_SIZE and 0 <= neighbor[1] < MAZE_SIZE and
                    neighbor in safe_set):
                    new_g = g + 1
                    if neighbor not in g_score or new_g < g_score[neighbor]:
                        g_score[neighbor] = new_g
                        f_score = new_g + heuristic(neighbor)
                        heappush(open_set, (f_score, new_g, neighbor, current))
                        came_from[neighbor] = current
        
        return None  # No path found
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2."""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===")
        print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Start: {self.start}, Goal: {self.goal}")
        print(f"Target: Optimal path = {OPTIMAL_PATH_LEN}, Rocks = {ROCK_COUNT}")
        print()
        
        # P0: Map negative space
        print("P0: MAPPING NEGATIVE SPACE (rocks)")
        self.generate_rocks()
        print(f"  • Identified {len(self.rocks)} rock positions")
        print("  • Rocks marked with '#'")
        print()
        
        # P1: Find safe channels
        print("P1: IDENTIFYING SAFE CHANNELS (deadbands)")
        safe_count = self.find_safe_channels()
        print(f"  • Found {safe_count} reachable safe positions")
        print(f"  • {len(self.safe_channels)} safe channel(s) identified")
        print()
        
        # P2: Optimize within channels
        print("P2: OPTIMIZING WITHIN SAFE CHANNELS")
        path_len = self.optimize_path()
        if path_len:
            print(f"  • Found optimal path length: {path_len}")
            if path_len == OPTIMAL_PATH_LEN:
                print("  ✓ Matches fleet optimal path length!")
            else:
                print(f"  Note: Fleet optimal is {OPTIMAL_PATH_LEN}")
        else:
            print("  ✗ No path found within safe channels")
        print()
        
        return path_len
    
    def visualize(self, show_path=True):
        """Create ASCII visualization."""
        viz_grid = [row[:] for row in self.grid]
        
        # Mark path
        if show_path and self.optimal_path:
            for x, y in self.optimal_path[1:-1]:  # Skip start and goal
                viz_grid[x][y] = 'o'
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        viz_grid[sx][sy] = 'S'
        viz_grid[gx][gy] = 'G'
        
        print("MAZE VISUALIZATION:")
        print("  S = Start, G = Goal, # = Rock, o = Path, . = Empty")
        print()
        for x in range(MAZE_SIZE):
            row = ''
            for y in range(MAZE_SIZE):
                row += viz_grid[x][y] + ' '
            print(f"  {row}")
        print()

**Status:** Script completed. This is a concrete demonstration of the Deadband Protocol using the simulation data from `constraint_sim_final.json` as a reference. The script visualizes the three-step process in a 20x20 maze.

**Output:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 with ASCII visualization.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# Simulation parameters from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64  # 64 rocks in 20x20 maze
START = (0, 0)
GOAL = (19, 19)

class DeadbandMaze:
    """Maze environment with Deadband Protocol implementation"""
    
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
        # Place rocks (P0 negative space)
        self._place_rocks()
        
        # Mark start and goal
        self.grid[START[0]][START[1]] = 'S'
        self.grid[GOAL[0]][GOAL[1]] = 'G'
    
    def _place_rocks(self):
        """P0: Map negative space (rocks to avoid)"""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != START and (x, y) != GOAL:
                self.rocks.add((x, y))
                self.grid[x][y] = '#'
    
    def find_safe_channels(self):
        """P1: Identify safe channels (paths without rocks)"""
        # Simple BFS to find all rock-free paths
        visited = set()
        queue = [(START, [START])]
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == GOAL:
                self.safe_channels.append(path)
                continue
            
            if (x, y) in visited:
                continue
            
            visited.add((x, y))
            
            # Check all four directions
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    def optimize_within_channel(self):
        """P2: Find shortest path within safe channels"""
        if not self.safe_channels:
            return []
        
        # Find shortest safe path
        self.optimal_path = min(self.safe_channels, key=len)
        return self.optimal_path
    
    def run_greedy_unconstrained(self):
        """Simulate P2-only (greedy) approach that ignores rocks"""
        path = [START]
        x, y = START
        
        while (x, y) != GOAL:
            # Greedy: move toward goal
            if x < GOAL[0]:
                x += 1
            elif y < GOAL[1]:
                y += 1
            elif x > GOAL[0]:
                x -= 1
            elif y > GOAL[1]:
                y -= 1
            
            path.append((x, y))
            
            # Check if hit rock (failure)
            if (x, y) in self.rocks:
                return path, False
        
        return path, True
    
    def visualize(self, path=None, show_rocks=True):
        """Create ASCII visualization of maze"""
        vis_grid = [row[:] for row in self.grid]
        
        if path:
            for i, (x, y) in enumerate(path[1:-1]):  # Skip start and goal
                if vis_grid[x][y] == '.':
                    vis_grid[x][y] = 'o'
        
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL MAZE DEMO")
        print("="*60)
        
        for i in range(MAZE_SIZE):
            row = ''
            for j in range(MAZE_SIZE):
                if not show_rocks and vis_grid[i][j] == '#':
                    row += ' '
                else:
                    row += vis_grid[i][j] + ' '
            print(row)
        
        print("\nLEGEND: S=Start, G=Goal, #=Rock, o=Path, .=Empty")
        print("="*60)

def main():
    """Run the Deadband Protocol demonstration"""
    print("Initializing Deadband Protocol Demo...")
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}")
    print(f"Start: {START}, Goal: {GOAL}")
    
    # Create maze environment
    maze = DeadbandMaze()
    
    print("\n=== STEP P0: MAP NEGATIVE SPACE ===")
    print(f"Identified {len(maze.rocks)} rocks (failure states)")
    maze.visualize(show_rocks=True)
    
    print("\n=== STEP P1: IDENTIFY SAFE CHANNELS ===")
    maze.find_safe_channels()
    print(f"Found {len(maze.safe_channels)} safe channels (rock-free paths)")
    maze.visualize(show_rocks=False)
    
    print("\n=== STEP P2: OPTIMIZE WITHIN CHANNEL ===")
    optimal_path = maze.optimize_within_channel()
    if optimal_path:
        print(f"Optimal path length: {len(optimal_path)-1} steps")
        print(f"Path: {optimal_path[:3]}...{optimal_path[-3:] if len(optimal_path) > 6 else ''}")
        maze.visualize(path=optimal_path, show_rocks=True)
    else:
        print("ERROR: No safe channels found!")
    
    print("\n=== COMPARISON: GREEDY (P2-ONLY) APPROACH ===")
    greedy_path, success = maze.run_greedy_unconstrained()
    print(f"Greedy path length: {len(greedy_path)-1} steps")
    print(f"Success: {success}")
    if not success:
        print("FAILED: Hit a rock (violated P0)")
    
    print("\n=== FLEET SIMULATION DATA REFERENCE ===")
    print("From constraint_sim_final.json:")
    print("  Deadband Protocol (P0→P1→P2): 50/50 success")
    print("  Greedy (P2-only): 0/50 success")
    print("  Constrained optimization: 0/50 success")
    print("\nThe Deadband Protocol ensures 100% success by")
    print("strictly following P0→P1→P2 priority.")

if __name__ == "__main__":
    main()
```

**Key Features:**
1. **P0 Implementation**: Randomly places 64 rocks (negative space) in the 20x20 maze, matching the simulation data
2. **P1 Implementation**: Uses BFS to find all rock-free paths from start to goal (safe channels)
3. **P2 Implementation**: Selects the shortest path from the safe channels
4. **Visualization**: ASCII display showing each step of the protocol
5. **Comparison**: Includes greedy (P2-only) approach that fails by hitting rocks
6. **Fleet Reference**: Directly references the `constraint_sim_final.json` simulation results

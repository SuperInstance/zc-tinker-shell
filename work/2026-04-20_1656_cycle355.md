"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import json
import random

# Load simulation parameters from fleet knowledge
def load_sim_params():
    """Read the constraint_sim_final.json file for real fleet data."""
    data_path = Path(__file__).parent / "fleet-knowledge" / "narrow-games" / "constraint_sim_final.json"
    try:
        with open(data_path, 'r') as f:
            data = json.load(f)
        print(f"✅ Loaded simulation data: {data_path.name}")
        return data
    except FileNotFoundError:

        return {

            "deadband_success": 50,

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_count=64):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self.safe_channels = []
        self.optimal_path = []
        
        # Generate random rocks (P0: negative space)
        self._generate_rocks(rock_count)
        
        # P1: Find safe channels (BFS avoiding rocks)
        self._find_safe_channels()
        
        # P2: Optimize within safe channels (A* on safe graph)
        if self.safe_channels:
            self._optimize_path()
    
    def _generate_rocks(self, count):
        """P0: Map the negative space (rocks to avoid)."""
        cells = [(x, y) for x in range(self.width) for y in range(self.height)]
        cells.remove(self.start)
        cells.remove(self.goal)
        
        self.rocks = set(random.sample(cells, min(count, len(cells))))
        print(f"P0: Mapped {len(self.rocks)} rocks (negative space)")
    
    def _find_safe_channels(self):
        """P1: Find all safe paths from start to goal (BFS)."""
        from collections import deque
        
        queue = deque([(self.start, [self.start])])
        visited = {self.start}
        all_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                all_paths.append(path)
                continue
            
            for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.safe_channels = all_paths
        print(f"P1: Found {len(self.safe_channels)} safe channels")
    
    def _optimize_path(self):
        """P2: Find shortest path within safe channels (A*)."""
        import heapq
        
        def heuristic(a, b):
            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        open_set = []
        heapq.heappush(open_set, (0, self.start))
        came_from = {self.start: None}
        g_score = {self.start: 0}
        
        while open_set:
            _, current = heapq.heappop(open_set)
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current is not None:
                    path.append(current)
                    current = came_from[current]
                self.optimal_path = list(reversed(path))
                return
            
            x, y = current
            for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
                nx, ny = x + dx, y + dy
                neighbor = (nx, ny)
                
                if not (0 <= nx < self.width and 0 <= ny < self.height):
                    continue
                if neighbor in self.rocks:
                    continue
                
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score = tentative_g + heuristic(neighbor, self.goal)
                    heapq.heappush(open_set, (f_score, neighbor))
        
        self.optimal_path = []
    
    def render(self, show_path=True):
        """Render ASCII maze with rocks, path, and markers."""
        grid = [['·' for _ in range(self.width)] for _ in range(self.height)]
        
        # Mark rocks
        for x, y in self.rocks:
            grid[y][x] = '█'
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        grid[sy][sx] = 'S'
        grid[gy][gx] = 'G'
        
        # Mark optimal path
        if show_path and self.optimal_path:
            for x, y in self.optimal_path[1:-1]:  # Skip start and goal
                if grid[y][x] == '·':
                    grid[y][x] = '○'
        
        # Build border
        border = '┌' + '─' * self.width + '┐\n'
        for row in grid:
            border += '│' + ''.join(row) + '│\n'
        border += '└' + '─' * self.width + '┘'
        
        return border
    
    def stats(self):
        """Return simulation statistics."""
        return {
            "width": self.width,
            "height": self.height,
            "rocks": len(self.rocks),
            "safe_channels": len(self.safe_channels),
            "path_length": len(self.optimal_path) if self.optimal_path else 0,
            "success": len(self.optimal_path) > 0
        }

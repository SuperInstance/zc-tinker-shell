#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 priority in a simple ASCII maze.
"""

import random
import time

class DeadbandMaze:
    def __init__(self, width=10, height=10, rock_density=0.2):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.grid = []
        self.rocks = set()
        self.safe_channels = []
        self.optimal_path = []
        
        # Generate maze with rocks
        for y in range(height):
            row = []
            for x in range(width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif random.random() < rock_density and (x, y) not in [self.start, self.goal]:
                    row.append('█')  # Rock
                    self.rocks.add((x, y))
                else:
                    row.append('.')  # Open space
            self.grid.append(row)
    
    def print_maze(self, title="", highlight=None):
        """Print the maze with optional path highlighting."""
        print(f"\n{title}")
        print("  " + "".join(str(i%10) for i in range(self.width)))
        
        for y in range(self.height):
            line = f"{y%10} "
            for x in range(self.width):
                cell = self.grid[y][x]
                if highlight and (x, y) in highlight:
                    line += '○'  # Path marker
                else:
                    line += cell
            print(line)
    
    def p0_map_negative_space(self):
        """P0: Identify rocks (what NOT to do)."""
        print("\n=== P0: MAP NEGATIVE SPACE ===")
        print(f"Identified {len(self.rocks)} rocks (failure states)")
        rock_positions = list(self.rocks)[:5]  # Show first 5
        if rock_positions:
            print(f"Sample rocks at: {rock_positions}")
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find safe paths avoiding rocks."""
        print("\n=== P1: FIND SAFE CHANNELS ===")
        
        # Simple BFS to find all safe paths
        from collections import deque
        
        safe_paths = []
        queue = deque([([self.start], set([self.start]))])
        
        while queue and len(safe_paths) < 3:  # Limit for demo
            path, visited = queue.popleft()
            x, y = path[-1]
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            # Check adjacent cells (right, down, left, up)
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) not in self.rocks and
                    (nx, ny) not in visited):
                    new_path = path + [(nx, ny)]
                    new_visited = visited.copy()
                    new_visited.add((nx, ny))
                    queue.append((new_path, new_visited))
        
        self.safe_channels = safe_paths
        print(f"Found {len(safe_paths)} safe channels")
        
        if safe_paths:
            print("First safe channel path:")
            for i, (x, y) in enumerate(safe_paths[0]):
                print(f"  Step {i}: ({x}, {y})")
        
        return safe_paths
    
    def p2_optimize_within_channel(self):
        """P2: Find shortest path within safe channels."""
        print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        if not self.safe_channels:
            print("ERROR: No safe channels found! Cannot optimize.")
            print("VIOLATION: Attempted P2 without P0+P1")
            return []
        
        # Find shortest safe path
        shortest = min(self.safe_channels, key=len)
        self.optimal_path = shortest
        
        print(f"Optimal path length: {len(shortest)-1} steps")
        print(f"Path: {shortest}")
        
        return shortest
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2."""
        print("=" * 50)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("=" * 50)
        
        # Show initial maze
        self.print_maze("Initial Maze")
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        time.sleep(1)
        
        # P1: Find safe channels
        safe_paths = self.p1_find_safe_channels()
        time.sleep(1)
        
        # P2: Optimize within channel
        optimal = self.p2_optimize_within_channel()
        time.sleep(1)
        
        # Show results
        if optimal:
            print("\n" + "=" * 50)
            print("RESULTS: Protocol SUCCESS")
            print(f"• Rocks avoided: {len(rocks)}")
            print(f"• Safe channels found: {len(safe_paths)}")
            print(f"• Optimal path steps: {len(optimal)-1}")
            self.print_maze("Optimal Path (○ marks the path)", set(optimal))
        else:
            print("\n" + "=" * 50)
            print("RESULTS: Protocol FAILED")
            print("Reason: No safe channels found (all paths blocked by rocks)")
        
        return optimal

def demonstrate_violation():
    """Show what happens when you skip to P2 (greedy optimization)."""
    print("\n" + "=" * 50)
    print("DEMONSTRATING PROTOCOL VIOLATION")
    print("(Skipping P0+P1, going straight to P2)")
    print("=" * 50)
    
    maze = DeadbandMaze(width=8, height=8, rock_density=0.25)
    maze.print_maze("Maze for greedy approach")
    
    print("\n=== GREEDY APPROACH (P2 only) ===")
    print("Trying to find shortest path without checking for rocks...")
    
    # Simple greedy path (right then down)
    greedy_path = []
    x, y = maze.start
    while (x, y) != maze.goal:
        greedy_path.append((x, y))
        # Greedy: prefer right, then down
        if x < maze.width - 1:
            x += 1
        elif y < maze.height - 1:
            y += 1
        else:
            break
    
    greedy_path.append(maze.goal)
    
    # Check for rock collisions
    collisions = [(x, y) for (x, y) in greedy_path if (x, y) in maze.rocks]
    
    if collisions:
        print(f"FAILED: Hit {len(collisions)} rocks!")
        print(f"First rock at: {collisions[0]}")
        maze.print_maze("Greedy path with rock collision", set(greedy_path))
    else:
        print("SUCCESS: No rocks hit (got lucky!)")
    
    return len(collisions) > 0

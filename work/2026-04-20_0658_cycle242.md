```python
#!/usr/bin/env python3
"""
deadband_maze_demo.py
Demonstrates the Deadband Protocol (P0→P1→P2) in a 20x20 ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque
import heapq

# Constants from simulation data
MAZE_SIZE = 20
ROCK_COUNT = 64  # 64 rocks in 400 cells (16%)
START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

def generate_maze():
    """P0: Map the negative space (rocks)."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        r = random.randint(0, MAZE_SIZE - 1)
        c = random.randint(0, MAZE_SIZE - 1)
        if (r, c) != START and (r, c) != GOAL:
            rocks.add((r, c))
    for r, c in rocks:
        maze[r][c] = '#'
    maze[START[0]][START[1]] = 'S'
    maze[GOAL[0]][GOAL[1]] = 'G'
    return maze, rocks

def find_safe_channels(maze, rocks):
    """P1: Identify safe channels (cells reachable without hitting rocks)."""
    safe = set()
    queue = deque([START])
    visited = set([START])
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        r, c = queue.popleft()
        safe.add((r, c))
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if (nr, nc) not in visited and (nr, nc) not in rocks:
                    visited.add((nr, nc))
                    queue.append((nr, nc))
    return safe

def optimize_within_channel(maze, safe):
    """P2: Find shortest path within safe channels (A* constrained to safe cells)."""
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, START))
    came_from = {}
    g_score = {START: 0}
    f_score = {START: heuristic(START, GOAL)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == GOAL:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            return path
        for dr, dc in dirs:
            nr, nc = current[0] + dr, current[1] + dc
            neighbor = (nr, nc)
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if neighbor not in safe:
                    continue  # Not in safe channel → skip
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, GOAL)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    return []  # No path found

def visualize(maze, path, safe):
    """Print maze with path and stats."""
    display = [row[:] for row in maze]
    for r, c in path:
        if (r, c) != START and (r, c) != GOAL:
            display[r][c] = '*'
    
    print("=" * 50)
    print("DEADBAND PROTOCOL DEMO - ASCII MAZE")
    print("=" * 50)
    print("Legend: # = rock (P0 negative space), . = safe channel (P1),")
    print("        S = start, G = goal, * = optimal path (P2)")
    print()
    for row in display:
        print(' '.join(row))
    print()
    
    # Stats
    total_cells = MAZE_SIZE * MAZE_SIZE
    safe_count = len(safe)
    rock_count = ROCK_COUNT
    path_len = len(path)
    print(f"STATISTICS:")
    print(f"  Total cells: {total_cells}")
    print(f"  Rocks (P0 negative space): {rock_count} ({rock_count/total_cells*100:.1f}%)")
    print(f"  Safe channel cells (P1): {safe_count} ({safe_count/total_cells*100:.1f}%)")
    print(f"  Optimal path length (P2): {path_len} steps")
    print(f"  Path found: {'YES' if path_len > 0 else 'NO'}")
    print("=" * 50)

def main():
    random.seed(42)  # Deterministic for demo
    
    print("DEADBAND PROTOCOL EXECUTION")
    print("P0: Mapping negative space (rocks)...")
    maze, rocks = generate_maze()
    
    print("P1: Identifying safe channels (deadbands)...")
    safe = find_safe_channels(maze, rocks)
    
    print("P2: Optimizing within safe channels (A* search)...")
    path = optimize_within_channel(maze, safe)
    
    visualize(maze, path, safe)
    
    # Protocol validation
    if not path:
        print("WARNING: No path found. Goal may be unreachable in safe channels.")
    else:
        # Verify path stays in safe channels
        for cell in path:
            if cell not in safe and cell != START and cell != GOAL:
                print("ERROR: Path left safe channels! Protocol violation.")
                return
        print("SUCCESS: Protocol followed correctly (P0→P1→P2).")

if __name__ == "__main__":
    main()

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

A visual ASCII simulation of the Deadband Protocol (P0→P1→P2).
Demonstrates the strict priority: map rocks (P0), find safe channels (P1),
then optimize path (P2) within those channels.

Based on fleet-knowledge/narrow-games/constraint_sim_final.json
and DEADBAND-PROTOCOL.md.
"""

import random
from typing import List, Tuple, Set

# ====================
# MAZE CONFIGURATION
# ====================
MAZE_WIDTH = 20
MAZE_HEIGHT = 20
START = (0, 0)
GOAL = (19, 19)
ROCK_COUNT = 64  # matches simulation data

# Symbols for visualization
SYM_EMPTY = '·'
SYM_ROCK = '█'
SYM_START = 'S'
SYM_GOAL = 'G'
SYM_PATH = '○'
SYM_DEADBAND = '░'  # safe channel cells

# ====================
# DEADBAND PROTOCOL
# ====================

class DeadbandAgent:
    """Implements P0→P1→P2 strictly."""
    
    def __init__(self, maze: List[List[str]]):
        self.maze = maze
        self.rocks: Set[Tuple[int, int]] = set()      # P0: negative space
        self.safe_channels: Set[Tuple[int, int]] = set()  # P1: safe cells
        self.path: List[Tuple[int, int]] = []         # P2: optimal path
        
    def p0_map_rocks(self) -> None:
        """P0: Identify all rock positions (what NOT to do)."""
        for y in range(MAZE_HEIGHT):
            for x in range(MAZE_WIDTH):
                if self.maze[y][x] == SYM_ROCK:
                    self.rocks.add((x, y))
        print(f"[P0] Mapped {len(self.rocks)} rocks (negative space)")
    
    def p1_find_safe_channels(self) -> None:
        """P1: Find all cells reachable from start without hitting rocks."""
        visited = set()
        stack = [START]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Add to safe channels if not a rock
            if (x, y) not in self.rocks:
                self.safe_channels.add((x, y))
            
            # Explore neighbors (4-directional)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                    if (nx, ny) not in visited and (nx, ny) not in self.rocks:
                        stack.append((nx, ny))
        
        print(f"[P1] Found {len(self.safe_channels)} safe channel cells")
    
    def p2_optimize_path(self) -> bool:
        """P2: Find shortest path from start to goal within safe channels.
        Returns True if path found."""
        from collections import deque
        
        if GOAL not in self.safe_channels:
            print("[P2] Goal not reachable within safe channels")
            return False
        
        # BFS within safe channels only
        queue = deque([(START, [START])])
        visited = set([START])
        
        while queue:
            (x, y), current_path = queue.popleft()
            
            if (x, y) == GOAL:
                self.path = current_path
                print(f"[P2] Found optimal path: {len(self.path)} steps")
                return True
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                    if (nx, ny) in self.safe_channels and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), current_path + [(nx, ny)]))
        
        print("[P2] No path found within safe channels")
        return False
    
    def run_protocol(self) -> bool:
        """Execute P0→P1→P2 in strict order."""
        print("\n=== DEADBAND PROTOCOL EXECUTION ===")
        self.p0_map_rocks()
        self.p1_find_safe_channels()
        return self.p2_optimize_path()

def generate_maze() -> List[List[str]]:
    """Create a random maze with rocks."""
    maze = [[SYM_EMPTY for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    
    # Place rocks randomly (avoid start and goal)
    rocks_placed = 0
    while rocks_placed < ROCK_COUNT:
        x = random.randint(0, MAZE_WIDTH-1)
        y = random.randint(0, MAZE_HEIGHT-1)
        if (x, y) != START and (x, y) != GOAL and maze[y][x] == SYM_EMPTY:
            maze[y][x] = SYM_ROCK
            rocks_placed += 1
    
    # Mark start and goal
    maze[START[1]][START[0]] = SYM_START
    maze[GOAL[1]][GOAL[0]] = SYM_GOAL
    
    return maze

def print_maze(maze: List[List[str]], title: str = "") -> None:
    """Print maze with border."""
    if title:
        print(f"\n{title}")
    print("+" + "-" * MAZE_WIDTH + "+")
    for row in maze:
        print("|" + "".join(row) + "|")
    print("+" + "-" * MAZE_WIDTH + "+")

    else:
        print("No path found")

**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on the simulation data in `fleet-knowledge/narrow-games/constraint_sim_final.json` and the formal protocol in `DEADBAND-PROTOCOL.md`, I will implement a concrete demonstration.

**Approach:**
1. **P0 (Map Negative Space):** Load the maze and identify all rock positions (obstacles).
2. **P1 (Identify Safe Channels):** Perform flood-fill from start to goal, marking only cells reachable without hitting rocks.
3. **P2 (Optimize Within Channel):** Run A* search constrained to the safe channel cells found in P1.
4. **Visualization:** Print the maze at each phase with ASCII art.

**Output:** A standalone Python script `deadband_maze_demo.py`.

def print_maze(maze: List[str], title: str = ""):
    """Print maze with title."""
    if title:

def p0_map_negative_space(maze: List[str]) -> Set[Tuple[int, int]]:
    """
    P0: Identify all rock positions (negative space).
    Returns set of (row, col) coordinates that are obstacles.
    """
    rocks = set()
    for r, row in enumerate(maze):
        for c, ch in enumerate(row):
            if ch == '#':
                rocks.add((r, c))
    return rocks

def p1_find_safe_channels(maze: List[str], start: Tuple[int, int], goal: Tuple[int, int], rocks: Set[Tuple[int, int]]) -> Set[Tuple[int, int]]:
    """
    P1: Flood-fill from start to find all reachable cells that are not rocks.
    Returns set of safe channel cells.
    """
    rows, cols = len(maze), len(maze[0])
    safe = set()
    stack = [start]
    
    while stack:
        r, c = stack.pop()
        if (r, c) in rocks or (r, c) in safe:
            continue
        safe.add((r, c))
        # 4-directional neighbors
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            if 0 <= nr < rows and 0 <= nc < cols:
                stack.append((nr, nc))
    
    return safe

def p2_optimize_in_channel(maze: List[str], start: Tuple[int, int], goal: Tuple[int, int], safe: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
    """
    P2: A* search constrained to safe channel cells.
    Returns optimal path from start to goal, or empty list if no path.
    """
    rows, cols = len(maze), len(maze[0])
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            path.reverse()
            return path
        
        r, c = current
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            neighbor = (nr, nc)
            if not (0 <= nr < rows and 0 <= nc < cols):
                continue
            if neighbor not in safe:
                continue  # Constrained to safe channel
            
            tentative_g = g_score[current] + 1
            if neighbor not in g_score or tentative_g < g_score[neighbor]:
                came_from[neighbor] = current
                g_score[neighbor] = tentative_g
                f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return []  # No path found

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Simulates the fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set

# Simulation parameters from fleet data
MAZE_SIZE = 20
NUM_ROCKS = 64
OPTIMAL_PATH_LEN = 21

class DeadbandAgent:
    def __init__(self, maze_size: int = MAZE_SIZE):
        self.size = maze_size
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (maze_size-1, maze_size-1)
        
    def generate_rocks(self, count: int = NUM_ROCKS) -> None:
        """P0: Map negative space (what NOT to do)"""
        while len(self.rocks) < count:
            x = random.randint(0, self.size-1)
            y = random.randint(0, self.size-1)
            # Don't block start or goal
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Find safe channels (where you CAN be)"""
        # Simple BFS that avoids rocks
        from collections import deque
        
        queue = deque([(self.start, [self.start])])
        visited = {self.start}
        safe_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            # Check neighbors (right and down only for simplicity)
            for dx, dy in [(1, 0), (0, 1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.size and 0 <= ny < self.size:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        return safe_paths
    
    def optimize_within_channel(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """P2: Optimize within channels (best path)"""
        if not safe_paths:
            return []
        
        # Find shortest safe path
        return min(safe_paths, key=len)
    
    def greedy_unconstrained(self) -> List[Tuple[int, int]]:
        """Violation: P2-only (greedy optimization without P0+P1)"""
        path = [self.start]
        x, y = self.start
        
        while (x, y) != self.goal:
            # Always move toward goal (greedy)
            if x < self.goal[0]:
                x += 1
            elif y < self.goal[1]:
                y += 1
            
            path.append((x, y))
            
            # Check if hit rock
            if (x, y) in self.rocks:
                return []  # Failed
        
        return path
    
    def visualize(self, path: List[Tuple[int, int]] = None) -> str:
        """Create ASCII visualization"""
        grid = [['·' for _ in range(self.size)] for _ in range(self.size)]
        
        # Mark rocks
        for x, y in self.rocks:
            grid[y][x] = '█'
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        grid[sy][sx] = 'S'
        grid[gy][gx] = 'G'
        
        # Mark path if provided
        if path:
            for x, y in path[1:-1]:  # Skip start and goal
                if grid[y][x] == '·':
                    grid[y][x] = '○'
        
        # Build output
        lines = []
        lines.append("=" * (self.size * 2 + 3))
        lines.append("MAZE (20×20) - Deadband Protocol Demo")
        lines.append("S=Start, G=Goal, █=Rock, ○=Safe Path, ·=Empty")
        lines.append("=" * (self.size * 2 + 3))
        
        for y in range(self.size):
            row = ' '.join(grid[y])
            lines.append(f"{y:2d} {row}")
        
        lines.append("  " + ' '.join(str(i%10) for i in range(self.size)))
        
        return '\n'.join(lines)

def run_simulation(num_runs: int = 5) -> dict:
    """Run multiple simulations to show protocol effectiveness"""
    results = {
        'deadband_success': 0,
        'greedy_success': 0,
        'deadband_path_lens': [],
        'greedy_path_lens': []
    }
    
    for run in range(num_runs):
        print(f"\n=== Run {run+1}/{num_runs} ===")
        
        agent = DeadbandAgent()
        agent.generate_rocks()
        
        # Deadband Protocol (P0→P1→P2)
        safe_paths = agent.find_safe_channels()
        if safe_paths:
            optimal_path = agent.optimize_within_channel(safe_paths)
            if optimal_path:
                results['deadband_success'] += 1
                results['deadband_path_lens'].append(len(optimal_path))
                print(f"Deadband: SUCCESS (path length: {len(optimal_path)})")
            else:
                print("Deadband: FAILED (no optimal path found)")
        else:
            print("Deadband: FAILED (no safe channels)")
        
        # Greedy (P2-only violation)
        greedy_path = agent.greedy_unconstrained()
        if greedy_path:
            results['greedy_success'] += 1
            results['greedy_path_lens'].append(len(greedy_path))
            print(f"Greedy: SUCCESS (path length: {len(greedy_path)})")
        else:
            print("Greedy: FAILED (hit rock)")
        
        # Show visualization for first run
        if run == 0 and optimal_path:
            print("\n" + agent.visualize(optimal_path))
    
    # Calculate averages
    if results['deadband_path_lens']:
        results['deadband_avg'] = sum(results['deadband_path_lens']) / len(results['deadband_path_lens'])
    else:
        results['deadband_avg'] = 0
    
    if results['greedy_path_lens']:
        results['greedy_avg'] = sum(results['greedy_path_lens']) / len(results['greedy_path_lens'])
    else:
        results['greedy_avg'] = 0
    
    return results

def main():
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("=" * 40)
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print(f"Parameters: {MAZE_SIZE}×{MAZE_SIZE} maze, {NUM_ROCKS} rocks")
    print(f"Optimal path length: {OPTIMAL_PATH_LEN}")
    print("\nProtocol phases:")
    print("P0: Map rocks (negative space)")
    print("P1: Find safe channels (avoid rocks)")
    print("P2: Optimize within channels (shortest safe path)")
    print("\nRunning simulation...")
    
    results = run_simulation(num_runs=5)
    
    print("\n" + "=" * 40)
    print("RESULTS SUMMARY (5 runs)")
    print("=" * 40)
    print(f"Deadband Protocol success: {results['deadband_success']}/5")
    print(f"Greedy (P2-only) success: {results['greedy_success']}/5")
    print(f"Deadband avg path length: {results['deadband_avg']:.1f}")
    print(f"Greedy avg path length: {results['greedy_avg']:.1f}")
    
    # Compare to fleet data
    print("\n" + "=" * 40)
    print("COMPARISON TO FLEET DATA")
    print("=" * 40)
    print("From constraint_sim_final.json:")
    print("  Deadband success: 50/50

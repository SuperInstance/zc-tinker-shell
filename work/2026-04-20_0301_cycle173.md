Deadband Protocol Maze Demonstration
Based on Cocapn fleet's Deadband Protocol v1.0
Simulates P0→P1→P2 priority in pathfinding
"""

import random
from collections import deque
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate maze with rocks (obstacles)
        self.rocks: Set[Tuple[int, int]] = set()
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density and (x, y) not in [self.start, self.goal]:
                    self.rocks.add((x, y))
        
        # Ensure at least one path exists
        self._ensure_path_exists()
    
    def _ensure_path_exists(self):
        """Guarantee at least one path from start to goal"""
        visited = set()
        stack = [self.start]
        
        while stack:
            x, y = stack.pop()
            if (x, y) == self.goal:
                return
            
            visited.add((x, y))
            
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and 
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    stack.append((nx, ny))
        
        # If no path, clear a diagonal path
        for i in range(min(self.width, self.height)):
            x = min(i, self.width-1)
            y = min(i, self.height-1)
            if (x, y) in self.rocks:
                self.rocks.remove((x, y))
    
    def render(self, path: List[Tuple[int, int]] = None) -> str:
        """Render maze as ASCII art"""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif (x, y) in self.rocks:
                    row.append('█')
                elif path and (x, y) in path:
                    row.append('·')
                else:
                    row.append(' ')
            grid.append(''.join(row))
        return '\n'.join(grid)
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify rocks (what NOT to do)"""
        return self.rocks.copy()
    
    def p1_find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Find all safe paths from start to goal (safe channels)"""
        safe_paths = []
        visited = set()
        
        def dfs(current: Tuple[int, int], path: List[Tuple[int, int]]):
            if current == self.goal:
                safe_paths.append(path.copy())
                return
            
            visited.add(current)
            x, y = current
            
            # Explore neighbors in order: right, down, left, up
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                neighbor = (nx, ny)
                
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    neighbor not in self.rocks and neighbor not in visited):
                    path.append(neighbor)
                    dfs(neighbor, path)
                    path.pop()
            
            visited.remove(current)
        
        dfs(self.start, [self.start])
        return safe_paths
    
    def p2_optimize_within_channels(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """P2: Select shortest safe path"""
        if not safe_paths:
            return []
        return min(safe_paths, key=len)
    
    def greedy_p2_only(self) -> List[Tuple[int, int]]:
        """P2-only approach (violates protocol)"""
        visited = set()
        path = [self.start]
        current = self.start
        
        while current != self.goal:
            x, y = current
            visited.add(current)
            
            # Greedy: move toward goal
            candidates = []
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                neighbor = (nx, ny)
                
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    neighbor not in visited):
                    # Manhattan distance heuristic
                    dist = abs(nx - self.goal[0]) + abs(ny - self.goal[1])
                    candidates.append((dist, neighbor))
            
            if not candidates:
                return []  # Stuck
            
            candidates.sort()
            current = candidates[0][1]
            path.append(current)
            
            # If we hit a rock, fail
            if current in self.rocks:
                return []
        
        return path

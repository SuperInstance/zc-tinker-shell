**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on the simulation data in `constraint_sim_final.json` and the formal protocol in `DEADBAND-PROTOCOL.md`, I will implement a concrete demonstration.

**Approach (P0→P1→P2):**
- **P0 (Map negative space):** Load the maze configuration (rocks) from the simulation data.
- **P1 (Identify safe channels):** Compute all paths that avoid rocks.
- **P2 (Optimize within channel):** Find the shortest safe path from start to goal.

**Implementation:** A Python script that generates a 20x20 maze matching the simulation parameters (64 rocks, optimal path length 21), visualizes it, and runs the deadband protocol.

---

**File:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 with a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque

# Simulation parameters from constraint_sim_final.json
MAZE_SIZE = 20
NUM_ROCKS = 64
OPTIMAL_PATH_LEN = 21

# Symbols
START = 'S'
GOAL = 'G'
ROCK = '█'
PATH = '·'
SAFE = ' '
VISITED = 'x'

def generate_maze():
    """Generate a 20x20 maze with 64 rocks, start at (0,0), goal at (19,19)."""
    maze = [[SAFE for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    
    # Place start and goal
    start = (0, 0)
    goal = (MAZE_SIZE-1, MAZE_SIZE-1)
    maze[start[0]][start[1]] = START
    maze[goal[0]][goal[1]] = GOAL
    
    # Place rocks randomly, avoiding start and goal
    rocks_placed = 0
    while rocks_placed < NUM_ROCKS:
        r = random.randint(0, MAZE_SIZE-1)
        c = random.randint(0, MAZE_SIZE-1)
        if (r, c) != start and (r, c) != goal and maze[r][c] == SAFE:
            maze[r][c] = ROCK
            rocks_placed += 1
    
    return maze, start, goal

def print_maze(maze, title="Maze"):
    """Print maze with border."""
    print(f"\n=== {title} ===")
    print('+' + '-' * MAZE_SIZE + '+')
    for row in maze:
        print('|' + ''.join(row) + '|')
    print('+' + '-' * MAZE_SIZE + '+')

def deadband_protocol(maze, start, goal):
    """
    Implement Deadband Protocol:
    P0: Rocks are negative space (already mapped in maze)
    P1: BFS to find all safe reachable cells (safe channels)
    P2: Shortest path within safe channels.
    """
    rows, cols = MAZE_SIZE, MAZE_SIZE
    directions = [(0,1), (1,0), (0,-1), (-1,0)]
    
    # P1: Find safe channels (all cells reachable without hitting rocks)
    safe_channels = [[False for _ in range(cols)] for _ in range(rows)]
    queue = deque([start])
    safe_channels[start[0]][start[1]] = True
    
    while queue:
        r, c = queue.popleft()
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < rows and 0 <= nc < cols:
                if not safe_channels[nr][nc] and maze[nr][nc] != ROCK:
                    safe_channels[nr][nc] = True
                    queue.append((nr, nc))
    
    # If goal not reachable, no safe channel exists
    if not safe_channels[goal[0]][goal[1]]:
        return None, safe_channels
    
    # P2: Shortest path within safe channels (BFS)
    parent = {start: None}
    queue = deque([start])
    found = False
    
    while queue and not found:
        r, c = queue.popleft()
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < rows and 0 <= nc < cols:
                if (nr, nc) not in parent and safe_channels[nr][nc]:
                    parent[(nr, nc)] = (r, c)
                    if (nr, nc) == goal:
                        found = True
                        break
                    queue.append((nr, nc))
    
    # Reconstruct path
    if not found:
        return None, safe_channels
    
    path = []
    node = goal
    while node:
        path.append(node)
        node = parent[node]
    path.reverse()
    
    return path, safe_channels

def visualize_solution(maze, path, safe_channels):
    """Create a visualization maze showing safe channels and path."""
    viz = [row[:] for row in maze]
    
    # Mark safe channels (P1 result)
    for r in range(MAZE_SIZE):
        for c in range(MAZE_SIZE):
            if safe_channels[r][c] and viz[r][c] == SAFE:
                viz[r][c] = '.'
    
    # Mark path (P2 result)
    for (r, c) in path[1:-1]:  # Exclude start and goal
        viz[r][c] = '*'
    
    return viz

def main():
    print("Deadband Protocol Maze Demonstration")
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {NUM_ROCKS}, Optimal: {OPTIMAL_PATH_LEN} steps")
    print("=" * 50)
    
    # Generate maze matching simulation parameters
    maze, start, goal = generate_maze()
    print_maze(maze, "Initial Maze (P0: Rocks = █)")
    
    # Run Deadband Protocol
    path, safe_channels = deadband_protocol(maze, start, goal)
    
    if path is None:
        print("\n❌ RESULT: No safe channel to goal (constrained failure)")
        # Show safe channels anyway
        viz = [row[:] for row in maze]
        for r in range(MAZE_SIZE):
            for c in range(MAZE_SIZE):
                if safe_channels[r][c] and viz[r][c] == SAFE:
                    viz[r][c] = '.'
        print_maze(viz, "Safe Channels (P1) - Goal Unreachable")
        return
    
    # Successful path found
    print(f"\n✅ RESULT: Safe path found ({len(path)-1} steps)")
    
    # Visualize
    viz = visualize_solution(maze, path, safe_channels)
    print_maze(viz, "Deadband Protocol Solution")
    print("Legend: S=Start, G=Goal, █=Rock, .=Safe channel, *=Optimal path")
    
    # Protocol summary
    print("\n--- Deadband Protocol Execution ---")
    print("P0 (Negative space): 64 rocks identified (avoid █)")
    print(f"P1 (Safe channels): {sum(sum(row) for row in safe_channels)} reachable cells")
    print(f"P2 (Optimization): Shortest safe path = {len(path)-1} steps")
    
    # Compare to optimal
    if len(path)-1 == OPTIMAL_PATH_LEN:
        print("✓ Matches optimal path length from simulation")
    else:
        print(f"Note: Simulation optimal was {OPTIMAL_PATH_LEN}, this maze differs due to random rock placement")

if __name__ == "__main__":
    main()
```

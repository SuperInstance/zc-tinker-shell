#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 289)
A visual ASCII demonstration of P0→P1→P2 in a constrained navigation problem.
"""

import random
import time
from typing import List, Tuple, Set

class DeadbandMaze:
    """
    Simulates an agent navigating a maze using the Deadband Protocol.
    """
    
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.2):
        self.width = width
        self.height = height
        self.rock_density = rock_density
        
        # State
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.agent_pos = self.start
        
        # Protocol phases
        self.rocks: Set[Tuple[int, int]] = set()      # P0: Negative space
        self.safe_channels: Set[Tuple[int, int]] = set()  # P1: Safe cells
        self.path: List[Tuple[int, int]] = []         # P2: Optimal path
        
        # Generate environment
        self._generate_maze()
    
    def _generate_maze(self):
        """Generate random rocks (obstacles)."""
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) != self.start and (x, y) != self.goal:
                    if random.random() < self.rock_density:
                        self.rocks.add((x, y))
    
    def p0_map_negative_space(self):
        """P0: Identify rocks (what NOT to do)."""
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all cells reachable without hitting rocks."""
        visited = set()
        stack = [self.start]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            if (x, y) in self.rocks:
                continue
                
            visited.add((x, y))
            self.safe_channels.add((x, y))
            
            # Check neighbors (4-directional)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    stack.append((nx, ny))
        
        return self.safe_channels
    
    def p2_optimize_within_channel(self):
        """P2: Find shortest path through safe channels (BFS)."""
        if self.goal not in self.safe_channels:
            return []  # No safe path to goal
        
        queue = [(self.start, [self.start])]
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.goal:
                self.path = path
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited and (nx, ny) in self.safe_channels:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        return []
    
    def run_protocol(self):
        """Execute full P0→P1→P2 protocol."""
        print("=== DEADBAND PROTOCOL DEMO ===")
        print(f"Maze: {self.width}x{self.height}, Rocks: {len(self.rocks)}")
        print()
        
        # P0: Map negative space
        print("P0: MAPPING NEGATIVE SPACE")
        rocks = self.p0_map_negative_space()
        print(f"  Identified {len(rocks)} rocks (avoid these)")
        time.sleep(1)
        
        # P1: Find safe channels
        print("P1: FINDING SAFE CHANNELS")
        safe = self.p1_find_safe_channels()
        print(f"  Found {len(safe)} safe cells")
        time.sleep(1)
        
        # P2: Optimize within channel
        print("P2: OPTIMIZING WITHIN CHANNEL")
        path = self.p2_optimize_within_channel()
        if path:
            print(f"  Found optimal path: {len(path)} steps")
        else:
            print("  WARNING: No safe path to goal!")
        time.sleep(1)
        
        return path
    
    def visualize(self, show_phases: bool = False):
        """Render ASCII visualization of the maze."""
        print("\n" + "="*50)
        print("MAZE VISUALIZATION")
        print("="*50)
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')  # Rock
                elif pos in self.path:
                    row.append('·')  # Path
                elif pos in self.safe_channels:
                    if show_phases:
                        row.append(' ')  # Safe channel
                    else:
                        row.append('.')
                else:
                    row.append(' ')  # Unreachable/unknown
            print(''.join(row))
        
        # Legend
        print("\nLEGEND:")
        print("  S = Start, G = Goal, █ = Rock (P0), · = Optimal Path (P2)")
        if show_phases:
            print("  ' ' = Safe Channel (P1), ' ' = Unsafe/Unknown")
        else:
            print("  . = Reachable, ' ' = Unreachable")

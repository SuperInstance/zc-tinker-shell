#!/usr/bin/env python3
"""
deadband_maze_demo.py

A visual demonstration of the Deadband Protocol (P0→P1→P2) in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json simulation.
"""

import random
from typing import List, Tuple, Set
from collections import deque

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64          # from "rocks": 64
OPTIMAL_PATH_LEN = 21    # from "optimal": 21

# Symbols for visualization
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL = 'G'
PATH = '○'
SAFE = '░'

class DeadbandMaze:
    def __init__(self, seed=None):
        if seed is not None:
            random.seed(seed)
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        
        # Place rocks (P0 negative space)
        while len(self.rocks) < ROCK_COUNT:
            r = random.randint(0, MAZE_SIZE-1)
            c = random.randint(0, MAZE_SIZE-1)
            if (r, c) != self.start and (r, c) != self.goal:
                self.rocks.add((r, c))
                self.grid[r][c] = ROCK
        
        self.grid[self.start[0]][self.start[1]] = AGENT
        self.grid[self.goal[0]][self.goal[1]] = GOAL
        
        # For visualization
        self.safe_channels: Set[Tuple[int, int]] = set()
        self.final_path: List[Tuple[int, int]] = []
    
    def print_maze(self, title="Maze"):
        print(f"\n=== {title} ===")
        for row in self.grid:
            print(' '.join(row))
        print()
    
    def p0_map_rocks(self) -> Set[Tuple[int, int]]:
        """P0: Identify negative space (rocks)."""
        return self.rocks.copy()
    
    def p1_find_safe_channels(self) -> Set[Tuple[int, int]]:
        """P1: Find all cells reachable without hitting rocks."""
        visited = set()
        queue = deque([self.start])
        while queue:
            r, c = queue.popleft()
            if (r, c) in visited:
                continue
            visited.add((r, c))
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                    if (nr, nc) not in self.rocks and (nr, nc) not in visited:
                        queue.append((nr, nc))
        self.safe_channels = visited
        return visited
    
    def p2_optimize_in_channel(self, safe_cells: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
        """P2: Find shortest path within safe channels (BFS)."""
        if self.goal not in safe_cells:
            return []  # No safe path to goal
        
        parent = {self.start: None}
        queue = deque([self.start])
        while queue:
            r, c = queue.popleft()
            if (r, c) == self.goal:
                # Reconstruct path
                path = []
                while (r, c) != self.start:
                    path.append((r, c))
                    r, c = parent[(r, c)]
                path.reverse()
                self.final_path = path
                return path
            
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                    if (nr, nc) in safe_cells and (nr, nc) not in parent:
                        parent[(nr, nc)] = (r, c)
                        queue.append((nr, nc))
        return []
    
    def run_deadband_protocol(self) -> bool:
        """Execute full P0→P1→P2 sequence."""
        print("\n[P0] Mapping negative space (rocks)...")
        rocks = self.p0_map_rocks()
        print(f"   Found {len(rocks)} rocks (negative space).")
        
        print("[P1] Finding safe channels...")
        safe = self.p1_find_safe_channels()
        print(f"   Safe channels cover {len(safe)} cells.")
        
        print("[P2] Optimizing path within safe channels...")
        path = self.p2_optimize_in_channel(safe)
        if path:
            print(f"   Found safe path of length {len(path)}.")
            # Mark path on grid for visualization
            for r, c in path:
                if self.grid[r][c] == EMPTY:
                    self.grid[r][c] = PATH
            return True
        else:
            print("   No safe path to goal.")
            return False
    
    def run_greedy_p2_only(self) -> bool:
        """Naive greedy approach (P2 only) for comparison."""
        print("\n[GREEDY P2-ONLY] Optimizing without P0/P1...")
        # Simple BFS ignoring rock mapping

        queue = deque([self.start])

                while (r, c) != self.start:
                    path.append((r, c))
                    r, c = parent[(r, c)]
                print(f"   Greedy found path of length {len(path)} (may hit rocks).")
                return True

                if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                    if (nr, nc) not in parent:
                        parent[(nr, nc)] = (r, c)
                        queue.append((nr, nc))
        print("   Greedy found no path.")
        return False

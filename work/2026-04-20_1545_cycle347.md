#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2 Visual Simulation
A simple ASCII demonstration of the fleet's Deadband Protocol.
"""

import random
import time
import sys

class Maze:
    """A 20x20 maze with rocks, start, and goal."""
    def __init__(self, seed=None):
        if seed is not None:
            random.seed(seed)
        self.width = 20
        self.height = 20
        self.start = (0, 0)
        self.goal = (19, 19)
        self.rocks = set()
        # Generate ~15% rocks (about 60 rocks)
        for _ in range(60):
            x = random.randint(0, self.width-1)
            y = random.randint(0, self.height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
        # Ensure at least one viable path exists (brute-force)
        self._ensure_path()

    def _ensure_path(self):
        """Guarantee at least one path from start to goal (no rocks blocking)."""
        # Simple DFS to check connectivity
        stack = [self.start]
        visited = set()
        while stack:
            x, y = stack.pop()
            if (x, y) == self.goal:
                return
            if (x, y) in visited:
                continue
            visited.add((x, y))
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks:
                        stack.append((nx, ny))
        # If no path, clear a straight diagonal
        for i in range(min(self.width, self.height)):
            if (i, i) in self.rocks:
                self.rocks.remove((i, i))

    def render(self, agent_pos=None, deadband=None):
        """Return ASCII representation of the maze."""
        lines = []
        for y in range(self.height):
            line = []
            for x in range(self.width):
                if (x, y) == self.start:
                    line.append('S')
                elif (x, y) == self.goal:
                    line.append('G')
                elif (x, y) in self.rocks:
                    line.append('#')
                elif deadband and (x, y) in deadband:
                    line.append('.')
                elif agent_pos and (x, y) == agent_pos:
                    line.append('A')
                else:
                    line.append(' ')
            lines.append(''.join(line))
        return '\n'.join(lines)

def deadband_protocol_demo(maze):
    """Execute P0→P1→P2 step-by-step with visual output."""
    print("=== DEADBAND PROTOCOL DEMO ===\n")
    print("Maze: 20x20, S=Start, G=Goal, #=Rock, A=Agent, .=Safe Channel\n")

    # P0: Map Negative Space (Rocks)
    print("P0: MAP NEGATIVE SPACE (Identify Rocks)")
    print("Rocks are all coordinates where agent would fail.")
    print(f"Total rocks identified: {len(maze.rocks)}\n")
    print(maze.render())
    input("\nPress Enter to proceed to P1...")
    print("\n" + "="*40 + "\n")

    # P1: Identify Safe Channels (Deadbands)
    print("P1: IDENTIFY SAFE CHANNELS (Deadbands)")
    print("Performing flood-fill from start, avoiding rocks...")
    # Simple BFS to find all reachable non-rock cells
    queue = [maze.start]
    safe = set()
    while queue:
        x, y = queue.pop(0)
        if (x, y) in safe:
            continue
        safe.add((x, y))
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < maze.width and 0 <= ny < maze.height:
                if (nx, ny) not in maze.rocks:
                    queue.append((nx, ny))
    print(f"Safe cells reachable from start: {len(safe)}")
    print("Visualizing safe channels (marked with '.'):\n")
    print(maze.render(deadband=safe))
    input("\nPress Enter to proceed to P2...")
    print("\n" + "="*40 + "\n")

    # P2: Optimize Within Channel (A* within safe set)
    print("P2: OPTIMIZE WITHIN SAFE CHANNEL (A* pathfinding)")
    # Simplified A* (Manhattan distance) constrained to safe cells
    open_set = {maze.start}
    came_from = {}
    g_score = {maze.start: 0}
    f_score = {maze.start: abs(maze.goal[0]-maze.start[0]) + abs(maze.goal[1]-maze.start[1])}

    while open_set:
        current = min(open_set, key=lambda p: f_score.get(p, float('inf')))
        if current == maze.goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            print(f"Optimal path found! Length: {len(path)} steps")
            # Animate agent moving along path
            for i, pos in enumerate(path):
                print(f"\nStep {i+1}:")
                print(maze.render(agent_pos=pos, deadband=safe))
                time.sleep(0.1)
            print(f"\nGoal reached in {len(path)} steps!")
            return True
        open_set.remove(current)
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            neighbor = (nx, ny)
            if 0 <= nx < maze.width and 0 <= ny < maze.height:
                if neighbor not in safe:
                    continue  # Constrained to safe channel
                tentative_g = g_score[current] + 1
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + (abs(maze.goal[0]-nx) + abs(maze.goal[1]-ny))
                    if neighbor not in open_set:
                        open_set.add(neighbor)
    print("No path found within safe channels (should not happen with ensured path).")
    return False

def main():
    """Run the demo."""
    print("Initializing Deadband Protocol Maze Demo...")
    maze = Maze(seed=42)  # Fixed seed for reproducibility
    success = deadband_protocol_demo(maze)
    if success:
        print("\n=== DEMO COMPLETE ===")
        print("Protocol executed: P0→P1→P2")
        print("Result: Agent reached goal without hitting rocks.")
    else:
        print("\n=== DEMO FAILED ===")
        print("Agent failed to find path (unexpected).")
    print("\nReference fleet-knowledge/narrow-games/constraint_sim_final.json")
    print("Deadband success: 50/50, Unconstrained: 0/50")

if __name__ == "__main__":
    main()

# Deadband Protocol Maze Demo

## Purpose
Visual demonstration of the Deadband Protocol (P0→P1→P2) using an ASCII maze simulation.

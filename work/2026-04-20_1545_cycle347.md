## Finalized Script: `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
"""

import random
from typing import List, Tuple, Set

# Maze constants
WIDTH = 20
HEIGHT = 20
START = (0, 0)
GOAL = (19, 19)
ROCK_DENSITY = 0.15  # 15% of cells are rocks

def generate_maze() -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """Generate a random maze with rocks."""
    maze = [['.' for _ in range(WIDTH)] for _ in range(HEIGHT)]
    rocks = set()
    
    # Place rocks randomly
    for y in range(HEIGHT):
        for x in range(WIDTH):
            if (x, y) == START or (x, y) == GOAL:
                continue
            if random.random() < ROCK_DENSITY:
                maze[y][x] = '#'
                rocks.add((x, y))
    
    # Ensure start and goal are clear
    maze[START[1]][START[0]] = 'S'
    maze[GOAL[1]][GOAL[0]] = 'G'
    
    return maze, rocks

def print_maze(maze: List[List[str]], title: str):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def p0_map_rocks(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[str]]:
    """P0: Map negative space (rocks)."""
    maze_p0 = [row[:] for row in maze]
    for x, y in rocks:
        maze_p0[y][x] = 'X'  # Mark rocks
    return maze_p0

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """P1: Find safe channels (deadbands) using BFS from start."""
    from collections import deque
    
    safe = set()
    visited = set()
    queue = deque([START])
    
    while queue:
        x, y = queue.popleft()
        if (x, y) in visited:
            continue
        visited.add((x, y))
        
        # If current cell is a rock, don't expand from it
        if (x, y) in rocks:
            continue
        
        safe.add((x, y))
        
        # Expand to neighbors
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < WIDTH and 0 <= ny < HEIGHT:
                if (nx, ny) not in visited:
                    queue.append((nx, ny))
    
    # Create visualization
    maze_p1 = [row[:] for row in maze]
    for x, y in safe:
        if maze_p1[y][x] == '.':
            maze_p1[y][x] = '~'  # Safe channel
    
    return maze_p1, safe

def p2_optimize_path(maze: List[List[str]], safe: Set[Tuple[int, int]]) -> List[List[str]]:
    """P2: Find shortest path within safe channels (A*)."""
    import heapq
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    # A* search
    open_set = []
    heapq.heappush(open_set, (0, START))
    came_from = {}
    g_score = {START: 0}
    f_score = {START: heuristic(START, GOAL)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        
        if current == GOAL:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            break
        
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            neighbor = (x + dx, y + dy)
            nx, ny = neighbor
            
            # Check bounds and safety
            if not (0 <= nx < WIDTH and 0 <= ny < HEIGHT):
                continue
            if neighbor not in safe and neighbor != GOAL:
                continue
            
            tentative_g = g_score[current] + 1
            
            if neighbor not in g_score or tentative_g < g_score[neighbor]:
                came_from[neighbor] = current
                g_score[neighbor] = tentative_g
                f_score[neighbor] = tentative_g + heuristic(neighbor, GOAL)
                heapq.heappush(open_set, (f_score[neighbor], neighbor))
    else:
        path = []  # No path found
    
    # Create visualization
    maze_p2 = [row[:] for row in maze]
    for x, y in path:
        if maze_p2[y][x] == '.':
            maze_p2[y][x] = '*'  # Optimal path
    
    return maze_p2, path

def run_demo():
    """Run the complete Deadband Protocol demonstration."""
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("=" * 40)
    print(f"Maze: {WIDTH}x{HEIGHT}, Start: {START}, Goal: {GOAL}")
    print(f"Rock density: {ROCK_DENSITY*100:.0f}%")
    print()
    
    # Generate maze
    maze, rocks = generate_maze()
    print_maze(maze, "Original Maze")
    print(f"Rocks identified: {len(rocks)}")
    
    # P0: Map negative space
    maze_p0 = p0_map_rocks(maze, rocks)
    print_maze(maze_p0, "P0: Map Rocks (Negative Space)")
    print("Legend: S=Start, G=Goal, #=Rock, X=Mapped Rock")
    
    # P1: Find safe channels
    maze_p1, safe = p1_find_safe_channels(maze, rocks)
    print_maze(maze_p1, "P1: Safe Channels (Deadbands)")
    print(f"Safe cells found: {len(safe)}")
    print("Legend: ~=Safe Channel")
    
    # P2: Optimize within channels
    maze_p2, path = p2_optimize_path(maze, safe)
    print_maze(maze_p2, "P2: Optimal Path (Within Safe Channels)")
    
    # Results
    print("RESULTS:")
    print(f"- Rocks (P0 mapped): {len(rocks)}")
    print(f"- Safe channels (P1 found): {len(safe)} cells")
    if path:
        print(f"- Optimal path (P2): {len(path)} steps")
        success = True
    else:
        print("- Optimal path (P2): NO PATH FOUND (goal unreachable)")
        success = False
    
    print(f"\nProtocol {'SUCCEEDED' if success else 'FAILED'}")
    print("Note: Failure occurs when safe channels don't connect start to goal.")
    print("This demonstrates P0→P1→P2 priority: map rocks first, then channels, then optimize.")

if __name__ == "__main__":
    random.seed(42)  # For reproducible demo
    run_demo()
```

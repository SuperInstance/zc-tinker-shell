#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 in a 20x20 ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import random
from collections import deque

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # From "rocks": 64
OPTIMAL_PATH_LENGTH = 21  # From "optimal": 21

def generate_maze(size, rocks):
    """Generate a random maze with rocks, start, and goal."""
    maze = [['.' for _ in range(size)] for _ in range(size)]
    
    # Place rocks randomly
    rock_positions = set()
    while len(rock_positions) < rocks:
        x, y = random.randint(0, size-1), random.randint(0, size-1)
        rock_positions.add((x, y))
    
    for x, y in rock_positions:
        maze[x][y] = '#'
    
    # Place start (top-left) and goal (bottom-right)
    start = (0, 0)
    goal = (size-1, size-1)
    maze[start[0]][start[1]] = 'S'
    maze[goal[0]][goal[1]] = 'G'
    
    # Ensure start and goal are not rocks
    if start in rock_positions:
        maze[start[0]][start[1]] = 'S'
    if goal in rock_positions:
        maze[goal[0]][goal[1]] = 'G'
    
    return maze, start, goal, rock_positions

def print_maze(maze, title="Maze"):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def p0_map_negative_space(maze, rocks):
    """P0: Identify all rock positions (negative space)."""
    print("P0: MAP NEGATIVE SPACE")
    print(f"Identified {len(rocks)} rock positions (states to avoid).")
    
    # Create visualization with rocks marked
    p0_viz = [row[:] for row in maze]
    for x, y in rocks:
        p0_viz[x][y] = '#'  # Already marked, but ensure visibility
    
    print_maze(p0_viz, "P0 - Rocks (Negative Space)")
    return list(rocks)

def p1_find_safe_channels(maze, start, goal, rocks):
    """P1: Find all safe paths (channels) avoiding rocks using BFS."""
    print("P1: FIND SAFE CHANNELS")
    
    # BFS to find all shortest safe paths
    queue = deque([(start, [start])])
    visited = {start}
    safe_paths = []
    
    while queue:
        (x, y), path = queue.popleft()
        
        if (x, y) == goal:
            safe_paths.append(path)
            continue
        
        # Explore neighbors (up, down, left, right)
        for dx, dy in [(-1,0), (1,0), (0,-1), (0,1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    # Create visualization of safe channels (all visited cells)
    p1_viz = [row[:] for row in maze]
    for x, y in visited:
        if p1_viz[x][y] == '.':
            p1_viz[x][y] = '~'  # Safe water
    
    print(f"Found {len(safe_paths)} safe channel(s) to goal.")
    print_maze(p1_viz, "P1 - Safe Channels (~ = safe water)")
    return safe_paths

def p2_optimize_within_channel(safe_paths):
    """P2: Select shortest path from safe channels."""
    print("P2: OPTIMIZE WITHIN CHANNEL")
    
    if not safe_paths:
        print("ERROR: No safe channels found (P1 failed).")
        return None
    
    # Find shortest path
    shortest_path = min(safe_paths, key=len)
    print(f"Selected optimal path length: {len(shortest_path)-1} steps")
    
    # Check if optimal matches fleet data
    if len(shortest_path)-1 == OPTIMAL_PATH_LENGTH:
        print(f"✓ Matches fleet optimal: {OPTIMAL_PATH_LENGTH}")
    else:
        print(f"✗ Fleet optimal: {OPTIMAL_PATH_LENGTH}, Found: {len(shortest_path)-1}")
    
    return shortest_path

def visualize_final_path(maze, path, rocks):
    """Visualize the final chosen path."""
    p2_viz = [row[:] for row in maze]
    
    # Mark path (excluding start and goal)
    for i, (x, y) in enumerate(path):
        if p2_viz[x][y] not in ['S', 'G']:
            p2_viz[x][y] = 'O' if i == 0 else '*'  # O for start, * for path
    
    print_maze(p2_viz, "P2 - Optimal Path (* = path, O = start)")

def run_deadband_simulation():
    """Run one complete Deadband Protocol simulation."""
    print("\n" + "="*50)
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("="*50)
    
    # Generate maze
    maze, start, goal, rocks = generate_maze(MAZE_SIZE, ROCK_COUNT)
    print_maze(maze, "Initial Maze (S=start, G=goal, #=rock)")
    
    # Step P0: Map negative space
    rocks_list = p0_map_negative_space(maze, rocks)

                print("✓ RESULT: Deadband Protocol SUCCESS")
                return True
            else:

                return True
        else:
            print("✗ RESULT: P2 optimization failed")
            return False
    else:
        print("✗ RESULT: No safe channels (P1 failed) - Protocol violation if P2 attempted")
        return False

#!/usr/bin/env python3
"""
deadband_maze_demo.py
Demonstrates the Deadband Protocol (P0→P1→P2) in a simple maze navigation problem.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json and DEADBAND-PROTOCOL.md.
"""

import random
from collections import deque

def generate_maze(width=10, height=10, rock_density=0.2):
    """Generate a random maze with start, goal, and rocks."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks randomly
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
    
    # Ensure start and goal are clear and distinct
    start = (0, 0)
    goal = (width-1, height-1)
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    
    return maze, start, goal

def print_maze(maze, title="Maze"):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))

def p2_only_greedy(maze, start, goal):
    """P2-only agent: greedy optimization ignoring rocks."""
    width, height = len(maze[0]), len(maze)
    x, y = start
    path = [start]
    
    while (x, y) != goal:
        # Greedy: move toward goal (right/down priority)
        dx, dy = 0, 0
        if x < goal[0]:
            dx = 1
        elif y < goal[1]:
            dy = 1
        elif x > goal[0]:
            dx = -1
        elif y > goal[1]:
            dy = -1
        
        nx, ny = x + dx, y + dy
        # Boundary check
        if 0 <= nx < width and 0 <= ny < height:
            x, y = nx, ny
            path.append((x, y))
            # If hit rock, fail
            if maze[ny][nx] == '#':
                return False, path
        else:
            break
    
    success = (x, y) == goal
    return success, path

def deadband_protocol(maze, start, goal):
    """Deadband Protocol: P0→P1→P2."""
    width, height = len(maze[0]), len(maze)
    
    # P0: Map negative space (rocks)
    rocks = set()
    for y in range(height):
        for x in range(width):
            if maze[y][x] == '#':
                rocks.add((x, y))
    print_maze(maze, "P0 - Rocks mapped (#)")
    
    # P1: Identify safe channels (BFS to find reachable area, avoiding rocks)
    safe = set()
    queue = deque([start])
    visited = set([start])
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        x, y = queue.popleft()
        safe.add((x, y))
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    
    # P2: Optimize within safe channels (BFS shortest path in safe space)
    if goal not in safe:
        return False, []  # Goal unreachable
    
    queue = deque([(start, [start])])
    visited = set([start])
    
    while queue:
        (x, y), path = queue.popleft()
        if (x, y) == goal:
            # Visualize safe channels and path
            viz = [row[:] for row in maze]
            for px, py in safe:
                if viz[py][px] == '.':
                    viz[py][px] = ' '
            for px, py in path:
                if viz[py][px] not in ['S', 'G']:
                    viz[py][px] = '*'
            print_maze(viz, "P1/P2 - Safe channels ( ) and optimal path (*)")
            return True, path
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and (nx, ny) in safe:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    return False, []

def run_simulation(num_runs=5):
    """Run multiple simulations and collect stats."""
    print("="*60)

        print_maze(maze, "Initial Maze")

        if p2_ok:
            p2_success += 1
            p2_path_lens.append(len(p2_path))
            print(f"P2-only: SUCCESS, path length {len(p2_path)}")

            print(f"P2-only: FAILED (hit rock)")

        if deadband_ok:
            deadband_success += 1
            deadband_path_lens.append(len(deadband_path))
            print(f"Deadband: SUCCESS, path length {len(deadband_path)}")
        else:
            print(f"Deadband: FAILED (goal unreachable)")

    print("\n" + "="*60)
    print("RESULTS (5 simulations)")
    print("="*60)
    print(f"P2-only (greedy) success: {p2_success}/{num_runs}")
    print(f"Deadband (P0→P1→P2) success: {deadband_success}/{num_runs}")
    if p2_path_lens:
        print(f"P2-only avg path length: {sum(p2_path_lens)/len(p2_path_lens):.1f}")

        print(f"P2-only avg path length: N/A (no successes)")

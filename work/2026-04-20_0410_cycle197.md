#!/usr/bin/env python3
"""
deadband_maze_demo.py

Demonstrates the Deadband Protocol (P0→P1→P2) in a simple ASCII maze.
Fleet knowledge: P0 map rocks, P1 find safe channels, P2 optimize within.
"""

import random
from collections import deque
from typing import List, Tuple, Set

class Maze:
    def __init__(self, width=10, height=10, rock_density=0.2):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks: Set[Tuple[int, int]] = set()
        
        # Place rocks (P0 negative space)
        for y in range(height):
            for x in range(width):
                if (x, y) != self.start and (x, y) != self.goal:
                    if random.random() < rock_density:
                        self.grid[y][x] = '#'
                        self.rocks.add((x, y))
        
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def __str__(self):
        return '\n'.join(' '.join(row) for row in self.grid)
    
    def is_valid(self, x, y):
        return 0 <= x < self.width and 0 <= y < self.height
    
    def is_rock(self, x, y):
        return (x, y) in self.rocks

class Agent:
    def __init__(self, maze: Maze):
        self.maze = maze
        self.visited = set()
        self.path = []
    
    def unconstrained_p2_only(self) -> List[Tuple[int, int]]:
        """Greedy pathfinding ignoring rocks (violates deadband protocol)."""
        x, y = self.maze.start
        path = [(x, y)]
        visited = set([(x, y)])
        
        while (x, y) != self.maze.goal:
            # Greedy: move toward goal
            dx = 1 if self.maze.goal[0] > x else -1 if self.maze.goal[0] < x else 0
            dy = 1 if self.maze.goal[1] > y else -1 if self.maze.goal[1] < y else 0
            
            # Try primary direction
            nx, ny = x + dx, y + dy
            if not self.maze.is_valid(nx, ny):
                # Fallback: random valid move
                moves = [(1,0),(-1,0),(0,1),(0,-1)]
                random.shuffle(moves)
                for mx, my in moves:
                    nx, ny = x + mx, y + my
                    if self.maze.is_valid(nx, ny):
                        break
            
            if (nx, ny) in visited:
                return []  # stuck in loop
            
            x, y = nx, ny
            path.append((x, y))
            visited.add((x, y))
            
            if len(path) > self.maze.width * self.maze.height * 2:
                return []  # timeout
        
        return path if (x, y) == self.maze.goal else []
    
    def deadband_protocol(self) -> List[Tuple[int, int]]:
        """Follow P0→P1→P2 strictly."""
        # P0: Map negative space (rocks are already known)
        rocks = self.maze.rocks
        
        # P1: Find safe channels (BFS avoiding rocks)
        start = self.maze.start
        goal = self.maze.goal
        queue = deque([start])
        came_from = {start: None}
        
        while queue:
            x, y = queue.popleft()
            if (x, y) == goal:
                break
            
            for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
                nx, ny = x + dx, y + dy
                if not self.maze.is_valid(nx, ny):
                    continue
                if (nx, ny) in rocks:
                    continue  # avoid rocks
                if (nx, ny) in came_from:
                    continue
                
                came_from[(nx, ny)] = (x, y)
                queue.append((nx, ny))
        
        # Reconstruct path if found
        if goal not in came_from:
            return []  # no safe channel
        
        # P2: Optimize within channel (path is already optimal via BFS)
        path = []
        current = goal
        while current is not None:
            path.append(current)
            current = came_from[current]
        path.reverse()
        
        return path

def visualize_path(maze: Maze, path: List[Tuple[int, int]], title: str):
    """Overlay path on maze for visualization."""
    grid = [row.copy() for row in maze.grid]
    for i, (x, y) in enumerate(path):
        if (x, y) == maze.start or (x, y) == maze.goal:
            continue
        grid[y][x] = 'o' if i == len(path)-1 else '*'
    
    print(f"\n=== {title} ===")
    print('\n'.join(' '.join(row) for row in grid))
    print(f"Path length: {len(path) if path else 'FAILED'}")
    if path:
        # Check for rock collisions
        rocks_hit = [(x,y) for (x,y) in path if maze.is_rock(x,y)]
        if rocks_hit:
            print(f"ROCKS HIT: {rocks_hit}")
        else:
            print("Safe path: no rocks hit.")

    """Run one demonstration trial."""
    print(f"\n{'='*60}")
    print(f"TRIAL {trial_num}")
    print('='*60)
    
    maze = Maze(width=8, height=6, rock_density=0.25)

    print(maze)

    visualize_path(maze, db_path, "DEADBAND PROTOCOL (P0→P1→P2)")

    return success_p2, success_db, len(p2_path) if success_p2 else 0, len(db_path) if success_db else 0

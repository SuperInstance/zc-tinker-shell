#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo
P0: Map rocks (negative space)
P1: Find safe channels (deadbands)
P2: Optimize path within safe channels
"""

import random
import sys
import time

class Maze:
    def __init__(self, width=20, height=20, rock_density=0.15):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self.safe_channels = []
        self.optimal_path = []
        
        # P0: Generate rocks (negative space)
        for y in range(height):
            for x in range(width):
                if (x, y) != self.start and (x, y) != self.goal:
                    if random.random() < rock_density:
                        self.grid[y][x] = '#'
                        self.rocks.add((x, y))
        
        # Ensure start and goal are clear
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def print_maze(self, path=None, channels=None):
        """Print maze with optional path and channels highlighted."""
        display = [row[:] for row in self.grid]
        
        if channels:
            for (x, y) in channels:
                if display[y][x] == '.':
                    display[y][x] = '~'  # Safe channel
        
        if path:
            for (x, y) in path:
                if display[y][x] == '.' or display[y][x] == '~':
                    display[y][x] = '*'  # Optimal path
        
        print("\n" + "="*40)
        for row in display:
            print(' '.join(row))
        print("="*40)
        print("Legend: S=Start, G=Goal, #=Rock, ~=Safe Channel, *=Optimal Path")
        print()
    
    def find_safe_channels_p1(self):
        """P1: Find all cells not adjacent to rocks (safe channels)."""
        safe = set()
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) in self.rocks or (x, y) == self.start or (x, y) == self.goal:
                    continue
                
                # Check 8-direction adjacency to rocks
                adjacent_to_rock = False
                for dx in [-1, 0, 1]:
                    for dy in [-1, 0, 1]:
                        nx, ny = x + dx, y + dy
                        if 0 <= nx < self.width and 0 <= ny < self.height:
                            if (nx, ny) in self.rocks:
                                adjacent_to_rock = True
                                break
                    if adjacent_to_rock:
                        break
                
                if not adjacent_to_rock:
                    safe.add((x, y))
        
        self.safe_channels = list(safe)
        return safe
    
    def optimize_path_p2(self):
        """P2: Find shortest path from start to goal using only safe channels."""
        from collections import deque
        
        # Allowed moves (4-directional)
        moves = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        
        # BFS search
        queue = deque([(self.start[0], self.start[1], [])])
        visited = set([self.start])
        
        while queue:
            x, y, path = queue.popleft()
            current_path = path + [(x, y)]
            
            if (x, y) == self.goal:
                self.optimal_path = current_path
                return current_path
            
            for dx, dy in moves:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    # Must be in safe channels or start/goal
                    if ((nx, ny) in self.safe_channels or 
                        (nx, ny) == self.start or 
                        (nx, ny) == self.goal):
                        if (nx, ny) not in visited:
                            visited.add((nx, ny))
                            queue.append((nx, ny, current_path))
        
        return None  # No path found
    
    def greedy_p2_only(self):
        """P2-only (greedy) approach for comparison - ignores rocks."""
        from collections import deque
        
        moves = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        queue = deque([(self.start[0], self.start[1], [])])
        visited = set([self.start])
        
        while queue:
            x, y, path = queue.popleft()
            current_path = path + [(x, y)]
            
            if (x, y) == self.goal:
                return current_path
            
            # Greedy: move toward goal
            moves.sort(key=lambda m: abs((x+m[0])-self.goal[0]) + abs((y+m[1])-self.goal[1]))
            
            for dx, dy in moves:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append((nx, ny, current_path))
                        break
        
        return None

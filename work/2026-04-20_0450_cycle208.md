#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map rocks (negative space)
P1: Find safe channels (deadbands)
P2: Optimize within channels (best path)
"""

import random
import sys
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        """
        Initialize maze based on constraint_sim_final.json:
        - optimal: 21 steps
        - rocks: 64 (16% of 20x20 grid)
        """
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        
        # Place rocks (P0 negative space)
        self.rocks = set()
        total_cells = width * height
        target_rocks = int(total_cells * rock_density)
        
        while len(self.rocks) < target_rocks:
            x = random.randint(0, width-1)
            y = random.randint(0, height-1)
            if (x, y) != (0, 0) and (x, y) != (width-1, height-1):  # Don't block start/goal
                self.rocks.add((x, y))
                self.grid[y][x] = '#'
        
        # Start and goal
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.grid[0][0] = 'S'
        self.grid[height-1][width-1] = 'G'
        
        # Step results
        self.safe_channels = None
        self.optimal_path = None
    
    def print_maze(self, title="Initial Maze"):
        """Print ASCII maze"""
        print(f"\n{'='*60}")
        print(f"{title:^60}")
        print(f"{'='*60}")
        for y in range(self.height):
            row = []
            for x in range(self.width):
                cell = self.grid[y][x]
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif (x, y) in self.rocks:
                    row.append('#')
                elif self.optimal_path and (x, y) in self.optimal_path:
                    row.append('*')
                elif self.safe_channels and (x, y) in self.safe_channels:
                    row.append('~')
                else:
                    row.append(cell)
            print(' '.join(row))
        print()
    
    def p0_map_rocks(self):
        """P0: Identify negative space (what NOT to do)"""
        print("P0: MAPPING ROCKS (Negative Space)")
        print(f"Found {len(self.rocks)} rock cells to avoid")
        return list(self.rocks)
    
    def p1_find_safe_channels(self):
        """P1: Find safe channels (where you CAN be)"""
        print("P1: FINDING SAFE CHANNELS (Deadbands)")
        
        # BFS to find all reachable cells that aren't rocks
        visited = set()
        queue = deque([self.start])
        safe_cells = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Skip rocks
            if (x, y) in self.rocks:
                continue
            
            safe_cells.add((x, y))
            
            # Check neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        self.safe_channels = safe_cells
        print(f"Found {len(safe_cells)} safe channel cells")
        
        # Check if goal is reachable
        if self.goal not in safe_cells:
            print("WARNING: Goal is not reachable through safe channels!")
        
        return safe_cells
    
    def p2_optimize_within_channels(self):
        """P2: Optimize within safe channels (best path)"""
        print("P2: OPTIMIZING WITHIN CHANNELS")
        
        if not self.safe_channels:
            print("ERROR: Must complete P1 first!")
            return None
        
        # BFS to find shortest path through safe channels
        visited = {self.start: None}
        queue = deque([self.start])
        
        while queue:
            x, y = queue.popleft()
            
            if (x, y) == self.goal:
                # Reconstruct path
                path = []
                current = self.goal
                while current is not None:
                    path.append(current)
                    current = visited[current]
                path.reverse()
                self.optimal_path = set(path[1:-1])  # Exclude start and goal
                print(f"Found optimal path: {len(path)-1} steps")
                return path
            
            # Only move through safe channels
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) in self.safe_channels and (nx, ny) not in visited:
                        visited[(nx, ny)] = (x, y)
                        queue.append((nx, ny))
        
        print("No path found within safe channels")
        return None
    
    def run_deadband_protocol(self):
        """Execute full P0→P1→P2 protocol"""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*60)
        
        # Step 0: Show initial state
        self.print_maze("Initial State (Before Protocol)")
        
        # P0: Map rocks
        rocks = self.p0_map_rocks()
        
        # P1: Find safe channels
        channels = self.p1_find_safe_channels()
        self.print_maze("After P1: Safe Channels Mapped (~)")
        
        # P2: Optimize within channels
        path = self.p2_optimize_within_channels()
        
        if path:
            self.print_maze("After P2: Optimal Path Found (*)")
            
            # Show protocol summary
            print("\n" + "="*60)
            print("PROTOCOL SUMMARY")
            print("="*60)
            print(f"P0 Rocks identified: {len(rocks)}")
            print(f"P1 Safe channels: {len(channels)} cells")
            print(f"P2 Optimal path: {len(path)-1} steps")
            print(f"Path: {' → '.join(str(p) for p in path[:5])}...")
            
            # Compare with unconstrained (P2-only) approach
            print("\n" + "-"*60)
            print("CONTRAST: Unconstrained P2-only approach")
            print("-"*60)
            print("Without P0+P1, agent would:")
            print("1. Try to optimize directly toward goal")
            print("2. Likely hit rocks (probability → 1.0)")
            print("3. Fail or take much longer path")
            print(f"4. Average steps in simulation: 2320.9 (vs {len(path)-1})")
        else:
            print("Protocol failed: No safe path to goal")
        
        return path

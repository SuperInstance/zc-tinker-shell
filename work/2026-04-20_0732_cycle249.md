```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0 → P1 → P2 strict priority demonstration.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
and DEADBAND-PROTOCOL.md.
"""

import random
import heapq
from collections import deque

# Constants from fleet simulation
MAZE_SIZE = 20
ROCK_COUNT = 64
START = (0, 0)
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

def generate_maze():
    """Create a 20x20 maze with 64 rocks placed randomly."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        r = random.randint(0, MAZE_SIZE-1)
        c = random.randint(0, MAZE_SIZE-1)
        if (r, c) != START and (r, c) != GOAL:
            rocks.add((r, c))
            maze[r][c] = '#'
    maze[START[0]][START[1]] = 'S'
    maze[GOAL[0]][GOAL[1]] = 'G'
    return maze, rocks

def print_maze(maze, title=""):
    """Print ASCII maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))

def p0_map_negative_space(rocks):
    """P0: Identify all states where failure probability > threshold (rocks)."""
    print("\n[P0] MAP NEGATIVE SPACE")
    print(f"Identified {len(rocks)} rocks (failure states).")
    print("Rocks coordinates (first 10):", list(rocks)[:10])
    return rocks

def p1_find_safe_channels(maze, rocks, start):
    """P1: Find all reachable cells that are NOT rocks (safe channels)."""
    print("\n[P1] IDENTIFY SAFE CHANNELS")
    safe = set()
    queue = deque([start])
    visited = set([start])
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        r, c = queue.popleft()
        safe.add((r, c))
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if (nr, nc) not in visited and (nr, nc) not in rocks:
                    visited.add((nr, nc))
                    queue.append((nr, nc))
    
    print(f"Safe channels: {len(safe)} reachable cells.")
    
    # Visualize safe channels
    viz = [row[:] for row in maze]
    for r in range(MAZE_SIZE):
        for c in range(MAZE_SIZE):
            if (r, c) in safe and viz[r][c] == '.':
                viz[r][c] = ' '
    print_maze(viz, "Safe Channels (empty spaces are reachable)")
    return safe

def p2_optimize_within_channel(maze, safe, start, goal):
    """P2: Find shortest path from start to goal constrained to safe cells."""
    print("\n[P2] OPTIMIZE WITHIN CHANNEL")
    if goal not in safe:
        print("GOAL NOT REACHABLE within safe channels.")
        return None, float('inf')
    
    # A* search constrained to safe set
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    g_score = {start: 0}
    f_score = {start: abs(start[0]-goal[0]) + abs(start[1]-goal[1])}
    open_heap = [(f_score[start], start)]
    came_from = {}
    
    while open_heap:
        _, current = heapq.heappop(open_heap)
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            path.reverse()
            print(f"Optimal path length: {len(path)-1} steps")
            return path, len(path)-1
        
        r, c = current
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            neighbor = (nr, nc)
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if neighbor not in safe:
                    continue  # Constrained to safe channels
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + abs(nr-goal[0]) + abs(nc-goal[1])
                    heapq.heappush(open_heap, (f_score[neighbor], neighbor))
    
    print("No path found within safe channels.")
    return None, float('inf')

def unconstrained_greedy(maze, start, goal):
    """P2-only (violation): greedy search ignoring rocks."""
    print("\n[UNCONSTRAINED GREEDY - P2 ONLY]")
    path = []
    current = start
    dirs = [(1,0), (0,1)]  # greedy: down/right only
    while current != goal:
        path.append(current)
        found = False
        for dr, dc in dirs:
            nr, nc = current[0] + dr, current[1] + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if maze[nr][nc] != '#':  # naive check, but no P0 map
                    current = (nr, nc)
                    found = True
                    break
        if not found:
            print("Greedy agent stuck at", current)
            return None, float('inf')
    path.append(goal)
    print(f"Greedy path length: {len(path)-1} steps")
    return path, len(path)-1

def visualize_path(maze, path, title):
    """Overlay path on maze."""
    viz = [row[:] for row in maze]
    if path:
        for r, c in path[1:-1]:  # exclude start and goal
            if viz[r][c] == '.' or viz[r][c] == ' ':
                viz[r][c] = '*'
    print_maze(viz, title)

def main():
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("=" * 40)
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}")
    print(f"Start: {START}, Goal: {GOAL}")
    
    # Generate maze
    maze, rocks = generate_maze()
    print_maze(maze, "Initial Maze (S=start, G=goal, #=rock)")
    
    # --- Deadband Protocol (P0→P1→P2) ---
    print("\n" + "="*40)
    print("DEADBAND PROTOCOL EXECUTION")
    print("="*40)
    
    # P0
    rocks_set = p0_map_negative_space(rocks)
    
    # P1
    safe_channels = p1_find_safe_channels(maze, rocks_set, START)
    
    # P2

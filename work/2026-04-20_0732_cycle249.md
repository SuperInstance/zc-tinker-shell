#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2

Simulates the 20×20 maze experiment from the constraint simulation.
Shows:
  P0: Map rocks (negative space)
  P1: Find safe channels (deadbands)
  P2: Optimize path within safe channels

Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# Simulation constants from the JSON data
MAZE_SIZE = 20
ROCK_COUNT = 64          # rocks = 64
OPTIMAL_PATH_LEN = 21    # optimal path length = 21

class DeadbandMaze:
    def __init__(self, seed: int = 42):
        random.seed(seed)
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        
        # P0: Rocks (negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[x][y] = '#'
        
        # P1: Safe channels (deadbands) — BFS to find reachable area
        self.safe: Set[Tuple[int, int]] = set()
        self._map_safe_channels()
        
        # P2: Optimal path within safe channels
        self.path: List[Tuple[int, int]] = []
        self._find_optimal_path()
    
    def _map_safe_channels(self):
        """P1: Find all cells reachable without hitting rocks."""
        queue = [self.start]
        visited = set()
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            self.safe.add((x, y))
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        queue.append((nx, ny))
    
    def _find_optimal_path(self):
        """P2: A* search constrained to safe cells only."""
        from heapq import heappush, heappop
        
        open_set = []
        heappush(open_set, (0, self.start))
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: self._heuristic(self.start)}
        
        while open_set:
            _, current = heappop(open_set)
            if current == self.goal:
                self.path = self._reconstruct_path(came_from, current)
                return
            
            x, y = current
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                neighbor = (nx, ny)
                if not (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE):
                    continue
                if neighbor in self.rocks:
                    continue  # P0 violation avoided
                if neighbor not in self.safe:
                    continue  # P1 violation avoided
                
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor)
                    heappush(open_set, (f_score[neighbor], neighbor))
        
        # No path found within safe channels
        self.path = []
    
    def _heuristic(self, pos: Tuple[int, int]) -> int:
        """Manhattan distance to goal."""
        return abs(pos[0] - self.goal[0]) + abs(pos[1] - self.goal[1])
    
    def _reconstruct_path(self, came_from, current):
        path = [current]
        while current in came_from:
            current = came_from[current]
            path.append(current)
        path.reverse()
        return path
    
    def display(self, phase: str = "full"):
        """Print maze with phase‑specific overlay."""
        print(f"\n=== Deadband Protocol: {phase.upper()} ===")
        for x in range(MAZE_SIZE):
            row = []
            for y in range(MAZE_SIZE):
                cell = (x, y)
                if cell == self.start:
                    row.append('S')
                elif cell == self.goal:
                    row.append('G')
                elif phase == "p0" and cell in self.rocks:
                    row.append('#')
                elif phase == "p1" and cell in self.safe:
                    row.append('·')
                elif phase == "p2" and cell in self.path:
                    row.append('*')
                elif cell in self.rocks:
                    row.append('#')
                else:
                    row.append('.')
            print(' '.join(row))
        print()
    
    def stats(self):
        """Print simulation statistics matching fleet data."""
        print("=== Simulation Statistics ===")
        print(f"Maze size: {MAZE_SIZE}×{MAZE_SIZE}")
        print(f"Rocks (P0 negative space): {len(self.rocks)}")
        print(f"Safe cells (P1 deadbands): {len(self.safe)}")
        print(f"Path length (P2 optimal): {len(self.path) if self.path else 'NO PATH'}")
        print(f"Optimal possible (from JSON): {OPTIMAL_PATH_LEN}")
        print()

def run_demo():
    """Run the three‑phase demonstration."""
    maze = DeadbandMaze(seed=249)  # seed = current cycle
    
    # P0: Map negative space (rocks)
    maze.display("p0")
    print("P0: Identified rocks (#) — what NOT to step on.")
    
    # P1: Find safe channels (deadbands)
    maze.display("p1")
    print("P1: Mapped safe channels (·) — where the agent CAN be.")
    
    # P2: Optimize within channels
    maze.display("p2")
    if maze.path:
        print(f"P2: Optimal path (*) length = {len(maze.path)}")
        print("   (Constrained to safe channels only)")
    else:
        print("P2: No safe path to goal — deadband is empty.")
    
    maze.stats()
    
    # Compare with fleet simulation results
    print("=== Fleet Simulation Reference ===")
    print("From constraint_sim_final.json:")
    print("  Unconstrained success: 33/50")
    print("  Constrained success:   0/50")
    print("  Deadband success:     50/50")
    print("  Deadband avg length:  21.0")

if __name__ == "__main__":
    run_demo()
```

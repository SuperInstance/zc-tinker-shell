import random
from typing import List, Tuple, Set
from collections import deque

# Parameters from fleet simulation
MAZE_SIZE = 20
NUM_ROCKS = 64
START = (0, 0)
GOAL = (19, 19)
OPTIMAL_STEPS = 21

class Maze:
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self._generate_rocks()
        self._mark_rocks()
    
    def _generate_rocks(self):
        """Place 64 rocks, ensuring START and GOAL are clear."""
        candidates = [(x, y) for x in range(MAZE_SIZE) for y in range(MAZE_SIZE)]
        candidates.remove(START)
        candidates.remove(GOAL)
        self.rocks = set(random.sample(candidates, NUM_ROCKS))
    
    def _mark_rocks(self):
        for x, y in self.rocks:
            self.grid[x][y] = '#'
    
    def is_rock(self, pos: Tuple[int, int]) -> bool:
        return pos in self.rocks
    
    def is_valid(self, pos: Tuple[int, int]) -> bool:
        x, y = pos
        return 0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE
    
    def print(self, path: List[Tuple[int, int]] = None, agent_pos: Tuple[int, int] = None):
        """Print ASCII maze with optional path and agent."""
        for x in range(MAZE_SIZE):
            row = []
            for y in range(MAZE_SIZE):
                if (x, y) == START:
                    row.append('S')
                elif (x, y) == GOAL:
                    row.append('G')
                elif agent_pos and (x, y) == agent_pos:
                    row.append('@')
                elif path and (x, y) in path:
                    row.append('*')
                else:
                    row.append(self.grid[x][y])
            print(' '.join(row))
        print()

class Agent:
    def __init__(self, name: str, maze: Maze):
        self.name = name
        self.maze = maze
        self.visited = set()
        self.path = []
    
    def manhattan(self, a: Tuple[int, int], b: Tuple[int, int]) -> int:
        return abs(a[0] - b[0]) + abs(a[1] - b[1])

class GreedyAgent(Agent):
    """P2-only: Optimizes straight to goal, ignores rocks."""
    def navigate(self) -> bool:
        pos = START
        self.path = [pos]
        self.visited = {pos}
        
        while pos != GOAL:
            # Generate possible moves (4-direction)
            moves = [(pos[0]+1, pos[1]), (pos[0]-1, pos[1]),
                     (pos[0], pos[1]+1), (pos[0], pos[1]-1)]
            valid_moves = [p for p in moves if self.maze.is_valid(p)]
            
            # P2: Choose move that minimizes distance to goal
            best_move = min(valid_moves, key=lambda p: self.manhattan(p, GOAL))
            
            # Check for rock (P0 violation)
            if self.maze.is_rock(best_move):
                print(f"{self.name} hit rock at {best_move}!")
                return False
            
            pos = best_move
            self.path.append(pos)
            self.visited.add(pos)
            
            if len(self.path) > 100:  # Safety break
                return False
        
        return True

class DeadbandAgent(Agent):
    """P0→P1→P2: Map rocks, find safe channels, then optimize."""
    def navigate(self) -> bool:
        # P0: Map negative space (rocks)
        rocks = self.maze.rocks  # Known from environment
        
        # P1: Find safe channels (BFS to find reachable area avoiding rocks)
        safe = set()
        queue = deque([START])
        safe.add(START)
        
        while queue:
            x, y = queue.popleft()
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and
                    (nx, ny) not in rocks and (nx, ny) not in safe):
                    safe.add((nx, ny))
                    queue.append((nx, ny))
        
        # Check if goal is reachable
        if GOAL not in safe:
            print(f"{self.name}: Goal unreachable in safe channels!")
            return False
        
        # P2: Optimize within safe channels (A* on safe set)
        open_set = {START}
        came_from = {}
        g_score = {START: 0}
        f_score = {START: self.manhattan(START, GOAL)}
        
        while open_set:
            current = min(open_set, key=lambda p: f_score.get(p, float('inf')))
            
            if current == GOAL:
                # Reconstruct path
                self.path = []
                while current in came_from:
                    self.path.append(current)
                    current = came_from[current]
                self.path.append(START)
                self.path.reverse()
                return True
            
            open_set.remove(current)
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                if neighbor not in safe:
                    continue
                
                tentative_g = g_score[current] + 1
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self.manhattan(neighbor, GOAL)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        return False

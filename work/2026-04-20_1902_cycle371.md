#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map rocks (obstacles)
P1: Find safe channels (deadbands)
P2: Optimize path within safe channels
"""

import random
from typing import List, Tuple, Set

# Simulation parameters from fleet-knowledge
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

def generate_maze(size: int, rocks: int) -> Tuple[List[List[str]], Tuple[int, int], Tuple[int, int]]:
    """Generate a size x size maze with rocks, start (top-left), goal (bottom-right)."""
    maze = [['.' for _ in range(size)] for _ in range(size)]
    
    # Place rocks randomly (avoid start and goal)
    start = (0, 0)
    goal = (size-1, size-1)
    blocked = {start, goal}
    
    for _ in range(rocks):
        while True:
            r, c = random.randint(0, size-1), random.randint(0, size-1)
            if (r, c) not in blocked:
                maze[r][c] = '#'
                blocked.add((r, c))
                break
    
    maze[start[0]][start[1]] = 'S'
    maze[goal[0]][goal[1]] = 'G'
    return maze, start, goal

def print_maze(maze: List[List[str]], path: Set[Tuple[int, int]] = None):
    """Print maze with optional path overlay."""
    for r in range(len(maze)):
        row = []
        for c in range(len(maze[r])):
            if path and (r, c) in path:
                row.append('*')
            else:
                row.append(maze[r][c])
        print(' '.join(row))
    print()

def p0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Identify negative space (rocks)."""
    rocks = set()
    for r in range(len(maze)):
        for c in range(len(maze[r])):
            if maze[r][c] == '#':
                rocks.add((r, c))
    return rocks

def p1_find_safe_channels(maze: List[List[str]], start: Tuple[int, int], goal: Tuple[int, int], rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
    """P1: Find all safe paths (deadbands) via BFS, avoiding rocks."""
    size = len(maze)
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]  # Right, Down, Left, Up
    
    # BFS to find all reachable positions
    queue = [start]
    visited = {start}
    parent = {start: None}
    
    while queue:
        r, c = queue.pop(0)
        if (r, c) == goal:
            break
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < size and 0 <= nc < size:
                if (nr, nc) not in rocks and (nr, nc) not in visited:
                    visited.add((nr, nc))
                    parent[(nr, nc)] = (r, c)
                    queue.append((nr, nc))
    
    # Reconstruct all safe paths (simplified: one shortest safe path)
    if goal not in parent:
        return []  # No safe path
    
    # Build one safe path (channel)
    path = []
    node = goal
    while node is not None:
        path.append(node)
        node = parent[node]
    path.reverse()
    return [path]

def p2_optimize_within_channel(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """P2: Select shortest path among safe channels."""
    if not safe_paths:
        return []
    # In this simple case, we already have the shortest safe path from BFS
    return min(safe_paths, key=len)

def run_deadband_protocol() -> None:
    """Main demonstration of P0→P1→P2."""
    print("=== DEADBAND PROTOCOL MAZE DEMO ===\n")
    
    # Generate maze matching simulation parameters
    maze, start, goal = generate_maze(MAZE_SIZE, ROCK_COUNT)
    print("Initial Maze (S=start, G=goal, #=rock, .=empty):")
    print_maze(maze)
    
    # P0: Map negative space
    rocks = p0_map_rocks(maze)
    print(f"P0 — Mapped {len(rocks)} rocks (negative space).")
    
    # P1: Find safe channels
    safe_channels = p1_find_safe_channels(maze, start, goal, rocks)
    if not safe_channels:
        print("P1 — NO SAFE CHANNELS FOUND. Agent would fail.")
        return
    print(f"P1 — Found {len(safe_channels)} safe channel(s).")
    
    # P2: Optimize within channel
    optimal_path = p2_optimize_within_channel(safe_channels)
    print(f"P2 — Optimal path length: {len(optimal_path)} steps (theoretical optimum: {OPTIMAL_PATH_LENGTH}).")
    
    # Visualize
    print("Final Path (* = agent path):")
    print_maze(maze, set(optimal_path))
    
    # Protocol validation
    print("=== PROTOCOL VALIDATION ===")
    print(f"P0 completed: {len(rocks) == ROCK_COUNT}")
    print(f"P1 completed: {len(safe_channels) > 0}")
    print(f"P2 completed: {len(optimal_path) >= OPTIMAL_PATH_LENGTH}")
    print(f"Path stays in safe channel: {all(pos not in rocks for pos in optimal_path)}")

if __name__ == "__main__":
    random.seed(42)  # Reproducible for demo
    run_deadband_protocol()
```

# Deadband Protocol Maze Demo

This demo visualizes the Deadband Protocol (P0→P1→P2) using an ASCII maze.

## Parameters
- Maze size: 20×20 (from constraint_sim_final.json)
- Rocks: 64 obstacles
- Optimal path: 21 steps (theoretical minimum)

## Protocol Steps

1. **P0: Map Negative Space**
   - Identify all rock positions (`#`)
   - These are states where failure probability > threshold
   - Without this map, agent cannot proceed safely

2. **P1: Find Safe Channels**
   - Perform BFS from start (`S`) to goal (`G`)
   - Avoid all rocks identified in P0
   - Result: one or more safe paths (deadbands)

3. **P2: Optimize Within Channel**
   - Select the shortest path among safe channels
   - Optimization is constrained to safe space only
   - Never optimize before P0+P1 are complete

## Running the Demo

```bash
python3 deadband_maze_demo.py
```

## Expected Output

The script will:
1. Print the generated maze with rocks
2. Show P0 rock count (should be 64)
3. Show P1 safe channel discovery
4. Show P2 optimal path length (≥21)
5. Display the final path overlaid on the maze
6. Validate protocol adherence

## Key Insight

The simulation data shows:
- Unconstrained optimization (P2-only): 0% success
- Deadband protocol (P0→P1→P2): 100% success

This demo illustrates why mapping negative space (P0) must come before optimization (P2).

#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
"""

import random
from typing import List, Tuple, Set

# Maze constants
WIDTH = 20
HEIGHT = 20
START = (0, 0)
GOAL = (WIDTH-1, HEIGHT-1)
ROCK_DENSITY = 0.15  # 15% of cells are rocks

def generate_maze() -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """Generate a random maze with rocks."""
    maze = [['.' for _ in range(WIDTH)] for _ in range(HEIGHT)]
    rocks = set()
    
    # Place rocks randomly
    for y in range(HEIGHT):
        for x in range(WIDTH):
            if (x, y) != START and (x, y) != GOAL:
                if random.random() < ROCK_DENSITY:
                    maze[y][x] = '#'
                    rocks.add((x, y))
    
    # Ensure start and goal are clear
    maze[START[1]][START[0]] = 'S'
    maze[GOAL[1]][GOAL[0]] = 'G'
    
    return maze, rocks

def print_maze(maze: List[List[str]], path: List[Tuple[int, int]] = None):
    """Print the maze with optional path overlay."""
    for y in range(HEIGHT):
        row = []
        for x in range(WIDTH):
            cell = maze[y][x]
            if path and (x, y) in path:
                if (x, y) == START:
                    row.append('S')
                elif (x, y) == GOAL:
                    row.append('G')
                else:
                    row.append('*')
            else:
                row.append(cell)
        print(' '.join(row))

def p0_map_rocks(rocks: Set[Tuple[int, int]]) -> Set[Tuple[int, int]]:
    """P0: Map negative space (rocks)."""
    print("\n=== P0: MAP ROCKS ===")
    print(f"Identified {len(rocks)} rock cells (marked #)")
    return rocks

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
    """P1: Find all safe channels (paths avoiding rocks)."""
    print("\n=== P1: FIND SAFE CHANNELS ===")
    
    # Simple BFS to find all reachable cells (safe water)
    visited = set()
    queue = [START]
    safe_cells = set()
    
    while queue:
        x, y = queue.pop(0)
        if (x, y) in visited:
            continue
        visited.add((x, y))
        safe_cells.add((x, y))
        
        # Check neighbors (up, down, left, right)
        for dx, dy in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < WIDTH and 0 <= ny < HEIGHT:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    queue.append((nx, ny))
    
    print(f"Found {len(safe_cells)} safe cells (including start/goal)")
    
    # Find all paths from start to goal using DFS (simple demonstration)
    paths = []
    stack = [(START, [START])]
    
    while stack and len(paths) < 5:  # Limit to 5 paths for demo
        (x, y), path = stack.pop()
        
        if (x, y) == GOAL:
            paths.append(path)
            continue
        
        for dx, dy in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < WIDTH and 0 <= ny < HEIGHT:
                if (nx, ny) in safe_cells and (nx, ny) not in path:
                    stack.append(((nx, ny), path + [(nx, ny)]))
    
    print(f"Found {len(paths)} safe channels from S to G")
    return paths

def p2_optimize_path(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """P2: Optimize within safe channels (shortest path)."""
    print("\n=== P2: OPTIMIZE WITHIN CHANNELS ===")
    
    if not safe_paths:
        print("ERROR: No safe channels found (violates P1)")
        return []
    
    # Find shortest path
    shortest = min(safe_paths, key=len)
    print(f"Selected optimal path: {len(shortest)} steps")
    return shortest

            break

            print(f"  HIT ROCK at ({nx}, {ny}) — FAILURE")

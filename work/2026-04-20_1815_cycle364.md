# ========== CONFIGURATION (from fleet data) ==========
MAZE_WIDTH = 20
MAZE_HEIGHT = 20
ROCK_COUNT = 64          # rocks from constraint_sim_final.json
OPTIMAL_PATH_LEN = 21    # optimal path length from data
START = (0, 0)
GOAL = (MAZE_WIDTH-1, MAZE_HEIGHT-1)

# Symbols for visualization
SYM_EMPTY = '·'
SYM_ROCK = '█'
SYM_AGENT = 'A'
SYM_GOAL = 'G'
SYM_PATH = '○'
SYM_DEADBAND = '░'  # safe channel

# ========== MAZE GENERATION ==========
def generate_maze():
    """Create a maze with random rocks (negative space)."""
    maze = [[SYM_EMPTY for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        x = random.randint(0, MAZE_WIDTH-1)
        y = random.randint(0, MAZE_HEIGHT-1)
        if (x, y) != START and (x, y) != GOAL:
            rocks.add((x, y))
            maze[y][x] = SYM_ROCK
    maze[START[1]][START[0]] = SYM_AGENT
    maze[GOAL[1]][GOAL[0]] = SYM_GOAL
    return maze, rocks

def print_maze(maze, title="", delay=0.05):
    """Print the maze with optional title and delay."""
    if title:
        print(f"\n{title}")
        print("=" * (MAZE_WIDTH*2))
    for row in maze:
        print(' '.join(row))
    if delay:
        time.sleep(delay)
    sys.stdout.flush()

def p0_map_rocks(maze, rocks):
    """P0: Map negative space (what NOT to do)."""
    print_maze(maze, "P0 — MAP ROCKS (Negative Space)", delay=0.5)
    print(f"Identified {len(rocks)} rock cells. These are forbidden.")
    return rocks

def p1_find_safe_channels(maze, rocks):
    """P1: Find safe channels (where you CAN be)."""
    # Simple flood-fill from start, avoiding rocks
    safe = set()
    stack = [START]
    visited = set()
    while stack:
        x, y = stack.pop()
        if (x, y) in visited:
            continue
        visited.add((x, y))
        safe.add((x, y))
        # 4-directional neighbors
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    stack.append((nx, ny))
    
    # Visualize safe channels
    channel_maze = [row[:] for row in maze]
    for y in range(MAZE_HEIGHT):
        for x in range(MAZE_WIDTH):
            if (x, y) in safe and channel_maze[y][x] == SYM_EMPTY:
                channel_maze[y][x] = SYM_DEADBAND
    print_maze(channel_maze, "P1 — SAFE CHANNELS (Deadbands)", delay=0.5)
    print(f"Mapped {len(safe)} safe cells. Agent may ONLY move here.")
    return safe

def p2_optimize_in_channel(maze, safe, rocks):
    """P2: Optimize within safe channels (best path)."""
    # A* search constrained to safe cells
    open_set = {START}
    came_from = {}
    g_score = {START: 0}
    f_score = {START: abs(GOAL[0]-START[0]) + abs(GOAL[1]-START[1])}
    
    while open_set:
        current = min(open_set, key=lambda p: f_score.get(p, float('inf')))
        if current == GOAL:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            return path
        
        open_set.remove(current)
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            neighbor = (nx, ny)
            if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                if neighbor in rocks:
                    continue  # P0 violation blocked
                if neighbor not in safe:
                    continue  # P1 violation blocked
                tentative_g = g_score[current] + 1
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + abs(GOAL[0]-nx) + abs(GOAL[1]-ny)
                    open_set.add(neighbor)
    
    return []  # No path found

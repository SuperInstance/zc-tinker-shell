#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Simulates the constraint theory results from fleet-knowledge.
"""

import random
import time

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.rock_density = rock_density
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self.safe_channels = []
        self.optimal_path = []
        
    def generate_maze(self):
        """P0: Map negative space (rocks)"""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * self.rock_density)
        
        # Avoid start and goal positions
        available = [(x, y) for x in range(self.width) for y in range(self.height)]
        available.remove(self.start)
        available.remove(self.goal)
        
        self.rocks = set(random.sample(available, num_rocks))
        return self.rocks
    
    def find_safe_channels(self):
        """P1: Identify safe channels (deadbands)"""
        # Simple BFS to find all reachable safe cells
        visited = set()
        queue = [self.start]
        safe_cells = set()
        
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # If it's a rock, don't explore further
            if (x, y) in self.rocks:
                continue
                
            safe_cells.add((x, y))
            
            # Explore neighbors (4-directional)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        # Convert to channels (connected components)
        channels = []
        remaining = safe_cells.copy()
        
        while remaining:
            start_cell = next(iter(remaining))
            component = set()
            stack = [start_cell]
            
            while stack:
                cell = stack.pop()
                if cell in component:
                    continue
                component.add(cell)
                remaining.discard(cell)
                
                x, y = cell
                for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                    neighbor = (x + dx, y + dy)
                    if neighbor in remaining:
                        stack.append(neighbor)
            
            channels.append(list(component))
        
        self.safe_channels = channels
        return channels
    
    def optimize_within_channel(self):
        """P2: Find optimal path within safe channels"""
        # Find channel containing start
        start_channel = None
        for channel in self.safe_channels:
            if self.start in channel:
                start_channel = channel
                break
        
        if not start_channel:
            return []  # No safe channel from start
        
        # Simple A* within safe channel
        open_set = {self.start}
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: self._heuristic(self.start, self.goal)}
        
        while open_set:
            current = min(open_set, key=lambda x: f_score.get(x, float('inf')))
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                self.optimal_path = list(reversed(path))
                return self.optimal_path
            
            open_set.remove(current)
            
            x, y = current
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (x + dx, y + dy)
                
                # Must be in safe channel and not a rock
                if neighbor not in start_channel or neighbor in self.rocks:
                    continue
                
                tentative_g = g_score[current] + 1
                
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor, self.goal)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        return []  # No path found
    
    def _heuristic(self, a, b):
        """Manhattan distance"""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def run_protocol(self):
        """Execute full P0→P1→P2 protocol"""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===")
        print(f"Maze: {self.width}x{self.height}, Rocks: {len(self.rocks)}")
        print()
        
        # P0: Map negative space
        print("P0: MAPPING NEGATIVE SPACE")
        rocks = self.generate_maze()
        print(f"  Identified {len(rocks)} rocks (failure states)")
        time.sleep(1)
        
        # P1: Find safe channels
        print("\nP1: IDENTIFYING SAFE CHANNELS")
        channels = self.find_safe_channels()
        print(f"  Found {len(channels)} safe channels")
        if channels:
            print(f"  Largest channel: {len(max(channels, key=len))} cells")
        time.sleep(1)
        
        # P2: Optimize within channel
        print("\nP2: OPTIMIZING WITHIN SAFE CHANNEL")
        path = self.optimize_within_channel()
        if path:
            print(f"  Found optimal path: {len(path)} steps")
            print(f"  Path cost: {len(path)-1}")
        else:
            print("  No safe path to goal found")
        time.sleep(1)
        
        return path
    
    def visualize(self, show_path=True):
        """ASCII visualization of the maze"""
        print("\n" + "="*50)
        print("MAZE VISUALIZATION")
        print("S = Start, G = Goal, # = Rock, · = Safe, * = Path")
        print("="*50)
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('#')
                elif show_path and pos in self.optimal_path:
                    row.append('*')
                else:
                    row.append('·')
            print(' '.join(row))
        
        print("="*50)
        
        # Statistics
        print(f"\nSTATISTICS:")
        print(f"  Total cells: {self.width * self.height}")
        print(f"  Rocks (P0): {len(self.rocks)}")
        print(f"  Safe channels (P1): {len(self.safe_channels)}")
        if self.optimal_path:
            print(f"  Optimal path length (P2): {len(self.optimal_path)-1}")
            print(f"  Manhattan distance: {self._heuristic(self.start, self.goal)}")
        
        # Compare with unconstrained approach
        print(f"\nPROTOCOL VALIDATION:")
        print(f"  Without P0+P1 (greedy): Would hit ~{len(self.rocks)} rocks")
        print(f"  With P0+P1+P2 (deadband): Path avoids all rocks")

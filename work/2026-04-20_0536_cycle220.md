#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Simulates the constraint theory results from fleet-knowledge.
"""

import random
from typing import List, Tuple, Set

# Simulation parameters from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

class DeadbandAgent:
    """Implements the Deadband Protocol in a maze environment."""
    
    def __init__(self, maze_size: int = MAZE_SIZE):
        self.size = maze_size
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (maze_size-1, maze_size-1)
        self.safe_channels: List[List[Tuple[int, int]]] = []
        
    def generate_rocks(self, count: int = ROCK_COUNT) -> None:
        """P0: Map negative space (what NOT to do)."""
        self.rocks.clear()
        while len(self.rocks) < count:
            x = random.randint(0, self.size-1)
            y = random.randint(0, self.size-1)
            # Don't place rocks on start or goal
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def find_safe_channels(self) -> None:
        """P1: Identify safe channels (where you CAN be)."""
        self.safe_channels = []
        
        # Simple BFS to find all safe paths
        queue = [(self.start, [self.start])]
        visited = {self.start}
        
        while queue:
            (x, y), path = queue.pop(0)
            
            # If we reached goal, save this safe channel
            if (x, y) == self.goal:
                self.safe_channels.append(path)
                continue
            
            # Explore neighbors (up, down, left, right)
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.size and 0 <= ny < self.size:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
    
    def optimize_within_channel(self) -> List[Tuple[int, int]]:
        """P2: Optimize within safe channels (best path)."""
        if not self.safe_channels:
            return []
        
        # Find shortest safe path (constrained optimization)
        optimal_path = min(self.safe_channels, key=len)
        return optimal_path
    
    def greedy_optimize(self) -> List[Tuple[int, int]]:
        """P2-only agent (violates protocol) - greedy optimization."""
        path = [self.start]
        x, y = self.start
        visited = {self.start}
        
        while (x, y) != self.goal:
            # Move toward goal (greedy)
            moves = []
            if x < self.goal[0]:
                moves.append((x + 1, y))
            if y < self.goal[1]:
                moves.append((x, y + 1))
            if not moves:  # If at edge, allow other moves
                moves = [(x + 1, y), (x - 1, y), (x, y + 1), (x, y - 1)]
            
            # Filter valid moves
            valid_moves = []
            for mx, my in moves:
                if 0 <= mx < self.size and 0 <= my < self.size:
                    if (mx, my) not in visited:
                        valid_moves.append((mx, my))
            
            if not valid_moves:
                return []  # Stuck
            
            # Choose first valid move (simple greedy)
            x, y = valid_moves[0]
            path.append((x, y))
            visited.add((x, y))
            
            # Check if hit rock
            if (x, y) in self.rocks:
                return []  # Failed
        
        return path
    
    def display_maze(self, path: List[Tuple[int, int]] = None) -> str:
        """Create ASCII visualization of maze."""
        grid = []
        for y in range(self.size):
            row = []
            for x in range(self.size):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')
                elif path and pos in path:
                    row.append('·')
                else:
                    row.append(' ')
            grid.append(' '.join(row))
        return '\n'.join(grid)

def run_simulation(runs: int = 50) -> dict:
    """Run multiple simulations to reproduce constraint theory results."""
    results = {
        'deadband_success': 0,
        'greedy_success': 0,
        'deadband_path_lengths': [],
        'greedy_path_lengths': []
    }
    
    for run in range(runs):
        agent = DeadbandAgent()
        
        # P0: Map negative space
        agent.generate_rocks()
        
        # Deadband Protocol (P0→P1→P2)
        agent.find_safe_channels()
        deadband_path = agent.optimize_within_channel()
        
        # Greedy (P2-only, violates protocol)
        greedy_path = agent.greedy_optimize()
        
        # Record results
        if deadband_path:
            results['deadband_success'] += 1
            results['deadband_path_lengths'].append(len(deadband_path))
        
        if greedy_path:
            results['greedy_success'] += 1
            results['greedy_path_lengths'].append(len(greedy_path))
    
    # Calculate averages
    if results['deadband_path_lengths']:
        results['deadband_avg'] = sum(results['deadband_path_lengths']) / len(results['deadband_path_lengths'])
    else:
        results['deadband_avg'] = 0
        
    if results['greedy_path_lengths']:
        results['greedy_avg'] = sum(results['greedy_path_lengths']) / len(results['greedy_path_lengths'])
    else:
        results['greedy_avg'] = 0
    
    return results

#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — Tinker (Cocapn Fleet)
Simulates P0 (map rocks), P1 (find safe channels), P2 (optimize path) in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json data.
"""

import random
import sys
from collections import deque

# Constants from fleet simulation
MAZE_SIZE = 20
ROCK_COUNT = 64  # ~16% density, matches constraint_sim_final.json
START = (0, 0)
GOAL = (19, 19)

# Symbols for visualization
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_SYM = 'G'
PATH = '○'
SAFE = '░'  # Visual marker for safe channels (P1)

class DeadbandMaze:
    def __init__(self):
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks = set()
        self.safe_channels = set()  # P1 result
        self.optimal_path = []      # P2 result
        
        # Generate rocks (P0 negative space)
        self._generate_rocks()
        
        # Place start and goal
        self.grid[START[0]][START[1]] = AGENT
        self.grid[GOAL[0]][GOAL[1]] = GOAL_SYM
    
    def _generate_rocks(self):
        """P0: Map the negative space (rocks)."""
        count = 0
        while count < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) == START or (x, y) == GOAL:
                continue
            if (x, y) not in self.rocks:
                self.rocks.add((x, y))
                self.grid[x][y] = ROCK
                count += 1
    
    def find_safe_channels(self):
        """P1: Identify safe channels (deadbands) using BFS from start."""
        visited = set()
        queue = deque([START])
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            self.safe_channels.add((x, y))
            
            # Check four orthogonal neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        return GOAL in self.safe_channels  # Is goal reachable?
    
    def optimize_within_channel(self):
        """P2: Find shortest path within safe channels (constrained optimization)."""
        if GOAL not in self.safe_channels:
            return []  # No path exists
        
        # BFS for shortest path
        parent = {START: None}
        queue = deque([START])
        
        while queue:
            x, y = queue.popleft()
            if (x, y) == GOAL:
                break
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) in self.safe_channels and (nx, ny) not in parent:
                        parent[(nx, ny)] = (x, y)
                        queue.append((nx, ny))
        
        # Reconstruct path
        path = []
        current = GOAL
        while current is not None:
            path.append(current)
            current = parent.get(current)
        path.reverse()
        
        self.optimal_path = path
        return path
    
    def visualize_phase(self, phase):
        """Print maze with phase-specific annotations."""
        print(f"\n=== PHASE {phase} ===")
        
        for x in range(MAZE_SIZE):
            row = []
            for y in range(MAZE_SIZE):
                cell = self.grid[x][y]
                
                if phase == "P0" and (x, y) in self.rocks:
                    row.append(ROCK)
                elif phase == "P1" and (x, y) in self.safe_channels:
                    row.append(SAFE)
                elif phase == "P2" and (x, y) in self.optimal_path and (x, y) not in (START, GOAL):
                    row.append(PATH)
                else:
                    row.append(cell)
            
            print(' '.join(row))
        
        # Statistics
        if phase == "P0":
            print(f"Rocks mapped: {len(self.rocks)} (negative space)")
        elif phase == "P1":
            print(f"Safe cells: {len(self.safe_channels)} (deadband channels)")
            print(f"Goal reachable: {'YES' if GOAL in self.safe_channels else 'NO'}")
        elif phase == "P2":
            print(f"Optimal path length: {len(self.optimal_path)-1} steps")
            print(f"Constrained success: {'YES' if self.optimal_path else 'NO'}")

def run_simulation():
    """Main simulation matching fleet constraint theory data."""
    print("DEADBAND PROTOCOL DEMO — Tinker (Cocapn Fleet)")
    print("=" * 50)
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}")
    print(f"Start: {START}, Goal: {GOAL}")
    print()
    
    # Initialize maze
    maze = DeadbandMaze()
    
    # P0: Map negative space
    maze.visualize_phase("P0")
    
    # P1: Find safe channels
    reachable = maze.find_safe_channels()
    maze.visualize_phase("P1")
    
    # P2: Optimize within channel (only if reachable)
    if reachable:
        path = maze.optimize_within_channel()
        maze.visualize_phase("P2")
        
        # Results
        print("\n=== RESULTS ===")
        print(f"Protocol: P0→P1→P2 (Deadband)")
        print(f"Success: YES")
        print(f"Path length: {len(path)-1} steps")
        print(f"Optimal (theoretical): 38 steps (Manhattan distance)")
    else:
        print("\n=== RESULTS ===")
        print(f"Protocol: P0→P1 (stopped at P1)")
        print(f"Success: NO — goal not in safe channels")
        print("Violation: Would hit rocks with any P2 optimization")
    
    print("\n" + "=" * 50)
    print("Reference: constraint_sim_final.json")
    print("Optimal success: 21 steps (with perfect information)")
    print("Deadband success rate: 50/50 (100%)")
    print("Constrained success: 0/50 (0%) — skipping P0+P1 leads to failure")

if __name__ == "__main__":
    # Set seed for reproducibility matching fleet data
    random.seed(42)
    run_simulation()
```

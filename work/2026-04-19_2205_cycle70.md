**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Complete. Based on `constraint_sim_final.json` (20×20 maze, 64 rocks, optimal path length 21). Implemented a simulation that strictly follows the Deadband Protocol: P0 (map rocks), P1 (find safe channels via BFS), P2 (optimize path within safe channels using A*). The script outputs an ASCII visualization of the maze, the safe channel map, and the final optimized path.

**Output:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (🔧)
A visual demonstration of P0→P1→P2 using a 20x20 maze with rocks.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque
import heapq

# Simulation parameters from constraint_sim_final.json
MAZE_SIZE = 20
NUM_ROCKS = 64
OPTIMAL_PATH_LENGTH = 21

# Symbols for visualization
ROCK = '█'
OPEN = '·'
AGENT = 'A'
GOAL = 'G'
PATH = '○'
SAFE = '░'  # Visual indicator of safe channel

def generate_maze(size, rocks):
    """Generate a random maze with rocks as obstacles."""
    maze = [[OPEN for _ in range(size)] for _ in range(size)]
    
    # Place rocks randomly (but not on start (0,0) or goal (size-1, size-1))
    placed = 0
    while placed < rocks:
        x = random.randint(0, size-1)
        y = random.randint(0, size-1)
        if (x == 0 and y == 0) or (x == size-1 and y == size-1):
            continue
        if maze[y][x] == OPEN:
            maze[y][x] = ROCK
            placed += 1
    
    # Set start and goal
    maze[0][0] = AGENT
    maze[size-1][size-1] = GOAL
    
    return maze

def print_maze(maze, title="MAZE"):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def deadband_protocol_demo(maze):
    """Execute Deadband Protocol: P0→P1→P2."""
    size = len(maze)
    
    # P0: MAP THE NEGATIVE SPACE (Identify rocks)
    rocks = []
    for y in range(size):
        for x in range(size):
            if maze[y][x] == ROCK:
                rocks.append((x, y))
    
    print(f"P0: Mapped {len(rocks)} rocks (negative space)")
    
    # P1: IDENTIFY SAFE CHANNELS (BFS to find reachable areas)
    start = (0, 0)
    goal = (size-1, size-1)
    
    # BFS to find all reachable cells from start
    visited = [[False for _ in range(size)] for _ in range(size)]
    queue = deque([start])
    visited[start[1]][start[0]] = True
    
    # Directions: up, down, left, right
    directions = [(0, -1), (0, 1), (-1, 0), (1, 0)]
    
    while queue:
        x, y = queue.popleft()
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            
            if 0 <= nx < size and 0 <= ny < size:
                if not visited[ny][nx] and maze[ny][nx] != ROCK:
                    visited[ny][nx] = True
                    queue.append((nx, ny))
    
    # Check if goal is reachable
    if not visited[goal[1]][goal[0]]:
        print("P1: No safe channel to goal! Protocol stops at P1.")
        return None
    
    # Create safe channel visualization
    safe_maze = [row[:] for row in maze]
    safe_cells = 0
    for y in range(size):
        for x in range(size):
            if visited[y][x] and maze[y][x] == OPEN:
                safe_maze[y][x] = SAFE
                safe_cells += 1
    
    print(f"P1: Found safe channel with {safe_cells} reachable cells")
    print_maze(safe_maze, "SAFE CHANNELS (P1)")
    
    # P2: OPTIMIZE WITHIN CHANNEL (A* search within safe area)
    # Heuristic: Manhattan distance
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    # A* search only through visited (safe) cells
    open_set = []
    heapq.heappush(open_set, (0, start))
    
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        current = heapq.heappop(open_set)[1]
        
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            
            print(f"P2: Found optimal path of length {len(path)} within safe channel")
            
            # Create final visualization
            final_maze = [row[:] for row in maze]
            for x, y in path:
                if final_maze[y][x] == OPEN:
                    final_maze[y][x] = PATH
            
            print_maze(final_maze, "OPTIMAL PATH (P2)")
            return path
        
        for dx, dy in directions:
            neighbor = (current[0] + dx, current[1] + dy)
            
            # Check bounds and safety
            if not (0 <= neighbor[0] < size and 0 <= neighbor[1] < size):
                continue
            if not visited[neighbor[1]][neighbor[0]]:
                continue  # Not in safe channel
            
            tentative_g = g_score[current] + 1
            
            if neighbor not in g_score or tentative_g < g_score[neighbor]:
                came_from[neighbor] = current
                g_score[neighbor] = tentative_g
                f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    print("P2: No path found within safe channel (should not happen if P1 passed)")
    return None

def main():
    """Run the Deadband Protocol demonstration."""
    print("=" * 50)
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("=" * 50)
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {NUM_ROCKS}")
    print(f"Optimal path from constraint_sim_final.json: {OPTIMAL_PATH_LENGTH}")
    print()
    
    # Generate maze matching simulation parameters
    random.seed(42)  # Deterministic for demo
    maze = generate_maze(MAZE_SIZE, NUM_ROCKS)
    
    print_maze(maze, "INITIAL MAZE (P0 Input)")
    
    # Run Deadband Protocol
    path = deadband_protocol_demo(maze)
    
    # Results
    print("\n" + "=" * 50)
    print("RESULTS SUMMARY")
    print("=" * 50)
    if path:
        print(f"✓ Protocol SUCCESS")
        print(f"  P0: Identified {NUM_ROCKS} rocks")
        print(f"  P1: Mapped safe channels (BFS reachability)")
        print(f"  P2: Optimized path length: {len(path)}")
        print(f"  Reference optimal: {OPTIMAL_PATH_LENGTH}")
    else:
        print("✗ Protocol FAILED at P1 (no safe channel to goal)")
    
    print("\nProtocol phases strictly enforced:")
    print("  P0 → P1 → P2 (no skipping, no P2 without P1)")

if __name__ == "__main__":
    main()
```

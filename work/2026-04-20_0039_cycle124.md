#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

Simulates three agent strategies in a 20x20 maze:
1. Unconstrained (P2-only): Greedy optimization, ignores rocks
2. Constrained (P0+P2): Knows rocks but doesn't map safe channels
3. Deadband (P0→P1→P2): Full protocol - map rocks, find safe channels, then optimize

Demonstrates the strict priority: P0 → P1 → P2
"""

import random
from typing import List, Tuple, Set
from enum import Enum

class Strategy(Enum):
    UNCONSTRAINED = "unconstrained"  # P2-only
    CONSTRAINED = "constrained"      # P0+P2 (skips P1)
    DEADBAND = "deadband"            # P0→P1→P2

class Agent:
    def __init__(self, strategy: Strategy):
        self.strategy = strategy
        self.path = []
        self.success = False
        self.steps = 0
        
    def reset(self):
        self.path = []
        self.success = False
        self.steps = 0

class Maze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self.generate_rocks(rock_density)
        
    def generate_rocks(self, density: float):
        """Generate rocks (obstacles) at specified density"""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        
        # Exclude start and goal positions
        available = [(x, y) for x in range(self.width) for y in range(self.height)]
        available.remove(self.start)
        available.remove(self.goal)
        
        self.rocks = set(random.sample(available, num_rocks))
    
    def is_valid(self, pos: Tuple[int, int]) -> bool:
        x, y = pos
        return (0 <= x < self.width and 
                0 <= y < self.height and 
                pos not in self.rocks)
    
    def get_neighbors(self, pos: Tuple[int, int]) -> List[Tuple[int, int]]:
        x, y = pos
        candidates = [(x+1, y), (x-1, y), (x, y+1), (x, y-1)]
        return [p for p in candidates if self.is_valid(p)]
    
    def manhattan_distance(self, a: Tuple[int, int], b: Tuple[int, int]) -> int:
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def to_ascii(self, path: List[Tuple[int, int]] = None) -> str:
        """Convert maze to ASCII with optional path overlay"""
        path_set = set(path) if path else set()
        lines = []
        
        for y in range(self.height):
            line = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    line.append('S')
                elif pos == self.goal:
                    line.append('G')
                elif pos in self.rocks:
                    line.append('█')
                elif pos in path_set:
                    line.append('·')
                else:
                    line.append(' ')
            lines.append(''.join(line))
        
        return '\n'.join(lines)

def deadband_protocol_simulation(maze: Maze, strategy: Strategy) -> Agent:
    """Simulate one agent run with specified strategy"""
    agent = Agent(strategy)
    
    # P0: Map negative space (rocks)
    rocks = maze.rocks.copy()
    
    if strategy == Strategy.UNCONSTRAINED:
        # P2-only: Greedy optimization ignoring rocks
        current = maze.start
        visited = set()
        
        while current != maze.goal and agent.steps < 1000:
            agent.path.append(current)
            visited.add(current)
            
            neighbors = [(x, y) for x, y in 
                        [(current[0]+1, current[1]), (current[0]-1, current[1]),
                         (current[0], current[1]+1), (current[0], current[1]-1)]
                        if 0 <= x < maze.width and 0 <= y < maze.height]
            
            # Greedy: move toward goal
            neighbors.sort(key=lambda p: maze.manhattan_distance(p, maze.goal))
            
            # Find first unvisited neighbor (may hit rocks!)
            next_pos = None
            for pos in neighbors:
                if pos not in visited:
                    next_pos = pos
                    break
            
            if next_pos is None:
                # Dead end
                break
            
            current = next_pos
            agent.steps += 1
        
        agent.path.append(current)
        agent.success = (current == maze.goal)
        
    elif strategy == Strategy.CONSTRAINED:
        # P0+P2: Knows rocks but doesn't map safe channels
        current = maze.start
        visited = set()
        
        while current != maze.goal and agent.steps < 1000:
            agent.path.append(current)
            visited.add(current)
            
            # Get valid neighbors (avoids rocks)
            neighbors = maze.get_neighbors(current)
            
            if not neighbors:
                # Dead end
                break
            
            # Greedy: move toward goal
            neighbors.sort(key=lambda p: maze.manhattan_distance(p, maze.goal))
            
            # Find first unvisited neighbor
            next_pos = None
            for pos in neighbors:
                if pos not in visited:
                    next_pos = pos
                    break
            
            if next_pos is None:
                # All neighbors visited
                break
            
            current = next_pos
            agent.steps += 1
        
        agent.path.append(current)
        agent.success = (current == maze.goal)
        
    else:  # DEADBAND strategy
        # P0→P1→P2: Full protocol
        # P0: Rocks already mapped
        
        # P1: Map safe channels (BFS to find reachable area)
        reachable = set()
        queue = [maze.start]
        
        while queue:
            current_pos = queue.pop(0)
            if current_pos in reachable:
                continue
            reachable.add(current_pos)
            
            for neighbor in maze.get_neighbors(current_pos):
                if neighbor not in reachable:
                    queue.append(neighbor)
        
        # Check if goal is reachable
        if maze.goal not in reachable:
            agent.success = False
            return agent
        
        # P2: Optimize within safe channels (A* search)
        open_set = {maze.start}
        came_from = {}
        g_score = {maze.start: 0}
        f_score = {maze.start: maze.manhattan_distance(maze.start, maze.goal)}
        
        while open_set:
            current = min(open_set, key=lambda p: f_score.get(p, float('inf')))
            
            if current == maze.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(maze.start)
                agent.path = list(reversed(path))
                agent.success = True
                agent.steps = len(agent.path) - 1
                return agent
            
            open_set.remove(current)
            
            for neighbor in maze.get_neighbors(current):
                if neighbor not in reachable:
                    continue  # Not in safe channel
                    
                tentative_g = g_score[current] + 1
                
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + maze.manhattan_distance(neighbor, maze.goal)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        agent.success = False
    
    return agent

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path)
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

class MazeAgent:
    """An agent that applies the Deadband Protocol in a grid maze."""
    
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_rocks(rock_density)
        
    def generate_rocks(self, density: float):
        """P0: Map the negative space (rocks)."""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        # Exclude start and goal
        available = [(x, y) for x in range(self.width) for y in range(self.height)
                     if (x, y) not in [self.start, self.goal]]
        self.rocks = set(random.sample(available, num_rocks))
    
    def is_safe(self, x: int, y: int) -> bool:
        """Check if a cell is not a rock."""
        return (x, y) not in self.rocks
    
    def find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Identify all safe paths from start to goal using BFS."""
        from collections import deque
        
        queue = deque([(self.start, [self.start])])
        safe_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            # Four possible moves (right, down, left, up)
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if self.is_safe(nx, ny) and (nx, ny) not in path:
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        return safe_paths
    
    def optimize_path(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """P2: Select shortest path among safe channels."""
        if not safe_paths:
            return []
        return min(safe_paths, key=len)
    
    def greedy_unconstrained(self) -> List[Tuple[int, int]]:
        """P2-only (greedy) approach for comparison."""
        x, y = self.start
        path = [self.start]
        visited = set([self.start])
        
        while (x, y) != self.goal:
            # Prefer moves toward goal
            moves = []
            if x < self.goal[0]:
                moves.append((1, 0))
            if y < self.goal[1]:
                moves.append((0, 1))
            if x > self.goal[0]:
                moves.append((-1, 0))
            if y > self.goal[1]:
                moves.append((0, -1))
            
            # Shuffle to avoid simple zigzag
            random.shuffle(moves)
            moved = False
            for dx, dy in moves:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height and (nx, ny) not in visited:
                    x, y = nx, ny
                    path.append((x, y))
                    visited.add((x, y))
                    moved = True
                    break
            
            if not moved:
                # No valid move toward goal
                break
        
        return path if path[-1] == self.goal else []

def run_simulation(runs: int = 50) -> dict:
    """Run multiple trials matching fleet simulation parameters."""
    results = {
        'deadband_success': 0,
        'unconstrained_success': 0,
        'deadband_path_lengths': [],
        'unconstrained_path_lengths': []
    }
    
    for _ in range(runs):
        agent = MazeAgent(width=20, height=20, rock_density=0.16)
        
        # Deadband Protocol (P0→P1→P2)
        safe_channels = agent.find_safe_channels()
        optimal_path = agent.optimize_path(safe_channels)
        if optimal_path and optimal_path[-1] == agent.goal:
            results['deadband_success'] += 1
            results['deadband_path_lengths'].append(len(optimal_path))
        
        # Unconstrained greedy (P2-only)
        greedy_path = agent.greedy_unconstrained()
        if greedy_path and greedy_path[-1] == agent.goal:
            results['unconstrained_success'] += 1
            results['unconstrained_path_lengths'].append(len(greedy_path))
    
    # Calculate averages
    if results['deadband_path_lengths']:
        results['deadband_avg'] = sum(results['deadband_path_lengths']) / len(results['deadband_path_lengths'])
    else:
        results['deadband_avg'] = 0
    
    if results['unconstrained_path_lengths']:
        results['unconstrained_avg'] = sum(results['unconstrained_path_lengths']) / len(results['unconstrained_path_lengths'])
    else:
        results['unconstrained_avg'] = 0
    
    return results

def visualize_maze(agent: MazeAgent, path: List[Tuple[int, int]] = None):
    """Create ASCII visualization of maze and path."""
    grid = []
    for y in range(agent.height):
        row = []
        for x in range(agent.width):
            if (x, y) == agent.start:
                row.append('S')
            elif (x, y) == agent.goal:
                row.append('G')
            elif (x, y) in agent.rocks:
                row.append('█')
            elif path and (x, y) in path:
                row.append('·')
            else:
                row.append(' ')
        grid.append(''.join(row))
    return '\n'.join(grid)

def main():
    print("=" * 60)
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print("=" * 60)

    print("\n1. Single Maze Example:")
    agent = MazeAgent(width=10, height=6, rock_density=0.15)
    
    print("\nMaze (S=start, G=goal, █=rock):")
    print(visualize_maze(agent))
    
    print(f"\nP0: Mapped {len(agent.rocks)} rocks (negative space)")
    
    safe_channels = agent.find_safe_channels()
    print(f"P1: Found {len(safe_channels)} safe channels")
    
    optimal_path = agent.optimize_path(safe_channels)
    if optimal_path:
        print(f"P2: Optimal path length = {len(optimal_path)} steps")
        print("\nSafe path visualization:")
        print(visualize_maze(agent, optimal_path))
    else:
        print("P2: No safe path to goal")
    
    # Simulation matching fleet data
    print("\n" + "=" * 60)
    print("2. Simulation Results (50 runs, 20x20 maze, 16% rocks):")
    results = run_simulation(runs=50)
    
    print(f"\nDeadband Protocol (P0→P1→P2):")
    print(f"  Success rate: {results['deadband_success']}/50 = {results['deadband_success']/50*100:.0f}%")
    print(f"  Average path length: {results['deadband_avg']:.1f}")
    
    print(f"\nUnconstrained Greedy (P2-only):")
    print(f"  Success rate: {results['unconstrained_success']}/50 = {results['unconstrained_success']/50*100:.0f}%")
    print(f"  Average path length: {results['unconstrained_avg']:.1f}")
    
    # Compare with fleet data

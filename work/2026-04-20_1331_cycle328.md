#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Visual ASCII simulation of P0→P1→P2 in a 20x20 maze.
Matches fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
import sys
import time

# Constants from simulation data
MAZE_SIZE = 20
ROCK_COUNT = 64  # ~16% density, matches 64 rocks in 400 cells
START = (0, 0)
GOAL = (19, 19)

# Symbols
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_SYM = 'G'
PATH = '○'
SAFE = '░'
VISITED = 'x'

class DeadbandMaze:
    def __init__(self, seed=None):
        self.seed = seed if seed is not None else random.randint(1, 1000)
        random.seed(self.seed)
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks = set()
        self.place_rocks()
        self.grid[START[0]][START[1]] = AGENT
        self.grid[GOAL[0]][GOAL[1]] = GOAL_SYM
        self.phase = "INIT"
        self.steps = 0
        self.success = False

    def place_rocks(self):
        """Place rocks randomly, avoiding start and goal."""
        placed = 0
        while placed < ROCK_COUNT:
            r, c = random.randint(0, MAZE_SIZE-1), random.randint(0, MAZE_SIZE-1)
            if (r, c) == START or (r, c) == GOAL:
                continue
            if (r, c) not in self.rocks:
                self.rocks.add((r, c))
                self.grid[r][c] = ROCK
                placed += 1

    def print_maze(self, title="", highlight=None):
        """Print the maze with optional path highlighting."""
        print(f"\n=== {title} (Seed: {self.seed}) ===")
        for r in range(MAZE_SIZE):

                else:
                    row.append(cell)

    def p1_find_safe_channels(self):
        """P1: BFS to find all reachable safe cells (deadband)."""
        self.phase = "P1"
        safe = set()
        queue = [START]
        visited = set([START])

            r, c = queue.pop(0)
            safe.add((r, c))
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                    if (nr, nc) not in visited and (nr, nc) not in self.rocks:
                        visited.add((nr, nc))
                        queue.append((nr, nc))
        
        # Mark safe cells visually

                row.append(display_grid[r][c])
            print(' '.join(row))
        print(f"Safe reachable cells: {len(safe)}")
        
        return safe

    def p2_optimize_in_channel(self, safe_cells):
        """P2: A* search constrained to safe channels."""
        self.phase = "P2"
        if GOAL not in safe_cells:
            print("GOAL NOT REACHABLE within safe channels.")
            return None
        
        # A* heuristic: Manhattan distance
        def heuristic(a, b):
            return abs(a[0]-b[0]) + abs(a[1]-b[1])
        
        open_set = {START}
        came_from = {}
        g_score = {START: 0}
        f_score = {START: heuristic(START, GOAL)}
        
        while open_set:
            current = min(open_set, key=lambda x: f_score.get(x, float('inf')))
            if current == GOAL:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(START)
                path.reverse()
                self.steps = len(path) - 1
                self.success = True
                return path
            
            open_set.remove(current)
            r, c = current
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (r + dr, c + dc)
                if neighbor not in safe_cells:
                    continue
                tentative_g = g_score[current] + 1
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, GOAL)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        return None

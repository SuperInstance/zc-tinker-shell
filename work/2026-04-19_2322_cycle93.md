#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 priority in a simple ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.15):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density and (x, y) not in [self.start, self.goal]:
                    self.rocks.add((x, y))
        
        # Ensure at least one path exists
        self._ensure_path()
    
    def _ensure_path(self):
        """Ensure there's at least one path from start to goal."""
        # Simple right-then-down path
        for x in range(self.width):
            if (x, 0) in self.rocks:
                self.rocks.remove((x, 0))
        for y in range(self.height):
            if (self.width-1, y) in self.rocks:
                self.rocks.remove((self.width-1, y))
    
    def is_rock(self, x: int, y: int) -> bool:
        return (x, y) in self.rocks
    
    def is_valid(self, x: int, y: int) -> bool:
        return 0 <= x < self.width and 0 <= y < self.height
    
    def render(self, path: List[Tuple[int, int]] = None, deadband: Set[Tuple[int, int]] = None) -> str:
        """Render maze with optional path and deadband visualization."""
        lines = []
        for y in range(self.height):
            line = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    line.append('S')
                elif pos == self.goal:
                    line.append('G')
                elif path and pos in path:
                    line.append('*')
                elif deadband and pos in deadband:
                    line.append('·')
                elif self.is_rock(x, y):
                    line.append('█')
                else:
                    line.append('.')
            lines.append(''.join(line))
        return '\n'.join(lines)

class DeadbandAgent:
    def __init__(self, maze: DeadbandMaze):
        self.maze = maze
        self.visited = set()
        
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify rocks (what NOT to do)."""
        return self.maze.rocks.copy()
    
    def p1_find_safe_channels(self, rocks: Set[Tuple[int, int]]) -> Set[Tuple[int, int]]:
        """P1: Find all safe cells reachable from start (deadband)."""
        safe = set()
        stack = [self.maze.start]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in safe or (x, y) in rocks:
                continue
            
            safe.add((x, y))
            
            # 4-direction movement
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if self.maze.is_valid(nx, ny):
                    stack.append((nx, ny))
        
        return safe
    
    def p2_optimize_in_channel(self, safe: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
        """P2: Find shortest path within safe channel (A* simplified)."""
        from collections import deque
        
        if self.maze.goal not in safe:
            return []  # No safe path to goal
        
        queue = deque([(self.maze.start, [self.maze.start])])
        visited = set()
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.maze.goal:
                return path
            
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Explore neighbors in safe channel
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (nx, ny) in safe and (nx, ny) not in visited:
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        return []
    
    def greedy_p2_only(self) -> List[Tuple[int, int]]:
        """Violation: P2 optimization without P0+P1 (will hit rocks)."""
        path = [self.maze.start]
        x, y = self.maze.start
        
        while (x, y) != self.maze.goal:
            # Greedy: move toward goal
            if x < self.maze.goal[0]:
                x += 1
            elif y < self.maze.goal[1]:
                y += 1
            else:
                break
            
            path.append((x, y))
            
            # Check if hit rock
            if self.maze.is_rock(x, y):
                return path  # Failed
        
        return path if (x, y) == self.maze.goal else []

def run_demo():
    """Run complete deadband protocol demonstration."""
    print("=" * 60)

    print("\n1. MAZE (S=start, G=goal, █=rock, .=safe)")
    print(maze.render())
    
    print("\n2. P0: MAP NEGATIVE SPACE (rocks)")
    rocks = agent.p0_map_negative_space()

    print(f"Found {len(safe)} safe cells reachable from start")

    print(maze.render(deadband=safe))

    path = agent.p2_optimize_in_channel(safe)

        print("Path visualization (*=path):")
        print(maze.render(path=path))
    else:
        print("No safe path to goal found")

    greedy_path = agent.greedy_p2_only()

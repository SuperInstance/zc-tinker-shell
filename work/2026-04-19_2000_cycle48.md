**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on the provided fleet knowledge, specifically the `constraint_sim_final.json` results and the formal Deadband Protocol steps in `DEADBAND-PROTOCOL.md`, I will implement a concrete simulation.

**Approach (P0→P1→P2):**
1. **P0 (Map negative space):** Load the maze, identify all rock (`#`) cells.
2. **P1 (Identify safe channels):** Perform flood-fill from start, marking all reachable cells that are not rocks. These are the deadband/safe channels.
3. **P2 (Optimize within channel):** Run A* pathfinding strictly within the safe channel cells to find the shortest path to goal.

**Implementation:** I'll create a script that reads a maze from a file, visualizes each step, and prints metrics matching the fleet's simulation format.

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0 → P1 → P2 strict priority demonstration.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import sys
from typing import List, Tuple, Set
import heapq

def load_maze(filename: str) -> Tuple[List[List[str]], Tuple[int, int], Tuple[int, int]]:
    """Load maze from file, find start (S) and goal (G)."""
    with open(filename, 'r') as f:
        maze = [list(line.rstrip()) for line in f]
    start = goal = None
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == 'S':
                start = (y, x)
            elif cell == 'G':
                goal = (y, x)
    if start is None or goal is None:
        raise ValueError("Maze must contain S (start) and G (goal)")
    return maze, start, goal

def p0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Identify all rock cells (#)."""
    rocks = set()
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.add((y, x))
    return rocks

def p1_find_safe_channels(maze: List[List[str]], start: Tuple[int, int], rocks: Set[Tuple[int, int]]) -> Set[Tuple[int, int]]:
    """P1: Flood-fill from start, avoiding rocks. Returns reachable safe cells."""
    h, w = len(maze), len(maze[0])
    safe = set()
    stack = [start]
    while stack:
        y, x = stack.pop()
        if (y, x) in safe or (y, x) in rocks:
            continue
        safe.add((y, x))
        for dy, dx in [(-1,0),(1,0),(0,-1),(0,1)]:
            ny, nx = y + dy, x + dx
            if 0 <= ny < h and 0 <= nx < w:
                stack.append((ny, nx))
    return safe

def p2_optimize_path(start: Tuple[int, int], goal: Tuple[int, int], safe: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
    """P2: A* search within safe cells only."""
    def heuristic(a, b):
        return abs(a[0]-b[0]) + abs(a[1]-b[1])
    
    frontier = []
    heapq.heappush(frontier, (0, start))
    came_from = {start: None}
    cost_so_far = {start: 0}
    
    while frontier:
        _, current = heapq.heappop(frontier)
        if current == goal:
            break
        y, x = current
        for dy, dx in [(-1,0),(1,0),(0,-1),(0,1)]:
            ny, nx = y + dy, x + dx
            neighbor = (ny, nx)
            if neighbor not in safe:
                continue  # constrained to safe channel
            new_cost = cost_so_far[current] + 1
            if neighbor not in cost_so_far or new_cost < cost_so_far[neighbor]:
                cost_so_far[neighbor] = new_cost
                priority = new_cost + heuristic(goal, neighbor)
                heapq.heappush(frontier, (priority, neighbor))
                came_from[neighbor] = current
    
    # reconstruct path
    if goal not in came_from:
        return []
    path = []
    current = goal
    while current != start:
        path.append(current)
        current = came_from[current]
    path.append(start)
    path.reverse()
    return path

def visualize(maze: List[List[str]], 
              rocks: Set[Tuple[int, int]], 
              safe: Set[Tuple[int, int]], 
              path: List[Tuple[int, int]]) -> None:
    """Print maze with annotations."""
    h, w = len(maze), len(maze[0])
    print("\n=== DEADBAND PROTOCOL DEMONSTRATION ===")
    print("Legend: # rock, S start, G goal, · safe channel, * optimal path")
    
    for y in range(h):
        row = []
        for x in range(w):
            cell = maze[y][x]
            if (y, x) in path and cell not in 'SG':
                row.append('*')
            elif (y, x) in safe and cell == '.':
                row.append('·')
            else:
                row.append(cell)
        print(''.join(row))

def run_simulation(maze_file: str = "sample_maze.txt"):
    """Full P0→P1→P2 simulation."""
    print(f"Loading maze from {maze_file}")
    maze, start, goal = load_maze(maze_file)
    
    # P0: Map rocks
    rocks = p0_map_rocks(maze)
    print(f"\n[P0] Negative space mapped: {len(rocks)} rock cells")
    
    # P1: Find safe channels
    safe = p1_find_safe_channels(maze, start, rocks)
    print(f"[P1] Safe channels identified: {len(safe)} reachable cells")
    
    # P2: Optimize within channel
    path = p2_optimize_path(start, goal, safe)
    success = len(path) > 0 and path[-1] == goal
    print(f"[P2] Path optimized: {'SUCCESS' if success else 'FAILED'}")
    if success:
        print(f"    Path length: {len(path)-1} steps (optimal: {len(path)-1})")
    
    # Metrics matching fleet format
    print("\n=== SIMULATION METRICS ===")
    print(f"optimal: {len(path)-1 if success else 0}")
    print(f"rocks: {len(rocks)}")
    print(f"deadband_success: {1 if success else 0}")
    print(f"deadband_avg: {len(path)-1 if success else 0}")
    
    visualize(maze, rocks, safe, path)
    return success

if __name__ == "__main__":
    maze_file = sys.argv[1] if len(sys.argv) > 1 else "sample_maze.txt"
    try:
        run_simulation(maze_file)
    except FileNotFoundError:
        print(f"Error: Maze file '{maze_file}' not found.")
        print("Creating a sample maze...")
        sample_maze = """10x10
S...#.....
.##.#.#...
...#.#.#..
.#..#.#.#.
..#....#..
.#.#.##.#.
..#.#...#.
.##.#.#.#.
....#...#G
"""
        with open("sample_maze.txt", "w") as f:
            f.write(sample_maze)
        print("Created sample_maze.txt. Run again.")
```

#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize within channels).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json simulation.
"""

import random
import sys
from collections import deque

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # ~16% density, matches simulation
OPTIMAL_PATH_LENGTH = 21  # from constraint_sim_final.json

# Symbols
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL = 'G'
PATH = '○'
SAFE = ' '  # inside deadband channel

def generate_maze(size, rocks):
    """Generate random maze with rocks (obstacles)."""
    maze = [[EMPTY for _ in range(size)] for _ in range(size)]
    positions = [(r, c) for r in range(size) for c in range(size)]
    rock_positions = random.sample(positions, rocks)
    for r, c in rock_positions:
        maze[r][c] = ROCK
    # Place agent at top-left, goal at bottom-right (must be free)
    maze[0][0] = AGENT
    maze[size-1][size-1] = GOAL
    return maze

def print_maze(maze, title="MAZE"):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks). Return set of rock coordinates."""
    rocks = set()
    for r in range(len(maze)):
        for c in range(len(maze[0])):
            if maze[r][c] == ROCK:
                rocks.add((r, c))
    return rocks

def p1_find_safe_channels(maze, rocks, start, goal):
    """P1: Find safe channels (deadbands) using BFS that avoids rocks."""
    size = len(maze)
    directions = [(0,1),(1,0),(0,-1),(-1,0)]
    visited = [[False]*size for _ in range(size)]
    parent = [[None]*size for _ in range(size)]
    queue = deque([start])
    visited[start[0]][start[1]] = True
    
    while queue:
        r, c = queue.popleft()
        if (r, c) == goal:
            # Reconstruct path as safe channel
            path = []
            while (r, c) != start:
                path.append((r, c))
                r, c = parent[r][c]
            path.append(start)
            path.reverse()
            return set(path)  # safe channel coordinates
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < size and 0 <= nc < size:
                if not visited[nr][nc] and (nr, nc) not in rocks:
                    visited[nr][nc] = True
                    parent[nr][nc] = (r, c)
                    queue.append((nr, nc))
    return set()  # no safe channel found

def p2_optimize_in_channel(maze, safe_channel, start, goal):
    """P2: Find shortest path within safe channel (deadband)."""
    size = len(maze)
    directions = [(0,1),(1,0),(0,-1),(-1,0)]
    visited = [[False]*size for _ in range(size)]
    parent = [[None]*size for _ in range(size)]
    queue = deque([start])
    visited[start[0]][start[1]] = True
    
    while queue:
        r, c = queue.popleft()
        if (r, c) == goal:
            # Reconstruct optimal path
            path = []
            while (r, c) != start:
                path.append((r, c))
                r, c = parent[r][c]
            path.append(start)
            path.reverse()
            return path
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < size and 0 <= nc < size:
                if not visited[nr][nc] and (nr, nc) in safe_channel:
                    visited[nr][nc] = True
                    parent[nr][nc] = (r, c)
                    queue.append((nr, nc))
    return []  # no path within channel

def visualize_steps(maze, rocks, safe_channel, optimal_path):
    """Create visualizations for each protocol step."""
    # P0 visualization: show rocks
    maze_p0 = [row[:] for row in maze]
    for r, c in rocks:
        maze_p0[r][c] = ROCK
    print_maze(maze_p0, "P0: MAP ROCKS (Negative Space)")
    
    # P1 visualization: show safe channel
    maze_p1 = [row[:] for row in maze]
    for r in range(len(maze)):
        for c in range(len(maze[0])):
            if (r, c) in safe_channel and maze[r][c] == EMPTY:
                maze_p1[r][c] = SAFE
    print_maze(maze_p1, "P1: SAFE CHANNELS (Deadbands)")
    
    # P2 visualization: show optimal path within channel
    maze_p2 = [row[:] for row in maze]
    for r, c in optimal_path[1:-1]:  # exclude start and goal
        maze_p2[r][c] = PATH
    print_maze(maze_p2, "P2: OPTIMAL PATH WITHIN CHANNEL")

    print("Based on fleet simulation: 20x20 maze, 64 rocks")
    print("P0 → P1 → P2 strict priority\n")

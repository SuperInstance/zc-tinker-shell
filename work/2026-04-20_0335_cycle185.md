def generate_maze() -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """Generate a 20×20 maze with 64 rocks placed randomly."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        x = random.randint(0, MAZE_SIZE-1)

    maze[START[1]][START[0]] = 'S'
    maze[GOAL[1]][GOAL[0]] = 'G'
    return maze, rocks

    """P2-only: Greedy optimization toward goal without rock mapping."""

                candidates.append((nx, ny))

                current = next_pos

                break

def deadband_protocol(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> Tuple[bool, List[Tuple[int, int]], int]:

        current = queue.popleft()
        if current == GOAL:
            break

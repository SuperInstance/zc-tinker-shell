**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on the fleet knowledge, specifically the `constraint_sim_final.json` data and the formal DEADBAND-PROTOCOL.md, I will implement a visual simulation. The simulation will:
1. **P0:** Map rocks (obstacles) in a maze.
2. **P1:** Find safe channels (deadbands) — paths that avoid all rocks.
3. **P2:** Optimize within a safe channel to find the shortest path to the goal.

I'll use a simple grid, ASCII visualization, and step-by-step output to show each protocol phase.

## Output: `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 in a simple navigational domain.
Based on fleet-knowledge/DEADBAND-PROTOCOL.md and constraint_sim_final.json.
"""

import random
from collections import deque

# ========== CONFIG ==========
GRID_WIDTH = 10
GRID_HEIGHT = 8
START = (0, 0)
GOAL = (9, 7)
ROCK_DENSITY = 0.25  # ~25% of cells are rocks
SEED = 42  # for reproducibility
# ============================

random.seed(SEED)

def generate_maze(width, height, rock_density):
    """Create a grid with rocks ('#'), empty cells ('.'), start ('S'), goal ('G')."""
    grid = [['.' for _ in range(width)] for _ in range(height)]
    rocks = []
    for y in range(height):
        for x in range(width):
            if (x, y) == START:
                grid[y][x] = 'S'
            elif (x, y) == GOAL:
                grid[y][x] = 'G'
            elif random.random() < rock_density:
                grid[y][x] = '#'
                rocks.append((x, y))
    return grid, rocks

def print_maze(grid, title="Maze"):
    """Print the maze with a title."""
    print(f"\n=== {title} ===")
    for row in grid:
        print(' '.join(row))
    print()

def p0_map_rocks(grid, rocks):
    """P0: Identify negative space (rocks)."""
    print("P0: MAP THE NEGATIVE SPACE")
    print(f"   Found {len(rocks)} rock cells (coordinates):")
    for i, (x, y) in enumerate(rocks[:5]):  # show first 5
        print(f"      ({x}, {y})", end='')
    if len(rocks) > 5:
        print(f" ... and {len(rocks)-5} more")
    else:
        print()
    print("   Rocks marked as '#' in maze.")
    return rocks

def p1_find_safe_channels(grid, start, goal):
    """P1: Find all safe paths (deadbands) using BFS, avoiding rocks."""
    print("\nP1: IDENTIFY SAFE CHANNELS")
    width, height = len(grid[0]), len(grid)
    # Directions: right, down, left, up
    dirs = [(1, 0), (0, 1), (-1, 0), (0, -1)]
    
    # BFS to find all reachable cells and shortest path distances
    visited = [[False for _ in range(width)] for _ in range(height)]
    parent = [[None for _ in range(width)] for _ in range(height)]
    queue = deque([start])
    visited[start[1]][start[0]] = True
    reachable_safe_cells = []
    
    while queue:
        x, y = queue.popleft()
        if grid[y][x] != '#':  # not a rock
            reachable_safe_cells.append((x, y))
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if not visited[ny][nx] and grid[ny][nx] != '#':
                    visited[ny][nx] = True
                    parent[ny][nx] = (x, y)
                    queue.append((nx, ny))
    
    # Check if goal is reachable
    if not visited[goal[1]][goal[0]]:
        print("   No safe channel to goal! Goal is blocked by rocks.")
        return [], None, reachable_safe_cells
    
    # Reconstruct one shortest safe path (for demonstration)
    path = []
    node = goal
    while node != start:
        path.append(node)
        node = parent[node[1]][node[0]]
    path.append(start)
    path.reverse()
    
    print(f"   Found {len(reachable_safe_cells)} safe reachable cells (excluding rocks).")
    print(f"   One safe channel (shortest path) has {len(path)} steps.")
    return path, parent, reachable_safe_cells

def p2_optimize_within_channel(grid, path, start, goal):
    """P2: Optimize within the safe channel (already the shortest path in this simple case)."""
    print("\nP2: OPTIMIZE WITHIN CHANNEL")
    if not path:
        print("   No safe channel available. Optimization impossible.")
        return None
    
    # In this simple BFS, the first path found is already shortest.
    # For demo, we could apply A* with same constraints, but result is identical.
    print(f"   Shortest safe path from {start} to {goal}:")
    for i, (x, y) in enumerate(path):
        print(f"      Step {i}: ({x}, {y})")
    return path

def visualize_path(grid, path):
    """Mark the path on the grid with '*'."""
    viz_grid = [row[:] for row in grid]
    for x, y in path[1:-1]:  # exclude start and goal
        if viz_grid[y][x] == '.':
            viz_grid[y][x] = '*'
    return viz_grid

def main():
    print("DEADBAND PROTOCOL DEMO - ASCII MAZE")
    print("=" * 40)
    
    # Generate maze
    grid, rocks = generate_maze(GRID_WIDTH, GRID_HEIGHT, ROCK_DENSITY)
    print_maze(grid, "Initial Maze (S=start, G=goal, #=rock)")
    
    # P0: Map rocks
    rocks_mapped = p0_map_rocks(grid, rocks)
    
    # P1: Find safe channels
    safe_path, parent, safe_cells = p1_find_safe_channels(grid, START, GOAL)
    
    # P2: Optimize within channel
    if safe_path:
        optimized_path = p2_optimize_within_channel(grid, safe_path, START, GOAL)
        
        # Visualization
        viz_grid = visualize_path(grid, optimized_path)
        print_maze(viz_grid, "Safe Path Visualization (S=start, G=goal, #=rock, *=path)")
        
        # Metrics
        print("METRICS:")
        print(f"   Rocks (P0 negatives): {len(rocks_mapped)}")
        print(f"   Safe reachable cells (P1 deadband): {len(safe_cells)}")
        print(f"   Path length (P2 optimized): {len(optimized_path)-1} steps")
        
        # Simulate failure of unconstrained optimization (P2-only)
        print("\n--- SIMULATED UNCONSTRAINED OPTIMIZATION (P2-only) ---")
        print("   An agent that skips P0+P1 would try straight-line or greedy path.")
        print("   High probability of hitting a rock → failure.")
    else:
        print("\nNo safe path exists. Deadband Protocol prevented certain failure.")
    
    print("\n" + "=" * 40)
    print("Demo complete. Protocol: P0→P1→P2 enforced.")

if __name__ == "__main__":
    main()
```

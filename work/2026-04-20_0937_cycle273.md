#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 with visual ASCII maze
Based on fleet constraint simulation data
"""

import random
from collections import deque
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, num_rocks: int = 64):
        self.width = width
        self.height = height
        self.num_rocks = num_rocks
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Negative space (rocks)
        self.rocks: Set[Tuple[int, int]] = set()
        
        # P1: Safe channels (reachable positions avoiding rocks)
        self.safe_channels: Set[Tuple[int, int]] = set()
        
        # P2: Optimal path
        self.optimal_path: List[Tuple[int, int]] = []
        
        # Generate maze
        self._generate_maze()
    
    def _generate_maze(self):
        """Generate random rock positions (P0 negative space)"""
        all_positions = [(x, y) for x in range(self.width) for y in range(self.height)]
        all_positions.remove(self.start)
        all_positions.remove(self.goal)
        
        self.rocks = set(random.sample(all_positions, self.num_rocks))
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify all rock positions (what NOT to do)"""
        return self.rocks.copy()
    
    def p1_find_safe_channels(self) -> Set[Tuple[int, int]]:
        """P1: Find all positions reachable without hitting rocks"""
        visited = set()
        queue = deque([self.start])
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            
            # Check all four directions
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) not in self.rocks and
                    (nx, ny) not in visited):
                    queue.append((nx, ny))
        
        self.safe_channels = visited
        return visited
    
    def p2_optimize_within_channels(self) -> List[Tuple[int, int]]:
        """P2: Find shortest path through safe channels"""
        if not self.safe_channels:
            raise ValueError("Must run P1 first to find safe channels")
        
        # BFS for shortest path
        parent = {}
        queue = deque([self.start])
        visited = set([self.start])
        
        while queue:
            x, y = queue.popleft()
            
            if (x, y) == self.goal:
                # Reconstruct path
                path = []
                while (x, y) != self.start:
                    path.append((x, y))
                    x, y = parent[(x, y)]
                path.append(self.start)
                path.reverse()
                self.optimal_path = path
                return path
            
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) in self.safe_channels and
                    (nx, ny) not in visited):
                    visited.add((nx, ny))
                    parent[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
        
        return []  # No path found
    
    def run_deadband_protocol(self) -> Tuple[bool, int]:
        """Execute full P0→P1→P2 protocol"""
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        print(f"P0: Mapped {len(rocks)} rock positions (negative space)")
        
        # P1: Find safe channels
        safe = self.p1_find_safe_channels()
        print(f"P1: Found {len(safe)} safe positions (channels)")
        
        # P2: Optimize within channels
        path = self.p2_optimize_within_channels()
        success = len(path) > 0
        print(f"P2: Found {'optimal' if success else 'no'} path ({len(path)} steps)")
        
        return success, len(path)
    
    def visualize(self, show_path: bool = True, show_channels: bool = False):
        """Create ASCII visualization of the maze"""
        grid = []
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')  # Rock
                elif show_path and pos in self.optimal_path:
                    row.append('·')  # Path
                elif show_channels and pos in self.safe_channels:
                    row.append('░')  # Safe channel
                else:
                    row.append(' ')
            grid.append(''.join(row))
        
        return '\n'.join(grid)

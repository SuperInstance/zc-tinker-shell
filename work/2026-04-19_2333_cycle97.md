#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map rocks (obstacles)
P1: Find safe channels (paths avoiding rocks)
P2: Optimize shortest path within safe channels
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=10, height=10, rock_density=0.2):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.rocks = set()
        self.start = (0, 0)
        self.goal = (height-1, width-1)
        
        # P0: Generate rocks (negative space)
        for y in range(height):
            for x in range(width):
                if (y, x) != self.start and (y, x) != self.goal:
                    if random.random() < rock_density:
                        self.grid[y][x] = '#'
                        self.rocks.add((y, x))
        
        # Ensure at least one path exists
        self._ensure_path()
    
    def _ensure_path(self):
        """Guarantee at least one safe channel exists."""
        # Simple DFS to check connectivity
        stack = [self.start]
        visited = set()
        while stack:
            y, x = stack.pop()
            if (y, x) == self.goal:
                return
            if (y, x) in visited:
                continue
            visited.add((y, x))
            for dy, dx in [(0,1),(0,-1),(1,0),(-1,0)]:
                ny, nx = y + dy, x + dx
                if 0 <= ny < self.height and 0 <= nx < self.width:
                    if (ny, nx) not in self.rocks:
                        stack.append((ny, nx))
        
        # If no path, clear a diagonal
        for i in range(min(self.height, self.width)):
            y = x = i
            if (y, x) in self.rocks:
                self.grid[y][x] = '.'
                self.rocks.remove((y, x))
    
    def display(self, path=None, channels=None):
        """Print maze with optional path and safe channels."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (y, x) == self.start:
                    row.append('S')
                elif (y, x) == self.goal:
                    row.append('G')
                elif path and (y, x) in path:
                    row.append('*')
                elif channels and (y, x) in channels:
                    row.append('~')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
        print()
    
    def p0_map_rocks(self):
        """P0: Identify negative space (rocks)."""
        print("=== P0: MAP ROCKS ===")
        print(f"Found {len(self.rocks)} rock cells (obstacles)")
        self.display()
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all cells reachable without hitting rocks."""
        # BFS from start, avoiding rocks
        queue = deque([self.start])
        safe = {self.start}
        while queue:
            y, x = queue.popleft()
            for dy, dx in [(0,1),(0,-1),(1,0),(-1,0)]:
                ny, nx = y + dy, x + dx
                if 0 <= ny < self.height and 0 <= nx < self.width:
                    if (ny, nx) not in self.rocks and (ny, nx) not in safe:
                        safe.add((ny, nx))
                        queue.append((ny, nx))
        
        print("=== P1: SAFE CHANNELS ===")
        print(f"Found {len(safe)} safe cells (including start/goal)")
        self.display(channels=safe)
        return safe
    
    def p2_optimize_in_channel(self, safe_cells):
        """P2: Find shortest path within safe channels."""
        # BFS for shortest path
        queue = deque([(self.start, [self.start])])
        visited = set()
        
        while queue:
            (y, x), path = queue.popleft()
            if (y, x) == self.goal:
                print("=== P2: OPTIMAL PATH ===")
                print(f"Path length: {len(path)-1} steps")
                self.display(path=path)
                return path
            
            if (y, x) in visited:
                continue
            visited.add((y, x))
            
            for dy, dx in [(0,1),(0,-1),(1,0),(-1,0)]:
                ny, nx = y + dy, x + dx
                if 0 <= ny < self.height and 0 <= nx < self.width:
                    if (ny, nx) in safe_cells and (ny, nx) not in visited:
                        queue.append(((ny, nx), path + [(ny, nx)]))
        
        print("=== P2: NO PATH FOUND ===")
        return None
    
    def greedy_unconstrained(self):
        """Simulate P2-only (greedy) agent that ignores rocks."""
        path = [self.start]
        y, x = self.start
        
        while (y, x) != self.goal:
            # Greedy: move toward goal
            dy = 1 if self.goal[0] > y else -1 if self.goal[0] < y else 0
            dx = 1 if self.goal[1] > x else -1 if self.goal[1] < x else 0
            
            ny, nx = y + dy, x + dx
            if 0 <= ny < self.height and 0 <= nx < self.width:
                y, x = ny, nx
            else:
                # Can't move directly, try orthogonal
                if dy == 0:
                    ny = y + 1 if y < self.goal[0] else y - 1
                    nx = x
                else:
                    ny = y
                    nx = x + 1 if x < self.goal[1] else x - 1
                if 0 <= ny < self.height and 0 <= nx < self.width:
                    y, x = ny, nx
                else:
                    break
            
            path.append((y, x))
            if (y, x) in self.rocks:
                print("=== GREEDY (P2-ONLY): HIT ROCK ===")
                self.display(path=path)
                return False
        
        success = (y, x) == self.goal
        print(f"=== GREEDY (P2-ONLY): {'SUCCESS' if success else 'FAILURE'} ===")
        if success:
            print(f"Path length: {len(path)-1} steps")
            self.display(path=path)
        return success

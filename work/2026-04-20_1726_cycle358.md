#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set

# Maze dimensions (small for ASCII display)
WIDTH = 10
HEIGHT = 8
START = (0, 0)
GOAL = (9, 7)

def generate_maze(rock_density: float = 0.2) -> List[List[str]]:
    """Generate a random maze with rocks (#), empty cells (.), start (S), goal (G)."""
    maze = [['.' for _ in range(WIDTH)] for _ in range(HEIGHT)]
    # Place rocks
    for y in range(HEIGHT):
        for x in range(WIDTH):
            if (x, y) != START and (x, y) != GOAL and random.random() < rock_density:
                maze[y][x] = '#'
    maze[START[1]][START[0]] = 'S'
    maze[GOAL[1]][GOAL[0]] = 'G'
    return maze

def print_maze(maze: List[List[str]], title: str):
    """Print maze with a title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def p0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Identify negative space (rocks)."""
    rocks = set()
    for y in range(HEIGHT):
        for x in range(WIDTH):
            if maze[y][x] == '#':
                rocks.add((x, y))
    print(f"P0 — Rocks mapped: {len(rocks)} rock cells")
    return rocks

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
    """P1: Find all safe paths (channels) from start to goal using BFS, avoiding rocks."""
    from collections import deque
    
    # Directions: right, down, left, up
    dirs = [(1, 0), (0, 1), (-1, 0), (0, -1)]
    
    # BFS to find all shortest safe paths
    queue = deque([(START, [START])])
    visited = {START}
    safe_paths = []
    
    while queue:
        (x, y), path = queue.popleft()
        
        if (x, y) == GOAL:
            safe_paths.append(path)
            continue
        
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < WIDTH and 0 <= ny < HEIGHT:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    print(f"P1 — Safe channels found: {len(safe_paths)} possible paths")
    return safe_paths

def p2_optimize_within_channel(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """P2: Select the shortest path among safe channels."""
    if not safe_paths:
        print("P2 — No safe channels available (trapped)")
        return []
    
    optimal_path = min(safe_paths, key=len)
    print(f"P2 — Optimal path length: {len(optimal_path)-1} steps")
    return optimal_path

def visualize_path(maze: List[List[str]], path: List[Tuple[int, int]]):
    """Mark the chosen path on the maze with '*'."""
    viz = [row[:] for row in maze]
    for (x, y) in path[1:-1]:  # exclude start and goal
        if viz[y][x] == '.':
            viz[y][x] = '*'
    print_maze(viz, "Final Path (P0→P1→P2)")

def run_unconstrained_greedy(maze: List[List[str]]) -> bool:
    """Simulate a P2-only (greedy) agent that ignores P0/P1."""
    x, y = START
    path = [START]
    dirs = [(1, 0), (0, 1)]  # biased toward right/down (goal direction)
    
    while (x, y) != GOAL:
        # Greedy: move toward goal if possible
        moves = []
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < WIDTH and 0 <= ny < HEIGHT:
                if maze[ny][nx] != '#':
                    moves.append((nx, ny))
        if not moves:
            # stuck
            print("Unconstrained greedy agent hit a rock or dead-end.")
            return False
        x, y = random.choice(moves)
        path.append((x, y))
        if len(path) > 30:  # prevent infinite loops
            return False
    
    print(f"Unconstrained greedy path length: {len(path)-1} steps")
    return True

def main():
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("=" * 40)
    
    # Seed for reproducibility
    random.seed(42)
    
    # Generate maze
    maze = generate_maze(rock_density=0.25)
    print_maze(maze, "Initial Maze")
    
    # --- Deadband Protocol (P0→P1→P2) ---
    print("\n--- Deadband Protocol Execution ---")
    rocks = p0_map_rocks(maze)
    safe_paths = p1_find_safe_channels(maze, rocks)
    optimal_path = p2_optimize_within_channel(safe_paths)
    
    if optimal_path:
        visualize_path(maze, optimal_path)
        print("SUCCESS: Deadband protocol found a safe optimal path.")
    else:
        print("FAILURE: No safe path exists (maze impossible).")
    
    # --- Unconstrained Greedy (P2-only) ---
    print("\n--- Unconstrained Greedy (P2-only) ---")
    success = run_unconstrained_greedy(maze)
    if success:
        print("Greedy agent reached goal (lucky).")
    else:
        print("Greedy agent failed (typical).")
    
    # Summary
    print("\n" + "=" * 40)
    print("SUMMARY:")
    print("- Deadband Protocol (P0→P1→P2): systematic, reliable.")
    print("- Unconstrained greedy (P2-only): often fails, hits rocks.")
    print("(Matches fleet simulation data: deadband 50/50 success, greedy 0/50)")

if __name__ == "__main__":
    main()
```

# Deadband Protocol Visual Demo

This Python script demonstrates the Deadband Protocol (P0→P1→P2) using an ASCII maze simulation.

## What It Shows

1. **P0 (Map Negative Space)**: Identifies all rock cells (`#`) in the maze.
2. **P1 (Find Safe Channels)**: Uses BFS to find all possible paths from start (`S`) to goal (`G`) that avoid rocks.
3. **P2 (Optimize Within Channel)**: Selects the shortest path among the safe channels.

The script also simulates an unconstrained greedy agent (P2-only) that often fails, illustrating why skipping P0/P1 is dangerous.

## How to Run

```bash
python3 deadband_demo.py
```

## Expected Output

- A randomly generated maze (rocks, start, goal).
- Step-by-step Deadband Protocol execution.
- Visualization of the optimal safe path.
- Comparison with a greedy agent that ignores P0/P1.

## Fleet Context

This demo is based on the simulation data in `fleet-knowledge/narrow-games/constraint_sim_final.json`, which shows:
- Deadband protocol: 50/50 success
- Unconstrained greedy: 0/50 success

The Deadband Protocol is a core fleet doctrine: always map rocks (P0) and safe channels (P1) before optimizing (P2).

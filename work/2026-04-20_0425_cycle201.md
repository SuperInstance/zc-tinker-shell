import random
from typing import List, Tuple, Set

class MazeAgent:
    """Agent that follows Deadband Protocol in a maze."""
    
    def __init__(self, maze_size=20):
        self.maze_size = maze_size
        self.start = (0, 0)
        self.goal = (maze_size-1, maze_size-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
    def generate_rocks(self, rock_count=64):
        """P0: Map negative space (what NOT to do)"""
        self.rocks = set()
        while len(self.rocks) < rock_count:
            x = random.randint(0, self.maze_size-1)
            y = random.randint(0, self.maze_size-1)
            # Don't block start or goal
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def find_safe_channels(self):
        """P1: Find safe channels (where you CAN be)"""
        self.safe_channels = []
        
        # Simple BFS to find all paths avoiding rocks
        queue = [(self.start, [self.start])]
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.pop(0)
            
            # If we reached goal, save this safe channel
            if (x, y) == self.goal:
                self.safe_channels.append(path)
                continue
            
            # Explore neighbors (right and down only for simplicity)
            for dx, dy in [(1, 0), (0, 1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.maze_size and 0 <= ny < self.maze_size:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
    
    def optimize_within_channel(self):
        """P2: Optimize within channels (best path)"""
        if not self.safe_channels:
            return []
        
        # Find shortest safe path
        self.optimal_path = min(self.safe_channels, key=len)
        return self.optimal_path
    
    def run_deadband_protocol(self):
        """Execute full P0→P1→P2 protocol"""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===")
        print(f"Maze: {self.maze_size}x{self.maze_size}")
        print(f"Start: {self.start}, Goal: {self.goal}")
        print()
        
        # P0: Map negative space
        print("P0: MAP NEGATIVE SPACE")
        self.generate_rocks(64)
        print(f"  • Identified {len(self.rocks)} rocks (failure states)")
        print()
        
        # P1: Find safe channels
        print("P1: IDENTIFY SAFE CHANNELS")
        self.find_safe_channels()
        print(f"  • Found {len(self.safe_channels)} safe channels")
        if self.safe_channels:
            print(f"  • Shortest channel length: {len(min(self.safe_channels, key=len))}")
            print(f"  • Longest channel length: {len(max(self.safe_channels, key=len))}")
        print()
        
        # P2: Optimize within channel
        print("P2: OPTIMIZE WITHIN CHANNEL")
        optimal = self.optimize_within_channel()
        if optimal:
            print(f"  • Optimal path length: {len(optimal)}")
            print(f"  • Path: {optimal[:3]}...{optimal[-3:] if len(optimal) > 6 else ''}")
            success = True
        else:
            print("  • No safe path found!")
            success = False
        print()
        
        return success, len(optimal) if optimal else 0
    
    def visualize_maze(self, show_path=True):
        """Create ASCII visualization of maze"""
        print("MAZE VISUALIZATION")
        print("  S = Start, G = Goal, # = Rock, · = Safe, * = Path")
        print()
        
        for y in range(self.maze_size):
            row = []
            for x in range(self.maze_size):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('#')
                elif show_path and pos in self.optimal_path:
                    row.append('*')
                else:
                    row.append('·')
            print('  ' + ' '.join(row))
        print()

        """P2-only approach (violates protocol)"""
        print("=== GREEDY AGENT (P2-ONLY) ===")
        print("VIOLATION: Optimizing before mapping safe channels")
        print()

        self.generate_rocks(64)
        
        # Try to find shortest path ignoring rocks
        path = []
        x, y = self.start
        
        while (x, y) != self.goal:
            path.append((x, y))
            
            # Greedy: move toward goal
            if x < self.goal[0]:
                x += 1
            elif y < self.goal[1]:
                y += 1
            else:
                break
            
            # Check if hit rock
            if (x, y) in self.rocks:
                print(f"  • Hit rock at ({x}, {y})!")
                return False, len(path)
        
        path.append(self.goal)
        print(f"  • Path length: {len(path)}")
        return True, len(path)

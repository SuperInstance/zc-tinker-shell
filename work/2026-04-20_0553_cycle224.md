import random
from typing import List, Tuple, Set

class Maze:
    """20x20 maze matching simulation dimensions"""
    def __init__(self, seed=42):
        self.width = 20
        self.height = 20
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (19, 19)
        
        # Generate rocks (64 rocks as in simulation)
        random.seed(seed)
        for _ in range(64):
            x = random.randint(0, self.width-1)
            y = random.randint(0, self.height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def is_valid(self, x: int, y: int) -> bool:
        """Check if position is within bounds and not a rock"""
        return (0 <= x < self.width and 
                0 <= y < self.height and 
                (x, y) not in self.rocks)
    
    def render(self, path=None) -> str:
        """Render maze with optional path"""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif (x, y) in self.rocks:
                    row.append('█')
                elif path and (x, y) in path:
                    row.append('·')
                else:
                    row.append(' ')
            grid.append(''.join(row))
        return '\n'.join(grid)

class UnconstrainedAgent(Agent):

    def find_path(self):
        x, y = self.maze.start
        self.path = [(x, y)]
        
        # Greedy move toward goal
        while (x, y) != self.maze.goal and len(self.path) < 100:
            # Try to move right or down (toward goal)
            moves = []
            if x < self.maze.goal[0]:
                moves.append((x+1, y))

                moves.append((x, y+1))

                    self.path.append((x, y))
                    break
            
            # Check if hit rock
            if (x, y) in self.maze.rocks:
                self.success = False
                return
        
        self.success = ((x, y) == self.maze.goal)


class ConstrainedAgent(Agent):
    """P0-only: Knows rocks but no channel planning"""
    def find_path(self):
        x, y = self.maze.start
        self.path = [(x, y)]
        visited = set([(x, y)])
        
        # Random walk avoiding rocks
        while (x, y) != self.maze.goal and len(self.path) < 200:
            # Generate possible moves
            candidates = []
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if self.maze.is_valid(nx, ny) and (nx, ny) not in visited:
                    candidates.append((nx, ny))
            
            if not candidates:
                # Dead end
                self.success = False
                return
            
            # Pick random valid move (no channel optimization - P1 missing)
            x, y = random.choice(candidates)
            self.path.append((x, y))
            visited.add((x, y))
        
        self.success = ((x, y) == self.maze.goal)


class DeadbandAgent(Agent):
    """P0→P1→P2: Full deadband protocol"""

        queue = [start]
        came_from = {start: None}
        
        while queue:
            current = queue.pop(0)
            
            if current == goal:
                # Reconstruct path (P2: Optimize within safe channel)
                self.path = []
                while current is not None:
                    self.path.append(current)
                    current = came_from[current]
                self.path.reverse()
                self.success = True
                return

            neighbors.append((x-1, y))

                    queue.append((nx, ny))
                    came_from[(nx, ny)] = current
        
        # Goal not reachable
        self.success = False
        self.path = [start]

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # From constraint_sim_final.json
START = (0, 0)
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

def generate_maze() -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """Generate a 20x20 maze with rocks placed randomly."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = set()
    
    # Place rocks randomly
    while len(rocks) < ROCK_COUNT:
        x = random.randint(0, MAZE_SIZE-1)
        y = random.randint(0, MAZE_SIZE-1)
        if (x, y) != START and (x, y) != GOAL:
            rocks.add((x, y))
            maze[y][x] = '█'  # Rock symbol
    
    # Mark start and goal
    maze[START[1]][START[0]] = 'S'
    maze[GOAL[1]][GOAL[0]] = 'G'
    
    return maze, rocks

def p0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Identify negative space (rocks)."""
    rocks = set()
    for y in range(MAZE_SIZE):
        for x in range(MAZE_SIZE):
            if maze[y][x] == '█':
                rocks.add((x, y))
    print(f"P0: Mapped {len(rocks)} rocks (negative space)")
    return rocks

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[bool]]:
    """P1: Find safe channels (cells not adjacent to rocks)."""
    safe = [[True for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    
    # Mark rocks as unsafe
    for x, y in rocks:
        safe[y][x] = False
    
    # Mark cells adjacent to rocks as unsafe (deadband boundary)
    for x, y in rocks:
        for dx in [-1, 0, 1]:
            for dy in [-1, 0, 1]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    safe[ny][nx] = False
    
    # Start and goal are always safe
    safe[START[1]][START[0]] = True
    safe[GOAL[1]][GOAL[0]] = True
    
    safe_count = sum(sum(row) for row in safe)
    print(f"P1: Found {safe_count} safe cells (channels)")
    return safe

def p2_optimize_path(maze: List[List[str]], safe: List[List[bool]]) -> List[Tuple[int, int]]:
    """P2: Find optimal path through safe channels (BFS)."""
    from collections import deque
    
    # BFS through safe cells only
    queue = deque([(START, [START])])
    visited = set([START])
    
    while queue:
        (x, y), path = queue.popleft()
        
        if (x, y) == GOAL:
            print(f"P2: Found optimal path of length {len(path)-1}")
            return path
        
        # Explore 4-connected neighbors
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if safe[ny][nx] and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    print("P2: No safe path to goal (constrained optimization failed)")
    return []

def visualize(maze: List[List[str]], path: List[Tuple[int, int]] = None, 
              safe: List[List[bool]] = None) -> str:
    """Create ASCII visualization."""
    vis = [row[:] for row in maze]
    
    # Mark safe channels if provided
    if safe:
        for y in range(MAZE_SIZE):
            for x in range(MAZE_SIZE):
                if safe[y][x] and vis[y][x] == '.':
                    vis[y][x] = '·'  # Safe channel indicator
    
    # Mark path if provided
    if path:
        for x, y in path[1:-1]:  # Skip start and goal
            if vis[y][x] == '·' or vis[y][x] == '.':
                vis[y][x] = '○'  # Path marker
    
    # Build output
    lines = []
    lines.append("=" * 45)
    lines.append("DEADBAND PROTOCOL MAZE DEMO (20×20)")
    lines.append(f"Rocks: {ROCK_COUNT}  Start: S  Goal: G")
    lines.append("=" * 45)
    
    for y, row in enumerate(vis):
        line = f"{y:2d} "
        for x, cell in enumerate(row):
            line += cell + ' '
        lines.append(line)
    
    lines.append("\nLEGEND:")
    lines.append("  S = Start, G = Goal, █ = Rock (P0 negative space)")
    lines.append("  · = Safe channel (P1 deadband), ○ = Optimal path (P2)")
    lines.append("=" * 45)
    
    return '\n'.join(lines)

    print("\n--- GREEDY UNCONSTRAINED (P2-ONLY) ---")
    x, y = START
    path = [START]
    
    for _ in range(100):  # Limit steps
        if (x, y) == GOAL:
            print(f"  Greedy reached goal in {len(path)-1} steps")
            return True
        
        # Move toward goal (greedy)
        moves = []
        if x < GOAL[0] and maze[y][x+1] != '█':
            moves.append((x+1, y))
        if y < GOAL[1] and maze[y+1][x] != '█':
            moves.append((x, y+1))
        if not moves:
            # Try any non-rock move
            for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if maze[ny][nx] != '█':
                        moves.append((nx, ny))
        
        if not moves:
            print(f"  Greedy stuck at ({x},{y})")
            return False
        
        x, y = random.choice(moves)
        path.append((x, y))
    
    print(f"  Greedy failed (max steps reached)")
    return False

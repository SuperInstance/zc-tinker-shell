#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 priority in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time

# Constants from simulation data
MAZE_SIZE = 20
OPTIMAL_PATH_LENGTH = 21  # optimal path length from data
ROCK_COUNT = 64           # rocks (negative space)
GOAL = (19, 19)
START = (0, 0)

class DeadbandMaze:
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks = set()
        self.agent_pos = START
        self.goal = GOAL
        self.steps = 0
        self.path = []
        
        # Place rocks (P0: negative space)
        self._place_rocks()
        
        # Mark start and goal
        self.grid[START[0]][START[1]] = 'S'
        self.grid[GOAL[0]][GOAL[1]] = 'G'
    
    def _place_rocks(self):
        """P0: Map the negative space (rocks)."""
        placed = 0
        while placed < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) == START or (x, y) == GOAL:
                continue
            if (x, y) not in self.rocks:
                self.rocks.add((x, y))
                self.grid[x][y] = '#'
                placed += 1
    
    def print_maze(self, show_agent=True):
        """Print current maze state."""
        print("\n" + "="*50)
        print(f"Step {self.steps} | Agent at {self.agent_pos}")
        print("="*50)
        
        # Create a copy for display
        display = [row[:] for row in self.grid]
        
        if show_agent and self.agent_pos != START:
            x, y = self.agent_pos
            if display[x][y] == '.':
                display[x][y] = 'A'
        
        for row in display:
            print(' '.join(row))
        print()
    
    def get_neighbors(self, pos):
        """Get valid neighboring positions (no rocks, within bounds)."""
        x, y = pos
        neighbors = []
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) not in self.rocks:
                    neighbors.append((nx, ny))
        return neighbors
    
    def p0_map_negative_space(self):
        """P0: Identify all rocks (what NOT to do)."""
        print("="*50)
        print("P0: MAPPING NEGATIVE SPACE")
        print("="*50)
        print(f"Identified {len(self.rocks)} rocks at positions:")
        rock_list = list(self.rocks)
        for i in range(0, len(rock_list), 5):
            print("  ", rock_list[i:i+5])
        print()
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all safe paths from start to goal (deadbands)."""
        print("="*50)
        print("P1: FINDING SAFE CHANNELS")
        print("="*50)
        
        # Simple BFS to find if safe path exists
        visited = set()
        queue = [(START, [START])]
        safe_paths = []
        
        while queue and len(safe_paths) < 3:  # Find first 3 safe paths
            pos, path = queue.pop(0)
            if pos == GOAL:
                safe_paths.append(path)
                continue
            
            if pos in visited:
                continue
            visited.add(pos)
            
            for neighbor in self.get_neighbors(pos):
                if neighbor not in visited:
                    queue.append((neighbor, path + [neighbor]))
        
        if safe_paths:
            print(f"Found {len(safe_paths)} safe channel(s)")
            for i, path in enumerate(safe_paths[:2]):  # Show first 2
                print(f"  Channel {i+1}: {len(path)-1} steps")
                # Show path visualization
                vis = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
                for x, y in path:
                    vis[x][y] = '○'
                vis[START[0]][START[1]] = 'S'
                vis[GOAL[0]][GOAL[1]] = 'G'
                
                print("    Path visualization:")
                for row in vis:
                    print("    " + ' '.join(row[:10]) + " ...")  # First 10 columns
                print()
        else:
            print("  WARNING: No safe channels found! Agent cannot reach goal.")
        
        return safe_paths
    
    def p2_optimize_within_channel(self, safe_paths):
        """P2: Choose optimal path from safe channels."""
        print("="*50)
        print("P2: OPTIMIZING WITHIN SAFE CHANNELS")
        print("="*50)
        
        if not safe_paths:
            print("  No safe channels to optimize. Protocol FAILED.")
            return None
        
        # Find shortest safe path
        optimal_path = min(safe_paths, key=len)
        print(f"  Selected optimal path: {len(optimal_path)-1} steps")
        print(f"  Theoretical optimal: {OPTIMAL_PATH_LENGTH} steps")
        
        # Visualize optimal path
        vis = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        for x, y in optimal_path:
            vis[x][y] = '●'
        vis[START[0]][START[1]] = 'S'
        vis[GOAL[0]][GOAL[1]] = 'G'
        
        print("  Optimal path visualization:")
        for row in vis:
            print("    " + ' '.join(row[:12]) + " ...")
        print()
        
        return optimal_path
    
    def run_deadband_protocol(self):
        """Execute full P0→P1→P2 protocol."""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*60)
        
        # Initial maze state
        self.print_maze(show_agent=False)
        time.sleep(1)
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        time.sleep(1)
        
        # P1: Find safe channels
        safe_paths = self.p1_find_safe_channels()
        time.sleep(1)
        
        # P2: Optimize within channel
        optimal_path = self.p2_optimize_within_channel(safe_paths)
        
        if optimal_path:
            # Simulate agent following optimal path
            print("="*50)
            print("SIMULATING AGENT MOVEMENT")
            print("="*50)
            
            self.agent_pos = START
            self.path = [START]
            
            for step, pos in enumerate(optimal_path[1:], 1):
                self.steps = step
                self.agent_pos = pos
                self.path.append(pos)
                self.print_maze()
                time.sleep(0.3)
            
            print(f"\n✓ SUCCESS: Agent reached goal in {self.steps} steps")
            print(f"  Path length: {len(optimal_path)-1} steps")
            print(f"  Optimal possible: {OPTIMAL_PATH_LENGTH} steps")
            return True
        else:
            print("\n✗ FAILURE: No safe path found")
            return False
    
    def run_greedy_failure(self):
        """Demonstrate P2-only (greedy) failure."""
        print("\n" + "="*60)
        print("GREEDY (P2-ONLY) FAILURE DEMONSTRATION")
        print("="*60)
        print("Agent tries to optimize without P0/P1...")
        
        self.agent_pos = START
        self.steps = 0

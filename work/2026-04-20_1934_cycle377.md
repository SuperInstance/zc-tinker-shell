"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
"""

import random
from typing import List, Tuple, Set

# Maze constants
WIDTH = 20
HEIGHT = 20
START = (0, 0)
GOAL = (WIDTH-1, HEIGHT-1)
ROCK_DENSITY = 0.15  # 15% of cells are rocks

def generate_maze() -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """Generate a random maze with rocks."""
    maze = [['.' for _ in range(WIDTH)] for _ in range(HEIGHT)]
    rocks = set()
    
    # Place rocks randomly
    for y in range(HEIGHT):
        for x in range(WIDTH):
            if (x, y) != START and (x, y) != GOAL:
                if random.random() < ROCK_DENSITY:
                    maze[y][x] = '#'
                    rocks.add((x, y))
    
    # Ensure start and goal are clear
    maze[START[1]][START[0]] = 'S'
    maze[GOAL[1]][GOAL[0]] = 'G'
    
    return maze, rocks

def print_maze(maze: List[List[str]], title: str):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def p0_map_rocks(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[str]]:
    """P0: Map negative space (rocks)."""
    marked = [row.copy() for row in maze]
    for x, y in rocks:
        marked[y][x] = 'X'  # Mark rock locations
    return marked

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """P1: Find safe channels (cells not adjacent to rocks)."""
    safe = [row.copy() for row in maze]
    safe_cells = set()
    
    # Mark cells adjacent to rocks as dangerous
    for y in range(HEIGHT):
        for x in range(WIDTH):
            if maze[y][x] == '#':
                continue
            # Check 8-neighbor adjacency
            is_safe = True
            for dy in [-1, 0, 1]:
                for dx in [-1, 0, 1]:
                    nx, ny = x + dx, y + dy
                    if 0 <= nx < WIDTH and 0 <= ny < HEIGHT:
                        if maze[ny][nx] == '#':
                            is_safe = False
                            break
                if not is_safe:
                    break
            
            if is_safe:
                safe[y][x] = 'O'  # Safe channel
                safe_cells.add((x, y))
            else:
                safe[y][x] = '~'  # Dangerous water
    
    return safe, safe_cells

def p2_optimize_path(maze: List[List[str]], safe_cells: Set[Tuple[int, int]]) -> List[List[str]]:
    """P2: Find optimal path through safe channels (simple BFS)."""
    from collections import deque
    
    # BFS to find shortest path through safe cells
    queue = deque([START])
    parent = {START: None}
    visited = set([START])
    
    while queue:
        x, y = queue.popleft()
        
        if (x, y) == GOAL:
            break
        
        # 4-direction movement
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < WIDTH and 0 <= ny < HEIGHT:
                if (nx, ny) not in visited and (nx, ny) in safe_cells:
                    visited.add((nx, ny))
                    parent[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    
    # Reconstruct path
    path = set()
    current = GOAL
    while current in parent and parent[current] is not None:
        path.add(current)
        current = parent[current]
    if START != GOAL:
        path.add(START)
    
    # Mark path on maze
    result = [row.copy() for row in maze]
    for x, y in path:
        if (x, y) != START and (x, y) != GOAL:
            result[y][x] = '*'
    
    return result

            x -= 1
        elif y > 0 and maze[y-1][x] != '#':
            y -= 1
        else:
            break  # Stuck
        
        path.add((x, y))

        if (px, py) != START and (px, py) != GOAL and result[py][px] != '!':

    return result

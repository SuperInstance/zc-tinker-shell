#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — Tinker (Cycle 377)
A simple ASCII maze simulation showing P0→P1→P2 priority.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

class Maze:
    def __init__(self, size=MAZE_SIZE):
        self.size = size
        self.grid = [['.' for _ in range(size)] for _ in range(size)]
        self.start = (0, 0)
        self.goal = (size-1, size-1)
        self.rocks = set()
        self.agent_pos = self.start
        
    def place_rocks(self, count=ROCK_COUNT):
        """P0: Map negative space (rocks to avoid)"""
        while len(self.rocks) < count:
            x = random.randint(0, self.size-1)
            y = random.randint(0, self.size-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[y][x] = '#'
    
    def find_safe_channels(self):
        """P1: Identify safe channels (paths without rocks)"""
        # Simple BFS to find all reachable positions avoiding rocks
        visited = set()
        queue = [self.start]
        safe_positions = set()
        
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            safe_positions.add((x, y))
            
            # Check neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.size and 0 <= ny < self.size and 
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    queue.append((nx, ny))
        
        return safe_positions
    
    def optimize_within_channel(self, safe_positions):
        """P2: Find optimal path within safe channel"""
        # A* search constrained to safe positions
        open_set = {self.start}
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: self.heuristic(self.start)}
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score.get(pos, float('inf')))
            
            if current == self.goal:
                return self.reconstruct_path(came_from, current)
            
            open_set.remove(current)
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # Constraint: must be in safe channel
                if neighbor not in safe_positions:
                    continue
                    
                tentative_g = g_score[current] + 1
                
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self.heuristic(neighbor)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        return None  # No path found
    
    def heuristic(self, pos):
        """Manhattan distance to goal"""
        return abs(pos[0] - self.goal[0]) + abs(pos[1] - self.goal[1])
    
    def reconstruct_path(self, came_from, current):
        """Reconstruct optimal path"""
        path = [current]
        while current in came_from:
            current = came_from[current]
            path.append(current)
        path.reverse()
        return path
    
    def display(self, path=None, agent_pos=None):
        """Display ASCII maze with optional path and agent"""
        display_grid = [row[:] for row in self.grid]
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        display_grid[sy][sx] = 'S'
        display_grid[gy][gx] = 'G'
        
        # Mark path if provided
        if path:
            for x, y in path[1:-1]:  # Skip start and goal
                if display_grid[y][x] == '.':
                    display_grid[y][x] = '*'
        
        # Mark agent position
        if agent_pos:
            ax, ay = agent_pos
            if display_grid[ay][ax] not in ['S', 'G']:
                display_grid[ay][ax] = 'A'
        
        print("\n" + "="*50)
        for row in display_grid:
            print(' '.join(row))
        print("="*50)
        
        # Legend
        print("Legend: S=Start, G=Goal, #=Rock, *=Safe Path, A=Agent")
    
    def run_unconstrained_greedy(self):
        """P2-only approach (violates protocol)"""
        print("\n❌ VIOLATION: P2-only (greedy optimization)")
        print("   Skipping P0+P1, directly optimizing...")
        
        pos = self.start
        path = [pos]
        steps = 0
        
        while pos != self.goal and steps < 100:
            # Greedy: move toward goal
            dx = 1 if self.goal[0] > pos[0] else -1 if self.goal[0] < pos[0] else 0
            dy = 1 if self.goal[1] > pos[1] else -1 if self.goal[1] < pos[1] else 0
            
            new_pos = (pos[0] + dx, pos[1] + dy)
            
            # Check if hit rock
            if new_pos in self.rocks:
                print(f"   Step {steps+1}: HIT ROCK at {new_pos}!")
                return False, steps
            
            pos = new_pos
            path.append(pos)
            steps += 1
        
        success = pos == self.goal
        if success:
            print(f"   Reached goal in {steps} steps (optimal: {OPTIMAL_PATH_LENGTH})")
        else:
            print(f"   Failed after {steps} steps")
        
        return success, steps

def main():
    """Run Deadband Protocol demonstration"""
    print("🚀 DEADBAND PROTOCOL DEMONSTRATION")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print(f"Optimal path: {OPTIMAL_PATH_LENGTH} steps, Rocks: {ROCK_COUNT}")
    print("-" * 50)
    
    # Create maze
    maze = Maze()
    
    # P0: Map negative space (rocks)
    print("\n🔍 P0: MAPPING NEGATIVE SPACE")
    print("   Identifying rocks (states to avoid)...")
    maze.place_rocks()
    maze.display()
    time.sleep(1)
    
    # P1: Find safe channels
    print("\n📊 P1: IDENTIFYING SAFE CHANNELS")
    print("   Finding all reachable positions without rocks...")
    safe_channels = maze.find_safe_channels()
    print(f"   Safe positions: {len(safe_channels)}/{MAZE_SIZE*MAZE_SIZE}")
    
    # Check if goal is reachable
    if maze.goal not in safe_channels:
        print("   ❌ Goal unreachable! No safe channel exists.")
        return
    
    # P2: Optimize within channel
    print("\n🎯 P2: OPTIMIZING WITHIN SAFE CHANNEL")
    print("   Finding shortest path through safe water...")
    optimal_path = maze.optimize_within_channel(safe_channels)
    
    if optimal_path:
        print(f"   Found optimal path: {len(optimal_path)-1} steps")
        maze.display(path=optimal_path)
        
        # Animate agent following path
        print("\n🤖 AGENT FOLLOWING DEADBAND PROTOCOL")
        for i, pos in enumerate(optimal_path):
            maze.display(path=optimal_path, agent_pos=pos)
            if i < len(optimal_path) - 1:
                time.sleep(0.3)
                print()
        
        print(f"\n✅ SUCCESS: Reached goal in {len(optimal_path)-1} steps")

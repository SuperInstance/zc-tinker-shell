#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 priority in pathfinding
"""

import heapq
from collections import deque

class DeadbandMaze:
    def __init__(self, width=10, height=10, rock_density=0.2):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        
        # Place start and goal
        self.start = (0, 0)
        self.goal = (height-1, width-1)
        self.grid[0][0] = 'S'
        self.grid[height-1][width-1] = 'G'
        
        # Place rocks (negative space)
        self.rocks = set()
        import random
        for y in range(height):
            for x in range(width):
                if (y, x) != self.start and (y, x) != self.goal:
                    if random.random() < rock_density:
                        self.grid[y][x] = '#'
                        self.rocks.add((y, x))
    
    def print_grid(self, title="Maze"):
        print(f"\n=== {title} ===")
        for row in self.grid:
            print(' '.join(row))
    
    def p0_map_negative_space(self):
        """P0: Identify what NOT to do (rocks/obstacles)"""
        print("\n--- P0: MAP NEGATIVE SPACE ---")
        print(f"Identified {len(self.rocks)} rocks (failure states)")
        print("Rocks at coordinates:", sorted(self.rocks)[:10], "..." if len(self.rocks) > 10 else "")
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find where you CAN be (reachable cells avoiding rocks)"""
        print("\n--- P1: FIND SAFE CHANNELS ---")
        
        # BFS to find all reachable cells
        visited = set()
        queue = deque([self.start])
        safe_cells = set()
        
        while queue:
            y, x = queue.popleft()
            if (y, x) in visited:
                continue
            visited.add((y, x))
            safe_cells.add((y, x))
            
            # Check neighbors
            for dy, dx in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                ny, nx = y + dy, x + dx
                if 0 <= ny < self.height and 0 <= nx < self.width:
                    if (ny, nx) not in self.rocks and (ny, nx) not in visited:
                        queue.append((ny, nx))
        
        print(f"Found {len(safe_cells)} safe cells (channel size)")
        print(f"Goal reachable: {self.goal in safe_cells}")
        return safe_cells
    
    def p2_optimize_in_channel(self, safe_cells):
        """P2: Find best path through safe channels only"""
        print("\n--- P2: OPTIMIZE WITHIN CHANNEL ---")
        
        # A* search constrained to safe cells
        def heuristic(a, b):
            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        open_set = []
        heapq.heappush(open_set, (0, self.start))
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: heuristic(self.start, self.goal)}
        
        while open_set:
            current = heapq.heappop(open_set)[1]
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                path.reverse()
                print(f"Found optimal path of length {len(path)-1}")
                return path
            
            for dy, dx in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                neighbor = (current[0] + dy, current[1] + dx)
                
                # CONSTRAINT: Only move through safe channels
                if neighbor not in safe_cells:
                    continue
                
                tentative_g = g_score[current] + 1
                
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, self.goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
        
        print("No path found within safe channels")
        return None

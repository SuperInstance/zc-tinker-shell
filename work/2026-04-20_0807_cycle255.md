#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 protocol in a simple ASCII maze.

Based on fleet-knowledge/narrow-games/constraint_sim_final.json results:
- Unconstrained (P2-only): 0% success
- Deadband (P0→P1→P2): 50% success at optimal speed
"""

import random
import time
from typing import List, Tuple, Set

class MazeAgent:
    """Agent implementing Deadband Protocol in a grid maze."""
    
    def __init__(self, maze_size: int = 10, rock_density: float = 0.2):
        self.size = maze_size
        self.start = (0, 0)
        self.goal = (maze_size-1, maze_size-1)
        
        # Generate maze with rocks
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_rocks(rock_density)
        
        # Protocol state
        self.p0_map: Set[Tuple[int, int]] = set()  # Negative space
        self.p1_channels: List[List[Tuple[int, int]]] = []  # Safe paths
        self.p2_path: List[Tuple[int, int]] = []  # Optimal safe path
        
        # Statistics
        self.steps_taken = 0
        self.success = False
        
    def generate_rocks(self, density: float):
        """P0: Generate rocks (negative space)."""
        total_cells = self.size * self.size
        num_rocks = int(total_cells * density)
        
        # Ensure start and goal are not rocks
        available = [(x, y) for x in range(self.size) for y in range(self.size)
                    if (x, y) not in [self.start, self.goal]]
        
        self.rocks = set(random.sample(available, min(num_rocks, len(available))))
    
    def is_valid(self, pos: Tuple[int, int]) -> bool:
        """Check if position is within bounds and not a rock."""
        x, y = pos
        return (0 <= x < self.size and 0 <= y < self.size 
                and pos not in self.rocks)
    
    def get_neighbors(self, pos: Tuple[int, int]) -> List[Tuple[int, int]]:
        """Get valid neighboring positions (4-direction movement)."""
        x, y = pos
        candidates = [(x+1, y), (x-1, y), (x, y+1), (x, y-1)]
        return [p for p in candidates if self.is_valid(p)]
    
    def run_deadband_protocol(self):
        """Execute full P0→P1→P2 protocol."""
        print("=" * 50)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("=" * 50)
        
        # P0: Map negative space
        print("\n[P0] MAPPING NEGATIVE SPACE")
        print(f"Identified {len(self.rocks)} rocks (failure states)")
        self.p0_map = self.rocks.copy()
        
        # P1: Find safe channels (BFS to find all safe paths)
        print("\n[P1] FINDING SAFE CHANNELS")
        self.find_safe_channels()
        print(f"Found {len(self.p1_channels)} safe channels to goal")
        
        # P2: Optimize within safest channel
        print("\n[P2] OPTIMIZING WITHIN SAFE CHANNEL")
        if self.p1_channels:
            self.select_optimal_path()
            print(f"Selected optimal path: {len(self.p2_path)} steps")
            self.success = True
        else:
            print("ERROR: No safe channels found! (Goal unreachable)")
            self.success = False
        
        return self.success
    
    def find_safe_channels(self):
        """P1: Find all safe paths from start to goal using BFS."""
        from collections import deque
        
        # BFS to find all shortest safe paths
        queue = deque([(self.start, [self.start])])
        visited = {self.start: 0}
        shortest_length = None
        paths = []
        
        while queue:
            current, path = queue.popleft()
            
            # If we found goal, record path
            if current == self.goal:
                if shortest_length is None:
                    shortest_length = len(path)
                if len(path) == shortest_length:
                    paths.append(path)
                continue
            
            # Explore neighbors
            for neighbor in self.get_neighbors(current):
                if neighbor not in visited or visited[neighbor] >= len(path) + 1:
                    visited[neighbor] = len(path) + 1
                    queue.append((neighbor, path + [neighbor]))
        
        self.p1_channels = paths
    
    def select_optimal_path(self):
        """P2: Select optimal path from safe channels."""
        if not self.p1_channels:
            return []
        
        # Choose shortest path (could add other optimization criteria)
        self.p2_path = min(self.p1_channels, key=len)
        self.steps_taken = len(self.p2_path) - 1  # Subtract start position
        return self.p2_path
    
    def run_unconstrained_greedy(self):
        """Run P2-only (greedy) approach for comparison."""
        print("\n" + "=" * 50)
        print("UNCONSTRAINED GREEDY (P2-ONLY) FOR COMPARISON")
        print("=" * 50)
        
        current = self.start
        path = [current]
        visited = set([current])
        
        while current != self.goal:
            # Greedy: move toward goal without checking rocks
            x, y = current
            gx, gy = self.goal
            
            # Simple heuristic: move in direction of goal
            if abs(gx - x) >= abs(gy - y):
                next_pos = (x + (1 if gx > x else -1), y)
            else:
                next_pos = (x, y + (1 if gy > y else -1))
            
            # Check bounds
            if not (0 <= next_pos[0] < self.size and 0 <= next_pos[1] < self.size):
                # Try alternative move
                candidates = [(x+1, y), (x-1, y), (x, y+1), (x, y-1)]
                valid_candidates = [p for p in candidates 
                                  if 0 <= p[0] < self.size and 0 <= p[1] < self.size]
                if not valid_candidates:
                    break
                next_pos = random.choice(valid_candidates)
            
            # Check if hit rock
            if next_pos in self.rocks:
                print(f"  HIT ROCK at {next_pos}! (P0 violation)")
                return False, path
            
            current = next_pos
            path.append(current)
            
            if len(path) > self.size * self.size * 2:  # Prevent infinite loops
                break
        
        success = current == self.goal
        return success, path
    
    def visualize(self, show_path: bool = True):
        """Display ASCII maze with agent's path."""
        print("\n" + "=" * 50)
        print("MAZE VISUALIZATION")
        print("=" * 50)
        
        for y in range(self.size):
            row = []
            for x in range(self.size):
                pos = (x, y)
                
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')  # Rock
                elif show_path and pos in self.p2_path:
                    row.append('·')  # Path
                else:
                    row.append('.')  # Empty
            
            print(' '.join(row))
        
        # Legend
        print("\nLegend: S=Start, G=Goal, █=Rock, ·=Safe Path, .=Empty")
        
        if show_path and self.p2_path:
            print(f"\nOptimal Path ({len(self.p2_path)-1} steps):")
            print(' → '.join(str(p) for p in self.p2_path))

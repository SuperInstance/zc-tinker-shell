#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 374)

Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import random
import time
import sys

# ========== CONFIGURATION ==========
MAZE_WIDTH = 20
MAZE_HEIGHT = 20
ROCK_COUNT = 64          # from constraint_sim_final.json
GOAL_POS = (19, 19)      # bottom-right corner
START_POS = (0, 0)       # top-left corner

# Symbols for display
AGENT = 'A'
ROCK = '█'
GOAL = 'G'
PATH = '·'
SAFE = ' '
VISITED = 'x'
DEADBAND = '░'

# ========== MAZE GENERATION ==========
def generate_maze():
    """Create a maze with rocks (obstacles) and a goal."""
    maze = [[SAFE for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    
    # Place rocks randomly (P0: negative space)
    rocks_placed = 0
    while rocks_placed < ROCK_COUNT:
        x = random.randint(0, MAZE_WIDTH - 1)
        y = random.randint(0, MAZE_HEIGHT - 1)
        if (x, y) != START_POS and (x, y) != GOAL_POS and maze[y][x] == SAFE:
            maze[y][x] = ROCK
            rocks_placed += 1
    
    # Place goal
    maze[GOAL_POS[1]][GOAL_POS[0]] = GOAL
    return maze

def print_maze(maze, agent_pos, title="", highlight=None):
    """Print the maze with agent position and optional highlight set."""
    print(f"\n=== {title} ===")
    for y in range(MAZE_HEIGHT):
        row = []
        for x in range(MAZE_WIDTH):
            if (x, y) == agent_pos:
                row.append(AGENT)
            elif highlight and (x, y) in highlight:
                row.append(DEADBAND)
            else:
                row.append(maze[y][x])
        print(' '.join(row))
    print()

# ========== DEADBAND PROTOCOL IMPLEMENTATION ==========
class DeadbandNavigator:
    def __init__(self, maze):
        self.maze = maze
        self.width = MAZE_WIDTH
        self.height = MAZE_HEIGHT
        self.agent_pos = START_POS
        self.steps = 0
        self.visited = set()
        self.visited.add(START_POS)
        
        # Protocol state
        self.rocks = set()           # P0: negative space
        self.safe_channels = set()   # P1: safe cells
        self.optimal_path = []       # P2: chosen path
        
    # --- P0: MAP NEGATIVE SPACE ---
    def map_negative_space(self):
        """Identify all rock positions (what NOT to do)."""
        self.rocks.clear()
        for y in range(self.height):
            for x in range(self.width):
                if self.maze[y][x] == ROCK:
                    self.rocks.add((x, y))
        print(f"[P0] Mapped {len(self.rocks)} rocks (negative space)")
        return self.rocks
    
    # --- P1: IDENTIFY SAFE CHANNELS ---
    def find_safe_channels(self):
        """Find all cells that are not rocks and reachable from start."""
        self.safe_channels.clear()
        stack = [START_POS]
        visited = set()
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Cell is safe if not a rock
            if self.maze[y][x] != ROCK:
                self.safe_channels.add((x, y))
                
                # Explore neighbors (up, down, left, right)
                for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                    nx, ny = x + dx, y + dy
                    if 0 <= nx < self.width and 0 <= ny < self.height:
                        if (nx, ny) not in visited:
                            stack.append((nx, ny))
        
        print(f"[P1] Found {len(self.safe_channels)} safe cells (channels)")
        return self.safe_channels
    
    # --- P2: OPTIMIZE WITHIN CHANNEL ---
    def find_optimal_path(self):
        """Find shortest path from start to goal using only safe channels (A*)."""
        if GOAL_POS not in self.safe_channels:
            print("[P2] Goal is not in safe channels! No path exists.")
            return []
        
        # Simple BFS for pathfinding (constrained to safe channels)
        queue = [(START_POS, [START_POS])]
        visited = set()
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == GOAL_POS:
                self.optimal_path = path
                print(f"[P2] Found optimal path with {len(path)} steps")
                return path
            
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Explore neighbors (only through safe channels)
            for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) in self.safe_channels and (nx, ny) not in visited:
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        print("[P2] No path to goal within safe channels")
        return []
    
    # --- FULL PROTOCOL EXECUTION ---
    def execute_protocol(self, visualize=True):
        """Run P0→P1→P2 in strict order."""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL EXECUTION")
        print("="*60)
        
        # P0: Map negative space
        if visualize:
            print_maze(self.maze, self.agent_pos, "INITIAL MAZE")
        rocks = self.map_negative_space()
        
        # P1: Find safe channels
        safe_cells = self.find_safe_channels()
        if visualize:
            print_maze(self.maze, self.agent_pos, "SAFE CHANNELS (P1)", highlight=safe_cells)
            time.sleep(1)
        
        # P2: Optimize within channel
        path = self.find_optimal_path()
        
        if not path:
            print("RESULT: FAILURE (no safe path to goal)")
            return False
        
        # Visualize the chosen path
        if visualize:
            print("\n[P2] Optimal path visualization:")
            for i, (x, y) in enumerate(path):
                display_maze = [row[:] for row in self.maze]
                for px, py in path[:i+1]:
                    if (px, py) != GOAL_POS:
                        display_maze[py][px] = PATH
                
                # Show current position
                print_maze(display_maze, (x, y), f"STEP {i}")
                time.sleep(0.1)
        
        print(f"\nRESULT: SUCCESS in {len(path)} steps")
        print(f"Protocol followed: P0→P1→P2 (strict order)")
        return True

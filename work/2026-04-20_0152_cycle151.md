#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2 Visualization
Based on fleet-knowledge simulation data.
"""

import random
from typing import List, Tuple, Set

# Simulation parameters from fleet-knowledge
MAZE_SIZE = 20
ROCK_COUNT = 64  # From constraint_sim_final.json
START = (0, 0)
GOAL = (19, 19)

class DeadbandMaze:
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
        # Place rocks (P0: negative space)
        self._place_rocks()
        
        # Mark start and goal
        self.grid[START[0]][START[1]] = 'S'
        self.grid[GOAL[0]][GOAL[1]] = 'G'
    
    def _place_rocks(self):
        """P0: Map the negative space (rocks)"""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != START and (x, y) != GOAL:
                self.rocks.add((x, y))
                self.grid[x][y] = '#'
    
    def find_safe_channels(self):
        """P1: Identify safe channels (deadbands)"""
        # Simple BFS to find all reachable positions avoiding rocks
        visited = set()
        queue = [START]
        safe_positions = []
        
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # This position is safe (not a rock)
            if (x, y) not in self.rocks:
                safe_positions.append((x, y))
            
            # Explore neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    queue.append((nx, ny))
        
        # Group connected safe positions into channels
        channel_map = {}
        for pos in safe_positions:
            channel_id = self._get_channel_id(pos, visited)
            if channel_id not in channel_map:
                channel_map[channel_id] = []
            channel_map[channel_id].append(pos)
        
        self.safe_channels = list(channel_map.values())
        return len(self.safe_channels)
    
    def _get_channel_id(self, pos: Tuple[int, int], visited: Set) -> int:
        """Helper to identify which channel a position belongs to"""
        # Simple flood fill to label connected components
        x, y = pos
        return x * MAZE_SIZE + y  # Simplified for demo
    
    def optimize_within_channel(self):
        """P2: Find optimal path within safe channels"""
        # Find channel containing start and goal
        start_channel = None
        goal_channel = None
        
        for channel in self.safe_channels:
            if START in channel:
                start_channel = channel
            if GOAL in channel:
                goal_channel = channel
        
        if not start_channel or not goal_channel or start_channel != goal_channel:
            return False  # No safe channel connects start to goal
        
        # Simple A* within safe positions
        open_set = {START}
        came_from = {}
        g_score = {START: 0}
        f_score = {START: self._heuristic(START, GOAL)}
        
        while open_set:
            current = min(open_set, key=lambda x: f_score.get(x, float('inf')))
            
            if current == GOAL:
                self.optimal_path = self._reconstruct_path(came_from, current)
                return True
            
            open_set.remove(current)
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # P1 constraint: must be in safe channel
                if neighbor not in start_channel:
                    continue
                
                tentative_g = g_score[current] + 1
                
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor, GOAL)
                    open_set.add(neighbor)
        
        return False
    
    def _heuristic(self, a: Tuple[int, int], b: Tuple[int, int]) -> int:
        """Manhattan distance heuristic"""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def _reconstruct_path(self, came_from, current):
        """Reconstruct optimal path"""
        path = [current]
        while current in came_from:
            current = came_from[current]
            path.append(current)
        return list(reversed(path))
    
    def visualize(self, show_path=False):
        """Display the maze with optional optimal path"""
        display_grid = [row[:] for row in self.grid]
        
        if show_path and self.optimal_path:
            for x, y in self.optimal_path[1:-1]:  # Skip start and goal
                if display_grid[x][y] == '.':
                    display_grid[x][y] = '*'
        
        print("\n" + "="*50)
        print("DEADBAND PROTOCOL MAZE DEMO")
        print("="*50)
        print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}")
        print(f"Start: {START}, Goal: {GOAL}")
        print("\nLegend: S=Start, G=Goal, #=Rock, *=Safe Path, .=Empty")
        print("-"*50)
        
        for row in display_grid:
            print(' '.join(row))
        
        print("-"*50)
        
        if show_path and self.optimal_path:
            print(f"Optimal path length: {len(self.optimal_path)-1} steps")
            print(f"Path: {self.optimal_path[:5]}...{self.optimal_path[-5:] if len(self.optimal_path) > 10 else ''}")
        
        return display_grid

#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_rocks(rock_density)
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
    def generate_rocks(self, density: float):
        """P0: Map negative space (rocks)."""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        # Avoid start and goal
        candidates = [(x, y) for x in range(self.width) for y in range(self.height)
                     if (x, y) not in [self.start, self.goal]]
        self.rocks = set(random.sample(candidates, num_rocks))
    
    def is_safe(self, x: int, y: int) -> bool:
        return (0 <= x < self.width and 0 <= y < self.height 
                and (x, y) not in self.rocks)
    
    def find_safe_channels(self):
        """P1: Identify safe channels (deadbands)."""
        # Use BFS to find all safe paths from start to goal
        from collections import deque
        queue = deque([(self.start, [self.start])])
        visited = {self.start}
        all_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            if (x, y) == self.goal:
                all_paths.append(path)
                continue
            
            for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
                nx, ny = x + dx, y + dy
                if self.is_safe(nx, ny) and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.safe_channels = all_paths
    
    def optimize_within_channels(self):
        """P2: Find shortest path within safe channels."""
        if not self.safe_channels:
            return []
        self.optimal_path = min(self.safe_channels, key=len)
    
    def run_protocol(self):
        """Execute full P0→P1→P2 protocol."""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===")
        print(f"Maze: {self.width}x{self.height}, Rocks: {len(self.rocks)}")
        print()
        
        # P0: Map negative space
        print("P0: MAP NEGATIVE SPACE")
        print(f"Identified {len(self.rocks)} rock positions (failure states)")
        print()
        
        # P1: Find safe channels
        print("P1: IDENTIFY SAFE CHANNELS")
        self.find_safe_channels()
        print(f"Found {len(self.safe_channels)} safe channels (deadbands)")
        if self.safe_channels:
            shortest_len = min(len(p) for p in self.safe_channels)
            longest_len = max(len(p) for p in self.safe_channels)
            print(f"Channel lengths: {shortest_len} to {longest_len} steps")
        print()
        
        # P2: Optimize within channels
        print("P2: OPTIMIZE WITHIN CHANNEL")
        self.optimize_within_channels()
        if self.optimal_path:
            print(f"Optimal path: {len(self.optimal_path)-1} steps")
            print(f"Path: {self.optimal_path[:3]}...{self.optimal_path[-3:]}")
        else:
            print("No safe path found (all channels blocked)")
        print()
        
        return len(self.safe_channels) > 0
    
    def render_ascii(self, show_path: bool = True):
        """Render maze with ASCII art."""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif (x, y) in self.rocks:
                    row.append('#')
                elif show_path and (x, y) in self.optimal_path:
                    row.append('·')
                else:
                    row.append('.')
            grid.append(' '.join(row))
        
        print("MAZE VISUALIZATION")
        print("S=Start, G=Goal, #=Rock, ·=Optimal Path, .=Safe Water")
        print("\n".join(grid))
        print()

def main():
    # Initialize with parameters matching fleet simulation
    maze = DeadbandMaze(width=20, height=20, rock_density=0.16)
    
    # Run full protocol
    success = maze.run_protocol()
    
    # Show visualization
    if success:
        maze.render_ascii(show_path=True)
        
        # Statistics
        print("=== STATISTICS ===")
        print(f"Optimal path length: {len(maze.optimal_path)-1}")
        print(f"Safe channels found: {len(maze.safe_channels)}")
        print(f"Rocks avoided: {len(maze.rocks)}")
        
        # Compare with unconstrained greedy (would fail)
        print("\nWithout Deadband Protocol (P2-only greedy):")
        print("- Would try straight-line path")
        print("- Would hit rocks with high probability")
        print("- Success rate: 0% (per fleet simulation)")
    else:
        print("PROTOCOL FAILED: No safe channels found")
        maze.render_ascii(show_path=False)

if __name__ == "__main__":
    main()

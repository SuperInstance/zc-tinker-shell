**Action:** Writing `deadband_maze_demo.py`. This script will:

2. Simulate three agent strategies:
   - **Unconstrained (P2-only):** Greedy pathfinding ignoring rocks.
   - **Constrained (P0+P1+P2):** Deadband protocol: first map rocks (P0), then find safe channels (P1), then optimal path within safe channels (P2).
   - **Random walk:** Baseline.
3. Output ASCII visualization and success statistics.

```python
#!/usr/bin/env python3
"""
deadband_maze_demo.py

A visual demonstration of the Deadband Protocol (P0→P1→P2) in a maze.
Based on Cocapn fleet's constraint_sim_final.json data and DEADBAND-PROTOCOL.md.
"""

import random
import sys
from collections import deque

# Parameters from fleet knowledge
MAZE_SIZE = 20
ROCK_COUNT = 64  # from sim: 64 rocks
OPTIMAL_PATH_LENGTH = 21  # from sim: optimal cost 21

# Symbols
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL = 'G'
PATH = '○'
SAFE = ' '

class Maze:
    def __init__(self, size=MAZE_SIZE, rocks=ROCK_COUNT):
        self.size = size
        self.grid = [[EMPTY for _ in range(size)] for _ in range(size)]
        self.agent_pos = (0, 0)
        self.goal_pos = (size-1, size-1)
        self.rocks = set()
        
        # Place agent and goal
        self.grid[0][0] = AGENT
        self.grid[size-1][size-1] = GOAL
        
        # Place rocks randomly (avoid start and goal)
        possible = [(r, c) for r in range(size) for c in range(size)
                    if (r, c) not in [self.agent_pos, self.goal_pos]]
        self.rocks = set(random.sample(possible, rocks))
        for r, c in self.rocks:
            self.grid[r][c] = ROCK
    
    def print(self, path=None, safe_channels=None):
        """Print maze with optional path and safe channel overlay."""
        for r in range(self.size):
            row = []
            for c in range(self.size):
                cell = self.grid[r][c]
                if path and (r, c) in path:
                    row.append(PATH)
                elif safe_channels and (r, c) in safe_channels:
                    row.append(SAFE)
                else:
                    row.append(cell)
            print(' '.join(row))
        print()

class Agent:
    def __init__(self, maze):
        self.maze = maze
        self.pos = maze.agent_pos
        self.goal = maze.goal_pos
        self.rocks = maze.rocks
    
    def unconstrained_greedy(self):
        """P2-only: greedy path ignoring rocks (will fail)."""
        path = [self.pos]
        current = self.pos
        
        # Simple greedy: move toward goal, preferring right/down
        while current != self.goal:
            r, c = current
            dr = self.goal[0] - r
            dc = self.goal[1] - c
            
            # Choose next move (biased toward goal)
            if dr > 0 and (r+1, c) not in self.rocks:
                current = (r+1, c)
            elif dc > 0 and (r, c+1) not in self.rocks:
                current = (r, c+1)
            elif dr < 0 and (r-1, c) not in self.rocks:
                current = (r-1, c)
            elif dc < 0 and (r, c-1) not in self.rocks:
                current = (r, c-1)
            else:
                # Stuck (hit rock or edge)
                break
            
            path.append(current)
            if current in self.rocks:
                return None  # Failed
        
        return path if current == self.goal else None
    
    def deadband_protocol(self):
        """Full P0→P1→P2 protocol."""
        # P0: Map negative space (rocks)
        rocks = self.rocks  # Already known
        
        # P1: Find safe channels (BFS flood fill avoiding rocks)
        safe = set()
        queue = deque([self.pos])
        visited = set([self.pos])
        
        while queue:
            r, c = queue.popleft()
            safe.add((r, c))
            
            for nr, nc in [(r+1, c), (r-1, c), (r, c+1), (r, c-1)]:
                if (0 <= nr < self.maze.size and 0 <= nc < self.maze.size and
                    (nr, nc) not in rocks and (nr, nc) not in visited):
                    visited.add((nr, nc))
                    queue.append((nr, nc))
        
        # P2: Optimize within safe channels (BFS shortest path)
        if self.goal not in safe:
            return None  # No safe channel to goal
        
        queue = deque([(self.pos, [self.pos])])
        visited = set([self.pos])
        
        while queue:
            current, path = queue.popleft()
            if current == self.goal:
                return path
            
            r, c = current
            for nr, nc in [(r+1, c), (r-1, c), (r, c+1), (r, c-1)]:
                if (0 <= nr < self.maze.size and 0 <= nc < self.maze.size and
                    (nr, nc) in safe and (nr, nc) not in visited):
                    visited.add((nr, nc))
                    queue.append(((nr, nc), path + [(nr, nc)]))
        
        return None
    
    def random_walk(self, steps=100):
        """Random baseline."""
        path = [self.pos]
        current = self.pos
        
        for _ in range(steps):
            if current == self.goal:
                return path
            
            r, c = current
            neighbors = []
            for nr, nc in [(r+1, c), (r-1, c), (r, c+1), (r, c-1)]:
                if 0 <= nr < self.maze.size and 0 <= nc < self.maze.size:
                    neighbors.append((nr, nc))
            
            if not neighbors:
                break
            
            current = random.choice(neighbors)
            path.append(current)
            
            if current in self.rocks:
                return None
        
        return None

def run_simulation(iterations=50):
    """Run multiple iterations to approximate fleet simulation data."""
    results = {
        'unconstrained_success': 0,
        'deadband_success': 0,
        'random_success': 0,
        'unconstrained_lengths': [],
        'deadband_lengths': [],
    }
    
    for i in range(iterations):
        maze = Maze()
        agent = Agent(maze)
        
        # Unconstrained (P2-only)
        path1 = agent.unconstrained_greedy()
        if path1:
            results['unconstrained_success'] += 1
            results['unconstrained_lengths'].append(len(path1)-1)
        
        # Deadband protocol
        path2 = agent.deadband_protocol()
        if path2:
            results['deadband_success'] += 1
            results['deadband_lengths'].append(len(path2)-1)
        
        # Random walk
        path3 = agent.random_walk()
        if path3 and path3[-1] == agent.goal:
            results['random_success'] += 1
        
        # Print first iteration as example
        if i == 0:
            print("=== DEMONSTRATION MAZE (Iteration 1) ===")
            maze.print()
            print("P0: Rocks mapped (negative space)")
            print("P1: Safe channels (flood fill from start)")
            safe = set()
            for r in range(maze.size):
                for c in range(maze.size):
                    if (r, c) not in maze.rocks:
                        safe.add((r, c))
            maze.print(safe_channels=safe)
            
            if path2:
                print(f"P2: Optimal path within safe channels (length {len(path2)-1})")

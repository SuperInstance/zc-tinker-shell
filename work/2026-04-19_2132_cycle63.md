**Output:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque

# Simulation parameters from fleet data
MAZE_SIZE = 20
NUM_ROCKS = 64
OPTIMAL_PATH_LEN = 21

ROCK = '█'
OPEN = '·'
AGENT = 'A'
GOAL = 'G'
PATH = '○'
SAFE = '░'  # Visual marker for safe channels (P1)

class DeadbandMaze:
    def __init__(self, seed=42):
        random.seed(seed)
        self.grid = [[OPEN for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        
        # P0: Map negative space (rocks)
        self.rocks = set()
        while len(self.rocks) < NUM_ROCKS:
            r, c = random.randint(0, MAZE_SIZE-1), random.randint(0, MAZE_SIZE-1)
            if (r, c) not in {self.start, self.goal}:
                self.rocks.add((r, c))
                self.grid[r][c] = ROCK
        
        self.grid[self.start[0]][self.start[1]] = AGENT
        self.grid[self.goal[0]][self.goal[1]] = GOAL
        
        # P1: Safe channels (deadbands)
        self.safe_channels = None
        self.optimal_path = None
    
    def print_maze(self, highlight=None):
        """Print maze with optional highlight set (e.g., safe channels or path)."""
        for r in range(MAZE_SIZE):
            row = []
            for c in range(MAZE_SIZE):
                pos = (r, c)
                if highlight and pos in highlight:
                    row.append(SAFE)
                else:
                    row.append(self.grid[r][c])
            print(''.join(row))
        print()
    
    def p0_map_rocks(self):
        """P0: Identify negative space (what NOT to do)."""
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Located {len(self.rocks)} rocks (failure states).")
        print("Maze with rocks (█):")
        self.print_maze()
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find safe channels (where agent CAN be)."""
        print("=== P1: FIND SAFE CHANNELS ===")
        
        # BFS to find all reachable positions avoiding rocks
        visited = set()
        queue = deque([self.start])
        visited.add(self.start)
        
        while queue:
            r, c = queue.popleft()
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                    pos = (nr, nc)
                    if pos not in self.rocks and pos not in visited:
                        visited.add(pos)
                        queue.append(pos)
        
        self.safe_channels = visited
        print(f"Found {len(self.safe_channels)} safe positions (deadband).")
        print("Safe channels marked with ░:")
        self.print_maze(highlight=self.safe_channels)
        return self.safe_channels
    
    def p2_optimize_within_channel(self):
        """P2: Optimize within safe channels (best path)."""
        if not self.safe_channels:
            raise RuntimeError("VIOLATION: Cannot optimize (P2) before mapping safe channels (P1).")
        
        print("=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        # BFS for shortest path constrained to safe channels
        parent = {self.start: None}
        queue = deque([self.start])
        
        while queue:
            r, c = queue.popleft()
            if (r, c) == self.goal:
                break
            
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                nr, nc = r + dr, c + dc
                pos = (nr, nc)
                if pos in self.safe_channels and pos not in parent:
                    parent[pos] = (r, c)
                    queue.append(pos)
        
        # Reconstruct path
        if self.goal not in parent:
            print("FAILURE: No path to goal within safe channels.")
            return None
        
        path = []
        curr = self.goal
        while curr is not None:
            path.append(curr)
            curr = parent[curr]
        path.reverse()
        
        self.optimal_path = set(path[1:-1])  # Exclude start and goal
        print(f"Optimal path length: {len(path)-1} steps")
        print(f"Reference optimal from fleet data: {OPTIMAL_PATH_LEN} steps")
        
        # Visualize path
        print("Optimal path (○) within safe channels:")
        for r in range(MAZE_SIZE):
            row = []
            for c in range(MAZE_SIZE):
                pos = (r, c)
                if pos == self.start:
                    row.append(AGENT)
                elif pos == self.goal:
                    row.append(GOAL)
                elif pos in self.optimal_path:
                    row.append(PATH)
                elif pos in self.safe_channels:
                    row.append(SAFE)
                else:
                    row.append(self.grid[r][c])
            print(''.join(row))
        
        return path
    
    def run_protocol(self):
        """Execute strict P0→P1→P2 priority."""
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("=" * 40)
        print(f"Maze: {MAZE_SIZE}×{MAZE_SIZE}, Rocks: {NUM_ROCKS}")
        print()
        
        # P0: Map rocks
        rocks = self.p0_map_rocks()
        
        # P1: Find safe channels
        safe = self.p1_find_safe_channels()
        
        # P2: Optimize within channels
        path = self.p2_optimize_within_channel()
        
        print()
        print("=" * 40)
        if path:
            print("SUCCESS: Reached goal via Deadband Protocol.")
            print(f"Path length: {len(path)-1} steps")
        else:
            print("FAILURE: Protocol violation or no safe path.")
        
        return path

def main():
    """Run the Deadband Protocol demonstration."""
    print("Cocapn Fleet — Tinker Demo")
    print("Deadband Protocol: P0→P1→P2 Priority")
    print()
    
    maze = DeadbandMaze(seed=42)
    maze.run_protocol()

if __name__ == "__main__":
    main()
```

1. **Strict P0→P1→P2 priority** — violates protocol if P2 attempted before P1
2. **Visual ASCII output** — shows rocks, safe channels, and optimal path
3. **Based on real fleet data** — uses parameters from `constraint_sim_final.json`
4. **Demonstrates deadband concept** — safe channels as reachable non-rock positions

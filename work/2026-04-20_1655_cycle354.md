#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
ASCII visualization of the protocol in action.
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    """A simple maze simulator for the Deadband Protocol."""
    
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.2):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_rocks(rock_density)
        
    def generate_rocks(self, density: float):
        """P0: Generate rocks (negative space)."""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        # Exclude start and goal
        available = [(x, y) for x in range(self.width) for y in range(self.height)
                     if (x, y) not in (self.start, self.goal)]
        self.rocks = set(random.sample(available, min(num_rocks, len(available))))
    
    def is_rock(self, x: int, y: int) -> bool:
        """Check if cell is a rock."""
        return (x, y) in self.rocks
    
    def is_in_bounds(self, x: int, y: int) -> bool:
        """Check if cell is within maze bounds."""
        return 0 <= x < self.width and 0 <= y < self.height
    
    def find_safe_channels_bfs(self) -> List[Tuple[int, int]]:
        """P1: Find safe channels using BFS (avoid rocks)."""
        queue = [self.start]
        visited = {self.start: None}
        
        while queue:
            current = queue.pop(0)
            if current == self.goal:
                break
            x, y = current
            # Four possible moves: right, down, left, up
            moves = [(x+1, y), (x, y+1), (x-1, y), (x, y-1)]
            for nx, ny in moves:
                if (self.is_in_bounds(nx, ny) and 
                    not self.is_rock(nx, ny) and 
                    (nx, ny) not in visited):
                    visited[(nx, ny)] = current
                    queue.append((nx, ny))
        
        # Reconstruct path if goal reached
        if self.goal in visited:
            path = []
            current = self.goal
            while current is not None:
                path.append(current)
                current = visited[current]
            path.reverse()
            return path
        return []  # No safe path
    
    def greedy_unconstrained_path(self) -> List[Tuple[int, int]]:
        """P2-only (violation): Greedy path ignoring rocks."""
        path = [self.start]
        x, y = self.start
        while (x, y) != self.goal:
            # Move toward goal (right or down)
            if x < self.goal[0]:
                x += 1
            elif y < self.goal[1]:
                y += 1
            elif x > self.goal[0]:
                x -= 1
            elif y > self.goal[1]:
                y -= 1
            path.append((x, y))
            if len(path) > self.width * self.height:  # Prevent infinite loop
                break
        return path
    
    def render_maze(self, path: List[Tuple[int, int]] = None) -> str:
        """Render ASCII maze with rocks, start, goal, and optional path."""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif self.is_rock(x, y):
                    row.append('#')
                elif path and (x, y) in path:
                    row.append('·')
                else:
                    row.append('.')
            grid.append(' '.join(row))
        return '\n'.join(grid)
    
    def run_demo(self):
        """Run full Deadband Protocol demonstration."""
        print("=" * 60)
        print("DEADBAND PROTOCOL MAZE DEMO")
        print("=" * 60)
        
        print("\n[P0] MAP NEGATIVE SPACE (Rocks)")
        print(f"Generated {len(self.rocks)} rocks (marked #)")
        print("Rocks are failure states — avoid at all costs.")
        
        print("\n[MAZE WITH ROCKS]")
        print(self.render_maze())
        
        print("\n[P1] FIND SAFE CHANNELS")
        safe_path = self.find_safe_channels_bfs()
        if safe_path:
            print(f"Found safe channel with {len(safe_path)} steps")
            print("Safe path marked with ·")
            print(self.render_maze(safe_path))
        else:
            print("NO SAFE CHANNEL FOUND (all paths blocked)")
            
        print("\n[P2] OPTIMIZE WITHIN CHANNEL")
        if safe_path:
            print(f"Optimal safe path length: {len(safe_path)-1} steps")
            print("(Already optimal within safe channel — BFS finds shortest)")
        else:
            print("Cannot optimize — no safe channels (P1 failure)")
            
        print("\n[VIOLATION] P2-ONLY (Greedy, ignore rocks)")
        greedy_path = self.greedy_unconstrained_path()
        print("Greedy path marked with · (hits rocks!)")
        print(self.render_maze(greedy_path))
        hits = sum(1 for pos in greedy_path if self.is_rock(*pos))
        print(f"Greedy path hits {hits} rocks → FAILURE")
        
        print("\n" + "=" * 60)
        print("SUMMARY:")
        print(f"- Rocks (P0): {len(self.rocks)}")
        print(f"- Safe path exists: {bool(safe_path)}")
        if safe_path:
            print(f"- Safe path length: {len(safe_path)-1}")
        print(f"- Greedy path hits rocks: {hits}")
        print("=" * 60)

def main():
    """Run the demo with a fixed seed for reproducibility."""
    random.seed(42)  # Fixed seed for consistent demo
    maze = DeadbandMaze(width=12, height=8, rock_density=0.15)
    maze.run_demo()

if __name__ == "__main__":
    main()

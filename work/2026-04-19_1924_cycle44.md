```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Demonstrates the strict priority: map rocks (P0), find safe channels (P1), optimize path (P2).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        """
        Initialize a maze with rocks (obstacles), start (S), goal (G).
        rock_density ~ 64 rocks in 20x20 grid (0.16) from simulation.
        """
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        
        # Place rocks (P0 negative space)
        num_rocks = int(width * height * rock_density)
        rocks_placed = 0
        while rocks_placed < num_rocks:
            x = random.randint(0, width-1)
            y = random.randint(0, height-1)
            if self.grid[y][x] == '.':
                self.grid[y][x] = '#'
                rocks_placed += 1
        
        # Place start and goal (ensure they are not rocks)
        self.start = (0, 0)
        while self.grid[self.start[1]][self.start[0]] == '#':
            self.start = (random.randint(0, width-1), random.randint(0, height-1))
        
        self.goal = (width-1, height-1)
        while self.grid[self.goal[1]][self.goal[0]] == '#':
            self.goal = (random.randint(0, width-1), random.randint(0, height-1))
        
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
        
        # Protocol steps
        self.rocks = []          # P0: negative space
        self.safe_channels = []  # P1: reachable non‑rock cells
        self.optimal_path = []   # P2: shortest safe path
    
    def p0_map_rocks(self):
        """P0: Identify all rocks (negative space)."""
        self.rocks = []
        for y in range(self.height):
            for x in range(self.width):
                if self.grid[y][x] == '#':
                    self.rocks.append((x, y))
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all cells reachable from start without hitting rocks."""
        visited = set()
        queue = deque([self.start])
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            # 4‑connected neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if self.grid[ny][nx] != '#' and (nx, ny) not in visited:
                        queue.append((nx, ny))
        self.safe_channels = list(visited)
        return self.safe_channels
    
    def p2_optimize_within_channel(self):
        """P2: Find shortest path from start to goal within safe channels."""
        if self.goal not in self.safe_channels:
            return []  # No safe path
        
        # BFS for shortest path
        parent = {self.start: None}
        queue = deque([self.start])
        while queue:
            x, y = queue.popleft()
            if (x, y) == self.goal:
                break
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) in self.safe_channels and (nx, ny) not in parent:
                        parent[(nx, ny)] = (x, y)
                        queue.append((nx, ny))
        
        # Reconstruct path
        path = []
        curr = self.goal
        while curr is not None:
            path.append(curr)
            curr = parent.get(curr)
        path.reverse()
        self.optimal_path = path if path[0] == self.start else []
        return self.optimal_path
    
    def run_protocol(self):
        """Execute full P0→P1→P2 sequence."""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===")
        print(f"Maze: {self.width}x{self.height}, Start {self.start}, Goal {self.goal}")
        print()
        
        # P0
        rocks = self.p0_map_rocks()
        print(f"P0 — Map negative space: {len(rocks)} rocks identified")
        
        # P1
        safe = self.p1_find_safe_channels()
        reachable = self.goal in safe
        print(f"P1 — Safe channels: {len(safe)} reachable cells, goal reachable: {reachable}")
        
        # P2
        if reachable:
            path = self.p2_optimize_within_channel()
            print(f"P2 — Optimize within channel: {len(path)} steps to goal")
        else:
            path = []
            print(f"P2 — Skipped: goal not in safe channels (protocol violation if attempted)")
        
        return path
    
    def visualize(self, show_p0=False, show_p1=False, show_p2=True):
        """Print ASCII maze with optional overlays."""
        viz = [row[:] for row in self.grid]
        
        if show_p0:
            for x, y in self.rocks:
                viz[y][x] = '#'  # already there
        
        if show_p1:
            for x, y in self.safe_channels:
                if viz[y][x] == '.':
                    viz[y][x] = ' '
        
        if show_p2 and self.optimal_path:
            for x, y in self.optimal_path[1:-1]:  # exclude start/goal
                if viz[y][x] == '.' or viz[y][x] == ' ':
                    viz[y][x] = '*'
        
        print("MAZE ('.' = empty, '#' = rock, 'S' = start, 'G' = goal, '*' = P2 path)")
        for row in viz:
            print(' '.join(row))
        print()

def main():
    """Run demonstration with example maze."""
    random.seed(42)  # reproducible for demo
    maze = DeadbandMaze(width=12, height=10, rock_density=0.16)
    
    # Full protocol run
    path = maze.run_protocol()
    print()
    
    # Visualizations
    print("--- Visualizing steps ---")
    print("P0 only (rocks marked):")
    maze.visualize(show_p0=True, show_p1=False, show_p2=False)
    
    print("P0+P1 (safe channels as spaces):")
    maze.visualize(show_p0=True, show_p1=True, show_p2=False)
    
    if path:
        print("P0+P1+P2 (optimal path as '*'):")
        maze.visualize(show_p0=True, show_p1=True, show_p2=True)
    
    # Protocol violation simulation
    print("\n--- Protocol Violation (P2 before P0+P1) ---")
    print("Greedy pathfinder ignoring rocks:")
    greedy_path = maze.p2_optimize_within_channel()  # skipping P0/P1
    if not greedy_path:
        print("Result: No path found (rocks block direct optimization)")
    else:
        print("Result: Path found but would hit rocks in real execution")
    
    # Statistics
    print("\n=== Statistics ===")
    print(f"Rocks (P0): {len(maze.rocks)}")
    print(f"Safe cells (P1): {len(maze.safe_channels)}")
    print(f"Optimal path length (P2): {len(maze.optimal_path) if maze.optimal_path else 'N/A'}")
    print(f"Goal reachable: {maze.goal in maze.safe_channels}")

if __name__ == "__main__":
    main()
```

**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Complete. Based on `constraint_sim_final.json` (20×20 maze, 50 runs) and DEADBAND-PROTOCOL.md.

**Output:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

Simulates the three-phase protocol:
P0: Map rocks (obstacles)
P1: Find safe channels (deadbands)
P2: Optimize path within safe channels

Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Optimal path length: 21, Rocks: 64, Deadband success: 50/50
"""

import random
from typing import List, Tuple, Set

# Constants from fleet data
MAZE_SIZE = 20
OPTIMAL_PATH_LEN = 21
NUM_ROCKS = 64

def generate_maze() -> Tuple[List[List[str]], Tuple[int, int], Tuple[int, int]]:
    """Generate a 20×20 maze with start (0,0), goal (19,19), and rocks."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    
    # Start and goal
    start = (0, 0)
    goal = (MAZE_SIZE-1, MAZE_SIZE-1)
    maze[start[0]][start[1]] = 'S'
    maze[goal[0]][goal[1]] = 'G'
    
    # Place rocks (64 total)
    rocks_placed = 0
    while rocks_placed < NUM_ROCKS:
        x = random.randint(0, MAZE_SIZE-1)
        y = random.randint(0, MAZE_SIZE-1)
        if maze[x][y] == '.':
            maze[x][y] = '#'
            rocks_placed += 1
    
    return maze, start, goal

def print_maze(maze: List[List[str]], path: List[Tuple[int, int]] = None):
    """Print maze with optional path overlay."""
    for i in range(MAZE_SIZE):
        row = []
        for j in range(MAZE_SIZE):
            cell = maze[i][j]
            if path and (i, j) in path:
                if cell == 'S':
                    row.append('S')
                elif cell == 'G':
                    row.append('G')
                else:
                    row.append('*')
            else:
                row.append(cell)
        print(' '.join(row))

def p0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Identify negative space (rocks)."""
    rocks = set()
    for i in range(MAZE_SIZE):
        for j in range(MAZE_SIZE):
            if maze[i][j] == '#':
                rocks.add((i, j))
    print(f"[P0] Mapped {len(rocks)} rocks")
    return rocks

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]], 
                         start: Tuple[int, int], goal: Tuple[int, int]) -> List[List[Tuple[int, int]]]:
    """P1: Find all safe paths (deadbands) avoiding rocks."""
    # Simple BFS to find all safe paths (max 3 for demo)
    from collections import deque
    
    safe_paths = []
    queue = deque([([start], set([start]))])
    
    while queue and len(safe_paths) < 3:
        path, visited = queue.popleft()
        x, y = path[-1]
        
        if (x, y) == goal:
            safe_paths.append(path)
            continue
        
        # Explore neighbors (down, right, up, left)
        for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    new_path = path + [(nx, ny)]
                    new_visited = visited.copy()
                    new_visited.add((nx, ny))
                    queue.append((new_path, new_visited))
    
    print(f"[P1] Found {len(safe_paths)} safe channels")
    return safe_paths

def p2_optimize_within_channel(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """P2: Select shortest path from safe channels."""
    if not safe_paths:
        return []
    
    optimal_path = min(safe_paths, key=len)
    print(f"[P2] Selected optimal path length: {len(optimal_path)} (optimal is {OPTIMAL_PATH_LEN})")
    return optimal_path

def greedy_unconstrained(maze: List[List[str]], start: Tuple[int, int], goal: Tuple[int, int]) -> List[Tuple[int, int]]:
    """Greedy P2-only approach (violates protocol)."""
    path = [start]
    x, y = start
    visited = set([start])
    
    while (x, y) != goal:
        # Move toward goal (down/right bias)
        moves = []
        if x < goal[0] and maze[x+1][y] != '#':
            moves.append((x+1, y))
        if y < goal[1] and maze[x][y+1] != '#':
            moves.append((x, y+1))
        if not moves:
            # Fallback
            if x > 0 and maze[x-1][y] != '#':
                moves.append((x-1, y))
            if y > 0 and maze[x][y-1] != '#':
                moves.append((x, y-1))
        
        if not moves:
            break  # Stuck
        
        next_pos = moves[0]
        if next_pos in visited:
            break
        
        path.append(next_pos)
        visited.add(next_pos)
        x, y = next_pos
    
    return path

def main():
    print("=" * 60)
    print("DEADBAND PROTOCOL DEMO — Tinker (Cocapn Fleet)")
    print("=" * 60)
    
    # Generate maze
    maze, start, goal = generate_maze()
    
    print("\n[MAZE] S=Start, G=Goal, #=Rock, .=Empty")
    print_maze(maze)
    
    print("\n" + "=" * 60)
    print("VIOLATION: P2-ONLY (GREEDY UNCONSTRAINED)")
    print("=" * 60)
    greedy_path = greedy_unconstrained(maze, start, goal)
    print(f"Path length: {len(greedy_path)}")
    if greedy_path[-1] == goal:
        print("Result: SUCCESS (but unreliable)")
    else:
        print("Result: FAILED (hit rock or stuck)")
    print_maze(maze, greedy_path)
    
    print("\n" + "=" * 60)
    print("PROTOCOL: P0→P1→P2 (DEADBAND APPROACH)")
    print("=" * 60)
    
    # P0: Map rocks
    rocks = p0_map_rocks(maze)
    
    # P1: Find safe channels
    safe_paths = p1_find_safe_channels(maze, rocks, start, goal)
    
    if not safe_paths:
        print("[ERROR] No safe channels found")
        return
    
    # P2: Optimize within safe channels
    optimal_path = p2_optimize_within_channel(safe_paths)
    
    print(f"\n[RESULT] Path reaches goal: {optimal_path[-1] == goal}")
    print_maze(maze, optimal_path)
    
    # Statistics
    print("\n" + "=" * 60)
    print("STATISTICS (based on fleet data)")
    print("=" * 60)
    print(f"Optimal path length: {OPTIMAL_PATH_LEN}")
    print(f"Rocks in maze: {NUM_ROCKS}")
    print("Greedy (P2-only) success: 0/50 (0%)")
    print("Deadband (P0→P1→P2) success: 50/50 (100%)")
    print("\nProtocol priority:")
    print("P0: Map negative space (rocks) — WHAT NOT TO DO")
    print("P1: Find safe channels — WHERE YOU CAN BE")
    print("P2: Optimize within channels — BEST PATH")

if __name__ == "__main__":
    main()
```

**Key Points Demonstrated:**
1. **P0 (Map Rocks)**: Identifies all `#` obstacles as negative space
2. **P1 (Safe Channels)**: Finds multiple paths avoiding rocks (deadbands)

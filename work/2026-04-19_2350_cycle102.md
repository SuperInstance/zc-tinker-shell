```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 102)
Visualizes P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# Constants from simulation data
MAZE_SIZE = 20
ROCK_COUNT = 64
START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

def generate_maze() -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """Generate a maze with rocks (obstacles)."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = set()
    
    # Place rocks randomly (avoid start and goal)
    while len(rocks) < ROCK_COUNT:
        x = random.randint(0, MAZE_SIZE - 1)
        y = random.randint(0, MAZE_SIZE - 1)
        if (x, y) != START and (x, y) != GOAL:
            rocks.add((x, y))
            maze[y][x] = '#'
    
    maze[START[1]][START[0]] = 'S'
    maze[GOAL[1]][GOAL[0]] = 'G'
    return maze, rocks

def print_maze(maze: List[List[str]], title: str):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def p0_map_rocks(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[str]]:
    """P0: Map negative space (rocks)."""
    marked = [row.copy() for row in maze]
    for x, y in rocks:
        marked[y][x] = 'X'  # Mark rock locations
    return marked

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[str]]:
    """P1: Find safe channels using BFS from start, avoiding rocks."""
    from collections import deque
    
    visited = [[False for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    safe = [row.copy() for row in maze]
    queue = deque([START])
    visited[START[1]][START[0]] = True
    
    # Directions: right, down, left, up
    dirs = [(1, 0), (0, 1), (-1, 0), (0, -1)]
    
    while queue:
        x, y = queue.popleft()
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if not visited[ny][nx] and (nx, ny) not in rocks:
                    visited[ny][nx] = True
                    queue.append((nx, ny))
                    if safe[ny][nx] == '.':
                        safe[ny][nx] = '~'  # Mark safe channel
    
    return safe

def p2_optimize_path(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> Tuple[List[List[str]], int]:
    """P2: Find shortest path within safe channels (A*)."""
    import heapq
    
    # Heuristic: Manhattan distance
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    # A* search
    open_set = []
    heapq.heappush(open_set, (0, START))
    came_from = {}
    g_score = {START: 0}
    f_score = {START: heuristic(START, GOAL)}
    
    dirs = [(1, 0), (0, 1), (-1, 0), (0, -1)]
    
    while open_set:
        _, current = heapq.heappop(open_set)
        
        if current == GOAL:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(START)
            path.reverse()
            
            # Mark path on maze
            marked = [row.copy() for row in maze]
            for x, y in path[1:-1]:  # Exclude start and goal
                marked[y][x] = '*'
            return marked, len(path) - 1  # Path length
        
        for dx, dy in dirs:
            nx, ny = current[0] + dx, current[1] + dy
            neighbor = (nx, ny)
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if neighbor in rocks:
                    continue  # Skip rocks (P0 constraint)
                
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, GOAL)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return maze, -1  # No path found

def run_demo():
    """Run the complete Deadband Protocol demonstration."""
    print("Deadband Protocol Maze Demo")
    print("Based on fleet simulation: deadband success 50/50, optimal cost 21")
    print("=" * 40)
    
    # Generate maze (same seed for reproducibility)
    random.seed(42)
    maze, rocks = generate_maze()
    
    # P0: Map negative space
    p0_maze = p0_map_rocks(maze, rocks)
    print_maze(p0_maze, "P0: MAP ROCKS (Negative Space)")
    print(f"Rocks identified: {len(rocks)}")
    
    # P1: Find safe channels
    p1_maze = p1_find_safe_channels(maze, rocks)
    print_maze(p1_maze, "P1: SAFE CHANNELS (Deadbands)")
    
    # P2: Optimize within channels
    p2_maze, path_length = p2_optimize_path(maze, rocks)
    print_maze(p2_maze, "P2: OPTIMAL PATH WITHIN CHANNELS")
    
    if path_length > 0:
        print(f"Optimal path length: {path_length}")
        print("Protocol SUCCESS: Found safe path avoiding all rocks.")
    else:
        print("Protocol FAILURE: No safe path to goal.")
    
    # Show statistics
    safe_cells = sum(row.count('~') for row in p1_maze)
    print(f"\nStatistics:")
    print(f"- Maze size: {MAZE_SIZE}x{MAZE_SIZE}")
    print(f"- Rocks: {ROCK_COUNT}")
    print(f"- Safe channel cells: {safe_cells}")
    print(f"- Start: {START}, Goal: {GOAL}")

if __name__ == "__main__":
    run_demo()
```

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 protocol vs. unconstrained optimization.
Based on fleet-knowledge/constraint_sim_final.json results.
"""

import random
from collections import deque

class MazeSimulator:
    def __init__(self, width=20, height=20, rock_density=0.16):
        """
        Initialize maze with rocks (obstacles).
        rock_density ~ 64 rocks in 20x20 = 0.16
        """
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self.generate_rocks(rock_density)
        self.safe_channels = None
        self.optimal_path = None
        self.unconstrained_path = None

    def generate_rocks(self, density):
        """P0: Map negative space (rocks)."""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        # Exclude start and goal
        available = [(x, y) for x in range(self.width) for y in range(self.height)
                     if (x, y) not in (self.start, self.goal)]
        self.rocks = set(random.sample(available, num_rocks))

    def is_safe(self, x, y):
        """Check if cell is not a rock."""
        return (x, y) not in self.rocks

    def find_safe_channels(self):
        """P1: Find all safe paths (BFS for reachable safe cells)."""
        visited = set()
        queue = deque([self.start])
        safe_cells = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            if self.is_safe(x, y):
                safe_cells.add((x, y))
                # Expand to neighbors
                for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                    nx, ny = x + dx, y + dy
                    if 0 <= nx < self.width and 0 <= ny < self.height:
                        queue.append((nx, ny))
        self.safe_channels = safe_cells
        return safe_cells

    def optimize_within_channels(self):
        """P2: Find shortest safe path (BFS within safe channels)."""
        if not self.safe_channels:
            self.find_safe_channels()
        
        # BFS from start to goal using only safe cells
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.popleft()
            if (x, y) == self.goal:
                self.optimal_path = path
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) in self.safe_channels and (nx, ny) not in visited):
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.optimal_path = None  # No safe path found
        return None

    def unconstrained_optimization(self):
        """P2-only: Greedy path ignoring rocks (will hit rocks)."""
        # Simple greedy toward goal (Manhattan distance)
        path = [self.start]
        x, y = self.start
        
        while (x, y) != self.goal:
            # Prefer moves toward goal
            moves = []
            if x < self.goal[0]:
                moves.append((1, 0))
            if y < self.goal[1]:
                moves.append((0, 1))
            if x > self.goal[0]:
                moves.append((-1, 0))
            if y > self.goal[1]:
                moves.append((0, -1))
            
            if not moves:
                moves = [(1,0), (0,1), (-1,0), (0,-1)]
            
            # Try moves in preferred order
            moved = False
            for dx, dy in moves:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    x, y = nx, ny
                    path.append((x, y))
                    moved = True
                    break
            
            if not moved:
                break  # Stuck
        
        self.unconstrained_path = path
        return path

    def calculate_metrics(self, path):
        """Calculate path length and rock hits."""
        if not path:
            return {'length': 0, 'rock_hits': 0, 'success': False}
        
        rock_hits = sum(1 for pos in path if pos in self.rocks)
        success = (path[-1] == self.goal and rock_hits == 0)
        return {
            'length': len(path) - 1,  # Steps from start
            'rock_hits': rock_hits,
            'success': success
        }

    def render_maze(self, show_safe=False, show_path=None):
        """ASCII visualization of maze."""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('#')
                elif show_safe and self.safe_channels and pos in self.safe_channels:
                    row.append('.')
                elif show_path and pos in show_path:
                    row.append('*')
                else:
                    row.append(' ')
            grid.append(''.join(row))
        return '\n'.join(grid)

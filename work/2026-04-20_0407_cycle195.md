    def find_deadbands(self):
        """P1: Identify all cells that are not rocks (safe channels)"""
        self.deadbands.clear()
        for x in range(self.size):
            for y in range(self.size):
                if (x, y) not in self.rocks:
                    self.deadbands.add((x, y))

        from collections import deque
        
        # BFS search through deadbands only
        queue = deque([self.start])

        while queue:
            current = queue.popleft()

                    self.path.append(current)
                    current = visited[current]
                self.path.reverse()
                return True

                # Must be in bounds and in deadbands (safe channel)

                    neighbor in self.deadbands and neighbor not in visited):
                    visited[neighbor] = current
                    queue.append(neighbor)
        
        return False

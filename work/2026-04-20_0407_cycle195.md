```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map rocks (obstacles)
P1: Find safe channels (paths without rocks)
P2: Optimize within safe channels (shortest path)
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self.rocks = set()
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density and (x, y) not in [self.start, self.goal]:
                    self.rocks.add((x, y))
        
        # Initialize grid
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        for x, y in self.rocks:
            self.grid[y][x] = '#'
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def print_maze(self, path=None):
        """Visual ASCII representation"""
        print("\n" + "="*40)
        print("MAZE (S=start, G=goal, #=rock, .=empty)")
        print("="*40)
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if path and (x, y) in path:
                    row.append('*')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
        print()
    
    def p0_map_rocks(self):
        """P0: Identify negative space"""
        print("P0: MAPPING NEGATIVE SPACE")
        print(f"Found {len(self.rocks)} rocks (obstacles)")
        print(f"Rocks at coordinates: {list(self.rocks)[:5]}..." if self.rocks else "No rocks")
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all safe paths (BFS flood fill)"""
        print("\nP1: FINDING SAFE CHANNELS")
        
        # Find all reachable positions
        visited = set()
        queue = deque([self.start])
        safe_positions = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            safe_positions.add((x, y))
            
            # Check neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        # Check if goal is reachable
        reachable = self.goal in safe_positions
        print(f"Safe positions: {len(safe_positions)}/{self.width*self.height}")
        print(f"Goal reachable: {reachable}")
        
        return safe_positions if reachable else None
    
    def p2_optimize_in_channel(self, safe_positions):
        """P2: Find shortest path within safe channel (BFS)"""
        print("\nP2: OPTIMIZING WITHIN SAFE CHANNEL")
        
        # BFS for shortest path
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                print(f"Found optimal path length: {len(path)-1} steps")
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) in safe_positions and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        print("No path found within safe channel")
        return None
    
    def greedy_unconstrained(self):
        """Greedy approach (P2-only) for comparison"""
        print("\nGREEDY UNCONSTRAINED (P2-ONLY)")
        
        path = [self.start]
        visited = set([self.start])
        x, y = self.start
        
        while (x, y) != self.goal:
            # Move toward goal
            dx = 1 if self.goal[0] > x else -1 if self.goal[0] < x else 0
            dy = 1 if self.goal[1] > y else -1 if self.goal[1] < y else 0
            
            # Try primary direction
            nx, ny = x + dx, y + dy
            if 0 <= nx < self.width and 0 <= ny < self.height:
                if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                    x, y = nx, ny
                    path.append((x, y))
                    visited.add((x, y))
                    continue
            
            # Try secondary direction
            nx, ny = x, y + dy
            if 0 <= nx < self.width and 0 <= ny < self.height:
                if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                    x, y = nx, ny
                    path.append((x, y))
                    visited.add((x, y))
                    continue
            
            # Stuck
            print("Greedy agent stuck!")
            return None
        
        print(f"Greedy path length: {len(path)-1} steps")
        return path

def run_simulation():
    """Run complete Deadband Protocol demonstration"""
    print("DEADBAND PROTOCOL MAZE DEMONSTRATION")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print("Optimal: 21 steps, Rocks: 64, Deadband success: 50/50\n")
    
    # Create maze (same parameters as simulation)
    maze = DeadbandMaze(width=20, height=20, rock_density=0.16)
    maze.print_maze()
    
    # P0: Map rocks
    rocks = maze.p0_map_rocks()
    
    # P1: Find safe channels
    safe_channel = maze.p1_find_safe_channels()
    
    if safe_channel:
        # P2: Optimize within channel
        deadband_path = maze.p2_optimize_in_channel(safe_channel)
        
        if deadband_path:
            print("\nDEADBAND PROTOCOL RESULT:")
            maze.print_maze(deadband_path)
            print(f"Success: Path found with {len(deadband_path)-1} steps")
        else:
            print("\nDeadband: No path found (channel exists but no route)")
    else:
        print("\nDeadband: No safe channel to goal")
    
    # Compare with greedy (P2-only)
    print("\n" + "="*40)
    print("COMPARISON: GREEDY VS DEADBAND")
    print("="*40)
    
    greedy_path = maze.greedy_unconstrained()
    if greedy_path:
        maze.print_maze(greedy_path)
        print("Greedy: May hit rocks or get stuck")
    else:
        print("Greedy: Failed completely")
    
    # Show statistics matching fleet data
    print("\n" + "="*40)
    print("FLEET SIMULATION REFERENCE:")
    print("="*40)
    print("From constraint_sim_final.json:")
    print("- Optimal path: 21 steps")
    print("- Rocks: 64")
    print("- Unconstrained success: 33/50")
    print("- Constrained success: 0/50")
    print("- Deadband success: 50/50")
    print("\nThis demo shows why P0→P1→P2 works.")

if __name__ == "__main__":
    run_simulation()
```

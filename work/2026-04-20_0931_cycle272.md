## deadband_maze_demo.py

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2 visualization
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Optimal path length: 21, Rocks: 64, Deadband success: 50/50
"""

import random
import time
from typing import List, Tuple, Set

# ========== MAZE GENERATION (matches simulation parameters) ==========
MAZE_SIZE = 20
START = (0, 0)
GOAL = (19, 19)
ROCK_COUNT = 64  # from JSON

def generate_maze() -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """Create a 20×20 maze with rocks, start (S), goal (G)."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    maze[START[0]][START[1]] = 'S'
    maze[GOAL[0]][GOAL[1]] = 'G'
    
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        r, c = random.randint(0, MAZE_SIZE-1), random.randint(0, MAZE_SIZE-1)
        if (r, c) not in {START, GOAL}:
            rocks.add((r, c))
            maze[r][c] = '#'
    return maze, rocks

def print_maze(maze: List[List[str]], path: Set[Tuple[int, int]] = None, 
               deadband: Set[Tuple[int, int]] = None, phase: str = None):
    """Print maze with optional path/deadband highlighting."""
    print(f"\n{'='*60}")
    if phase:
        print(f"PHASE {phase}")
    print()
    for r in range(MAZE_SIZE):
        row = []
        for c in range(MAZE_SIZE):
            cell = maze[r][c]
            if path and (r, c) in path:
                row.append('*')
            elif deadband and (r, c) in deadband:
                row.append('~')
            else:
                row.append(cell)
        print(' '.join(row))
    print()

# ========== DEADBAND PROTOCOL IMPLEMENTATION ==========
def p0_map_negative_space(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> Set[Tuple[int, int]]:
    """P0: Identify all rocks (failure states)."""
    print("P0 — MAP NEGATIVE SPACE")
    print(f"Identified {len(rocks)} rock cells (failure states).")
    return rocks

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> Set[Tuple[int, int]]:
    """P1: Find all cells reachable without hitting rocks (deadband)."""
    print("P1 — FIND SAFE CHANNELS")
    # Simple flood‑fill from start, avoiding rocks
    deadband = set()
    stack = [START]
    visited = set()
    
    while stack:
        r, c = stack.pop()
        if (r, c) in visited:
            continue
        visited.add((r, c))
        deadband.add((r, c))
        
        # 4‑neighbor expansion
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if (nr, nc) not in rocks and (nr, nc) not in visited:
                    stack.append((nr, nc))
    
    print(f"Safe channel (deadband) contains {len(deadband)} reachable cells.")
    if GOAL in deadband:
        print("✓ Goal is reachable within safe channel.")
    else:
        print("✗ Goal is NOT reachable — no safe path exists.")
    return deadband

def p2_optimize_within_channel(maze: List[List[str]], deadband: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
    """P2: Find shortest path from start to goal staying within deadband (A*)."""
    print("P2 — OPTIMIZE WITHIN CHANNEL")
    if GOAL not in deadband:
        return []
    
    # Manhattan distance heuristic
    def h(pos):
        return abs(pos[0] - GOAL[0]) + abs(pos[1] - GOAL[1])
    
    open_set = {START}
    came_from = {}
    g_score = {START: 0}
    f_score = {START: h(START)}
    
    while open_set:
        current = min(open_set, key=lambda p: f_score.get(p, float('inf')))
        if current == GOAL:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(START)
            path.reverse()
            print(f"Found optimal path of length {len(path)-1} steps.")
            return path
        
        open_set.remove(current)
        r, c = current
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            neighbor = (nr, nc)
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if neighbor not in deadband:
                    continue  # outside safe channel — skip
                tentative_g = g_score[current] + 1
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + h(neighbor)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
    
    return []

# ========== SIMULATION RUN ==========
def run_deadband_protocol():
    """Execute full P0→P1→P2 protocol with visualization."""
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print(f"Maze: {MAZE_SIZE}×{MAZE_SIZE}, Rocks: {ROCK_COUNT}, Start: S, Goal: G")
    print("="*60)
    
    # Generate environment
    maze, rocks = generate_maze()
    print_maze(maze, phase="Initial Maze")
    time.sleep(1.5)
    
    # P0: Map negative space
    rocks_set = p0_map_negative_space(maze, rocks)
    print_maze(maze, deadband=rocks_set, phase="P0 — Rocks (#)")
    time.sleep(2)
    
    # P1: Find safe channels
    deadband = p1_find_safe_channels(maze, rocks_set)
    print_maze(maze, deadband=deadband, phase="P1 — Safe Channel (~)")
    time.sleep(2)
    
    # P2: Optimize within channel
    path = p2_optimize_within_channel(maze, deadband)
    if path:
        print_maze(maze, path=set(path), phase="P2 — Optimal Path (*)")
        print("PROTOCOL RESULT: SUCCESS")
        print(f"Path length: {len(path)-1} steps (optimal: 21)")
    else:
        print("PROTOCOL RESULT: FAILURE — No safe path to goal.")
    
    # Show statistics matching fleet data
    print("\n" + "="*60)
    print("SIMULATION STATISTICS (vs fleet data)")
    print(f"Rocks identified (P0): {len(rocks_set)} / {ROCK_COUNT}")
    print(f"Safe channel size (P1): {len(deadband)} cells")
    print(f"Path found: {'Yes' if path else 'No'}")
    if path:
        print(f"Path length: {len(path)-1} (optimal: 21)")
    print("\nReference (constraint_sim_final.json):")
    print("  Deadband success: 50/50 runs")
    print("  Deadband avg steps: 21.0")
    print("  Constrained success: 0/50 (P2‑only fails)")

# ========== MAIN ==========
if __name__ == "__main__":
    random.seed(42)  # reproducible for demo
    run_deadband_protocol()
```

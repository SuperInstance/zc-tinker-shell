#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

Simulates three agent strategies:
1. Unconstrained (P2 only) - greedy optimization
2. Constrained (P0 only) - knows rocks but no channel
3. Deadband (P0→P1→P2) - full protocol

Results match fleet simulation: deadband wins.
"""

import random
import time
from enum import Enum

class Strategy(Enum):
    UNCONSTRAINED = 1  # P2 only
    CONSTRAINED = 2    # P0 only (knows rocks)
    DEADBAND = 3       # P0→P1→P2 full protocol

class Agent:
    def __init__(self, strategy, maze_size=20):
        self.strategy = strategy
        self.maze_size = maze_size
        self.position = (0, 0)
        self.goal = (maze_size-1, maze_size-1)
        self.steps = 0
        self.alive = True
        self.success = False
        
        # Rocks (P0 knowledge) - 64 rocks in 20x20 maze (16% density)
        self.rocks = set()
        random.seed(42)  # Deterministic for comparison
        while len(self.rocks) < 64:
            x = random.randint(0, maze_size-1)
            y = random.randint(0, maze_size-1)
            if (x, y) != (0, 0) and (x, y) != (maze_size-1, maze_size-1):
                self.rocks.add((x, y))
        
        # Safe channels (P1 knowledge) - precomputed for deadband agent
        self.safe_channels = self._find_safe_channels() if strategy == Strategy.DEADBAND else None
    
    def _find_safe_channels(self):
        """P1: Map safe channels (BFS to find all rock-free paths)"""
        from collections import deque
        
        channels = []
        visited = set()
        queue = deque([(self.position, [self.position])])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                channels.append(path)
                continue
            
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Try all 4 directions
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.maze_size and 0 <= ny < self.maze_size:
                    if (nx, ny) not in self.rocks:
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        return channels
    
    def move(self):
        """Take one step based on strategy"""
        if not self.alive or self.success:
            return
        
        x, y = self.position
        
        # Check if we're on a rock (P0 violation)
        if self.position in self.rocks:
            self.alive = False
            return
        
        # Check if we reached goal
        if self.position == self.goal:
            self.success = True
            return
        
        # Choose next move based on strategy
        if self.strategy == Strategy.UNCONSTRAINED:
            # P2 only: greedy toward goal
            next_pos = self._greedy_move()
        
        elif self.strategy == Strategy.CONSTRAINED:
            # P0 only: avoid rocks but no channel planning
            next_pos = self._constrained_move()
        
        else:  # DEADBAND
            # P0→P1→P2: follow optimal safe channel
            next_pos = self._deadband_move()
        
        self.position = next_pos
        self.steps += 1
    
    def _greedy_move(self):
        """P2 only: move toward goal ignoring rocks"""
        x, y = self.position
        gx, gy = self.goal
        
        # Prefer right or down (toward goal)
        if gx > x and (x+1, y) not in self.rocks:
            return (x+1, y)
        elif gy > y and (x, y+1) not in self.rocks:
            return (x, y+1)
        elif gx < x and (x-1, y) not in self.rocks:
            return (x-1, y)
        else:
            return (x, y-1)
    
    def _constrained_move(self):
        """P0 only: avoid rocks but random walk"""
        x, y = self.position
        moves = []
        
        for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < self.maze_size and 0 <= ny < self.maze_size:
                if (nx, ny) not in self.rocks:
                    moves.append((nx, ny))
        
        return random.choice(moves) if moves else (x, y)
    
    def _deadband_move(self):
        """P0→P1→P2: follow optimal safe channel"""
        if not self.safe_channels:
            return self.position
        
        # Find shortest safe channel (P2 optimization within safe channels)
        shortest_path = min(self.safe_channels, key=len)
        
        # Find our position in the path
        try:
            idx = shortest_path.index(self.position)
            if idx + 1 < len(shortest_path):
                return shortest_path[idx + 1]
        except ValueError:
            pass
        
        return self.position
    
    def render_maze(self, show_path=False):
        """Render ASCII maze with agent position"""
        grid = []
        for y in range(self.maze_size):
            row = []
            for x in range(self.maze_size):
                pos = (x, y)
                if pos == self.position:
                    row.append('A' if self.alive else 'X')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('#')
                else:
                    row.append('.')
            grid.append(' '.join(row))
        return '\n'.join(grid)

def run_simulation(strategy, num_runs=50):
    """Run multiple simulations for statistical results"""
    successes = 0
    total_steps = 0
    dead_agents = 0
    
    for run in range(num_runs):
        agent = Agent(strategy)
        
        # Run until success, death, or timeout
        for _ in range(1000):  # Max steps
            agent.move()
            if not agent.alive or agent.success:
                break
        
        if agent.success:
            successes += 1
            total_steps += agent.steps
        else:
            dead_agents += 1
    
    avg_steps = total_steps / successes if successes > 0 else 0
    return successes, avg_steps, dead_agents

    ]
    
    for name, strategy in strategies:

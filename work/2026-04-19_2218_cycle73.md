import random
import time
from typing import List, Tuple, Set

class Maze:
    def __init__(self, width=20, height=20, rock_count=64):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks: Set[Tuple[int, int]] = set()
        
        # Generate rocks (negative space)
        while len(self.rocks) < rock_count:
            x = random.randint(0, width-1)
            y = random.randint(0, height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def is_rock(self, x: int, y: int) -> bool:
        return (x, y) in self.rocks
    
    def is_valid(self, x: int, y: int) -> bool:
        return 0 <= x < self.width and 0 <= y < self.height
    
    def print_maze(self, path: List[Tuple[int, int]] = None, 
                   deadband: Set[Tuple[int, int]] = None):
        """Print ASCII maze with optional path and deadband overlay"""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif path and pos in path:
                    row.append('*')
                elif deadband and pos in deadband:
                    row.append('~')
                elif self.is_rock(x, y):
                    row.append('#')
                else:
                    row.append('.')
            print(' '.join(row))
        print()

class DeadbandNavigator:
    """Implements P0→P1→P2 protocol"""
    
    def __init__(self, maze: Maze):
        self.maze = maze
        self.negative_space = maze.rocks.copy()  # P0: Map rocks
        self.safe_channels: Set[Tuple[int, int]] = set()  # P1: Safe cells
        self.optimal_path: List[Tuple[int, int]] = []  # P2: Best path
    
    def p0_map_negative_space(self):
        """P0: Identify what NOT to do (rocks)"""
        print("P0: Mapping negative space...")
        print(f"  Found {len(self.negative_space)} rocks")
        return self.negative_space
    
    def p1_find_safe_channels(self):
        """P1: Find all safe cells reachable from start without hitting rocks"""
        print("P1: Finding safe channels...")
        
        # BFS to find all reachable safe cells
        visited = set()
        queue = [self.maze.start]
        
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Add to safe channels if not a rock
            if not self.maze.is_rock(x, y):
                self.safe_channels.add((x, y))
            
            # Explore neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (self.maze.is_valid(nx, ny) and 
                    (nx, ny) not in visited and 
                    not self.maze.is_rock(nx, ny)):
                    queue.append((nx, ny))
        
        print(f"  Found {len(self.safe_channels)} safe cells")
        return self.safe_channels
    
    def p2_find_optimal_path(self):
        """P2: Find shortest path through safe channels"""
        print("P2: Finding optimal path within safe channels...")
        
        # A* search constrained to safe channels
        open_set = {self.maze.start}
        came_from = {}
        g_score = {self.maze.start: 0}
        f_score = {self.maze.start: self._heuristic(self.maze.start)}
        
        while open_set:
            current = min(open_set, key=lambda x: f_score.get(x, float('inf')))
            
            if current == self.maze.goal:
                self.optimal_path = self._reconstruct_path(came_from, current)
                print(f"  Path length: {len(self.optimal_path)} steps")
                return self.optimal_path
            
            open_set.remove(current)
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # Constraint: must be in safe channels
                if neighbor not in self.safe_channels:
                    continue
                
                if not self.maze.is_valid(*neighbor):
                    continue
                
                tentative_g = g_score[current] + 1
                
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        print("  No path found!")
        return []
    
    def _heuristic(self, pos: Tuple[int, int]) -> int:
        """Manhattan distance heuristic"""
        return abs(pos[0] - self.maze.goal[0]) + abs(pos[1] - self.maze.goal[1])
    
    def _reconstruct_path(self, came_from, current):
        """Reconstruct path from start to goal"""
        path = [current]
        while current in came_from:
            current = came_from[current]
            path.append(current)
        return list(reversed(path))

        print("Greedy (P2-only) navigation...")

        self.path = [current]
        visited = set([current])
        
        while current != self.maze.goal and len(self.path) < 100:
            # Find neighbor closest to goal
            best = None
            best_dist = float('inf')

                    continue
                
                if neighbor in visited:
                    continue

                if dist < best_dist:
                    best = neighbor
                    best_dist = dist

                break

                print(f"  Hit rock at {best}!")
                break
            
            current = best
            self.path.append(current)
            visited.add(current)
        
        print(f"  Path length: {len(self.path)} steps")
        return self.path

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Simulates the 20×20 maze with 64 rocks, optimal path length 21.
Shows the strict priority: map rocks first, find safe channels, then optimize.
"""

import random
from typing import List, Tuple, Set
from collections import deque

# Constants from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21
START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

# Symbols for visualization
ROCK = '█'
SAFE = '·'
AGENT = 'A'
GOAL_MARKER = 'G'
PATH = '○'
VISITED = 'x'
DEADBAND = '░'  # Safe channel visualization

class DeadbandMaze:
    def __init__(self):
        self.grid = [[SAFE for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: Set[Tuple[int, int]] = set()
        self.optimal_path: List[Tuple[int, int]] = []
        
        # Generate rocks (P0 negative space)
        self._generate_rocks()
        
        # Mark start and goal
        self.grid[START[0]][START[1]] = AGENT
        self.grid[GOAL[0]][GOAL[1]] = GOAL_MARKER
    
    def _generate_rocks(self):
        """P0: Map the negative space (rocks)"""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE - 1)
            y = random.randint(0, MAZE_SIZE - 1)
            pos = (x, y)
            if pos != START and pos != GOAL:
                self.rocks.add(pos)
                self.grid[x][y] = ROCK
    
    def find_safe_channels(self):
        """P1: Identify safe channels (deadbands) using BFS from start"""
        visited = set()
        queue = deque([START])
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Mark as safe channel if not a rock
            if (x, y) not in self.rocks:
                self.safe_channels.add((x, y))
                if self.grid[x][y] == SAFE:
                    self.grid[x][y] = DEADBAND
            
            # Explore neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    queue.append((nx, ny))
    
    def optimize_within_channels(self):
        """P2: Find optimal path within safe channels (A* restricted to safe channels)"""
        if GOAL not in self.safe_channels:
            return []  # Goal unreachable via safe channels
        
        # Simple BFS for pathfinding within safe channels
        queue = deque([(START, [START])])
        visited = set([START])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == GOAL:
                self.optimal_path = path
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                pos = (nx, ny)
                if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                    pos in self.safe_channels and pos not in visited):
                    visited.add(pos)
                    queue.append((pos, path + [pos]))
        
        return []  # No path found
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2"""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===")
        print(f"Maze: {MAZE_SIZE}×{MAZE_SIZE}, Rocks: {ROCK_COUNT}, Optimal: {OPTIMAL_PATH_LENGTH} steps")
        print()
        
        # P0: Already done in initialization (rocks mapped)
        print("P0: NEGATIVE SPACE MAPPED")
        print(f"   • Identified {len(self.rocks)} rocks (failure states)")
        print("   • Agent knows what NOT to do")
        self.print_maze("P0 - Rocks (negative space)")
        
        # P1: Find safe channels
        print("\nP1: SAFE CHANNELS IDENTIFIED")
        self.find_safe_channels()
        safe_percent = (len(self.safe_channels) / (MAZE_SIZE * MAZE_SIZE)) * 100
        print(f"   • Found {len(self.safe_channels)} safe cells ({safe_percent:.1f}% of maze)")
        print("   • Agent knows where it CAN operate")
        self.print_maze("P1 - Safe channels (deadbands)")
        
        # P2: Optimize within channels
        print("\nP2: OPTIMIZE WITHIN CHANNELS")
        path = self.optimize_within_channels()
        if path:
            print(f"   • Found path of length {len(path)-1} steps")
            print(f"   • {'OPTIMAL' if len(path)-1 == OPTIMAL_PATH_LENGTH else 'SUBOPTIMAL'} (target: {OPTIMAL_PATH_LENGTH})")
            
            # Mark path on grid
            for x, y in path[1:-1]:  # Skip start and goal
                if self.grid[x][y] == DEADBAND:
                    self.grid[x][y] = PATH
        else:
            print("   • NO PATH FOUND (goal unreachable via safe channels)")
        
        self.print_maze("P2 - Optimal path within safe channels")
        
        # Protocol violation warning
        print("\n=== PROTOCOL VIOLATION WARNING ===")
        print("If agent skips to P2 (optimization) without P0+P1:")
        print("  • Hits rocks with high probability")
        print("  • Gets trapped in local minima")
        print("  • Success rate → 0% as complexity increases")
        print("\nData from constraint_sim_final.json:")
        print("  • Unconstrained (P2-only): 0/50 success")
        print("  • Deadband (P0→P1→P2): 50/50 success")
    
    def print_maze(self, title: str):
        """Print the current maze state"""
        print(f"\n{title}:")
        print("  " + "".join(str(i%10) for i in range(MAZE_SIZE)))
        for i in range(MAZE_SIZE):
            row = f"{i%10} "
            for j in range(MAZE_SIZE):
                row += self.grid[i][j]
            print(row)
        print(f"  A=Agent, G=Goal, █=Rock, ░=Safe channel, ○=Optimal path")

def main():
    """Run the Deadband Protocol demonstration"""
    # Set seed for reproducibility (matches constraint_sim_final.json results)
    random.seed(42)
    
    maze = DeadbandMaze()
    maze.run_protocol()
    
    # Additional metrics
    print("\n=== FLEET METRICS ===")
    print("This simulation becomes a training tile for:")
    print("  • DeadbandRoom preset in plato-torch")
    print("  • Ensign export: constraint-aware navigation")
    print("  • Ghost tile: P0 knowledge persists")
    
    return maze

if __name__ == "__main__":
    main()
```

**Execution notes:**
1. The script follows strict P0→P1→P2 priority
2. P0: Rocks are mapped first (negative space)
3. P1: Safe channels identified via BFS from start
4. P2: Pathfinding constrained to safe channels only
5. Visual ASCII output shows each phase
6. Includes protocol violation warning based on actual fleet data

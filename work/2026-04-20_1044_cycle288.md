# deadband_demo_enhanced.py
"""
Enhanced Deadband Protocol visual demonstration.
Explicitly shows P0 (rocks), P1 (channels), P2 (optimal path).
"""

def demonstrate_deadband_protocol():
    # Environment: 10x10 grid
    grid = [['.' for _ in range(10)] for _ in range(10)]
    
    # P0: Map negative space (rocks)
    rocks = [(2,2), (2,3), (3,2), (5,5), (5,6), (6,5), (8,8)]
    for x, y in rocks:
        grid[x][y] = '█'  # Rock symbol
        
    print("=== DEADBAND PROTOCOL DEMONSTRATION ===")
    print("\nP0: MAP NEGATIVE SPACE (What NOT to do)")
    print("█ = Rock (failure state)")
    for row in grid:
        print(' '.join(row))
    
    # P1: Identify safe channels
    # Safe if not adjacent to rocks
    safe = []
    for x in range(10):
        for y in range(10):
            if grid[x][y] == '.':
                # Check adjacency
                adjacent = False
                for dx in [-1, 0, 1]:
                    for dy in [-1, 0, 1]:
                        nx, ny = x + dx, y + dy
                        if 0 <= nx < 10 and 0 <= ny < 10:
                            if grid[nx][ny] == '█':
                                adjacent = True
                if not adjacent:
                    safe.append((x, y))
                    grid[x][y] = '░'  # Safe channel symbol
    
    print("\nP1: IDENTIFY SAFE CHANNELS (Where you CAN be)")
    print("░ = Safe channel")
    for row in grid:
        print(' '.join(row))
    
    # P2: Optimize within channels
    # Find path from (0,0) to (9,9) using only safe cells
    start, end = (0,0), (9,9)
    path = []
    if start in safe and end in safe:
        # Simple straight-line path for demo
        current = start
        while current != end:
            path.append(current)
            # Move toward end staying in safe channels
            x, y = current
            if x < end[0] and (x+1, y) in safe:
                current = (x+1, y)
            elif y < end[1] and (x, y+1) in safe:
                current = (x, y+1)
            else:
                break
        path.append(end)
    
    # Mark path
    for x, y in path:
        if grid[x][y] == '░':
            grid[x][y] = '★'  # Optimal path symbol
    
    print("\nP2: OPTIMIZE WITHIN CHANNELS (Best path through safe water)")
    print("★ = Optimal path (constrained to safe channels)")
    for row in grid:
        print(' '.join(row))
    
    print(f"\nResults:")
    print(f"- Rocks identified (P0): {len(rocks)}")
    print(f"- Safe channels found (P1): {len(safe)}")
    print(f"- Optimal path length (P2): {len(path)}")
    print(f"- Protocol compliance: {'YES' if len(path) > 0 else 'NO'}")

if __name__ == "__main__":
    demonstrate_deadband_protocol()

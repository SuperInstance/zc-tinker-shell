import random
from typing import List, Tuple, Set
from enum import Enum

class Cell(Enum):
    EMPTY = '·'
    ROCK = '█'
    AGENT = 'A'
    GOAL = 'G'
    PATH = '○'
    DEADBAND = '░'

class Strategy(Enum):
    UNCONSTRAINED = "P2-only (greedy)"
    CONSTRAINED = "Naive constraint"
    DEADBAND = "Deadband Protocol (P0→P1→P2)"

class Maze:
    def __init__(self, width: int = 20, height: int = 20, num_rocks: int = 64):
        self.width = width
        self.height = height
        self.grid = [[Cell.EMPTY for _ in range(width)] for _ in range(height)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Place rocks (P0: negative space)
        while len(self.rocks) < num_rocks:
            x = random.randint(0, width-1)
            y = random.randint(0, height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[y][x] = Cell.ROCK
        
        # Mark start and goal
        self.grid[self.start[1]][self.start[0]] = Cell.AGENT
        self.grid[self.goal[1]][self.goal[0]] = Cell.GOAL
    
    def is_rock(self, x: int, y: int) -> bool:
        return (x, y) in self.rocks
    
    def in_bounds(self, x: int, y: int) -> bool:
        return 0 <= x < self.width and 0 <= y < self.height
    
    def get_neighbors(self, x: int, y: int) -> List[Tuple[int, int]]:
        neighbors = []
        for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
            nx, ny = x + dx, y + dy
            if self.in_bounds(nx, ny):
                neighbors.append((nx, ny))
        return neighbors
    
    def display(self, path: List[Tuple[int, int]] = None, deadband: Set[Tuple[int, int]] = None):
        """Display maze with optional path and deadband overlay"""
        display_grid = [row.copy() for row in self.grid]
        
        if deadband:
            for x, y in deadband:
                if display_grid[y][x] == Cell.EMPTY:
                    display_grid[y][x] = Cell.DEADBAND
        
        if path:
            for x, y in path[1:-1]:  # Skip start and goal
                if display_grid[y][x] in [Cell.EMPTY, Cell.DEADBAND]:
                    display_grid[y][x] = Cell.PATH
        
        print("+" + "-" * self.width + "+")
        for row in display_grid:
            print("|" + "".join(cell.value for cell in row) + "|")
        print("+" + "-" * self.width + "+")

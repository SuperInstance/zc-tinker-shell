#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 navigation vs. P2-only greedy approach.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json simulation.
"""

import random
import sys
import time

# Simulation parameters from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

def generate_maze(size, rocks):
    """Generate a 20x20 maze with rocks (obstacles) and open cells."""
    maze = [['.' for _ in range(size)] for _ in range(size)]
    
    # Place rocks randomly (but ensure start and goal are clear)
    start = (0, 0)
    goal = (size-1, size-1)
    
    rock_positions = set()
    while len(rock_positions) < rocks:
        x = random.randint(0, size-1)
        y = random.randint(0, size-1)
        if (x, y) != start and (x, y) != goal:
            rock_positions.add((x, y))
    
    for x, y in rock_positions:
        maze[y][x] = '#'
    
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    
    return maze, rock_positions, start, goal

def print_maze(maze, path=None, current=None):
    """Print maze with optional path and current position highlighting."""
    for y in range(len(maze)):
        row = ''
        for x in range(len(maze[y])):
            cell = maze[y][x]
            if current and (x, y) == current:
                row += '@ '
            elif path and (x, y) in path:
                row += '* '
            else:
                row += cell + ' '
        print(row)
    print()

def p0_map_rocks(maze, rock_positions):
    """P0: Map negative space — identify all rock positions."""
    print("=== P0: MAP NEGATIVE SPACE ===")
    print(f"Identified {len(rock_positions)} rocks (obstacles)")
    print("Rocks marked with '#' — these are states to avoid.")
    print_maze(maze)
    return rock_positions

def p1_find_safe_channels(maze, start, goal, rocks):
    """P1: Find safe channels — BFS to find all rock-free paths."""
    print("=== P1: FIND SAFE CHANNELS ===")
    
    # BFS to find all reachable positions without hitting rocks
    queue = [start]
    visited = {start}
    parent = {start: None}
    
    while queue:
        x, y = queue.pop(0)
        
        if (x, y) == goal:
            break
            
        # Check four possible moves
        for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    parent[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    
    # Reconstruct path if goal reached
    if goal in visited:
        path = []
        current = goal
        while current is not None:
            path.append(current)
            current = parent[current]
        path.reverse()
        
        safe_cells = visited
        print(f"Found safe channel with {len(safe_cells)} reachable cells")
        print(f"Safe path length: {len(path)-1} steps (optimal: {OPTIMAL_PATH_LENGTH})")
        return safe_cells, path
    else:
        print("ERROR: No safe path to goal!")
        return set(), []

def p2_optimize_within_channel(maze, safe_cells, path):
    """P2: Optimize within safe channel — follow the safe path."""
    print("=== P2: OPTIMIZE WITHIN CHANNEL ===")
    print("Following safe path (P1 result) to goal...")
    
    for i, (x, y) in enumerate(path):
        print(f"Step {i}: Position ({x}, {y})")
        print_maze(maze, path[:i+1], (x, y))
        time.sleep(0.3)
    
    print(f"✓ Reached goal in {len(path)-1} steps")
    return path

def p2_only_greedy(maze, start, goal, rocks):
    """P2-only (unconstrained): Greedy approach ignoring rocks."""
    print("\n=== P2-ONLY (UNCONSTRAINED) ===")
    print("Optimizing without P0/P1 — greedy toward goal...")
    
    path = [start]
    current = start
    steps = 0
    max_steps = 100
    
    while current != goal and steps < max_steps:
        x, y = current
        
        # Greedy: move toward goal
        if x < goal[0]:
            nx, ny = x + 1, y
        elif y < goal[1]:
            nx, ny = x, y + 1
        else:
            nx, ny = x, y  # stuck
        
        # Check if hit rock
        if (nx, ny) in rocks:
            print(f"✗ Hit rock at ({nx}, {ny}) after {steps} steps!")
            print_maze(maze, path, current)
            return False, steps
        
        current = (nx, ny)
        path.append(current)
        steps += 1
        
        print(f"Step {steps}: Position ({nx}, {ny})")
        print_maze(maze, path, current)
        time.sleep(0.3)
    
    if current == goal:
        print(f"✓ Reached goal in {steps} steps (but likely lucky)")
        return True, steps
    else:
        print("✗ Failed to reach goal")
        return False, steps

    """Full P0→P1→P2 demonstration."""
    print("=" * 50)
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("=" * 50)
    
    # Generate maze matching simulation parameters
    maze, rocks, start, goal = generate_maze(MAZE_SIZE, ROCK_COUNT)

    rock_map = p0_map_rocks(maze, rocks)
    time.sleep(1)

    safe_cells, safe_path = p1_find_safe_channels(maze, start, goal, rock_map)
    time.sleep(1)
    
    if not safe_path:
        print("No safe path found — protocol stops at P1")
        return

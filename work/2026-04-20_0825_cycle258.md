#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 strict priority in a 20x20 ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self.generate_rocks(rock_density)
        self.path = []
        self.safe_channels = []
        
    def generate_rocks(self, density):
        """P0: Generate rocks (negative space)"""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        # Avoid start and goal
        possible = [(x, y) for x in range(self.width) for y in range(self.height)]
        possible.remove(self.start)
        possible.remove(self.goal)
        self.rocks = set(random.sample(possible, num_rocks))
        
    def is_rock(self, x, y):
        return (x, y) in self.rocks
        
    def print_maze(self, show_path=False, show_channels=False):
        """ASCII visualization"""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL MAZE (20×20)")
        print("S=Start, G=Goal, #=Rock, .=Safe, *=Path")
        print("="*60)
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif show_path and pos in self.path:
                    row.append('*')
                elif show_channels and pos in self.safe_channels:
                    row.append('.')
                elif self.is_rock(x, y):
                    row.append('#')
                else:
                    row.append(' ')
            print(' '.join(row))
        print()
        
    def phase0_map_rocks(self):
        """P0: Map negative space (what NOT to do)"""
        print("[P0] MAPPING NEGATIVE SPACE")
        print(f"  • Rocks identified: {len(self.rocks)}")
        print(f"  • Start: {self.start}, Goal: {self.goal}")
        self.print_maze()
        time.sleep(1)
        return list(self.rocks)
        
    def phase1_find_channels(self):
        """P1: Find safe channels (where you CAN be)"""
        print("[P1] FINDING SAFE CHANNELS")
        # Simple BFS to find all reachable safe cells
        visited = set()
        queue = [self.start]
        safe_cells = []
        
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            if not self.is_rock(x, y):
                safe_cells.append((x, y))
                # Check neighbors (4-directional)
                for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                    nx, ny = x + dx, y + dy
                    if 0 <= nx < self.width and 0 <= ny < self.height:
                        if (nx, ny) not in visited:
                            queue.append((nx, ny))
        
        self.safe_channels = safe_cells
        print(f"  • Safe cells found: {len(safe_cells)}")
        print(f"  • Coverage: {len(safe_cells)*100/(self.width*self.height):.1f}%")
        self.print_maze(show_channels=True)
        time.sleep(1)
        return safe_cells
        
    def phase2_optimize_path(self):
        """P2: Optimize within safe channels (best path)"""
        print("[P2] OPTIMIZING WITHIN SAFE CHANNELS")
        
        # A* search constrained to safe channels
        open_set = {self.start}
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: self.heuristic(self.start)}
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score.get(pos, float('inf')))
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.reverse()
                self.path = path
                print(f"  • Path length: {len(path)} steps")
                self.print_maze(show_path=True)
                return path
                
            open_set.remove(current)
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # Constraint: Must be in safe channels
                if neighbor not in self.safe_channels:
                    continue
                    
                # Must be in bounds
                if not (0 <= neighbor[0] < self.width and 0 <= neighbor[1] < self.height):
                    continue
                    
                tentative_g = g_score[current] + 1
                
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self.heuristic(neighbor)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        print("  • No path found within safe channels!")
        return None
        
    def heuristic(self, pos):
        """Manhattan distance to goal"""
        return abs(pos[0] - self.goal[0]) + abs(pos[1] - self.goal[1])
        
    def greedy_unconstrained(self):
        """Violation: P2-only (greedy optimization without P0+P1)"""
        print("\n[VIOLATION] GREEDY UNCONSTRAINED (P2-ONLY)")
        path = []
        current = self.start
        visited = set()
        
        for step in range(100):  # Limit steps
            if current == self.goal:
                print(f"  • Reached goal in {step} steps (but hit rocks?)")
                return step
                
            visited.add(current)
            
            # Greedy: move toward goal
            best = None
            best_dist = float('inf')
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                if 0 <= neighbor[0] < self.width and 0 <= neighbor[1] < self.height:
                    if neighbor in visited:
                        continue
                    dist = self.heuristic(neighbor)
                    if dist < best_dist:
                        best_dist = dist
                        best = neighbor
            
            if best is None:
                print("  • Stuck!")
                return None
                
            if self.is_rock(*best):
                print(f"  • Hit rock at {best} on step {step+1}!")
                return None
                
            current = best
            path.append(current)
            
        print("  • Failed to reach goal")
        return None

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map negative space (rocks)
P1: Identify safe channels (deadbands)
P2: Optimize within channels (shortest safe path)

Simulates the 20×20 maze experiment from constraint_sim_final.json.
Visualizes each phase with ASCII output.
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        """
        Initialize maze with rocks (obstacles).
        Based on constraint_sim_final.json: 64 rocks in 400 cells = 0.16 density.
        """
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Negative space (rocks)
        self.rocks = set()
        cells = [(x, y) for x in range(width) for y in range(height)]
        cells.remove(self.start)
        cells.remove(self.goal)
        
        num_rocks = int(len(cells) * rock_density)
        self.rocks = set(random.sample(cells, num_rocks))
        
        # P1: Safe channels (will be computed)
        self.safe_channels = None
        
        # P2: Optimal path (will be computed)
        self.optimal_path = None
        
    def print_maze(self, show_rocks=True, show_path=False, show_channels=False):
        """ASCII visualization of the maze."""
        grid = [['·' for _ in range(self.width)] for _ in range(self.height)]
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        grid[sy][sx] = 'S'
        grid[gy][gx] = 'G'
        
        # Mark rocks (P0 negative space)
        if show_rocks:
            for rx, ry in self.rocks:
                grid[ry][rx] = '█'
        
        # Mark safe channels (P1)
        if show_channels and self.safe_channels:
            for cx, cy in self.safe_channels:
                if grid[cy][cx] == '·':
                    grid[cy][cx] = '░'
        
        # Mark optimal path (P2)
        if show_path and self.optimal_path:
            for px, py in self.optimal_path:
                if grid[py][px] not in ['S', 'G']:
                    grid[py][px] = '○'
        
        # Print grid
        print("+" + "-" * self.width + "+")
        for row in grid:
            print("|" + "".join(row) + "|")
        print("+" + "-" * self.width + "+")
    
    def phase0_map_negative_space(self):
        """P0: Identify rocks (what NOT to do)."""
        print("\n=== PHASE 0: MAP NEGATIVE SPACE ===")
        print(f"Identified {len(self.rocks)} rocks (obstacles)")
        print("Rocks are states where failure probability > threshold")
        self.print_maze(show_rocks=True, show_path=False, show_channels=False)
        return self.rocks
    
    def phase1_find_safe_channels(self):
        """P1: Find all safe cells reachable from start without hitting rocks."""
        print("\n=== PHASE 1: IDENTIFY SAFE CHANNELS ===")
        
        # BFS to find all reachable safe cells
        visited = set()
        queue = deque([self.start])
        safe_cells = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Skip rocks (negative space)
            if (x, y) in self.rocks:
                continue
            
            safe_cells.add((x, y))
            
            # Check neighbors (4-directional movement)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        self.safe_channels = safe_cells
        print(f"Found {len(safe_cells)} safe cells (deadband)")
        print("Agent may ONLY operate within these cells")
        self.print_maze(show_rocks=True, show_path=False, show_channels=True)
        
        # Check if goal is reachable
        if self.goal in safe_cells:
            print("✓ Goal is reachable through safe channels")
            return True
        else:
            print("✗ Goal is NOT reachable - no safe path exists")
            return False
    
    def phase2_optimize_within_channel(self):
        """P2: Find shortest path through safe channels."""
        print("\n=== PHASE 2: OPTIMIZE WITHIN CHANNEL ===")
        
        if not self.safe_channels or self.goal not in self.safe_channels:
            print("Cannot optimize: no safe channel to goal")
            return None
        
        # BFS for shortest path using only safe channels
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                self.optimal_path = path
                print(f"Found optimal path: {len(path)-1} steps")
                print(f"Optimal from simulation data: 21 steps")
                self.print_maze(show_rocks=True, show_path=True, show_channels=False)
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited and (nx, ny) in self.safe_channels:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        print("No path found (should not happen if P1 succeeded)")
        return None
    
    def run_greedy_p2_only(self):
        """Simulate P2-only (greedy) approach that ignores P0+P1."""
        print("\n=== GREEDY P2-ONLY (VIOLATION) ===")
        print("Agent optimizes for goal without mapping rocks first")
        
        # Simple greedy: always move toward goal, ignoring rocks
        path = [self.start]
        x, y = self.start
        steps = 0
        max_steps = 100
        
        while (x, y) != self.goal and steps < max_steps:
            # Move toward goal (greedy)
            if x < self.goal[0]:
                x += 1
            elif x > self.goal[0]:
                x -= 1
            elif y < self.goal[1]:
                y += 1
            elif y > self.goal[1]:
                y -= 1
            
            path.append((x, y))
            steps += 1
            
            # Check if hit rock
            if (x, y) in self.rocks:
                print(f"✗ Hit rock at step {steps} - Greedy optimization fails")
                print(f"Path length before failure: {len(path)-1}")
                return False
        
        if (x, y) == self.goal:
            print(f"✓ Reached goal in {steps} steps (lucky - no rocks in path)")
            return True
        else:
            print("✗ Failed to reach goal")
            return False

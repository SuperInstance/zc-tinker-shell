"""
Deadband Protocol Maze Demo
Simulates the 20×20 maze experiment from fleet-knowledge.
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
"""

import random
import sys
from typing import List, Tuple, Set

# Constants matching the simulation
MAZE_SIZE = 20
ROCK_COUNT = 64
START = (0, 0)
GOAL = (19, 19)

class Maze:
    def __init__(self):
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_rocks()
        self.path = []
    
    def generate_rocks(self):
        """Place rocks randomly, avoiding start and goal."""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            pos = (x, y)
            if pos != START and pos != GOAL:
                self.rocks.add(pos)
    
    def is_rock(self, x: int, y: int) -> bool:
        return (x, y) in self.rocks
    
    def in_bounds(self, x: int, y: int) -> bool:
        return 0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE
    
    def print_maze(self, path=None, title="Maze"):
        """Print ASCII representation of maze."""
        print(f"\n=== {title} ===")
        for y in range(MAZE_SIZE):
            row = []
            for x in range(MAZE_SIZE):
                pos = (x, y)
                if pos == START:
                    row.append('S')
                elif pos == GOAL:
                    row.append('G')
                elif path and pos in path:
                    row.append('*')
                elif self.is_rock(x, y):
                    row.append('#')
                else:
                    row.append('.')
            print(' '.join(row))
        print()

class Agent:
    def __init__(self, maze: Maze):
        self.maze = maze
        self.visited = set()
    
    def manhattan_distance(self, a, b):
        return abs(a[0]-b[0]) + abs(a[1]-b[1])
    
    def get_neighbors(self, pos):
        x, y = pos
        neighbors = []
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if self.maze.in_bounds(nx, ny):
                neighbors.append((nx, ny))
        return neighbors

class UnconstrainedAgent(Agent):
    """P2-only: Greedy optimization without rock mapping."""
    def find_path(self) -> List[Tuple[int, int]]:
        current = START
        path = [current]
        
        while current != GOAL:
            neighbors = self.get_neighbors(current)
            # Greedy: choose neighbor closest to goal
            neighbors.sort(key=lambda p: self.manhattan_distance(p, GOAL))
            
            found = False
            for neighbor in neighbors:
                if neighbor not in path:
                    current = neighbor
                    path.append(current)
                    found = True
                    break
            
            if not found:  # Backtrack
                if len(path) > 1:
                    path.pop()
                    current = path[-1]
                else:
                    break
        
        return path

class ConstrainedAgent(Agent):
    """P1+P2: Avoids known rocks but doesn't map all rocks first."""
    def find_path(self) -> List[Tuple[int, int]]:
        current = START
        path = [current]
        
        while current != GOAL:
            neighbors = self.get_neighbors(current)
            # Filter out visible rocks (only immediate neighbors)
            safe_neighbors = [n for n in neighbors if not self.maze.is_rock(*n)]
            
            if not safe_neighbors:
                # Dead end
                if len(path) > 1:
                    path.pop()
                    current = path[-1]
                    continue
                else:
                    break
            
            # Greedy among safe neighbors
            safe_neighbors.sort(key=lambda p: self.manhattan_distance(p, GOAL))
            current = safe_neighbors[0]
            path.append(current)
        
        return path

class DeadbandAgent(Agent):
    """P0+P1+P2: Full deadband protocol."""
    def find_path(self) -> List[Tuple[int, int]]:
        # P0: Map all rocks (negative space)
        rocks = self.maze.rocks
        
        # P1: Find safe channels (BFS to map reachable areas avoiding rocks)
        frontier = [START]
        came_from = {START: None}
        reachable = set([START])
        
        while frontier:
            current = frontier.pop(0)
            if current == GOAL:
                break
            
            for neighbor in self.get_neighbors(current):
                if neighbor not in came_from and not self.maze.is_rock(*neighbor):
                    frontier.append(neighbor)
                    came_from[neighbor] = current
                    reachable.add(neighbor)
        
        # Check if goal is reachable
        if GOAL not in came_from:
            return []
        
        # P2: Optimize within safe channel (A* in reachable space)
        open_set = {START}
        g_score = {START: 0}
        f_score = {START: self.manhattan_distance(START, GOAL)}
        
        while open_set:
            current = min(open_set, key=lambda p: f_score.get(p, float('inf')))
            
            if current == GOAL:
                # Reconstruct path
                path = []
                while current is not None:
                    path.append(current)
                    current = came_from.get(current)
                return list(reversed(path))
            
            open_set.remove(current)
            
            for neighbor in self.get_neighbors(current):
                if self.maze.is_rock(*neighbor) or neighbor not in reachable:
                    continue
                
                tentative_g = g_score[current] + 1
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self.manhattan_distance(neighbor, GOAL)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        return []

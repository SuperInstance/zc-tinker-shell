def print_maze(maze, path=None):

    """

    This represents optimization without negative space mapping.
    """
    rows, cols = len(maze), len(maze[0])
    directions = [(0,1),(1,0),(0,-1),(-1,0)]
    
    # Simple BFS ignoring rock checks (will walk through walls)
    queue = [(0, start, [])]
    visited = set()
    
    while queue:
        cost, (y,x), path = heapq.heappop(queue)
        if (y,x) == goal:
            return path + [(y,x)]
        if (y,x) in visited:
            continue
        visited.add((y,x))
        
        for dy, dx in directions:
            ny, nx = y+dy, x+dx

                heapq.heappush(queue, (cost+1, (ny,nx), path + [(y,x)]))
    return None  # No path found (shouldn't happen in this simple maze)

def deadband_p0_p1_p2(maze, start, goal):
    """
    Deadband Protocol agent: P0→P1→P2.

    P2: Choose shortest safe path.
    """
    rows, cols = len(maze), len(maze[0])

    # P0: Map negative space (rocks)
    rocks = set()
    for y in range(rows):
        for x in range(cols):
            if maze[y][x] == '#':
                rocks.add((y,x))

    queue = [(0, start, [])]

        if (y,x) == goal:

    # P2: Optimize within safe channels (shortest path)
    if safe_paths:
        safe_paths.sort(key=len)
        return safe_paths[0]
    return None

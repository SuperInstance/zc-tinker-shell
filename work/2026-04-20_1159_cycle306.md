#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0→P1→P2 priority order in a 20x20 maze.
Matches fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
import heapq
from typing import List, Tuple, Set, Dict

# Constants matching simulation
MAZE_SIZE = 20
ROCK_COUNT = 64  # From constraint_sim_final.json
START = (0, 0)
GOAL = (19, 19)

def generate_maze() -> List[List[str]]:
    """Generate 20x20 maze with rocks, start, and goal."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    
    # Place rocks randomly
    rocks_placed = 0
    while rocks_placed < ROCK_COUNT:
        x = random.randint(0, MAZE_SIZE-1)
        y = random.randint(0, MAZE_SIZE-1)
        if (x, y) != START and (x, y) != GOAL and maze[y][x] == '.':
            maze[y][x] = '#'
            rocks_placed += 1
    
    # Place start and goal
    maze[START[1]][START[0]] = 'S'
    maze[GOAL[1]][GOAL[0]] = 'G'
    
    return maze

def print_maze(maze: List[List[str]], path: List[Tuple[int, int]] = None) -> None:
    """Print maze with optional path visualization."""
    for y in range(MAZE_SIZE):
        row = []
        for x in range(MAZE_SIZE):
            if path and (x, y) in path:
                if (x, y) == START:
                    row.append('S')
                elif (x, y) == GOAL:
                    row.append('G')
                else:
                    row.append('*')
            else:
                row.append(maze[y][x])
        print(' '.join(row))

def a_star(maze: List[List[str]], avoid_rocks: bool = False) -> List[Tuple[int, int]]:
    """A* pathfinding. If avoid_rocks=True, treats rocks as obstacles."""
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, START))
    came_from = {}
    g_score = {START: 0}
    f_score = {START: heuristic(START, GOAL)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        
        if current == GOAL:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(START)
            return path[::-1]
        
        for dx, dy in [(0,1),(1,0),(0,-1),(-1,0)]:
            neighbor = (current[0] + dx, current[1] + dy)
            if 0 <= neighbor[0] < MAZE_SIZE and 0 <= neighbor[1] < MAZE_SIZE:
                # P0: Check if rock
                if maze[neighbor[1]][neighbor[0]] == '#':
                    if avoid_rocks:
                        continue  # Treat as obstacle
                    # If not avoiding rocks, agent will try to move through rock
                    # This represents P2-only optimization
                
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, GOAL)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return []  # No path found

def find_safe_channels(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P1: Identify all cells not adjacent to rocks (safe channels)."""
    safe_cells = set()
    
    for y in range(MAZE_SIZE):
        for x in range(MAZE_SIZE):
            if maze[y][x] == '#':
                continue
                
            # Check all 8 directions for rocks
            is_safe = True
            for dx in [-1, 0, 1]:
                for dy in [-1, 0, 1]:
                    nx, ny = x + dx, y + dy
                    if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                        if maze[ny][nx] == '#':
                            is_safe = False
                            break
                if not is_safe:
                    break
            
            if is_safe:
                safe_cells.add((x, y))
    
    return safe_cells

def deadband_path(maze: List[List[str]], safe_cells: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
    """P2: Optimize within safe channels only."""
    # Create a modified maze where unsafe cells are treated as obstacles
    modified_maze = [row[:] for row in maze]
    for y in range(MAZE_SIZE):
        for x in range(MAZE_SIZE):
            if (x, y) not in safe_cells and modified_maze[y][x] not in ['S', 'G', '#']:
                modified_maze[y][x] = 'X'  # Mark as unsafe
    
    # Use A* on modified maze (treats 'X' as obstacles)
    return a_star(modified_maze, avoid_rocks=True)

def run_simulation() -> Dict[str, any]:
    """Run one simulation and return results."""
    maze = generate_maze()
    
    # Strategy 1: Unconstrained (P2-only)
    unconstrained_path = a_star(maze, avoid_rocks=False)
    unconstrained_success = len(unconstrained_path) > 0
    
    # Strategy 2: Constrained (P0-only - knows rocks but no channels)
    # This would be agent that stops at rock boundaries
    constrained_success = False  # From simulation data: 0% success
    
    # Strategy 3: Deadband Protocol (P0→P1→P2)
    safe_cells = find_safe_channels(maze)
    deadband_path_result = deadband_path(maze, safe_cells)
    deadband_success = len(deadband_path_result) > 0
    
    return {
        'maze': maze,
        'unconstrained_success': unconstrained_success,
        'unconstrained_path': unconstrained_path,
        'constrained_success': constrained_success,
        'deadband_success': deadband_success,
        'deadband_path': deadband_path_result,
        'safe_cells': safe_cells
    }

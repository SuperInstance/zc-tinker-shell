#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 with visual ASCII maze
Based on fleet constraint theory data
"""

import random
from collections import deque
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        """
        Initialize maze based on fleet simulation parameters:
        - 20x20 grid (from constraint_sim_final.json)
        - 64 rocks (16% density)
        - Start at (0,0), goal at (19,19)
        """
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self.rocks: Set[Tuple[int, int]] = set()
        self._generate_rocks(rock_density)
        
        # P1: Safe channels (will be computed)
        self.safe_channels: List[List[Tuple[int, int]]] = []
        
        # P2: Optimal path
        self.optimal_path: List[Tuple[int, int]] = []
        
        # Statistics
        self.stats = {
            'rocks_count': 0,
            'safe_paths_found': 0,
            'optimal_length': 0,
            'p0_complete': False,
            'p1_complete': False,
            'p2_complete': False
        }
    
    def _generate_rocks(self, density: float):
        """Generate random obstacles (rocks) avoiding start and goal"""
        total_cells = self.width * self.height
        target_rocks = int(total_cells * density)
        
        # Ensure exactly 64 rocks for 20x20 grid (16%)
        if self.width == 20 and self.height == 20:
            target_rocks = 64
        
        candidates = []
        for x in range(self.width):
            for y in range(self.height):
                if (x, y) != self.start and (x, y) != self.goal:
                    candidates.append((x, y))
        
        self.rocks = set(random.sample(candidates, target_rocks))
        self.stats['rocks_count'] = len(self.rocks)
    
    def run_deadband_protocol(self):
        """Execute full P0→P1→P2 protocol"""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===")
        print(f"Maze: {self.width}x{self.height}, Rocks: {self.stats['rocks_count']}")
        print(f"Start: {self.start}, Goal: {self.goal}")
        print()
        
        # P0: Map negative space
        print("P0: MAPPING NEGATIVE SPACE")
        print(f"  Identified {len(self.rocks)} rock positions")
        print("  Rocks marked as '█' in visualization")
        self.stats['p0_complete'] = True
        print()
        
        # P1: Find safe channels
        print("P1: FINDING SAFE CHANNELS")
        self.safe_channels = self._find_all_safe_paths()
        self.stats['safe_paths_found'] = len(self.safe_channels)
        print(f"  Found {len(self.safe_channels)} safe channels (rock-free paths)")
        self.stats['p1_complete'] = True
        print()
        
        # P2: Optimize within channels
        print("P2: OPTIMIZING WITHIN CHANNELS")
        if self.safe_channels:
            self.optimal_path = min(self.safe_channels, key=len)
            self.stats['optimal_length'] = len(self.optimal_path)
            print(f"  Selected optimal path: {self.stats['optimal_length']} steps")
            self.stats['p2_complete'] = True
        else:
            print("  WARNING: No safe channels found! Protocol failed at P1.")
        print()
        
        # Display statistics
        self._display_stats()
    
    def _find_all_safe_paths(self, max_paths: int = 1000) -> List[List[Tuple[int, int]]]:
        """Find all paths from start to goal avoiding rocks (BFS variant)"""
        safe_paths = []
        queue = deque([[self.start]])
        visited = set([self.start])
        
        while queue and len(safe_paths) < max_paths:
            path = queue.popleft()
            x, y = path[-1]
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            # Explore neighbors (4-directional movement)
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        new_path = path + [(nx, ny)]
                        queue.append(new_path)
                        visited.add((nx, ny))
        
        return safe_paths
    
    def visualize(self, show_path: bool = True):
        """Display ASCII maze with rocks and optimal path"""
        print("MAZE VISUALIZATION")
        print("Legend: S=Start, G=Goal, █=Rock, ·=Empty, ○=Path")
        print()
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')
                elif show_path and pos in self.optimal_path:
                    row.append('○')
                else:
                    row.append('·')
            print(' '.join(row))
        print()
    
    def _display_stats(self):
        """Display protocol statistics"""
        print("=== PROTOCOL STATISTICS ===")
        print(f"P0 Complete: {'✓' if self.stats['p0_complete'] else '✗'}")
        print(f"P1 Complete: {'✓' if self.stats['p1_complete'] else '✗'}")
        print(f"P2 Complete: {'✓' if self.stats['p2_complete'] else '✗'}")
        print(f"Rocks mapped: {self.stats['rocks_count']}")
        print(f"Safe channels found: {self.stats['safe_paths_found']}")
        if self.stats['optimal_length'] > 0:
            print(f"Optimal path length: {self.stats['optimal_length']} steps")
            # Compare with fleet's optimal of 21
            if self.width == 20 and self.height == 20:
                print(f"Fleet optimal: 21 steps (from constraint_sim_final.json)")
                if self.stats['optimal_length'] == 21:
                    print("  ✓ Matches fleet optimal!")
                else:
                    print(f"  Δ Difference: {self.stats['optimal_length'] - 21} steps")
        print()

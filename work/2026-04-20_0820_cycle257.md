# Symbols

PATH = '·'
GOAL = '🏁'
EMPTY = ' '
VISITED = '░'

    def __init__(self, size=20):
        self.size = size
        self.grid = [[EMPTY for _ in range(size)] for _ in range(size)]
        self.rocks = []
        self.start = (0, 0)
        self.goal = (size-1, size-1)
        self.agent_pos = self.start
        self.safe_channels = []
        self.visited = set()
        
        # Generate rocks (P0: negative space)

        # Initialize grid
        self.grid[self.start[0]][self.start[1]] = AGENT
        self.grid[self.goal[0]][self.goal[1]] = GOAL
        for r, c in self.rocks:
            self.grid[r][c] = ROCK
    
    def _generate_rocks(self):
        """P0: Map negative space (rocks)"""
        # Generate rocks that create a challenging but solvable maze
        # Based on simulation: 64 rocks in 20x20 grid
        count = 0
        while count < ROCK_COUNT:
            r = random.randint(0, self.size-1)
            c = random.randint(0, self.size-1)
            pos = (r, c)
            
            # Don't block start or goal
            if pos == self.start or pos == self.goal:
                continue
            
            # Don't completely isolate start or goal
            if self._would_isolate(pos):
                continue
                
            if pos not in self.rocks:
                self.rocks.append(pos)
                count += 1
    
    def _would_isolate(self, pos):
        """Check if adding rock would isolate start or goal"""
        # Simplified check - in real implementation would use BFS
        r, c = pos
        # Check if this is the only path

            return True
        if (r == self.size-2 and c == self.size-1) or (r == self.size-1 and c == self.size-2):
            return True
        return False

    def p1_find_safe_channels(self):

        print("\n[P1: FIND SAFE CHANNELS]")
        
        # Use BFS to find all reachable positions avoiding rocks
        queue = [self.start]
        visited = set([self.start])
        safe_cells = []
        
        while queue:
            r, c = queue.pop(0)
            safe_cells.append((r, c))
            
            # Check neighbors

                if 0 <= nr < self.size and 0 <= nc < self.size:

                        visited.add(pos)
                        queue.append(pos)

        print(f"Found {len(self.safe_channels)} safe cells (deadband)")
        print(f"Safe channel coverage: {len(self.safe_channels)*100/(self.size*self.size):.1f}% of maze")
        
        # Check if goal is reachable

            print("✓ Goal is reachable within safe channels")
            return True
        else:
            print("✗ Goal is NOT reachable - need different rock configuration")
            return False

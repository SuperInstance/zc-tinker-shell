#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

A visual ASCII simulation of P0→P1→P2 in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# Simulation constants from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64          # 64 rocks in simulation
OPTIMAL_PATH_LEN = 21    # optimal path length

# Symbols for visualization
AGENT = 'A'
GOAL = 'G'
ROCK = '█'
SAFE = '·'
PATH = '○'
DEADBAND = '░'  # safe channel
EMPTY = ' '

class DeadbandMaze:
    """Maze environment with Deadband Protocol implementation."""
    
    def __init__(self, seed=42):
        random.seed(seed)
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        
        # Initialize
        self._place_rocks()
        self.grid[self.start[0]][self.start[1]] = AGENT
        self.grid[self.goal[0]][self.goal[1]] = GOAL
    
    def _place_rocks(self):
        """Place rocks randomly (P0: negative space)."""
        placed = 0
        while placed < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            pos = (x, y)
            
            # Don't place on start or goal
            if pos == self.start or pos == self.goal:
                continue
            
            if pos not in self.rocks:
                self.rocks.add(pos)
                self.grid[x][y] = ROCK
                placed += 1
    
    def display(self, title="", highlight=None):
        """Display the maze with optional highlighted positions."""
        print(f"\n{'='*60}")
        print(f"{title}")
        print(f"{'='*60}")
        
        for x in range(MAZE_SIZE):
            row = []
            for y in range(MAZE_SIZE):
                if highlight and (x, y) in highlight:
                    row.append(DEADBAND)
                else:
                    row.append(self.grid[x][y])
            print(''.join(row))
        print()
    
    def p0_map_negative_space(self):
        """P0: Identify all rocks (what NOT to do)."""
        print("P0: MAPPING NEGATIVE SPACE")
        print(f"Identified {len(self.rocks)} rocks (failure states)")
        return self.rocks.copy()
    
    def p1_find_safe_channels(self, rocks: Set[Tuple[int, int]]):
        """P1: Find safe channels (where agent CAN move)."""
        print("P1: FINDING SAFE CHANNELS")
        
        # Simple flood fill to find reachable safe cells
        safe_cells = set()
        queue = [self.start]
        visited = set()
        
        # Valid moves (up, down, left, right)
        moves = [(-1, 0), (1, 0), (0, -1), (0, 1)]
        
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # If not a rock and within bounds, it's safe
            if (x, y) not in rocks and 0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE:
                safe_cells.add((x, y))
                
                # Explore neighbors
                for dx, dy in moves:
                    nx, ny = x + dx, y + dy
                    if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                        queue.append((nx, ny))
        
        # Remove start and goal for cleaner display
        display_safe = safe_cells - {self.start, self.goal}
        print(f"Found {len(safe_cells)} safe cells in reachable channels")
        
        return safe_cells, display_safe
    
    def p2_optimize_within_channel(self, safe_cells: Set[Tuple[int, int]]):
        """P2: Find optimal path through safe channels."""
        print("P2: OPTIMIZING WITHIN SAFE CHANNELS")
        
        # A* search constrained to safe cells
        open_set = {self.start}
        came_from = {}
        
        g_score = {self.start: 0}
        f_score = {self.start: self._heuristic(self.start, self.goal)}
        
        moves = [(-1, 0), (1, 0), (0, -1), (0, 1)]
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score.get(pos, float('inf')))
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.reverse()
                return path
            
            open_set.remove(current)
            
            x, y = current
            for dx, dy in moves:
                neighbor = (x + dx, y + dy)
                
                # Constraint: must be in safe channel
                if neighbor not in safe_cells:
                    continue
                
                tentative_g = g_score[current] + 1
                
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor, self.goal)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        return []  # No path found
    
    def _heuristic(self, a: Tuple[int, int], b: Tuple[int, int]) -> int:
        """Manhattan distance heuristic."""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def run_deadband_protocol(self):
        """Execute full P0→P1→P2 protocol."""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*60)
        
        # Step 0: Show initial maze
        self.display("INITIAL MAZE (A=Agent, G=Goal, █=Rock)")
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        
        # P1: Find safe channels
        safe_cells, display_safe = self.p1_find_safe_channels(rocks)
        self.display("SAFE CHANNELS (░=Safe reachable cells)", display_safe)
        
        # P2: Optimize within channel
        path = self.p2_optimize_within_channel(safe_cells)
        
        if path:
            print(f"✓ Found optimal path of length {len(path)}")
            print(f"  (Theoretical optimal: {OPTIMAL_PATH_LEN})")
            
            # Show final path
            path_set = set(path)
            for x, y in path:
                if (x, y) not in {self.start, self.goal}:
                    self.grid[x][y] = PATH
            
            self.display("FINAL PATH (○=Optimal path through safe channels)")
            
            # Protocol validation
            print("\nPROTOCOL VALIDATION:")
            print(f"  P0: Mapped {len(rocks)} rocks ✓")
            print(f"  P1: Found {len(safe_cells)} safe cells ✓")
            print(f"  P2: Path length {len(path)} (optimal: {OPTIMAL_PATH_LEN}) ✓")
            
            # Check for protocol violations
            path_in_safe = all(pos in safe_cells for pos in path)
            path_avoids_rocks = all(pos not in rocks for pos in path)
            
            if path_in_safe and path_avoids_rocks:
                print("  ✓ No protocol violations - all constraints respected")
            else:
                print("  ⚠ Protocol violation detected!")
            
            return True
        else:
            print("✗ No path found through safe channels")
            return False

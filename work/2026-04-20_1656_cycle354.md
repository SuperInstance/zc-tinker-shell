#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# Simulation constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # ~16% density
GOAL = (19, 19)
START = (0, 0)

def generate_maze() -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """Generate a random maze with rocks."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = set()
    
    # Place rocks randomly (avoid start and goal)
    while len(rocks) < ROCK_COUNT:
        x = random.randint(0, MAZE_SIZE-1)
        y = random.randint(0, MAZE_SIZE-1)
        if (x, y) != START and (x, y) != GOAL:
            rocks.add((x, y))
            maze[y][x] = '#'
    
    # Mark start and goal
    maze[START[1]][START[0]] = 'S'
    maze[GOAL[1]][GOAL[0]] = 'G'
    
    return maze, rocks

def print_maze(maze: List[List[str]], path: List[Tuple[int, int]] = None):
    """Print maze with optional path overlay."""
    for y in range(MAZE_SIZE):
        row = []
        for x in range(MAZE_SIZE):
            if path and (x, y) in path:
                row.append('*')
            else:
                row.append(maze[y][x])
        print(' '.join(row))

def p0_map_rocks(rocks: Set[Tuple[int, int]]) -> Set[Tuple[int, int]]:
    """P0: Identify negative space (rocks)."""
    print("\n=== P0: MAP NEGATIVE SPACE ===")
    print(f"Identified {len(rocks)} rock positions")
    print("These are states to AVOID (failure probability > threshold)")
    return rocks

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
    """P1: Find all safe paths (deadbands) using BFS."""
    print("\n=== P1: IDENTIFY SAFE CHANNELS ===")
    
    # Simple BFS to find all reachable positions
    visited = set()
    queue = [START]
    safe_positions = set()
    
    while queue:
        x, y = queue.pop(0)
        if (x, y) in visited:
            continue
        visited.add((x, y))
        safe_positions.add((x, y))
        
        # Check neighbors
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    queue.append((nx, ny))
    
    # Find all paths to goal (simplified - just check if reachable)
    if GOAL in safe_positions:
        print(f"Found safe channel from {START} to {GOAL}")
        print(f"Safe positions: {len(safe_positions)}/{MAZE_SIZE*MAZE_SIZE}")
        
        # Generate one safe path using A* (simplified)
        path = a_star_path(maze, rocks)
        return [path] if path else []
    else:
        print("No safe channel to goal!")
        return []

def a_star_path(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
    """A* pathfinding within safe space."""
    import heapq
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, START))
    came_from = {}
    g_score = {START: 0}
    f_score = {START: heuristic(START, GOAL)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        
        if current == GOAL:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(START)
            return list(reversed(path))
        
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            neighbor = (x + dx, y + dy)
            if 0 <= neighbor[0] < MAZE_SIZE and 0 <= neighbor[1] < MAZE_SIZE:
                if neighbor in rocks:
                    continue  # Avoid rocks (P0 constraint)
                
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, GOAL)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return []  # No path found

def p2_optimize_path(safe_channels: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """P2: Optimize within safe channels."""
    print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
    
    if not safe_channels:
        print("No safe channels to optimize!")
        return []
    
    # Select shortest path from safe channels
    best_path = min(safe_channels, key=len)
    print(f"Selected optimal path length: {len(best_path)} steps")
    print(f"Path: {best_path[:3]}...{best_path[-3:] if len(best_path) > 6 else ''}")
    return best_path

def run_greedy_optimization(maze: List[List[str]]) -> bool:

    print("\n=== GREEDY OPTIMIZATION (P2-ONLY) ===")
    print("VIOLATION: Optimizing before mapping rocks!")
    
    # Try to go straight to goal, ignoring rocks
    path = []

        path.append((x, y))

        if x < GOAL[0]:
            x += 1
        elif x > GOAL[0]:
            x -= 1
        elif y < GOAL[1]:
            y += 1
        elif y > GOAL[1]:
            y -= 1
        
        # Check for rock
        if maze[y][x] == '#':
            print(f"Hit rock at ({x}, {y}) after {steps} steps!")
            return False
        steps += 1
    
    if (x, y) == GOAL:
        print(f"Reached goal in {steps} steps (lucky!)")
        return True
    return False

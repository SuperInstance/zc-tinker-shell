## Deadband Protocol Visual Demo (ASCII Maze)

**File:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    """
    A simple 10x10 maze simulation of the Deadband Protocol.
    'G' = goal, 'A' = agent, '#' = rock, '.' = open, '*' = safe channel, 'o' = optimal path.
    """
    
    def __init__(self, seed=42):
        random.seed(seed)
        self.width = 10
        self.height = 10
        self.grid = [['.' for _ in range(self.width)] for _ in range(self.height)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (9, 9)
        self._generate_maze()
        
    def _generate_maze(self):
        """Place rocks (~20% of cells) and ensure start/goal are clear."""
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) in (self.start, self.goal):
                    continue
                if random.random() < 0.2:
                    self.grid[y][x] = '#'
                    self.rocks.add((x, y))
        self.grid[self.start[1]][self.start[0]] = 'A'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def p0_map_rocks(self) -> Set[Tuple[int, int]]:
        """P0: Identify negative space (rocks)."""
        print("=== P0: MAP ROCKS (Negative Space) ===")
        print(f"Found {len(self.rocks)} rock cells.")
        self._print_grid()
        return self.rocks
    
    def p1_find_safe_channels(self, rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
        """P1: Find all paths from start to goal avoiding rocks (simple BFS)."""
        print("\n=== P1: FIND SAFE CHANNELS ===")
        
        # BFS to find all shortest safe paths
        from collections import deque
        queue = deque([(self.start, [self.start])])
        visited = {self.start}
        safe_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        # Mark safe channels on grid
        channel_cells = set()
        for path in safe_paths:
            for (x, y) in path:
                if (x, y) not in (self.start, self.goal):
                    channel_cells.add((x, y))
        
        # Create visualization
        viz = [row[:] for row in self.grid]
        for (x, y) in channel_cells:
            viz[y][x] = '*'
        
        print(f"Found {len(safe_paths)} safe channel(s).")
        print("Safe channels marked with '*':")
        for row in viz:
            print(' '.join(row))
        
        return safe_paths
    
    def p2_optimize_in_channel(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """P2: Select shortest path within safe channels."""
        if not safe_paths:
            print("No safe channels found. Cannot optimize.")
            return []
        
        optimal_path = min(safe_paths, key=len)
        print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
        print(f"Selected optimal path length: {len(optimal_path)-1} steps")
        
        # Create final visualization
        viz = [row[:] for row in self.grid]
        for (x, y) in optimal_path[1:-1]:  # Exclude start and goal
            viz[y][x] = 'o'
        
        print("Optimal path marked with 'o':")
        for row in viz:
            print(' '.join(row))
        
        return optimal_path
    
    def run_protocol(self):
        """Execute full P0→P1→P2 sequence."""
        print("DEADBAND PROTOCOL DEMO - ASCII MAZE")
        print("=" * 40)
        
        # P0: Map rocks
        rocks = self.p0_map_rocks()
        
        # P1: Find safe channels
        safe_paths = self.p1_find_safe_channels(rocks)
        
        # P2: Optimize within channel
        optimal_path = self.p2_optimize_in_channel(safe_paths)
        
        # Summary
        print("\n" + "=" * 40)
        print("PROTOCOL COMPLETE")
        if optimal_path:
            print(f"✓ Successfully navigated from {self.start} to {self.goal}")
            print(f"✓ Path avoids all {len(rocks)} rocks")
            print(f"✓ Optimal steps: {len(optimal_path)-1}")
        else:
            print("✗ No safe path exists")
        
        return optimal_path is not None
    
    def _print_grid(self):
        """Print current grid state."""
        for row in self.grid:
            print(' '.join(row))

def main():
    """Run the demo with three different seeds to show variability."""
    print("Deadband Protocol Visual Demo")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print("Optimal: 21 steps, Rocks: 64, Deadband success: 50/50")
    print("\n" + "=" * 60)
    
    successes = 0
    for i, seed in enumerate([42, 123, 777], 1):
        print(f"\n\nMAZE SIMULATION #{i} (Seed: {seed})")
        print("-" * 40)
        maze = DeadbandMaze(seed=seed)
        if maze.run_protocol():
            successes += 1
    
    print("\n" + "=" * 60)
    print(f"SUMMARY: {successes}/3 mazes successfully navigated")
    print("\nProtocol demonstrates:")
    print("1. P0: First identify rocks (failure states)")
    print("2. P1: Then find ALL safe channels")
    print("3. P2: Finally optimize within safe channels")
    print("\nViolation (skipping to P2 first) → 0% success in constraint_sim_final.json")

if __name__ == "__main__":
    main()
```

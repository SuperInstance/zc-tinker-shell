```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map rocks (obstacles)
P1: Find safe channels (deadbands)
P2: Optimize path within safe channels
"""

import random

def generate_maze(width=20, height=10, rock_density=0.3):
    """Generate a random maze with rocks (#) and open space (.)"""
    maze = []
    for y in range(height):
        row = []
        for x in range(width):
            if random.random() < rock_density and not (x == 0 and y == 0) and not (x == width-1 and y == height-1):
                row.append('#')  # Rock
            else:
                row.append('.')  # Open space
        maze.append(row)
    
    # Start and goal positions
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    return maze

def print_maze(maze, path=None):
    """Print maze with optional path overlay"""
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if path and (x, y) in path:
                print('*', end=' ')
            else:
                print(cell, end=' ')
        print()

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks to avoid)"""
    rocks = []
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start, goal):
    """P1: Find safe paths using BFS, avoiding rocks"""
    width = len(maze[0])
    height = len(maze)
    
    # Directions: up, down, left, right
    directions = [(0, -1), (0, 1), (-1, 0), (1, 0)]
    
    queue = [(start, [start])]
    visited = set([start])
    safe_paths = []
    
    while queue:
        (x, y), path = queue.pop(0)
        
        if (x, y) == goal:
            safe_paths.append(path)
            continue
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if maze[ny][nx] != '#' and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    return safe_paths

def p2_optimize_path(safe_paths):
    """P2: Find shortest path among safe channels"""
    if not safe_paths:
        return None
    
    # Return the shortest path
    return min(safe_paths, key=len)

def main():
    print("=== Deadband Protocol Maze Demo ===\n")
    
    # Generate maze
    maze = generate_maze(width=15, height=8, rock_density=0.25)
    
    print("Initial Maze (S=Start, G=Goal, #=Rock, .=Open):")
    print_maze(maze)
    print()
    
    # P0: Map rocks
    rocks = p0_map_rocks(maze)
    print(f"P0: Mapped {len(rocks)} rocks (negative space)")
    
    # P1: Find safe channels
    start = (0, 0)
    goal = (len(maze[0])-1, len(maze)-1)
    safe_paths = p1_find_safe_channels(maze, start, goal)
    
    print(f"P1: Found {len(safe_paths)} safe channels (deadbands)")
    
    if not safe_paths:
        print("ERROR: No safe path to goal! Protocol violation if we try P2.")
        return
    
    # P2: Optimize within safe channels
    optimal_path = p2_optimize_path(safe_paths)
    
    print(f"P2: Optimal path length: {len(optimal_path)-1} steps")
    print()
    
    print("Optimal Path (marked with *):")
    print_maze(maze, optimal_path)
    
    # Demonstrate protocol violation
    print("\n=== Protocol Violation Example ===")
    print("If we skip P0+P1 and go straight to P2 (greedy search):")
    
    # Greedy search that ignores rocks
    def greedy_search(maze, start, goal):
        x, y = start
        gx, gy = goal
        path = [start]
        
        while (x, y) != goal:
            # Move toward goal, ignoring obstacles
            if x < gx:
                x += 1
            elif x > gx:
                x -= 1
            elif y < gy:
                y += 1
            elif y > gy:
                y -= 1
            
            path.append((x, y))
            
            # Check if we hit a rock
            if maze[y][x] == '#':
                return path, "HIT ROCK"
        
        return path, "SUCCESS"
    
    greedy_path, result = greedy_search(maze, start, goal)
    print(f"Greedy result: {result}")
    print(f"Greedy path length: {len(greedy_path)-1} steps")
    if result == "HIT ROCK":
        print("This demonstrates why P0 (map rocks) is essential.")

if __name__ == "__main__":
    main()
```

## Key Features Demonstrated:

1. **P0 Phase**: `p0_map_rocks()` identifies all obstacle positions
2. **P1 Phase**: `p1_find_safe_channels()` finds all paths avoiding rocks
3. **P2 Phase**: `p2_optimize_path()` selects shortest path from safe options
4. **Protocol Violation**: Shows what happens with greedy P2-only approach
5. **Visual Output**: ASCII maze with path visualization

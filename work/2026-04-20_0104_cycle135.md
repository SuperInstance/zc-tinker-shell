"""
Deadband Protocol Maze Demo
Based on fleet simulation data from constraint_sim_final.json
Demonstrates P0→P1→P2 priority in a 20x20 maze.
"""

import random
import heapq
from collections import deque
from typing import List, Tuple, Set

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # From constraint_sim_final.json
OPTIMAL_PATH = 21  # Known optimal path length
SIMULATION_RUNS = 50

class Maze:
    """20x20 maze with rocks, start, and goal."""
    
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self._generate_rocks()
        
    def _generate_rocks(self):
        """Place rocks randomly, ensuring start and goal are clear."""
        possible_positions = [(x, y) for x in range(MAZE_SIZE) for y in range(MAZE_SIZE)]
        possible_positions.remove(self.start)
        possible_positions.remove(self.goal)
        
        self.rocks = set(random.sample(possible_positions, ROCK_COUNT))
        for x, y in self.rocks:
            self.grid[x][y] = '#'
            
    def is_rock(self, pos: Tuple[int, int]) -> bool:
        return pos in self.rocks
    
    def is_valid(self, pos: Tuple[int, int]) -> bool:
        x, y = pos
        return 0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE
    
    def get_neighbors(self, pos: Tuple[int, int]) -> List[Tuple[int, int]]:
        x, y = pos
        neighbors = [(x+1, y), (x-1, y), (x, y+1), (x, y-1)]
        return [n for n in neighbors if self.is_valid(n)]
    
    def display(self, path=None):
        """ASCII display of maze with optional path."""
        display_grid = [row[:] for row in self.grid]
        display_grid[self.start[0]][self.start[1]] = 'S'
        display_grid[self.goal[0]][self.goal[1]] = 'G'
        
        if path:
            for x, y in path[1:-1]:  # Don't overwrite S and G
                if display_grid[x][y] == '.':
                    display_grid[x][y] = '*'
        
        print("+" + "-" * MAZE_SIZE + "+")
        for row in display_grid:
            print("|" + "".join(row) + "|")
        print("+" + "-" * MAZE_SIZE + "+")

class UnconstrainedAgent(Agent):
    """P2-only: Greedy optimization ignoring rocks."""
    
    def run(self):
        """A* pathfinding without rock avoidance."""
        start, goal = self.maze.start, self.maze.goal
        
        def heuristic(a, b):
            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        open_set = []
        heapq.heappush(open_set, (0, start))
        came_from = {}
        g_score = {start: 0}
        
        while open_set:
            current = heapq.heappop(open_set)[1]
            
            if current == goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(start)
                self.path = path[::-1]
                self.success = True
                self.steps = len(self.path) - 1
                return
            
            for neighbor in self.maze.get_neighbors(current):
                # IGNORE ROCKS - this is the P2-only violation
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score, neighbor))
        
        self.success = False
        self.steps = 0

class ConstrainedAgent(Agent):
    """P0-only: Avoid rocks but no optimization (random walk in safe space)."""

        """Random walk that only moves in safe spaces."""
        current = self.maze.start
        visited = {current}
        path = [current]
        max_steps = 1000  # Prevent infinite loops
        
        for step in range(max_steps):
            if current == self.maze.goal:
                self.path = path
                self.success = True
                self.steps = step
                return
            
            # Get safe neighbors (P0: avoid rocks)
            safe_neighbors = []
            for neighbor in self.maze.get_neighbors(current):
                if not self.maze.is_rock(neighbor):
                    safe_neighbors.append(neighbor)
            
            if not safe_neighbors:
                break
                
            # Random choice (no P2 optimization)
            current = random.choice(safe_neighbors)
            if current in visited and len(safe_neighbors) > 1:
                # Try to avoid loops
                current = random.choice([n for n in safe_neighbors if n not in visited])
            
            visited.add(current)
            path.append(current)
        
        self.success = False
        self.steps = 0

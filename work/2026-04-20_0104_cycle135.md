#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Simulates P0→P1→P2 protocol in a 20x20 maze.
"""

import random
from typing import List, Tuple, Set

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # From "rocks": 64
OPTIMAL_PATH_LENGTH = 21  # From "optimal": 21

class DeadbandMaze:
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
    def generate_rocks(self):
        """P0: Map negative space (what NOT to do)"""
        # Generate random obstacles (rocks)
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            # Don't block start or goal
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[x][y] = '█'
    
    def find_safe_channels(self):
        """P1: Find safe channels (where you CAN be)"""
        # Simple BFS to find all safe paths
        visited = set()
        queue = [(self.start, [self.start])]
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.goal:
                self.safe_channels.append(path)
                continue
            
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            
            # Check all 4 directions
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                    (nx, ny) not in self.rocks and 
                    (nx, ny) not in visited):
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    def optimize_within_channel(self):
        """P2: Optimize within channels (best path)"""
        if not self.safe_channels:
            return []
        
        # Find shortest safe path
        self.optimal_path = min(self.safe_channels, key=len)
        return self.optimal_path
    
    def run_protocol(self) -> bool:
        """Execute full P0→P1→P2 protocol"""
        print("=== DEADBAND PROTOCOL DEMO ===")
        print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}")
        print(f"Start: {self.start}, Goal: {self.goal}")
        print()
        
        # P0: Map negative space
        print("P0: MAP NEGATIVE SPACE")
        self.generate_rocks()
        print(f"  • Identified {len(self.rocks)} rocks (failure states)")
        print()
        
        # P1: Find safe channels
        print("P1: IDENTIFY SAFE CHANNELS")
        self.find_safe_channels()
        print(f"  • Found {len(self.safe_channels)} safe channels")
        if self.safe_channels:
            print(f"  • Channel lengths: {[len(c) for c in self.safe_channels[:3]]}...")
        print()
        
        # P2: Optimize within channel
        print("P2: OPTIMIZE WITHIN CHANNEL")
        self.optimize_within_channel()
        if self.optimal_path:
            print(f"  • Optimal path length: {len(self.optimal_path)}")
            print(f"  • Theoretical optimal: {OPTIMAL_PATH_LENGTH}")
            print(f"  • Success: {len(self.optimal_path) >= OPTIMAL_PATH_LENGTH}")
        else:
            print("  • No safe path found (all channels blocked)")
        print()
        
        return len(self.optimal_path) > 0
    
    def visualize(self, show_path=True):
        """Create ASCII visualization"""
        vis_grid = [row[:] for row in self.grid]
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        vis_grid[sx][sy] = 'S'
        vis_grid[gx][gy] = 'G'
        
        # Mark optimal path
        if show_path and self.optimal_path:
            for x, y in self.optimal_path[1:-1]:  # Skip start and goal
                if vis_grid[x][y] == '.':
                    vis_grid[x][y] = '·'
        
        print("MAZE VISUALIZATION:")
        print("  S = Start, G = Goal, █ = Rock, · = Safe Path")
        print()
        for i in range(MAZE_SIZE):
            row = '  ' + ''.join(vis_grid[i])
            print(row)
        print()
    
    def print_stats(self):
        """Print simulation statistics matching fleet data format"""
        print("=== SIMULATION STATISTICS ===")
        print(f"Optimal theoretical path: {OPTIMAL_PATH_LENGTH}")
        print(f"Rocks placed: {len(self.rocks)}")
        print(f"Safe channels found: {len(self.safe_channels)}")
        
        if self.optimal_path:
            success = len(self.optimal_path) >= OPTIMAL_PATH_LENGTH
            print(f"Deadband success: {'YES' if success else 'NO'}")
            print(f"Deadband path length: {len(self.optimal_path)}")
        else:
            print("Deadband success: NO (no path)")
            print("Deadband path length: N/A")
        
        # Compare with fleet simulation results
        print("\nFleet Reference Data (from constraint_sim_final.json):")
        print("  unconstrained_success: 33% (hits rocks)")
        print("  constrained_success: 0% (too restrictive)")
        print("  deadband_success: 50% (balanced approach)")
        print("  deadband_avg: 21.0 (optimal path length)")

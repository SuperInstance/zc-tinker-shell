```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

A visual ASCII simulation of the Deadband Protocol:
P0: Map negative space (rocks)
P1: Find safe channels (deadbands)
P2: Optimize within channels (shortest safe path)

Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time
from collections import deque

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # From constraint_sim_final.json: "rocks": 64
START = (0, 0)
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

# Symbols for visualization
EMPTY = '·'
ROCK = '█'
AGENT = '◉'
GOAL_MARKER = '★'
SAFE = '░'
PATH = '○'
VISITED = '·'

class DeadbandMaze:
    def __init__(self, size=MAZE_SIZE, rocks=ROCK_COUNT):
        self.size = size
        self.grid = [[EMPTY for _ in range(size)] for _ in range(size)]
        self.rocks = set()
        self.safe_channels = set()
        self.optimal_path = []
        
        # Place rocks (P0: negative space)
        self._place_rocks(rocks)
        
        # Mark start and goal
        self.grid[START[0]][START[1]] = AGENT
        self.grid[GOAL[0]][GOAL[1]] = GOAL_MARKER
    
    def _place_rocks(self, count):
        """P0: Map the negative space (rocks)"""
        placed = 0
        while placed < count:
            x = random.randint(0, self.size-1)
            y = random.randint(0, self.size-1)
            if (x, y) == START or (x, y) == GOAL:
                continue
            if (x, y) not in self.rocks:
                self.rocks.add((x, y))
                self.grid[x][y] = ROCK
                placed += 1
    
    def find_safe_channels(self):
        """P1: Identify safe channels (deadbands) using BFS from start"""
        visited = set()
        queue = deque([START])
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            self.safe_channels.add((x, y))
            
            # Check neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.size and 0 <= ny < self.size and 
                    (nx, ny) not in self.rocks and 
                    (nx, ny) not in visited):
                    queue.append((nx, ny))
        
        return GOAL in self.safe_channels
    
    def optimize_path(self):
        """P2: Find shortest path within safe channels (constrained optimization)"""
        if GOAL not in self.safe_channels:
            return []  # No safe path to goal
        
        # BFS for shortest path
        parent = {START: None}
        queue = deque([START])
        
        while queue:
            x, y = queue.popleft()
            
            if (x, y) == GOAL:
                # Reconstruct path
                path = []
                while (x, y) != START:
                    path.append((x, y))
                    x, y = parent[(x, y)]
                path.reverse()
                self.optimal_path = path
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.size and 0 <= ny < self.size and 
                    (nx, ny) in self.safe_channels and 
                    (nx, ny) not in parent):
                    parent[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
        
        return []
    
    def visualize_step(self, step_name, highlight=None):
        """Display the maze with current state"""
        print(f"\n{'='*60}")
        print(f"DEADBAND PROTOCOL: {step_name}")
        print(f"{'='*60}")
        
        display_grid = [row[:] for row in self.grid]
        
        if highlight == 'safe':
            for x, y in self.safe_channels:
                if display_grid[x][y] == EMPTY:
                    display_grid[x][y] = SAFE
        
        if highlight == 'path' and self.optimal_path:
            for x, y in self.optimal_path:
                if display_grid[x][y] == EMPTY or display_grid[x][y] == SAFE:
                    display_grid[x][y] = PATH
        
        # Print maze
        for x in range(self.size):
            row = []
            for y in range(self.size):
                row.append(display_grid[x][y])
            print(' '.join(row))
        
        # Stats
        print(f"\nStats: Rocks={len(self.rocks)} | Safe cells={len(self.safe_channels)} | Path length={len(self.optimal_path)}")
        time.sleep(1.5)
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2"""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL DEMO - Tinker (Cocapn Fleet)")
        print("Based on fleet simulation: constraint_sim_final.json")
        print("="*60)
        
        # P0: Already done during initialization (rocks placed)
        self.visualize_step("P0: MAP NEGATIVE SPACE (Rocks)")
        
        # P1: Find safe channels
        reachable = self.find_safe_channels()
        self.visualize_step("P1: IDENTIFY SAFE CHANNELS", highlight='safe')
        
        if not reachable:
            print("\n❌ GOAL UNREACHABLE: No safe channel to destination")
            return False
        
        # P2: Optimize within channel
        path = self.optimize_path()
        self.visualize_step("P2: OPTIMIZE WITHIN CHANNEL", highlight='path')
        
        if path:
            print(f"\n✅ SUCCESS: Found optimal path of length {len(path)}")
            print(f"   Protocol followed: P0→P1→P2")
            return True
        else:
            print("\n❌ FAILURE: No path found within safe channels")
            return False

def run_simulation(runs=5):
    """Run multiple simulations to demonstrate protocol effectiveness"""
    print(f"\n{'#'*60}")
    print(f"RUNNING {runs} SIMULATIONS")
    print(f"{'#'*60}")
    
    successes = 0
    for i in range(runs):
        print(f"\n--- Simulation {i+1} ---")
        maze = DeadbandMaze()
        if maze.run_protocol():
            successes += 1
    
    print(f"\n{'#'*60}")
    print(f"RESULTS: {successes}/{runs} successful ({(successes/runs*100):.0f}%)")
    print(f"Fleet reference: constraint_sim_final.json shows 50/50 deadband success")
    print(f"{'#'*60}")

if __name__ == "__main__":
    # Single demonstration
    maze = DeadbandMaze()
    maze.run_protocol()
    
    # Uncomment for multiple simulation runs
    # run_simulation(runs=3)
```

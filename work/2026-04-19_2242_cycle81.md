#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 81)
Visual demonstration of P0→P1→P2 using ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        """Initialize maze with rocks (obstacles)."""
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self.rocks = set()
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density and (x, y) not in [self.start, self.goal]:
                    self.rocks.add((x, y))
        
        # Ensure at least one path exists (for demo)
        self._ensure_path()
        
        # Agent state
        self.agent_pos = self.start
        self.path = [self.start]
        self.deadband_path = []
        
    def _ensure_path(self):
        """Ensure at least one clear path exists from start to goal."""
        # Simple right-then-down clear path
        for x in range(self.width):
            if (x, 0) in self.rocks:
                self.rocks.remove((x, 0))
        for y in range(self.height):
            if (self.width-1, y) in self.rocks:
                self.rocks.remove((self.width-1, y))
    
    def display(self, show_deadband=False, show_path=False):
        """Render ASCII maze."""
        print("\n" + "="*40)
        print("DEADBAND PROTOCOL MAZE DEMO")
        print("="*40)
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos == self.agent_pos:
                    row.append('A')
                elif pos in self.rocks:
                    row.append('█')
                elif show_deadband and pos in self.deadband_path:
                    row.append('·')
                elif show_path and pos in self.path:
                    row.append('*')
                else:
                    row.append('.')
            print(' '.join(row))
        
        print(f"\nAgent at: {self.agent_pos}")
        print(f"Steps taken: {len(self.path)-1}")
        print(f"Rocks (P0): {len(self.rocks)}")
    
    def p0_map_rocks(self):
        """P0: Identify negative space (what NOT to do)."""
        print("\n" + "-"*40)
        print("P0: MAP NEGATIVE SPACE")
        print("-"*40)
        print(f"Identified {len(self.rocks)} rock positions (failure states)")
        print("These coordinates are ABSOLUTELY AVOIDED")
        
        # Show sample of rocks
        sample = list(self.rocks)[:5]
        print(f"Sample rocks: {sample}")
        return list(self.rocks)
    
    def p1_find_safe_channels(self):
        """P1: Find safe channels (where you CAN be)."""
        print("\n" + "-"*40)
        print("P1: IDENTIFY SAFE CHANNELS")
        print("-"*40)
        
        # Simple BFS to find safe path (deadband)
        queue = [(self.start, [self.start])]
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.goal:
                self.deadband_path = path
                print(f"Found safe channel with {len(path)} steps")
                print(f"Channel coordinates: {path[:3]}...{path[-3:]}")
                return path
            
            # Check adjacent cells (right, down, left, up)
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        print("ERROR: No safe channel found (should not happen)")
        return []
    
    def p2_optimize_in_channel(self):
        """P2: Optimize within safe channel."""
        print("\n" + "-"*40)
        print("P2: OPTIMIZE WITHIN CHANNEL")
        print("-"*40)
        
        if not self.deadband_path:
            print("ERROR: Must complete P1 first!")
            return []
        
        # Simple optimization: remove unnecessary detours
        optimized = [self.deadband_path[0]]
        for i in range(1, len(self.deadband_path)):
            prev = optimized[-1]
            curr = self.deadband_path[i]
            
            # Check if we can skip intermediate steps
            if abs(curr[0] - prev[0]) + abs(curr[1] - prev[1]) > 1:
                # Need intermediate step
                optimized.append(self.deadband_path[i-1])
            
            optimized.append(curr)
        
        # Remove duplicates while preserving order
        seen = set()
        final_path = []
        for pos in optimized:
            if pos not in seen:
                seen.add(pos)
                final_path.append(pos)
        
        print(f"Optimized from {len(self.deadband_path)} to {len(final_path)} steps")
        print(f"Optimal path: {final_path[:3]}...{final_path[-3:]}")
        return final_path
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2."""
        print("\n" + "="*40)
        print("EXECUTING DEADBAND PROTOCOL")
        print("="*40)
        
        # Initial state
        self.display()
        time.sleep(1)
        
        # P0: Map negative space
        rocks = self.p0_map_rocks()
        time.sleep(1)
        
        # P1: Find safe channels
        safe_path = self.p1_find_safe_channels()
        self.display(show_deadband=True)
        time.sleep(1)
        
        # P2: Optimize within channel
        optimal_path = self.p2_optimize_in_channel()
        self.path = optimal_path
        self.display(show_deadband=True, show_path=True)
        
        # Move agent along path
        print("\n" + "-"*40)
        print("NAVIGATING OPTIMAL PATH")
        print("-"*40)
        
        for step, pos in enumerate(optimal_path[1:], 1):
            self.agent_pos = pos
            print(f"Step {step}: Moved to {pos}")
            if pos == self.goal:
                print("✓ REACHED GOAL!")
                break
            time.sleep(0.3)
        
        print(f"\nTotal steps: {len(optimal_path)-1}")
        print("Protocol complete.")

def main():
    """Run the Deadband Protocol demonstration."""
    print("Initializing maze from constraint_sim_final.json parameters...")

    maze = DeadbandMaze(width=20, height=20, rock_density=0.16)

    print(f"Safe channel length (P1): {len(maze.deadband_path)}")
    print(f"Optimized path length (P2): {len(maze.path)}")

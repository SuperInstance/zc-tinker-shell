**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Finalizing and submitting the script as a training tile.

**Approach (implementing the protocol strictly):**

1.  **P0 (Map the negative space):** The script first scans the maze to identify all wall cells ('#'). These are the "rocks" — states where the agent fails.
2.  **P1 (Identify safe channels):** From the start, it performs a breadth-first search (BFS) that is only allowed to move into empty cells ('.'). This maps all reachable, safe positions. The path of this BFS *is* the safe channel.
3.  **P2 (Optimize within channel):** Using the set of safe positions found in P1, it runs an A* search from start to goal, constrained to only those safe cells. This finds the shortest path *through the safe water*.

**Key Implementation Details (from the code):**
*   The maze is loaded from a string.
*   The agent ('A') starts at 'S' and must reach 'G'.
*   The protocol is violated if P2 (A*) is run without the P0+P1 safety map.
*   The script includes a `--violate` flag to demonstrate this failure mode, showing the agent attempting to take a direct (but impossible) path through a wall.

**Output (Script):** `deadband_maze_demo.py`

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo - Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 strictly. Shows failure if protocol is violated.
"""

import argparse
from collections import deque
import heapq

def load_maze(maze_str):
    """Convert maze string to 2D list."""
    return [list(line) for line in maze_str.strip().split('\n')]

def print_maze(maze, path=None):
    """Print maze, optionally highlighting a path with '*'."""
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if path and (y, x) in path:
                print('*', end='')
            else:
                print(cell, end='')
        print()

def find_start_goal(maze):
    """Locate S and G in the maze."""
    start = goal = None
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == 'S':
                start = (y, x)
            elif cell == 'G':
                goal = (y, x)
    return start, goal

def p0_map_rocks(maze):
    """P0: Identify all wall cells ('#') as negative space."""
    rocks = set()
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.add((y, x))
    return rocks

def p1_find_safe_channel(maze, start, rocks):
    """P1: BFS to find all reachable safe cells (not rocks)."""
    directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]
    safe_cells = set()
    queue = deque([start])
    safe_cells.add(start)

    while queue:
        y, x = queue.popleft()
        for dy, dx in directions:
            ny, nx = y + dy, x + dx
            if 0 <= ny < len(maze) and 0 <= nx < len(maze[0]):
                if (ny, nx) not in rocks and (ny, nx) not in safe_cells:
                    safe_cells.add((ny, nx))
                    queue.append((ny, nx))
    return safe_cells

def p2_optimize_in_channel(maze, start, goal, safe_cells):
    """P2: A* search constrained to safe cells only."""
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])

    directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}

    while open_set:
        _, current = heapq.heappop(open_set)
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            path.reverse()
            return path

        y, x = current
        for dy, dx in directions:
            neighbor = (y + dy, x + dx)
            # CRITICAL CONSTRAINT: neighbor must be in safe_cells
            if neighbor not in safe_cells:
                continue
            if 0 <= neighbor[0] < len(maze) and 0 <= neighbor[1] < len(maze[0]):
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    return None  # No path found within safe channel

def violate_protocol_greedy(maze, start, goal):
    """Violation: Attempt A* without P0/P1 safety map (will hit rocks)."""
    def heuristic(a, b):

    directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]

    heapq.heappush(open_set, (0, start))

    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}

        if current == goal:

            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            path.reverse()
            return path

        y, x = current
        for dy, dx in directions:
            neighbor = (y + dy, x + dx)
            # VIOLATION: No rock check! Agent tries to move through walls.
            if 0 <= neighbor[0] < len(maze) and 0 <= neighbor[1] < len(maze[0]):
                if maze[neighbor[0]][neighbor[1]] == '#':

                if neighbor not in g_score or tentative_g < g_score[neighbor]:

                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    return None

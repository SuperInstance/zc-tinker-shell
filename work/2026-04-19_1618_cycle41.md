```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Simulates the 20×20 maze with 64 rocks, optimal path length 21.
"""

import random
from typing import List, Tuple, Set

# Constants from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

# Symbols
ROCK = '█'
AGENT = 'A'
GOAL = 'G'
PATH = '·'
SAFE = ' '
UNKNOWN = '?'

class DeadbandMaze:
    def __init__(self):
        self.grid = [[UNKNOWN for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.agent_pos = (0, 0)
        self.goal_pos = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.safe_channels: List[Set[Tuple[int, int]]] = []
        self.visited = set()
        
        # Initialize rocks (P0 negative space)
        self._place_rocks()
        
        # Mark start and goal as safe initially
        self.grid[self.agent_pos[0]][self.agent_pos[1]] = AGENT
        self.grid[self.goal_pos[0]][self.goal_pos[1]] = GOAL
    
    def _place_rocks(self):
        """P0: Map negative space (rocks)"""
        positions = [(r, c) for r in range(MAZE_SIZE) for c in range(MAZE_SIZE)]
        positions.remove(self.agent_pos)
        positions.remove(self.goal_pos)
        
        self.rocks = set(random.sample(positions, ROCK_COUNT))
        for r, c in self.rocks:
            self.grid[r][c] = ROCK
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """Step P0: Identify all failure states (rocks)"""
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Identified {len(self.rocks)} rocks (failure states)")
        print("Do NOT proceed to P1 without this map.")
        return self.rocks
    
    def p1_find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """Step P1: Find all paths that avoid rocks"""
        print("\n=== P1: FIND SAFE CHANNELS ===")
        
        # Simple BFS to find safe paths
        def bfs_safe_paths():
            queue = [(self.agent_pos, [self.agent_pos])]
            safe_paths = []
            
            while queue and len(safe_paths) < 5:  # Limit for demo
                (r, c), path = queue.pop(0)
                
                if (r, c) == self.goal_pos:
                    safe_paths.append(path)
                    continue
                
                # Explore neighbors (up, down, left, right)
                for dr, dc in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
                    nr, nc = r + dr, c + dc
                    if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                        if (nr, nc) not in self.rocks and (nr, nc) not in [p for p, _ in queue]:
                            queue.append(((nr, nc), path + [(nr, nc)]))
            
            return safe_paths
        
        safe_paths = bfs_safe_paths()
        
        if not safe_paths:
            print("NO SAFE CHANNELS FOUND! Agent trapped.")
            return []
        
        print(f"Found {len(safe_paths)} safe channels (rock-free paths)")
        for i, path in enumerate(safe_paths[:3]):  # Show first 3
            print(f"  Channel {i+1}: {len(path)} steps")
        
        return safe_paths
    
    def p2_optimize_within_channel(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """Step P2: Find shortest path within safe channels"""
        print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        if not safe_paths:
            print("No safe channels to optimize. Protocol violation: P2 before P1.")
            return []
        
        # Find shortest safe path
        optimal_path = min(safe_paths, key=len)
        
        print(f"Selected optimal path: {len(optimal_path)} steps")
        print(f"Optimal from simulation: {OPTIMAL_PATH_LENGTH} steps")
        
        # Mark the path on grid
        for r, c in optimal_path[1:-1]:  # Exclude start and goal
            if self.grid[r][c] == UNKNOWN:
                self.grid[r][c] = PATH
        
        return optimal_path
    
    def run_deadband_protocol(self):
        """Execute full P0→P1→P2 protocol"""
        print("=" * 50)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("=" * 50)
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        
        # P1: Find safe channels
        safe_paths = self.p1_find_safe_channels()
        
        # P2: Optimize within channel
        if safe_paths:
            optimal_path = self.p2_optimize_within_channel(safe_paths)
            
            # Results
            print("\n" + "=" * 50)
            print("RESULTS:")
            print(f"  Rocks (P0): {len(rocks)}")
            print(f"  Safe channels found (P1): {len(safe_paths)}")
            print(f"  Optimal path length (P2): {len(optimal_path) if optimal_path else 'N/A'}")
            print(f"  Theoretical optimal: {OPTIMAL_PATH_LENGTH}")
            
            if optimal_path and len(optimal_path) <= OPTIMAL_PATH_LENGTH + 5:
                print("  STATUS: SUCCESS (Deadband protocol followed)")
            else:
                print("  STATUS: FAILURE (Protocol violation or no path)")
        else:
            print("\nPROTOCOL VIOLATION: Attempted P2 without safe channels")
    
    def display_maze(self, show_legend=True):
        """Display the current maze state"""
        print("\n" + "=" * 50)
        print("MAZE STATE (20×20):")
        print("=" * 50)
        
        for r in range(MAZE_SIZE):
            row = ''
            for c in range(MAZE_SIZE):
                row += self.grid[r][c] + ' '
            print(row)
        
        if show_legend:
            print("\nLEGEND:")
            print(f"  {AGENT} = Agent start")
            print(f"  {GOAL} = Goal")
            print(f"  {ROCK} = Rock (failure state)")
            print(f"  {PATH} = Safe path (P2 optimal)")
            print(f"  {SAFE} = Safe but unexplored")
            print(f"  {UNKNOWN} = Unknown")

def main():
    """Run the deadband protocol demonstration"""
    # Create maze
    maze = DeadbandMaze()
    
    # Show initial state
    maze.display_maze(show_legend=True)
    
    # Run protocol
    maze.run_deadband_protocol()
    
    # Show final state with optimal path
    maze.display_maze(show_legend=False)
    
    print("\n" + "=" * 50)
    print("SIMULATION DATA FROM FLEET:")
    print("  Unconstrained (P2-only): 0/50 success")
    print("  Deadband (P0→P1→P2): 50/50 success")
    print("=" * 50)

if __name__ == "__main__":
    main()
```

**Key Features:**
1. **P0 Implementation**: Randomly places 64 rocks (negative space) as per `constraint_sim_final.json`
2. **P1 Implementation**: BFS to find all rock-free paths (safe channels)
3. **P2 Implementation**: Selects shortest path from safe channels only
4. **Visual ASCII Display**: Shows 20×20 maze with symbols matching the protocol
5. **Protocol Validation**: Checks for P2-before-P1 violations
6. **Simulation Accuracy**: Uses actual data from fleet-knowledge (64 rocks, optimal 21 steps)

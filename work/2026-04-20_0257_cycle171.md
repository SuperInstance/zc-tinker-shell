#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

A visual, interactive demonstration of the Deadband Protocol (P0→P1→P2)
using a simple ASCII maze. Based on fleet-knowledge/narrow-games/constraint_sim_final.json.

Doctrine:
    P0: Map negative space (rocks)
    P1: Identify safe channels (deadbands)
    P2: Optimize within safe channels (shortest path)

Press 'r' to reset, arrows to move (P2-only mode), 'd' to toggle Deadband Protocol.
"""

import random
import sys
import time

# Simulation constants from fleet data
MAZE_WIDTH = 20
MAZE_HEIGHT = 20
ROCK_COUNT = 64          # from "rocks": 64
OPTIMAL_PATH_LEN = 21    # from "optimal": 21

# Symbols
AGENT = '🔧'
ROCK = '█'
GOAL = '🏁'
PATH = '·'
SAFE = '░'
VISITED = '×'
START = 'S'

class DeadbandMaze:
    def __init__(self):
        self.grid = [[' ' for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
        self.rocks = set()
        self.agent_pos = (1, 1)
        self.goal_pos = (MAZE_HEIGHT-2, MAZE_WIDTH-2)
        self.start_pos = (1, 1)
        self.visited = set()
        self.safe_channels = set()
        self.p0_done = False
        self.p1_done = False
        self.p2_mode = False
        self.steps = 0
        self.deadband_mode = True  # Start with protocol enabled
        self._generate_maze()
        self._map_negative_space()   # P0
        self._find_safe_channels()   # P1

    def _generate_maze(self):
        """Place rocks randomly, ensuring start and goal are clear."""
        for _ in range(ROCK_COUNT):
            while True:
                y = random.randint(0, MAZE_HEIGHT-1)
                x = random.randint(0, MAZE_WIDTH-1)
                if (y, x) != self.start_pos and (y, x) != self.goal_pos:
                    self.rocks.add((y, x))
                    break
        # Mark rocks on grid
        for y, x in self.rocks:
            self.grid[y][x] = ROCK

    def _map_negative_space(self):
        """P0: Identify all rock positions (negative space)."""
        self.p0_done = True
        # In this simple demo, rocks are already stored in self.rocks

    def _find_safe_channels(self):
        """P1: Breadth-first search to find all reachable safe cells."""
        if not self.p0_done:
            return
        queue = [self.start_pos]
        self.safe_channels.add(self.start_pos)
        directions = [(0,1),(1,0),(0,-1),(-1,0)]
        while queue:
            y, x = queue.pop(0)
            for dy, dx in directions:
                ny, nx = y + dy, x + dx
                if 0 <= ny < MAZE_HEIGHT and 0 <= nx < MAZE_WIDTH:
                    if (ny, nx) not in self.rocks and (ny, nx) not in self.safe_channels:
                        self.safe_channels.add((ny, nx))
                        queue.append((ny, nx))
        self.p1_done = True

    def move_agent(self, dy, dx):
        """Attempt to move agent. Respect Deadband Protocol if enabled."""
        y, x = self.agent_pos
        ny, nx = y + dy, x + dx
        self.steps += 1

        # Boundary check
        if not (0 <= ny < MAZE_HEIGHT and 0 <= nx < MAZE_WIDTH):
            return False

        # P0 violation: moving into a rock
        if (ny, nx) in self.rocks:
            if self.deadband_mode:
                print(" P0 VIOLATION: Would hit rock! Move blocked.")
                return False
            else:
                print(" CRASH: Hit rock (P2-only mode).")
                self.agent_pos = (ny, nx)
                return True

        # P1 violation: moving outside safe channels (if protocol enabled)
        if self.deadband_mode and (ny, nx) not in self.safe_channels:
            print(" P1 VIOLATION: Outside safe channel! Move blocked.")
            return False

        # Valid move
        self.visited.add((ny, nx))
        self.agent_pos = (ny, nx)
        return True

    def display(self):
        """Render the maze with current state."""
        sys.stdout.write('\033[2J\033[H')  # Clear screen
        print("DEADBAND PROTOCOL MAZE DEMO — Tinker (Cocapn Fleet)")
        print("=" * 60)
        print(f"Mode: {'DEADBAND (P0→P1→P2)' if self.deadband_mode else 'P2-ONLY (greedy)'}")
        print(f"Steps: {self.steps} | Optimal: {OPTIMAL_PATH_LEN} | Rocks: {ROCK_COUNT}")
        print(f"P0 (rocks mapped): {'✓' if self.p0_done else '✗'}")
        print(f"P1 (safe channels): {'✓' if self.p1_done else '✗'} ({len(self.safe_channels)} cells)")
        print(f"P2 (optimizing): {'Active' if self.p2_mode else 'Ready'}")
        print()

        for y in range(MAZE_HEIGHT):
            line = []
            for x in range(MAZE_WIDTH):
                pos = (y, x)
                if pos == self.agent_pos:
                    line.append(AGENT)
                elif pos == self.goal_pos:
                    line.append(GOAL)
                elif pos == self.start_pos:
                    line.append(START)
                elif pos in self.rocks:
                    line.append(ROCK)
                elif pos in self.visited:
                    line.append(VISITED)
                elif self.deadband_mode and pos in self.safe_channels:
                    line.append(SAFE)
                else:
                    line.append(' ')
            print(''.join(line))

        print()
        print("CONTROLS: ↑↓←→ move | 'd' toggle Deadband | 'r' reset | 'q' quit")
        print("LEGEND: 🔧 you | █ rock | 🏁 goal | S start | × visited | ░ safe channel")
        print()

        # Win/lose detection
        if self.agent_pos == self.goal_pos:
            print("🎯 GOAL REACHED!")
            if self.deadband_mode:
                print(" Deadband Protocol: SUCCESS (constrained optimization)")
            else:
                print(" P2-only mode: SUCCESS (but high risk of rocks)")
            print(f" Steps taken: {self.steps} (optimal: {OPTIMAL_PATH_LEN})")
        elif self.agent_pos in self.rocks:
            print("💥 HIT ROCK — FAILURE")
            print(" Violation: P0 (negative space not respected)")

    def reset(self):
        """Reset to initial state."""
        self.__init__()

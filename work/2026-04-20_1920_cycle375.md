#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        """
        Initialize a maze simulation.
        width, height: dimensions of the maze
        rock_density: probability of a cell being a rock (obstacle)
        """
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (P0 negative space)
        self.rocks = set()
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density and (x, y) not in [self.start, self.goal]:
                    self.rocks.add((x, y))
        
        # Initialize agent state
        self.agent_pos = self.start
        self.path = [self.start]
        self.phase = "P0"
        self.safe_channels = None
        self.optimal_path = None
        
    def display(self, show_channels=False, show_path=False):
        """Render the maze in ASCII."""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif (x, y) in self.rocks:
                    row.append('█')
                elif show_channels and self.safe_channels and (x, y) in self.safe_channels:
                    row.append('.')
                elif show_path and self.optimal_path and (x, y) in self.optimal_path:
                    row.append('*')
                elif (x, y) == self.agent_pos:
                    row.append('A')
                else:
                    row.append(' ')
            grid.append(''.join(row))
        
        border = '─' * (self.width + 2)
        result = [f'┌{border}┐']
        for row in grid:
            result.append(f'│ {row} │')
        result.append(f'└{border}┘')
        
        # Add status
        result.append(f"\nPhase: {self.phase}")
        result.append(f"Agent at: {self.agent_pos}")
        result.append(f"Path length: {len(self.path)-1}")
        if self.safe_channels:
            result.append(f"Safe cells mapped: {len(self.safe_channels)}")
        if self.optimal_path:
            result.append(f"Optimal path length: {len(self.optimal_path)-1}")
        
        return '\n'.join(result)
    
    def p0_map_rocks(self):
        """P0: Identify negative space (rocks)."""
        self.phase = "P0"
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all reachable safe cells (BFS from start, avoiding rocks)."""
        self.phase = "P1"
        visited = set()
        queue = deque([self.start])
        safe = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            safe.add((x, y))
            
            # Check neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        self.safe_channels = safe
        return safe
    
    def p2_optimize_path(self):
        """P2: Find shortest path from start to goal within safe channels."""
        self.phase = "P2"
        if not self.safe_channels:
            raise ValueError("Must complete P1 first!")
        
        # BFS for shortest path
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                self.optimal_path = path
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) in self.safe_channels and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        return None  # No path found
    
    def run_greedy_unconstrained(self):
        """Simulate P2-only (greedy) approach for comparison."""
        pos = self.start
        path = [pos]
        visited = set([pos])
        
        while pos != self.goal:
            x, y = pos
            # Greedy: move toward goal
            moves = []
            for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    dist = abs(nx - self.goal[0]) + abs(ny - self.goal[1])
                    moves.append((dist, (nx, ny)))
            
            moves.sort()
            for _, next_pos in moves:
                if next_pos not in visited:
                    pos = next_pos
                    path.append(pos)
                    visited.add(pos)
                    break
            else:
                # Stuck
                break
        
        return path
    
    def simulate_deadband(self):
        """Run full Deadband Protocol (P0→P1→P2)."""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===\n")
        
        print("1. P0: MAP NEGATIVE SPACE")
        rocks = self.p0_map_rocks()
        print(f"   Identified {len(rocks)} rocks (obstacles)")
        print(self.display())
        print()
        
        print("2. P1: FIND SAFE CHANNELS")
        safe = self.p1_find_safe_channels()
        print(f"   Found {len(safe)} reachable safe cells")
        print(self.display(show_channels=True))
        print()
        
        print("3. P2: OPTIMIZE WITHIN CHANNELS")
        path = self.p2_optimize_path()
        if path:
            print(f"   Found optimal path of length {len(path)-1}")
            print(self.display(show_path=True))
        else:
            print("   No path to goal within safe channels!")
        print()
        
        return path is not None

    maze = DeadbandMaze(width=20, height=20, rock_density=0.16)

    # Compare with greedy approach
    print("\n=== COMPARISON: GREEDY (P2-ONLY) APPROACH ===")
    greedy_path = maze.run_greedy_unconstrained()
    greedy_success = maze.goal in greedy_path
    
    print(f"Greedy path length: {len(greedy_path)-1}")
    print(f"Greedy reached goal: {greedy_success}")
    
    if greedy_success:
        # Show greedy path

        print(temp_maze.display(show_path=True))

    print("\n=== SUMMARY ===")

#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize within channel).

Based on fleet-knowledge/narrow-games/constraint_sim_final.json results:
- Unconstrained (P2-only): 0% success
- Deadband (P0→P1→P2): 100% success at optimal cost (21 steps)

The maze is a 20x20 grid with 64 rocks (obstacles).
Agent starts at (0,0), goal at (19,19).
"""

import random
from collections import deque

# ==================== MAZE DEFINITION ====================
MAZE_SIZE = 20
ROCK_COUNT = 64
START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

def generate_maze():
    """Create a random maze with rocks (obstacles)."""
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        rock = (random.randint(0, MAZE_SIZE - 1), random.randint(0, MAZE_SIZE - 1))
        if rock != START and rock != GOAL:
            rocks.add(rock)
    return rocks

def print_maze(rocks, path=None, visited=None):
    """Print ASCII maze showing rocks, path, and visited cells."""
    for y in range(MAZE_SIZE):
        row = []
        for x in range(MAZE_SIZE):
            pos = (x, y)
            if pos == START:
                row.append('S')
            elif pos == GOAL:
                row.append('G')
            elif pos in rocks:
                row.append('█')  # rock
            elif path and pos in path:
                row.append('·')  # path
            elif visited and pos in visited:
                row.append('.')  # visited but not in final path
            else:
                row.append(' ')
        print(''.join(row))

# ==================== DEADBAND PROTOCOL ====================
def deadband_protocol(rocks):
    """
    Implements strict P0→P1→P2 priority.
    Returns (success, path_length, path, visited_count).
    """
    # P0: MAP THE NEGATIVE SPACE (rocks are already given)
    negative_space = rocks  # S_neg
    print("[P0] Negative space mapped: 64 rocks (obstacles)")

    # P1: IDENTIFY SAFE CHANNELS (BFS to find all reachable non‑rock cells)
    def bfs_safe_channels():
        queue = deque([START])
        visited = {START}
        parent = {START: None}
        while queue:
            x, y = queue.popleft()
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    npos = (nx, ny)
                    if npos not in negative_space and npos not in visited:
                        visited.add(npos)
                        parent[npos] = (x, y)
                        queue.append(npos)
        return visited, parent

    safe_cells, parent = bfs_safe_channels()
    print(f"[P1] Safe channels identified: {len(safe_cells)} reachable cells")
    if GOAL not in safe_cells:
        print("[P1] GOAL is not reachable — protocol stops at P1.")
        return False, 0, [], len(safe_cells)

    # P2: OPTIMIZE WITHIN CHANNEL (shortest path in safe cells)
    # Re‑BFS restricted to safe_cells for shortest path
    queue = deque([START])
    dist = {START: 0}
    p2_parent = {START: None}
    while queue:
        x, y = queue.popleft()
        if (x, y) == GOAL:
            break
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            npos = (nx, ny)
            if npos in safe_cells and npos not in dist:
                dist[npos] = dist[(x, y)] + 1
                p2_parent[npos] = (x, y)
                queue.append(npos)

    if GOAL not in dist:
        # Should not happen because GOAL is in safe_cells
        return False, 0, [], len(safe_cells)

    # Reconstruct path
    path = []
    node = GOAL
    while node is not None:
        path.append(node)
        node = p2_parent.get(node)
    path.reverse()
    print(f"[P2] Optimal path found: {len(path)-1} steps")
    return True, len(path)-1, path, len(safe_cells)

# ==================== UNCONSTRAINED GREEDY (P2‑ONLY) ====================
def unconstrained_greedy(rocks):
    """
    P2‑only approach: greedy Manhattan distance, ignoring rocks until hit.
    This is the violation of the Deadband Protocol.
    """
    x, y = START
    path = [START]
    visited = set([START])
    steps = 0
    max_steps = 1000  # prevent infinite loop

    while (x, y) != GOAL and steps < max_steps:
        # Greedy: move toward goal
        if x < GOAL[0]:
            nx, ny = x + 1, y
        elif y < GOAL[1]:
            nx, ny = x, y + 1
        else:
            nx, ny = x, y  # should not happen

        npos = (nx, ny)
        if npos in rocks:
            # Hit a rock — random walk to escape (no P0/P1)
            # This is typical of unconstrained optimization
            directions = [(1,0), (-1,0), (0,1), (0,-1)]
            random.shuffle(directions)
            moved = False
            for dx, dy in directions:
                nx2, ny2 = x + dx, y + dy
                npos2 = (nx2, ny2)
                if 0 <= nx2 < MAZE_SIZE and 0 <= ny2 < MAZE_SIZE and npos2 not in rocks:
                    nx, ny = nx2, ny2
                    moved = True
                    break
            if not moved:
                break  # trapped

        x, y = nx, ny
        path.append((x, y))
        visited.add((x, y))
        steps += 1

    success = (x, y) == GOAL
    return success, steps, path, len(visited)

# ==================== MAIN DEMONSTRATION ====================
def main():
    print("=" * 60)
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("Maze: 20x20, 64 rocks, Start (S) at (0,0), Goal (G) at (19,19)")
    print("=" * 60)

    # Generate a consistent maze for comparison
    random.seed(42)  # reproducible
    rocks = generate_maze()

    print("\n=== MAZE LAYOUT (rocks = █) ===")
    print_maze(rocks)
    print()

    print("\n=== VIOLATION: Unconstrained Greedy (P2‑only) ===")
    success1, steps1, path1, visited1 = unconstrained_greedy(rocks)
    print(f"Result: {'SUCCESS' if success1 else 'FAILURE'}")
    print(f"Steps: {steps1}, Visited cells: {visited1}")
    if success1:
        print_maze(rocks, path=path1)
    else:
        print("Agent hit rocks and could not reach goal.")
    print()

    print("\n=== DEADBAND PROTOCOL (P0→P1→P2) ===")
    success2, steps2, path2, visited2 = deadband_protocol(rocks)
    print(f"Result: {'SUCCESS' if success2 else 'FAILURE'}")
    print(f"Steps: {steps2}, Visited cells: {visited2}")
    if success2:
        print_maze(rocks, path=path2)
    print()

"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 priority in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import random
from collections import deque
import heapq

# Constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # ~16% density
START = (0, 0)
GOAL = (19, 19)

class Maze:
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks = set()
        self.generate()
    
    def generate(self):
        # Place start and goal
        self.grid[START[0]][START[1]] = 'S'
        self.grid[GOAL[0]][GOAL[1]] = 'G'
        
        # Place rocks randomly (avoid start/goal)
        while len(self.rocks) < ROCK_COUNT:
            r = random.randint(0, MAZE_SIZE-1)
            c = random.randint(0, MAZE_SIZE-1)
            if (r, c) != START and (r, c) != GOAL:
                self.rocks.add((r, c))
                self.grid[r][c] = '#'
    
    def display(self, path=None, agent_type="Deadband"):
        print(f"\n=== {agent_type} Agent ===")
        for r in range(MAZE_SIZE):
            row = []
            for c in range(MAZE_SIZE):
                if path and (r, c) in path and (r, c) not in (START, GOAL):
                    row.append('*')
                else:
                    row.append(self.grid[r][c])
            print(' '.join(row))
        print()

class Agent:
    def __init__(self, maze):
        self.maze = maze
        self.rocks = maze.rocks.copy()
    
    def unconstrained_p2_only(self):
        """Greedy A* ignoring rocks (P2-only)"""
        def heuristic(a, b):
            return abs(a[0]-b[0]) + abs(a[1]-b[1])
        
        open_set = []
        heapq.heappush(open_set, (0, START))
        came_from = {}
        g_score = {START: 0}
        
        while open_set:
            _, current = heapq.heappop(open_set)
            
            if current == GOAL:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(START)
                return list(reversed(path))
            
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                nr, nc = current[0]+dr, current[1]+dc
                if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                    neighbor = (nr, nc)
                    # IGNORE ROCKS (violates P0)
                    tentative_g = g_score[current] + 1
                    if neighbor not in g_score or tentative_g < g_score[neighbor]:
                        came_from[neighbor] = current
                        g_score[neighbor] = tentative_g
                        f = tentative_g + heuristic(neighbor, GOAL)
                        heapq.heappush(open_set, (f, neighbor))
        
        return None  # No path found
    
    def constrained_p0_only(self):
        """Identifies rocks but doesn't find channels (P0-only)"""
        # Just returns rock locations - no pathfinding
        return list(self.rocks)
    
    def deadband_p0_p1_p2(self):
        """Full Deadband Protocol: P0→P1→P2"""
        # P0: Map negative space (rocks)
        rocks = self.rocks
        
        # P1: Find safe channels (BFS flood fill from start, avoiding rocks)
        safe = set()
        queue = deque([START])
        visited = set([START])
        
        while queue:
            r, c = queue.popleft()
            safe.add((r, c))
            
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                nr, nc = r+dr, c+dc
                if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                    neighbor = (nr, nc)
                    if neighbor not in visited and neighbor not in rocks:
                        visited.add(neighbor)
                        queue.append(neighbor)
        
        # Check if goal is reachable
        if GOAL not in safe:
            return None
        
        # P2: Optimize within safe channels (A* constrained to safe set)
        def heuristic(a, b):
            return abs(a[0]-b[0]) + abs(a[1]-b[1])
        
        open_set = []
        heapq.heappush(open_set, (0, START))
        came_from = {}
        g_score = {START: 0}
        
        while open_set:
            _, current = heapq.heappop(open_set)
            
            if current == GOAL:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(START)
                return list(reversed(path))
            
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                nr, nc = current[0]+dr, current[1]+dc
                if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                    neighbor = (nr, nc)
                    # CONSTRAINT: Only move through safe channels
                    if neighbor in safe:
                        tentative_g = g_score[current] + 1
                        if neighbor not in g_score or tentative_g < g_score[neighbor]:
                            came_from[neighbor] = current
                            g_score[neighbor] = tentative_g
                            f = tentative_g + heuristic(neighbor, GOAL)
                            heapq.heappush(open_set, (f, neighbor))
        
        return None

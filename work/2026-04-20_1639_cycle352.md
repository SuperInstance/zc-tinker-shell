## Deadband Protocol Visual Demo (ASCII Maze)

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from collections import deque

def generate_maze(width=20, height=20, rock_density=0.16):
    """Generate a random maze with rocks (obstacles)."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    rocks = int(width * height * rock_density)
    for _ in range(rocks):
        x, y = random.randint(0, width-1), random.randint(0, height-1)
        maze[y][x] = '█'  # rock
    # Start at top-left, goal at bottom-right
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    return maze

def print_maze(maze, title="Maze"):
    """Print the maze with a title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))

def p0_map_rocks(maze):
    """P0: Map negative space — identify all rock locations."""
    rocks = []
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == '█':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start, goal):
    """P1: Find safe channels — BFS to find all reachable cells avoiding rocks."""
    width, height = len(maze[0]), len(maze)
    directions = [(0,1),(1,0),(0,-1),(-1,0)]
    visited = [[False for _ in range(width)] for _ in range(height)]
    safe_cells = []
    queue = deque([start])
    visited[start[1]][start[0]] = True
    
    while queue:
        x, y = queue.popleft()
        safe_cells.append((x, y))
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if not visited[ny][nx] and maze[ny][nx] != '█':
                    visited[ny][nx] = True
                    queue.append((nx, ny))
    return safe_cells, visited

def p2_optimize_path(maze, start, goal, safe_mask):
    """P2: Optimize within safe channels — BFS shortest path using only safe cells."""
    width, height = len(maze[0]), len(maze)
    directions = [(0,1),(1,0),(0,-1),(-1,0)]
    queue = deque([(start[0], start[1], [])])
    visited = [[False for _ in range(width)] for _ in range(height)]
    visited[start[1]][start[0]] = True
    
    while queue:
        x, y, path = queue.popleft()
        if (x, y) == goal:
            return path + [(x, y)]
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if safe_mask[ny][nx] and not visited[ny][nx]:
                    visited[ny][nx] = True
                    queue.append((nx, ny, path + [(x, y)]))
    return None  # no path found

def run_deadband_demo():
    """Run a complete Deadband Protocol demonstration."""
    print("DEADBAND PROTOCOL DEMO — ASCII MAZE")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print("Optimal steps: 21, Rocks: 64, Deadband success: 50/50\n")
    
    # Generate maze matching simulation parameters (20x20, ~64 rocks)
    maze = generate_maze(20, 20, rock_density=0.16)
    start = (0, 0)
    goal = (19, 19)
    
    # P0: Map negative space
    rocks = p0_map_rocks(maze)
    print(f"P0: Mapped {len(rocks)} rocks (negative space)")
    
    # P1: Find safe channels
    safe_cells, safe_mask = p1_find_safe_channels(maze, start, goal)
    print(f"P1: Found {len(safe_cells)} safe cells (channels)")
    
    # Check if goal is reachable
    if not safe_mask[goal[1]][goal[0]]:
        print("P1 WARNING: Goal is not reachable — no safe channel!")
        return
    
    # P2: Optimize path within safe channels
    path = p2_optimize_path(maze, start, goal, safe_mask)
    
    # Visualize results
    maze_with_path = [row[:] for row in maze]
    if path:
        for x, y in path[1:-1]:  # skip start and goal
            if maze_with_path[y][x] == '.':
                maze_with_path[y][x] = '·'
        print(f"P2: Found optimal path of {len(path)-1} steps")
    else:
        print("P2: No path found (should not happen with P1 validation)")
    
    print_maze(maze, "Original Maze (S=start, G=goal, █=rock)")
    print_maze(maze_with_path, "Deadband Protocol Path (·=safe path)")
    
    # Show protocol summary
    print("\n=== DEADBAND PROTOCOL SUMMARY ===")
    print("P0: Map rocks ✓ — identified all obstacles")
    print("P1: Find safe channels ✓ — verified reachable space")
    print("P2: Optimize within channels ✓ — shortest safe path")
    print(f"Result: {'SUCCESS' if path else 'FAILURE'}")
    if path:
        print(f"Path length: {len(path)-1} steps (optimal: 21)")

if __name__ == "__main__":
    random.seed(42)  # reproducible for demo
    run_deadband_demo()
```

**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on reading `fleet-knowledge/narrow-games/constraint_sim_final.json` and `fleet-knowledge/docs/DEADBAND-PROTOCOL.md`, I will implement a concrete demonstration. The simulation data shows: unconstrained (P2-only) success 33%, deadband (P0→P1→P2) success 50%. I will model this.

**Approach:**
1. **P0 (Map negative space):** Generate a maze with rocks ('#') and open cells ('.'). Identify all rock positions.
2. **P1 (Find safe channels):** From start to goal, find all paths that avoid rocks.
3. **P2 (Optimize within channels):** Select the shortest safe path.
4. **Visualization:** Print the maze with the chosen path marked.

**Output:** A Python script `deadband_maze_demo.py` that runs the protocol and prints each step.

## File: `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map rocks (negative space)
P1: Find safe channels (paths avoiding rocks)
P2: Optimize within channels (shortest safe path)
"""

import random
from collections import deque

def generate_maze(width=10, height=10, rock_density=0.2):
    """Generate a random maze with rocks."""
    maze = []
    for y in range(height):
        row = []
        for x in range(width):
            if random.random() < rock_density:
                row.append('#')  # rock
            else:
                row.append('.')  # open
        maze.append(row)
    # Ensure start (0,0) and goal (width-1, height-1) are open
    maze[0][0] = '.'
    maze[height-1][width-1] = '.'
    return maze

def print_maze(maze, path=None):
    """Print maze with optional path marked."""
    for y, row in enumerate(maze):
        line = []
        for x, cell in enumerate(row):
            if path and (x, y) in path:
                line.append('*')
            else:
                line.append(cell)
        print(''.join(line))

def p0_map_rocks(maze):
    """P0: Identify all rock positions (negative space)."""
    rocks = []
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start, goal, rocks):
    """P1: Find all paths from start to goal that avoid rocks."""
    width, height = len(maze[0]), len(maze)
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    # BFS to find all safe paths
    queue = deque([(start, [start])])
    safe_paths = []
    
    while queue:
        (x, y), path = queue.popleft()
        if (x, y) == goal:
            safe_paths.append(path)
            continue
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in rocks and (nx, ny) not in path:
                    queue.append(((nx, ny), path + [(nx, ny)]))
    return safe_paths

def p2_optimize_within_channels(safe_paths):
    """P2: Select the shortest safe path."""
    if not safe_paths:
        return None
    return min(safe_paths, key=len)

def run_deadband_protocol():
    """Execute the full Deadband Protocol."""
    print("=== Deadband Protocol Maze Demo ===\n")
    
    # Generate maze
    maze = generate_maze(width=8, height=8, rock_density=0.25)
    print("Initial Maze ('.' = open, '#' = rock):")
    print_maze(maze)
    print()
    
    start = (0, 0)
    goal = (len(maze[0])-1, len(maze)-1)
    
    # P0: Map negative space
    rocks = p0_map_rocks(maze)
    print(f"P0: Mapped {len(rocks)} rock positions (negative space).")
    
    # P1: Find safe channels
    safe_paths = p1_find_safe_channels(maze, start, goal, rocks)
    print(f"P1: Found {len(safe_paths)} safe channels (paths avoiding rocks).")
    
    if not safe_paths:
        print("No safe channels found. Goal unreachable.")
        return
    
    # P2: Optimize within channels
    optimal_path = p2_optimize_within_channels(safe_paths)
    print(f"P2: Selected optimal path length = {len(optimal_path)} steps.")
    
    print("\nOptimal Safe Path (marked '*'):")
    print_maze(maze, set(optimal_path))
    
    # Simulate unconstrained (P2-only) failure
    print("\n--- Contrast: Unconstrained (P2-only) Approach ---")
    # A naive greedy BFS that ignores rocks
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    queue = deque([(start, [start])])
    visited = set([start])
    unconstrained_path = None
    
    while queue:
        (x, y), path = queue.popleft()
        if (x, y) == goal:
            unconstrained_path = path
            break
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < len(maze[0]) and 0 <= ny < len(maze):
                if (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    if unconstrained_path:
        # Check if it hits rocks
        hits_rock = any(maze[y][x] == '#' for (x, y) in unconstrained_path)
        if hits_rock:
            print("Unconstrained path hits rocks (failure).")
        else:
            print("Unconstrained path accidentally safe (lucky).")
        print(f"Unconstrained path length = {len(unconstrained_path)}")
    else:
        print("Unconstrained search failed to find any path.")

if __name__ == "__main__":
    run_deadband_protocol()
```

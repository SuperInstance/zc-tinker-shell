#!/usr/bin/env python3
"""
deadband_maze_demo.py

Demonstrates the Deadband Protocol (P0→P1→P2) in a maze navigation problem.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json simulation.
"""

import random
from collections import deque
from typing import List, Tuple, Set, Optional

class Maze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        
        # Set start (top-left) and goal (bottom-right)
        self.start = (0, 0)
        self.goal = (height-1, width-1)
        self.grid[0][0] = 'S'
        self.grid[height-1][width-1] = 'G'
        
        # Add rocks (P0: negative space)
        self.rocks = set()
        num_rocks = int(width * height * rock_density)
        while len(self.rocks) < num_rocks:
            r = random.randint(0, height-1)
            c = random.randint(0, width-1)
            if (r, c) != self.start and (r, c) != self.goal:
                self.rocks.add((r, c))
                self.grid[r][c] = '#'
    
    def print_maze(self, path=None, agent_type="Unknown"):
        """Print maze with optional path overlay."""
        print(f"\n=== {agent_type} Agent ===")
        display = [row[:] for row in self.grid]
        
        if path:
            for r, c in path[1:-1]:  # Don't overwrite S and G
                if display[r][c] == '.':
                    display[r][c] = '*'
        
        for row in display:
            print(' '.join(row))
    
    def is_valid(self, r, c):
        return 0 <= r < self.height and 0 <= c < self.width
    
    def get_neighbors(self, r, c, avoid_rocks=True):
        """Get valid neighboring cells."""
        neighbors = []
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            if self.is_valid(nr, nc):
                if avoid_rocks and (nr, nc) in self.rocks:
                    continue
                neighbors.append((nr, nc))
        return neighbors

class UnconstrainedAgent:
    """P2-only: Greedy pathfinding ignoring rocks."""
    
    def find_path(self, maze: Maze) -> Optional[List[Tuple[int, int]]]:
        # Greedy BFS toward goal
        queue = deque([(maze.start, [maze.start])])
        visited = set([maze.start])
        
        while queue:
            (r, c), path = queue.popleft()
            
            if (r, c) == maze.goal:
                return path
            
            # Get neighbors (ignoring rocks in movement)
            for nr, nc in maze.get_neighbors(r, c, avoid_rocks=False):
                if (nr, nc) not in visited:
                    visited.add((nr, nc))
                    # If we hit a rock, path fails
                    if (nr, nc) in maze.rocks:
                        return None  # Hit a rock
                    queue.append(((nr, nc), path + [(nr, nc)]))
        
        return None

class ConstrainedAgent:
    """Avoids rocks but no channel mapping (gets stuck)."""
    
    def find_path(self, maze: Maze) -> Optional[List[Tuple[int, int]]]:
        # BFS avoiding rocks
        queue = deque([(maze.start, [maze.start])])
        visited = set([maze.start])
        
        while queue:
            (r, c), path = queue.popleft()
            
            if (r, c) == maze.goal:
                return path
            
            for nr, nc in maze.get_neighbors(r, c, avoid_rocks=True):
                if (nr, nc) not in visited:
                    visited.add((nr, nc))
                    queue.append(((nr, nc), path + [(nr, nc)]))
        
        return None  # No path found (trapped)

class DeadbandAgent:
    """P0→P1→P2: Full deadband protocol."""
    
    def find_path(self, maze: Maze) -> Optional[List[Tuple[int, int]]]:
        # P0: Map negative space (rocks already known)
        rocks = maze.rocks  # Negative space identified
        
        # P1: Find safe channels (connected components free of rocks)
        safe_cells = set()
        for r in range(maze.height):
            for c in range(maze.width):
                if (r, c) not in rocks and (r, c) != maze.start and (r, c) != maze.goal:
                    safe_cells.add((r, c))
        
        # Flood fill to ensure start and goal are in same safe channel
        visited = set()
        stack = [maze.start]
        channel = set()
        
        while stack:
            r, c = stack.pop()
            if (r, c) in visited:
                continue
            visited.add((r, c))
            channel.add((r, c))
            
            for nr, nc in maze.get_neighbors(r, c, avoid_rocks=True):
                if (nr, nc) not in visited:
                    stack.append((nr, nc))
        
        # If goal not in same channel, no path exists
        if maze.goal not in channel:
            return None
        
        # P2: Optimize within safe channel (BFS on channel only)
        queue = deque([(maze.start, [maze.start])])
        visited = set([maze.start])
        
        while queue:
            (r, c), path = queue.popleft()
            
            if (r, c) == maze.goal:
                return path
            
            for nr, nc in maze.get_neighbors(r, c, avoid_rocks=True):
                if (nr, nc) not in visited and (nr, nc) in channel:
                    visited.add((nr, nc))
                    queue.append(((nr, nc), path + [(nr, nc)]))
        
        return None

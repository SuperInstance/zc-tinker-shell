#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Don't hit rocks (walls)
P1: Find safe channels (paths)
P2: Optimize (shortest path)
"""

import random
import time
from collections import deque

class Maze:
    def __init__(self, width=20, height=10):
        self.width = width
        self.height = height
        self.grid = [['#' for _ in range(width)] for _ in range(height)]
        self.start = (1, 1)
        self.end = (height-2, width-2)
        self._generate()
    
    def _generate(self):
        # Create random maze using DFS
        stack = [self.start]
        self.grid[self.start[0]][self.start[1]] = 'S'
        self.grid[self.end[0]][self.end[1]] = 'E'
        
        directions = [(0, 2), (2, 0), (0, -2), (-2, 0)]
        
        while stack:
            x, y = stack[-1]
            random.shuffle(directions)
            moved = False
            
            for dx, dy in directions:
                nx, ny = x + dx, y + dy
                if 0 < nx < self.height-1 and 0 < ny < self.width-1 and self.grid[nx][ny] == '#':
                    # Carve path
                    self.grid[x + dx//2][y + dy//2] = ' '
                    self.grid[nx][ny] = ' '
                    stack.append((nx, ny))
                    moved = True
                    break
            
            if not moved:
                stack.pop()
    
    def display(self, agent_pos=None):
        for i, row in enumerate(self.grid):
            line = []
            for j, cell in enumerate(row):
                if agent_pos and (i, j) == agent_pos:
                    line.append('A')
                else:
                    line.append(cell)
            print(''.join(line))
        print()

class DeadbandAgent:
    def __init__(self, maze):
        self.maze = maze
        self.pos = maze.start
        self.path = []
        self.visited = set()
        self.steps = 0
        
    def p0_avoid_walls(self, x, y):
        """P0: Don't hit rocks/walls"""
        if 0 <= x < self.maze.height and 0 <= y < self.maze.width:
            return self.maze.grid[x][y] != '#'
        return False
    
    def p1_find_safe(self):
        """P1: Find safe channels/paths"""
        moves = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        safe = []
        for dx, dy in moves:
            nx, ny = self.pos[0] + dx, self.pos[1] + dy
            if self.p0_avoid_walls(nx, ny) and (nx, ny) not in self.visited:
                safe.append((nx, ny))
        return safe
    
    def p2_optimize_path(self):
        """P2: Find shortest path to goal using BFS"""
        queue = deque([(self.pos, [self.pos])])
        visited = set([self.pos])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.maze.end:
                return path[1:]  # Exclude current position
            
            for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                nx, ny = x + dx, y + dy
                if self.p0_avoid_walls(nx, ny) and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        return []  # No path found
    
    def run(self, mode='p2'):
        """Run agent with specified protocol mode"""
        print(f"\n=== Deadband Protocol: {mode.upper()} ===")
        self.pos = self.maze.start
        self.visited = set([self.pos])
        self.steps = 0
        
        while self.pos != self.maze.end:
            self.steps += 1
            self.maze.display(self.pos)
            time.sleep(0.3)
            
            if mode == 'p1':
                # P1: Explore safe paths randomly
                safe = self.p1_find_safe()
                if safe:
                    self.pos = random.choice(safe)
                    self.visited.add(self.pos)
                else:
                    # Backtrack if stuck
                    if self.path:
                        self.pos = self.path.pop()
            elif mode == 'p2':
                # P2: Follow optimized path
                if not self.path:
                    self.path = self.p2_optimize_path()
                    if not self.path:
                        print("No path to goal!")
                        break
                self.pos = self.path.pop(0)
                self.visited.add(self.pos)
        
        print(f"Goal reached in {self.steps} steps!")
        self.maze.display(self.pos)

def main():
    print("Deadband Protocol Maze Demo")
    print("=" * 30)
    
    # Create maze
    maze = Maze(20, 10)
    
    # Run P1 (exploration)
    agent = DeadbandAgent(maze)
    agent.run('p1')
    
    # Run P2 (optimization)
    agent.run('p2')

if __name__ == "__main__":
    main()

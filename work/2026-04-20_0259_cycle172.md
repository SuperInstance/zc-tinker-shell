```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 172)
A visual demonstration of P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json and DEADBAND-PROTOCOL.md.
"""

import random
from collections import deque

# ========== MAZE DEFINITION ==========
# 10x10 maze: '.' = open, '#' = rock, 'S' = start, 'G' = goal
MAZE = [
    "..........",
    ".#.#.#.#..",
    "..........",
    ".#.#.#.#..",
    "..........",
    ".#.#.#.#..",
    "..........",
    ".#.#.#.#..",
    "..........",
    "..........",
]
START = (0, 0)
GOAL = (9, 9)

# ========== HELPER FUNCTIONS ==========
def print_maze(maze, title="", path=None, rocks=None, channels=None):
    """Print maze with optional overlays."""
    print(f"\n=== {title} ===")
    for r in range(len(maze)):
        line = ""
        for c in range(len(maze[r])):
            pos = (r, c)
            if pos == START:
                line += "S"
            elif pos == GOAL:
                line += "G"
            elif path and pos in path:
                line += "*"
            elif channels and pos in channels:
                line += "~"
            elif rocks and pos in rocks:
                line += "#"
            else:
                line += maze[r][c]
        print(line)

def get_neighbors(pos, maze):
    """Return valid orthogonal neighbors within bounds."""
    r, c = pos
    neighbors = []
    for dr, dc in [(-1,0),(1,0),(0,-1),(0,1)]:
        nr, nc = r + dr, c + dc
        if 0 <= nr < len(maze) and 0 <= nc < len(maze[0]):
            neighbors.append((nr, nc))
    return neighbors

# ========== DEADBAND PROTOCOL IMPLEMENTATION ==========
def p0_map_rocks(maze):
    """P0: Identify all rocks (negative space)."""
    rocks = set()
    for r in range(len(maze)):
        for c in range(len(maze[r])):
            if maze[r][c] == '#':
                rocks.add((r, c))
    return rocks

def p1_find_safe_channels(maze, rocks, start, goal):
    """P1: Find all cells reachable from start without hitting rocks."""
    visited = set()
    queue = deque([start])
    while queue:
        pos = queue.popleft()
        if pos in visited:
            continue
        visited.add(pos)
        for nr, nc in get_neighbors(pos, maze):
            if (nr, nc) not in rocks and (nr, nc) not in visited:
                queue.append((nr, nc))
    return visited

def p2_optimize_path(maze, safe_cells, start, goal):
    """P2: Find shortest path from start to goal within safe cells (BFS)."""
    if goal not in safe_cells:
        return None  # Goal unreachable within safe channels
    queue = deque([(start, [start])])
    visited = set([start])
    while queue:
        pos, path = queue.popleft()
        if pos == goal:
            return path
        for nr, nc in get_neighbors(pos, maze):
            npos = (nr, nc)
            if npos in safe_cells and npos not in visited:
                visited.add(npos)
                queue.append((npos, path + [npos]))
    return None

def unconstrained_greedy(maze, start, goal):
    """P2-only (greedy) approach: optimize without mapping rocks."""
    # Simple greedy that moves toward goal, ignoring rock mapping
    path = [start]
    pos = start
    visited = set([start])
    for _ in range(100):  # limit steps
        if pos == goal:
            return path
        # Pick neighbor that reduces Manhattan distance to goal
        neighbors = get_neighbors(pos, maze)
        neighbors = [n for n in neighbors if n not in visited]
        if not neighbors:
            break
        # Greedy choice
        best = min(neighbors, key=lambda n: abs(n[0]-goal[0]) + abs(n[1]-goal[1]))
        # BUT: might step on a rock!
        r, c = best
        if maze[r][c] == '#':
            return None  # hit a rock → failure
        path.append(best)
        visited.add(best)
        pos = best
    return None

# ========== MAIN DEMONSTRATION ==========
def main():
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("Based on fleet simulation: deadband success 50/50, unconstrained 0/50")
    print("=" * 50)
    
    # Original maze
    print_maze(MAZE, "Original Maze")
    
    # ----- P0: Map Rocks -----
    rocks = p0_map_rocks(MAZE)
    print_maze(MAZE, "P0: Map Rocks (negative space)", rocks=rocks)
    print(f"Rocks identified: {len(rocks)}")
    
    # ----- P1: Find Safe Channels -----
    safe_cells = p1_find_safe_channels(MAZE, rocks, START, GOAL)
    print_maze(MAZE, "P1: Safe Channels (deadbands)", channels=safe_cells)
    print(f"Safe cells reachable from start: {len(safe_cells)}")
    print(f"Goal in safe channels? {GOAL in safe_cells}")
    
    # ----- P2: Optimize Within Channels -----
    deadband_path = p2_optimize_path(MAZE, safe_cells, START, GOAL)
    if deadband_path:
        print_maze(MAZE, "P2: Optimal Path Within Safe Channels", path=deadband_path)
        print(f"Deadband path length: {len(deadband_path)-1} steps")
        print(f"Path: {deadband_path}")
    else:
        print("Deadband protocol: Goal unreachable within safe channels.")
    
    # ----- Unconstrained (P2-only) Greedy -----
    greedy_path = unconstrained_greedy(MAZE, START, GOAL)
    if greedy_path:
        print_maze(MAZE, "Unconstrained Greedy (P2-only)", path=greedy_path)
        print(f"Greedy path length: {len(greedy_path)-1} steps")
    else:
        print_maze(MAZE, "Unconstrained Greedy (P2-only) — FAILED", path=[])
        print("Greedy hit a rock (failure).")
    
    # ----- Summary -----
    print("\n" + "=" * 50)
    print("SUMMARY (simulating 1 run):")
    print(f"  Deadband protocol (P0→P1→P2): {'SUCCESS' if deadband_path else 'FAIL'}")
    print(f"  Unconstrained greedy (P2-only): {'SUCCESS' if greedy_path else 'FAIL'}")
    print("\nNote: In fleet simulation (50 runs):")
    print("  - Deadband success: 50/50")
    print("  - Unconstrained success: 0/50")
    print("  (Single run may not reflect aggregate statistics.)")

if __name__ == "__main__":
    main()
```

**Explanation:**
1. **P0 (Map Rocks)**: Identifies all `#` cells as negative space.
2. **P1 (Safe Channels)**: Flood-fills from start, avoiding rocks, to find all reachable safe cells.
3. **P2 (Optimize)**: Performs BFS within the safe channel set to find shortest path to goal.
4. **Unconstrained Greedy**: A P2-only agent that moves toward goal without rock mapping — often hits a rock.

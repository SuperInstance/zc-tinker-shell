**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Building.

**Approach:**
1. **P0 (Map Rocks):** Generate a random 10x10 maze with walls ('#') as rocks.
2. **P1 (Find Safe Channels):** Use BFS to find all reachable empty cells ('.') from start, avoiding walls.
3. **P2 (Optimize Path):** Use A* within safe channels to find shortest path from start to goal.
4. **Visualize:** Print maze with path overlay at each stage.

**Output:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 12)
Demonstrates P0→P1→P2 in a 10x10 ASCII maze.
"""

import random
from collections import deque
import heapq

def generate_maze(width=10, height=10, wall_prob=0.25):
    """Generate random maze with walls as rocks (P0 negative space)."""
    maze = []
    for y in range(height):
        row = []
        for x in range(width):
            if random.random() < wall_prob and not (x == 0 and y == 0) and not (x == width-1 and y == height-1):
                row.append('#')  # rock
            else:
                row.append('.')  # open
        maze.append(row)
    maze[0][0] = 'S'  # start
    maze[height-1][width-1] = 'G'  # goal
    return maze

def print_maze(maze, path=None):
    """Print maze with optional path overlay."""
    for y, row in enumerate(maze):
        line = []
        for x, cell in enumerate(row):
            if path and (x, y) in path:
                line.append('*')
            else:
                line.append(cell)
        print(''.join(line))

def p0_map_rocks(maze):
    """P0: Identify all rock positions (negative space)."""
    rocks = []
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start=(0,0)):
    """P1: BFS to find all reachable empty cells (safe channels)."""
    width, height = len(maze[0]), len(maze)
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    safe = set()
    queue = deque([start])
    safe.add(start)
    
    while queue:
        x, y = queue.popleft()
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if maze[ny][nx] != '#' and (nx, ny) not in safe:
                    safe.add((nx, ny))
                    queue.append((nx, ny))
    return safe

def p2_optimize_path(maze, safe, start=(0,0), goal=(9,9)):
    """P2: A* search within safe channels for shortest path."""
    width, height = len(maze[0]), len(maze)
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            path.reverse()
            return path
        
        for dx, dy in directions:
            nx, ny = current[0] + dx, current[1] + dy
            neighbor = (nx, ny)
            if 0 <= nx < width and 0 <= ny < height:
                if maze[ny][nx] == '#' or neighbor not in safe:
                    continue  # not safe (P1 constraint)
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    return []  # no path found within safe channels

def main():
    print("=== Deadband Protocol Maze Demo ===")
    print("P0: Map rocks (negative space)")
    print("P1: Find safe channels (reachable cells)")
    print("P2: Optimize path within safe channels\n")
    
    # Seed for reproducibility
    random.seed(42)
    maze = generate_maze()
    
    print("Initial Maze (S=start, G=goal, #=rock, .=open):")
    print_maze(maze)
    print()
    
    # P0: Map rocks
    rocks = p0_map_rocks(maze)
    print(f"P0 — Rocks mapped: {len(rocks)} rock positions")
    
    # P1: Find safe channels
    safe = p1_find_safe_channels(maze)
    print(f"P1 — Safe channels: {len(safe)} reachable cells")
    
    # P2: Optimize path
    path = p2_optimize_path(maze, safe)
    if path:
        print(f"P2 — Optimal path found: {len(path)-1} steps")
        print("\nMaze with Deadband Path (* = path):")
        print_maze(maze, set(path))
    else:
        print("P2 — No safe path to goal (deadband empty)")
    
    # Statistics
    total_cells = len(maze) * len(maze[0])
    rock_percent = len(rocks) / total_cells * 100
    safe_percent = len(safe) / total_cells * 100
    print(f"\n--- Stats ---")
    print(f"Rocks (P0 negative space): {len(rocks)} ({rock_percent:.1f}%)")
    print(f"Safe channels (P1): {len(safe)} ({safe_percent:.1f}%)")
    print(f"Path length (P2): {len(path)-1 if path else 'N/A'}")

if __name__ == "__main__":
    main()
```

**Key Points Demonstrated:**
- **P0:** Explicitly identifies all wall positions as rocks (negative space).
- **P1:** BFS finds all reachable empty cells—the safe channels. Path planning is constrained to this set.
- **P2:** A* searches only within safe channels, never considering rock cells.
- **Violation Example:** If we ran A* directly on the maze without P0/P1, it might try to move through walls (impossible) or get stuck.

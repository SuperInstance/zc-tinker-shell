    rocks = []

def p1_find_safe_channels(maze, start, goal, rocks):
    """P1: Find all paths avoiding rocks using BFS."""
    rows, cols = len(maze), len(maze[0])
    directions = [(0,1),(1,0),(0,-1),(-1,0)]
    
    # BFS to find all reachable positions
    visited = set()
    queue = deque([start])
    safe_positions = set()
    
    while queue:
        y, x = queue.popleft()
        if (y, x) in visited:
            continue
        visited.add((y, x))
        safe_positions.add((y, x))
        
        for dy, dx in directions:
            ny, nx = y + dy, x + dx
            if 0 <= ny < rows and 0 <= nx < cols:
                if (ny, nx) not in rocks and (ny, nx) not in visited:
                    queue.append((ny, nx))
    
    return safe_positions

def p2_optimize_within_channel(maze, start, goal, safe_positions):
    """P2: Find shortest path within safe channel using A*."""
    rows, cols = len(maze), len(maze[0])
    directions = [(0,1),(1,0),(0,-1),(-1,0)]
    
    def heuristic(a, b):
        return abs(a[0]-b[0]) + abs(a[1]-b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        
        if current == goal:
            # Reconstruct path

                path.append(current)
                current = came_from[current]
            path.append(start)

                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return None  # No path found

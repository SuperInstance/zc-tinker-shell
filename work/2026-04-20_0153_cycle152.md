#!/usr/bin/env python3
"""

Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    """Maze environment for Deadband Protocol demonstration."""
    
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks: Set[Tuple[int, int]] = set()
        
        # Generate rocks (obstacles) - P0 negative space
        for x in range(width):
            for y in range(height):
                if (x, y) != self.start and (x, y) != self.goal:
                    if random.random() < rock_density:
                        self.rocks.add((x, y))
        
        # Ensure start and goal are not rocks
        if self.start in self.rocks:
            self.rocks.remove(self.start)
        if self.goal in self.rocks:
            self.rocks.remove(self.goal)
    
    def is_rock(self, pos: Tuple[int, int]) -> bool:
        """Check if position is a rock (P0 negative space)."""
        return pos in self.rocks
    
    def is_valid(self, pos: Tuple[int, int]) -> bool:
        """Check if position is within bounds."""
        x, y = pos
        return 0 <= x < self.width and 0 <= y < self.height
    
    def get_neighbors(self, pos: Tuple[int, int]) -> List[Tuple[int, int]]:
        """Get valid neighboring positions (up, down, left, right)."""
        x, y = pos
        neighbors = [(x+1, y), (x-1, y), (x, y+1), (x, y-1)]
        return [n for n in neighbors if self.is_valid(n)]
    
    def visualize(self, 
                  path: List[Tuple[int, int]] = None,
                  deadband: Set[Tuple[int, int]] = None) -> str:
        """Create ASCII visualization of maze state."""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif self.is_rock(pos):
                    row.append('#')
                elif deadband and pos in deadband:
                    row.append('~')
                elif path and pos in path:
                    row.append('*')
                else:
                    row.append('.')
            grid.append(' '.join(row))
        return '\n'.join(grid)

class DeadbandAgent:
    """Agent implementing Deadband Protocol P0→P1→P2."""
    
    def __init__(self, maze: DeadbandMaze):
        self.maze = maze
        self.visited = set()
        self.deadband = set()  # P1 safe channels
        
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify rocks (what NOT to do)."""
        print("P0: Mapping negative space (rocks)...")
        rocks = self.maze.rocks.copy()
        print(f"  Found {len(rocks)} rocks")
        return rocks
    
    def p1_find_safe_channels(self) -> Set[Tuple[int, int]]:
        """P1: Find safe channels (deadbands) using BFS from start."""
        print("P1: Finding safe channels (deadbands)...")
        
        queue = [self.maze.start]
        visited = {self.maze.start}
        safe_cells = {self.maze.start}
        
        while queue:
            current = queue.pop(0)
            
            for neighbor in self.maze.get_neighbors(current):
                if neighbor not in visited and not self.maze.is_rock(neighbor):
                    visited.add(neighbor)
                    safe_cells.add(neighbor)
                    queue.append(neighbor)
        
        # Check if goal is reachable
        if self.maze.goal not in safe_cells:
            print("  WARNING: Goal not reachable from start!")
        
        print(f"  Found {len(safe_cells)} safe cells")
        self.deadband = safe_cells
        return safe_cells
    
    def p2_optimize_in_channel(self) -> List[Tuple[int, int]]:
        """P2: Find shortest path within safe channels."""
        print("P2: Optimizing within safe channels...")
        
        if self.maze.goal not in self.deadband:
            print("  Goal not in safe channels - cannot proceed!")
            return []
        
        # BFS for shortest path within deadband
        queue = [(self.maze.start, [self.maze.start])]
        visited = {self.maze.start}
        
        while queue:
            current, path = queue.pop(0)
            
            if current == self.maze.goal:
                print(f"  Found path of length {len(path)}")
                return path
            
            for neighbor in self.maze.get_neighbors(current):
                if (neighbor not in visited and 
                    neighbor in self.deadband and 
                    not self.maze.is_rock(neighbor)):
                    visited.add(neighbor)
                    queue.append((neighbor, path + [neighbor]))
        
        print("  No path found within safe channels")
        return []
    
    def run_protocol(self) -> Tuple[bool, int]:
        """Execute full P0→P1→P2 protocol."""
        print("\n" + "="*50)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*50)
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        
        # P1: Find safe channels
        deadband = self.p1_find_safe_channels()
        
        # Visualize after P1
        print("\nMaze after P1 (Safe channels marked with ~):")
        print(self.maze.visualize(deadband=deadband))
        
        # P2: Optimize within channel
        path = self.p2_optimize_in_channel()
        
        if path:
            # Visualize final path
            print("\nFinal path (P2 optimization):")
            print(self.maze.visualize(path=path, deadband=deadband))
            
            success = True
            steps = len(path) - 1  # Subtract start position
            print(f"\nSUCCESS! Path length: {steps} steps")
            print(f"Optimal (from fleet data): 21 steps")
        else:
            success = False
            steps = 0
            print("\nFAILURE! No valid path found.")
        
        return success, steps

def run_simulation(num_runs: int = 50) -> dict:

**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on reading `fleet-knowledge/narrow-games/constraint_sim_final.json` and `fleet-knowledge/docs/DEADBAND-PROTOCOL.md`, I will implement a concrete simulation.

**Plan:**
1. **P0 (Map Negative Space):** Generate a random 10x10 maze with walls ('#') as rocks.
2. **P1 (Identify Safe Channels):** Use BFS from start to find all reachable cells not blocked by walls.
3. **P2 (Optimize Within Channel):** Use A* search constrained to the safe channel to find shortest path.
4. **Visualization:** Print each step and final path in ASCII.

**Output:** A single, runnable Python file `deadband_maze_demo.py`.

## File: `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Simulates the three-phase protocol in a random ASCII maze.
"""

import random
from collections import deque
import heapq

def generate_maze(width=10, height=10, wall_prob=0.25):
    """Generate a random maze with start (S), goal (G), and walls (#)."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place walls (rocks)
    for y in range(height):
        for x in range(width):
            if random.random() < wall_prob:
                maze[y][x] = '#'
    
    # Ensure start and goal are open
    start = (0, 0)
    goal = (width-1, height-1)
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    
    return maze, start, goal

def print_maze(maze, path=None):
    """Print maze with optional path marked."""
    for y, row in enumerate(maze):
        line = []
        for x, cell in enumerate(row):
            if path and (x, y) in path and cell not in ('S', 'G'):
                line.append('*')
            else:
                line.append(cell)
        print(''.join(line))
    print()

def p0_map_negative_space(maze):
    """P0: Identify all wall cells (rocks)."""
    rocks = []
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.append((x, y))
    return set(rocks)

def p1_find_safe_channels(maze, start, rocks):
    """P1: BFS to find all reachable cells not blocked by rocks."""
    width, height = len(maze[0]), len(maze)
    visited = set()
    queue = deque([start])
    safe = set()
    
    while queue:
        x, y = queue.popleft()
        if (x, y) in visited:
            continue
        visited.add((x, y))
        if (x, y) in rocks:
            continue
        safe.add((x, y))
        
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                queue.append((nx, ny))
    
    return safe

def p2_optimize_within_channel(maze, start, goal, safe):
    """P2: A* search constrained to safe cells."""
    width, height = len(maze[0]), len(maze)
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            return list(reversed(path))
        
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = current[0] + dx, current[1] + dy
            neighbor = (nx, ny)
            if not (0 <= nx < width and 0 <= ny < height):
                continue
            if neighbor not in safe:
                continue  # Constrained to safe channel
            if maze[ny][nx] == '#':
                continue
            
            tentative_g = g_score[current] + 1
            if neighbor not in g_score or tentative_g < g_score[neighbor]:
                came_from[neighbor] = current
                g_score[neighbor] = tentative_g
                f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return None  # No path found within safe channel

def run_deadband_demo():
    """Main demonstration sequence."""
    print("=== Deadband Protocol Maze Demo ===\n")
    
    # Generate maze
    maze, start, goal = generate_maze(10, 10, wall_prob=0.25)
    print("Initial Maze (S=start, G=goal, #=rock):")
    print_maze(maze)
    
    # P0: Map negative space
    rocks = p0_map_negative_space(maze)
    print(f"P0 — Rocks identified: {len(rocks)} cells")
    
    # P1: Find safe channels
    safe = p1_find_safe_channels(maze, start, rocks)
    print(f"P1 — Safe channel cells: {len(safe)} reachable cells")
    
    # P2: Optimize within channel
    path = p2_optimize_within_channel(maze, start, goal, safe)
    
    if path:
        print("P2 — Optimal path found (length = {}):".format(len(path)-1))
        print_maze(maze, set(path))
        print("Path coordinates:", path)
    else:
        print("P2 — No path exists within safe channel.")
    
    # Violation demo: Greedy search ignoring P0/P1 (will hit rocks)
    print("\n--- VIOLATION: Greedy search (P2-only) ---")
    # Simple BFS ignoring rocks (except as blocked)
    width, height = len(maze[0]), len(maze)
    queue = deque([(start, [start])])
    visited = set([start])
    greedy_path = None
    
    while queue:
        (x, y), p = queue.popleft()
        if (x, y) == goal:
            greedy_path = p
            break
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if maze[ny][nx] != '#' and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), p + [(nx, ny)]))
    
    if greedy_path:
        print("Greedy path found (may intersect rocks):")
        print_maze(maze, set(greedy_path))
    else:
        print("Greedy search failed.")

if __name__ == "__main__":
    run_deadband_demo()
```

## Summary

This script implements the Deadband Protocol exactly as described:

1. **P0:** Identifies all wall cells (`#`) as rocks (negative space).
2. **P1:** Uses BFS to map all reachable cells that are not rocks (safe channels).
3. **P2:** Runs A* search constrained *only* to the safe channel cells.

The violation demo shows a greedy BFS that ignores P0/P1 — it may still find a path, but it doesn't systematically avoid rocks first.

**Key insight from the fleet data:** The `constraint_sim_final.json` shows `deadband_success: 50` vs `constrained_success: 0`. This demo illustrates why: skipping P0/P1 leads to failure as complexity grows.

**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Implementing based on fleet knowledge. Using `constraint_sim_final.json` as reference for success rates and the DEADBAND-PROTOCOL.md for the algorithm.

**Action:** I will write a Python script that simulates a 20x20 maze with rocks (obstacles), an agent, and a goal. The script will visually demonstrate three strategies:
1. **Unconstrained (P2-only)**: Greedy pathfinding toward the goal, ignoring rocks.
2. **Constrained (P0-only)**: Only avoids rocks but doesn't move toward goal (stuck).
3. **Deadband Protocol (P0→P1→P2)**: First map rocks, then find safe channels, then optimize path within safe channels.

The output will be ASCII mazes printed to console, with statistics matching the fleet's simulation data trend.

**File:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Demonstrates P0→P1→P2 in a 20x20 maze with rocks.
"""

import random
from collections import deque

# Constants from fleet simulation
MAZE_SIZE = 20
ROCK_COUNT = 64  # matches "rocks": 64 in JSON
GOAL = (19, 19)
START = (0, 0)

# Symbols for visualization
AGENT = 'A'
GOAL_SYM = 'G'
ROCK = '█'
PATH = '.'
SAFE = ' '
VISITED = '·'

def generate_maze():
    """Create a 20x20 maze with random rocks."""
    maze = [[SAFE for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks_set = set()
    while len(rocks_set) < ROCK_COUNT:
        r = random.randint(0, MAZE_SIZE-1)
        c = random.randint(0, MAZE_SIZE-1)
        if (r, c) != START and (r, c) != GOAL:
            rocks_set.add((r, c))
    for r, c in rocks_set:
        maze[r][c] = ROCK
    maze[START[0]][START[1]] = AGENT
    maze[GOAL[0]][GOAL[1]] = GOAL_SYM
    return maze, rocks_set

def print_maze(maze, title):
    """Print ASCII maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(''.join(row))
    print()

def unconstrained_p2_only(maze, rocks):
    """Greedy move toward goal, ignoring rocks -> hits rocks."""
    path = []
    r, c = START
    maze_copy = [row[:] for row in maze]
    
    while (r, c) != GOAL:
        path.append((r, c))
        if (r, c) in rocks:
            maze_copy[r][c] = 'X'  # collision
            break
        # greedy: move down or right toward goal
        if r < GOAL[0]:
            r += 1
        elif c < GOAL[1]:
            c += 1
        else:
            break
        if (r, c) == GOAL:
            path.append(GOAL)
            break
    
    # Visualize path
    for pr, pc in path:
        if maze_copy[pr][pc] not in (AGENT, GOAL_SYM, ROCK, 'X'):
            maze_copy[pr][pc] = PATH
    print_maze(maze_copy, "Unconstrained (P2-only) - Hits Rocks")
    return len(path), (r, c) == GOAL

def constrained_p0_only(maze, rocks):
    """Only avoid rocks, no goal seeking -> stuck."""
    maze_copy = [row[:] for row in maze]
    # Mark all safe cells around start
    queue = deque([START])
    visited = set()
    while queue:
        r, c = queue.popleft()
        if (r, c) in visited:
            continue
        visited.add((r, c))
        if maze_copy[r][c] not in (AGENT, GOAL_SYM, ROCK):
            maze_copy[r][c] = VISITED
        # Explore neighbors that are not rocks
        for dr, dc in [(1,0),(-1,0),(0,1),(0,-1)]:
            nr, nc = r+dr, c+dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if (nr, nc) not in rocks and (nr, nc) not in visited:
                    queue.append((nr, nc))
    print_maze(maze_copy, "Constrained (P0-only) - Safe Map, No Goal")
    return len(visited), False  # never reaches goal

def deadband_p0_p1_p2(maze, rocks):
    """Full deadband protocol: map rocks, find safe channels, then optimal path."""
    # P0: Map negative space (rocks already known)
    # P1: Identify safe channels (BFS through non-rock cells)
    queue = deque([START])
    visited = {START: None}
    found = False
    
    while queue:
        r, c = queue.popleft()
        if (r, c) == GOAL:
            found = True
            break
        for dr, dc in [(1,0),(-1,0),(0,1),(0,-1)]:
            nr, nc = r+dr, c+dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if (nr, nc) not in rocks and (nr, nc) not in visited:
                    visited[(nr, nc)] = (r, c)
                    queue.append((nr, nc))
    
    # P2: Optimize within safe channel (reconstruct path)
    if found:
        path = []
        curr = GOAL
        while curr != START:
            path.append(curr)
            curr = visited[curr]
        path.append(START)
        path.reverse()
        
        # Visualize
        maze_copy = [row[:] for row in maze]
        for r, c in path[1:-1]:  # exclude start and goal
            maze_copy[r][c] = PATH
        print_maze(maze_copy, "Deadband Protocol (P0→P1→P2) - Optimal Safe Path")
        return len(path), True
    else:
        maze_copy = [row[:] for row in maze]
        print_maze(maze_copy, "Deadband Protocol - No Safe Channel Found")
        return 0, False

def main():
    print("Deadband Protocol Maze Demonstration")
    print("Based on fleet simulation: 20x20 maze, 64 rocks")
    print("=" * 40)
    
    # Generate consistent maze for comparison
    random.seed(42)  # reproducible for demo
    maze, rocks = generate_maze()
    print_maze(maze, "Initial Maze (A=Agent, G=Goal, █=Rock)")
    
    # Run three strategies
    steps1, success1 = unconstrained_p2_only(maze, rocks)
    steps2, success2 = constrained_p0_only(maze, rocks)
    steps3, success3 = deadband_p0_p1_p2(maze, rocks)
    
    # Summary
    print("\n=== Results Summary ===")
    print(f"Unconstrained (P2-only):  {steps1} steps, Success: {success1}")
    print(f"Constrained (P0-only):    {steps2} safe cells mapped, Success: {success2}")
    print(f"Deadband (P0→P1→P2):      {steps3} steps, Success: {success3}")
    print("\nNote: Single run for demonstration. Fleet data (50 runs) shows:")
    print("  - Unconstrained success: 0/50 (hits rocks)")
    print("  - Deadband success: 50/50 at optimal speed")
    print("  (See constraint_sim_final.json)")

if __name__ == "__main__":
    main()
```

**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on the simulation data in `constraint_sim_final.json` and the formal protocol in `DEADBAND-PROTOCOL.md`, I will implement a concrete, runnable demonstration.

**Approach (P0→P1→P2):**
- **P0 (Map negative space):** Read the maze file, identify walls/rocks.
- **P1 (Find safe channels):** Compute all traversable paths that avoid rocks.
- **P2 (Optimize within channel):** Select shortest safe path using BFS.

**Output:** A Python script that loads a maze, applies the protocol step‑by‑step, and prints the maze with the discovered safe path.

## File: `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0 → P1 → P2 demonstration with ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
and fleet-knowledge/docs/DEADBAND-PROTOCOL.md
"""

import sys
from collections import deque

def load_maze(filename):
    """Load maze from file, return list of strings."""
    with open(filename, 'r') as f:
        return [line.rstrip('\n') for line in f]

def print_maze(maze, path=None):
    """Print maze, optionally overlay path with '*'."""
    for y, row in enumerate(maze):
        line = []
        for x, ch in enumerate(row):
            if path and (x, y) in path:
                line.append('*')
            else:
                line.append(ch)
        print(''.join(line))

def p0_map_negative_space(maze):
    """P0: Identify all rock/wall positions."""
    rocks = set()
    for y, row in enumerate(maze):
        for x, ch in enumerate(row):
            if ch == '#':  # rock/wall
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(maze, rocks, start, goal):
    """P1: Find all traversable cells (safe channels)."""
    rows = len(maze)
    cols = len(maze[0])
    safe = set()
    stack = [start]
    visited = set()
    while stack:
        x, y = stack.pop()
        if (x, y) in visited:
            continue
        visited.add((x, y))
        if (x, y) in rocks:
            continue
        safe.add((x, y))
        if (x, y) == goal:
            continue  # keep exploring all reachable safe cells
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < cols and 0 <= ny < rows:
                stack.append((nx, ny))
    return safe

def p2_optimize_within_channel(maze, safe, start, goal):
    """P2: Find shortest path within safe channels using BFS."""
    rows = len(maze)
    cols = len(maze[0])
    queue = deque([start])
    parent = {start: None}
    while queue:
        x, y = queue.popleft()
        if (x, y) == goal:
            # reconstruct path
            path = []
            while (x, y) != start:
                path.append((x, y))
                x, y = parent[(x, y)]
            path.append(start)
            path.reverse()
            return path
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < cols and 0 <= ny < rows:
                if (nx, ny) in safe and (nx, ny) not in parent:
                    parent[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    return []  # no safe path

def main(maze_file):
    print("=== Deadband Protocol Maze Demo ===\n")
    maze = load_maze(maze_file)
    print("Original Maze:")
    print_maze(maze)
    print()

    # Define start (S) and goal (G) from typical maze format
    start = None
    goal = None
    for y, row in enumerate(maze):
        for x, ch in enumerate(row):
            if ch == 'S':
                start = (x, y)
            elif ch == 'G':
                goal = (x, y)
    if not start or not goal:
        print("Error: Maze must contain 'S' (start) and 'G' (goal).")
        sys.exit(1)

    # P0: Map negative space
    rocks = p0_map_negative_space(maze)
    print(f"P0: Mapped {len(rocks)} rock positions (negative space).")

    # P1: Find safe channels
    safe = p1_find_safe_channels(maze, rocks, start, goal)
    print(f"P1: Found {len(safe)} safe cells (channels).")

    # P2: Optimize within channel
    path = p2_optimize_within_channel(maze, safe, start, goal)
    if path:
        print(f"P2: Optimal safe path length = {len(path)-1} steps.")
        print("\nMaze with Safe Path (P2 result):")
        print_maze(maze, set(path))
    else:
        print("P2: No safe path to goal exists.")

    # Demonstrate violation: greedy (P2-only) approach
    print("\n--- Violation Example: Greedy (P2-only) ---")
    # Simple greedy that moves toward goal ignoring rocks
    greedy_path = []
    x, y = start
    for _ in range(100):  # limit steps
        greedy_path.append((x, y))
        if (x, y) == goal:
            break
        # move toward goal (naive)
        if x < goal[0]:
            x += 1
        elif x > goal[0]:
            x -= 1
        elif y < goal[1]:
            y += 1
        elif y > goal[1]:
            y -= 1
        else:
            break
        if (x, y) in rocks:
            greedy_path.append((x, y))
            break
    if goal in greedy_path:
        print("Greedy reached goal (lucky in simple maze).")
    else:
        print("Greedy hit a rock at", greedy_path[-1])
    print("Greedy path length attempted:", len(greedy_path)-1)

if __name__ == '__main__':
    if len(sys.argv) != 2:
        print("Usage: python deadband_maze_demo.py <maze_file.txt>")
        print("Example maze format: '#' = rock, 'S' = start, 'G' = goal, '.' = open")
        sys.exit(1)
    main(sys.argv[1])
```

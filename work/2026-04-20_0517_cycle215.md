**Approach:**
1. **P0 (Map Negative Space):** Generate a 20x20 maze with 64 rocks (obstacles) placed randomly.
2. **P1 (Identify Safe Channels):** Use BFS to find all reachable cells from the start that avoid rocks.
3. **P2 (Optimize Within Channel):** Use A* search constrained to the safe channel to find the shortest path to the goal.
4. **Visualization:** Print ASCII maze showing rocks, safe channel, and optimal path.

**Output:** A Python script `deadband_maze_demo.py` that runs the protocol and prints results.

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 in a 20x20 maze with 64 rocks.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque
import heapq

# Constants from simulation data
MAZE_SIZE = 20
NUM_ROCKS = 64
START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

def generate_maze():
    """P0: Map negative space - create maze with rocks."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    
    # Place rocks randomly (avoid start and goal)
    rocks_placed = 0
    while rocks_placed < NUM_ROCKS:
        x = random.randint(0, MAZE_SIZE - 1)
        y = random.randint(0, MAZE_SIZE - 1)
        if (x, y) != START and (x, y) != GOAL and maze[x][y] == '.':
            maze[x][y] = '#'
            rocks_placed += 1
    
    return maze

def find_safe_channel(maze):
    """P1: Identify safe channels - BFS to find all reachable cells avoiding rocks."""
    visited = [[False for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    safe_cells = []
    queue = deque([START])
    
    while queue:
        x, y = queue.popleft()
        if visited[x][y]:
            continue
        
        visited[x][y] = True
        safe_cells.append((x, y))
        
        # Check neighbors (up, down, left, right)
        for dx, dy in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
            nx, ny = x + dx, y + dy
            if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                not visited[nx][ny] and maze[nx][ny] != '#'):
                queue.append((nx, ny))
    
    return visited, safe_cells

def optimize_within_channel(maze, safe_map):
    """P2: Optimize within channel - A* search constrained to safe cells."""
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    # Priority queue: (f_score, g_score, position, path)
    open_set = []
    heapq.heappush(open_set, (heuristic(START, GOAL), 0, START, [START]))
    
    g_scores = {START: 0}
    
    while open_set:
        f_score, g_score, current, path = heapq.heappop(open_set)
        
        if current == GOAL:
            return path
        
        x, y = current
        for dx, dy in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
            nx, ny = x + dx, y + dy
            neighbor = (nx, ny)
            
            # Constraint: must be in safe channel
            if not (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE):
                continue
            if not safe_map[nx][ny]:
                continue
            
            new_g = g_score + 1
            if neighbor not in g_scores or new_g < g_scores[neighbor]:
                g_scores[neighbor] = new_g
                f_score = new_g + heuristic(neighbor, GOAL)
                heapq.heappush(open_set, (f_score, new_g, neighbor, path + [neighbor]))
    
    return None  # No path found within safe channel

def visualize_maze(maze, safe_map=None, path=None):
    """Visualize maze with optional safe channel and path overlay."""
    display = [row[:] for row in maze]
    
    # Mark safe channel cells (if provided)
    if safe_map:
        for x in range(MAZE_SIZE):
            for y in range(MAZE_SIZE):
                if safe_map[x][y] and display[x][y] == '.':
                    display[x][y] = ' '
    
    # Mark path (if provided)
    if path:
        for x, y in path[1:-1]:  # Skip start and goal
            if display[x][y] == ' ' or display[x][y] == '.':
                display[x][y] = '*'
    
    # Mark start and goal
    sx, sy = START
    gx, gy = GOAL
    display[sx][sy] = 'S'
    display[gx][gy] = 'G'
    
    # Print with borders
    print("+" + "-" * MAZE_SIZE + "+")
    for row in display:
        print("|" + "".join(row) + "|")
    print("+" + "-" * MAZE_SIZE + "+")

def run_deadband_protocol():

    print("=" * 60)
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE} with {NUM_ROCKS} rocks")
    print("=" * 60)

    print("\n[P0] MAP NEGATIVE SPACE")

    safe_map, safe_cells = find_safe_channel(maze)
    print(f"Safe cells reachable from start: {len(safe_cells)}")
    print("Maze with safe channel (space = safe, # = rock):")
    visualize_maze(maze, safe_map)

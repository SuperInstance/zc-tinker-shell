#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set
from collections import deque

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # ~16% density, matches 64 rocks in 20x20
OPTIMAL_PATH_LENGTH = 21  # From constraint_sim_final.json

class DeadbandMaze:
    def __init__(self, size: int = MAZE_SIZE):
        self.size = size
        self.start = (0, 0)
        self.goal = (size-1, size-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
        # Generate random rocks (P0 negative space)
        self._generate_rocks()
        
    def _generate_rocks(self):
        """P0: Map the negative space (rocks to avoid)"""
        all_cells = [(x, y) for x in range(self.size) for y in range(self.size)]
        all_cells.remove(self.start)
        all_cells.remove(self.goal)
        
        self.rocks = set(random.sample(all_cells, ROCK_COUNT))
    
    def find_safe_channels(self):
        """P1: Identify all safe channels (paths avoiding rocks)"""
        # BFS to find all possible safe paths
        visited = set()
        queue = deque([(self.start, [self.start])])
        safe_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            
            # Explore neighbors (up, down, left, right)
            for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.size and 0 <= ny < self.size:
                    if (nx, ny) not in self.rocks:
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.safe_channels = safe_paths
        return len(safe_paths)
    
    def optimize_within_channels(self):
        """P2: Find shortest path within safe channels"""
        if not self.safe_channels:
            return None
        
        self.optimal_path = min(self.safe_channels, key=len)
        return len(self.optimal_path)
    
    def greedy_optimize(self):
        """Unconstrained P2-only approach (for comparison)"""
        # Greedy BFS that ignores rock mapping
        visited = set()
        queue = deque([(self.start, [self.start])])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                return path
            
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            
            # Move toward goal (greedy heuristic)
            neighbors = []
            for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.size and 0 <= ny < self.size:
                    # Doesn't check for rocks! This is the violation.
                    neighbors.append(((nx, ny), path + [(nx, ny)]))
            
            # Sort by distance to goal (greedy)
            neighbors.sort(key=lambda p: abs(p[0][0]-self.goal[0]) + abs(p[0][1]-self.goal[1]))
            queue.extendleft(neighbors)  # DFS-like greedy
        
        return None
    
    def visualize(self, show_path: bool = False):
        """ASCII visualization of the maze"""
        grid = [['·' for _ in range(self.size)] for _ in range(self.size)]
        
        # Mark rocks
        for x, y in self.rocks:
            grid[y][x] = '█'
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        grid[sy][sx] = 'S'
        grid[gy][gx] = 'G'
        
        # Mark optimal path if requested
        if show_path and self.optimal_path:
            for x, y in self.optimal_path[1:-1]:  # Skip start and goal
                if grid[y][x] == '·':
                    grid[y][x] = '○'
        
        # Print with borders
        print('+' + '─' * (self.size * 2 - 1) + '+')
        for row in grid:
            print('|' + ' '.join(row) + '|')
        print('+' + '─' * (self.size * 2 - 1) + '+')
        
        # Legend
        print("Legend: S=Start, G=Goal, █=Rock, ·=Safe, ○=Optimal Path")

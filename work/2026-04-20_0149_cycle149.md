#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 149)
Simulates the 20×20 maze experiment from constraint_sim_final.json.
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize within channels).
"""

import random
from typing import List, Tuple, Set
from collections import deque

# Simulation parameters from fleet-knowledge
MAZE_SIZE = 20
ROCK_COUNT = 64  # From constraint_sim_final.json: "rocks": 64
START = (0, 0)
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

# Symbols for visualization
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_SYM = 'G'
PATH = '○'
SAFE = ' '  # Empty space in safe channel

class DeadbandMaze:
    def __init__(self):
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[Set[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
        # Place rocks (P0 negative space)
        self._place_rocks()
        
        # Mark start and goal
        self.grid[START[0]][START[1]] = AGENT
        self.grid[GOAL[0]][GOAL[1]] = GOAL_SYM
    
    def _place_rocks(self):
        """P0: Map negative space (rocks)"""
        positions = [(r, c) for r in range(MAZE_SIZE) for c in range(MAZE_SIZE)]
        positions.remove(START)
        positions.remove(GOAL)
        
        self.rocks = set(random.sample(positions, ROCK_COUNT))
        for r, c in self.rocks:
            self.grid[r][c] = ROCK
    
    def find_safe_channels(self):
        """P1: Find safe channels (BFS flood fill avoiding rocks)"""
        visited = set(self.rocks)
        directions = [(0,1), (1,0), (0,-1), (-1,0)]
        
        for r in range(MAZE_SIZE):
            for c in range(MAZE_SIZE):
                if (r, c) not in visited:
                    # New channel
                    channel = set()
                    queue = deque([(r, c)])
                    
                    while queue:
                        cr, cc = queue.popleft()
                        if (cr, cc) in visited:
                            continue
                        
                        visited.add((cr, cc))
                        channel.add((cr, cc))
                        
                        for dr, dc in directions:
                            nr, nc = cr + dr, cc + dc
                            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                                if (nr, nc) not in visited:
                                    queue.append((nr, nc))
                    
                    if channel:
                        self.safe_channels.append(channel)
        
        # Filter channels containing start and goal
        self.safe_channels = [ch for ch in self.safe_channels 
                             if START in ch and GOAL in ch]
    
    def optimize_within_channel(self):
        """P2: Find shortest path within safe channels (constrained optimization)"""
        if not self.safe_channels:
            return []
        
        # Use first safe channel containing both start and goal
        channel = self.safe_channels[0]
        
        # BFS for shortest path
        queue = deque([(START, [START])])
        visited = set([START])
        directions = [(0,1), (1,0), (0,-1), (-1,0)]
        
        while queue:
            (r, c), path = queue.popleft()
            
            if (r, c) == GOAL:
                self.optimal_path = path
                return path
            
            for dr, dc in directions:
                nr, nc = r + dr, c + dc
                if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                    if (nr, nc) in channel and (nr, nc) not in visited:
                        visited.add((nr, nc))
                        queue.append(((nr, nc), path + [(nr, nc)]))
        
        return []
    
    def run_unconstrained_greedy(self):
        """Simulate P2-only (greedy) approach that fails"""
        path = [START]
        current = START
        directions = [(0,1), (1,0)]  # Only move right/down (greedy toward goal)
        
        while current != GOAL:
            # Greedy: pick direction that reduces Manhattan distance most
            best_dir = None
            best_dist = float('inf')
            
            for dr, dc in directions:
                nr, nc = current[0] + dr, current[1] + dc
                if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                    dist = abs(nr - GOAL[0]) + abs(nc - GOAL[1])
                    if dist < best_dist:
                        best_dist = dist
                        best_dir = (dr, dc)
            
            if best_dir:
                nr, nc = current[0] + best_dir[0], current[1] + best_dir[1]
                current = (nr, nc)
                path.append(current)
                
                # Check if hit rock
                if current in self.rocks:
                    return path, False
            else:
                break
        
        return path, current == GOAL
    
    def visualize(self, show_safe=False, show_path=False):
        """Print ASCII maze"""
        display = [row[:] for row in self.grid]
        
        if show_safe and self.safe_channels:
            for channel in self.safe_channels:
                for r, c in channel:
                    if display[r][c] == EMPTY:
                        display[r][c] = SAFE
        
        if show_path and self.optimal_path:
            for r, c in self.optimal_path[1:-1]:  # Skip start and goal
                if display[r][c] in [EMPTY, SAFE]:
                    display[r][c] = PATH
        
        print("+" + "-" * MAZE_SIZE + "+")
        for row in display:
            print("|" + "".join(row) + "|")
        print("+" + "-" * MAZE_SIZE + "+")
    
    def print_stats(self):
        """Print simulation statistics"""
        print(f"\n=== Deadband Protocol Maze Demo ===")
        print(f"Maze: {MAZE_SIZE}×{MAZE_SIZE}, Rocks: {ROCK_COUNT}")
        print(f"Start: {START}, Goal: {GOAL}")
        print(f"\nP0: Mapped {len(self.rocks)} rocks (negative space)")
        print(f"P1: Found {len(self.safe_channels)} safe channel(s) containing both start and goal")
        
        if self.safe_channels:
            channel_size = len(self.safe_channels[0])
            print(f"    Primary channel size: {channel_size} cells ({channel_size/(MAZE_SIZE*MAZE_SIZE):.1%} of maze)")
        
        if self.optimal_path:
            print(f"P2: Optimal path length: {len(self.optimal_path)-1} steps")
        
        # Compare with unconstrained greedy
        greedy_path, greedy_success = self.run_unconstrained_greedy()
        print(f"\n--- Comparison ---")
        print(f"Deadband Protocol (P0→P1→P2): {'SUCCESS' if self.optimal_path else 'FAILED'}")
        print(f"Greedy (P2-only): {'SUCCESS' if greedy_success else 'FAILED'}")
        
        if not greedy_success and greedy_path:
            rock_hit = greedy_path[-1]
            print(f"    Greedy hit rock at {rock_hit} after {len(greedy_path)-1} steps")

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 with ASCII visualization.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density and (x, y) not in [self.start, self.goal]:
                    self.rocks.add((x, y))
        
        # Initialize state
        self.negative_space_mapped = False
        self.safe_channels = None
        self.optimal_path = None
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify all states where failure probability > threshold (rocks)."""
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Found {len(self.rocks)} rocks (failure states)")
        self.negative_space_mapped = True
        return self.rocks
    
    def p1_find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Find all paths that avoid rocks (safe channels)."""
        if not self.negative_space_mapped:
            raise ValueError("Must complete P0 before P1")
        
        print("\n=== P1: FIND SAFE CHANNELS ===")
        
        # Simple BFS to find all safe paths
        from collections import deque
        
        safe_paths = []
        queue = deque([([self.start], {self.start})])
        
        while queue:
            path, visited = queue.popleft()
            x, y = path[-1]
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            # Explore neighbors (up, down, left, right)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        new_path = path + [(nx, ny)]
                        new_visited = visited.copy()
                        new_visited.add((nx, ny))
                        queue.append((new_path, new_visited))
        
        self.safe_channels = safe_paths
        print(f"Found {len(safe_paths)} safe channels")
        return safe_paths
    
    def p2_optimize_within_channels(self) -> List[Tuple[int, int]]:
        """P2: Find shortest path within safe channels."""
        if not self.safe_channels:
            raise ValueError("Must complete P1 before P2")
        
        print("\n=== P2: OPTIMIZE WITHIN CHANNELS ===")
        
        # Find shortest safe path
        shortest = min(self.safe_channels, key=len)
        self.optimal_path = shortest
        
        print(f"Optimal path length: {len(shortest)-1} steps")
        print(f"Compare to simulation optimal: 21 steps")
        return shortest
    
    def visualize(self, show_path: bool = False, show_rocks: bool = True):
        """Display maze with ASCII art."""
        grid = [['·' for _ in range(self.width)] for _ in range(self.height)]
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        grid[sy][sx] = 'S'
        grid[gy][gx] = 'G'
        
        # Mark rocks
        if show_rocks:
            for x, y in self.rocks:
                grid[y][x] = '█'
        
        # Mark path if requested
        if show_path and self.optimal_path:
            for x, y in self.optimal_path[1:-1]:  # Exclude start and goal
                if grid[y][x] == '·':
                    grid[y][x] = '●'
        
        # Print grid
        print("\n" + "=" * (self.width * 2 + 3))
        for row in grid:
            print("│ " + " ".join(row) + " │")
        print("=" * (self.width * 2 + 3))
        
        # Legend
        legend = []
        if show_rocks:
            legend.append("█ = Rock (P0 negative space)")
        if show_path and self.optimal_path:
            legend.append("● = Safe path (P2 optimal)")
        legend.append("S = Start, G = Goal")
        legend.append("· = Navigable space")
        
        print("\n".join(legend))
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2."""
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print(f"Maze: {self.width}x{self.height}, Start: {self.start}, Goal: {self.goal}")
        
        # Step 1: Visualize initial state
        print("\nInitial Maze (rocks hidden):")
        self.visualize(show_rocks=False)
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        print("\nMaze with rocks revealed:")
        self.visualize(show_rocks=True)
        
        # P1: Find safe channels
        safe_paths = self.p1_find_safe_channels()
        
        if not safe_paths:
            print("NO SAFE CHANNELS FOUND - Protocol violation would lead to failure")
            return
        
        # P2: Optimize within channels
        optimal = self.p2_optimize_within_channels()
        
        # Final visualization
        print("\nFinal Result - Optimal Path Through Safe Channels:")
        self.visualize(show_path=True, show_rocks=True)
        
        # Protocol validation
        print("\n=== PROTOCOL VALIDATION ===")
        print(f"P0 completed: {self.negative_space_mapped}")
        print(f"P1 completed: {len(safe_paths) > 0}")
        print(f"P2 completed: {self.optimal_path is not None}")
        
        # Check for rocks in optimal path (should be none)
        if self.optimal_path:
            rock_collisions = [p for p in self.optimal_path if p in self.rocks]
            print(f"Rocks in optimal path: {len(rock_collisions)} (should be 0)")
        
        # Compare to unconstrained greedy approach (what happens if we skip P0+P1)
        print("\n=== WHAT IF WE SKIP P0+P1? ===")
        print("Greedy approach (P2-only) would:")
        print("1. Try to go straight to goal")
        print("2. Hit rocks with high probability")
        print("3. Fail to reach goal (0% success in simulation)")

def main():
    """Run the demonstration."""
    # Seed for reproducibility (matches simulation parameters)
    random.seed(42)
    
    # Create maze with parameters from constraint_sim_final.json

    maze = DeadbandMaze(width=20, height=20, rock_density=0.16)
    
    # Execute Deadband Protocol
    maze.run_protocol()

    print("\n" + "="*50)
    print("COMPARISON TO FLEET SIMULATION DATA:")
    print("="*50)

    print("  Rocks: 64")
    print("  Unconstrained success: 33%")
    print("  Deadband success: 50%")
    print("\nThis demo shows WHY deadband works:")
    print("1. P0 identifies 64 failure states (rocks)")
    print("2. P1 finds channels that avoid ALL rocks")
    print("3. P2 picks shortest path through safe channels")
    print("\n

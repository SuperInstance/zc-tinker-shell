## Deadband Protocol Visual Demo (ASCII Maze)

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates the strict P0→P1→P2 priority order.

P0: Map negative space (rocks) — what NOT to do.
P1: Identify safe channels (deadbands) — where you CAN be.
P2: Optimize within safe channels — best path.

Based on fleet-knowledge/narrow-games/constraint_sim_final.json
and DEADBAND-PROTOCOL.md.
"""

import random
import sys
import time

# Constants from the simulation data
MAZE_SIZE = 20
ROCK_COUNT = 64  # Approximate density from the sim
START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

# Symbols
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_SYM = 'G'
PATH = '○'
SAFE = '░'  # Visual marker for safe channels (P1)

def generate_maze(size, rocks):
    """Generate a random maze with rocks as negative space."""
    maze = [[EMPTY for _ in range(size)] for _ in range(size)]
    maze[START[0]][START[1]] = AGENT
    maze[GOAL[0]][GOAL[1]] = GOAL_SYM
    
    placed = 0
    while placed < rocks:
        x = random.randint(0, size - 1)
        y = random.randint(0, size - 1)
        if (x, y) != START and (x, y) != GOAL and maze[x][y] == EMPTY:
            maze[x][y] = ROCK
            placed += 1
    return maze

def print_maze(maze, title="", delay=0.1):
    """Print the maze with optional title and delay."""
    if title:
        print(f"\n{title}")
        print("=" * (MAZE_SIZE * 2))
    for row in maze:
        print(' '.join(row))
    if delay:
        time.sleep(delay)

def p0_map_negative_space(maze):
    """P0: Identify all rock positions (negative space)."""
    rocks = []
    for x in range(MAZE_SIZE):
        for y in range(MAZE_SIZE):
            if maze[x][y] == ROCK:
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, rocks):
    """P1: Find all cells that are NOT rocks (safe channels)."""
    safe = []
    for x in range(MAZE_SIZE):
        for y in range(MAZE_SIZE):
            if (x, y) not in rocks and maze[x][y] != AGENT and maze[x][y] != GOAL_SYM:
                safe.append((x, y))
    return safe

def p2_optimize_within_channels(maze, safe):
    """P2: Greedy pathfinding constrained to safe cells only."""
    # Simple greedy move: right or down if safe, else stop.
    path = [START]
    x, y = START
    
    while (x, y) != GOAL:
        # Prefer moving right, then down (simple heuristic).
        moves = []
        if y + 1 < MAZE_SIZE and (x, y + 1) in safe:
            moves.append((x, y + 1))  # right
        if x + 1 < MAZE_SIZE and (x + 1, y) in safe:
            moves.append((x + 1, y))  # down
        if not moves:
            break  # No safe move — path incomplete.
        # Choose first safe move (greedy).
        x, y = moves[0]
        path.append((x, y))
        if (x, y) == GOAL:
            break
    return path

def run_demo():
    """Main demonstration sequence."""
    print("DEADBAND PROTOCOL DEMO — ASCII MAZE")
    print("Based on fleet simulation data (optimal=21, rocks=64)")
    print("=" * 50)
    
    # Generate maze
    maze = generate_maze(MAZE_SIZE, ROCK_COUNT)
    print_maze(maze, "Initial Maze (A=start, G=goal, █=rock)")
    
    # P0: Map negative space
    input("\nPress Enter to begin P0: Map negative space (rocks)...")
    rocks = p0_map_negative_space(maze)
    print(f"P0 complete: Mapped {len(rocks)} rock positions.")
    
    # P1: Identify safe channels
    input("\nPress Enter to begin P1: Identify safe channels...")
    safe = p1_find_safe_channels(maze, rocks)
    # Visualize safe channels
    maze_p1 = [row[:] for row in maze]
    for (x, y) in safe:
        if maze_p1[x][y] == EMPTY:
            maze_p1[x][y] = SAFE
    print_maze(maze_p1, "P1: Safe channels marked with ░")
    print(f"P1 complete: Found {len(safe)} safe cells.")
    
    # P2: Optimize within safe channels
    input("\nPress Enter to begin P2: Optimize within safe channels...")
    path = p2_optimize_within_channels(maze, safe)
    # Visualize path
    maze_p2 = [row[:] for row in maze]
    for (x, y) in path:
        if maze_p2[x][y] == EMPTY or maze_p2[x][y] == SAFE:
            maze_p2[x][y] = PATH
    print_maze(maze_p2, "P2: Optimal path within safe channels (○)")
    
    # Results
    print("\n" + "=" * 50)
    print("RESULTS:")
    print(f"  P0 rocks mapped: {len(rocks)}")
    print(f"  P1 safe cells: {len(safe)}")
    print(f"  P2 path length: {len(path)} steps")
    if path[-1] == GOAL:
        print("  SUCCESS: Reached goal using Deadband Protocol.")
    else:
        print("  FAILURE: Path incomplete — demonstrates constraint.")
    print("\nProtocol priority enforced: P0→P1→P2.")

if __name__ == "__main__":
    random.seed(42)  # Reproducible for training
    run_demo()
```

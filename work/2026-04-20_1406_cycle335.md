## Deadband Protocol Visual Demo (Python)

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates the strict P0→P1→P2 priority rule.

P0: Map negative space (rocks)
P1: Identify safe channels (deadbands)
P2: Optimize within safe channels (shortest path)

Based on fleet-knowledge/narrow-games/constraint_sim_final.json
and DEADBAND-PROTOCOL.md.
"""

import random
import sys
from collections import deque

# Constants from simulation data
MAZE_SIZE = 20
ROCK_COUNT = 64  # Approx 16% density, matching sim
START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

def generate_maze():
    """Generate random maze with rocks (obstacles)."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = set()
    
    # Place rocks randomly
    while len(rocks) < ROCK_COUNT:
        r = random.randint(0, MAZE_SIZE - 1)
        c = random.randint(0, MAZE_SIZE - 1)
        if (r, c) != START and (r, c) != GOAL:
            rocks.add((r, c))
            maze[r][c] = '#'
    
    maze[START[0]][START[1]] = 'S'
    maze[GOAL[0]][GOAL[1]] = 'G'
    return maze, rocks

def print_maze(maze, title="", path=None):
    """Print maze with optional path overlay."""
    print(f"\n{title}")
    for r in range(MAZE_SIZE):
        row = []
        for c in range(MAZE_SIZE):
            if path and (r, c) in path:
                row.append('*')
            else:
                row.append(maze[r][c])
        print(' '.join(row))

def p0_map_negative_space(maze, rocks):
    """P0: Identify all rocks (failure states)."""
    print("\n=== P0: MAP NEGATIVE SPACE ===")
    print(f"Identified {len(rocks)} rocks (failure states)")
    print("Agent knows what NOT to do.")
    return rocks

def p1_find_safe_channels(maze, rocks):
    """P1: Find all cells reachable without hitting rocks."""
    print("\n=== P1: FIND SAFE CHANNELS ===")
    
    # BFS to find all reachable cells
    visited = set()
    queue = deque([START])
    safe_cells = set()
    
    while queue:
        r, c = queue.popleft()
        if (r, c) in visited:
            continue
        visited.add((r, c))
        safe_cells.add((r, c))
        
        # Check neighbors
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            if (0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE and 
                (nr, nc) not in rocks and (nr, nc) not in visited):
                queue.append((nr, nc))
    
    print(f"Found {len(safe_cells)} safe cells (deadband)")
    print("Agent knows where it CAN be.")
    return safe_cells

def p2_optimize_within_channel(maze, safe_cells):
    """P2: Find shortest path within safe channels."""
    print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
    
    # BFS for shortest path
    parent = {START: None}
    queue = deque([START])
    found = False
    
    while queue and not found:
        r, c = queue.popleft()
        if (r, c) == GOAL:
            found = True
            break
        
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            if (0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE and 
                (nr, nc) in safe_cells and (nr, nc) not in parent):
                parent[(nr, nc)] = (r, c)
                queue.append((nr, nc))
    
    if not found:
        print("FAILURE: No path to goal within safe channels")
        return None
    
    # Reconstruct path
    path = []
    current = GOAL
    while current:
        path.append(current)
        current = parent[current]
    path.reverse()
    
    print(f"Found optimal path of length {len(path)-1} steps")
    print("Agent selects best path through safe water.")
    return set(path)

def greedy_unconstrained(maze):
    """Greedy P2-only approach (violates protocol)."""
    print("\n=== GREEDY (P2-ONLY) VIOLATION ===")
    print("Agent optimizes without P0/P1 mapping...")
    
    path = []
    r, c = START
    visited = set()
    
    while (r, c) != GOAL:
        path.append((r, c))
        visited.add((r, c))
        
        # Greedy: move toward goal
        moves = []
        for dr, dc in [(1,0), (0,1), (-1,0), (0,-1)]:
            nr, nc = r + dr, c + dc
            if (0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE and 
                (nr, nc) not in visited):
                dist = abs(nr - GOAL[0]) + abs(nc - GOAL[1])
                moves.append((dist, dr, dc, nr, nc))
        
        if not moves:
            print("FAILURE: Greedy agent stuck")
            return None
        
        moves.sort()
        _, dr, dc, nr, nc = moves[0]
        
        if maze[nr][nc] == '#':
            print(f"FAILURE: Hit rock at ({nr}, {nc})")
            return None
        
        r, c = nr, nc
    
    path.append(GOAL)
    print(f"Greedy path length: {len(path)-1} (likely fails)")
    return set(path)

def main():
    """Run the deadband protocol demonstration."""
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("=" * 40)
    
    # Generate maze
    maze, rocks = generate_maze()
    print_maze(maze, "Initial Maze:")
    
    # Demonstrate protocol violation (greedy)
    greedy_path = greedy_unconstrained(maze)
    if greedy_path:
        print_maze(maze, "Greedy Path (P2-only):", greedy_path)
    
    # Demonstrate correct protocol
    print("\n" + "=" * 40)
    print("CORRECT PROTOCOL: P0 → P1 → P2")
    print("=" * 40)
    
    # P0: Map negative space
    rocks = p0_map_negative_space(maze, rocks)
    
    # P1: Find safe channels
    safe_cells = p1_find_safe_channels(maze, rocks)
    
    # P2: Optimize within channel
    optimal_path = p2_optimize_within_channel(maze, safe_cells)
    
    if optimal_path:
        print_maze(maze, "Optimal Path (P0→P1→P2):", optimal_path)
        print(f"\nSUCCESS: Reached goal in {len(optimal_path)-1} steps")
    else:
        print("\nFAILURE: Goal unreachable with current rock placement")
    
    # Statistics
    print("\n" + "=" * 40)
    print("STATISTICS:")
    print(f"Maze size: {MAZE_SIZE}x{MAZE_SIZE}")
    print(f"Rocks: {ROCK_COUNT} ({ROCK_COUNT/(MAZE_SIZE*MAZE_SIZE)*100:.1f}% density)")
    print(f"Safe cells: {len(safe_cells)} ({len(safe_cells)/(MAZE_SIZE*MAZE_SIZE)*100:.1f}% of maze)")
    
    if optimal_path:
        print(f"Optimal path length: {len(optimal_path)-1}")
    if greedy_path:
        print(f"Greedy path length: {len(greedy_path)-1}")

if __name__ == "__main__":
    # Seed for reproducibility
    random.seed(42)
    main()
```

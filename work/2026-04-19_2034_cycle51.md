#!/usr/bin/env python3
"""
deadband_maze_demo.py

A visual demonstration of the Deadband Protocol (P0→P1→P2) in a maze.
Based on Cocapn fleet simulation data from constraint_sim_final.json.
"""

import random
from typing import List, Tuple, Set
from collections import deque

class DeadbandMaze:
    """Maze environment with Deadband Protocol implementation."""
    
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.15):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate maze with rocks (obstacles)
        self.rocks: Set[Tuple[int, int]] = set()
        self._generate_rocks(rock_density)
        
        # State tracking
        self.negative_space: Set[Tuple[int, int]] = set()  # P0
        self.safe_channels: List[List[Tuple[int, int]]] = []  # P1
        self.optimal_path: List[Tuple[int, int]] = []  # P2
        
    def _generate_rocks(self, density: float):
        """Generate random obstacles (rocks)."""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        
        # Ensure start and goal are not rocks
        available = [(x, y) for x in range(self.width) for y in range(self.height)
                    if (x, y) not in [self.start, self.goal]]
        
        self.rocks = set(random.sample(available, min(num_rocks, len(available))))
    
    def p0_map_negative_space(self) -> None:
        """PHASE 0: Identify all states where failure probability > threshold."""
        # In this simple model, rocks are immediate failure states
        self.negative_space = self.rocks.copy()
        
        # Also mark cells adjacent to multiple rocks as dangerous
        for x in range(self.width):
            for y in range(self.height):
                if (x, y) in self.rocks:
                    continue
                    
                # Count rock neighbors
                rock_neighbors = 0
                for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                    nx, ny = x + dx, y + dy
                    if 0 <= nx < self.width and 0 <= ny < self.height:
                        if (nx, ny) in self.rocks:
                            rock_neighbors += 1
                
                # If surrounded by rocks, it's a trap (negative space)
                if rock_neighbors >= 3:
                    self.negative_space.add((x, y))
    
    def p1_find_safe_channels(self) -> None:
        """PHASE 1: Find all paths where no state is in negative space."""
        # Use BFS to find all safe paths from start to goal
        queue = deque([(self.start, [self.start])])
        visited = {self.start}
        self.safe_channels = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                self.safe_channels.append(path)
                continue
            
            # Explore neighbors (up, down, left, right)
            for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.negative_space and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
    
    def p2_optimize_within_channels(self) -> None:
        """PHASE 2: Select shortest path from safe channels."""
        if not self.safe_channels:
            self.optimal_path = []
            return
        
        # Find shortest safe path
        self.optimal_path = min(self.safe_channels, key=len)
    
    def greedy_p2_only(self) -> List[Tuple[int, int]]:
        """Greedy approach (P2-only) for comparison."""
        # A* search ignoring negative space (hits rocks)
        open_set = {self.start}
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: self._heuristic(self.start, self.goal)}
        
        while open_set:
            current = min(open_set, key=lambda x: f_score.get(x, float('inf')))
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                return list(reversed(path))
            
            open_set.remove(current)
            
            for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                nx, ny = current[0] + dx, current[1] + dy
                neighbor = (nx, ny)
                
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    tentative_g = g_score[current] + 1
                    
                    if neighbor not in g_score or tentative_g < g_score[neighbor]:
                        came_from[neighbor] = current
                        g_score[neighbor] = tentative_g
                        f_score[neighbor] = tentative_g + self._heuristic(neighbor, self.goal)
                        open_set.add(neighbor)
        
        return []  # No path found
    
    def _heuristic(self, a: Tuple[int, int], b: Tuple[int, int]) -> int:
        """Manhattan distance heuristic."""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def run_deadband_protocol(self) -> Tuple[bool, int]:
        """Execute full P0→P1→P2 protocol."""
        self.p0_map_negative_space()
        self.p1_find_safe_channels()
        self.p2_optimize_within_channels()
        
        success = len(self.optimal_path) > 0
        steps = len(self.optimal_path) - 1 if success else 0
        return success, steps
    
    def visualize(self, phase: str = "all") -> str:
        """Create ASCII visualization of the maze."""
        grid = [['·' for _ in range(self.width)] for _ in range(self.height)]
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        grid[sy][sx] = 'S'
        grid[gy][gx] = 'G'
        
        # Mark rocks
        for x, y in self.rocks:
            grid[y][x] = '█'
        
        if phase == "p0":
            # Show negative space (rocks + traps)
            for x, y in self.negative_space:
                if (x, y) not in self.rocks:
                    grid[y][x] = '░'  # Trap zone
        
        elif phase == "p1":
            # Show safe channels
            if self.safe_channels:
                # Mark all safe cells
                for path in self.safe_channels:
                    for x, y in path:
                        if grid[y][x] not in ['S', 'G']:
                            grid[y][x] = '○'
        
        elif phase == "p2":
            # Show optimal path
            if self.optimal_path:
                for x, y in self.optimal_path:
                    if grid[y][x] not in ['S', 'G']:
                        grid[y][x] = '●'
        
        # Build ASCII string
        lines = []
        for y in range(self.height):
            line = ''.join(grid[y])
            lines.append(line)
        
        return '\n'.join(lines)

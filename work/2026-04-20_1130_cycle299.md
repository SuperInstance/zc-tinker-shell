**Task:** Complete the Deadband Protocol visual demo (ASCII maze simulation).

**Status:** I have written a Python script that generates a random 20x20 maze and visually demonstrates the strict P0→P1→P2 priority order of the Deadband Protocol. The script simulates an agent navigating from start (S) to goal (G), showing three distinct phases:

1. **P0 (Map Negative Space):** Identifies all obstacles ("rocks") in the maze.
2. **P1 (Find Safe Channels):** Uses flood-fill to find all reachable, safe cells that avoid rocks.
3. **P2 (Optimize Within Channel):** Finds the shortest path from start to goal using only safe cells.

The output is a series of ASCII maps printed to the console, showing each phase step-by-step.

**Code:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
A visual ASCII simulation of P0→P1→P2 priority.
"""

import random
from collections import deque

def generate_maze(width=20, height=20, rock_density=0.2):
    """Generate a random maze with start, goal, and rocks."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
    
    # Ensure start and goal are clear
    start = (1, 1)
    goal = (width-2, height-2)
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    
    return maze, start, goal

def print_maze(maze, title=""):
    """Print maze with optional title."""
    if title:
        print(f"\n=== {title} ===")
    for row in maze:
        print(''.join(row))

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks)."""
    rocks = []
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start):
    """P1: Flood-fill to find all reachable safe cells."""
    width, height = len(maze[0]), len(maze)
    safe = [[False for _ in range(width)] for _ in range(height)]
    
    # Directions: up, down, left, right
    dirs = [(0, -1), (0, 1), (-1, 0), (1, 0)]
    queue = deque([start])
    safe[start[1]][start[0]] = True
    
    while queue:
        x, y = queue.popleft()
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if not safe[ny][nx] and maze[ny][nx] != '#':
                    safe[ny][nx] = True
                    queue.append((nx, ny))
    
    # Mark safe cells on a copy for visualization
    viz = [row[:] for row in maze]
    for y in range(height):
        for x in range(width):
            if safe[y][x] and viz[y][x] == '.':
                viz[y][x] = '~'  # Safe channel marker
    return safe, viz

def p2_optimize_path(maze, safe, start, goal):
    """P2: Find shortest path within safe channels using BFS."""
    width, height = len(maze[0]), len(maze)
    dirs = [(0, -1), (0, 1), (-1, 0), (1, 0)]
    
    # BFS setup
    parent = {start: None}
    queue = deque([start])
    
    while queue:
        x, y = queue.popleft()
        if (x, y) == goal:
            break
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if safe[ny][nx] and (nx, ny) not in parent:
                    parent[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    
    # Reconstruct path if goal reached
    if goal not in parent:
        return None, maze  # No safe path
    
    path = []
    curr = goal
    while curr is not None:
        path.append(curr)
        curr = parent[curr]
    path.reverse()
    
    # Mark path on a copy
    viz = [row[:] for row in maze]
    for (x, y) in path[1:-1]:  # Exclude start and goal
        viz[y][x] = '*'  # Path marker
    return path, viz

def main():
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("=" * 40)
    
    # Generate maze
    maze, start, goal = generate_maze(20, 20, rock_density=0.25)
    print_maze(maze, "Initial Maze (S=start, G=goal, #=rock)")
    
    # P0: Map negative space
    rocks = p0_map_rocks(maze)
    print(f"\nP0: Mapped {len(rocks)} rocks (negative space).")
    
    # P1: Find safe channels
    safe, safe_viz = p1_find_safe_channels(maze, start)
    print_maze(safe_viz, "P1: Safe Channels (~ = reachable, safe)")
    
    # P2: Optimize within channel
    path, path_viz = p2_optimize_path(maze, safe, start, goal)
    if path:
        print_maze(path_viz, "P2: Optimal Path (* = path, within safe channels)")
        print(f"P2: Found path of length {len(path)} steps.")
    else:
        print_maze(path_viz, "P2: No Safe Path to Goal")
        print("P2: Goal unreachable within safe channels.")
    
    # Demonstrate violation: greedy (P2-only) approach
    print("\n--- VIOLATION DEMO: P2-Only (Greedy) ---")
    print("Agent that optimizes before mapping hits rocks.")
    # Simple greedy: move toward goal, ignoring rocks
    greedy_viz = [row[:] for row in maze]
    x, y = start
    steps = 0
    max_steps = 100
    while (x, y) != goal and steps < max_steps:
        # Move toward goal
        dx = 1 if goal[0] > x else -1 if goal[0] < x else 0
        dy = 1 if goal[1] > y else -1 if goal[1] < y else 0
        nx, ny = x + dx, y + dy
        
        if 0 <= nx < len(maze[0]) and 0 <= ny < len(maze):
            if maze[ny][nx] == '#':
                greedy_viz[ny][nx] = 'X'  # Hit a rock!
                break
            else:
                if greedy_viz[ny][nx] == '.':
                    greedy_viz[ny][nx] = 'o'  # Greedy path
                x, y = nx, ny
        steps += 1
    
    print_maze(greedy_viz, "Greedy Path (o = attempted, X = rock hit)")
    if steps >= max_steps:
        print("Greedy: Stuck in local minima (max steps exceeded).")
    else:
        print("Greedy: Hit a rock — failure.")

if __name__ == "__main__":
    main()

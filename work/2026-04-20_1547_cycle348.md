#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 348)
Visual demonstration of P0→P1→P2 in a 10x10 ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
import time

class DeadbandMaze:
    def __init__(self, width=10, height=10, rock_density=0.2):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self.generate_maze(rock_density)
        self.agent_pos = self.start
        self.path = []
        self.safe_channels = []
        self.visited = set()
        
    def generate_maze(self, density):
        """P0: Rocks are negative space."""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        # Exclude start and goal
        possible = [(x, y) for x in range(self.width) for y in range(self.height)
                   if (x, y) not in [self.start, self.goal]]
        self.rocks = set(random.sample(possible, num_rocks))
    
    def print_maze(self, show_agent=True, show_channels=False):
        """ASCII visualization."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')
                elif show_agent and pos == self.agent_pos:
                    row.append('A')
                elif show_channels and pos in self.safe_channels:
                    row.append('.')
                elif pos in self.path:
                    row.append('*')
                else:
                    row.append(' ')
            print(' '.join(row))
        print()
    
    def p0_map_negative_space(self):
        """P0: Identify rocks (what NOT to do)."""
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Rocks: {len(self.rocks)} positions")
        self.print_maze(show_agent=False)
        time.sleep(1)
        return list(self.rocks)
    
    def p1_find_safe_channels(self):
        """P1: Find safe paths (deadbands)."""
        print("=== P1: FIND SAFE CHANNELS ===")
        # Simple BFS to find all reachable safe cells
        from collections import deque
        queue = deque([self.start])
        visited = set([self.start])
        safe = set()
        
        while queue:
            x, y = queue.popleft()
            safe.add((x, y))
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    npos = (nx, ny)
                    if npos not in self.rocks and npos not in visited:
                        visited.add(npos)
                        queue.append(npos)
        
        self.safe_channels = list(safe)
        print(f"Safe channels: {len(self.safe_channels)} positions")
        self.print_maze(show_agent=False, show_channels=True)
        time.sleep(1)
        return self.safe_channels
    
    def p2_optimize_within_channel(self):
        """P2: Find shortest path within safe channels."""
        print("=== P2: OPTIMIZE WITHIN CHANNEL ===")
        # A* search constrained to safe channels
        from heapq import heappush, heappop
        
        start, goal = self.start, self.goal
        safe_set = set(self.safe_channels)
        
        if goal not in safe_set:
            print("Goal is in rocks! No safe path.")
            return []
        
        def heuristic(a, b):
            return abs(a[0]-b[0]) + abs(a[1]-b[1])
        
        open_set = []
        heappush(open_set, (0, start))
        came_from = {}
        g_score = {start: 0}
        f_score = {start: heuristic(start, goal)}
        
        while open_set:
            _, current = heappop(open_set)
            
            if current == goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.reverse()
                self.path = [start] + path
                print(f"Optimal path length: {len(self.path)} steps")
                self.print_maze(show_agent=True)
                return self.path
            
            x, y = current
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                neighbor = (nx, ny)
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if neighbor not in safe_set:
                        continue  # Skip rocks (enforce deadband)
                    tentative_g = g_score[current] + 1
                    if neighbor not in g_score or tentative_g < g_score[neighbor]:
                        came_from[neighbor] = current
                        g_score[neighbor] = tentative_g
                        f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                        heappush(open_set, (f_score[neighbor], neighbor))
        
        print("No path found within safe channels.")
        return []
    
    def run_unconstrained_greedy(self):
        """Demonstrate failure without P0+P1."""
        print("\n=== UNCONSTRAINED GREEDY (P2 ONLY) ===")
        pos = self.start
        path = [pos]
        for _ in range(50):  # Max steps
            if pos == self.goal:
                print("Reached goal (lucky!)")
                break
            # Greedy move toward goal
            x, y = pos
            gx, gy = self.goal
            if random.random() < 0.7:  # Mostly toward goal
                dx = 1 if gx > x else -1 if gx < x else 0
                dy = 1 if gy > y else -1 if gy < y else 0
            else:
                dx, dy = random.choice([(1,0), (-1,0), (0,1), (0,-1)])
            nx, ny = x + dx, y + dy
            if 0 <= nx < self.width and 0 <= ny < self.height:
                new_pos = (nx, ny)
                if new_pos in self.rocks:
                    print(f"Hit rock at {new_pos}! Failed.")
                    return False
                pos = new_pos
                path.append(pos)
        return pos == self.goal

def main():
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("=" * 40)
    
    # Initialize maze
    maze = DeadbandMaze(width=10, height=10, rock_density=0.2)
    
    # P0: Map negative space
    rocks = maze.p0_map_negative_space()
    
    # P1: Find safe channels
    safe = maze.p1_find_safe_channels()
    
    # P2: Optimize within channel
    optimal_path = maze.p2_optimize_within_channel()
    
    # Demonstrate failure mode
    success = maze.run_unconstrained_greedy()
    
    print("\n=== RESULTS ===")
    print(f"Rocks (P0): {len(rocks)}")
    print(f"Safe channels (P1): {len(safe)}")
    print(f"Optimal path length (P2): {len(optimal_path) if optimal_path else 'No path'}")
    print(f"Unconstrained greedy success: {success}")
    
    # Compare with fleet simulation data
    print("\n=== FLEET SIMULATION DATA (from constraint_sim_final.json) ===")
    print("Optimal steps: 21")
    print("Deadband success: 50/50")
    print("Unconstrained success: 33/50")
    print("Constrained (no P0) success: 0/50")

if __name__ == "__main__":
    main()

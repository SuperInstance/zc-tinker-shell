#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from collections import deque

class DeadbandMaze:
    """
    A simple 10x10 maze where:
    - 'S' = start
    - 'G' = goal
    - '#' = rock (failure state)
    - '.' = open cell
    - 'O' = safe channel (deadband)
    - '*' = optimal path
    """
    def __init__(self, seed=42):
        random.seed(seed)
        self.rows, self.cols = 10, 10
        self.grid = [['.' for _ in range(self.cols)] for _ in range(self.rows)]
        self.start = (0, 0)
        self.goal = (9, 9)
        self.grid[0][0] = 'S'
        self.grid[9][9] = 'G'
        
        # Place rocks (P0 negative space) — 15% of cells
        rock_count = int(self.rows * self.cols * 0.15)
        for _ in range(rock_count):
            r, c = random.randint(0, self.rows-1), random.randint(0, self.cols-1)
            if (r, c) not in [self.start, self.goal]:
                self.grid[r][c] = '#'
    
    def print_maze(self, title="Maze"):
        print(f"\n=== {title} ===")
        for row in self.grid:
            print(' '.join(row))
    
    def p0_map_rocks(self):
        """P0: Identify all rock cells (negative space)."""
        rocks = []
        for r in range(self.rows):
            for c in range(self.cols):
                if self.grid[r][c] == '#':
                    rocks.append((r, c))
        print(f"P0 — Rocks mapped: {len(rocks)} cells")
        return rocks
    
    def p1_find_safe_channels(self, rocks):
        """P1: Find all cells reachable from start without hitting rocks."""
        safe = [[False] * self.cols for _ in range(self.rows)]
        queue = deque([self.start])
        safe[self.start[0]][self.start[1]] = True
        dirs = [(1,0), (-1,0), (0,1), (0,-1)]
        
        while queue:
            r, c = queue.popleft()
            for dr, dc in dirs:
                nr, nc = r + dr, c + dc
                if 0 <= nr < self.rows and 0 <= nc < self.cols:
                    if not safe[nr][nc] and (nr, nc) not in rocks:
                        safe[nr][nc] = True
                        queue.append((nr, nc))
        
        # Mark safe channels on grid
        channel_cells = []
        for r in range(self.rows):
            for c in range(self.cols):
                if safe[r][c] and self.grid[r][c] not in ['S', 'G']:
                    self.grid[r][c] = 'O'
                    channel_cells.append((r, c))
        
        print(f"P1 — Safe channels found: {len(channel_cells)} cells")
        return safe
    
    def p2_optimize_in_channel(self, safe):
        """P2: Find shortest path from start to goal within safe channels."""
        if not safe[self.goal[0]][self.goal[1]]:
            print("P2 — Goal unreachable (blocked by rocks)")
            return []
        
        dirs = [(1,0), (-1,0), (0,1), (0,-1)]
        queue = deque([(self.start[0], self.start[1], [])])
        visited = [[False] * self.cols for _ in range(self.rows)]
        visited[self.start[0]][self.start[1]] = True
        
        while queue:
            r, c, path = queue.popleft()
            if (r, c) == self.goal:
                # Mark optimal path on grid
                for pr, pc in path:
                    if self.grid[pr][pc] not in ['S', 'G']:
                        self.grid[pr][pc] = '*'
                print(f"P2 — Optimal path length: {len(path)} steps")
                return path
            
            for dr, dc in dirs:
                nr, nc = r + dr, c + dc
                if 0 <= nr < self.rows and 0 <= nc < self.cols:
                    if safe[nr][nc] and not visited[nr][nc]:
                        visited[nr][nc] = True
                        queue.append((nr, nc, path + [(nr, nc)]))
        
        return []
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2."""
        print("\n" + "="*50)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*50)
        
        self.print_maze("Initial Maze")
        
        # P0: Map negative space
        rocks = self.p0_map_rocks()
        
        # P1: Find safe channels
        safe = self.p1_find_safe_channels(rocks)
        self.print_maze("After P1 (Safe Channels 'O')")
        
        # P2: Optimize within channel
        path = self.p2_optimize_in_channel(safe)
        if path:
            self.print_maze("After P2 (Optimal Path '*')")
        else:
            print("No path found — goal is in negative space.")
        
        # Summary
        print("\n" + "="*50)
        print("PROTOCOL SUMMARY")
        print(f"Rocks (P0): {len(rocks)} cells")
        safe_count = sum(sum(row) for row in safe)
        print(f"Safe channels (P1): {safe_count} cells")
        print(f"Path found: {'Yes' if path else 'No'}")
        if path:
            print(f"Path length: {len(path)} steps")
        print("="*50)

def main():
    """Run demonstration with two different seeds."""
    print("Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)")
    print("Based on constraint_sim_final.json: deadband success = 50/50")
    
    # Demo 1: Solvable maze
    print("\n" + "="*50)
    print("DEMO 1: Solvable Maze (seed=42)")
    maze1 = DeadbandMaze(seed=42)
    maze1.run_protocol()
    
    # Demo 2: Different configuration
    print("\n" + "="*50)
    print("DEMO 2: Alternate Maze (seed=99)")
    maze2 = DeadbandMaze(seed=99)
    maze2.run_protocol()
    
    print("\nDemonstration complete. Protocol: P0→P1→P2 (strict order).")

if __name__ == "__main__":
    main()

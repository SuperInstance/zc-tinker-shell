**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on the provided `constraint_sim_final.json` and `DEADBAND-PROTOCOL.md`, I will create a concrete, runnable simulation. The data shows a 20×20 maze with 64 rocks. The protocol's three strict phases (P0, P1, P2) must be followed in order.

**Approach:**
1. **P0 (Map Negative Space):** Generate a random 20×20 maze with 64 obstacle cells ('#'). The agent cannot enter these.
2. **P1 (Identify Safe Channels):** From the start (0,0) to goal (19,19), compute all cells reachable via BFS that avoid obstacles. These form the "safe water."
3. **P2 (Optimize Within Channel):** Use A* search, but only allowed to move through cells marked as safe in P1. This ensures the agent stays within the deadband.
4. **Visualization:** Print the maze at each phase, showing obstacles, safe channels, and the final optimal path.

**Output:** A single, self-contained Python script.

```python
#!/usr/bin/env python3
"""
deadband_maze_demo.py

A visual demonstration of the Deadband Protocol (P0→P1→P2) in a 20×20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
and fleet-knowledge/docs/DEADBAND-PROTOCOL.md

Simulation parameters:
- Grid: 20 rows × 20 columns
- Rocks (obstacles): 64 cells (16% density)
- Start: (0, 0)
- Goal: (19, 19)
- Movement: 4‑connected (up, down, left, right)

The protocol is followed strictly:
P0: Identify all obstacle cells (negative space).
P1: Compute all reachable safe cells (safe channels).
P2: Find shortest path within safe channels only.
"""

import random
from collections import deque
import heapq

# ----------------------------------------------------------------------
# Parameters (from constraint_sim_final.json)
# ----------------------------------------------------------------------
GRID_SIZE = 20
NUM_ROCKS = 64
START = (0, 0)
GOAL = (GRID_SIZE - 1, GRID_SIZE - 1)

# Symbols for visualization
ROCK = '#'
SAFE = '.'
START_SYM = 'S'
GOAL_SYM = 'G'
PATH = 'o'
EMPTY = ' '

# ----------------------------------------------------------------------
# P0: Map the Negative Space
# ----------------------------------------------------------------------
def generate_rocks(grid_size, num_rocks, start, goal):
    """Place random obstacles, avoiding start and goal."""
    cells = [(r, c) for r in range(grid_size) for c in range(grid_size)]
    cells.remove(start)
    cells.remove(goal)
    rocks = random.sample(cells, num_rocks)
    return set(rocks)

# ----------------------------------------------------------------------
# P1: Identify Safe Channels (BFS from start, avoiding rocks)
# ----------------------------------------------------------------------
def find_safe_channels(rocks, grid_size, start):
    """Return set of all cells reachable from start without hitting a rock."""
    safe = set()
    q = deque([start])
    safe.add(start)
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    while q:
        r, c = q.popleft()
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if 0 <= nr < grid_size and 0 <= nc < grid_size:
                if (nr, nc) not in rocks and (nr, nc) not in safe:
                    safe.add((nr, nc))
                    q.append((nr, nc))
    return safe

# ----------------------------------------------------------------------
# P2: Optimize Within Channel (A* restricted to safe cells)
# ----------------------------------------------------------------------
def astar_in_channel(safe_cells, start, goal):
    """A* search that only moves through safe_cells."""
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            return list(reversed(path))
        for dr, dc in dirs:
            nr, nc = current[0] + dr, current[1] + dc
            neighbor = (nr, nc)
            if neighbor not in safe_cells:
                continue
            tentative_g = g_score[current] + 1
            if neighbor not in g_score or tentative_g < g_score[neighbor]:
                came_from[neighbor] = current
                g_score[neighbor] = tentative_g
                f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                heapq.heappush(open_set, (f_score[neighbor], neighbor))
    return []  # No path found within safe channels

# ----------------------------------------------------------------------
# Visualization
# ----------------------------------------------------------------------
def print_grid(grid_size, rocks, safe_cells, path, title):
    """Print an ASCII grid showing rocks, safe cells, path, start, goal."""
    print(f"\n{title}")
    print("=" * (grid_size + 2))
    for r in range(grid_size):
        row = []
        for c in range(grid_size):
            pos = (r, c)
            if pos == START:
                row.append(START_SYM)
            elif pos == GOAL:
                row.append(GOAL_SYM)
            elif pos in path:
                row.append(PATH)
            elif pos in rocks:
                row.append(ROCK)
            elif pos in safe_cells:
                row.append(SAFE)
            else:
                row.append(EMPTY)
        print('|' + ''.join(row) + '|')
    print("=" * (grid_size + 2))

# ----------------------------------------------------------------------
# Main Simulation
# ----------------------------------------------------------------------
def main():
    random.seed(42)  # Deterministic for demonstration
    
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print(f"Grid: {GRID_SIZE}×{GRID_SIZE}, Rocks: {NUM_ROCKS}")
    print(f"Start: {START}, Goal: {GOAL}")
    print("-" * 40)
    
    # --- P0: Map Negative Space ---
    rocks = generate_rocks(GRID_SIZE, NUM_ROCKS, START, GOAL)
    print_grid(GRID_SIZE, rocks, set(), [], "P0: NEGATIVE SPACE (Rocks = '#')")
    print(f"P0 complete: {len(rocks)} obstacle cells identified.")
    
    # --- P1: Identify Safe Channels ---
    safe_cells = find_safe_channels(rocks, GRID_SIZE, START)
    print_grid(GRID_SIZE, rocks, safe_cells, [], "P1: SAFE CHANNELS (Safe = '.')")
    print(f"P1 complete: {len(safe_cells)} reachable safe cells.")
    
    # --- P2: Optimize Within Channel ---
    path = astar_in_channel(safe_cells, START, GOAL)
    if path:
        print_grid(GRID_SIZE, rocks, safe_cells, path, "P2: OPTIMAL PATH WITHIN CHANNEL (Path = 'o')")
        print(f"P2 complete: Optimal path length = {len(path)-1} steps.")
        print(f"Path: {path}")
    else:
        print("P2: No path to goal within safe channels (dead‑end).")
    
    # --- Protocol Violation Warning ---
    print("\n" + "!"*50)
    print("VIOLATION EXAMPLE: If you skip P0/P1 and run unconstrained A*")
    print("(i.e., ignore rocks), the agent will hit a rock with probability → 1.0")
    print("as maze complexity increases (see constraint_sim_final.json).")
    print("!"*50)

if __name__ == "__main__":
    main()
```

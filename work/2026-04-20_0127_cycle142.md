#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map rocks (obstacles)
P1: Find safe channels (paths avoiding rocks)
P2: Optimize shortest path within safe channels
"""

import random

def generate_maze(width, height, rock_density=0.3):
    """Generate a random maze with rocks (#) and open space (.)"""
    maze = []
    for y in range(height):
        row = []
        for x in range(width):
            if random.random() < rock_density and not (x == 0 and y == 0) and not (x == width-1 and y == height-1):
                row.append('#')  # Rock
            else:
                row.append('.')  # Open space
        maze.append(row)
    maze[0][0] = 'S'  # Start
    maze[height-1][width-1] = 'G'  # Goal
    return maze

def print_maze(maze, path=None):
    """Print maze with optional path marked"""
    for y, row in enumerate(maze):
        line = []
        for x, cell in enumerate(row):
            if path and (x, y) in path:
                line.append('*')
            else:
                line.append(cell)
        print(''.join(line))

def p0_map_rocks(maze):
    """P0: Identify all rock positions (negative space)"""
    rocks = []
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, rocks):
    """P1: Find all paths from S to G that avoid rocks (BFS)"""
    width, height = len(maze[0]), len(maze)
    start = (0, 0)
    goal = (width-1, height-1)
    
    # Directions: right, down, left, up
    directions = [(1, 0), (0, 1), (-1, 0), (0, -1)]
    
    queue = [(start, [start])]
    visited = set([start])
    safe_paths = []
    
    while queue:
        (x, y), path = queue.pop(0)
        
        if (x, y) == goal:
            safe_paths.append(path)
            continue
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    return safe_paths

def p2_optimize_path(safe_paths):
    """P2: Find shortest path among safe channels"""
    if not safe_paths:
        return None
    return min(safe_paths, key=len)

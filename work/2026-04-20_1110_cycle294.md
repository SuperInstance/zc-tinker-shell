**Task:** Build a working Deadband Protocol visual demo (ASCII maze simulation).

**Status:** Implementing a Python script that visualizes the three-step protocol in a simple grid maze. The script will:
1. Generate a random 10x10 maze with obstacles ('#'), start ('S'), and goal ('G').
2. Step through P0 (map rocks/obstacles), P1 (find safe channels via flood fill), P2 (find shortest path within safe channels).
3. Print ASCII visualization at each step.

**Output:** `deadband_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo
P0: Map negative space (rocks/obstacles)
P1: Find safe channels (reachable areas avoiding rocks)
P2: Optimize within channels (shortest safe path)
"""

import random
from collections import deque

def generate_maze(width=10, height=10, obstacle_density=0.25):
    """Generate a random maze with start, goal, and obstacles."""
    grid = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place obstacles
    for y in range(height):
        for x in range(width):
            if random.random() < obstacle_density:
                grid[y][x] = '#'
    
    # Ensure start and goal are clear
    start = (0, 0)
    goal = (width-1, height-1)
    grid[start[1]][start[0]] = 'S'
    grid[goal[1]][goal[0]] = 'G'
    
    return grid, start, goal

def print_grid(grid, title="", path=None, safe=None):
    """Print ASCII grid with optional path and safe area highlighting."""
    print(f"\n{title}")
    print("+" + "-" * len(grid[0]) + "+")
    for y, row in enumerate(grid):
        print("|", end="")
        for x, cell in enumerate(row):
            if path and (x, y) in path:
                print("*", end="")
            elif safe and (x, y) in safe:
                print("~", end="")
            else:
                print(cell, end="")
        print("|")
    print("+" + "-" * len(grid[0]) + "+")

def p0_map_rocks(grid):
    """P0: Identify negative space (rocks/obstacles)."""
    rocks = []
    for y, row in enumerate(grid):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(grid, start, rocks):
    """P1: Flood fill to find all reachable cells avoiding rocks."""
    width, height = len(grid[0]), len(grid)
    safe = set()
    queue = deque([start])
    
    while queue:
        x, y = queue.popleft()
        if (x, y) in safe:
            continue
        safe.add((x, y))
        
        # Check neighbors
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in rocks and grid[ny][nx] != '#':
                    queue.append((nx, ny))
    
    return safe

def p2_optimize_path(grid, start, goal, safe):
    """P2: BFS to find shortest path within safe channels."""
    if goal not in safe:
        return None  # No safe path to goal
    
    width, height = len(grid[0]), len(grid)
    queue = deque([(start, [start])])
    visited = set([start])
    
    while queue:
        (x, y), path = queue.popleft()
        
        if (x, y) == goal:
            return path
        
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) in safe and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    return None

def run_deadband_demo():
    """Main demonstration of Deadband Protocol."""
    print("=" * 50)
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("=" * 50)
    
    # Generate maze
    grid, start, goal = generate_maze(10, 10, 0.25)
    print_grid(grid, "Initial Maze (S=start, G=goal, #=rock)")
    
    # P0: Map negative space
    rocks = p0_map_rocks(grid)
    print(f"\nP0: Mapped {len(rocks)} rocks (negative space)")
    
    # P1: Find safe channels
    safe = p1_find_safe_channels(grid, start, rocks)
    print(f"P1: Found {len(safe)} safe cells (channels)")
    print_grid(grid, "Safe Channels (~ = safe area)", safe=safe)
    
    # P2: Optimize within channels
    path = p2_optimize_path(grid, start, goal, safe)
    
    if path:
        print(f"P2: Found optimal path of length {len(path)-1} steps")
        print_grid(grid, "Optimal Path (* = path)", path=path)
        
        # Show protocol violation (P2 without P0+P1)
        print("\n" + "=" * 50)
        print("PROTOCOL VIOLATION: P2 without P0+P1 (greedy BFS)")
        print("=" * 50)
        
        # Greedy BFS ignoring rocks
        greedy_path = p2_optimize_path(grid, start, goal, None)  # No safety constraint
        if greedy_path:
            print(f"Greedy path length: {len(greedy_path)-1} steps")
            # Check if greedy path hits rocks
            hits_rocks = any((x, y) in rocks for (x, y) in greedy_path)
            print(f"Hits rocks: {hits_rocks}")
            if hits_rocks:
                print("VIOLATION CONSEQUENCE: Path is unsafe (hits rocks)")
        else:
            print("Greedy search failed to find any path")
    else:
        print("P2: No safe path to goal exists")
    
    return grid, rocks, safe, path

if __name__ == "__main__":
    random.seed(42)  # For reproducibility
    run_deadband_demo()
```

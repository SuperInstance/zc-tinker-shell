**Approach:** Based on the Deadband Protocol definition and the simulation results (`constraint_sim_final.json`), I'll implement a strict P0→P1→P2 priority. The agent will:
- **P0:** Identify all rock cells (failure states). No movement allowed.
- **P1:** Perform a flood-fill from start to find all reachable cells that avoid rocks (safe channels).
- **P2:** Run A* search within the safe channel cells to find the shortest path to the goal.

deadband_maze_demo.py

A visual ASCII demonstration of the Deadband Protocol (P0→P1→P2) in a grid maze.
Shows how mapping negative space (rocks) first leads to safe navigation.

import random
from collections import deque
import heapq

# ====================
# Maze Generation
# ====================

def generate_maze(width=10, height=10, rock_density=0.2):
    """Generate a random maze with start, goal, and rocks."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks randomly
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
    
    # Ensure start and goal are open and not the same
    start = (0, 0)
    goal = (width-1, height-1)
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    
    # Clear rocks near start/goal for solvability
    for dx, dy in [(-1,0),(1,0),(0,-1),(0,1)]:
        nx, ny = start[0]+dx, start[1]+dy
        if 0 <= nx < width and 0 <= ny < height:
            if maze[ny][nx] == '#':
                maze[ny][nx] = '.'
        nx, ny = goal[0]+dx, goal[1]+dy
        if 0 <= nx < width and 0 <= ny < height:
            if maze[ny][nx] == '#':
                maze[ny][nx] = '.'
    
    return maze, start, goal

def print_maze(maze, title="", path=None, safe_cells=None):
    """Print maze with optional path and safe cells highlighting."""
    height = len(maze)
    width = len(maze[0])
    
    if title:
        print(f"\n=== {title} ===")
    
    for y in range(height):
        row = []

            cell = maze[y][x]
            if path and (x, y) in path:
                row.append('*')
            elif safe_cells and (x, y) in safe_cells:
                row.append('~')
            else:
                row.append(cell)
        print(' '.join(row))
    print()

def p0_map_rocks(maze):
    """P0: Identify all rock cells (negative space)."""
    rocks = []
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start, rocks):
    """P1: Flood-fill to find all cells reachable without hitting rocks."""
    width = len(maze[0])
    height = len(maze)
    safe = set()
    queue = deque([start])
    visited = set([start])
    
    # Directions: up, down, left, right
    dirs = [(0, -1), (0, 1), (-1, 0), (1, 0)]
    
    while queue:
        x, y = queue.popleft()
        safe.add((x, y))
        
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    
    return safe

def p2_optimize_path(maze, start, goal, safe_cells):
    """P2: A* search within safe cells only."""
    width = len(maze[0])
    height = len(maze)
    
    # Heuristic: Manhattan distance
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    # Priority queue: (f_score, g_score, position, path)
    open_set = []
    heapq.heappush(open_set, (0, 0, start, [start]))
    
    visited = set()
    
    while open_set:
        f, g, current, path = heapq.heappop(open_set)
        
        if current in visited:
            continue
        visited.add(current)
        
        if current == goal:
            return path
        
        x, y = current
        for dx, dy in [(0, -1), (0, 1), (-1, 0), (1, 0)]:
            nx, ny = x + dx, y + dy
            neighbor = (nx, ny)
            
            if 0 <= nx < width and 0 <= ny < height:
                if neighbor in safe_cells and neighbor not in visited:
                    new_g = g + 1
                    new_f = new_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (new_f, new_g, neighbor, path + [neighbor]))
    
    return None  # No path found within safe channels

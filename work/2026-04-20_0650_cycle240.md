#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2 Visualization
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import json
import random
from pathlib import Path

# Load simulation data
try:
    sim_path = Path("fleet-knowledge/narrow-games/constraint_sim_final.json")
    with open(sim_path) as f:
        sim_data = json.load(f)
except FileNotFoundError:
    # Fallback to known values from the file
    sim_data = {
        "optimal": 21,
        "rocks": 64,
        "unconstrained_success": 33,
        "constrained_success": 0,
        "deadband_success": 50,
        "unconstrained_avg": 2320.9,
        "constrained_avg": 0,
        "deadband_avg": 21.0
    }

class DeadbandMaze:
    """Visual demonstration of Deadband Protocol in a maze"""
    
    def __init__(self, width=20, height=20):
        self.width = width
        self.height = height
        self.rocks = []  # P0: Negative space
        self.channels = []  # P1: Safe channels
        self.path = []  # P2: Optimal path within channels
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks based on simulation data (64 rocks in 20x20)
        self._generate_rocks()
        
    def _generate_rocks(self):
        """P0: Map negative space (rocks)"""
        num_rocks = min(sim_data["rocks"], self.width * self.height - 2)
        
        # Ensure start and goal are not rocks
        all_cells = [(x, y) for x in range(self.width) for y in range(self.height)]
        all_cells.remove(self.start)
        all_cells.remove(self.goal)
        
        self.rocks = random.sample(all_cells, num_rocks)
        
    def _find_safe_channels(self):
        """P1: Identify safe channels (BFS avoiding rocks)"""
        from collections import deque
        
        visited = set()
        queue = deque([(self.start, [self.start])])
        safe_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
                
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            
            # Check all four directions
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.channels = safe_paths
        return len(safe_paths) > 0
    
    def _optimize_path(self):
        """P2: Find shortest path within safe channels"""
        if not self.channels:
            return False
            
        # Find shortest path (optimal = 21 steps from simulation)
        self.path = min(self.channels, key=len)
        
        # Check if we found the optimal path
        optimal_length = sim_data["optimal"]
        return len(self.path) - 1 == optimal_length  # -1 because path includes start
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2"""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===")
        print(f"Based on simulation: {sim_data['deadband_success']}/50 success")
        print(f"Optimal path length: {sim_data['optimal']} steps")
        print()
        
        # P0: Map negative space
        print("P0: MAPPING NEGATIVE SPACE")
        print(f"  • Identified {len(self.rocks)} rocks (failure states)")
        print(f"  • Rocks density: {len(self.rocks)/(self.width*self.height):.1%}")
        print()
        
        # P1: Find safe channels
        print("P1: IDENTIFYING SAFE CHANNELS")
        has_channels = self._find_safe_channels()
        if has_channels:
            print(f"  • Found {len(self.channels)} safe channels")
            print(f"  • Channel lengths: {[len(c)-1 for c in self.channels[:5]]}...")
        else:
            print("  ✗ NO SAFE CHANNELS FOUND")
            return False
        print()
        
        # P2: Optimize within channels
        print("P2: OPTIMIZING WITHIN CHANNELS")
        is_optimal = self._optimize_path()
        if is_optimal:
            print(f"  • Found optimal path: {len(self.path)-1} steps")
            print(f"  • Matches simulation optimal: {sim_data['optimal']}")
        else:
            print(f"  • Found path: {len(self.path)-1} steps")
            print(f"  • Simulation optimal: {sim_data['optimal']}")
        print()
        
        return True
    
    def visualize(self, show_p0=True, show_p1=True, show_p2=True):
        """Create ASCII visualization of the protocol steps"""
        grid = [['·' for _ in range(self.width)] for _ in range(self.height)]
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        grid[sy][sx] = 'S'
        grid[gy][gx] = 'G'
        
        if show_p0:
            # Show rocks (P0)
            for x, y in self.rocks:
                grid[y][x] = '█'
        
        if show_p1 and self.channels:
            # Show all safe channels (P1) as faint dots
            for path in self.channels[:3]:  # Show first 3 channels
                for x, y in path[1:-1]:  # Exclude start and goal
                    if grid[y][x] == '·':
                        grid[y][x] = '◦'
        
        if show_p2 and self.path:
            # Show optimal path (P2)
            for x, y in self.path[1:-1]:  # Exclude start and goal
                grid[y][x] = '●'
        
        # Print the grid
        print("MAZE VISUALIZATION (20×20)")
        print("S = Start, G = Goal, █ = Rock, ◦ = Safe channel, ● = Optimal path")
        print()
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                row.append(grid[y][x])
            print(' '.join(row))
        
        # Legend
        print()
        print("PROTOCOL STEPS VISUALIZED:")
        if show_p0: print("  P0 (Rocks): █ - Negative space to avoid")
        if show_p1: print("  P1 (Channels): ◦ - Safe passage options")
        if show_p2: print("  P2 (Optimal): ● - Best path within safe channels")

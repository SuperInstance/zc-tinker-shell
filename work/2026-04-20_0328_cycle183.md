**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on the fleet knowledge, specifically the `constraint_sim_final.json` data and the formal DEADBAND-PROTOCOL.md, I will implement a concrete simulation that mirrors the proof. The data shows: unconstrained (P2-only) fails (0% success), deadband protocol (P0→P1→P2) succeeds (50% success in the sim, but 100% in the proof description). I'll create a small, runnable Python demo.

**Approach:**
1. **P0 (Map negative space):** Generate a random 10x10 maze with obstacles ('#') and open cells ('.'). The agent knows the full map.
2. **P1 (Identify safe channels):** Use BFS from start to goal to find *all* shortest paths that avoid obstacles. This defines the deadband — the set of cells that belong to at least one shortest safe path.
3. **P2 (Optimize within channel):** From the safe channels, pick the path that minimizes a simple cost (e.g., prefers straight lines). An unconstrained agent (P2-only) will greedily move toward the goal ignoring obstacles, hitting rocks.

**Output:** A script that prints each step, visualizes the maze, and shows success/failure for both strategies.

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Simulates P0→P1→P2 vs P2-only in a random maze.
Based on fleet-knowledge/constraint_sim_final.json and DEADBAND-PROTOCOL.md.
"""

import random
from collections import deque

def generate_maze(width=10, height=10, obstacle_ratio=0.2):
    """Generate a random maze with start (S), goal (G), and obstacles (#)."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    # Place obstacles
    for y in range(height):
        for x in range(width):
            if random.random() < obstacle_ratio:
                maze[y][x] = '#'
    # Ensure start and goal are clear
    start = (0, 0)
    goal = (width-1, height-1)
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    return maze, start, goal

def print_maze(maze, path=None):
    """Print maze with optional path marked."""
    for y, row in enumerate(maze):
        line = []
        for x, cell in enumerate(row):
            if path and (x, y) in path:
                line.append('*')
            else:
                line.append(cell)
        print(' '.join(line))

def bfs_all_shortest_paths(maze, start, goal):
    """BFS to find all shortest paths from start to goal avoiding obstacles."""
    width, height = len(maze[0]), len(maze)
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    queue = deque([(start, [start])])
    visited = {start: 0}
    shortest_paths = []
    shortest_length = None
    
    while queue:
        (x, y), path = queue.popleft()
        if (x, y) == goal:
            if shortest_length is None:
                shortest_length = len(path)
            if len(path) == shortest_length:
                shortest_paths.append(path)
            continue
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height and maze[ny][nx] != '#':
                new_pos = (nx, ny)
                new_len = len(path) + 1
                if new_pos not in visited or visited[new_pos] >= new_len:
                    visited[new_pos] = new_len
                    queue.append((new_pos, path + [new_pos]))
    return shortest_paths

def deadband_protocol(maze, start, goal):
    """P0→P1→P2: Map obstacles, find safe channels, optimize within."""
    print("\n=== DEADBAND PROTOCOL (P0→P1→P2) ===")
    # P0: Map negative space (obstacles already known from maze)
    obstacles = [(x, y) for y in range(len(maze)) for x in range(len(maze[0])) if maze[y][x] == '#']
    print(f"P0: Mapped {len(obstacles)} rocks.")
    
    # P1: Identify safe channels (all shortest paths)
    safe_paths = bfs_all_shortest_paths(maze, start, goal)
    if not safe_paths:
        print("P1: No safe channels to goal. FAIL.")
        return False, []
    # Safe channels = all cells in any shortest path
    safe_cells = set()
    for path in safe_paths:
        safe_cells.update(path)
    print(f"P1: Found {len(safe_paths)} shortest safe paths, {len(safe_cells)} safe cells.")
    
    # P2: Optimize within channel (pick first path for simplicity)
    chosen_path = safe_paths[0]
    print(f"P2: Selected path length {len(chosen_path)}.")
    return True, chosen_path

def unconstrained_greedy(maze, start, goal):
    """P2-only: Greedy move toward goal, ignoring obstacles."""
    print("\n=== UNCONSTRAINED GREEDY (P2-only) ===")
    x, y = start
    path = [start]
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    for step in range(100):  # limit steps
        if (x, y) == goal:
            print(f"P2-only: Reached goal in {len(path)} steps.")
            return True, path
        # Greedy: move in direction that reduces Manhattan distance
        best_dist = float('inf')
        best_move = None
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < len(maze[0]) and 0 <= ny < len(maze):
                dist = abs(nx - goal[0]) + abs(ny - goal[1])
                if dist < best_dist:
                    best_dist = dist
                    best_move = (nx, ny)
        if best_move is None:
            break
        nx, ny = best_move
        # Check for obstacle
        if maze[ny][nx] == '#':
            print(f"P2-only: Hit rock at ({nx},{ny}) on step {step+1}. FAIL.")
            return False, path
        x, y = nx, ny
        path.append((x, y))
    print("P2-only: Failed to reach goal within step limit.")
    return False, path

def main():
    random.seed(42)  # reproducible for demo
    maze, start, goal = generate_maze(width=8, height=8, obstacle_ratio=0.25)
    
    print("Generated Maze (S=start, G=goal, #=rock, .=open):")
    print_maze(maze)
    
    # Run Deadband Protocol
    success1, path1 = deadband_protocol(maze, start, goal)
    if success1:
        print("\nDeadband path (cells):", path1)
        print("\nMaze with Deadband path (*):")
        print_maze(maze, set(path1))
    
    # Run Unconstrained Greedy
    success2, path2 = unconstrained_greedy(maze, start, goal)
    if success2:
        print("\nUnconstrained path (cells):", path2)
        print("\nMaze with Unconstrained path (*):")
        print_maze(maze, set(path2))
    
    # Summary
    print("\n=== SUMMARY ===")
    print(f"Deadband Protocol (P0→P1→P2): {'SUCCESS' if success1 else 'FAIL'}")
    print(f"Unconstrained Greedy (P2-only): {'SUCCESS' if success2 else 'FAIL'}")
    print("\nNote: In constrained environments, P2-only fails with high probability.")
    print("Reference: constraint_sim_final.json shows 0% success for constrained, 50% for deadband.")

if __name__ == "__main__":
    main()
```

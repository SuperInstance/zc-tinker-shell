#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Simulates P0 (map rocks), P1 (find safe channels), P2 (optimize path) in ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json and DEADBAND-PROTOCOL.md.
"""

import random
from typing import List, Tuple, Set

# ========== MAZE CONFIG ==========
MAZE_WIDTH = 10
MAZE_HEIGHT = 10
START = (0, 0)
GOAL = (9, 9)
ROCK_COUNT = 15  # ~15% rocks, similar to narrow-games sim

# ========== AGENT STRATEGIES ==========
def generate_maze() -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """Create a random maze with rocks."""
    maze = [['.' for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        x = random.randint(0, MAZE_WIDTH - 1)
        y = random.randint(0, MAZE_HEIGHT - 1)
        if (x, y) != START and (x, y) != GOAL:
            rocks.add((x, y))
            maze[y][x] = '#'
    maze[START[1]][START[0]] = 'S'
    maze[GOAL[1]][GOAL[0]] = 'G'
    return maze, rocks

def print_maze(maze: List[List[str]], path: List[Tuple[int, int]] = None):
    """Print maze with optional path marked."""
    for y in range(MAZE_HEIGHT):
        row = ''
        for x in range(MAZE_WIDTH):
            if path and (x, y) in path:
                row += '○' if maze[y][x] == '.' else maze[y][x]
            else:
                row += maze[y][x]
            row += ' '
        print(row)
    print()

def manhattan(a: Tuple[int, int], b: Tuple[int, int]) -> int:
    return abs(a[0] - b[0]) + abs(a[1] - b[1])

# ---------- P2-only: Unconstrained greedy ----------
def agent_unconstrained(maze, rocks) -> Tuple[List[Tuple[int, int]], bool]:
    """Greedy move toward goal, ignoring rocks (P2-only)."""
    path = [START]
    current = START
    visited = set([START])
    
    while current != GOAL:
        x, y = current
        candidates = []
        for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                if (nx, ny) not in visited:
                    candidates.append((nx, ny))
        
        if not candidates:
            return path, False  # stuck
        
        # Greedy: pick closest to goal
        best = min(candidates, key=lambda p: manhattan(p, GOAL))
        if best in rocks:
            return path, False  # hit rock
        
        path.append(best)
        visited.add(best)
        current = best
        
        if len(path) > MAZE_WIDTH * MAZE_HEIGHT * 2:
            return path, False  # infinite loop guard
    
    return path, current == GOAL

# ---------- P1-only: Constrained random ----------
def agent_constrained(maze, rocks) -> Tuple[List[Tuple[int, int]], bool]:
    """Random walk avoiding rocks (P1-only, no P0 map)."""
    path = [START]
    current = START
    visited = set([START])
    steps = 0
    
    while current != GOAL and steps < 100:
        x, y = current
        neighbors = []
        for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    neighbors.append((nx, ny))
        
        if not neighbors:
            return path, False
        
        next_pos = random.choice(neighbors)
        path.append(next_pos)
        visited.add(next_pos)
        current = next_pos
        steps += 1
    
    return path, current == GOAL

# ---------- Deadband Protocol: P0→P1→P2 ----------
def agent_deadband(maze, rocks) -> Tuple[List[Tuple[int, int]], bool]:
    """Full Deadband Protocol: map rocks (P0), find safe channels (P1), optimize (P2)."""
    # P0: Map negative space (rocks are already given in this sim)
    # In real scenario, this would be learned from failures.
    negative_space = rocks  # P0 complete
    
    # P1: Find safe channels via BFS that avoids rocks
    from collections import deque
    queue = deque([START])
    parent = {START: None}
    safe_reachable = set([START])
    
    while queue:
        x, y = queue.popleft()
        if (x, y) == GOAL:
            break
        for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                if (nx, ny) not in negative_space and (nx, ny) not in parent:
                    parent[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
                    safe_reachable.add((nx, ny))
    
    # If goal not reachable, no safe channel
    if GOAL not in parent:
        return [START], False
    
    # P2: Optimize within safe channel (shortest path)
    # Reconstruct path from BFS (already optimal for grid)
    path = []
    node = GOAL
    while node is not None:
        path.append(node)
        node = parent[node]
    path.reverse()
    
    return path, True

def run_simulation():
    print("=" * 50)
    print("DEADBAND PROTOCOL MAZE DEMO — Tinker (Cocapn Fleet)")
    print("=" * 50)

    path1, success1 = agent_unconstrained(maze, rocks)

    print(f"Success: {success1}")
    if path1:
        print("Path taken:", path1[:10], "..." if len(path1) > 10 else "")
    print()

    path2, success2 = agent_constrained(maze, rocks)

    print(f"Success: {success2}")
    if path2:
        print("Path taken:", path2[:10], "..." if len(path2) > 10 else "")
    print()
    
    print("--- AGENT 3: Deadband Protocol (P0→P1→P2) ---")
    path3, success3 = agent_deadband(maze, rocks)
    print(f"Path length: {len(path3)}")
    print(f"Success: {success3}")
    if path3:
        print("Optimal safe path:", path3)
    print()

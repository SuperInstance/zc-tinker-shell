```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set

# ========== CONFIGURATION ==========
MAZE_SIZE = 20
ROCK_COUNT = 64
START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

# Symbols for visualization
SYM_EMPTY = '·'
SYM_ROCK = '█'
SYM_START = 'S'
SYM_GOAL = 'G'
SYM_PATH = '○'
SYM_SAFE = ' '  # inside safe channel

# ========== MAZE GENERATION ==========
def generate_maze(size: int, rocks: int) -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """Create a maze with random rocks (negative space)."""
    maze = [[SYM_EMPTY for _ in range(size)] for _ in range(size)]
    rock_positions = set()
    
    while len(rock_positions) < rocks:
        x = random.randint(0, size - 1)
        y = random.randint(0, size - 1)
        if (x, y) != START and (x, y) != GOAL:
            rock_positions.add((x, y))
            maze[y][x] = SYM_ROCK
    
    maze[START[1]][START[0]] = SYM_START
    maze[GOAL[1]][GOAL[0]] = SYM_GOAL
    return maze, rock_positions

# ========== DEADBAND PROTOCOL ==========
def deadband_protocol(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> Tuple[List[Tuple[int, int]], Set[Tuple[int, int]]]:
    """
    Implements P0→P1→P2:
    P0: Map rocks (negative space)
    P1: Find safe channels (BFS avoiding rocks)
    P2: Optimize path within safe channel (shortest path)
    """
    size = len(maze)
    
    # P0: Rocks are already mapped in `rocks` set
    
    # P1: Find safe channels using BFS that avoids rocks
    # Safe channel = all reachable cells without hitting rocks
    queue = [START]
    visited = {START}
    safe_cells = {START}
    
    # 4-direction movements
    directions = [(1, 0), (-1, 0), (0, 1), (0, -1)]
    
    while queue:
        x, y = queue.pop(0)
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < size and 0 <= ny < size:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    safe_cells.add((nx, ny))
                    queue.append((nx, ny))
    
    # P2: Optimize path within safe channel (BFS for shortest path)
    if GOAL not in safe_cells:
        return [], safe_cells  # No path to goal
    
    queue = [(START, [START])]
    visited_goal = {START}
    
    while queue:
        (x, y), path = queue.pop(0)
        if (x, y) == GOAL:
            return path, safe_cells
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < size and 0 <= ny < size:
                if (nx, ny) in safe_cells and (nx, ny) not in visited_goal:
                    visited_goal.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    return [], safe_cells

# ========== VISUALIZATION ==========
def print_maze(maze: List[List[str]], path: List[Tuple[int, int]] = None, safe: Set[Tuple[int, int]] = None):
    """Print maze with optional path and safe channel highlighting."""
    size = len(maze)
    display = [row[:] for row in maze]
    
    # Mark safe channel cells (P1 result)
    if safe:
        for x, y in safe:
            if display[y][x] == SYM_EMPTY:
                display[y][x] = SYM_SAFE
    
    # Mark path (P2 result)
    if path:
        for x, y in path[1:-1]:  # Skip start and goal
            if display[y][x] == SYM_EMPTY or display[y][x] == SYM_SAFE:
                display[y][x] = SYM_PATH
    
    # Print with borders
    print("+" + "-" * size + "+")
    for y in range(size):
        print("|", end="")
        for x in range(size):
            print(display[y][x], end="")
        print("|")
    print("+" + "-" * size + "+")

# ========== SIMULATION ==========
def run_simulation():
    """Run one complete demonstration of the Deadband Protocol."""
    print("=" * 60)
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("=" * 60)
    
    # Generate maze
    maze, rocks = generate_maze(MAZE_SIZE, ROCK_COUNT)
    
    print("\n1. INITIAL MAZE (P0: Rocks are negative space)")
    print(f"   Rocks: {len(rocks)}, Start: {START}, Goal: {GOAL}")
    print_maze(maze)
    
    # Apply Deadband Protocol
    path, safe_cells = deadband_protocol(maze, rocks)
    
    print("\n2. P1: SAFE CHANNELS (where agent CAN move)")
    print(f"   Safe cells: {len(safe_cells)} / {MAZE_SIZE * MAZE_SIZE}")
    print_maze(maze, safe=safe_cells)
    
    print("\n3. P2: OPTIMAL PATH WITHIN SAFE CHANNEL")
    if path:
        print(f"   Path length: {len(path)-1} steps (optimal: 38 for Manhattan distance)")
        print_maze(maze, path=path, safe=safe_cells)
        
        # Validate
        for x, y in path:
            if (x, y) in rocks:
                print("ERROR: Path goes through rock!")
                return False
        print("✓ Path is valid (avoids all rocks)")
        return True
    else:
        print("✗ No path to goal within safe channels")
        return False

# ========== MAIN ==========
if __name__ == "__main__":
    random.seed(42)  # Reproducible for demonstration
    
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json:")
    print("  - Unconstrained (P2-only): 0/50 success")
    print("  - Deadband (P0→P1→P2): 50/50 success")
    print()
    
    success = run_simulation()
    
    print("\n" + "=" * 60)
    print("PROTOCOL SUMMARY:")
    print("  P0: Mapped 64 rocks (negative space)")
    print("  P1: Found safe channels (white spaces)")
    print("  P2: Optimized path within channels (○ marks)")
    print(f"  Result: {'SUCCESS' if success else 'FAILURE'}")
    print("=" * 60)
```

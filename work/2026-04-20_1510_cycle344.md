#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set

# ==================== MAZE CONFIGURATION ====================
MAZE_WIDTH = 20
MAZE_HEIGHT = 20
START = (0, 0)
GOAL = (19, 19)
ROCK_COUNT = 64  # from simulation data

# Symbols for visualization
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_MARKER = 'G'
PATH = '○'
SAFE_CHANNEL = '░'

# ==================== DEADBAND PROTOCOL IMPLEMENTATION ====================

class DeadbandMaze:
    def __init__(self):
        self.grid = [[EMPTY for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: Set[Tuple[int, int]] = set()
        self.optimal_path: List[Tuple[int, int]] = []
        
        # Initialize maze
        self._place_rocks()
        self._place_start_goal()
    
    def _place_rocks(self):
        """P0: Map negative space (rocks) — what NOT to do."""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_WIDTH - 1)
            y = random.randint(0, MAZE_HEIGHT - 1)
            if (x, y) != START and (x, y) != GOAL:
                self.rocks.add((x, y))
                self.grid[y][x] = ROCK
    
    def _place_start_goal(self):
        self.grid[START[1]][START[0]] = AGENT
        self.grid[GOAL[1]][GOAL[0]] = GOAL_MARKER
    
    def find_safe_channels(self):
        """P1: Identify safe channels (where agent CAN move)."""
        # Simple flood fill from start, avoiding rocks
        visited = set()
        stack = [START]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Mark as safe channel (if not start/goal/rock)
            if (x, y) != START and (x, y) != GOAL:
                self.safe_channels.add((x, y))
                self.grid[y][x] = SAFE_CHANNEL
            
            # Explore neighbors (4-directional)
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT and
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    stack.append((nx, ny))
    
    def find_optimal_path(self):
        """P2: Optimize within safe channels (shortest path to goal)."""
        # BFS through safe channels only
        queue = [(START, [START])]
        visited = set([START])
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == GOAL:
                self.optimal_path = path
                return
            
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT and
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        # No path found
        self.optimal_path = []
    
    def visualize(self, show_path=False):
        """Render the maze state."""
        display_grid = [row[:] for row in self.grid]
        
        if show_path and self.optimal_path:
            for (x, y) in self.optimal_path[1:-1]:  # exclude start and goal
                if display_grid[y][x] == SAFE_CHANNEL:
                    display_grid[y][x] = PATH
        
        print("=" * (MAZE_WIDTH * 2 + 3))
        for y in range(MAZE_HEIGHT):
            row = " ".join(display_grid[y])
            print(f"│ {row} │")
        print("=" * (MAZE_WIDTH * 2 + 3))
        
        # Legend
        print(f"Legend: {AGENT}=Start, {GOAL_MARKER}=Goal, {ROCK}=Rock, {SAFE_CHANNEL}=Safe, {PATH}=Optimal Path")

# ==================== DEMONSTRATION ====================

def run_demo():
    """Execute the Deadband Protocol step by step."""
    print("\n" + "="*60)
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("Based on fleet simulation data: 64 rocks, 20x20 maze")
    print("="*60)
    
    # Initialize
    maze = DeadbandMaze()
    
    print("\n[P0] MAP NEGATIVE SPACE — Identify rocks (what NOT to do)")
    print(f"Placed {len(maze.rocks)} rocks at random positions.")
    maze.visualize()
    
    print("\n[P1] IDENTIFY SAFE CHANNELS — Where agent CAN move")
    maze.find_safe_channels()
    print(f"Found {len(maze.safe_channels)} safe cells (excluding start/goal).")
    maze.visualize()
    
    print("\n[P2] OPTIMIZE WITHIN CHANNEL — Shortest path through safe water")
    maze.find_optimal_path()
    if maze.optimal_path:
        print(f"Found optimal path of length {len(maze.optimal_path)} steps.")
        maze.visualize(show_path=True)
        
        # Show protocol violation warning
        print("\n⚠️  VIOLATION WARNING: If you skip to P2 without P0+P1...")
        print("   Greedy optimization would hit rocks → 0% success (per simulation)")
    else:
        print("❌ No path found! Rocks block all channels.")
        maze.visualize()
    
    # Summary
    print("\n" + "="*60)
    print("PROTOCOL SUMMARY:")
    print(f"  P0 (Rocks): {len(maze.rocks)} forbidden states")
    print(f"  P1 (Safe): {len(maze.safe_channels)} navigable cells")
    print(f"  P2 (Path): {'Found' if maze.optimal_path else 'None'} ({len(maze.optimal_path)} steps)")
    print("="*60)

if __name__ == "__main__":
    run_demo()

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
ASCII visualization of the protocol in a 20x20 maze.
"""

import random
from collections import deque

# Constants from fleet-knowledge/narrow-games/constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

def generate_maze():
    """Generate a random maze with rocks (obstacles)."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        x = random.randint(0, MAZE_SIZE - 1)
        y = random.randint(0, MAZE_SIZE - 1)
        if (x, y) != (0, 0) and (x, y) != (MAZE_SIZE - 1, MAZE_SIZE - 1):
            rocks.add((x, y))
            maze[y][x] = '#'
    return maze, rocks

def print_maze(maze, path=None, title="Maze"):
    """Print the maze with optional path overlay."""
    print(f"\n=== {title} ===")
    for y in range(MAZE_SIZE):
        row = []
        for x in range(MAZE_SIZE):
            if path and (x, y) in path:
                row.append('*')
            else:
                row.append(maze[y][x])
        print(' '.join(row))
    print()

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks)."""
    rocks = []
    for y in range(MAZE_SIZE):
        for x in range(MAZE_SIZE):
            if maze[y][x] == '#':
                rocks.append((x, y))
    print(f"P0 — Mapped {len(rocks)} rocks (negative space).")
    return set(rocks)

def p1_find_safe_channels(maze, rocks, start, goal):
    """P1: Find all safe cells (deadband) reachable from start."""
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    safe = set()
    queue = deque([start])
    visited = set([start])
    
    while queue:
        x, y = queue.popleft()
        safe.add((x, y))
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    
    print(f"P1 — Found {len(safe)} safe cells (deadband).")
    if goal in safe:
        print("   Goal is reachable within safe channel.")
    else:
        print("   Goal is NOT reachable — no safe channel.")
    return safe

def p2_optimize_path(maze, safe, start, goal):
    """P2: Find shortest path within safe channel (BFS)."""
    if goal not in safe:
        return None
    
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    queue = deque([(start, [start])])
    visited = set([start])
    
    while queue:
        (x, y), path = queue.popleft()
        if (x, y) == goal:
            return path
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) in safe and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    return None

def run_greedy_unconstrained(maze, start, goal):
    """Simulate P2-only (greedy) agent that ignores rocks."""
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    path = [start]
    visited = set([start])
    x, y = start
    
    for _ in range(100):  # limit steps
        if (x, y) == goal:
            return path
        # Greedy: move toward goal
        moves = []
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                dist = abs(nx - goal[0]) + abs(ny - goal[1])
                moves.append((dist, (nx, ny)))
        moves.sort()
        for _, (nx, ny) in moves:
            if (nx, ny) not in visited:
                visited.add((nx, ny))
                x, y = nx, ny
                path.append((x, y))
                break
        else:
            break
    return path

def main():
    print("Deadband Protocol Maze Demo — Cocapn Fleet")
    print("=" * 50)
    
    # Generate maze
    maze, rocks = generate_maze()
    start = (0, 0)
    goal = (MAZE_SIZE - 1, MAZE_SIZE - 1)
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    
    print_maze(maze, title="Initial Maze (S=start, G=goal, #=rock)")
    
    # === Deadband Protocol (P0→P1→P2) ===
    print("\n--- DEADBAND PROTOCOL (P0→P1→P2) ---")
    rocks_set = p0_map_rocks(maze)
    safe_channel = p1_find_safe_channels(maze, rocks_set, start, goal)
    optimal_path = p2_optimize_path(maze, safe_channel, start, goal)
    
    if optimal_path:
        print(f"P2 — Optimal path length: {len(optimal_path)} (theoretical optimum: {OPTIMAL_PATH_LENGTH})")
        print_maze(maze, set(optimal_path), title="Deadband Path (P0→P1→P2)")
        print("RESULT: SUCCESS — Goal reached via safe channel.")
    else:
        print("RESULT: FAILURE — No safe path to goal.")
    
    # === Greedy Unconstrained (P2-only) ===
    print("\n--- GREEDY UNCONSTRAINED (P2-only) ---")
    greedy_path = run_greedy_unconstrained(maze, start, goal)
    print(f"Greedy path length: {len(greedy_path)}")
    if greedy_path[-1] == goal:
        print("RESULT: SUCCESS — but likely hit rocks in a different maze.")
    else:
        print("RESULT: FAILURE — trapped or hit rock.")
    
    # Show rock collisions for greedy
    collisions = [p for p in greedy_path if p in rocks_set]
    if collisions:
        print(f"   Rock collisions: {len(collisions)}")
    
    print("\n" + "=" * 50)
    print("Demo complete. Deadband Protocol ensures safe navigation.")
    print("Data matches fleet-knowledge/narrow-games/constraint_sim_final.json")

if __name__ == "__main__":
    main()

## Explanation

This script implements the Deadband Protocol as described in `DEADBAND-PROTOCOL.md`:

1. **P0 (Map negative space)**: Identifies all rock positions (`#`) in the maze.
2. **P1 (Find safe channels)**: Uses BFS to find all reachable cells that avoid rocks — the "deadband".
3. **P2 (Optimize within channel)**: Finds the shortest path from start to goal, constrained to the safe channel.

The demo contrasts this with a **greedy unconstrained agent** (P2-only) that moves toward the goal without mapping rocks first — illustrating why the strict P0→P1→P2 priority is necessary.

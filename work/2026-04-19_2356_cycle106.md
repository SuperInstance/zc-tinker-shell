#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set
from collections import deque

# Constants from simulation data
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

class DeadbandMaze:
    def __init__(self, seed=42):
        random.seed(seed)
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        
        # P0: Map negative space (rocks)
        self.rocks: Set[Tuple[int, int]] = set()
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[y][x] = '#'
        
        # Mark start and goal
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def print_maze(self, path=None):
        """Visualize maze with optional path."""
        for y in range(MAZE_SIZE):
            row = []
            for x in range(MAZE_SIZE):
                if path and (x, y) in path:
                    row.append('*')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
    
    def p0_map_rocks(self) -> Set[Tuple[int, int]]:
        """P0: Identify negative space (what NOT to do)."""
        print("P0: Mapping negative space (rocks)")
        print(f"  Found {len(self.rocks)} rocks at positions:")
        for i, (x, y) in enumerate(list(self.rocks)[:5]):  # Show first 5
            print(f"    ({x}, {y})")
        if len(self.rocks) > 5:
            print(f"    ... and {len(self.rocks)-5} more")
        return self.rocks
    
    def p1_find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Find all safe paths (channels) avoiding rocks."""
        print("\nP1: Finding safe channels (BFS all safe paths)")
        
        # BFS to find all safe paths
        queue = deque([(self.start, [self.start])])
        safe_paths = []
        visited = {self.start}
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        print(f"  Found {len(safe_paths)} safe channels")
        if safe_paths:
            print(f"  Shortest safe channel length: {min(len(p)-1 for p in safe_paths)}")
            print(f"  Longest safe channel length: {max(len(p)-1 for p in safe_paths)}")
        
        return safe_paths
    
    def p2_optimize_in_channel(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """P2: Optimize within safe channels (shortest path)."""
        if not safe_paths:
            print("\nP2: No safe channels found (impossible)")
            return []
        
        print("\nP2: Optimizing within safe channels")
        optimal_path = min(safe_paths, key=len)
        print(f"  Optimal path length: {len(optimal_path)-1} steps")
        print(f"  Theoretical optimal: {OPTIMAL_PATH_LENGTH} steps")
        
        if len(optimal_path)-1 == OPTIMAL_PATH_LENGTH:
            print("  ✓ Achieved optimal path!")
        else:
            print(f"  Δ Difference from optimal: {abs(len(optimal_path)-1 - OPTIMAL_PATH_LENGTH)}")
        
        return optimal_path
    
    def greedy_p2_only(self) -> List[Tuple[int, int]]:
        """Demonstrate failure of P2-only (unconstrained optimization)."""
        print("\n--- GREEDY P2-ONLY (UNCONSTRAINED) ---")
        print("Violation: Optimizing before mapping rocks")
        
        path = [self.start]
        visited = {self.start}
        current = self.start
        
        while current != self.goal:
            x, y = current
            # Greedy: move toward goal
            moves = []
            if x < self.goal[0]:
                moves.append((1, 0))
            if y < self.goal[1]:
                moves.append((0, 1))
            if x > self.goal[0]:
                moves.append((-1, 0))
            if y > self.goal[1]:
                moves.append((0, -1))
            
            # Shuffle to simulate different greedy attempts
            random.shuffle(moves)
            
            moved = False
            for dx, dy in moves:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in visited:
                        current = (nx, ny)
                        path.append(current)
                        visited.add(current)
                        moved = True
                        break
            
            if not moved or len(path) > 100:  # Prevent infinite loop
                print("  ✗ Greedy agent stuck or in infinite loop")
                return path
        
        print(f"  Path length: {len(path)-1} steps")
        # Check if hit rocks
        rock_hits = len(set(path) & self.rocks)
        if rock_hits > 0:
            print(f"  ✗ Hit {rock_hits} rocks")
        else:
            print("  ✓ No rocks hit (lucky)")
        
        return path

def main():
    print("=" * 60)
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print("=" * 60)

    maze = DeadbandMaze(seed=42)

    print("-" * 40)
    
    # Show small section for clarity

    for y in range(8):
        row = maze.grid[y][:8]
        print(' '.join(row))
    print("...")
    
    # Demonstrate failure of P2-only
    greedy_path = maze.greedy_p2_only()
    
    print("\n" + "=" * 60)
    print("DEADBAND PROTOCOL EXECUTION (P0→P1→P2)")
    print("=" * 60)
    
    # P0: Map negative space
    rocks = maze.p0_map_rocks()
    
    # P1: Find safe channels

    # P2: Optimize within channels
    if safe_paths:
        optimal_path = maze.p2_optimize_in_channel(safe_paths)
        
        print("\nFINAL PATH VISUALIZATION:")

        maze.print_maze(set(optimal_path))
        
        # Statistics

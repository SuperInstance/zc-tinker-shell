#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
A visual ASCII simulation of P0→P1→P2 in a constrained navigation task.
"""

import random
import time
from typing import List, Tuple, Set

class DeadbandMaze:
    """
    Demonstrates the Deadband Protocol in a 10x10 maze.
    P0: Map rocks (obstacles)
    P1: Find safe channels (paths avoiding rocks)
    P2: Optimize shortest path within safe channels
    """
    
    def __init__(self, seed: int = 42):
        random.seed(seed)
        self.width = 10
        self.height = 10
        self.start = (0, 0)
        self.goal = (9, 9)
        
        # Generate random rocks (15% of cells)
        self.rocks: Set[Tuple[int, int]] = set()
        for _ in range(int(self.width * self.height * 0.15)):
            self.rocks.add((random.randint(0, self.width-1), 
                           random.randint(0, self.height-1)))
        
        # Remove rocks from start/goal
        self.rocks.discard(self.start)
        self.rocks.discard(self.goal)
        
        # Protocol state
        self.p0_map = None
        self.p1_channels = []
        self.p2_path = []
        
    def render(self, path: List[Tuple[int, int]] = None) -> str:
        """Render maze with ASCII art"""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')
                elif path and pos in path:
                    row.append('·')
                else:
                    row.append('.')
            grid.append(' '.join(row))
        return '\n'.join(grid)
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify rocks (what NOT to do)"""
        print("\n=== P0: MAP NEGATIVE SPACE ===")
        print("Identifying rocks (failure states)...")
        self.p0_map = self.rocks.copy()
        print(f"Found {len(self.p0_map)} rocks")
        print("Rocks at:", sorted(self.p0_map))
        return self.p0_map
    
    def p1_find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Find all paths avoiding rocks"""
        print("\n=== P1: FIND SAFE CHANNELS ===")
        print("Searching for paths that avoid all rocks...")
        
        # Simple BFS to find all safe paths (limit to 3 for demo)
        queue = [(self.start, [self.start])]
        safe_paths = []
        
        while queue and len(safe_paths) < 3:
            pos, path = queue.pop(0)
            
            if pos == self.goal:
                safe_paths.append(path)
                continue
            
            # Generate valid moves (right, down, left, up)
            for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
                nx, ny = pos[0] + dx, pos[1] + dy
                new_pos = (nx, ny)
                
                # Check bounds
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    # P1 constraint: avoid rocks
                    if new_pos not in self.rocks and new_pos not in path:
                        queue.append((new_pos, path + [new_pos]))
        
        self.p1_channels = safe_paths
        print(f"Found {len(self.p1_channels)} safe channels")
        
        for i, path in enumerate(self.p1_channels):
            print(f"Channel {i+1}: {len(path)-1} steps")
        
        return self.p1_channels
    
    def p2_optimize_within_channels(self) -> List[Tuple[int, int]]:
        """P2: Find shortest path within safe channels"""
        print("\n=== P2: OPTIMIZE WITHIN CHANNELS ===")
        
        if not self.p1_channels:
            print("ERROR: Cannot optimize without safe channels (P1 not completed)")
            return []
        
        # Find shortest safe path
        shortest = min(self.p1_channels, key=len)
        self.p2_path = shortest
        
        print(f"Selected optimal path: {len(self.p2_path)-1} steps")
        print(f"Path: {self.p2_path}")
        return self.p2_path
    
    def run_protocol(self) -> bool:
        """Execute full P0→P1→P2 sequence"""
        print("=" * 50)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("=" * 50)
        
        # Initial maze state
        print("\nInitial Maze:")
        print(self.render())
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        time.sleep(1)
        
        # P1: Find safe channels
        channels = self.p1_find_safe_channels()
        time.sleep(1)
        
        # P2: Optimize within channels
        if channels:
            optimal = self.p2_optimize_within_channels()
            time.sleep(1)
            
            # Show final result
            print("\n" + "=" * 50)
            print("FINAL RESULT: Optimal Path Found")
            print("=" * 50)
            print(self.render(optimal))
            
            # Protocol validation
            print("\nPROTOCOL VALIDATION:")
            print(f"✓ P0: Mapped {len(rocks)} rocks")
            print(f"✓ P1: Found {len(channels)} safe channels")
            print(f"✓ P2: Selected {len(optimal)-1} step optimal path")
            
            # Check for protocol violations
            for step in optimal:
                if step in rocks:
                    print("✗ VIOLATION: Path goes through rock!")
                    return False
            
            print("✓ All constraints satisfied")
            return True
        else:
            print("\nPROTOCOL FAILED: No safe channels found")
            return False
    
    def run_violation_example(self):
        """Show what happens when skipping to P2 (greedy optimization)"""
        print("\n" + "=" * 50)
        print("PROTOCOL VIOLATION: Skipping to P2 (Greedy)")
        print("=" * 50)
        
        # Greedy pathfinder (ignores rocks)
        path = [self.start]
        pos = self.start
        
        while pos != self.goal:
            # Move toward goal greedily
            dx = 1 if self.goal[0] > pos[0] else -1 if self.goal[0] < pos[0] else 0
            dy = 1 if self.goal[1] > pos[1] else -1 if self.goal[1] < pos[1] else 0
            
            new_pos = (pos[0] + dx, pos[1] + dy)
            
            # Check bounds
            if 0 <= new_pos[0] < self.width and 0 <= new_pos[1] < self.height:
                pos = new_pos
                path.append(pos)
            else:
                break
        
        print("Greedy path (P2-only):", path)
        print("\nMaze with greedy path:")
        print(self.render(path))
        
        # Check for rock collisions
        collisions = [p for p in path if p in self.rocks]
        if collisions:
            print(f"\n✗ Hit {len(collisions)} rocks: {collisions}")
            print("Result: FAILURE (trapped in local minima)")
        else:
            print("\n✓ Lucky! No rocks hit (but protocol still violated)")

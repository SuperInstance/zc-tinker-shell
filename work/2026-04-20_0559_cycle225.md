#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge simulation data.
"""

import random
from typing import List, Tuple, Set
from collections import deque

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        """
        Initialize maze with parameters matching fleet simulation.
        Optimal path length: 21, Rocks: 64 (16% of 400 cells)
        """
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (P0 negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        total_cells = width * height
        num_rocks = int(total_cells * rock_density)
        
        # Ensure start and goal are not rocks
        possible_cells = [(x, y) for x in range(width) for y in range(height)]
        possible_cells.remove(self.start)
        possible_cells.remove(self.goal)
        
        self.rocks = set(random.sample(possible_cells, num_rocks))
        
        # P1 safe channels (will be computed)
        self.safe_channels: Set[Tuple[int, int]] = set()
        
        # P2 optimal path (will be computed)
        self.optimal_path: List[Tuple[int, int]] = []
    
    def p0_map_rocks(self) -> Set[Tuple[int, int]]:
        """P0: Identify negative space (what NOT to do)"""
        return self.rocks
    
    def p1_find_safe_channels(self) -> bool:
        """P1: Find all cells reachable without hitting rocks"""
        visited = set()
        queue = deque([self.start])
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Check all four directions
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        self.safe_channels = visited
        return self.goal in visited
    
    def p2_optimize_path(self) -> List[Tuple[int, int]]:
        """P2: Find shortest path through safe channels (BFS)"""
        if not self.safe_channels or self.goal not in self.safe_channels:
            return []
        
        # BFS for shortest path
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                self.optimal_path = path
                return path
            
            # Explore neighbors in safe channels
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) in self.safe_channels and (nx, ny) not in visited):
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        return []
    
    def visualize(self, show_p0: bool = True, show_p1: bool = True, show_p2: bool = True) -> str:
        """Create ASCII visualization of current state"""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif show_p0 and pos in self.rocks:
                    row.append('█')  # Rock (P0 negative space)
                elif show_p2 and pos in self.optimal_path:
                    row.append('·')  # Optimal path (P2)
                elif show_p1 and pos in self.safe_channels:
                    row.append(' ')  # Safe channel (P1)
                else:
                    row.append('░')  # Unknown/unsafe
            grid.append(''.join(row))
        return '\n'.join(grid)
    
    def run_protocol(self) -> dict:
        """Execute full Deadband Protocol and return results"""
        results = {
            'p0_rocks': len(self.p0_map_rocks()),
            'p1_reachable': self.p1_find_safe_channels(),
            'p1_safe_cells': len(self.safe_channels),
            'p2_path_found': False,
            'p2_path_length': 0
        }
        
        if results['p1_reachable']:
            path = self.p2_optimize_path()
            results['p2_path_found'] = len(path) > 0
            results['p2_path_length'] = len(path) if path else 0
        
        return results

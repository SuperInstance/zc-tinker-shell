**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on the provided fleet knowledge, specifically the `constraint_sim_final.json` results and the formal Deadband Protocol steps, I will implement a concrete simulation. The data shows: unconstrained greedy (P2-only) fails (0% success), while deadband protocol (P0→P1→P2) achieves 50% success at optimal cost (21 steps). I'll model this.

**Approach:**
1. **P0 (Map negative space):** Identify all maze cells that are obstacles ('rocks').
2. **P1 (Identify safe channels):** From the start, find all reachable cells that avoid rocks (breadth-first search).
3. **P2 (Optimize within channel):** From the safe reachable set, compute shortest path to goal using A* (constrained to safe cells only).

**Implementation:** Python script with ASCII visualization, step-by-step protocol logging, and metrics matching the simulation results.

**Output:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 strict priority in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque
import heapq

# ------------------------------------------------------------
# 1. MAZE GENERATION (matching simulation parameters)
# ------------------------------------------------------------
def generate_maze(width=20, height=20, rock_density=0.16):
    """
    Generates a maze with start (S), goal (G), and rocks (#).
    Rock count ~64 for 20x20 (0.16 density) per constraint_sim_final.json.
    """
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks
    rocks = 0
    target_rocks = int(width * height * rock_density)
    while rocks < target_rocks:
        x = random.randint(0, width-1)
        y = random.randint(0, height-1)
        if maze[y][x] == '.':
            maze[y][x] = '#'
            rocks += 1
    
    # Place start (top-left) and goal (bottom-right)
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    
    return maze

def print_maze(maze, path=None):
    """Print maze with optional path overlay."""
    for y, row in enumerate(maze):
        line = []
        for x, cell in enumerate(row):
            if path and (x, y) in path:
                line.append('*')
            else:
                line.append(cell)
        print(''.join(line))
    print()

# ------------------------------------------------------------
# 2. DEADBAND PROTOCOL IMPLEMENTATION
# ------------------------------------------------------------
def p0_map_negative_space(maze):
    """
    P0: Identify all rocks (states where failure probability > threshold).
    Returns set of (x,y) coordinates that are obstacles.
    """
    rocks = set()
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(maze, start, rocks):
    """
    P1: From start, find all reachable cells that avoid rocks.
    Returns set of safe (x,y) coordinates (the deadband).
    """
    width, height = len(maze[0]), len(maze)
    safe = set()
    queue = deque([start])
    visited = set([start])
    
    while queue:
        x, y = queue.popleft()
        safe.add((x, y))
        
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    return safe

def p2_optimize_within_channel(maze, start, goal, safe):
    """
    P2: Find shortest path from start to goal constrained to safe cells.
    Uses A* with Manhattan distance heuristic.
    Returns path as list of (x,y) or empty list if no path.
    """
    if goal not in safe:
        return []  # Goal not reachable within safe channel
    
    width, height = len(maze[0]), len(maze)
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            path.reverse()
            return path
        
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            neighbor = (nx, ny)
            if 0 <= nx < width and 0 <= ny < height:
                if neighbor not in safe:
                    continue  # Violates P1 constraint
                
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return []  # No path found

def unconstrained_greedy(maze, start, goal):
    """
    P2-only (violation): Greedy best-first search ignoring rocks.
    This is what fails 100% in simulation.
    """
    width, height = len(maze[0]), len(maze)
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    visited = set([start])
    
    while open_set:
        _, current = heapq.heappop(open_set)
        
        if current == goal:
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            path.reverse()
            return path
        
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            neighbor = (nx, ny)
            if 0 <= nx < width and 0 <= ny < height:
                if neighbor in visited:
                    continue
                # CRITICAL: Doesn't check for rocks!
                visited.add(neighbor)
                came_from[neighbor] = current
                priority = heuristic(neighbor, goal)
                heapq.heappush(open_set, (priority, neighbor))
    
    return []

# ------------------------------------------------------------
# 3. SIMULATION AND VISUALIZATION
# ------------------------------------------------------------
def run_simulation():
    print("=" * 60)
    print("DEADBAND PROTOCOL MAZE DEMO — Tinker (Cocapn Fleet)")
    print("=" * 60)
    
    # Generate maze matching simulation parameters
    maze = generate_maze(20, 20, 0.16)
    start = (0, 0)
    goal = (19, 19)
    
    print("\n[MAZE] 20x20 (S=start, G=goal, #=rock, .=empty)")
    print("Rock density: 16% (~64 rocks)")
    print_maze(maze)
    
    # --------------------------------------------------------
    # P0: MAP NEGATIVE SPACE
    # --------------------------------------------------------
    print("[P0] Map negative space (identify rocks)")
    rocks = p0_map_negative_space(maze)
    print(f"  Found {len(rocks)} rocks (failure states)")
    
    # --------------------------------------------------------
    # P1: IDENTIFY SAFE CHANNELS
    # --------------------------------------------------------
    print("[P1] Identify safe channels (reachable cells avoiding rocks)")
    safe = p1_find_safe_channels(maze, start, rocks)
    print(f"  Safe reachable cells: {len(safe)}")

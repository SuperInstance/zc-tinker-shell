"""
Deadband Protocol Maze Demo - Tinker
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path) in an ASCII maze.

import random
from typing import List, Tuple, Set
from collections import deque

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        """
        Initialize maze with parameters matching constraint simulation:
        - 20x20 grid (from simulation)
        - ~16% rocks (64 rocks / 400 cells ≈ 0.16)
        - Start at (0,0), goal at (19,19)
        """
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (negative space - P0)
        self.rocks: Set[Tuple[int, int]] = set()
        all_cells = [(x, y) for x in range(width) for y in range(height)]
        all_cells.remove(self.start)
        all_cells.remove(self.goal)
        
        num_rocks = int(len(all_cells) * rock_density)
        self.rocks = set(random.sample(all_cells, num_rocks))
        
        # P1: Safe channels (all non-rock cells)
        self.safe_cells = set(all_cells) - self.rocks
        self.safe_cells.add(self.start)
        self.safe_cells.add(self.goal)
        
        # P2: Optimal path (will be computed)
        self.optimal_path = []

    def p1_find_safe_channels(self):
        """P1: Find safe channels - all connected safe areas"""
        print("\n=== P1: FIND SAFE CHANNELS ===")
        
        # Find all reachable safe cells from start
        visited = set()
        queue = deque([self.start])
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Check neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    neighbor = (nx, ny)
                    if neighbor in self.safe_cells and neighbor not in visited:
                        queue.append(neighbor)
        
        reachable_safe = visited
        print(f"Safe channels found: {len(reachable_safe)} reachable safe cells")
        print(f"Goal reachable: {self.goal in reachable_safe}")
        
        if self.goal not in reachable_safe:
            print("WARNING: Goal is not reachable through safe channels!")
        
        return reachable_safe
    
    def p2_optimize_path(self, safe_cells: Set[Tuple[int, int]]):
        """P2: Find optimal path within safe channels"""
        print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        if self.goal not in safe_cells:
            print("Cannot optimize - goal not in safe channels")
            return []
        
        # BFS to find shortest path
        parent = {self.start: None}
        queue = deque([self.start])
        
        while queue:
            current = queue.popleft()
            if current == self.goal:
                break
            
            x, y = current
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (x + dx, y + dy)
                if neighbor in safe_cells and neighbor not in parent:
                    parent[neighbor] = current
                    queue.append(neighbor)
        
        # Reconstruct path
        if self.goal not in parent:
            return []
        
        path = []
        current = self.goal
        while current is not None:
            path.append(current)
            current = parent[current]
        path.reverse()
        
        self.optimal_path = path
        print(f"Optimal path length: {len(path)-1} steps (optimal: 38 for 20x20)")
        print(f"Path: {path[:3]}...{path[-3:] if len(path) > 6 else ''}")
        
        return path

#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import sys
import time

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64          # rocks (obstacles)
OPTIMAL_PATH_LEN = 21    # shortest possible path length
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

def generate_maze(size, rocks):
    """Generate a random maze with rocks (obstacles)."""
    maze = [['.' for _ in range(size)] for _ in range(size)]
    maze[0][0] = 'S'  # start
    maze[size-1][size-1] = 'G'  # goal
    
    placed = 0
    while placed < rocks:
        x = random.randint(0, size-1)
        y = random.randint(0, size-1)
        if maze[x][y] == '.':
            maze[x][y] = '#'
            placed += 1
    return maze

def print_maze(maze, title="", path=None, rocks=None, channels=None):
    """Print maze with optional overlays."""
    print(f"\n=== {title} ===")
    for i in range(len(maze)):
        row = []
        for j in range(len(maze[i])):
            cell = maze[i][j]
            if path and (i, j) in path:
                row.append('*')
            elif channels and (i, j) in channels:
                row.append('~')
            elif rocks and (i, j) in rocks:
                row.append('#')
            else:
                row.append(cell)
        print(' '.join(row))
    print()

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks)."""
    rocks = []
    for i in range(len(maze)):
        for j in range(len(maze[i])):
            if maze[i][j] == '#':
                rocks.append((i, j))
    return rocks

def p1_find_safe_channels(maze, rocks):
    """P1: Find all cells not adjacent to rocks (deadband channels)."""
    safe = []
    for i in range(len(maze)):
        for j in range(len(maze[i])):
            if maze[i][j] == '#':
                continue
            # Check 8-neighbor adjacency
            is_safe = True
            for dx in [-1, 0, 1]:
                for dy in [-1, 0, 1]:
                    ni, nj = i + dx, j + dy
                    if 0 <= ni < len(maze) and 0 <= nj < len(maze[0]):
                        if (ni, nj) in rocks:
                            is_safe = False
                            break
                if not is_safe:
                    break
            if is_safe:
                safe.append((i, j))
    return safe

def p2_optimize_path(maze, start, goal, safe_channels):
    """P2: Find shortest path within safe channels (A* simplified)."""
    # Simplified BFS since we only need demonstration
    from collections import deque
    
    if start not in safe_channels or goal not in safe_channels:
        return None
    
    queue = deque([(start, [start])])
    visited = set([start])
    
    while queue:
        (x, y), path = queue.popleft()
        if (x, y) == goal:
            return path
        
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < len(maze) and 0 <= ny < len(maze[0]):
                if (nx, ny) in safe_channels and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    return None

def run_deadband_protocol():
    """Execute full P0→P1→P2 protocol with visualization."""
    print("DEADBAND PROTOCOL DEMO")
    print("=" * 40)
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}, Optimal: {OPTIMAL_PATH_LEN} steps")
    
    # Generate maze
    maze = generate_maze(MAZE_SIZE, ROCK_COUNT)
    print_maze(maze, "Initial Maze")
    time.sleep(1)
    
    # P0: Map negative space
    rocks = p0_map_rocks(maze)
    print(f"P0: Mapped {len(rocks)} rocks (negative space)")
    print_maze(maze, "P0 - Rocks (Negative Space)", rocks=rocks)
    time.sleep(1)
    
    # P1: Find safe channels
    safe_channels = p1_find_safe_channels(maze, rocks)
    print(f"P1: Found {len(safe_channels)} safe channel cells (deadband)")
    print_maze(maze, "P1 - Safe Channels (~)", channels=safe_channels)
    time.sleep(1)
    
    # P2: Optimize within channels
    path = p2_optimize_path(maze, (0,0), GOAL, safe_channels)
    
    if path:
        print(f"P2: Found path of length {len(path)-1} (optimal: {OPTIMAL_PATH_LEN})")
        print_maze(maze, "P2 - Optimal Path (*)", path=path)
        
        # Validate against fleet data
        if len(path)-1 == OPTIMAL_PATH_LEN:
            print("✓ MATCHES FLEET DATA: Deadband success at optimal speed")
        else:
            print(f"Note: Path length {len(path)-1} vs optimal {OPTIMAL_PATH_LEN}")
    else:
        print("P2: No path found within safe channels")
    
    # Demonstrate protocol violation (P2-only greedy)
    print("\n" + "=" * 40)
    print("PROTOCOL VIOLATION DEMO: P2-only (greedy optimization)")
    print("=" * 40)
    
    # Simple greedy that ignores rocks
    greedy_path = []
    x, y = 0, 0
    for _ in range(MAZE_SIZE*2):  # Limit steps
        greedy_path.append((x, y))
        if (x, y) == GOAL:
            break
        # Move toward goal, ignoring obstacles
        if x < GOAL[0]:
            x += 1
        elif y < GOAL[1]:
            y += 1
    
    # Check if hit rock
    hit_rock = any(maze[x][y] == '#' for (x, y) in greedy_path)
    
    print_maze(maze, "Greedy Path (Ignores P0/P1)", path=greedy_path)
    if hit_rock:
        print("✗ VIOLATION RESULT: Hit rock (failure)")
        print("Matches fleet data: 0/50 success for unconstrained optimization")
    else:
        print("Note: Got lucky (low probability as complexity increases)")
    
    return path is not None

def main():
    """Run multiple trials to match fleet statistics."""
    print("Running 5 trials to demonstrate protocol effectiveness...")
    print("-" * 40)
    
    successes = 0
    trials = 5
    
    for trial in range(trials):
        print(f"\nTrial {trial+1}/{trials}")
        print("-" * 20)
        success = run_deadband_protocol()
        if success:
            successes += 1
        time.sleep(0.5)
    
    print("\n" + "=" * 40)
    print("SUMMARY")
    print("=" * 40)
    print(f"Deadband Protocol (P0→P1→P2): {successes}/{trials} successes")
    print(f"Expected from fleet data: 50/50 successes")
    print("\nKey insight: Protocol prevents hitting rocks by enforcing")
    print("P0 (map obstacles) and P1 (find safe channels) before P2 (optimize).")

if __name__ == "__main__":
    main()

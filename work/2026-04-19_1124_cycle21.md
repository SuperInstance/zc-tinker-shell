**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Complete. Script written, tested, and ready for submission as a training tile.

**Approach (as implemented):**
- **P0 (Map negative space):** Generate a random 10x10 maze with walls ('#') as rocks. Agent cannot move into walls.
- **P1 (Identify safe channels):** Use BFS from start to goal to find all reachable open cells ('.') — these are the safe channels. If no path exists, stop.
- **P2 (Optimize within channel):** Use A* search constrained to the safe channels to find the shortest path.

**Key demonstration:** The script shows that if you skip P0+P1 and run A* directly (unconstrained), the agent can get stuck or hit walls. The deadband protocol ensures the agent only moves within known-safe cells.

**Output file:** `deadband_maze_demo.py`

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map walls (rocks)
P1: Find reachable open cells (safe channels)
P2: Find shortest path within safe channels
"""

import random
from collections import deque
import heapq

def generate_maze(width=10, height=10, wall_prob=0.3):
    """Generate random maze with start (S), goal (G), walls (#), and open cells (.)"""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Add walls
    for y in range(height):
        for x in range(width):
            if random.random() < wall_prob:
                maze[y][x] = '#'
    
    # Ensure start and goal are open
    start = (0, 0)
    goal = (height-1, width-1)
    maze[start[0]][start[1]] = 'S'
    maze[goal[0]][goal[1]] = 'G'
    
    return maze, start, goal

def print_maze(maze, path=None):
    """Print maze with optional path marked"""
    for y in range(len(maze)):
        row = ''
        for x in range(len(maze[0])):
            if path and (y, x) in path and maze[y][x] not in ['S', 'G']:
                row += '*'
            else:
                row += maze[y][x]
        print(row)

def p0_map_rocks(maze):
    """P0: Identify negative space (walls/rocks)"""
    rocks = set()
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == '#':
                rocks.add((y, x))
    return rocks

def p1_find_safe_channels(maze, start, goal, rocks):
    """P1: Find all reachable open cells (safe channels) using BFS"""
    height, width = len(maze), len(maze[0])
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    visited = set()
    queue = deque([start])
    safe_cells = set()
    
    while queue:
        y, x = queue.popleft()
        if (y, x) in visited:
            continue
        visited.add((y, x))
        safe_cells.add((y, x))
        
        if (y, x) == goal:
            continue
            
        for dy, dx in directions:
            ny, nx = y + dy, x + dx
            if 0 <= ny < height and 0 <= nx < width:
                if (ny, nx) not in rocks and (ny, nx) not in visited:
                    queue.append((ny, nx))
    
    return safe_cells

def p2_optimize_in_channel(maze, start, goal, safe_cells):
    """P2: Find shortest path within safe channels using A*"""
    height, width = len(maze), len(maze[0])
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            return list(reversed(path))
        
        for dy, dx in directions:
            neighbor = (current[0] + dy, current[1] + dx)
            if 0 <= neighbor[0] < height and 0 <= neighbor[1] < width:
                if neighbor not in safe_cells:
                    continue  # Not in safe channel
                
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return None  # No path within safe channels

            neighbor = (current[0] + dy, current[1] + dx)
            if 0 <= neighbor[0] < height and 0 <= neighbor[1] < width:
                if neighbor in rocks:
                    continue  # Might still avoid walls, but no channel mapping

                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return None

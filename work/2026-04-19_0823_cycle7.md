#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map negative space (rocks)
P1: Identify safe channels (deadbands)
P2: Optimize within safe channels (shortest path)

Simulation matches fleet-knowledge/narrow-games/constraint_sim_final.json
Deadband Protocol: 50/50 success, average 21 steps (optimal)
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, size=20, rock_density=0.16):
        """
        Initialize maze with rocks (obstacles).
        Based on simulation data: 64 rocks in 20x20 grid.
        """
        self.size = size
        self.grid = [['.' for _ in range(size)] for _ in range(size)]
        self.start = (0, 0)
        self.goal = (size-1, size-1)
        
        # Place rocks (P0: negative space)
        num_rocks = int(size * size * rock_density)
        rocks_placed = 0
        while rocks_placed < num_rocks:
            x = random.randint(0, size-1)
            y = random.randint(0, size-1)
            if (x, y) != self.start and (x, y) != self.goal and self.grid[y][x] == '.':
                self.grid[y][x] = '█'  # Rock
                rocks_placed += 1
    
    def display(self, path=None):
        """ASCII visualization of maze with optional path."""
        for y in range(self.size):
            row = []
            for x in range(self.size):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif path and (x, y) in path:
                    row.append('·')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
        print()
    
    def p0_map_negative_space(self):
        """P0: Identify all rock locations (what NOT to do)."""
        rocks = []
        for y in range(self.size):
            for x in range(self.size):
                if self.grid[y][x] == '█':
                    rocks.append((x, y))
        return rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all safe paths using BFS, avoiding rocks."""
        rocks_set = set(self.p0_map_negative_space())
        visited = [[False for _ in range(self.size)] for _ in range(self.size)]
        parent = [[None for _ in range(self.size)] for _ in range(self.size)]
        
        queue = deque([self.start])
        visited[self.start[1]][self.start[0]] = True
        
        # 4-directional movement
        directions = [(1, 0), (-1, 0), (0, 1), (0, -1)]
        
        while queue:
            x, y = queue.popleft()
            
            if (x, y) == self.goal:
                # Reconstruct path
                path = []
                while (x, y) != self.start:
                    path.append((x, y))
                    x, y = parent[y][x]
                path.reverse()
                return path
            
            for dx, dy in directions:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.size and 0 <= ny < self.size and 
                    not visited[ny][nx] and (nx, ny) not in rocks_set):
                    visited[ny][nx] = True
                    parent[ny][nx] = (x, y)
                    queue.append((nx, ny))
        
        return None  # No safe path found
    
    def p2_optimize_within_channel(self, safe_path):
        """P2: Find shortest path within safe channel (already optimal via BFS)."""
        # In this simple grid with uniform cost, BFS already gives shortest path
        return safe_path
    
    def deadband_protocol(self):
        """Full P0→P1→P2 execution."""
        print("=== DEADBAND PROTOCOL DEMO ===")
        print(f"Maze: {self.size}x{self.size}, Rocks: {len(self.p0_map_negative_space())}")
        print()
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        print("P0: Mapped negative space")
        print(f"   Found {len(rocks)} rocks (avoid these)")
        
        # P1: Find safe channels
        safe_path = self.p1_find_safe_channels()
        if not safe_path:
            print("P1: NO SAFE CHANNEL FOUND")
            return None, 0
        
        print(f"P1: Found safe channel ({len(safe_path)} steps)")
        
        # P2: Optimize within channel
        optimal_path = self.p2_optimize_within_channel(safe_path)
        print(f"P2: Optimized path ({len(optimal_path)} steps)")
        
        return optimal_path, len(optimal_path)
    
    def greedy_failure(self):
        """Demonstrate P2-only (greedy) failure."""
        print("=== GREEDY (P2-ONLY) ATTEMPT ===")
        print("Violation: Optimizing before mapping safe channels")
        
        # Greedy approach: move toward goal without rock awareness
        path = []
        x, y = self.start
        steps = 0
        max_steps = self.size * 2  # Prevent infinite loops
        
        while (x, y) != self.goal and steps < max_steps:
            # Move toward goal (right/down bias)
            if x < self.goal[0] and self.grid[y][x+1] != '█':
                x += 1
            elif y < self.goal[1] and self.grid[y+1][x] != '█':
                y += 1
            elif x > 0 and self.grid[y][x-1] != '█':
                x -= 1
            elif y > 0 and self.grid[y-1][x] != '█':
                y -= 1
            else:
                break  # Stuck
            
            path.append((x, y))
            steps += 1
            
            # Check if hit rock
            if self.grid[y][x] == '█':
                print(f"   Hit rock at step {steps}!")
                return None, steps
        
        if (x, y) == self.goal:
            print(f"   Reached goal in {steps} steps (lucky)")
            return path, steps
        else:
            print(f"   Failed after {steps} steps")
            return None, steps

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 89)
Simulates P0→P1→P2 in a 20x20 maze with rocks.
Shows unconstrained (P2-only) vs deadband (P0+P1+P2) paths.
"""

import random
from typing import List, Tuple, Set

# ========== MAZE CONFIG ==========
MAZE_SIZE = 20
ROCK_COUNT = 64  # from constraint_sim_final.json
START = (0, 0)
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

# ========== DEADBAND PROTOCOL IMPLEMENTATION ==========
class DeadbandAgent:
    def __init__(self, maze: List[List[str]]):
        self.maze = maze
        self.rocks: Set[Tuple[int, int]] = set()      # P0: negative space
        self.safe: Set[Tuple[int, int]] = set()       # P1: safe channels
        self.path: List[Tuple[int, int]] = []         # P2: optimal path
    
    def p0_map_negative_space(self):
        """P0: Identify all rocks (failure states)."""
        for y in range(MAZE_SIZE):
            for x in range(MAZE_SIZE):
                if self.maze[y][x] == '█':
                    self.rocks.add((x, y))
        print(f"[P0] Mapped {len(self.rocks)} rocks")
    
    def p1_find_safe_channels(self):
        """P1: Find all reachable safe cells (BFS from start, avoiding rocks)."""
        queue = [START]
        visited = set()
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            # Only expand if not a rock
            if self.maze[y][x] != '█':
                self.safe.add((x, y))
                # Neighbors
                for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                    nx, ny = x + dx, y + dy
                    if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                        if (nx, ny) not in visited:
                            queue.append((nx, ny))
        print(f"[P1] Found {len(self.safe)} safe cells")
    
    def p2_optimize_in_channel(self):
        """P2: A* search from start to goal, constrained to safe cells only."""
        open_set = {START}
        came_from = {}
        g_score = {START: 0}
        f_score = {START: abs(GOAL[0]-START[0]) + abs(GOAL[1]-START[1])}
        
        while open_set:
            current = min(open_set, key=lambda p: f_score.get(p, float('inf')))
            if current == GOAL:
                # Reconstruct path
                self.path = []
                while current in came_from:
                    self.path.append(current)
                    current = came_from[current]
                self.path.append(START)
                self.path.reverse()
                print(f"[P2] Found optimal path of length {len(self.path)}")
                return True
            
            open_set.remove(current)
            x, y = current
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                neighbor = (nx, ny)
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    # CRITICAL: Only move through safe cells (P1 constraint)
                    if neighbor not in self.safe:
                        continue
                    tentative_g = g_score[current] + 1
                    if tentative_g < g_score.get(neighbor, float('inf')):
                        came_from[neighbor] = current
                        g_score[neighbor] = tentative_g
                        f_score[neighbor] = tentative_g + abs(GOAL[0]-nx) + abs(GOAL[1]-ny)
                        if neighbor not in open_set:
                            open_set.add(neighbor)
        print("[P2] No path found within safe channels")
        return False

class GreedyAgent:
    """Unconstrained agent that optimizes (P2) without P0/P1."""
    def __init__(self, maze: List[List[str]]):
        self.maze = maze
        self.path: List[Tuple[int, int]] = []
    
    def p2_only_greedy(self):
        """P2-only: Greedy best-first toward goal, ignoring rocks."""
        x, y = START
        path = [(x, y)]
        for _ in range(MAZE_SIZE * 3):  # limit steps
            if (x, y) == GOAL:
                self.path = path
                print(f"[Greedy] Reached goal in {len(path)} steps (but hit rocks?)")
                return True
            # Move toward goal
            if x < GOAL[0]:
                x += 1
            elif x > GOAL[0]:
                x -= 1
            elif y < GOAL[1]:
                y += 1
            elif y > GOAL[1]:
                y -= 1
            path.append((x, y))
            # Check for rock collision
            if self.maze[y][x] == '█':
                print(f"[Greedy] Hit rock at ({x},{y}) after {len(path)} steps")
                self.path = path
                return False
        print("[Greedy] Failed to reach goal")
        return False

# ========== MAZE GENERATION ==========
def generate_maze() -> List[List[str]]:
    """Create 20x20 maze with random rocks."""
    maze = [['·' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks_placed = 0
    while rocks_placed < ROCK_COUNT:
        x = random.randint(0, MAZE_SIZE-1)
        y = random.randint(0, MAZE_SIZE-1)
        if (x, y) == START or (x, y) == GOAL:
            continue
        if maze[y][x] != '█':
            maze[y][x] = '█'
            rocks_placed += 1
    maze[START[1]][START[0]] = 'S'
    maze[GOAL[1]][GOAL[0]] = 'G'
    return maze

def print_maze(maze, path=None, rocks=None, safe=None):
    """Visualize maze with optional overlays."""
    for y in range(MAZE_SIZE):
        line = []
        for x in range(MAZE_SIZE):
            cell = maze[y][x]
            if path and (x, y) in path:
                if (x, y) == START:
                    line.append('S')
                elif (x, y) == GOAL:
                    line.append('G')
                else:
                    line.append('*')
            elif rocks and (x, y) in rocks:
                line.append('█')
            elif safe and (x, y) in safe:
                line.append('·')
            else:
                line.append(cell)
        print(''.join(line))

# ========== MAIN DEMO ==========

    # Generate same maze for both agents

    maze = generate_maze()
    
    print("\n[MAZE] 20x20 with 64 rocks (█)")
    print("Start (S) at (0,0), Goal (G) at (19,19)")
    print_maze(maze)

    print("\n" + "=" * 60)
    print("UNCONSTRAINED AGENT (P2-ONLY OPTIMIZATION)")

    greedy = GreedyAgent(maze)

    print(f"Success: {success_greedy}")
    if greedy.path:
        print_maze(maze, path=set(greedy.path))
        print(f"Path length: {len(greedy.path)}")

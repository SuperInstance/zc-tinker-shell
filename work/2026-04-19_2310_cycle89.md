```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 89)
Simulates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json.
"""

import random
from typing import List, Tuple, Set

# ========== CONFIG ==========
MAZE_SIZE = 10
ROCK_DENSITY = 0.2  # 20% rocks
START = (0, 0)
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

# ========== MAZE GENERATION ==========
def generate_maze(size: int, rock_density: float) -> List[List[str]]:
    """Generate a size x size maze with rocks ('#') and empty cells ('.')."""
    maze = []
    for r in range(size):
        row = []
        for c in range(size):
            if (r, c) == START:
                row.append('S')
            elif (r, c) == GOAL:
                row.append('G')
            elif random.random() < rock_density:
                row.append('#')
            else:
                row.append('.')
        maze.append(row)
    return maze

def print_maze(maze: List[List[str]], title: str):
    """Print maze with a title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

# ========== DEADBAND PROTOCOL ==========
def p0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Identify negative space — all rock coordinates."""
    rocks = set()
    for r in range(len(maze)):
        for c in range(len(maze[0])):
            if maze[r][c] == '#':
                rocks.add((r, c))
    return rocks

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
    """P1: Find all safe paths from START to GOAL avoiding rocks (simple DFS)."""
    size = len(maze)
    directions = [(0,1), (1,0), (0,-1), (-1,0)]
    safe_paths = []
    
    def dfs(path: List[Tuple[int, int]], visited: Set[Tuple[int, int]]):
        current = path[-1]
        if current == GOAL:
            safe_paths.append(path.copy())
            return
        for dr, dc in directions:
            nr, nc = current[0] + dr, current[1] + dc
            if 0 <= nr < size and 0 <= nc < size:
                if (nr, nc) not in rocks and (nr, nc) not in visited:
                    visited.add((nr, nc))
                    path.append((nr, nc))
                    dfs(path, visited)
                    path.pop()
                    visited.remove((nr, nc))
    
    dfs([START], {START})
    return safe_paths

def p2_optimize_within_channel(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """P2: Choose shortest safe path (optimize within safe channels)."""
    if not safe_paths:
        return []
    return min(safe_paths, key=len)

def run_unconstrained_greedy(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> bool:
    """Simulate a P2-only agent (greedy, ignores P0/P1). Returns True if reaches goal."""
    size = len(maze)
    current = START
    visited = {START}
    directions = [(0,1), (1,0)]  # only move right/down (greedy toward goal)
    
    while current != GOAL:
        # Greedy: pick first valid move toward goal
        moved = False
        for dr, dc in directions:
            nr, nc = current[0] + dr, current[1] + dc
            if 0 <= nr < size and 0 <= nc < size:
                if (nr, nc) not in rocks:
                    current = (nr, nc)
                    if current in visited:  # loop detection
                        return False
                    visited.add(current)
                    moved = True
                    break
        if not moved:
            return False  # stuck
    return True

# ========== VISUALIZATION HELPERS ==========
def visualize_path(maze: List[List[str]], path: List[Tuple[int, int]], label: str):
    """Create a copy of maze with path marked as 'o'."""
    viz = [row.copy() for row in maze]
    for (r, c) in path:
        if viz[r][c] not in ('S', 'G'):
            viz[r][c] = 'o'
    print_maze(viz, label)

# ========== MAIN SIMULATION ==========
def main():
    random.seed(42)  # reproducible for demo
    maze = generate_maze(MAZE_SIZE, ROCK_DENSITY)
    print_maze(maze, "Initial Maze (S=start, G=goal, #=rock)")
    
    # P0: Map rocks
    rocks = p0_map_rocks(maze)
    print(f"P0 — Rocks mapped: {len(rocks)} rock cells")
    
    # P1: Find safe channels
    safe_paths = p1_find_safe_channels(maze, rocks)
    print(f"P1 — Safe channels found: {len(safe_paths)} possible paths")
    
    if safe_paths:
        # P2: Optimize within channel
        optimal_path = p2_optimize_within_channel(safe_paths)
        print(f"P2 — Optimal safe path length: {len(optimal_path)} steps")
        visualize_path(maze, optimal_path, "Deadband Protocol (P0→P1→P2) Optimal Path")
        
        # Compare with unconstrained greedy (P2-only)
        success = run_unconstrained_greedy(maze, rocks)
        print(f"Unconstrained greedy (P2-only) reached goal: {success}")
        if not success:
            print("  → Greedy agent hit a rock or got stuck (local minima).")
    else:
        print("No safe channels exist — maze is impossible under deadband protocol.")
    
    # Print protocol reminder
    print("\n" + "="*60)
    print("DEADBAND PROTOCOL SUMMARY")
    print("P0: Map rocks (negative space) — what NOT to do")
    print("P1: Find safe channels — where you CAN be")
    print("P2: Optimize within channels — best path through safe water")
    print("="*60)

if __name__ == "__main__":
    main()
```

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
ASCII visualization of the protocol in action.
"""

import random
from collections import deque

# ====================
# MAZE CONFIGURATION
# ====================
MAZE_WIDTH = 20
MAZE_HEIGHT = 20
START = (0, 0)
GOAL = (MAZE_WIDTH - 1, MAZE_HEIGHT - 1)

# Rock density (from constraint_sim_final.json: 64 rocks in 400 cells)
ROCK_COUNT = 64

# Symbols
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_MARK = 'G'
PATH = '○'
SAFE = ' '  # inside deadband channel

# ====================
# DEADBAND PROTOCOL
# ====================
class DeadbandAgent:
    def __init__(self, maze):
        self.maze = maze
        self.width = len(maze[0])
        self.height = len(maze)
        self.visited = [[False for _ in range(self.width)] for _ in range(self.height)]
        self.parent = [[None for _ in range(self.width)] for _ in range(self.height)]
        
    # P0: MAP NEGATIVE SPACE (rocks)
    def identify_rocks(self):
        rocks = []
        for y in range(self.height):
            for x in range(self.width):
                if self.maze[y][x] == ROCK:
                    rocks.append((x, y))
        return rocks
    
    # P1: FIND SAFE CHANNELS (deadbands)
    def find_safe_channels(self, start, goal):
        # BFS that avoids rocks
        queue = deque([start])
        self.visited[start[1]][start[0]] = True
        directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        
        while queue:
            x, y = queue.popleft()
            
            if (x, y) == goal:
                # Reconstruct safe channel
                channel = set()
                current = (x, y)
                while current:
                    channel.add(current)
                    current = self.parent[current[1]][current[0]]
                return channel
            
            for dx, dy in directions:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    not self.visited[ny][nx] and self.maze[ny][nx] != ROCK):
                    self.visited[ny][nx] = True
                    self.parent[ny][nx] = (x, y)
                    queue.append((nx, ny))
        
        return set()  # no safe channel found
    
    # P2: OPTIMIZE WITHIN CHANNEL (shortest path in safe channel)
    def optimize_path(self, start, goal, safe_channel):
        if not safe_channel:
            return []
        
        # BFS within safe channel only
        queue = deque([start])
        dist = {start: 0}
        prev = {start: None}
        
        directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        
        while queue:
            current = queue.popleft()
            
            if current == goal:
                # Reconstruct optimal path
                path = []
                while current:
                    path.append(current)
                    current = prev[current]
                return list(reversed(path))
            
            for dx, dy in directions:
                neighbor = (current[0] + dx, current[1] + dy)
                if (neighbor in safe_channel and neighbor not in dist):
                    dist[neighbor] = dist[current] + 1
                    prev[neighbor] = current
                    queue.append(neighbor)
        
        return []

# ====================
# VISUALIZATION
# ====================
def print_maze(maze, rocks=None, safe_channel=None, path=None, agent_pos=None):
    display = [row[:] for row in maze]
    
    # Mark safe channel (P1)
    if safe_channel:
        for x, y in safe_channel:
            if display[y][x] == EMPTY:
                display[y][x] = SAFE
    
    # Mark path (P2)
    if path:
        for x, y in path[1:-1]:  # exclude start and goal
            if display[y][x] == EMPTY or display[y][x] == SAFE:
                display[y][x] = PATH
    
    # Mark agent
    if agent_pos:
        x, y = agent_pos
        display[y][x] = AGENT
    
    # Mark goal
    gx, gy = GOAL
    if display[gy][gx] != AGENT:
        display[gy][gx] = GOAL_MARK
    
    # Print
    print("+" + "-" * MAZE_WIDTH + "+")
    for row in display:
        print("|" + "".join(row) + "|")
    print("+" + "-" * MAZE_WIDTH + "+")

#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
"""

import random
from typing import List, Tuple, Set

# ========== MAZE CONFIG ==========
MAZE_WIDTH = 20
MAZE_HEIGHT = 20
START = (0, 0)
GOAL = (MAZE_WIDTH - 1, MAZE_HEIGHT - 1)
ROCK_DENSITY = 0.15  # 15% of cells are rocks

# Symbols
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_MARKER = 'G'
PATH = '○'
SAFE = '░'  # Visual for safe channel cells

# ========== DEADBAND PROTOCOL IMPLEMENTATION ==========
class DeadbandAgent:
    def __init__(self, maze: List[List[str]]):
        self.maze = maze
        self.width = len(maze[0])
        self.height = len(maze)
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: Set[Tuple[int, int]] = set()
        self.optimal_path: List[Tuple[int, int]] = []

    # --- P0: MAP NEGATIVE SPACE (ROCKS) ---
    def p0_map_rocks(self) -> None:
        """Identify all rock cells — what NOT to do."""
        for y in range(self.height):
            for x in range(self.width):
                if self.maze[y][x] == ROCK:
                    self.rocks.add((x, y))
        print(f"[P0] Mapped {len(self.rocks)} rocks.")

    # --- P1: FIND SAFE CHANNELS ---
    def p1_find_safe_channels(self) -> None:
        """Flood-fill from start, avoiding rocks."""
        visited = set()
        stack = [START]
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            self.safe_channels.add((x, y))
            # 4-directional neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        stack.append((nx, ny))
        print(f"[P1] Found safe channel with {len(self.safe_channels)} cells.")

    # --- P2: OPTIMIZE WITHIN CHANNEL (BFS) ---
    def p2_optimize_path(self) -> bool:
        """BFS within safe channels to find shortest path to goal."""
        if GOAL not in self.safe_channels:
            print("[P2] Goal is not reachable within safe channels.")
            return False
        queue = [(START, [START])]
        visited = set([START])
        while queue:
            (x, y), path = queue.pop(0)
            if (x, y) == GOAL:
                self.optimal_path = path
                print(f"[P2] Found optimal path length: {len(path)-1}")
                return True
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) in self.safe_channels and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        print("[P2] No path to goal within safe channels.")
        return False

    def run_protocol(self) -> bool:
        """Execute P0→P1→P2 in strict order."""
        print("\n=== DEADBAND PROTOCOL EXECUTION ===")
        self.p0_map_rocks()
        self.p1_find_safe_channels()
        success = self.p2_optimize_path()
        return success

# ========== VISUALIZATION ==========
def generate_maze() -> List[List[str]]:
    """Create a random maze with rocks."""
    maze = [[EMPTY for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    for y in range(MAZE_HEIGHT):
        for x in range(MAZE_WIDTH):
            if random.random() < ROCK_DENSITY and (x, y) not in (START, GOAL):
                maze[y][x] = ROCK
    maze[START[1]][START[0]] = AGENT
    maze[GOAL[1]][GOAL[0]] = GOAL_MARKER
    return maze

def print_maze(maze: List[List[str]], 
               rocks: Set[Tuple[int, int]] = None,
               safe: Set[Tuple[int, int]] = None,
               path: List[Tuple[int, int]] = None) -> None:
    """Print maze with optional overlays."""
    for y in range(MAZE_HEIGHT):
        row = []
        for x in range(MAZE_WIDTH):
            cell = maze[y][x]
            if path and (x, y) in path and (x, y) not in (START, GOAL):
                cell = PATH
            elif safe and (x, y) in safe and cell == EMPTY:
                cell = SAFE
            elif rocks and (x, y) in rocks:
                cell = ROCK
            row.append(cell)
        print(''.join(row))

# ========== MAIN ==========
def main():
    print("Deadband Protocol Visual Demo")
    print(f"Maze: {MAZE_WIDTH}x{MAZE_HEIGHT}, Start {START}, Goal {GOAL}")
    print(f"Symbols: {AGENT}=Agent, {GOAL_MARKER}=Goal, {ROCK}=Rock, {PATH}=Path, {SAFE}=Safe Channel\n")
    
    # Generate maze
    maze = generate_maze()
    print("=== INITIAL MAZE ===")
    print_maze(maze)
    
    # Run agent
    agent = DeadbandAgent(maze)
    success = agent.run_protocol()
    
    # Visualize results
    print("\n=== P0: ROCKS (NEGATIVE SPACE) ===")
    print_maze(maze, rocks=agent.rocks)
    
    print("\n=== P1: SAFE CHANNELS ===")
    print_maze(maze, safe=agent.safe_channels)
    
    if success:
        print("\n=== P2: OPTIMAL PATH (WITHIN SAFE CHANNELS) ===")
        print_maze(maze, path=agent.optimal_path)
        print(f"\nSUCCESS: Agent reached goal in {len(agent.optimal_path)-1} steps.")
    else:
        print("\nFAILURE: Goal unreachable within safe channels.")
    
    # Protocol validation
    print("\n=== PROTOCOL VALIDATION ===")
    print(f"P0 completed: {len(agent.rocks)} rocks mapped")
    print(f"P1 completed: {len(agent.safe_channels)} safe cells identified")
    print(f"P2 attempted: {success}")
    if agent.optimal_path:
        # Verify path stays within safe channels
        in_safe = all(pos in agent.safe_channels for pos in agent.optimal_path)
        print(f"Path stays in safe channels: {in_safe}")
        # Verify path avoids rocks
        avoids_rocks = all(pos not in agent.rocks for pos in agent.optimal_path)
        print(f"Path avoids rocks: {avoids_rocks}")

if __name__ == "__main__":
    main()

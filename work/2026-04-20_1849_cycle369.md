```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time
from typing import List, Tuple, Set

# Simulation parameters from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

def generate_maze(size: int, rocks: int) -> Tuple[List[List[str]], Tuple[int, int], Tuple[int, int]]:
    """Generate a size x size maze with rocks, start (top-left), goal (bottom-right)."""
    maze = [['.' for _ in range(size)] for _ in range(size)]
    
    # Place rocks randomly, avoiding start and goal
    start = (0, 0)
    goal = (size-1, size-1)
    rock_positions = set()
    while len(rock_positions) < rocks:
        r, c = random.randint(0, size-1), random.randint(0, size-1)
        if (r, c) != start and (r, c) != goal:
            rock_positions.add((r, c))
    
    for r, c in rock_positions:
        maze[r][c] = '#'
    
    maze[start[0]][start[1]] = 'S'
    maze[goal[0]][goal[1]] = 'G'
    
    return maze, start, goal

def print_maze(maze: List[List[str]], path: List[Tuple[int, int]] = None):
    """Print maze with optional path marked."""
    for r in range(len(maze)):
        row = ''
        for c in range(len(maze[r])):
            if path and (r, c) in path:
                row += '○'  # path marker
            else:
                row += maze[r][c]
        print(row)
    print()

def p0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Identify negative space (rocks)."""
    rocks = set()
    for r in range(len(maze)):
        for c in range(len(maze[r])):
            if maze[r][c] == '#':
                rocks.add((r, c))
    return rocks

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]], 
                          start: Tuple[int, int], goal: Tuple[int, int]) -> List[List[Tuple[int, int]]]:
    """P1: Find all safe paths (deadbands) using BFS, avoiding rocks."""
    size = len(maze)
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]  # right, down, left, up
    
    # BFS to find all shortest safe paths
    from collections import deque
    queue = deque([(start, [start])])
    safe_paths = []
    shortest_length = None
    
    while queue:
        (r, c), path = queue.popleft()
        
        # If we already found a shorter path, stop exploring longer ones
        if shortest_length and len(path) > shortest_length:
            continue
        
        if (r, c) == goal:
            if shortest_length is None:
                shortest_length = len(path)
            if len(path) == shortest_length:
                safe_paths.append(path)
            continue
        
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < size and 0 <= nc < size:
                if (nr, nc) not in rocks and (nr, nc) not in path:
                    queue.append(((nr, nc), path + [(nr, nc)]))
    
    return safe_paths

def p2_optimize_within_channel(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """P2: Select the optimal path among safe channels (shortest)."""
    if not safe_paths:
        return []
    # In this simple case, all safe paths are same length (shortest).
    # For demo, pick first.
    return safe_paths[0]

def greedy_unconstrained(maze: List[List[str]], start: Tuple[int, int], goal: Tuple[int, int]) -> List[Tuple[int, int]]:
    """Greedy P2-only approach (no P0/P1) -> hits rocks."""
    path = [start]
    r, c = start
    size = len(maze)
    
    while (r, c) != goal:
        # Prefer moving right or down (toward goal)
        if c < size-1 and maze[r][c+1] != '#':
            c += 1
        elif r < size-1 and maze[r+1][c] != '#':
            r += 1
        elif c > 0 and maze[r][c-1] != '#':
            c -= 1
        elif r > 0 and maze[r-1][c] != '#':
            r -= 1
        else:
            break  # stuck
        
        if maze[r][c] == '#':
            path.append((r, c))
            break  # hit rock
        
        path.append((r, c))
    
    return path

def run_demo():
    """Run the full Deadband Protocol demonstration."""
    print("=" * 60)
    print("DEADBAND PROTOCOL DEMO — ASCII MAZE")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print("=" * 60)
    
    # Generate maze matching simulation parameters
    random.seed(42)  # reproducible for demo
    maze, start, goal = generate_maze(MAZE_SIZE, ROCK_COUNT)
    
    print("\n1. MAZE (20x20, 64 rocks, S=start, G=goal, #=rock)")
    print_maze(maze)
    
    # P0: Map rocks
    rocks = p0_map_rocks(maze)
    print(f"2. P0: MAP NEGATIVE SPACE — Found {len(rocks)} rocks.")
    
    # P1: Find safe channels
    safe_paths = p1_find_safe_channels(maze, rocks, start, goal)
    print(f"3. P1: FIND SAFE CHANNELS — Found {len(safe_paths)} shortest safe paths (deadbands).")
    
    # P2: Optimize within channel
    optimal_path = p2_optimize_within_channel(safe_paths)
    print(f"4. P2: OPTIMIZE WITHIN CHANNEL — Selected path length: {len(optimal_path)} steps.")
    
    print("\n5. DEADBAND PATH (P0→P1→P2) — SUCCESS")
    print_maze(maze, optimal_path)
    
    # Contrast with greedy (P2-only)
    print("\n6. GREEDY P2-ONLY (no P0/P1) — FAILURE")
    greedy_path = greedy_unconstrained(maze, start, goal)
    if greedy_path[-1] == goal:
        print("   (Lucky — no rock hit this time)")
    else:
        print(f"   Hit rock at {greedy_path[-1]} after {len(greedy_path)} steps.")
    print_maze(maze, greedy_path)
    
    # Summary
    print("\n" + "=" * 60)
    print("SUMMARY")
    print(f"Optimal path length (theoretical): {OPTIMAL_PATH_LENGTH}")
    print(f"Deadband path length (actual): {len(optimal_path)}")
    print(f"Safe channels found: {len(safe_paths)}")
    print(f"Rocks avoided: {len(rocks)}")
    print("=" * 60)
    print("\nProtocol followed: P0 → P1 → P2 (strict priority)")
    print("Result: Path found within safe deadband, no rock hits.")

if __name__ == "__main__":
    run_demo()
```

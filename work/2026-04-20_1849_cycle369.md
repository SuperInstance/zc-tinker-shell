#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — Tinker (Cycle 369)
A simple ASCII maze simulation showing P0 (map rocks), P1 (find safe channels),
and P2 (optimize path) as per fleet doctrine.
"""

import random
from typing import List, Tuple, Set

# Constants from fleet-knowledge/narrow-games/constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

def generate_maze(size: int, rocks: int) -> Tuple[List[List[str]], Tuple[int, int], Tuple[int, int]]:
    """Generate a size x size maze with rocks, start (top-left), goal (bottom-right)."""
    maze = [['.' for _ in range(size)] for _ in range(size)]
    
    # Place rocks randomly, avoiding start and goal
    start = (0, 0)
    goal = (size-1, size-1)
    rock_positions = set()
    while len(rock_positions) < rocks:
        r, c = random.randint(0, size-1), random.randint(0, size-1)
        if (r, c) != start and (r, c) != goal:
            rock_positions.add((r, c))
    
    for r, c in rock_positions:
        maze[r][c] = '#'
    
    maze[start[0]][start[1]] = 'S'
    maze[goal[0]][goal[1]] = 'G'
    
    return maze, start, goal

def print_maze(maze: List[List[str]], title: str):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def p0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Identify negative space (rocks)."""
    rocks = set()
    for r in range(len(maze)):
        for c in range(len(maze[0])):
            if maze[r][c] == '#':
                rocks.add((r, c))
    print(f"P0: Mapped {len(rocks)} rocks (negative space).")
    return rocks

def p1_find_safe_channels(maze: List[List[str]], start: Tuple[int, int], goal: Tuple[int, int], rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
    """P1: Find all safe paths (channels) via BFS, avoiding rocks."""
    size = len(maze)
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]  # right, down, left, up
    
    # BFS to find all shortest safe paths
    from collections import deque
    queue = deque([(start, [start])])
    safe_paths = []
    shortest_length = None
    
    while queue:
        (r, c), path = queue.popleft()
        
        if (r, c) == goal:
            if shortest_length is None:
                shortest_length = len(path)
            if len(path) == shortest_length:
                safe_paths.append(path)
            continue
        
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < size and 0 <= nc < size and (nr, nc) not in rocks and (nr, nc) not in [p for p in path]:
                queue.append(((nr, nc), path + [(nr, nc)]))
    
    print(f"P1: Found {len(safe_paths)} safe channels (shortest length = {shortest_length}).")
    return safe_paths

def p2_optimize_within_channel(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """P2: Select optimal path from safe channels (here, first shortest path)."""
    if not safe_paths:
        print("P2: No safe channels — optimization impossible.")
        return []
    
    # Simple optimization: pick first shortest path (could add cost metrics)
    optimal_path = safe_paths[0]
    print(f"P2: Selected optimal path length = {len(optimal_path)}.")
    return optimal_path

def visualize_path(maze: List[List[str]], path: List[Tuple[int, int]]):
    """Mark path on maze with '*'."""
    viz = [row[:] for row in maze]
    for r, c in path[1:-1]:  # exclude start and goal
        if viz[r][c] == '.':
            viz[r][c] = '*'
    print_maze(viz, "P2: Optimal Path Visualization")

def run_deadband_demo():
    """Main demo sequence."""
    print("DEADBAND PROTOCOL DEMO — Tinker (Fleet Cocapn)")
    print("=" * 50)
    
    # Generate maze matching simulation parameters
    maze, start, goal = generate_maze(MAZE_SIZE, ROCK_COUNT)
    print_maze(maze, "Initial Maze (S=start, G=goal, #=rock)")
    
    # P0: Map negative space
    rocks = p0_map_rocks(maze)
    
    # P1: Find safe channels
    safe_paths = p1_find_safe_channels(maze, start, goal, rocks)
    
    if not safe_paths:
        print("FAILURE: No safe channels found. Protocol stops at P1.")
        return
    
    # P2: Optimize within channel
    optimal_path = p2_optimize_within_channel(safe_paths)
    
    # Visualize
    visualize_path(maze, optimal_path)
    
    # Summary
    print("=" * 50)
    print("PROTOCOL SUMMARY:")
    print(f"  P0 (rocks): {len(rocks)} obstacles mapped")
    print(f"  P1 (channels): {len(safe_paths)} safe paths found")
    print(f"  P2 (optimize): path length {len(optimal_path)} (optimal = {OPTIMAL_PATH_LENGTH})")
    print(f"  Success: {'YES' if len(optimal_path) <= OPTIMAL_PATH_LENGTH + 5 else 'NO'}")
    print("\nDoctrine followed: P0 → P1 → P2 (strict priority).")

if __name__ == "__main__":
    random.seed(42)  # reproducible for demo
    run_deadband_demo()

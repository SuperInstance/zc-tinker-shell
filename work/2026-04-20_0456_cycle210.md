**Approach:**
1. **P0 (Map negative space):** Generate a random 10x10 maze with obstacles ('#') as rocks.
2. **P1 (Identify safe channels):** Use BFS to find all reachable paths from start to goal that avoid rocks.
3. **P2 (Optimize within channel):** Select the shortest safe path from the valid channels.
4. **Visualization:** Print each step with ASCII art.

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo â€” Tinker (Cycle 210)
Demonstrates P0â†’P1â†’P2 in a navigational domain.
Based on fleet-knowledge/constraint_sim_final.json and DEADBAND-PROTOCOL.md.
"""

import random
from collections import deque

def generate_maze(width=10, height=10, rock_density=0.25):
    """P0: Generate environment with rocks (negative space)."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    rocks = 0
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density and not (x == 0 and y == 0) and not (x == width-1 and y == height-1):
                maze[y][x] = '#'  # Rock
                rocks += 1
    maze[0][0] = 'S'  # Start
    maze[height-1][width-1] = 'G'  # Goal
    return maze, rocks

def print_maze(maze, path=None):
    """Visualize maze with optional path."""
    for y, row in enumerate(maze):
        line = []
        for x, cell in enumerate(row):
            if path and (x, y) in path:
                line.append('*')
            else:
                line.append(cell)
        print(' '.join(line))
    print()

def find_safe_channels(maze):
    """P1: BFS to find all safe paths from S to G (avoid rocks)."""
    height, width = len(maze), len(maze[0])
    start = (0, 0)
    goal = (width-1, height-1)
    
    # Directions: up, down, left, right
    dirs = [(0, -1), (0, 1), (-1, 0), (1, 0)]
    
    queue = deque([(start, [start])])
    visited = set([start])
    safe_channels = []
    
    while queue:
        (x, y), path = queue.popleft()
        
        if (x, y) == goal:
            safe_channels.append(path)
            continue
        
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if maze[ny][nx] != '#' and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    return safe_channels

def deadband_protocol_demo():
    """Main demonstration of P0â†’P1â†’P2."""
    print("=== DEADBAND PROTOCOL MAZE DEMO ===")
    print("P0: Map negative space (rocks)\n")
    
    # P0: Generate maze with rocks
    maze, rocks = generate_maze()
    print("Generated Maze (S=start, G=goal, #=rock):")
    print_maze(maze)
    print(f"Rocks identified: {rocks}")
    
    print("\nP1: Identify safe channels (paths avoiding rocks)\n")
    safe_channels = find_safe_channels(maze)
    
    if not safe_channels:
        print("NO SAFE CHANNELS FOUND â€” Goal unreachable.")
        return
    
    print(f"Found {len(safe_channels)} safe channel(s).")
    print("Example safe channel (first found):")
    print_maze(maze, safe_channels[0])
    
    print("P2: Optimize within safe channels\n")
    # Select shortest path among safe channels
    optimal_path = min(safe_channels, key=len)
    print(f"Optimal path length: {len(optimal_path)-1} steps")
    print("Optimal path visualization:")
    print_maze(maze, optimal_path)
    
    # Simulation metrics
    print("=== SIMULATION RESULTS ===")
    print(f"Rocks (P0 negative space): {rocks}")
    print(f"Safe channels found (P1): {len(safe_channels)}")
    print(f"Optimal path steps (P2): {len(optimal_path)-1}")
    
    # Compare with unconstrained (P2-only) approach
    print("\nHypothetical unconstrained (P2-only) agent:")
    print("Would attempt straight-line path, likely hitting rocks.")
    print("Success probability: low (as shown in constraint_sim_final.json)")

if __name__ == "__main__":
    random.seed(42)  # Deterministic for demonstration
    deadband_protocol_demo()
```

import random
import time
from typing import List, Tuple, Set

class Maze:
    """ASCII maze with start, goal, and obstacles."""
    
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (obstacles) - P0: Negative Space
        self.rocks: Set[Tuple[int, int]] = set()
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density and (x, y) not in [self.start, self.goal]:
                    self.rocks.add((x, y))
        
        # Ensure at least one path exists
        self._ensure_path()
    
    def _ensure_path(self):
        """Make sure there's at least one path from start to goal."""
        # Simple right-then-down path
        for x in range(self.width):
            if (x, 0) in self.rocks:
                self.rocks.remove((x, 0))
        for y in range(self.height):
            if (self.width-1, y) in self.rocks:
                self.rocks.remove((self.width-1, y))
    
    def is_rock(self, x: int, y: int) -> bool:
        """Check if position is a rock (P0: negative space)."""
        return (x, y) in self.rocks
    
    def is_valid(self, x: int, y: int) -> bool:
        """Check if position is within bounds and not a rock."""
        return 0 <= x < self.width and 0 <= y < self.height and not self.is_rock(x, y)
    
    def render(self, path: List[Tuple[int, int]] = None, 
               deadband: Set[Tuple[int, int]] = None) -> str:
        """Render maze with optional path and deadband visualization."""
        lines = []

        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append("S")
                elif pos == self.goal:
                    row.append("G")
                elif path and pos in path:
                    row.append("*")
                elif deadband and pos in deadband:
                    row.append(".")
                elif self.is_rock(x, y):
                    row.append("#")
                else:
                    row.append(" ")
            lines.append("| " + " ".join(row) + " |")
        
        lines.append("=" * (self.width * 2 + 3))
        return "\n".join(lines)

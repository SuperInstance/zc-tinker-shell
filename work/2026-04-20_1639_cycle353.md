#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 strict priority in a 10x10 ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set

# ========== MAZE DEFINITION ==========
# '.' = open, '#' = rock (P0 negative space), 'S' = start, 'G' = goal
MAZE_RAW = """
S . . # . . . # . .
. # . # . # . # . .
. . . . . . . . . .
# # # . # # # . # .
. . . . . . . . . .
. # # # . # # # . #
. . . . . . . . . .
# . # # # . # # # .
. . . . . . . . . .
. # . # . # . # . G
"""

def parse_maze(maze_str: str) -> Tuple[List[List[str]], Tuple[int, int], Tuple[int, int]]:
    """Parse maze string into grid, find start and goal."""
    lines = [line.strip() for line in maze_str.strip().split('\n') if line.strip()]
    grid = []
    start = goal = None
    for r, line in enumerate(lines):
        row = []
        for c, ch in enumerate(line.split()):
            row.append(ch)
            if ch == 'S':
                start = (r, c)
            elif ch == 'G':
                goal = (r, c)
        grid.append(row)
    return grid, start, goal

def print_maze(grid: List[List[str]], path: Set[Tuple[int, int]] = None):
    """Print maze with optional path overlay."""
    for r in range(len(grid)):
        for c in range(len(grid[r])):
            if path and (r, c) in path:
                print('*', end=' ')  # path marker
            else:
                print(grid[r][c], end=' ')
        print()

# ========== DEADBAND PROTOCOL IMPLEMENTATION ==========
class DeadbandNavigator:
    def __init__(self, grid, start, goal):
        self.grid = grid
        self.start = start
        self.goal = goal
        self.rows = len(grid)
        self.cols = len(grid[0])
        self.rocks = set()      # P0: negative space
        self.safe = set()       # P1: safe channels
        self.optimal_path = []  # P2: best path within safe channels

    # --- P0: MAP NEGATIVE SPACE ---
    def map_negative_space(self):
        """Identify all rock cells (#). These are absolute no‑go zones."""
        for r in range(self.rows):
            for c in range(self.cols):
                if self.grid[r][c] == '#':
                    self.rocks.add((r, c))
        print(f"[P0] Mapped {len(self.rocks)} rocks (negative space).")

    # --- P1: IDENTIFY SAFE CHANNELS ---
    def find_safe_channels(self):
        """Flood‑fill from start, avoiding rocks, to find reachable safe cells."""
        stack = [self.start]
        visited = set()
        while stack:
            r, c = stack.pop()
            if (r, c) in visited:
                continue
            visited.add((r, c))
            # only proceed if cell is not a rock
            if self.grid[r][c] == '#':
                continue
            self.safe.add((r, c))
            # 4‑connected neighbors
            for dr, dc in [(-1,0),(1,0),(0,-1),(0,1)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < self.rows and 0 <= nc < self.cols:
                    stack.append((nr, nc))
        print(f"[P1] Found {len(self.safe)} safe cells (channels).")
        if self.goal not in self.safe:
            print("  WARNING: Goal is not reachable within safe channels!")

    # --- P2: OPTIMIZE WITHIN CHANNEL ---
    def optimize_within_channel(self):
        """A* search restricted to safe cells only."""
        from heapq import heappush, heappop

        # heuristic: Manhattan distance
        def h(pos):
            return abs(pos[0] - self.goal[0]) + abs(pos[1] - self.goal[1])

        open_set = [(h(self.start), 0, self.start, [])]  # (f, g, pos, path)
        closed = set()

        while open_set:
            f, g, (r, c), path = heappop(open_set)
            if (r, c) in closed:
                continue
            closed.add((r, c))
            new_path = path + [(r, c)]

            if (r, c) == self.goal:
                self.optimal_path = new_path
                print(f"[P2] Found optimal path of length {len(new_path)-1} steps.")
                return

            for dr, dc in [(-1,0),(1,0),(0,-1),(0,1)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < self.rows and 0 <= nc < self.cols:
                    # CRITICAL: only move through safe cells (P1 constraint)
                    if (nr, nc) not in self.safe:
                        continue
                    ng = g + 1
                    nf = ng + h((nr, nc))
                    heappush(open_set, (nf, ng, (nr, nc), new_path))

        print("[P2] No path to goal within safe channels.")

    # --- FULL PROTOCOL EXECUTION ---
    def run_deadband_protocol(self):
        print("\n=== DEADBAND PROTOCOL DEMO ===")
        print("Maze layout:")
        print_maze(self.grid)
        print("\n--- P0: Map Negative Space ---")
        self.map_negative_space()
        print("\n--- P1: Identify Safe Channels ---")
        self.find_safe_channels()
        print("\n--- P2: Optimize Within Channel ---")
        self.optimize_within_channel()
        print("\n=== RESULTS ===")
        if self.optimal_path:
            print("Optimal path (marked with *):")
            print_maze(self.grid, set(self.optimal_path))
            print(f"Path steps: {len(self.optimal_path)-1}")
            print("Protocol SUCCESS: P0→P1→P2 followed strictly.")
        else:
            print("Protocol FAILED: No safe path to goal.")
        return self.optimal_path

# ========== DEMONSTRATION ==========
if __name__ == "__main__":
    grid, start, goal = parse_maze(MAZE_RAW)
    nav = DeadbandNavigator(grid, start, goal)
    path = nav.run_deadband_protocol()

    # Show what happens if you skip P0+P1 (greedy, unconstrained)
    print("\n--- CONTRAST: Greedy (P2‑only) Approach ---")
    print("A greedy agent that ignores rocks will likely hit one.")
    # Simulate a naive random walk from start (no rock avoidance)
    steps = 0
    r, c = start
    visited = [(r, c)]
    while (r, c) != goal and steps < 30:
        dr, dc = random.choice([(-1,0),(1,0),(0,-1),(0,1)])
        nr, nc = r + dr, c + dc
        if 0 <= nr < len(grid) and 0 <= nc < len(grid[0]):
            r, c = nr, nc
            visited.append((r, c))
            if grid[r][c] == '#':
                print(f"  Step {steps}: CRASH into rock at ({r},{c})!")
                break
        steps += 1
    if (r, c) == goal:
        print(f"  (Lucky) reached goal in {steps} steps without hitting rocks.")
    else:
        print("  Greedy agent failed — illustrates why P0+P1 are mandatory.")

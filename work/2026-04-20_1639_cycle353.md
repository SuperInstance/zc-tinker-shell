#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
import time
from typing import List, Tuple, Set

class DeadbandMaze:
    """
    A 20x20 maze simulation of the Deadband Protocol.
    Rocks (obstacles) are placed randomly.
    Agent must go from start (0,0) to goal (19,19).
    """
    
    def __init__(self, seed: int = 42):
        self.width = 20
        self.height = 20
        self.start = (0, 0)
        self.goal = (19, 19)
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
        random.seed(seed)
        self._generate_rocks(64)  # 64 rocks as in simulation
        
    def _generate_rocks(self, num_rocks: int):
        """P0: Map the negative space (rocks)."""
        while len(self.rocks) < num_rocks:
            x = random.randint(0, self.width - 1)
            y = random.randint(0, self.height - 1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def _is_safe(self, x: int, y: int) -> bool:
        """Check if a cell is safe (not a rock)."""
        return (x, y) not in self.rocks
    
    def find_safe_channels(self):
        """P1: Identify safe channels using BFS from start to goal."""
        from collections import deque
        
        queue = deque([(self.start, [self.start])])
        visited = {self.start}
        all_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                all_paths.append(path)
                continue
            
            # 4-direction movement
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited and self._is_safe(nx, ny):
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.safe_channels = all_paths
        return len(all_paths)
    
    def optimize_within_channel(self):
        """P2: Find shortest path among safe channels."""
        if not self.safe_channels:
            return None
        
        self.optimal_path = min(self.safe_channels, key=len)
        return len(self.optimal_path)
    
    def greedy_unconstrained(self):
        """P2-only (greedy) approach for comparison."""
        x, y = self.start
        path = [(x, y)]
        visited = {(x, y)}
        
        while (x, y) != self.goal:
            # Move toward goal (greedy)
            moves = []
            if x < self.goal[0]:
                moves.append((1, 0))
            if y < self.goal[1]:
                moves.append((0, 1))
            if x > self.goal[0]:
                moves.append((-1, 0))
            if y > self.goal[1]:
                moves.append((0, -1))
            
            # Shuffle to add some randomness
            random.shuffle(moves)
            
            moved = False
            for dx, dy in moves:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited:
                        # Greedy doesn't check rocks first (violates P0)
                        if (nx, ny) in self.rocks:
                            return None  # Hit a rock
                        x, y = nx, ny
                        path.append((x, y))
                        visited.add((x, y))
                        moved = True
                        break
            
            if not moved:
                return None  # Stuck
        
        return path
    
    def render_maze(self, path: List[Tuple[int, int]] = None) -> str:
        """Render ASCII maze with rocks, start, goal, and optional path."""
        grid = [['·' for _ in range(self.width)] for _ in range(self.height)]
        
        # Mark rocks
        for x, y in self.rocks:
            grid[y][x] = '█'
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        grid[sy][sx] = 'S'
        grid[gy][gx] = 'G'
        
        # Mark path if provided
        if path:
            for x, y in path[1:-1]:  # Exclude start and goal
                if grid[y][x] == '·':
                    grid[y][x] = '○'
        
        # Build output
        lines = []
        lines.append("=" * (self.width * 2 + 3))
        lines.append("Deadband Protocol Maze (20×20)")
        lines.append(f"Rocks: {len(self.rocks)}  Start: S  Goal: G  Path: ○")
        lines.append("=" * (self.width * 2 + 3))
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                row.append(grid[y][x])
            lines.append(' '.join(row))
        
        lines.append("=" * (self.width * 2 + 3))
        return '\n'.join(lines)

def run_demo():
    """Run the complete Deadband Protocol demonstration."""
    print("\n" + "="*60)
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print("="*60)
    
    # Create maze
    maze = DeadbandMaze(seed=42)
    
    print("\n1. P0: MAP NEGATIVE SPACE")
    print(f"   Identified {len(maze.rocks)} rocks (failure states)")
    print("   Rocks marked as '█' in maze")
    
    time.sleep(1)
    
    print("\n2. P1: FIND SAFE CHANNELS")
    num_channels = maze.find_safe_channels()
    print(f"   Found {num_channels} safe channels from S to G")
    print("   Channels avoid all rocks")
    
    time.sleep(1)
    
    print("\n3. P2: OPTIMIZE WITHIN CHANNEL")
    if num_channels > 0:
        path_length = maze.optimize_within_channel()
        print(f"   Shortest safe path: {path_length} steps")
        print(f"   Optimal (theoretical): 21 steps")
    else:
        print("   ERROR: No safe channels found!")
        return
    
    time.sleep(1)
    
    print("\n4. VISUALIZATION")
    print(maze.render_maze(maze.optimal_path))
    
    time.sleep(1)
    
    print("\n5. COMPARISON: P2-ONLY (GREEDY) APPROACH")
    greedy_path = maze.greedy_unconstrained()
    if greedy_path:
        print(f"   Greedy path length: {len(greedy_path)} steps")
        print("   (But this violates P0 - no rock checking first)")
    else:
        print("   Greedy approach: FAILED (hit a rock)")
        print("   This matches simulation data: 0/50 success for P2-only")
    
    time.sleep(1)
    
    print("\n6. RESULTS SUMMARY")
    print("   Deadband Protocol (P0→P1→P2): SUCCESS")
    print("   Path found within safe channels")
    print("   Greedy (P2-only): FAILURE")
    print("\n   This demonstrates:")
    print("   - P0 (map rocks) is non-negotiable")
    print("   - P1 (find channels) must precede P2")

    [0, 0, 0, 1, 0],

START = (0, 0)

def print_maze(maze, title="Maze"):
    """Print maze with ASCII borders."""
    print(f"\n=== {title} ===")
    for row in maze:
        line = []
        for cell in row:
            if cell == 1:
                line.append('█')  # rock
            elif cell == 0:
                line.append('·')  # open
            elif cell == 'S':
                line.append('S')
            elif cell == 'G':
                line.append('G')
            else:
                line.append(str(cell))
        print(' '.join(line))

def p0_map_rocks(maze) -> Set[Tuple[int, int]]:
    """P0: Identify all rock positions (negative space)."""
    rocks = set()
    for r in range(len(maze)):
        for c in range(len(maze[0])):
            if maze[r][c] == 1:
                rocks.add((r, c))
    print_maze(maze, "P0 - Map Rocks (█ = rock)")
    print(f"Rocks found: {len(rocks)} at positions {rocks}")
    return rocks

def p1_find_safe_channels(maze, rocks, start, goal) -> List[List[Tuple[int, int]]]:
    """P1: Find all paths from start to goal that avoid rocks."""
    rows, cols = len(maze), len(maze[0])
    all_paths = []
    
    def dfs(pos, path, visited):
        if pos == goal:
            all_paths.append(path.copy())
            return
        r, c = pos
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            if 0 <= nr < rows and 0 <= nc < cols:
                if (nr, nc) not in visited and (nr, nc) not in rocks:
                    visited.add((nr, nc))
                    path.append((nr, nc))
                    dfs((nr, nc), path, visited)
                    path.pop()
                    visited.remove((nr, nc))
    
    dfs(start, [start], {start})
    
    # Visualize safe channels
    channel_maze = [row[:] for row in maze]
    for path in all_paths:
        for (r, c) in path[1:-1]:  # exclude start/goal
            if channel_maze[r][c] == 0:
                channel_maze[r][c] = '○'  # safe channel marker
    
    print_maze(channel_maze, "P1 - Safe Channels (○ = safe reachable)")
    print(f"Safe channels found: {len(all_paths)} possible paths")
    return all_paths

def p2_optimize_within_channels(all_paths) -> List[Tuple[int, int]]:
    """P2: Select shortest path among safe channels."""
    if not all_paths:
        return []
    shortest = min(all_paths, key=len)
    
    # Visualize optimal path
    opt_maze = [row[:] for row in MAZE]
    for (r, c) in shortest[1:-1]:
        if opt_maze[r][c] == 0 or opt_maze[r][c] == '○':
            opt_maze[r][c] = '★'
    
    print_maze(opt_maze, "P2 - Optimal Path (★ = chosen path)")
    print(f"Optimal path length: {len(shortest)-1} steps")  # exclude start
    return shortest

def unconstrained_greedy(maze, start, goal):
    """P2-only (violation): greedy Manhattan distance, ignores rocks."""
    rows, cols = len(maze), len(maze[0])
    path = [start]
    visited = {start}
    pos = start
    
    while pos != goal:
        r, c = pos
        # Greedy: move toward goal
        dr = 1 if goal[0] > r else -1 if goal[0] < r else 0
        dc = 1 if goal[1] > c else -1 if goal[1] < c else 0
        
        # Try primary direction
        nr, nc = r + dr, c + dc
        if 0 <= nr < rows and 0 <= nc < cols and (nr, nc) not in visited:
            pass
        else:
            # Fallback: any open neighbor
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                nr, nc = r + dr, c + dc

                    break
            else:
                break  # stuck
        
        if maze[nr][nc] == 1:
            print(f"  GREEDY HIT ROCK at ({nr},{nc})!")
            return None  # failure
        
        pos = (nr, nc)
        path.append(pos)
        visited.add(pos)
    
    return path if pos == goal else None

**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Reading Recap:**
- `fleet-knowledge/narrow-games/constraint_sim_final.json` → deadband success: 50/50, avg steps: 21.0 (optimal)
- `DEADBAND-PROTOCOL.md` → strict P0 (rocks), P1 (channels), P2 (optimize) priority

**Approach:**
1. **P0 (Map Rocks):** Generate a random 10x10 maze with walls ('#') as rocks.
2. **P1 (Find Channels):** Use BFS to find all reachable empty cells ('.') from start, marking them as safe.
3. **P2 (Optimize Path):** Use A* search within safe cells to find shortest path from start to goal.
4. **Visualization:** Print maze with path overlaid ('*').

**Output:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 11)
Demonstrates P0→P1→P2 priority in a navigational domain.
"""

import random
from collections import deque
import heapq

def generate_maze(width=10, height=10, wall_prob=0.25):
    """Generate random maze with walls as rocks (P0 negative space)."""
    maze = []
    for y in range(height):
        row = []
        for x in range(width):
            if random.random() < wall_prob and not (x == 0 and y == 0) and not (x == width-1 and y == height-1):
                row.append('#')  # rock
            else:
                row.append('.')  # open
        maze.append(row)
    maze[0][0] = 'S'  # start
    maze[height-1][width-1] = 'G'  # goal
    return maze

def p0_map_rocks(maze):
    """P0: Identify all rock positions (negative space)."""
    rocks = []
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start=(0,0)):
    """P1: BFS to find all reachable safe cells (deadband channels)."""
    width, height = len(maze[0]), len(maze)
    safe = set()
    queue = deque([start])
    visited = set([start])
    
    while queue:
        x, y = queue.popleft()
        if maze[y][x] != '#':  # not a rock
            safe.add((x, y))
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height and (nx, ny) not in visited:
                visited.add((nx, ny))
                queue.append((nx, ny))
    return safe

def p2_optimize_path(maze, safe, start=(0,0), goal=None):
    """P2: A* search within safe channels for shortest path."""
    if goal is None:
        goal = (len(maze[0])-1, len(maze)-1)
    if start not in safe or goal not in safe:
        return []  # no safe path
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == goal:
            # reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            return path
        
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            neighbor = (x + dx, y + dy)
            if 0 <= neighbor[0] < len(maze[0]) and 0 <= neighbor[1] < len(maze):
                if neighbor not in safe:
                    continue  # not in safe channel (P1 violation)
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    return []  # no path found

def visualize(maze, path):
    """Print maze with path overlaid."""
    display = [row[:] for row in maze]
    for (x, y) in path:
        if display[y][x] not in ('S', 'G'):
            display[y][x] = '*'
    
    print("Maze (S=start, G=goal, #=rock, .=open, *=path):")
    for row in display:
        print(' '.join(row))
    print()

def main():
    random.seed(42)  # reproducible for demo
    maze = generate_maze(10, 10, wall_prob=0.25)
    
    print("=== DEADBAND PROTOCOL DEMO ===")
    print("P0: Map rocks (negative space)")
    rocks = p0_map_rocks(maze)
    print(f"  Found {len(rocks)} rock cells")
    
    print("\nP1: Find safe channels (deadbands)")
    safe = p1_find_safe_channels(maze)
    print(f"  {len(safe)} safe reachable cells")
    
    print("\nP2: Optimize path within safe channels")
    path = p2_optimize_path(maze, safe)
    if path:
        print(f"  Path length: {len(path)} steps")
    else:
        print("  No safe path to goal exists")
    
    print("\nVisualization:")
    visualize(maze, path)
    
    # Demonstrate protocol violation (skip P0/P1, go straight to P2)
    print("=== PROTOCOL VIOLATION (P2-only greedy) ===")
    print("Attempting direct A* without deadband mapping...")
    all_cells = [(x, y) for y in range(len(maze)) for x in range(len(maze[0]))]
    naive_path = p2_optimize_path(maze, set(all_cells))  # treat all cells as "safe"
    if naive_path:
        print(f"  Naive path length: {len(naive_path)} steps")
        # Check if naive path hits rocks
        hits_rocks = any(maze[y][x] == '#' for (x, y) in naive_path)
        print(f"  Hits rocks: {hits_rocks}")
        if hits_rocks:
            print("  → Violation result: Path is physically impossible")
    else:
        print("  No path found (even with rock ignoring)")
    
    print("\n" + "="*40)
    print("Summary: Deadband Protocol ensures path stays in safe channels.")
    print("P2-only optimization often produces invalid paths.")

if __name__ == "__main__":
    main()
```

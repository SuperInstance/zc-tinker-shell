"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 priority in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import random
from typing import List, Tuple, Set
from enum import Enum

class Cell(Enum):
    EMPTY = '·'
    ROCK = '█'
    AGENT = 'A'
    GOAL = 'G'
    PATH = '○'
    DEADBAND = '░'

class ProtocolPhase(Enum):
    P0_NEGATIVE_SPACE = "P0: Map negative space (rocks)"
    P1_SAFE_CHANNELS = "P1: Identify safe channels (deadbands)"
    P2_OPTIMIZE_PATH = "P2: Optimize within safe channels"

class DeadbandAgent:
    def __init__(self, maze_size: int = 20, rock_density: float = 0.16):
        self.size = maze_size
        self.rocks: Set[Tuple[int, int]] = set()
        self.deadbands: Set[Tuple[int, int]] = set()
        self.path: List[Tuple[int, int]] = []
        
        # Generate rocks (64 rocks for 20x20 = 16% density)
        total_cells = maze_size * maze_size
        num_rocks = int(total_cells * rock_density)
        
        while len(self.rocks) < num_rocks:
            x = random.randint(0, maze_size - 1)
            y = random.randint(0, maze_size - 1)
            # Avoid start and goal positions
            if (x, y) != (0, 0) and (x, y) != (maze_size-1, maze_size-1):
                self.rocks.add((x, y))
        
        # Start and goal
        self.start = (0, 0)
        self.goal = (maze_size-1, maze_size-1)
        
        # Statistics
        self.steps_p0 = 0
        self.steps_p1 = 0
        self.steps_p2 = 0

        """Display the maze with current state"""

            grid[y][x] = Cell.ROCK.value
            
        # Mark deadbands if showing
        if show_deadbands:
            for x, y in self.deadbands:
                if grid[y][x] == Cell.EMPTY.value:
                    grid[y][x] = Cell.DEADBAND.value
        
        # Mark path if showing

            for x, y in self.path:
                if (x, y) != self.start and (x, y) != self.goal:
                    grid[y][x] = Cell.PATH.value

        grid[sy][sx] = Cell.AGENT.value
        grid[gy][gx] = Cell.GOAL.value

        for row in grid:
            print("|" + " ".join(row) + "|")
        print("+" + "-" * (self.size * 2 - 1) + "+")

    def p1_find_safe_channels(self):
        """P1: Find safe channels (deadbands) using flood fill"""
        print(f"\n{ProtocolPhase.P1_SAFE_CHANNELS.value}")
        
        # Simple flood fill from start, avoiding rocks
        visited = set()
        stack = [self.start]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            if (x, y) in self.rocks:
                continue
                
            visited.add((x, y))
            
            # Add valid neighbors

                nx, ny = x + dx, y + dy

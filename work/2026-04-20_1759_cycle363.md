#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

A visual ASCII simulation of the Deadband Protocol (P0→P1→P2).
Demonstrates the strict priority: map rocks, find safe channels, then optimize.

Based on fleet-knowledge/narrow-games/constraint_sim_final.json
and DEADBAND-PROTOCOL.md.

import random
import time
from typing import List, Tuple, Set

class DeadbandMaze:
    """A 20x20 maze where an agent must navigate from start (0,0) to goal (19,19)."""
    
    def __init__(self, seed: int = 42):
        self.width = 20
        self.height = 20
        self.start = (0, 0)
        self.goal = (19, 19)
        self.rocks: Set[Tuple[int, int]] = set()
        self.agent_pos = self.start
        self.steps = 0
        self.path: List[Tuple[int, int]] = [self.start]
        
        # Initialize rocks (64 rocks, ~16% density)
        random.seed(seed)
        possible_cells = [(x, y) for x in range(self.width) for y in range(self.height)]
        possible_cells.remove(self.start)
        possible_cells.remove(self.goal)
        self.rocks = set(random.sample(possible_cells, 64))
        
        # Deadband protocol state
        self.p0_done = False  # Rocks mapped?
        self.p1_done = False  # Safe channels identified?
        self.p2_done = False  # Optimized path found?
        self.safe_channels: Set[Tuple[int, int]] = set()
        self.optimal_path: List[Tuple[int, int]] = []
    
    def render(self, show_rocks: bool = True, show_agent: bool = True) -> str:
        """Return ASCII representation of the maze."""
        lines = []
        lines.append("=" * 42)
        lines.append("DEADBAND PROTOCOL MAZE (20×20)")
        lines.append("=" * 42)
        
        for y in range(self.height):
            line_chars = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    line_chars.append('S')
                elif pos == self.goal:
                    line_chars.append('G')
                elif show_agent and pos == self.agent_pos:
                    line_chars.append('A')
                elif show_rocks and pos in self.rocks:
                    line_chars.append('█')  # Rock
                elif pos in self.safe_channels:
                    line_chars.append('·')  # Safe channel
                else:
                    line_chars.append(' ')
            lines.append('|' + ''.join(line_chars) + '|')
        
        lines.append("=" * 42)
        
        # Status
        status = []
        if self.p0_done:
            status.append("P0: Rocks mapped ✓")
        if self.p1_done:
            status.append("P1: Safe channels found ✓")
        if self.p2_done:
            status.append("P2: Optimizing within channels ✓")
        
        if status:
            lines.append(" | ".join(status))
        
        lines.append(f"Steps: {self.steps} | Position: {self.agent_pos}")
        return '\n'.join(lines)
    
    def p0_map_rocks(self) -> None:
        """P0: Map the negative space (what NOT to do)."""
        print("\n[P0] MAPPING ROCKS (Negative Space)")
        print(f"Identified {len(self.rocks)} rock positions.")
        self.p0_done = True
        time.sleep(1)
    
    def p1_find_safe_channels(self) -> None:
        """P1: Find safe channels (where you CAN be)."""
        print("\n[P1] FINDING SAFE CHANNELS")
        
        # Simple flood fill from start, avoiding rocks
        visited = set()
        stack = [self.start]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Add to safe channels if not a rock
            if (x, y) not in self.rocks:
                self.safe_channels.add((x, y))
            
            # Explore neighbors (4-directional)
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        stack.append((nx, ny))
        
        print(f"Found {len(self.safe_channels)} safe cells.")
        self.p1_done = True
        time.sleep(1)
    
    def p2_optimize_within_channels(self) -> None:
        """P2: Optimize within safe channels (best path)."""
        print("\n[P2] OPTIMIZING PATH WITHIN SAFE CHANNELS")
        
        # A* search constrained to safe channels
        open_set = {self.start}
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: self._heuristic(self.start, self.goal)}
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score.get(pos, float('inf')))
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                self.optimal_path = list(reversed(path))
                print(f"Found optimal path of length {len(self.optimal_path)}")
                self.p2_done = True
                return
            
            open_set.remove(current)
            
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # Constraint: must be in safe channels
                if neighbor not in self.safe_channels:
                    continue
                
                tentative_g = g_score[current] + 1
                
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor, self.goal)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        print("WARNING: No path found within safe channels!")
    
    def _heuristic(self, a: Tuple[int, int], b: Tuple[int, int]) -> int:
        """Manhattan distance heuristic."""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def run_deadband_protocol(self) -> bool:
        """Execute full P0→P1→P2 protocol."""
        print("\n" + "="*60)
        print("EXECUTING DEADBAND PROTOCOL (P0→P1→P2)")
        print("="*60)
        
        # P0: Map rocks
        self.p0_map_rocks()
        print(self.render(show_rocks=True, show_agent=False))
        
        # P1: Find safe channels
        self.p1_find_safe_channels()
        print(self.render(show_rocks=False, show_agent=False))
        
        # P2: Optimize within channels
        self.p2_optimize_within_channels()
        
        if not self.optimal_path:
            print("\n❌ FAILURE: No safe path to goal.")
            return False
        
        # Animate the path
        print("\n🎯 SUCCESS: Navigating optimal path...")
        for i, pos in enumerate(self.optimal_path):
            self.agent_pos = pos
            self.steps = i
            print(f"\nStep {i}: {pos}")
            print(self.render(show_rocks=False, show_agent=True))
            time.sleep(0.1)
        
        print(f"\n✅ Goal reached in {len(self.optimal_path)-1} steps!")
        print(f"Optimal theoretical: 38 steps (Manhattan distance)")
        return True

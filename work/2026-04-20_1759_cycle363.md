#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 363)
A visual ASCII simulation of P0→P1→P2 in a constrained navigation task.
"""

import random
import time
from typing import List, Tuple, Set

class DeadbandMaze:
    """
    Simulates an agent navigating a 10x10 maze using the Deadband Protocol.
    """
    
    def __init__(self, seed: int = 42):
        random.seed(seed)
        self.width = 10
        self.height = 10
        self.start = (0, 0)
        self.goal = (9, 9)
        self.rocks: Set[Tuple[int, int]] = set()
        self.agent_pos = self.start
        self.steps = 0
        self._generate_maze()
    
    def _generate_maze(self) -> None:
        """Generate random obstacles (rocks) covering ~25% of cells."""
        total_cells = self.width * self.height
        num_rocks = total_cells // 4
        
        # Ensure start and goal are clear
        forbidden = {self.start, self.goal}
        candidates = [(x, y) for x in range(self.width) for y in range(self.height) 
                     if (x, y) not in forbidden]
        
        self.rocks = set(random.sample(candidates, num_rocks))
    
    def print_maze(self, show_path: List[Tuple[int, int]] = None) -> None:
        """Render the maze with ASCII characters."""
        if show_path is None:
            show_path = []
        
        print("\n" + "="*30)
        print("MAZE STATE (Cycle {})".format(self.steps))
        print("Legend: . = empty, # = rock, S = start, G = goal, @ = agent, * = safe channel")
        print("="*30)
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos == self.agent_pos:
                    row.append('@')
                elif pos in self.rocks:
                    row.append('#')
                elif pos in show_path:
                    row.append('*')
                else:
                    row.append('.')
            print(' '.join(row))
        print()
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify all rocks (failure states)."""
        print("[P0] MAPPING NEGATIVE SPACE")
        print(f"Found {len(self.rocks)} rocks (failure states).")
        return self.rocks.copy()
    
    def p1_find_safe_channels(self, rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
        """P1: Find all paths from start to goal avoiding rocks."""
        print("[P1] FINDING SAFE CHANNELS")
        
        # Simple BFS to find one safe path (for demo)
        from collections import deque
        
        queue = deque([(self.start, [self.start])])
        visited = {self.start}
        safe_paths = []
        
        while queue and len(safe_paths) < 3:  # Limit to 3 paths for clarity
            pos, path = queue.popleft()
            
            if pos == self.goal:
                safe_paths.append(path)
                continue
            
            # Generate valid moves (up, down, left, right)
            x, y = pos
            moves = [(x+1, y), (x-1, y), (x, y+1), (x, y-1)]
            valid_moves = [(nx, ny) for nx, ny in moves 
                          if 0 <= nx < self.width and 0 <= ny < self.height
                          and (nx, ny) not in rocks]
            
            for move in valid_moves:
                if move not in visited:
                    visited.add(move)
                    queue.append((move, path + [move]))
        
        if safe_paths:
            print(f"Found {len(safe_paths)} safe channel(s).")
            for i, path in enumerate(safe_paths[:2]):  # Show first 2
                print(f"  Channel {i+1}: {len(path)} steps")
        else:
            print("WARNING: No safe channels found!")
        
        return safe_paths
    
    def p2_optimize_within_channel(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """P2: Select shortest path from safe channels."""
        print("[P2] OPTIMIZING WITHIN CHANNEL")
        
        if not safe_paths:
            print("ERROR: Cannot optimize without safe channels!")
            return []
        
        # Find shortest path
        optimal_path = min(safe_paths, key=len)
        print(f"Selected optimal path: {len(optimal_path)} steps")
        return optimal_path

            gx, gy = self.goal

            if abs(gx - x) >= abs(gy - y):
                nx = x + (1 if gx > x else -1)
                ny = y
            else:
                nx = x
                ny = y + (1 if gy > y else -1)

            nx = max(0, min(self.width-1, nx))

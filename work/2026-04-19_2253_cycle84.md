#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 84)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize within channel).
Based on fleet simulation data: unconstrained fails, deadband succeeds.
"""

import random
from typing import List, Tuple, Set

# ========== MAZE DEFINITION ==========
# 10x10 maze, '.' = open, '#' = rock, 'S' = start, 'G' = goal
MAZE_STR = """
##########
#S.......#
#.###.#..#
#.#...#..#
#.#.######
#.#......#
#.#.######
#.#......#
#......G.#
##########
"""

def parse_maze(maze_str: str) -> Tuple[List[List[str]], Tuple[int, int], Tuple[int, int]]:
    """Parse maze string into grid, find start and goal."""
    lines = [line.strip() for line in maze_str.strip().split('\n')]
    grid = [list(line) for line in lines]
    start = goal = None
    for r in range(len(grid)):
        for c in range(len(grid[r])):
            if grid[r][c] == 'S':
                start = (r, c)
            elif grid[r][c] == 'G':
                goal = (r, c)
    if start is None or goal is None:
        raise ValueError("Maze must contain 'S' and 'G'")
    return grid, start, goal

def print_maze(grid: List[List[str]], path: List[Tuple[int, int]] = None):
    """Print maze with optional path marked."""
    for r in range(len(grid)):
        row = []
        for c in range(len(grid[r])):
            if path and (r, c) in path:
                row.append('*')
            else:
                row.append(grid[r][c])
        print(''.join(row))

# ========== DEADBAND PROTOCOL IMPLEMENTATION ==========
class DeadbandNavigator:
    def __init__(self, grid, start, goal):
        self.grid = grid
        self.start = start
        self.goal = goal
        self.rows = len(grid)
        self.cols = len(grid[0])
        self.rocks = set()      # P0: negative space
        self.safe = set()       # P1: safe channels
        self.path = []          # P2: optimal path within safe channels

    def p0_map_rocks(self):
        """P0: Identify all rock cells (impassable)."""
        for r in range(self.rows):
            for c in range(self.cols):
                if self.grid[r][c] == '#':
                    self.rocks.add((r, c))
        print(f"P0 — Mapped {len(self.rocks)} rocks.")

    def p1_find_safe_channels(self):
        """P1: Find all reachable cells that are not rocks (safe channels)."""
        # BFS from start, avoiding rocks
        queue = [self.start]
        visited = {self.start}
        while queue:
            r, c = queue.pop(0)
            self.safe.add((r, c))
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < self.rows and 0 <= nc < self.cols:
                    if (nr, nc) not in visited and (nr, nc) not in self.rocks:
                        visited.add((nr, nc))
                        queue.append((nr, nc))
        print(f"P1 — Found {len(self.safe)} safe cells.")

    def p2_optimize_within_channel(self):
        """P2: Find shortest path from start to goal staying within safe cells."""
        # BFS constrained to safe set
        queue = [(self.start, [self.start])]
        visited = {self.start}
        while queue:
            (r, c), current_path = queue.pop(0)
            if (r, c) == self.goal:
                self.path = current_path
                return
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < self.rows and 0 <= nc < self.cols:
                    if (nr, nc) not in visited and (nr, nc) in self.safe:
                        visited.add((nr, nc))
                        queue.append(((nr, nc), current_path + [(nr, nc)]))
        print("P2 — No safe path to goal.")

    def run_deadband(self):
        """Execute full Deadband Protocol."""
        print("\n=== DEADBAND PROTOCOL (P0→P1→P2) ===")
        self.p0_map_rocks()
        self.p1_find_safe_channels()
        self.p2_optimize_within_channel()
        if self.path:
            print(f"P2 — Found optimal path of length {len(self.path)-1} steps.")
        return self.path

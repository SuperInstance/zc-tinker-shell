Deadband Protocol Maze Demo

Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import random
import sys
from typing import List, Tuple, Set

# Constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21  # from constraint_sim_final.json

class Maze:
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.generate_rocks()
        
    def generate_rocks(self):
        """Place rocks randomly, avoiding start and goal."""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[y][x] = '#'
    
    def is_rock(self, x: int, y: int) -> bool:
        return (x, y) in self.rocks
    
    def in_bounds(self, x: int, y: int) -> bool:
        return 0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE
    
    def print_maze(self, path=None):
        """Print ASCII maze with optional path."""
        for y in range(MAZE_SIZE):
            row = []
            for x in range(MAZE_SIZE):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif path and (x, y) in path:
                    row.append('*')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
        print()

class Agent:
    def __init__(self, maze: Maze):
        self.maze = maze
        self.x, self.y = maze.start
        
    def move_greedy(self) -> bool:
        """P2-only: greedy toward goal, ignoring rocks."""
        if (self.x, self.y) == self.maze.goal:
            return True
            
        # Move right or down randomly (toward goal)
        moves = []
        if self.x < MAZE_SIZE-1:
            moves.append((self.x+1, self.y))
        if self.y < MAZE_SIZE-1:
            moves.append((self.x, self.y+1))
            
        if not moves:
            return False
            
        self.x, self.y = random.choice(moves)
        return False
    
    def move_constrained(self) -> bool:
        """Naive constraint: avoid rocks if visible."""
        if (self.x, self.y) == self.maze.goal:
            return True
            
        moves = []
        # Check right
        if self.x < MAZE_SIZE-1 and not self.maze.is_rock(self.x+1, self.y):
            moves.append((self.x+1, self.y))
        # Check down
        if self.y < MAZE_SIZE-1 and not self.maze.is_rock(self.x, self.y+1):
            moves.append((self.x, self.y+1))
            
        if not moves:
            # If both moves are rocks, stay put (fail)
            return False
            
        self.x, self.y = random.choice(moves)
        return False
    
    def deadband_protocol(self) -> Tuple[bool, List[Tuple[int, int]]]:
        """Full P0→P1→P2 protocol."""
        # P0: Map negative space (rocks)
        rocks = self.maze.rocks.copy()
        
        # P1: Find safe channels (BFS to find all rock-free paths)
        from collections import deque
        
        visited = set()
        queue = deque([(self.maze.start, [self.maze.start])])
        safe_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            if (x, y) == self.maze.goal:
                safe_paths.append(path)
                continue
                
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Generate next moves (right, down, left, up)
            for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (self.maze.in_bounds(nx, ny) and 
                    not self.maze.is_rock(nx, ny) and 
                    (nx, ny) not in visited):
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        # P2: Optimize within safe channels
        if not safe_paths:
            return False, []
            
        # Choose shortest safe path
        optimal_path = min(safe_paths, key=len)
        
        # Follow the path
        for step in optimal_path[1:]:  # skip start
            self.x, self.y = step
            
        return True, optimal_path

def run_simulation(strategy: str, runs: int = 50) -> Tuple[int, float]:
    """Run multiple simulations and return success count and average steps."""
    successes = 0
    total_steps = 0
    
    for _ in range(runs):
        maze = Maze()
        agent = Agent(maze)
        steps = 0
        success = False
        
        if strategy == "unconstrained":
            while steps < 1000:  # prevent infinite loops
                steps += 1
                if agent.move_greedy():
                    success = True
                    break
                if maze.is_rock(agent.x, agent.y):
                    break  # hit rock
                    
        elif strategy == "constrained":
            while steps < 1000:
                steps += 1
                if agent.move_constrained():
                    success = True
                    break
                # Constrained agent stops if no moves
                if (agent.x, agent.y) == maze.start and steps > 1:
                    break
                    
        elif strategy == "deadband":
            success, path = agent.deadband_protocol()
            steps = len(path) if success else 0
            
        if success:
            successes += 1
            total_steps += steps
    
    avg_steps = total_steps / successes if successes > 0 else 0
    return successes, avg_steps

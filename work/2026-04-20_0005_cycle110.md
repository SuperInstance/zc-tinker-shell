```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 strict priority in a simple 10x10 ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random

class DeadbandMaze:
    def __init__(self, width=10, height=10, rock_density=0.2):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (height-1, width-1)
        self.rocks = set()
        self.safe_channels = []
        self.optimal_path = []
        
        # P0: Map negative space (rocks)
        self._place_rocks(rock_density)
        
    def _place_rocks(self, density):
        """P0: Identify all states where failure probability > threshold."""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        
        # Avoid start and goal
        available = [(r, c) for r in range(self.height) for c in range(self.width)]
        available.remove(self.start)
        available.remove(self.goal)
        
        self.rocks = set(random.sample(available, min(num_rocks, len(available))))
        for r, c in self.rocks:
            self.grid[r][c] = '#'
            
    def _find_safe_channels(self):
        """P1: Find all paths where ∀ s ∈ path: s ∉ rocks."""
        # Simple BFS to find all safe paths from start to goal
        from collections import deque
        
        visited = set()
        queue = deque([(self.start, [self.start])])
        safe_paths = []
        
        while queue:
            (r, c), path = queue.popleft()
            
            if (r, c) == self.goal:
                safe_paths.append(path)
                continue
                
            if (r, c) in visited:
                continue
            visited.add((r, c))
            
            # Generate safe neighbors (not rocks, within bounds)
            for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
                nr, nc = r + dr, c + dc
                if (0 <= nr < self.height and 0 <= nc < self.width and 
                    (nr, nc) not in self.rocks):
                    queue.append(((nr, nc), path + [(nr, nc)]))
        
        self.safe_channels = safe_paths
        return safe_paths
    
    def _optimize_within_channel(self):
        """P2: From safe channels, select shortest path."""
        if not self.safe_channels:
            return []
        
        # Find path with minimum cost (length)
        self.optimal_path = min(self.safe_channels, key=len)
        return self.optimal_path
    
    def run_protocol(self):
        """Execute full P0→P1→P2 deadband protocol."""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===\n")
        
        # P0: Show negative space
        print("P0: MAP NEGATIVE SPACE (Rocks)")
        print(f"Identified {len(self.rocks)} rock cells (failure states)")
        self.display_maze()
        print()
        
        # P1: Find safe channels
        print("P1: IDENTIFY SAFE CHANNELS")
        safe_paths = self._find_safe_channels()
        print(f"Found {len(safe_paths)} safe channels (rock-free paths)")
        
        if safe_paths:
            # Show one example safe path
            example_path = safe_paths[0]
            print(f"Example safe channel length: {len(example_path)} steps")
            self.display_maze_with_path(example_path, 'O')
        else:
            print("WARNING: No safe channels found! Goal unreachable.")
        print()
        
        # P2: Optimize within channel
        print("P2: OPTIMIZE WITHIN CHANNEL")
        if safe_paths:
            optimal = self._optimize_within_channel()
            print(f"Optimal path length: {len(optimal)} steps")
            self.display_maze_with_path(optimal, '*')
            
            # Compare with naive approach (P2-only)
            naive_length = abs(self.goal[0]-self.start[0]) + abs(self.goal[1]-self.start[1])
            print(f"\nNaive Manhattan distance (P2-only): {naive_length} steps")
            print(f"Deadband optimal (P0+P1+P2): {len(optimal)} steps")
            
            if len(optimal) > naive_length:
                print("NOTE: Safe path is longer than naive distance (avoids rocks)")
        else:
            print("Cannot optimize: no safe channels (P1 failed)")
        
        return len(self.safe_channels) > 0
    
    def display_maze(self):
        """Display maze with rocks, start, goal."""
        for r in range(self.height):
            row = []
            for c in range(self.width):
                if (r, c) == self.start:
                    row.append('S')
                elif (r, c) == self.goal:
                    row.append('G')
                elif (r, c) in self.rocks:
                    row.append('#')
                else:
                    row.append('.')
            print(' '.join(row))
    
    def display_maze_with_path(self, path, path_char):
        """Display maze with path overlay."""
        path_set = set(path)
        for r in range(self.height):
            row = []
            for c in range(self.width):
                if (r, c) == self.start:
                    row.append('S')
                elif (r, c) == self.goal:
                    row.append('G')
                elif (r, c) in self.rocks:
                    row.append('#')
                elif (r, c) in path_set:
                    row.append(path_char)
                else:
                    row.append('.')
            print(' '.join(row))

def main():
    """Run demonstration with different random seeds."""
    print("Deadband Protocol Maze Demo")
    print("Based on Cocapn Fleet constraint simulation\n")
    
    # Seed for reproducibility
    random.seed(42)
    
    # Create and run maze
    maze = DeadbandMaze(width=10, height=10, rock_density=0.2)
    success = maze.run_protocol()
    
    print("\n" + "="*50)
    print("SIMULATION RESULTS:")
    print(f"- Rocks (P0 negative space): {len(maze.rocks)}")
    print(f"- Safe channels found (P1): {len(maze.safe_channels)}")
    
    if maze.safe_channels:
        optimal_len = len(maze.optimal_path) if maze.optimal_path else 0
        print(f"- Optimal path length (P2): {optimal_len}")
        print(f"- Success: {'YES' if success else 'NO'}")
        
        # Show protocol violation warning
        if len(maze.safe_channels) == 0:
            print("\n⚠️  PROTOCOL VIOLATION: Attempting P2 without P1 would fail")
    else:
        print("- Success: NO (no safe channels)")
        print("\n⚠️  PROTOCOL VIOLATION: P2-only approach guaranteed to fail")

if __name__ == "__main__":
    main()
```

**Explanation:**
1. **P0 (Map negative space)**: Randomly places rocks ('#') as failure states, avoiding start (S) and goal (G).
2. **P1 (Identify safe channels)**: Uses BFS to find all rock-free paths from S to G.
3. **P2 (Optimize within channel)**: Selects shortest path among safe channels.
4. **Visualization**: Shows maze at each stage with ASCII art.
5. **Protocol validation**: Compares deadband approach with naive P2-only (Manhattan distance).

**Key insight from fleet data:** The simulation showed 50/50 success with deadband vs 0/50 with P2-only. This demo visualizes why: P2-only ignores rocks and fails, while P0+P1+P2 finds safe channels first.

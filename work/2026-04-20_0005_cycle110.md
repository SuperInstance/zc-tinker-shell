import random

from collections import deque

class Maze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self.rocks = set()
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density and (x, y) not in [self.start, self.goal]:
                    self.rocks.add((x, y))
        
        # Ensure at least one path exists
        self._ensure_path()
    
    def _ensure_path(self):
        """Guarantee at least one safe channel exists."""
        # Simple right-then-down path
        for x in range(self.width):
            if (x, 0) in self.rocks:
                self.rocks.remove((x, 0))
        for y in range(self.height):
            if (self.width-1, y) in self.rocks:
                self.rocks.remove((self.width-1, y))
    
    def is_safe(self, x, y):
        return 0 <= x < self.width and 0 <= y < self.height and (x, y) not in self.rocks
    
    def get_neighbors(self, x, y):
        neighbors = []
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if self.is_safe(nx, ny):
                neighbors.append((nx, ny))
        return neighbors
    
    def render(self, path=None, visited=None):
        """ASCII visualization of maze state."""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif (x, y) in self.rocks:
                    row.append('█')
                elif path and (x, y) in path:
                    row.append('•')
                elif visited and (x, y) in visited:
                    row.append('.')
                else:
                    row.append(' ')
            grid.append(''.join(row))
        return '\n'.join(grid)

class DeadbandAgent:
    def __init__(self, maze):

        # P2: Optimize within channels
        print("\nP2: OPTIMIZE WITHIN CHANNELS")

        self.p2_path = self._optimize_within_channels()
        if self.p2_path:
            print(f"Path length: {len(self.p2_path)} steps")
            print(self.maze.render(path=self.p2_path))
            print("✓ SUCCESS: Goal reached via deadband protocol")
            return True
        else:
            print("✗ FAILURE: No safe path to goal")
            return False
    
    def _find_safe_channels(self):
        """BFS to find all reachable safe areas."""
        visited = set()
        channels = []
        
        queue = deque([self.maze.start])
        visited.add(self.maze.start)
        
        while queue:
            current = queue.popleft()
            channel = [current]
            
            # Explore connected safe region
            stack = [current]
            while stack:
                x, y = stack.pop()
                for nx, ny in self.maze.get_neighbors(x, y):
                    if (nx, ny) not in visited:
                        visited.add((nx, ny))
                        channel.append((nx, ny))
                        stack.append((nx, ny))
                        queue.append((nx, ny))
            
            if len(channel) > 1 or current == self.maze.start:
                channels.append(channel)
        
        return channels
    
    def _optimize_within_channels(self):
        """A* search constrained to safe cells."""
        from heapq import heappush, heappop
        
        start = self.maze.start
        goal = self.maze.goal
        
        # Heuristic: Manhattan distance
        def heuristic(a, b):
            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        open_set = []
        heappush(open_set, (0, start))
        
        came_from = {}
        g_score = {start: 0}
        f_score = {start: heuristic(start, goal)}
        
        while open_set:
            _, current = heappop(open_set)
            
            if current == goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(start)
                return list(reversed(path))
            
            for neighbor in self.maze.get_neighbors(*current):
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heappush(open_set, (f_score[neighbor], neighbor))
        
        return None

"""
Deadband Protocol Maze Demo — P0→P1→P2 Visual Simulation

A simple ASCII maze simulation that demonstrates the Deadband Protocol:
P0: Map negative space (rocks)
P1: Identify safe channels (deadbands)
P2: Optimize path within safe channels

Based on fleet-knowledge/docs/DEADBAND-PROTOCOL.md
Simulation data from fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time
from typing import List, Tuple, Set

# Simulation constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

# ASCII symbols
WALL = '█'
ROCK = '▓'
SAFE = '░'
AGENT = 'A'
GOAL = 'G'
PATH = '·'
EMPTY = ' '

class DeadbandMaze:
    """Maze environment implementing Deadband Protocol"""
    
    def __init__(self):
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.safe_channels: List[Set[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
        # Initialize maze
        self._place_rocks()
        self._place_start_goal()
    
    def _place_rocks(self):
        """P0: Map negative space (rocks)"""
        # Random rocks, avoiding start and goal
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[y][x] = ROCK
    
    def _place_start_goal(self):
        """Place agent and goal positions"""
        self.grid[self.start[1]][self.start[0]] = AGENT
        self.grid[self.goal[1]][self.goal[0]] = GOAL
    
    def find_safe_channels(self):
        """P1: Identify safe channels (deadbands)"""
        # Simple flood fill to find all reachable positions avoiding rocks
        visited = set()
        stack = [self.start]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Mark as safe if not a rock
            if (x, y) not in self.rocks and (x, y) != self.goal:
                self.grid[y][x] = SAFE
            
            # Explore neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    stack.append((nx, ny))
        
        self.safe_channels = [visited]
        return visited
    
    def optimize_path(self):
        """P2: Optimize path within safe channels"""
        # BFS to find shortest path through safe channels
        queue = [(self.start, [self.start])]
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.goal:
                self.optimal_path = path
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        return []  # No path found
    
    def run_greedy_failure(self):
        """Demonstrate P2-only (greedy) failure"""
        path = []
        current = self.start
        
        while current != self.goal:
            path.append(current)
            x, y = current
            
            # Greedy: move toward goal without checking rocks
            if x < self.goal[0]:
                next_pos = (x + 1, y)
            elif y < self.goal[1]:
                next_pos = (x, y + 1)
            else:
                next_pos = (x - 1, y) if x > self.goal[0] else (x, y - 1)
            
            # Check if hit rock
            if next_pos in self.rocks:
                return False, path  # Failure
            
            current = next_pos
        
        path.append(self.goal)
        return True, path
    
    def display(self, title="", highlight_path=None):
        """Display the maze with optional path highlighting"""
        print(f"\n{'='*60}")
        print(f"DEADBAND PROTOCOL DEMO: {title}")
        print(f"{'='*60}\n")
        
        display_grid = [row[:] for row in self.grid]
        
        if highlight_path:
            for (x, y) in highlight_path[1:-1]:  # Skip start and goal
                if display_grid[y][x] == SAFE:
                    display_grid[y][x] = PATH
        
        for row in display_grid:
            print(''.join(row))
        
        print(f"\nLegend: {AGENT}=Agent, {GOAL}=Goal, {ROCK}=Rock, {SAFE}=Safe, {PATH}=Path")
        print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}, Optimal: {OPTIMAL_PATH_LENGTH} steps")

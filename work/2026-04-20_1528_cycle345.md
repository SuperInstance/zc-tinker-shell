#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — ASCII Visualization
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
import time
from collections import deque

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # ~16% density, matches 64 rocks in 20x20
START = (0, 0)
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

def generate_maze():
    """Create a maze with random rocks (obstacles)."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        r = random.randint(0, MAZE_SIZE-1)
        c = random.randint(0, MAZE_SIZE-1)
        if (r, c) != START and (r, c) != GOAL:
            rocks.add((r, c))
            maze[r][c] = '#'
    maze[START[0]][START[1]] = 'S'
    maze[GOAL[0]][GOAL[1]] = 'G'
    return maze, rocks

def print_maze(maze, path=None, visited=None):
    """Display maze with optional path and visited cells."""
    for r in range(MAZE_SIZE):
        row = []
        for c in range(MAZE_SIZE):
            cell = maze[r][c]
            if path and (r, c) in path:
                row.append('*')
            elif visited and (r, c) in visited:
                row.append('~')
            else:
                row.append(cell)
        print(' '.join(row))
    print()

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks)."""
    rocks = []
    for r in range(MAZE_SIZE):
        for c in range(MAZE_SIZE):
            if maze[r][c] == '#':
                rocks.append((r, c))
    print(f"P0: Mapped {len(rocks)} rocks (negative space)")
    return set(rocks)

def p1_find_safe_channels(maze, rocks):
    """P1: Find all safe cells (not rocks)."""
    safe = set()
    for r in range(MAZE_SIZE):
        for c in range(MAZE_SIZE):
            if (r, c) not in rocks:
                safe.add((r, c))
    print(f"P1: Found {len(safe)} safe cells (channel)")
    return safe

def p2_optimize_path(maze, safe, start, goal):
    """P2: BFS to find shortest path within safe channel."""
    if start not in safe or goal not in safe:
        return None, set()
    
    queue = deque([start])
    parent = {start: None}
    visited = set([start])
    
    while queue:
        current = queue.popleft()
        if current == goal:
            break
        r, c = current
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            neighbor = (nr, nc)
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if neighbor in safe and neighbor not in visited:
                    visited.add(neighbor)
                    parent[neighbor] = current
                    queue.append(neighbor)
    
    # Reconstruct path
    if goal not in parent:
        return None, visited
    
    path = []
    node = goal
    while node is not None:
        path.append(node)
        node = parent[node]
    path.reverse()
    print(f"P2: Found optimal path of length {len(path)} steps")
    return path, visited

def run_unconstrained_greedy(maze, start, goal):
    """Naive greedy approach (P2-only) for comparison."""
    path = [start]
    visited = set([start])
    current = start
    
    while current != goal:
        r, c = current
        # Simple heuristic: move toward goal
        if r < goal[0]:
            nr, nc = r+1, c
        elif c < goal[1]:
            nr, nc = r, c+1
        else:
            nr, nc = r-1, c-1  # fallback
        
        if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
            if maze[nr][nc] == '#':
                print("  Greedy agent hit a rock!")
                return None, visited
            current = (nr, nc)
            path.append(current)
            visited.add(current)
        else:
            break
    
    return path, visited

def main():
    print("=" * 60)
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("=" * 60)
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}")
    print(f"Start: {START}, Goal: {GOAL}")
    print()
    
    # Generate maze
    maze, rocks = generate_maze()
    print("Initial Maze (S=start, G=goal, #=rock, .=empty):")
    print_maze(maze)
    time.sleep(1)
    
    print("\n" + "=" * 60)
    print("SCENARIO 1: Unconstrained Greedy (P2-only)")
    print("=" * 60)
    greedy_path, greedy_visited = run_unconstrained_greedy(maze, START, GOAL)
    if greedy_path:
        print("Path found (but likely suboptimal):")
        print_maze(maze, path=set(greedy_path))
    else:
        print("FAILED: Greedy agent hit rocks (common outcome)")
        print_maze(maze, visited=greedy_visited)
    time.sleep(2)
    
    print("\n" + "=" * 60)
    print("SCENARIO 2: Deadband Protocol (P0→P1→P2)")
    print("=" * 60)
    
    # P0: Map rocks
    rocks_set = p0_map_rocks(maze)
    time.sleep(0.5)
    
    # P1: Find safe channels
    safe = p1_find_safe_channels(maze, rocks_set)
    time.sleep(0.5)
    
    # P2: Optimize within channel
    deadband_path, deadband_visited = p2_optimize_path(maze, safe, START, GOAL)
    time.sleep(0.5)
    
    if deadband_path:
        print("\nDeadband Protocol SUCCESS:")
        print(f"  Path length: {len(deadband_path)} steps")
        print(f"  Cells visited: {len(deadband_visited)}")
        print("\nVisualization (*=path, ~=visited, #=rock):")
        print_maze(maze, path=set(deadband_path), visited=deadband_visited)
        
        # Compare with optimal from fleet data
        print("Fleet simulation results (from constraint_sim_final.json):")
        print("  Deadband success rate: 50/50 (100%)")
        print("  Deadband avg steps: 21.0")
        print("  Unconstrained success: 33/50 (66%)")
        print("  Constrained success: 0/50 (0%)")
    else:
        print("\nDeadband Protocol FAILED: No path exists in this maze")
        print_maze(maze, visited=deadband_visited)
    
    print("\n" + "=" * 60)
    print("CONCLUSION")
    print("=" * 60)
    print("Deadband Protocol ensures success by:")
    print("  1. First identifying what NOT to do (rocks)")
    print("  2. Then finding where you CAN operate (safe cells)")
    print("  3. Only then optimizing path within safe space")
    print("\nViolating priority (P2 before P0+P1) leads to rock collisions.")

if __name__ == "__main__":
    main()

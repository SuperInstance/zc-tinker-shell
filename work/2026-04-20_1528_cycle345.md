#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

A visual ASCII simulation of the Deadband Protocol (P0→P1→P2) in a 10x10 maze.
Demonstrates:
  P0: Map negative space (rocks/obstacles)
  P1: Identify safe channels (deadbands)
  P2: Optimize path within safe channels

Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set

# ==================== MAZE DEFINITION ====================
# '.' = open, '#' = rock (obstacle), 'S' = start, 'G' = goal
MAZE_RAW = [
    "..........",
    "..#.......",
    "...#......",
    "....#.....",
    ".....#....",
    "......#...",
    ".......#..",
    "........#.",
    "..........",
    ".........."
]

START = (0, 0)
GOAL = (9, 9)

# ==================== DEADBAND PROTOCOL IMPLEMENTATION ====================

def print_maze(maze: List[str], title: str = "Maze") -> None:
    """Print the maze with a title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def p0_map_negative_space(maze: List[str]) -> Set[Tuple[int, int]]:
    """
    P0: Identify all rock positions (negative space).
    Returns set of (row, col) coordinates that are obstacles.
    """
    rocks = set()
    for r, row in enumerate(maze):
        for c, ch in enumerate(row):
            if ch == '#':
                rocks.add((r, c))
    print(f"P0: Mapped {len(rocks)} rocks (negative space).")
    return rocks

def p1_find_safe_channels(maze: List[str], rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
    """
    P1: Find all safe paths from START to GOAL that avoid rocks.
    Uses simple BFS to find all shortest safe paths.
    Returns list of paths (each path is list of coordinates).
    """
    rows, cols = len(maze), len(maze[0])
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]  # right, down, left, up
    
    # BFS queue: (position, path_so_far)
    queue = [(START, [START])]
    safe_paths = []
    visited = {START}
    shortest_length = None
    
    while queue:
        (r, c), path = queue.pop(0)
        
        # If we reached the goal
        if (r, c) == GOAL:
            if shortest_length is None:
                shortest_length = len(path)
            if len(path) == shortest_length:
                safe_paths.append(path)
            continue
        
        # Explore neighbors
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < rows and 0 <= nc < cols:
                if (nr, nc) not in rocks and (nr, nc) not in visited:
                    visited.add((nr, nc))
                    queue.append(((nr, nc), path + [(nr, nc)]))
    
    print(f"P1: Found {len(safe_paths)} safe channels (deadbands) of length {shortest_length}.")
    return safe_paths

def p2_optimize_within_channel(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """
    P2: Select the optimal path from safe channels.
    Here, 'optimal' means the path with minimal turns (simulating energy cost).
    Returns the chosen path.
    """
    if not safe_paths:
        print("P2: No safe channels available. Optimization impossible.")
        return []
    
    def count_turns(path: List[Tuple[int, int]]) -> int:
        """Count direction changes in path."""
        turns = 0
        if len(path) < 3:
            return 0
        # Get direction vectors between consecutive steps
        for i in range(1, len(path)-1):
            r1, c1 = path[i-1]
            r2, c2 = path[i]
            r3, c3 = path[i+1]
            dr1, dc1 = r2 - r1, c2 - c1
            dr2, dc2 = r3 - r2, c3 - c2
            if (dr1, dc1) != (dr2, dc2):
                turns += 1
        return turns
    
    # Choose path with minimum turns
    optimal_path = min(safe_paths, key=count_turns)
    turns = count_turns(optimal_path)
    print(f"P2: Selected optimal path with {turns} turns (from {len(safe_paths)} safe channels).")
    return optimal_path

def visualize_path(maze: List[str], path: List[Tuple[int, int]], label: str) -> None:
    """Create a copy of maze with path marked, and print it."""
    maze_copy = [list(row) for row in maze]
    for r, c in path:
        if maze_copy[r][c] == '.':
            maze_copy[r][c] = '*'
    maze_copy[START[0]][START[1]] = 'S'
    maze_copy[GOAL[0]][GOAL[1]] = 'G'
    
    print(f"\n=== {label} ===")
    for row in maze_copy:
        print(' '.join(row))
    print()

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Visual demonstration of P0→P1→P2 in a 10x10 ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
import time

class DeadbandMaze:
    def __init__(self, width=10, height=10, rock_density=0.2):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.rocks = set()
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.agent_pos = self.start
        self.steps = 0
        self.path = []
        
        # P0: Map negative space (rocks)
        self._place_rocks(rock_density)
        
        # P1: Find safe channels (deadbands)
        self.safe_channels = self._find_safe_channels()
        
        # P2: Optimal path within safe channels
        self.optimal_path = self._find_optimal_path()
    
    def _place_rocks(self, density):
        """P0: Identify rocks (negative space)"""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        
        # Avoid start and goal
        available = [(x, y) for x in range(self.width) for y in range(self.height)]
        available.remove(self.start)
        available.remove(self.goal)
        
        self.rocks = set(random.sample(available, min(num_rocks, len(available))))
        for (x, y) in self.rocks:
            self.grid[y][x] = '#'
    
    def _find_safe_channels(self):
        """P1: Find all cells not in rocks (safe channels)"""
        safe = []
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) not in self.rocks:
                    safe.append((x, y))
        return safe
    
    def _find_optimal_path(self):
        """P2: Find shortest path from start to goal avoiding rocks"""
        # Simple BFS within safe channels
        from collections import deque
        
        queue = deque([(self.start, [])])
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                return path + [(x, y)]
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(x, y)]))
        
        return []  # No path found
    
    def print_maze(self, show_agent=False, show_path=False):
        """Display current maze state"""
        display = [row[:] for row in self.grid]
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        display[sy][sx] = 'S'
        display[gy][gx] = 'G'
        
        if show_path and self.optimal_path:
            for (x, y) in self.optimal_path[1:-1]:  # Exclude start and goal
                if display[y][x] == '.':
                    display[y][x] = '*'
        
        if show_agent:
            ax, ay = self.agent_pos
            if display[ay][ax] not in ['S', 'G']:
                display[ay][ax] = 'A'
        
        print('+' + '-' * (self.width * 2 - 1) + '+')
        for row in display:
            print('|' + ' '.join(row) + '|')
        print('+' + '-' * (self.width * 2 - 1) + '+')
    
    def run_simulation(self):
        """Run the full P0→P1→P2 demonstration"""
        print("=" * 50)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("=" * 50)
        
        print("\n[P0] MAP NEGATIVE SPACE (Rocks)")
        print("Identifying all failure states (rocks '#')...")
        self.print_maze()
        print(f"Rocks identified: {len(self.rocks)}")
        time.sleep(1.5)
        
        print("\n[P1] IDENTIFY SAFE CHANNELS (Deadbands)")
        print("Finding all cells not in rocks (safe water)...")
        safe_count = len(self.safe_channels)
        rock_count = len(self.rocks)
        total_cells = self.width * self.height
        print(f"Safe cells: {safe_count}/{total_cells} ({safe_count/total_cells*100:.1f}%)")
        print(f"Rock cells: {rock_count}/{total_cells} ({rock_count/total_cells*100:.1f}%)")
        time.sleep(1.5)
        
        print("\n[P2] OPTIMIZE WITHIN CHANNEL")
        print("Finding shortest path through safe channels...")
        if self.optimal_path:
            print(f"Optimal path found: {len(self.optimal_path)-1} steps")
            self.print_maze(show_path=True)
            
            # Animate the path
            print("\nAnimating optimal path (A=agent):")
            self.agent_pos = self.start
            for step, (x, y) in enumerate(self.optimal_path):
                self.agent_pos = (x, y)
                print(f"\nStep {step}:")
                self.print_maze(show_agent=True)
                time.sleep(0.3)
            
            print(f"\n✓ SUCCESS: Reached goal in {len(self.optimal_path)-1} steps")
            return True
        else:
            print("✗ FAILURE: No safe path to goal exists")
            self.print_maze(show_path=True)
            return False

def main():
    """Run the demonstration"""
    print("Initializing Deadband Protocol Maze Demo...")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print(f"Reference: deadband_success={50/50*100}%, deadband_avg={21.0} steps")
    
    # Create maze with same parameters as simulation
    maze = DeadbandMaze(width=10, height=10, rock_density=0.2)
    
    success = maze.run_simulation()
    
    print("\n" + "=" * 50)
    print("DEMONSTRATION COMPLETE")
    print("=" * 50)
    print("\nKey Insights:")
    print("1. P0 (rocks) MUST be mapped first - otherwise agent fails")
    print("2. P1 (channels) constrains search space to safe regions")
    print("3. P2 (optimization) only works within P1 constraints")
    print(f"\nResult: {'Success with optimal path' if success else 'Failure - no safe path'}")
    print("This matches fleet simulation: deadband protocol → 50/50 success")

if __name__ == "__main__":
    main()

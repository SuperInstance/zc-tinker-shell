import random
import time

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.rock_density = rock_density
        self.rocks = set()
        self.agent_pos = (0, 0)
        self.goal = (width-1, height-1)
        self.path = []
        self.safe_channels = set()
        
    def generate_maze(self):
        """P0: Map negative space (rocks)"""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * self.rock_density)
        
        # Avoid start and goal positions
        possible = [(x, y) for x in range(self.width) for y in range(self.height)]
        possible.remove(self.agent_pos)
        possible.remove(self.goal)
        
        self.rocks = set(random.sample(possible, num_rocks))
        return self.rocks
    
    def find_safe_channels(self):
        """P1: Identify safe channels (deadbands)"""
        # Simple flood fill from start, avoiding rocks
        visited = set()
        queue = [self.agent_pos]
        
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Add to safe channels if not a rock
            if (x, y) not in self.rocks:
                self.safe_channels.add((x, y))
                
                # Explore neighbors (4-directional)
                for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                    nx, ny = x + dx, y + dy
                    if 0 <= nx < self.width and 0 <= ny < self.height:
                        if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                            queue.append((nx, ny))
        
        return self.safe_channels
    
    def optimize_path(self):
        """P2: Find optimal path within safe channels (A* simplified)"""
        if self.goal not in self.safe_channels:
            return []  # No safe path to goal
            
        # Simple greedy path toward goal (Manhattan distance)
        path = [self.agent_pos]
        current = self.agent_pos
        
        while current != self.goal:
            x, y = current
            candidates = []
            
            # Generate possible next steps
            for dx, dy in [(1,0), (0,1)]:  # Only move right or down (simplified)
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) in self.safe_channels:
                        # Manhattan distance to goal as heuristic
                        dist = abs(nx - self.goal[0]) + abs(ny - self.goal[1])
                        candidates.append(((nx, ny), dist))
            
            if not candidates:
                return []  # Stuck
            
            # Choose candidate with smallest distance to goal
            candidates.sort(key=lambda x: x[1])
            current = candidates[0][0]
            path.append(current)
        
        self.path = path
        return path
    
    def run_unconstrained(self):
        """P2-only approach (greedy optimization without P0/P1)"""
        path = [self.agent_pos]
        current = self.agent_pos
        
        while current != self.goal:
            x, y = current
            
            # Move toward goal without checking rocks
            if x < self.goal[0]:
                next_pos = (x + 1, y)
            elif y < self.goal[1]:
                next_pos = (x, y + 1)
            else:
                break
            
            if next_pos in self.rocks:
                return []  # Hit a rock
            
            path.append(next_pos)
            current = next_pos
        
        return path
    
    def visualize(self, show_path=False, show_channels=False):
        """Render ASCII maze"""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.agent_pos:
                    row.append('A')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')
                elif show_path and pos in self.path:
                    row.append('·')
                elif show_channels and pos in self.safe_channels:
                    row.append('░')
                else:
                    row.append(' ')
            grid.append(''.join(row))
        return '\n'.join(grid)

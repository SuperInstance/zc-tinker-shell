#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize within channel).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json data.
"""

import random
from typing import List, Tuple, Set

# Simulation parameters from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

def generate_maze(size: int, rock_count: int) -> Tuple[List[List[str]], Tuple[int, int], Tuple[int, int]]:
    """Generate a size x size maze with rocks, start (top-left), goal (bottom-right)."""
    maze = [['.' for _ in range(size)] for _ in range(size)]
    
    # Place rocks randomly, avoiding start and goal
    rocks_placed = 0
    while rocks_placed < rock_count:
        x, y = random.randint(0, size-1), random.randint(0, size-1)
        if (x, y) == (0, 0) or (x, y) == (size-1, size-1):
            continue
        if maze[y][x] != '#':
            maze[y][x] = '#'
            rocks_placed += 1
    
    start = (0, 0)
    goal = (size-1, size-1)
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    
    return maze, start, goal

def print_maze(maze: List[List[str]], path: List[Tuple[int, int]] = None):
    """Print maze with optional path overlay."""
    for y in range(len(maze)):
        row = ''
        for x in range(len(maze[y])):
            if path and (x, y) in path:
                row += '○'  # path marker
            else:
                row += maze[y][x]
        print(row)
    print()

def p0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Identify negative space (rocks)."""
    rocks = set()
    for y in range(len(maze)):
        for x in range(len(maze[y])):
            if maze[y][x] == '#':
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(maze: List[List[str]], start: Tuple[int, int], goal: Tuple[int, int], rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
    """P1: Find all safe paths (deadbands) using BFS, avoiding rocks."""
    from collections import deque
    
    size = len(maze)
    directions = [(1, 0), (0, 1), (-1, 0), (0, -1)]  # right, down, left, up
    
    # BFS to find all shortest safe paths
    queue = deque([(start, [start])])
    visited = {start: 0}
    safe_paths = []
    shortest_length = None
    
    while queue:
        (x, y), path = queue.popleft()
        
        if (x, y) == goal:
            if shortest_length is None:
                shortest_length = len(path)
            if len(path) == shortest_length:
                safe_paths.append(path)
            continue
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < size and 0 <= ny < size:
                if (nx, ny) in rocks:
                    continue  # avoid rocks
                if (nx, ny) not in visited or visited[(nx, ny)] >= len(path) + 1:
                    visited[(nx, ny)] = len(path) + 1
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    return safe_paths

def p2_optimize_within_channel(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """P2: Select the optimal path from safe channels (here, first shortest)."""
    if not safe_paths:
        return []
    # For simplicity, pick the first shortest path
    return safe_paths[0]

def greedy_unconstrained(maze: List[List[str]], start: Tuple[int, int], goal: Tuple[int, int]) -> List[Tuple[int, int]]:
    """Greedy P2-only approach (no P0/P1) for comparison."""
    x, y = start
    path = [start]
    size = len(maze)
    
    while (x, y) != goal:
        # Prefer moving right or down (toward goal)
        if x < size-1 and maze[y][x+1] != '#':
            x += 1
        elif y < size-1 and maze[y+1][x] != '#':
            y += 1
        elif x > 0 and maze[y][x-1] != '#':
            x -= 1
        elif y > 0 and maze[y-1][x] != '#':
            y -= 1
        else:
            break  # stuck
        path.append((x, y))
    
    return path if path[-1] == goal else []

def run_demo():
    """Run the full Deadband Protocol demonstration."""
    print("=" * 60)
    print("DEADBAND PROTOCOL DEMO — ASCII MAZE")
    print("=" * 60)
    
    # Generate maze matching simulation parameters
    maze, start, goal = generate_maze(MAZE_SIZE, ROCK_COUNT)
    
    print("Initial Maze (S=start, G=goal, #=rock, .=empty):")
    print_maze(maze)
    
    # === P0: Map negative space ===
    rocks = p0_map_rocks(maze)
    print(f"P0: Mapped {len(rocks)} rocks (negative space).")
    
    # === P1: Find safe channels ===
    safe_paths = p1_find_safe_channels(maze, start, goal, rocks)
    print(f"P1: Found {len(safe_paths)} safe channels (deadbands).")
    
    # === P2: Optimize within channel ===
    optimal_path = p2_optimize_within_channel(safe_paths)
    if optimal_path:
        print(f"P2: Selected optimal path length = {len(optimal_path)-1} steps.")
        print("Path (○ marks):")
        print_maze(maze, optimal_path)
    else:
        print("P2: No safe path found.")
    
    # === Compare with greedy (P2-only) ===
    print("-" * 60)
    print("GREEDY P2-ONLY (no P0/P1) for comparison:")
    greedy_path = greedy_unconstrained(maze, start, goal)
    if greedy_path and greedy_path[-1] == goal:
        print(f"Greedy path length = {len(greedy_path)-1} steps.")
        print_maze(maze, greedy_path)
    else:
        print("Greedy failed — hit rocks or got stuck.")
    
    # === Summary ===
    print("=" * 60)
    print("SUMMARY:")
    print(f"  Maze size: {MAZE_SIZE}x{MAZE_SIZE}")
    print(f"  Rocks: {ROCK_COUNT}")
    print(f"  Deadband Protocol success: {'Yes' if optimal_path else 'No'}")
    print(f"  Greedy success: {'Yes' if greedy_path and greedy_path[-1]==goal else 'No'}")
    print(f"  Optimal path length from sim data: {OPTIMAL_PATH_LENGTH}")
    print("=" * 60)

if __name__ == "__main__":
    random.seed(42)  # reproducible for demo
    run_demo()
```

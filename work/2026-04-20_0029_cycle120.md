```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo - Tinker Cycle 120
Demonstrates P0→P1→P2 protocol with visual ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    """Maze solver implementing Deadband Protocol."""
    
    def __init__(self, width: int = 10, height: int = 10, rock_density: float = 0.2):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
        # Generate random rocks (P0 negative space)
        for y in range(height):
            for x in range(width):
                if (x, y) not in [self.start, self.goal] and random.random() < rock_density:
                    self.rocks.add((x, y))
    
    def p0_map_rocks(self) -> Set[Tuple[int, int]]:
        """P0: Map negative space (what NOT to do)."""
        print("=== P0: MAP ROCKS ===")
        print(f"Identified {len(self.rocks)} rock cells (failure states)")
        print("Rocks are positions where agent would fail/crash.")
        return self.rocks
    
    def p1_find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Find safe channels (where you CAN be)."""
        print("\n=== P1: FIND SAFE CHANNELS ===")
        
        # Simple BFS to find all safe paths from start to goal
        from collections import deque
        
        safe_paths = []
        queue = deque([([self.start], {self.start})])
        
        while queue:
            path, visited = queue.popleft()
            x, y = path[-1]
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            # Check adjacent cells (right, down, left, up)
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        new_path = path + [(nx, ny)]
                        new_visited = visited | {(nx, ny)}
                        queue.append((new_path, new_visited))
        
        self.safe_channels = safe_paths
        print(f"Found {len(safe_paths)} safe channels from start to goal")
        
        if safe_paths:
            print(f"Shortest safe channel: {len(safe_paths[0])} steps")
            print(f"Longest safe channel: {len(safe_paths[-1])} steps")
        
        return safe_paths
    
    def p2_optimize_in_channel(self) -> List[Tuple[int, int]]:
        """P2: Optimize within safe channels (best path)."""
        print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        if not self.safe_channels:
            print("ERROR: Cannot optimize without safe channels (P1 not completed)")
            return []
        
        # Find shortest path among safe channels
        self.optimal_path = min(self.safe_channels, key=len)
        print(f"Selected optimal path: {len(self.optimal_path)} steps")
        print("Optimization constrained to pre-identified safe channels only.")
        
        return self.optimal_path
    
    def display_maze(self, path: List[Tuple[int, int]] = None):
        """Display ASCII maze with rocks, start, goal, and optional path."""
        print("\n" + "="*40)
        print("MAZE VISUALIZATION")
        print("="*40)
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')  # Rock
                elif path and pos in path:
                    row.append('·')  # Path marker
                else:
                    row.append('.')  # Empty space
            print(' '.join(row))
        
        legend = [
            "S = Start",
            "G = Goal", 
            "█ = Rock (P0 negative space)",
            "· = Safe path (if shown)",
            ". = Navigable space"
        ]
        print("\nLegend:", ' | '.join(legend))
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2."""
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*50)
        
        # P0: Map rocks
        rocks = self.p0_map_rocks()
        self.display_maze()
        
        # P1: Find safe channels
        channels = self.p1_find_safe_channels()
        
        if not channels:
            print("\nRESULT: No safe channels found. Agent cannot reach goal.")
            print("This demonstrates P0 preventing certain failure.")
            return
        
        # P2: Optimize within channel
        optimal = self.p2_optimize_in_channel()
        
        # Show final result
        print("\n" + "="*50)
        print("FINAL RESULT: SUCCESS")
        print(f"Optimal path length: {len(optimal)} steps")
        print("Path respects all constraints (never hits rocks).")
        
        self.display_maze(optimal)
        
        # Compare with unconstrained approach
        print("\n" + "="*50)
        print("CONTRAST WITH UNCONSTRAINED APPROACH:")
        print("An agent skipping P0/P1 would:")
        print("1. Try straight-line path to goal")
        print("2. Likely hit rocks (probability → 1.0 as maze grows)")
        print("3. Fail to reach goal")
        print("\nBased on constraint_sim_final.json:")
        print("- Unconstrained success: 33/100")
        print("- Deadband success: 50/50")

def main():
    """Run the demonstration."""
    # Seed for reproducible example
    random.seed(42)
    
    # Create maze with moderate rock density
    maze = DeadbandMaze(width=8, height=8, rock_density=0.15)
    
    # Execute Deadband Protocol
    maze.run_protocol()
    
    print("\n" + "="*50)
    print("PROTOCOL VALIDATION:")
    print("✓ P0 completed before P1 (rocks mapped)")
    print("✓ P1 completed before P2 (channels identified)")
    print("✓ P2 constrained to safe channels only")
    print("✓ No optimization attempted without safety mapping")

if __name__ == "__main__":
    main()
```

**Verification against Deadband Protocol:**

1. **P0 (Map rocks)**: First step identifies failure states (rocks)
2. **P1 (Safe channels)**: Finds all paths avoiding rocks before optimization  
3. **P2 (Optimize)**: Only selects shortest path from safe channels
4. **Strict priority**: Cannot skip to P2 without P0+P1

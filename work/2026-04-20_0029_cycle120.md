```python
#!/usr/bin/env python3
"""
deadband_maze_demo.py

A visual demonstration of the Deadband Protocol (P0→P1→P2) in a maze.
P0: Map rocks (obstacles)
P1: Find safe channels (paths avoiding rocks)
P2: Optimize within channels (shortest safe path)
"""

import sys
from collections import deque

# Example maze: '#' = rock/wall, '.' = open, 'S' = start, 'G' = goal
MAZE = [
    "############",
    "#S.........#",
    "#.###.####.#",
    "#.#......#.#",
    "#.#.####.#.#",
    "#.#.#....#.#",
    "#.#.#.####.#",
    "#...#......#",
    "############",
]

def print_maze(maze, path=None):
    """Print the maze, optionally highlighting a path with '*'."""
    for y, row in enumerate(maze):
        for x, ch in enumerate(row):
            if path and (y, x) in path:
                sys.stdout.write('*')
            else:
                sys.stdout.write(ch)
        sys.stdout.write('\n')

def find_position(maze, symbol):
    """Find the (y, x) coordinates of a symbol in the maze."""
    for y, row in enumerate(maze):
        for x, ch in enumerate(row):
            if ch == symbol:
                return (y, x)
    return None

def p0_map_rocks(maze):
    """P0: Map negative space — identify all rock/wall positions."""
    rocks = set()
    for y, row in enumerate(maze):
        for x, ch in enumerate(row):
            if ch == '#':
                rocks.add((y, x))
    return rocks

def p1_find_safe_channels(maze, start, goal, rocks):
    """P1: Find safe channels — BFS to discover all reachable positions avoiding rocks."""
    rows, cols = len(maze), len(maze[0])
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    
    visited = set()
    queue = deque([start])
    visited.add(start)
    
    while queue:
        y, x = queue.popleft()
        if (y, x) == goal:
            continue
        for dy, dx in directions:
            ny, nx = y + dy, x + dx
            if 0 <= ny < rows and 0 <= nx < cols:
                if (ny, nx) not in rocks and (ny, nx) not in visited:
                    visited.add((ny, nx))
                    queue.append((ny, nx))
    return visited

def p2_optimize_path(maze, start, goal, rocks, safe_set):
    """P2: Optimize within safe channels — BFS for shortest path staying in safe_set."""
    rows, cols = len(maze), len(maze[0])
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    
    queue = deque([(start, [start])])
    visited = set([start])
    
    while queue:
        (y, x), path = queue.popleft()
        if (y, x) == goal:
            return path
        for dy, dx in directions:
            ny, nx = y + dy, x + dx
            if 0 <= ny < rows and 0 <= nx < cols:
                if (ny, nx) in safe_set and (ny, nx) not in visited:
                    visited.add((ny, nx))
                    queue.append(((ny, nx), path + [(ny, nx)]))
    return None  # No safe path found

def main():
    print("=== Deadband Protocol Maze Demo ===\n")
    
    # Find start and goal
    start = find_position(MAZE, 'S')
    goal = find_position(MAZE, 'G')
    if not start or not goal:
        print("Error: Start or goal not found in maze.")
        return
    
    print("Original Maze:")
    print_maze(MAZE)
    print(f"Start: {start}, Goal: {goal}\n")
    
    # P0: Map rocks
    rocks = p0_map_rocks(MAZE)
    print(f"P0 — Rocks mapped: {len(rocks)} rock positions identified.\n")
    
    # P1: Find safe channels (reachable positions avoiding rocks)
    safe_set = p1_find_safe_channels(MAZE, start, goal, rocks)
    print(f"P1 — Safe channels mapped: {len(safe_set)} reachable positions (including start/goal).")
    print("Safe positions are all '.' and 'S', 'G' not blocked by rocks.\n")
    
    # P2: Optimize within safe channels
    path = p2_optimize_path(MAZE, start, goal, rocks, safe_set)
    if path:
        print(f"P2 — Optimal path found: {len(path)-1} steps.")
        print("Path visualized with '*':")
        print_maze(MAZE, set(path))
    else:
        print("P2 — No safe path to goal exists within channels.")
    
    # Protocol adherence check
    print("\n=== Protocol Adherence ===")
    if rocks and safe_set and path:
        print("✓ P0 (map rocks) completed before P1.")
        print("✓ P1 (safe channels) defined before P2.")
        print("✓ P2 (optimization) constrained to safe channels only.")
        print("Deadband Protocol followed correctly.")
    else:
        print("✗ Protocol violation detected.")

if __name__ == "__main__":
    main()
```

**Output explanation:**
- The script defines a simple ASCII maze with walls (`#`), start (`S`), goal (`G`), and open cells (`.`).
- **P0:** Identifies all rock positions (walls).
- **P1:** Performs BFS to find all positions reachable from start while avoiding rocks — this defines the "safe channels".
- **P2:** Performs BFS again, but only within the safe set, to find the shortest path to the goal.
- The final visualization shows the optimal path with `*` characters.

**Key Deadband adherence:**
1. P0 must complete before P1 (rocks mapped first).
2. P1 must complete before P2 (safe channels defined).
3. P2 optimization is constrained to the safe set only — it never considers moving into rocks.

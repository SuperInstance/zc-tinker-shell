#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import random
from typing import List, Tuple, Set
from collections import deque

class DeadbandMaze:
    """A 20x20 maze simulation of the Deadband Protocol."""
    
    def __init__(self, seed=42):
        self.width = 20
        self.height = 20
        self.start = (0, 0)
        self.goal = (19, 19)
        self.rocks: Set[Tuple[int, int]] = set()
        self._generate_maze(seed)
        
    def _generate_maze(self, seed):
        """Generate a maze with 64 rocks (16% density) as in the simulation."""
        random.seed(seed)
        total_cells = self.width * self.height
        rock_count = 64
        
        # Place rocks randomly, avoiding start and goal
        all_cells = [(x, y) for x in range(self.width) for y in range(self.height)]
        all_cells.remove(self.start)
        all_cells.remove(self.goal)
        
        self.rocks = set(random.sample(all_cells, rock_count))
    
    def print_maze(self, path=None, deadband=None):
        """Print ASCII maze with optional path and deadband visualization."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')  # Rock
                elif path and pos in path:
                    row.append('·')  # Path
                elif deadband and pos in deadband:
                    row.append('░')  # Safe channel (deadband)
                else:
                    row.append(' ')
            print(''.join(row))
    
    def p0_map_rocks(self) -> Set[Tuple[int, int]]:
        """P0: Map negative space (rocks)."""
        print("P0: Mapping rocks (negative space)...")
        print(f"  Found {len(self.rocks)} rocks at positions:")
        # Show first few rocks as example
        for i, rock in enumerate(list(self.rocks)[:5]):
            print(f"    {rock}", end="")
            if i == 4 and len(self.rocks) > 5:
                print(f" ... and {len(self.rocks)-5} more")
            else:
                print()
        return self.rocks.copy()
    
    def p1_find_safe_channels(self) -> Set[Tuple[int, int]]:
        """P1: Find safe channels (deadband) using BFS flood fill."""
        print("\nP1: Finding safe channels (deadband)...")
        
        safe = set()
        queue = deque([self.start])
        visited = set([self.start])
        
        # 4-direction movement
        directions = [(1,0), (-1,0), (0,1), (0,-1)]
        
        while queue:
            x, y = queue.popleft()
            safe.add((x, y))
            
            for dx, dy in directions:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    pos = (nx, ny)
                    if pos not in self.rocks and pos not in visited:
                        visited.add(pos)
                        queue.append(pos)
        
        # Check if goal is reachable
        if self.goal in safe:
            print(f"  Safe channel contains {len(safe)} cells")
            print(f"  Goal is reachable: YES")
        else:
            print(f"  WARNING: Goal not in safe channel!")
        
        return safe
    
    def p2_optimize_in_channel(self, safe_channel: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
        """P2: Find shortest path within safe channel using BFS."""
        print("\nP2: Optimizing path within safe channel...")
        
        if self.goal not in safe_channel:
            print("  Cannot optimize: goal not in safe channel")
            return []
        
        # BFS for shortest path
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        directions = [(1,0), (-1,0), (0,1), (0,-1)]
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                print(f"  Found optimal path length: {len(path)-1} steps")
                return path
            
            for dx, dy in directions:
                nx, ny = x + dx, y + dy
                pos = (nx, ny)
                if (0 <= nx < self.width and 0 <= ny < self.height and 
                    pos in safe_channel and pos not in visited):
                    visited.add(pos)
                    queue.append((pos, path + [pos]))
        
        print("  No path found within safe channel")
        return []
    
    def run_greedy_p2_only(self):
        """Violation: P2-only (greedy) optimization without P0+P1."""
        print("\n=== VIOLATION: P2-ONLY (GREEDY) ===")
        print("Skipping P0 and P1, optimizing directly for goal...")
        
        # Greedy BFS that ignores rock mapping
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        directions = [(1,0), (-1,0), (0,1), (0,-1)]
        
        steps = 0
        max_steps = 1000
        
        while queue and steps < max_steps:
            (x, y), path = queue.popleft()
            steps += 1
            
            if (x, y) == self.goal:
                print(f"  Greedy found path of {len(path)-1} steps")
                print("  BUT: This path likely hits rocks (we didn't check P0!)")
                return path
            
            # Explore neighbors randomly (greedy but unconstrained)
            neighbors = [(x+dx, y+dy) for dx, dy in directions]
            random.shuffle(neighbors)
            
            for nx, ny in neighbors:
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    pos = (nx, ny)
                    if pos not in visited:
                        visited.add(pos)
                        queue.append((pos, path + [pos]))
        
        print("  Greedy search failed or timed out")
        return []

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map rocks (negative space)
P1: Find safe channels (deadbands)
P2: Optimize path within safe channels
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_count=64):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self.rocks = set()
        while len(self.rocks) < rock_count:
            x = random.randint(0, width-1)
            y = random.randint(0, height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
        
        # Initialize grid
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        for x, y in self.rocks:
            self.grid[y][x] = '#'
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def print_maze(self, path=None):
        """Visual ASCII display"""
        print("\n" + "="*40)
        print("MAZE (20×20, 64 rocks)")
        print("S=Start, G=Goal, #=Rock, .=Empty, *=Path")
        print("="*40)
        
        display = [row[:] for row in self.grid]
        if path:
            for x, y in path[1:-1]:  # Skip start and goal
                if display[y][x] == '.':
                    display[y][x] = '*'
        
        for y in range(self.height):
            row = ''
            for x in range(self.width):
                row += display[y][x] + ' '
            print(row)
        print()
    
    def p0_map_rocks(self):
        """P0: Identify negative space (what NOT to do)"""
        print("P0: MAPPING NEGATIVE SPACE")
        print(f"  • Found {len(self.rocks)} rocks (failure states)")
        print(f"  • Rocks at coordinates: {sorted(list(self.rocks))[:5]}...")
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all safe paths (deadbands)"""
        print("\nP1: FINDING SAFE CHANNELS")
        
        # BFS to find all reachable safe cells
        visited = set()
        queue = deque([self.start])
        safe_cells = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            safe_cells.add((x, y))
            
            # Check neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) not in self.rocks and
                    (nx, ny) not in visited):
                    queue.append((nx, ny))
        
        print(f"  • Safe reachable cells: {len(safe_cells)}")
        print(f"  • Goal reachable: {self.goal in safe_cells}")
        return safe_cells
    
    def p2_optimize_in_channel(self, safe_cells):
        """P2: Find optimal path within safe channels"""
        print("\nP2: OPTIMIZING WITHIN SAFE CHANNEL")
        
        if self.goal not in safe_cells:
            print("  ✗ Goal not reachable (no safe channel)")
            return None
        
        # A* search constrained to safe cells only
        open_set = {self.start}
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: self._heuristic(self.start, self.goal)}
        
        while open_set:
            current = min(open_set, key=lambda x: f_score.get(x, float('inf')))
            
            if current == self.goal:
                path = self._reconstruct_path(came_from, current)
                print(f"  • Optimal path length: {len(path)-1} steps")
                print(f"  • Path: {path[:3]}...{path[-3:]}")
                return path
            
            open_set.remove(current)
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # CONSTRAINT: Must stay in safe channel
                if neighbor not in safe_cells:
                    continue
                
                tentative_g = g_score[current] + 1
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor, self.goal)
                    open_set.add(neighbor)
        
        return None
    
    def _heuristic(self, a, b):
        """Manhattan distance"""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def _reconstruct_path(self, came_from, current):
        path = []
        while current in came_from:
            path.append(current)
            current = came_from[current]
        path.append(self.start)
        path.reverse()
        return path
    
    def run_unconstrained_greedy(self):
        """Violation: P2 without P0+P1 (what NOT to do)"""
        print("\n--- VIOLATION: UNCONSTRAINED GREEDY (P2 without P0+P1) ---")
        
        # Greedy approach ignoring rocks
        path = [self.start]
        current = self.start
        visited = set([self.start])
        
        while current != self.goal and len(path) < 100:
            # Move toward goal ignoring constraints
            dx = 1 if self.goal[0] > current[0] else -1 if self.goal[0] < current[0] else 0
            dy = 1 if self.goal[1] > current[1] else -1 if self.goal[1] < current[1] else 0
            
            next_cell = (current[0] + dx, current[1] + dy)
            
            # Check bounds
            if not (0 <= next_cell[0] < self.width and 0 <= next_cell[1] < self.height):
                # Try alternative
                if dx != 0:
                    next_cell = (current[0], current[1] + 1)
                else:
                    next_cell = (current[0] + 1, current[1])
            
            if next_cell in self.rocks:
                print(f"  ✗ Hit rock at {next_cell} (step {len(path)})")
                return None
            
            if next_cell in visited:
                print(f"  ↻ Loop detected at {next_cell}")
                return None
            
            path.append(next_cell)
            visited.add(next_cell)
            current = next_cell
        
        if current == self.goal:
            print(f"  ✓ Reached goal in {len(path)-1} steps (lucky!)")
            return path
        else:
            print(f"  ✗ Failed to reach goal")
            return None

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 on a 20×20 maze with 64 rocks.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time
from typing import List, Tuple, Set

# Simulation parameters from fleet data
MAZE_SIZE = 20
NUM_ROCKS = 64
OPTIMAL_PATH_LENGTH = 21

# Symbols for visualization
ROCK = '█'
AGENT = 'A'
GOAL = 'G'
PATH = '·'
SAFE = ' '
UNKNOWN = '?'

class DeadbandMaze:
    """Implements the Deadband Protocol in a maze environment."""
    
    def __init__(self):
        self.grid = [[UNKNOWN for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.agent_pos = self.start
        
        # Generate rocks (P0: negative space)
        self._generate_rocks()
        
        # Initialize grid
        self._update_grid()
    
    def _generate_rocks(self):
        """P0: Map the negative space (rocks)."""
        while len(self.rocks) < NUM_ROCKS:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            pos = (x, y)
            
            # Don't place rocks on start or goal
            if pos != self.start and pos != self.goal:
                self.rocks.add(pos)
    
    def _update_grid(self):
        """Update visualization grid."""
        for x in range(MAZE_SIZE):
            for y in range(MAZE_SIZE):
                pos = (x, y)
                if pos in self.rocks:
                    self.grid[x][y] = ROCK
                elif pos == self.start:
                    self.grid[x][y] = AGENT
                elif pos == self.goal:
                    self.grid[x][y] = GOAL
                else:
                    self.grid[x][y] = SAFE
    
    def print_maze(self, title: str = ""):
        """Print current maze state."""
        print(f"\n{'='*60}")
        print(f"{title}")
        print(f"{'='*60}")
        
        for y in range(MAZE_SIZE):
            row = []
            for x in range(MAZE_SIZE):
                if (x, y) == self.agent_pos:
                    row.append(AGENT)
                else:
                    row.append(self.grid[x][y])
            print(' '.join(row))
        
        print(f"Agent: {self.agent_pos} | Goal: {self.goal}")
        print(f"Rocks: {len(self.rocks)} | Optimal path: {OPTIMAL_PATH_LENGTH} steps")
    
    def find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Identify safe channels (paths avoiding rocks)."""
        print("\n[P1] Identifying safe channels...")
        
        # Simple BFS to find all safe paths
        safe_paths = []
        visited = set()
        queue = [[self.start]]
        
        while queue and len(safe_paths) < 5:  # Limit to 5 paths for demo
            path = queue.pop(0)
            current = path[-1]
            
            if current == self.goal:
                safe_paths.append(path)
                continue
            
            if current in visited:
                continue
            
            visited.add(current)
            
            # Generate possible moves (up, down, left, right)
            x, y = current
            moves = [
                (x+1, y), (x-1, y),
                (x, y+1), (x, y-1)
            ]
            
            for move in moves:
                mx, my = move
                if (0 <= mx < MAZE_SIZE and 0 <= my < MAZE_SIZE and
                    move not in self.rocks and move not in visited):
                    new_path = list(path)
                    new_path.append(move)
                    queue.append(new_path)
        
        return safe_paths
    
    def optimize_within_channel(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """P2: Optimize within safe channels (shortest safe path)."""
        print("\n[P2] Optimizing within safe channels...")
        
        if not safe_paths:
            print("No safe channels found!")
            return []
        
        # Find shortest path among safe channels
        shortest_path = min(safe_paths, key=len)
        
        print(f"Found {len(safe_paths)} safe channels")
        print(f"Shortest safe path: {len(shortest_path)-1} steps")
        
        return shortest_path
    
    def greedy_optimization(self) -> List[Tuple[int, int]]:
        """Demonstrate P2-only (greedy) approach that hits rocks."""
        print("\n[GREEDY] P2-only optimization (will hit rocks)...")
        
        path = [self.start]
        current = self.start
        visited = set([current])
        
        while current != self.goal and len(path) < 100:  # Safety limit
            x, y = current
            
            # Greedy: move toward goal
            if x < self.goal[0]:
                next_pos = (x+1, y)
            elif y < self.goal[1]:
                next_pos = (x, y+1)
            else:
                # Already at goal row/col, move diagonally
                next_pos = (x+1, y+1) if x < self.goal[0] and y < self.goal[1] else self.goal
            
            # Check bounds
            if (0 <= next_pos[0] < MAZE_SIZE and 0 <= next_pos[1] < MAZE_SIZE):
                if next_pos in self.rocks:
                    print(f"  Hit rock at {next_pos}! Greedy optimization failed.")
                    return path  # Failed
                
                path.append(next_pos)
                current = next_pos
                visited.add(current)
            else:
                break
        
        return path
    
    def run_deadband_protocol(self):
        """Execute full P0→P1→P2 protocol."""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*60)
        
        # P0: Already done in initialization (rocks mapped)
        print("\n[P0] Negative space mapped:")
        print(f"  - {NUM_ROCKS} rocks identified")
        print(f"  - Agent knows where NOT to go")
        
        self.print_maze("Initial Maze (P0 Complete)")
        time.sleep(1)
        
        # P1: Find safe channels
        safe_paths = self.find_safe_channels()
        
        if not safe_paths:
            print("No safe path to goal exists!")
            return
        
        # P2: Optimize within safe channels
        optimal_path = self.optimize_within_channel(safe_paths)
        
        # Visualize the optimal path
        print("\n[RESULT] Deadband Protocol Success:")
        print(f"  - Safe path found: {len(optimal_path)-1} steps")
        print(f"  - Optimal from simulation: {OPTIMAL_PATH_LENGTH} steps")
        print(f"  - Within {abs((len(optimal_path)-1) - OPTIMAL_PATH_LENGTH)} steps of optimal")
        
        # Mark path on grid for visualization
        for pos in optimal_path[1:-1]:  # Skip start and goal
            x, y = pos
            self.grid[x][y] = PATH
        
        self.print_maze("Deadband Protocol Result (P0→P1→P2)")
        
        return optimal_path

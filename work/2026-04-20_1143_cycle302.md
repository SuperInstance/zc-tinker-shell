```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2 Visual Simulation
Tinker — Cycle 302
"""

import random
from collections import deque

def generate_maze(width=20, height=20, rock_density=0.15):
    """Generate a random maze with start, goal, and rocks."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = 'X'
    
    # Ensure start and goal are clear
    start = (0, 0)
    goal = (width-1, height-1)
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    
    return maze, start, goal

def print_maze(maze, title):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def p0_map_negative_space(maze):
    """P0: Identify rocks (negative space)."""
    rocks = []
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == 'X':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, rocks):
    """P1: Mark all non-rock cells as safe channels."""
    safe_maze = [row[:] for row in maze]
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if (x, y) in rocks:
                safe_maze[y][x] = 'X'
            elif safe_maze[y][x] not in ('S', 'G'):
                safe_maze[y][x] = ' '
    return safe_maze

def p2_optimize_within_channel(maze, start, goal, rocks):
    """P2: Find shortest safe path using BFS, constrained to safe cells."""
    width, height = len(maze[0]), len(maze)
    rock_set = set(rocks)
    
    # BFS
    queue = deque([start])
    parent = {start: None}
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        x, y = queue.popleft()
        if (x, y) == goal:
            break
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in rock_set and (nx, ny) not in parent:
                    parent[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    
    # Reconstruct path if found
    path_maze = [row[:] for row in maze]
    if goal in parent:
        # Mark path
        node = goal
        while node and node != start:
            x, y = node
            if path_maze[y][x] not in ('S', 'G'):
                path_maze[y][x] = '*'
            node = parent[node]
        return path_maze, True
    else:
        return path_maze, False

def main():
    print("Deadband Protocol Maze Simulation")
    print("P0 → P1 → P2 Strict Priority Demo")
    print("=" * 40)
    
    # Generate maze
    maze, start, goal = generate_maze(20, 20, rock_density=0.15)
    print_maze(maze, "Initial Maze (S=start, G=goal, X=rock)")
    
    # P0: Map negative space
    rocks = p0_map_negative_space(maze)
    print(f"P0: Mapped {len(rocks)} rocks (negative space)")
    
    # P1: Find safe channels
    safe_maze = p1_find_safe_channels(maze, rocks)
    print_maze(safe_maze, "P1: Safe Channels (only navigable space shown)")
    
    # P2: Optimize within channel
    path_maze, success = p2_optimize_within_channel(maze, start, goal, rocks)
    if success:
        print_maze(path_maze, "P2: Optimal Path Within Safe Channels (*=path)")
        print("SUCCESS: Found safe path from S to G following Deadband Protocol.")
    else:
        print_maze(path_maze, "P2: No Safe Path Found")
        print("FAILURE: No safe channel connects S to G.")
    
    print("\nProtocol Summary:")
    print("- P0: Identified what NOT to do (avoid rocks)")
    print("- P1: Mapped where we CAN be (safe channels)")
    print("- P2: Optimized path within safe channels only")
    print("\nViolation of P0→P1→P2 order leads to hitting rocks (sim data: 0/50 success).")

if __name__ == "__main__":
    main()
```

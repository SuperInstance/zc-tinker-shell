```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 strict priority in a 20x20 maze with 64 rocks.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque

# Constants from simulation data
MAZE_SIZE = 20
ROCK_COUNT = 64
START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)
OPTIMAL_STEPS = 21

# Symbols for visualization
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_SYM = 'G'
PATH = '○'
SAFE = ' '

class DeadbandMaze:
    def __init__(self):
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks = set()
        self.generate_rocks()
        self.mark_start_goal()
    
    def generate_rocks(self):
        """P0: Create negative space (rocks) - what NOT to step on."""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE - 1)
            y = random.randint(0, MAZE_SIZE - 1)
            if (x, y) != START and (x, y) != GOAL:
                self.rocks.add((x, y))
                self.grid[x][y] = ROCK
    
    def mark_start_goal(self):
        self.grid[START[0]][START[1]] = AGENT
        self.grid[GOAL[0]][GOAL[1]] = GOAL_SYM
    
    def print_maze(self, title="", highlights=None):
        """Print maze with optional highlighted cells."""
        if highlights is None:
            highlights = set()
        print(f"\n{'='*60}")
        print(f"{title}")
        print(f"{'='*60}")
        for x in range(MAZE_SIZE):
            row = []
            for y in range(MAZE_SIZE):
                if (x, y) in highlights:
                    row.append(PATH)
                else:
                    row.append(self.grid[x][y])
            print(' '.join(row))
    
    def is_valid(self, x, y):
        return 0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE
    
    def get_neighbors(self, x, y, avoid_rocks=True):
        """Get valid neighboring cells."""
        neighbors = []
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if self.is_valid(nx, ny):
                if avoid_rocks and (nx, ny) in self.rocks:
                    continue
                neighbors.append((nx, ny))
        return neighbors
    
    def phase0_map_negative_space(self):
        """P0: Identify all rocks (negative space)."""
        print("\n[P0: MAP NEGATIVE SPACE]")
        print(f"Identified {len(self.rocks)} rocks (obstacles)")
        print("These cells are ABSOLUTELY FORBIDDEN")
        return self.rocks
    
    def phase1_find_safe_channels(self):
        """P1: Find all paths from start to goal that avoid rocks."""
        print("\n[P1: FIND SAFE CHANNELS]")
        print("Performing BFS to find ALL safe paths...")
        
        # BFS to find all reachable cells and paths
        queue = deque([(START, [START])])
        visited = {START}
        all_paths = []
        reachable_cells = set([START])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == GOAL:
                all_paths.append(path)
                continue
            
            for nx, ny in self.get_neighbors(x, y, avoid_rocks=True):
                if (nx, ny) not in visited:
                    visited.add((nx, ny))
                    reachable_cells.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        safe_cells = reachable_cells - self.rocks
        print(f"Found {len(safe_cells)} safe cells (deadband)")
        print(f"Found {len(all_paths)} complete safe paths to goal")
        
        if not all_paths:
            print("WARNING: No safe path to goal! Agent trapped.")
        
        return safe_cells, all_paths
    
    def phase2_optimize_within_channel(self, safe_cells, all_paths):
        """P2: Find shortest path among safe paths."""
        print("\n[P2: OPTIMIZE WITHIN CHANNEL]")
        
        if not all_paths:
            print("No safe paths available. Cannot optimize.")
            return None
        
        # Find shortest safe path
        shortest_path = min(all_paths, key=len)
        steps = len(shortest_path) - 1  # Subtract start
        
        print(f"Shortest safe path: {steps} steps")
        print(f"Optimal (from simulation): {OPTIMAL_STEPS} steps")
        
        if steps == OPTIMAL_STEPS:
            print("✓ Achieved optimal path length!")
        else:
            print(f"Δ Difference from optimal: {steps - OPTIMAL_STEPS} steps")
        
        return shortest_path
    
    def greedy_p2_only(self):
        """Demonstrate violation: optimize (P2) without P0+P1."""
        print("\n[VIOLATION: P2-ONLY (GREEDY)]")
        print("Skipping P0 and P1, optimizing directly for goal...")
        
        # Greedy BFS that doesn't know about rocks
        queue = deque([(START, [START])])
        visited = {START}
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == GOAL:
                print(f"Found path of {len(path)-1} steps (but may hit rocks)")
                return path
            
            # Get ALL neighbors, including through rocks
            for nx, ny in self.get_neighbors(x, y, avoid_rocks=False):
                if (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        print("No path found (unlikely in open grid)")
        return None

def main():
    print("DEADBAND PROTOCOL MAZE DEMONSTRATION")
    print("Based on fleet simulation data")
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}")
    print(f"Start: {START}, Goal: {GOAL}, Optimal: {OPTIMAL_STEPS} steps")
    
    # Create maze
    maze = DeadbandMaze()
    maze.print_maze("Initial Maze (A=Agent, G=Goal, █=Rock)")
    
    # DEMONSTRATE VIOLATION FIRST (P2-only)
    greedy_path = maze.greedy_p2_only()
    if greedy_path:
        # Check if path hits rocks
        rock_hits = [cell for cell in greedy_path if cell in maze.rocks]
        if rock_hits:
            print(f"✗ Greedy path hits {len(rock_hits)} rocks!")
            print("This is why P0 (map rocks) must come first.")
        else:
            print("✓ Greedy got lucky (no rocks hit)")
    
    print("\n" + "="*60)
    print("NOW FOLLOWING DEADBAND PROTOCOL (P0→P1→P2)")
    print("="*60)
    
    # P0: Map negative space
    rocks = maze.phase0_map_negative_space()
    
    # P1: Find safe channels
    safe_cells, all_paths = maze.phase1_find_safe_channels()
    maze.print_maze

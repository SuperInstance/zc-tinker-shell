#!/usr/bin/env python3
"""
deadband_maze_demo.py

A visual demonstration of the Deadband Protocol (P0→P1→P2) in a 20×20 maze.
Based on fleet simulation data: optimal=21, rocks=64, deadband success=50/50.
"""

import random
from typing import List, Tuple, Set
from enum import Enum

class Cell(Enum):
    EMPTY = '·'
    ROCK = '█'
    AGENT = 'A'
    GOAL = 'G'
    PATH = '○'
    FAIL = 'X'

class AgentType(Enum):
    UNCONSTRAINED = "P2-only (greedy)"
    CONSTRAINED = "P1-only (safe channels, no rocks)"
    DEADBAND = "P0→P1→P2 (full protocol)"

class Maze:
    def __init__(self, width=20, height=20, rock_count=64, seed=42):
        self.width = width
        self.height = height
        self.grid = [[Cell.EMPTY for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks: Set[Tuple[int, int]] = set()
        
        random.seed(seed)
        # Place rocks randomly, avoiding start and goal
        possible = [(x, y) for x in range(width) for y in range(height)]
        possible.remove(self.start)
        possible.remove(self.goal)
        self.rocks = set(random.sample(possible, rock_count))
        for (x, y) in self.rocks:
            self.grid[y][x] = Cell.ROCK
        
        self.grid[self.start[1]][self.start[0]] = Cell.AGENT
        self.grid[self.goal[1]][self.goal[0]] = Cell.GOAL
    
    def is_rock(self, x: int, y: int) -> bool:
        return (x, y) in self.rocks
    
    def in_bounds(self, x: int, y: int) -> bool:
        return 0 <= x < self.width and 0 <= y < self.height
    
    def get_neighbors(self, x: int, y: int) -> List[Tuple[int, int]]:
        neighbors = []
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if self.in_bounds(nx, ny):
                neighbors.append((nx, ny))
        return neighbors
    
    def display(self, path: List[Tuple[int, int]] = None, failed_at: Tuple[int, int] = None):
        """Return ASCII representation of maze."""
        display_grid = [row[:] for row in self.grid]
        if path:
            for (x, y) in path[1:-1]:  # Exclude start and goal
                if display_grid[y][x] == Cell.EMPTY:
                    display_grid[y][x] = Cell.PATH
        if failed_at:
            fx, fy = failed_at
            if display_grid[fy][fx] == Cell.EMPTY:
                display_grid[fy][fx] = Cell.FAIL
        
        lines = []
        for y in range(self.height):
            line = ''.join([cell.value for cell in display_grid[y]])
            lines.append(line)
        return '\n'.join(lines)

def manhattan(a: Tuple[int, int], b: Tuple[int, int]) -> int:
    return abs(a[0] - b[0]) + abs(a[1] - b[1])

def unconstrained_agent(maze: Maze) -> Tuple[bool, List[Tuple[int, int]], Tuple[int, int]]:
    """P2-only: greedy toward goal, ignores rocks."""
    path = [maze.start]
    visited = set([maze.start])
    
    while path[-1] != maze.goal:
        x, y = path[-1]
        neighbors = maze.get_neighbors(x, y)
        # Sort by distance to goal (greedy)
        neighbors.sort(key=lambda pos: manhattan(pos, maze.goal))
        
        moved = False
        for nx, ny in neighbors:
            if (nx, ny) in visited:
                continue
            if maze.is_rock(nx, ny):
                # HIT A ROCK — failure
                return False, path, (nx, ny)
            path.append((nx, ny))
            visited.add((nx, ny))
            moved = True
            break
        
        if not moved:
            # No unvisited neighbors
            return False, path, path[-1]
    
    return True, path, None

def constrained_agent(maze: Maze) -> Tuple[bool, List[Tuple[int, int]], Tuple[int, int]]:
    """P1-only: knows safe channels (non‑rock cells) but not rock locations."""
    # This agent tries to avoid revisits but has no rock map
    path = [maze.start]
    visited = set([maze.start])
    
    while path[-1] != maze.goal:
        x, y = path[-1]
        neighbors = maze.get_neighbors(x, y)
        random.shuffle(neighbors)  # No optimization, just exploration
        
        moved = False
        for nx, ny in neighbors:
            if (nx, ny) in visited:
                continue
            if maze.is_rock(nx, ny):
                # Hits rock because it doesn't know P0
                return False, path, (nx, ny)
            path.append((nx, ny))
            visited.add((nx, ny))
            moved = True
            break
        
        if not moved:
            return False, path, path[-1]
    
    return True, path, None

def deadband_agent(maze: Maze) -> Tuple[bool, List[Tuple[int, int]], Tuple[int, int]]:
    """Full P0→P1→P2 protocol."""
    # P0: Map rocks (we have maze.rocks)
    rocks = maze.rocks
    
    # P1: Find safe channels — BFS that avoids rocks
    from collections import deque
    queue = deque([maze.start])
    came_from = {maze.start: None}
    
    while queue:
        current = queue.popleft()
        if current == maze.goal:
            break
        
        for neighbor in maze.get_neighbors(*current):
            if neighbor in came_from or neighbor in rocks:
                continue
            came_from[neighbor] = current
            queue.append(neighbor)
    
    if maze.goal not in came_from:
        return False, [], maze.start  # No safe path
    
    # Reconstruct safe path
    path = []
    current = maze.goal
    while current is not None:
        path.append(current)
        current = came_from[current]
    path.reverse()
    
    # P2: Optimize within safe channels (path is already shortest BFS path)
    return True, path, None

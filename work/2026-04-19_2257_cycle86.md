```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 86)
A visual ASCII simulation of P0→P1→P2 in a 10x10 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 10, height: int = 10, rock_density: float = 0.2):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Generate rocks (negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        for y in range(height):
            for x in range(width):
                if (x, y) not in [self.start, self.goal] and random.random() < rock_density:
                    self.rocks.add((x, y))
        
        # For visualization
        self.agent_path = []
        self.safe_channels = set()
    
    def print_maze(self, show_channels: bool = False, show_path: bool = False):
        """Render ASCII maze with rocks, agent, goal, and optional overlays."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')  # Rock
                elif show_channels and pos in self.safe_channels:
                    row.append('·')  # Safe channel
                elif show_path and pos in self.agent_path:
                    row.append('*')  # Agent path
                else:
                    row.append('.')  # Open water
            print(' '.join(row))
        print()
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify all rocks (states where failure probability > threshold)."""
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Found {len(self.rocks)} rocks (failure states)")
        self.print_maze()
        return self.rocks.copy()
    
    def p1_find_safe_channels(self) -> Set[Tuple[int, int]]:
        """P1: Find all paths from start to goal that avoid rocks."""
        print("=== P1: FIND SAFE CHANNELS ===")
        
        # Simple BFS to find reachable positions avoiding rocks
        visited = set()
        queue = [self.start]
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Explore neighbors (up, down, left, right)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        self.safe_channels = visited
        print(f"Safe channels contain {len(self.safe_channels)} reachable positions")
        self.print_maze(show_channels=True)
        return self.safe_channels
    
    def p2_optimize_within_channel(self) -> List[Tuple[int, int]]:
        """P2: Find shortest path through safe channels (constrained optimization)."""
        print("=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        # BFS for shortest path constrained to safe_channels
        queue = [(self.start, [self.start])]
        visited = set()
        
        while queue:
            (x, y), path = queue.pop(0)
            if (x, y) == self.goal:
                self.agent_path = path
                print(f"Found optimal path length: {len(path)-1} steps")
                self.print_maze(show_path=True)
                return path
            
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) in self.safe_channels and (nx, ny) not in visited:
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        print("ERROR: No path found within safe channels")
        return []
    
    def run_unconstrained_greedy(self) -> bool:
        """Simulate P2-only (greedy) agent that ignores P0/P1."""
        print("\n=== UNCONSTRAINED GREEDY (P2-ONLY) ===")
        path = [self.start]
        x, y = self.start
        
        # Greedy move toward goal, ignoring rocks
        steps = 0
        while (x, y) != self.goal and steps < 50:
            # Move toward goal
            if x < self.goal[0]:
                x += 1
            elif x > self.goal[0]:
                x -= 1
            elif y < self.goal[1]:
                y += 1
            elif y > self.goal[1]:
                y -= 1
            
            pos = (x, y)
            path.append(pos)
            
            # Check for rock hit
            if pos in self.rocks:
                print(f"FAIL: Hit rock at {pos} after {steps} steps")
                self.agent_path = path
                self.print_maze(show_path=True)
                return False
            
            steps += 1
        
        success = (x, y) == self.goal
        print(f"{'SUCCESS' if success else 'FAIL'}: Reached goal in {steps} steps")
        self.agent_path = path
        self.print_maze(show_path=True)
        return success

def main():
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("=" * 40)
    
    # Seed for reproducibility (matches fleet simulation approach)
    random.seed(42)
    
    # Create maze environment
    maze = DeadbandMaze(width=10, height=10, rock_density=0.2)
    
    # Run Deadband Protocol (P0→P1→P2)
    print("\n[DEADBAND PROTOCOL EXECUTION]")
    rocks = maze.p0_map_negative_space()
    safe_channels = maze.p1_find_safe_channels()
    optimal_path = maze.p2_optimize_within_channel()
    
    success = len(optimal_path) > 0 and optimal_path[-1] == maze.goal
    print(f"\nDeadband Protocol Result: {'SUCCESS' if success else 'FAIL'}")
    
    # Run unconstrained greedy for comparison
    print("\n" + "=" * 40)
    maze_unconstrained = DeadbandMaze(width=10, height=10, rock_density=0.2)
    greedy_success = maze_unconstrained.run_unconstrained_greedy()
    
    # Summary
    print("\n" + "=" * 40)
    print("SUMMARY (matches fleet constraint theory):")
    print(f"  Deadband Protocol (P0→P1→P2): {'Success' if success else 'Failure'}")
    print(f"  Unconstrained Greedy (P2-only): {'Success' if greedy_success else 'Failure'}")
    print("\nNote: In complex environments, unconstrained optimization")
    print("      hits rocks with probability → 1.0 (per DEADBAND-PROTOCOL.md)")

if __name__ == "__main__":
    main()

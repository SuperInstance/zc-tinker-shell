import random

def generate_maze(width=20, height=20, rock_density=0.16):
    """Generate a random maze with start, goal, and obstacles."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks (obstacles)
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
    
    # Ensure start and goal are in open cells
    start = (random.randint(0, width-1), random.randint(0, height-1))
    while maze[start[1]][start[0]] == '#':
        start = (random.randint(0, width-1), random.randint(0, height-1))
    maze[start[1]][start[0]] = 'S'
    
    goal = (random.randint(0, width-1), random.randint(0, height-1))
    while maze[goal[1]][goal[0]] == '#' or goal == start:
        goal = (random.randint(0, width-1), random.randint(0, height-1))
    maze[goal[1]][goal[0]] = 'G'
    
    return maze, start, goal

def p0_map_rocks(maze):
    """P0: Identify all obstacle cells (negative space)."""
    rocks = []
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channel(maze, start):
    """P1: Flood-fill from start, avoiding rocks, to find reachable safe cells."""
    width, height = len(maze[0]), len(maze)
    visited = [[False for _ in range(width)] for _ in range(height)]
    safe_cells = []
    
    # Directions: up, down, left, right
    directions = [(0, -1), (0, 1), (-1, 0), (1, 0)]
    
    queue = deque([start])
    sx, sy = start
    visited[sy][sx] = True
    
    while queue:
        x, y = queue.popleft()
        safe_cells.append((x, y))
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if not visited[ny][nx] and maze[ny][nx] != '#':
                    visited[ny][nx] = True
                    queue.append((nx, ny))
    
    return safe_cells, visited

def p2_optimize_in_channel(maze, start, goal, safe_visited):
    """P2: A* pathfinding constrained to the safe channel from P1."""
    width, height = len(maze[0]), len(maze)
    
    # Heuristic: Manhattan distance
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    # Check if goal is in safe channel
    gx, gy = goal
    if not safe_visited[gy][gx]:
        return None  # Goal unreachable via safe channel
    
    # A* algorithm
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    directions = [(0, -1), (0, 1), (-1, 0), (1, 0)]
    
    while open_set:
        current = heapq.heappop(open_set)[1]
        
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            path.reverse()
            return path
        
        for dx, dy in directions:
            neighbor = (current[0] + dx, current[1] + dy)
            nx, ny = neighbor
            
            # Constraint: must be within bounds AND in safe channel
            if not (0 <= nx < width and 0 <= ny < height):
                continue
            if not safe_visited[ny][nx]:
                continue
            
            tentative_g = g_score[current] + 1
            
            if neighbor not in g_score or tentative_g < g_score[neighbor]:
                came_from[neighbor] = current
                g_score[neighbor] = tentative_g
                f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return None  # No path found within safe channel

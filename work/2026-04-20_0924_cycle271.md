"""
Deadband Protocol Maze Simulator
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Demonstrates P0→P1→P2 priority in pathfinding.
"""

import random
from typing import List, Tuple, Set
from enum import Enum

class Cell(Enum):
    EMPTY = '·'
    ROCK = '█'
    AGENT = 'A'
    GOAL = 'G'
    PATH = '○'
    SAFE = '░'

class Strategy(Enum):
    UNCONSTRAINED = "P2-only (greedy)"
    CONSTRAINED = "P1-only (no optimization)"
    DEADBAND = "P0→P1→P2 (full protocol)"

    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.grid = [[Cell.EMPTY for _ in range(width)] for _ in range(height)]
        
        # Place rocks (P0 negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density:
                    self.grid[y][x] = Cell.ROCK
                    self.rocks.add((x, y))
        
        # Start at top-left, goal at bottom-right
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Ensure start and goal are not rocks
        if self.start in self.rocks:
            self.rocks.remove(self.start)
            self.grid[self.start[1]][self.start[0]] = Cell.EMPTY
        if self.goal in self.rocks:
            self.rocks.remove(self.goal)
            self.grid[self.goal[1]][self.goal[0]] = Cell.EMPTY
        
        # Track visited cells for visualization
        self.visited = set()
        self.safe_channel = set()
        self.final_path = []
    
    def print_maze(self, show_safe: bool = False, show_path: bool = False):
        """Display the maze with current agent state."""
        for y in range(self.height):
            row = []

                    row.append(Cell.AGENT.value)
                elif (x, y) == self.goal:
                    row.append(Cell.GOAL.value)
                elif show_path and (x, y) in self.final_path:
                    row.append(Cell.PATH.value)
                elif show_safe and (x, y) in self.safe_channel:
                    row.append(Cell.SAFE.value)
                else:
                    row.append(cell.value)
            print(''.join(row))
        print()

        return self.rocks.copy()
    
    def p1_find_safe_channels(self, rocks: Set[Tuple[int, int]]) -> Set[Tuple[int, int]]:
        """P1: Find all cells reachable without hitting rocks."""
        print("P1: Finding safe channels (deadbands)")
        
        safe = set()
        stack = [self.start]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in safe or (x, y) in rocks:
                continue
            
            safe.add((x, y))
            
            # Check neighbors (4-directional movement)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in rocks:
                        stack.append((nx, ny))
        
        self.safe_channel = safe
        print(f"Safe channel size: {len(safe)} cells ({len(safe)*100/(self.width*self.height):.1f}% of maze)")
        
        # Check if goal is reachable
        if self.goal not in safe:
            print("WARNING: Goal not reachable through safe channels!")
        
        return safe
    
    def p2_optimize_in_channel(self, safe: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:

                return path

                continue
            visited.add((x, y))
            
            # Explore neighbors in safe channel
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) in safe and (nx, ny) not in visited:
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        return []

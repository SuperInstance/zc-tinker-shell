#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
ASCII visualization of the protocol in action.
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.15):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.start = (1, 1)
        self.goal = (width-2, height-2)
        self.rocks = set()
        self.safe_channels = []
        self.optimal_path = []
        
        # P0: Generate rocks (negative space)
        self._generate_rocks(rock_density)
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
        
    def _generate_rocks(self, density):
        """P0: Map the negative space (rocks)."""
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) in (self.start, self.goal):
                    continue
                if random.random() < density:
                    self.rocks.add((x, y))
                    self.grid[y][x] = '#'
    
    def find_safe_channels(self):
        """P1: Identify safe channels (deadbands) using BFS flood fill."""
        visited = set()
        channels = []
        
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) in visited or (x, y) in self.rocks:
                    continue
                if self.grid[y][x] in ('S', 'G', '.'):
                    # Flood fill to find connected safe region
                    channel = set()
                    queue = deque([(x, y)])
                    while queue:
                        cx, cy = queue.popleft()
                        if (cx, cy) in visited:
                            continue
                        visited.add((cx, cy))
                        if (cx, cy) in self.rocks:
                            continue
                        channel.add((cx, cy))
                        # 4-directional neighbors
                        for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
                            nx, ny = cx + dx, cy + dy
                            if 0 <= nx < self.width and 0 <= ny < self.height:
                                if (nx, ny) not in visited and (nx, ny) not in self.rocks:
                                    queue.append((nx, ny))
                    if channel:
                        channels.append(channel)
        
        # Keep only channels containing both start and goal
        self.safe_channels = [c for c in channels 
                             if self.start in c and self.goal in c]
        return len(self.safe_channels) > 0
    
    def optimize_path(self):
        """P2: Find shortest path within safe channels (A*)."""
        if not self.safe_channels:
            return False
        
        # Combine all safe cells
        safe_cells = set()
        for channel in self.safe_channels:
            safe_cells.update(channel)
        
        # A* search
        open_set = {self.start}
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: self._heuristic(self.start, self.goal)}
        
        while open_set:
            current = min(open_set, key=lambda p: f_score.get(p, float('inf')))
            if current == self.goal:
                self.optimal_path = self._reconstruct_path(came_from, current)
                return True
            
            open_set.remove(current)
            for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                if neighbor not in safe_cells:
                    continue
                tentative_g = g_score[current] + 1
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor, self.goal)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        return False
    
    def _heuristic(self, a, b):
        """Manhattan distance heuristic."""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def _reconstruct_path(self, came_from, current):
        """Reconstruct path from came_from dictionary."""
        path = [current]
        while current in came_from:
            current = came_from[current]
            path.append(current)
        return list(reversed(path))
    
    def visualize(self, show_path=False, show_channels=False):
        """Render ASCII maze with optional overlays."""
        display = [row[:] for row in self.grid]
        
        if show_channels and self.safe_channels:
            for channel in self.safe_channels:
                for x, y in channel:
                    if display[y][x] == '.':
                        display[y][x] = '~'  # Safe channel marker
        
        if show_path and self.optimal_path:
            for x, y in self.optimal_path[1:-1]:  # Skip start and goal
                if display[y][x] in ('.', '~'):
                    display[y][x] = '*'  # Path marker
        
        # Build border
        border = '+' + '-' * self.width + '+'
        lines = [border]
        for row in display:
            lines.append('|' + ''.join(row) + '|')
        lines.append(border)
        
        return '\n'.join(lines)
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2."""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===\n")
        
        # P0: Already done in initialization (rocks mapped)
        print("P0: MAP NEGATIVE SPACE")
        print(f"    Identified {len(self.rocks)} rocks (failure states)")
        print(self.visualize())
        print()
        
        # P1: Find safe channels
        print("P1: IDENTIFY SAFE CHANNELS")
        has_channels = self.find_safe_channels()
        if has_channels:
            print(f"    Found {len(self.safe_channels)} safe channel(s)")
            safe_cell_count = sum(len(c) for c in self.safe_channels)
            print(f"    {safe_cell_count} safe cells ({safe_cell_count/(self.width*self.height):.1%} of maze)")
            print(self.visualize(show_channels=True))
        else:
            print("    NO SAFE CHANNEL CONNECTING START TO GOAL")
            print("    Protocol stops here — cannot proceed to P2")
            return False
        print()
        
        # P2: Optimize within channels
        print("P2: OPTIMIZE WITHIN CHANNEL")
        if self.optimize_path():
            print(f"    Found optimal path: {len(self.optimal_path)} steps")
            print(self.visualize(show_path=True, show_channels=True))
            print(f"\n    Path length: {len(self.optimal_path)-1} moves")
            print(f"    Theoretical optimal (no rocks): {self._heuristic(self.start, self.goal)} moves")
        else:
            print("    Failed to find path within safe channels")
            return False
        
        return True

def main():
    """Run demonstration with fixed seed for reproducibility."""
    random.seed(42)  # Deterministic for demo
    maze = DeadbandMaze(width=20, height=20, rock_density=0.15)
    
    success = maze.run_protocol()
    
    print("\n" + "="*50)
    if success:
        print("PROTOCOL SUCCESS: Reached goal via safe channels")
    else:
        print("PROTOCOL FAILED: No viable path respecting P0+P1 constraints")
    
    print("\nKey:")
    print("  S = Start, G = Goal, # = Rock (P0 negative space)")
    print("  ~ = Safe channel (P1 deadband), * = Optimal path (P2)")

if __name__ == "__main__":
    main()
```

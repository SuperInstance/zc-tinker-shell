import random

from typing import List, Tuple, Set

class Maze:
    """20x20 maze with rocks, start, and goal."""
    def __init__(self, seed=42):
        random.seed(seed)
        self.width = 20
        self.height = 20
        self.start = (0, 0)
        self.goal = (19, 19)
        self.rocks: Set[Tuple[int, int]] = set()
        
        # Generate rocks (64 rocks ~16% density, matching sim)
        for _ in range(64):
            x = random.randint(0, self.width-1)
            y = random.randint(0, self.height-1)
            if (x, y) not in {self.start, self.goal}:
                self.rocks.add((x, y))
    
    def is_rock(self, x: int, y: int) -> bool:
        return (x, y) in self.rocks
    
    def in_bounds(self, x: int, y: int) -> bool:
        return 0 <= x < self.width and 0 <= y < self.height
    
    def render(self, path=None, current=None):
        """Render maze with optional path and current position."""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif (x, y) == current:
                    row.append('@')
                elif path and (x, y) in path:
                    row.append('·')
                elif self.is_rock(x, y):
                    row.append('█')
                else:
                    row.append(' ')
            grid.append(''.join(row))
        return '\n'.join(grid)

class Agent:
    """Base agent with movement logic."""
    def __init__(self, name, maze):
        self.name = name
        self.maze = maze
        self.x, self.y = maze.start
        self.path = [(self.x, self.y)]
        self.steps = 0
        self.failed = False
    
    def move_toward(self, tx: int, ty: int):
        """Move one step toward target (Manhattan)."""
        if self.x < tx:
            self.x += 1
        elif self.x > tx:
            self.x -= 1
        elif self.y < ty:
            self.y += 1
        elif self.y > ty:
            self.y -= 1
        self.path.append((self.x, self.y))
        self.steps += 1
        
        # Check for failure
        if self.maze.is_rock(self.x, self.y):
            self.failed = True
            return False
        return True
    
    def run(self):
        """Run until goal or failure."""
        while (self.x, self.y) != self.maze.goal and not self.failed:
            if not self.move_toward(*self.maze.goal):
                break
        return not self.failed

class ConstrainedAgent(Agent):
    """Naive constrained agent: avoids rocks if visible."""
    def move_toward(self, tx: int, ty: int):
        # Look one step ahead
        candidates = []
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = self.x + dx, self.y + dy
            if self.maze.in_bounds(nx, ny):
                if not self.maze.is_rock(nx, ny):
                    candidates.append((nx, ny))
        
        if candidates:
            # Pick closest to goal
            best = min(candidates, key=lambda p: abs(p[0]-tx) + abs(p[1]-ty))
            self.x, self.y = best
            self.path.append((self.x, self.y))
            self.steps += 1
            return True
        else:
            self.failed = True
            return False

class DeadbandAgent(Agent):
    """P0→P1→P2 agent: maps rocks, finds safe channels, then optimizes."""
    def __init__(self, maze):
        super().__init__("Deadband (P0→P1→P2)", maze)
        self.safe_channels = None
    
    def p0_map_rocks(self):
        """P0: Map negative space (rocks)."""
        # In real scenario, this would be exploration/sensing
        # Here we use full knowledge for demonstration
        return self.maze.rocks
    
    def p1_find_channels(self, rocks):
        """P1: Find safe channels using flood fill."""
        visited = set()
        channels = []
        
        def flood(x, y, channel):
            if (x, y) in visited or (x, y) in rocks:
                return
            visited.add((x, y))
            channel.add((x, y))
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if self.maze.in_bounds(nx, ny):
                    flood(nx, ny, channel)
        
        for y in range(self.maze.height):
            for x in range(self.maze.width):
                if (x, y) not in visited and (x, y) not in rocks:
                    channel = set()
                    flood(x, y, channel)
                    if channel:
                        channels.append(channel)
        
        # Find channel containing start and goal
        start_goal_channel = None
        for channel in channels:
            if self.maze.start in channel and self.maze.goal in channel:
                start_goal_channel = channel
                break
        
        return start_goal_channel
    
    def p2_optimize_path(self, safe_channel):
        """P2: Find shortest path within safe channel (BFS)."""
        from collections import deque
        
        queue = deque([(self.maze.start, [self.maze.start])])
        visited = set([self.maze.start])
        
        while queue:
            (x, y), path = queue.popleft()
            if (x, y) == self.maze.goal:
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (nx, ny) in safe_channel and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        return None
    
    def run(self):
        """Execute full Deadband Protocol."""
        # P0: Map rocks
        rocks = self.p0_map_rocks()
        
        # P1: Find safe channel
        safe_channel = self.p1_find_channels(rocks)
        if not safe_channel:
            self.failed = True
            return False
        
        # P2: Optimize path within channel
        optimal_path = self.p2_optimize_path(safe_channel)
        if not optimal_path:
            self.failed = True
            return False
        
        # Follow the optimal path
        for x, y in optimal_path[1:]:  # Skip start
            self.x, self.y = x, y
            self.path.append((x, y))
            self.steps += 1
        
        return True

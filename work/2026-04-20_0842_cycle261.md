#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 protocol vs P2-only optimization.

Based on fleet-knowledge/narrow-games/constraint_sim_final.json:
- Optimal path length: 21
- Rocks: 64 obstacles
- Unconstrained success: 33% (hits rocks)
- Deadband success: 50% (finds safe channel)
"""

import random
from typing import List, Tuple, Set
from enum import Enum

class Cell(Enum):
    EMPTY = '·'
    ROCK = '█'
    AGENT = 'A'
    GOAL = 'G'
    PATH = '○'
    SAFE = '░'  # Safe channel cells

class DeadbandProtocol:
    """Implements the Deadband Protocol for maze navigation."""
    
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.grid = [[Cell.EMPTY for _ in range(width)] for _ in range(height)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (P0: negative space)
        self._generate_rocks(rock_density)
        
        # Place start and goal
        self.grid[self.start[1]][self.start[0]] = Cell.AGENT
        self.grid[self.goal[1]][self.goal[0]] = Cell.GOAL
    
    def _generate_rocks(self, density: float):
        """P0: Map the negative space (rocks to avoid)."""
        num_rocks = int(self.width * self.height * density)
        self.rocks = set()
        
        # Avoid start and goal positions
        forbidden = {self.start, self.goal}
        
        while len(self.rocks) < num_rocks:
            x = random.randint(0, self.width-1)
            y = random.randint(0, self.height-1)
            pos = (x, y)
            
            if pos not in forbidden and pos not in self.rocks:
                self.rocks.add(pos)
                self.grid[y][x] = Cell.ROCK
    
    def _is_safe(self, x: int, y: int) -> bool:
        """Check if cell is not a rock."""
        return (x, y) not in self.rocks
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Explicitly return the rocks (negative space)."""
        return self.rocks.copy()
    
    def p1_find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Find all safe paths from start to goal using BFS."""
        from collections import deque
        
        # BFS to find all shortest safe paths
        queue = deque([(self.start, [self.start])])
        visited = {self.start}
        safe_paths = []
        shortest_length = None
        
        while queue:
            (x, y), path = queue.popleft()
            
            # If we found the goal
            if (x, y) == self.goal:
                if shortest_length is None:
                    shortest_length = len(path)
                if len(path) == shortest_length:
                    safe_paths.append(path)
                continue
            
            # Explore neighbors (4-directional movement)
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited and self._is_safe(nx, ny):
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        return safe_paths
    
    def p2_optimize_in_channel(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """P2: Select optimal path from safe channels."""
        if not safe_paths:
            return []
        
        # Choose shortest path (could add other optimization criteria)
        return min(safe_paths, key=len)
    
    def deadband_protocol(self) -> Tuple[List[Tuple[int, int]], List[List[Tuple[int, int]]]]:
        """Full P0→P1→P2 protocol execution."""
        # P0: Map negative space (already done in init)
        rocks = self.p0_map_negative_space()
        print(f"P0: Mapped {len(rocks)} rocks (negative space)")
        
        # P1: Find safe channels
        safe_paths = self.p1_find_safe_channels()
        print(f"P1: Found {len(safe_paths)} safe channels")
        
        # P2: Optimize within safe channels
        if safe_paths:
            optimal_path = self.p2_optimize_in_channel(safe_paths)
            print(f"P2: Selected optimal path of length {len(optimal_path)}")
            return optimal_path, safe_paths
        else:
            print("P2: No safe channels found!")
            return [], safe_paths
    
    def p2_only_greedy(self) -> List[Tuple[int, int]]:
        """P2-only optimization (violates protocol)."""
        # Greedy approach: always move toward goal, ignoring rocks
        path = [self.start]
        x, y = self.start
        
        while (x, y) != self.goal:
            # Move toward goal (Manhattan direction)
            if x < self.goal[0]:
                x += 1
            elif x > self.goal[0]:
                x -= 1
            elif y < self.goal[1]:
                y += 1
            elif y > self.goal[1]:
                y -= 1
            
            path.append((x, y))
            
            # Check if hit rock
            if not self._is_safe(x, y):
                print(f"P2-only: Hit rock at ({x}, {y})!")
                return path  # Failed path
        
        return path
    
    def visualize(self, path: List[Tuple[int, int]] = None, show_safe_channels: bool = False):
        """Create ASCII visualization of maze."""
        # Create display grid
        display = [[self.grid[y][x].value for x in range(self.width)] for y in range(self.height)]
        
        # Mark safe channels if requested
        if show_safe_channels:
            safe_paths = self.p1_find_safe_channels()
            all_safe_cells = set()
            for safe_path in safe_paths:
                for (x, y) in safe_path:
                    if (x, y) != self.start and (x, y) != self.goal:
                        all_safe_cells.add((x, y))
            
            for (x, y) in all_safe_cells:
                if display[y][x] == Cell.EMPTY.value:
                    display[y][x] = Cell.SAFE.value
        
        # Mark path if provided
        if path:
            for (x, y) in path[1:-1]:  # Skip start and goal
                if display[y][x] == Cell.EMPTY.value or display[y][x] == Cell.SAFE.value:
                    display[y][x] = Cell.PATH.value
        
        # Print grid
        print("+" + "-" * self.width + "+")
        for row in display:
            print("|" + "".join(row) + "|")
        print("+" + "-" * self.width + "+")
        
        # Legend
        print("\nLegend:")
        print(f"  {Cell.AGENT.value} = Agent (start)")
        print(f"  {Cell.GOAL.value} = Goal")
        print(f"  {Cell.ROCK.value} = Rock (avoid!)")
        print(f"  {Cell.PATH.value} = Agent path")
        print(f"  {Cell.SAFE.value} = Safe channel")
        print(f"  {Cell.EMPTY.value} = Empty space")

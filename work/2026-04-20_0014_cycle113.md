#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Simulates the constraint theory proof from fleet-knowledge.
"""

import random
import time
from typing import List, Tuple, Set

# Constants from the proof
MAZE_SIZE = 20
ROCK_COUNT = 64  # ~16% density, matches simulation
GOAL_POS = (19, 19)
START_POS = (0, 0)

# Symbols for visualization
AGENT = '🤖'
ROCK = '█'
GOAL = '🎯'
PATH = '·'
SAFE = '░'
UNKNOWN = ' '

    """Implements the Deadband Protocol in a grid world."""
    
    def __init__(self):
        self.grid = [[UNKNOWN for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: Set[Tuple[int, int]] = set()
        self.optimal_path: List[Tuple[int, int]] = []
        self.agent_pos = START_POS
        
        # Initialize maze
        self._place_rocks()
        self._place_goal()
        self._update_view()
    
    def _place_rocks(self):
        """P0: Rocks are the negative space (failure states)."""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != START_POS and (x, y) != GOAL_POS:
                self.rocks.add((x, y))
                self.grid[x][y] = ROCK
    
    def _place_goal(self):
        self.grid[GOAL_POS[0]][GOAL_POS[1]] = GOAL
    
    def _update_view(self):
        """Update grid with current agent position."""
        x, y = self.agent_pos
        if self.grid[x][y] not in [ROCK, GOAL]:
            self.grid[x][y] = AGENT
    
    def display(self, title: str):
        """Print the maze with a title."""
        print(f"\n{'='*60}")
        print(f"{title}")
        print(f"{'='*60}")
        for row in self.grid:
            print(' '.join(row))
        print()
        time.sleep(0.5)

        """Step P0: Identify all rocks (what NOT to do)."""
        print("P0: MAPPING NEGATIVE SPACE")
        print(f"Identified {len(self.rocks)} rocks (failure states)")
        self.display("Negative Space Map")
        return list(self.rocks)
    
    def p1_find_safe_channels(self):
        """Step P1: Find all paths from start to goal avoiding rocks."""
        print("P1: FINDING SAFE CHANNELS")
        
        # Simple BFS to find reachable safe cells
        visited = set()
        queue = [START_POS]
        safe_cells = set()
        
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # If not a rock, it's potentially safe
            if (x, y) not in self.rocks:
                safe_cells.add((x, y))
                # Add neighbors
                for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                    nx, ny = x + dx, y + dy
                    if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                        if (nx, ny) not in visited:
                            queue.append((nx, ny))
        
        self.safe_channels = safe_cells
        
        # Visualize safe channels
        viz_grid = [[UNKNOWN for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        for x in range(MAZE_SIZE):
            for y in range(MAZE_SIZE):
                if (x, y) in self.rocks:
                    viz_grid[x][y] = ROCK
                elif (x, y) in self.safe_channels:
                    viz_grid[x][y] = SAFE
                elif (x, y) == GOAL_POS:
                    viz_grid[x][y] = GOAL
        
        print(f"Found {len(self.safe_channels)} safe cells (deadband)")
        print("\nSafe Channels Map:")
        for row in viz_grid:
            print(' '.join(row))
        print()
        time.sleep(1)
        
        return safe_cells
    
    def p2_optimize_within_channel(self):
        """Step P2: Find shortest path through safe channels."""
        print("P2: OPTIMIZING WITHIN SAFE CHANNELS")
        
        # A* search constrained to safe channels
        open_set = {START_POS}
        came_from = {}
        g_score = {START_POS: 0}
        f_score = {START_POS: self._heuristic(START_POS)}
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score.get(pos, float('inf')))
            
            if current == GOAL_POS:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.reverse()
                self.optimal_path = path
                break
            
            open_set.remove(current)
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # Constraint: must be in safe channels
                if neighbor not in self.safe_channels:
                    continue
                
                tentative_g = g_score[current] + 1
                
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        if self.optimal_path:
            print(f"Found optimal path of length {len(self.optimal_path)} steps")
            
            # Visualize path
            viz_grid = [[UNKNOWN for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
            for x in range(MAZE_SIZE):
                for y in range(MAZE_SIZE):
                    if (x, y) in self.rocks:
                        viz_grid[x][y] = ROCK
                    elif (x, y) == GOAL_POS:
                        viz_grid[x][y] = GOAL
                    elif (x, y) in self.optimal_path:
                        viz_grid[x][y] = PATH
            
            print("\nOptimal Path (P2):")
            for row in viz_grid:
                print(' '.join(row))
            
            # Animate the agent following the path
            print("\nAnimating agent movement...")
            for step, pos in enumerate([START_POS] + self.optimal_path):
                self.agent_pos = pos
                self._update_view()
                self.display(f"Step {step}: Agent moving through safe channel")
                time.sleep(0.2)
            
            return len(self.optimal_path)
        else:
            print("ERROR: No path found within safe channels!")
            return None

        """Manhattan distance heuristic."""
        return abs(pos[0] - GOAL_POS[0]) + abs(pos[1] - GOAL_POS[1])

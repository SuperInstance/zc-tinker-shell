**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Implementing a visual simulation based on `constraint_sim_final.json` and `DEADBAND-PROTOCOL.md`. The script will generate a random 20x20 maze, map rocks (P0), find safe channels (P1), then compute an optimal path (P2) within those channels.

**Output:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 protocol with visual ASCII maze.
Based on constraint_sim_final.json and DEADBAND-PROTOCOL.md.
"""

import random
import sys
from collections import deque

# Constants from simulation data
MAZE_SIZE = 20
ROCK_COUNT = 64  # Approximate from 64 rocks in 20x20 grid
START = (0, 0)
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

# Symbols for visualization
ROCK = '█'
SAFE = '·'
AGENT = 'A'
GOAL_MARKER = 'G'
PATH = '○'
WALL = '▓'
EMPTY = ' '

class DeadbandMaze:
    def __init__(self, seed=None):
        if seed is not None:
            random.seed(seed)
        
        # P0: Map negative space (rocks)
        self.rocks = set()
        self.generate_rocks()
        
        # P1: Safe channels (non-rock cells)
        self.safe_cells = self.map_safe_channels()
        
        # P2: Optimal path (initialized later)
        self.optimal_path = []
        
        # Visualization grid
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.update_grid()
    
    def generate_rocks(self):
        """P0: Generate random obstacles (negative space)"""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            # Don't place rocks on start or goal
            if (x, y) != START and (x, y) != GOAL:
                self.rocks.add((x, y))
    
    def map_safe_channels(self):
        """P1: Identify all safe cells (not rocks)"""
        safe = set()
        for x in range(MAZE_SIZE):
            for y in range(MAZE_SIZE):
                if (x, y) not in self.rocks:
                    safe.add((x, y))
        return safe
    
    def find_optimal_path(self):
        """P2: Find shortest path through safe channels using BFS"""
        if START not in self.safe_cells or GOAL not in self.safe_cells:
            return []
        
        queue = deque([(START, [START])])
        visited = set([START])
        
        # 4-direction movement
        directions = [(1, 0), (-1, 0), (0, 1), (0, -1)]
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == GOAL:
                return path
            
            for dx, dy in directions:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) in self.safe_cells and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        return []  # No path found
    
    def update_grid(self):
        """Update visualization grid with current state"""
        # Clear grid
        for x in range(MAZE_SIZE):
            for y in range(MAZE_SIZE):
                self.grid[x][y] = EMPTY
        
        # Mark rocks (P0 negative space)
        for x, y in self.rocks:
            self.grid[x][y] = ROCK
        
        # Mark safe cells (P1 channels)
        for x, y in self.safe_cells:
            if self.grid[x][y] == EMPTY:
                self.grid[x][y] = SAFE
        
        # Mark start and goal
        sx, sy = START
        gx, gy = GOAL
        self.grid[sx][sy] = AGENT
        self.grid[gx][gy] = GOAL_MARKER
        
        # Mark optimal path if found
        for x, y in self.optimal_path[1:-1]:  # Exclude start and goal
            self.grid[x][y] = PATH
    
    def print_maze(self, title="", show_stats=True):
        """Print ASCII visualization of maze"""
        print(f"\n{'='*60}")
        print(f"{title:^60}")
        print(f"{'='*60}")
        
        # Print column numbers
        print("   " + "".join(f"{i%10}" for i in range(MAZE_SIZE)))
        
        for y in range(MAZE_SIZE):
            # Print row number
            print(f"{y:2} ", end="")
            for x in range(MAZE_SIZE):
                print(self.grid[x][y], end="")
            print()
        
        if show_stats:
            print(f"\nStatistics:")
            print(f"  Maze Size: {MAZE_SIZE}x{MAZE_SIZE}")
            print(f"  Rocks (P0): {len(self.rocks)}")
            print(f"  Safe Cells (P1): {len(self.safe_cells)}")
            print(f"  Optimal Path Length (P2): {len(self.optimal_path)-1 if self.optimal_path else 'No path'}")
            
            # Protocol compliance check
            if self.optimal_path:
                path_in_safe = all(pos in self.safe_cells for pos in self.optimal_path)
                print(f"  Protocol Compliance: {'✓ P0→P1→P2 followed' if path_in_safe else '✗ Violation detected'}")
    
    def run_protocol_demo(self):
        """Execute full P0→P1→P2 demonstration"""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL DEMONSTRATION - Tinker (Cocapn Fleet)")
        print("="*60)
        
        # Step P0: Show negative space
        self.print_maze("P0: MAP NEGATIVE SPACE (Rocks)", show_stats=False)
        print(f"\nP0 Complete: Identified {len(self.rocks)} rocks to avoid")
        
        # Step P1: Show safe channels
        self.update_grid()
        self.print_maze("P1: IDENTIFY SAFE CHANNELS (Deadbands)", show_stats=False)
        print(f"\nP1 Complete: Mapped {len(self.safe_cells)} safe cells")
        
        # Step P2: Find and show optimal path
        self.optimal_path = self.find_optimal_path()
        self.update_grid()
        
        if self.optimal_path:
            self.print_maze("P2: OPTIMIZE WITHIN CHANNEL (Optimal Path)")
            print(f"\nP2 Complete: Found optimal path of length {len(self.optimal_path)-1}")
        else:
            self.print_maze("P2: OPTIMIZE WITHIN CHANNEL (No Path Found)")
            print(f"\nP2 Complete: No safe path exists to goal")
        
        # Show comparison with unconstrained approach
        print(f"\n{'='*60}")
        print("PROTOCOL EFFECTIVENESS COMPARISON")
        print(f"{'='*60}")
        print("Based on constraint_sim_final.json:")
        print(f"  Unconstrained (P2-only): {33}% success, avg cost {2320.9}")
        print(f"  Deadband (P0→P1→P2): {50}% success, avg cost {21.0}")
        print(f"  This demo: {'Success' if self.optimal_path else 'Failure'}")

def main():
    """Main demonstration function"""
    # Run with fixed seed for reproducibility
    maze = DeadbandMaze(seed=42)
    maze.run_protocol_demo()
    
    # Interactive option
    print(f"\n{'='*60}")
    response = input("Run another random simulation? (y/n): ")
    if response.lower() == 'y':
        maze2 = DeadbandMaze()  # Random seed
        maze2.run_protocol_demo()

if __name__ == "__main__":
    main()
```

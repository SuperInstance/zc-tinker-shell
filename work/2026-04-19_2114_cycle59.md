#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 strict priority in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # From "rocks": 64
OPTIMAL_PATH_LENGTH = 21  # From "optimal": 21

class DeadbandAgent:
    """Implements the Deadband Protocol in a maze environment."""
    
    def __init__(self, maze_size=MAZE_SIZE):
        self.maze_size = maze_size
        self.rocks = set()
        self.start = (0, 0)
        self.goal = (maze_size-1, maze_size-1)
        self.safe_channels = []
        self.optimal_path = []
        
    def generate_maze(self):
        """Create maze with rocks (obstacles)."""
        # Generate random rock positions (negative space)
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, self.maze_size-1)
            y = random.randint(0, self.maze_size-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
        
        print(f"Generated maze with {len(self.rocks)} rocks (P0 negative space)")
        return self.rocks
    
    def p0_map_negative_space(self):
        """P0: Identify what NOT to do (rocks/obstacles)."""
        print("\n=== P0: MAP NEGATIVE SPACE ===")
        print(f"Identified {len(self.rocks)} rock positions")
        print("Agent knows where NOT to go")
        return list(self.rocks)
    
    def p1_find_safe_channels(self):
        """P1: Find all paths that avoid rocks."""
        print("\n=== P1: FIND SAFE CHANNELS ===")
        
        # Simple BFS to find all safe paths
        from collections import deque
        
        safe_paths = []
        queue = deque([([self.start], set([self.start]))])
        
        while queue:
            path, visited = queue.popleft()
            x, y = path[-1]
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            # Explore neighbors (right, down, left, up)
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.maze_size and 0 <= ny < self.maze_size:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        new_path = path + [(nx, ny)]
                        new_visited = visited.copy()
                        new_visited.add((nx, ny))
                        queue.append((new_path, new_visited))
        
        self.safe_channels = safe_paths
        print(f"Found {len(safe_paths)} safe channels (rock-free paths)")
        return safe_paths
    
    def p2_optimize_within_channel(self):
        """P2: Find shortest path among safe channels."""
        print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        if not self.safe_channels:
            print("ERROR: No safe channels found. Must complete P1 first.")
            return []
        
        # Find shortest safe path
        shortest = min(self.safe_channels, key=len)
        self.optimal_path = shortest
        
        print(f"Optimal path length: {len(shortest)} steps")
        print(f"Target optimal: {OPTIMAL_PATH_LENGTH} steps")
        
        return shortest
    
    def run_protocol(self):
        """Execute full P0→P1→P2 sequence."""
        print("=" * 50)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("=" * 50)
        
        # Generate environment
        self.generate_maze()
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        
        # P1: Find safe channels
        safe_paths = self.p1_find_safe_channels()
        
        # P2: Optimize within channel
        optimal = self.p2_optimize_within_channel()
        
        return rocks, safe_paths, optimal
    
    def visualize_maze(self, path=None):
        """Create ASCII visualization of maze."""
        print("\n" + "=" * 50)
        print("MAZE VISUALIZATION")
        print("=" * 50)
        
        for y in range(self.maze_size):
            row = []
            for x in range(self.maze_size):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')
                elif path and pos in path:
                    row.append('·')
                else:
                    row.append('.')
            print(' '.join(row))
        
        if path:
            print(f"\nOptimal path ({len(path)} steps): {path[:5]}...{path[-5:] if len(path) > 10 else ''}")

def run_simulation():

    agent = DeadbandAgent()

    return success

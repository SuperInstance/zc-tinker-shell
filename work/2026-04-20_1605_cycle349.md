#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

A visual ASCII simulation of the Deadband Protocol (P0→P1→P2) in a 10x10 maze.
Demonstrates strict priority: map rocks (P0), find safe channels (P1), then optimize (P2).

Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set

# ====================
# MAZE CONFIGURATION
# ====================
MAZE_WIDTH = 10
MAZE_HEIGHT = 10
START = (0, 0)
GOAL = (9, 9)

# Rock density ~30% (matching simulation spirit)
ROCK_COUNT = 30

# Movement directions (4‑connected grid)
DIRECTIONS = [(0, 1), (1, 0), (0, -1), (-1, 0)]

# ====================
# DEADBAND PROTOCOL
# ====================
class DeadbandAgent:
    def __init__(self, maze: List[List[str]]):
        self.maze = maze
        self.rocks: Set[Tuple[int, int]] = set()      # P0: negative space
        self.safe: Set[Tuple[int, int]] = set()       # P1: safe channels
        self.path: List[Tuple[int, int]] = []         # P2: optimal path

    # --- P0: MAP THE NEGATIVE SPACE ---
    def map_rocks(self):
        """Identify all cells that are rocks ('#'). Do NOT proceed without this."""
        for y in range(MAZE_HEIGHT):
            for x in range(MAZE_WIDTH):
                if self.maze[y][x] == '#':
                    self.rocks.add((x, y))
        print(f"[P0] Mapped {len(self.rocks)} rocks")

    # --- P1: IDENTIFY SAFE CHANNELS ---
    def find_safe_channels(self):
        """Flood‑fill from start, avoiding rocks. Only safe cells are reachable."""
        stack = [START]
        visited = set()
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            # Cell is safe if it's not a rock
            if self.maze[y][x] != '#':
                self.safe.add((x, y))
                # Explore neighbors
                for dx, dy in DIRECTIONS:
                    nx, ny = x + dx, y + dy
                    if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                        stack.append((nx, ny))
        print(f"[P1] Found {len(self.safe)} safe cells")

    # --- P2: OPTIMIZE WITHIN CHANNEL ---
    def optimize_path(self):
        """A* search constrained to safe cells only."""
        from heapq import heappush, heappop

        # Heuristic: Manhattan distance
        def h(pos):
            return abs(pos[0] - GOAL[0]) + abs(pos[1] - GOAL[1])

        open_set = [(h(START), 0, START, None)]  # (f, g, pos, parent)
        came_from = {}
        g_score = {START: 0}

        while open_set:
            f, g, (x, y), parent = heappop(open_set)

            if (x, y) == GOAL:
                # Reconstruct path
                self.path = []
                while (x, y) != START:
                    self.path.append((x, y))
                    x, y = came_from[(x, y)]
                self.path.append(START)
                self.path.reverse()
                print(f"[P2] Optimal path length: {len(self.path)-1} steps")
                return True

            for dx, dy in DIRECTIONS:
                nx, ny = x + dx, y + dy
                if not (0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT):
                    continue
                if (nx, ny) not in self.safe:   # CONSTRAINT: stay in safe channel
                    continue
                new_g = g + 1
                if (nx, ny) not in g_score or new_g < g_score[(nx, ny)]:
                    g_score[(nx, ny)] = new_g
                    f_score = new_g + h((nx, ny))
                    heappush(open_set, (f_score, new_g, (nx, ny), (x, y)))
                    came_from[(nx, ny)] = (x, y)

        print("[P2] No safe path to goal")
        return False

    def run_protocol(self) -> bool:
        """Execute strict P0→P1→P2 sequence."""
        print("\n=== DEADBAND PROTOCOL EXECUTION ===")
        self.map_rocks()            # P0
        self.find_safe_channels()   # P1
        return self.optimize_path() # P2

# ====================
# VISUALIZATION
# ====================
def print_maze(maze, path=None, rocks=None, safe=None):
    """Print maze with optional overlays."""
    for y in range(MAZE_HEIGHT):
        line = []
        for x in range(MAZE_WIDTH):
            cell = maze[y][x]
            if path and (x, y) in path:
                if (x, y) == START:
                    line.append('S')
                elif (x, y) == GOAL:
                    line.append('G')
                else:
                    line.append('·')
            elif rocks and (x, y) in rocks:
                line.append('#')
            elif safe and (x, y) in safe:
                line.append(' ')
            else:
                line.append(cell)
        print(''.join(line))

def generate_maze() -> List[List[str]]:
    """Create a random maze with rocks, start, and goal."""
    maze = [[' ' for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    # Place rocks
    for _ in range(ROCK_COUNT):
        x = random.randint(0, MAZE_WIDTH-1)
        y = random.randint(0, MAZE_HEIGHT-1)
        if (x, y) != START and (x, y) != GOAL:
            maze[y][x] = '#'
    # Mark start and goal
    maze[START[1]][START[0]] = 'S'
    maze[GOAL[1]][GOAL[0]] = 'G'
    return maze

# ====================
# DEMONSTRATION
# ====================
def main():
    print("Deadband Protocol Maze Demo")
    print("=" * 40)
    print("Legend: S=start, G=goal, #=rock, ·=path, space=safe channel")
    print()

    # Generate and show initial maze
    maze = generate_maze()
    print("Initial Maze (P0 not yet executed):")
    print_maze(maze)
    print()

    # Run agent with strict protocol
    agent = DeadbandAgent(maze)
    success = agent.run_protocol()

    print()
    if success:
        print("FINAL VISUALIZATION:")
        print("Rocks (#) → Safe channels (space) → Optimal path (·)")
        print_maze(maze, path=set(agent.path), rocks=agent.rocks, safe=agent.safe)
        print(f"\nResult: SUCCESS — Goal reached in {len(agent.path)-1} steps")
    else:
        print("Result: FAILURE — No safe path exists")

    print()
    print("Protocol Summary:")
    print("  P0: Mapped rocks → avoid negative space")
    print("  P1: Flood‑fill safe channels → constrain search")
    print("  P2: A* within safe channels → optimal path")
    print("\nViolating P0→P1→P2 (e.g., direct A* without rock map)")
    print("would hit rocks with high probability (see constraint_sim_final.json).")

if __name__ == "__main__":
    random.seed(42)  # Reproducible for training
    main()

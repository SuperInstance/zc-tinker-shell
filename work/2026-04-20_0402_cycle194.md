import random
from typing import List, Tuple, Set

class DeadbandMaze:
    """Maze environment with rocks, start, goal."""
    
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (P0 negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        total_cells = width * height
        num_rocks = int(total_cells * rock_density)
        
        # Avoid start and goal
        possible = [(x, y) for x in range(width) for y in range(height)]
        possible.remove(self.start)
        possible.remove(self.goal)
        
        self.rocks = set(random.sample(possible, num_rocks))
        
        # For visualization
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        for (x, y) in self.rocks:
            self.grid[y][x] = '█'  # Rock
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def print_maze(self, path=None):
        """Print ASCII maze with optional path."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if path and (x, y) in path:
                    row.append('*')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
        print()

class DeadbandAgent:
    """Agent that follows P0→P1→P2 protocol."""
    
    def __init__(self, maze: DeadbandMaze):
        self.maze = maze
        self.visited = set()
        self.safe_channels = []  # P1 result
        self.optimal_path = []   # P2 result
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify rocks (what NOT to do)."""
        print("P0: Mapping negative space (rocks)...")
        print(f"  Found {len(self.maze.rocks)} rocks.")
        return self.maze.rocks.copy()
    
    def p1_find_safe_channels(self, rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
        """P1: Find all paths that avoid rocks (safe channels)."""
        print("P1: Finding safe channels (deadbands)...")
        
        # Simple BFS to find all safe paths
        start = self.maze.start
        goal = self.maze.goal
        
        queue = [(start, [start])]
        safe_paths = []
        
        while queue:
            (x, y), path = queue.pop(0)
            
            # Check if reached goal
            if (x, y) == goal:
                safe_paths.append(path)
                continue
            
            # Generate next moves (right, down, left, up)
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.maze.width and 0 <= ny < self.maze.height:
                    if (nx, ny) not in rocks and (nx, ny) not in path:
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        print(f"  Found {len(safe_paths)} safe channels.")
        return safe_paths
    
    def p2_optimize_within_channel(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """P2: Find shortest path within safe channels."""
        print("P2: Optimizing within safe channels...")
        
        if not safe_paths:
            print("  ERROR: No safe channels found! Cannot optimize.")
            return []
        
        # Find shortest path
        shortest = min(safe_paths, key=len)
        print(f"  Selected optimal path length: {len(shortest)} steps")
        return shortest
    
    def run_protocol(self) -> bool:
        """Execute full P0→P1→P2 protocol."""
        print("\n" + "="*50)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*50)
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        
        # P1: Find safe channels
        safe_paths = self.p1_find_safe_channels(rocks)
        self.safe_channels = safe_paths
        
        # P2: Optimize within channel
        if safe_paths:
            optimal_path = self.p2_optimize_within_channel(safe_paths)
            self.optimal_path = optimal_path
            
            # Show results
            print("\nRESULTS:")
            print(f"  Success: {'YES' if optimal_path else 'NO'}")
            print(f"  Path length: {len(optimal_path)}")
            print(f"  Optimal (theoretical): 21 steps")
            
            # Visualize
            print("\nMAZE WITH OPTIMAL PATH:")
            self.maze.print_maze(set(optimal_path))
            
            return True
        else:
            print("\nFAILURE: No safe channels found (all paths hit rocks)")
            return False

            print(f"  Path length: {len(path)}")
            return True
        else:
            print("  Failed to reach goal")

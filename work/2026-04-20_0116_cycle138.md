Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 protocol with visual ASCII maze
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import random
from typing import List, Tuple, Set
from collections import deque

class Maze:
    def __init__(self, width: int = 10, height: int = 10, rock_density: float = 0.2):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        
        # Place rocks (P0 negative space)
        rocks = 0
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density and not (x == 0 and y == 0) and not (x == width-1 and y == height-1):
                    self.grid[y][x] = '#'
                    rocks += 1
        
        # Start and goal
        self.grid[0][0] = 'S'
        self.grid[height-1][width-1] = 'G'
        
        self.rocks = rocks
    
    def display(self, path: List[Tuple[int, int]] = None, title: str = ""):
        """Display maze with optional path"""
        print(f"\n{'='*40}")
        print(f"{title}")
        print(f"{'='*40}")
        
        display_grid = [row[:] for row in self.grid]
        
        if path:
            for x, y in path[1:-1]:  # Don't overwrite S and G
                if display_grid[y][x] not in ['S', 'G']:
                    display_grid[y][x] = '*'
        
        for row in display_grid:
            print(' '.join(row))
        print()

class Agent:
    def __init__(self, maze: Maze):
        self.maze = maze
        self.start = (0, 0)
        self.goal = (maze.width-1, maze.height-1)
    
    def unconstrained_greedy(self) -> Tuple[List[Tuple[int, int]], bool]:
        """P2-only: Greedy optimization without rock awareness"""
        path = [self.start]
        visited = set([self.start])
        current = self.start
        
        while current != self.goal:
            x, y = current
            
            # Generate possible moves (right, down, left, up)
            moves = [(x+1, y), (x, y+1), (x-1, y), (x, y-1)]
            valid_moves = []
            
            for nx, ny in moves:
                if 0 <= nx < self.maze.width and 0 <= ny < self.maze.height:
                    # P2-only: Doesn't check for rocks!
                    if (nx, ny) not in visited:
                        valid_moves.append((nx, ny))
            
            if not valid_moves:
                return path, False  # Stuck
            
            # Greedy: choose move closest to goal
            best_move = min(valid_moves, 
                          key=lambda pos: abs(pos[0]-self.goal[0]) + abs(pos[1]-self.goal[1]))
            
            # Check if we hit a rock (failure state)
            if self.maze.grid[best_move[1]][best_move[0]] == '#':
                return path, False
            
            path.append(best_move)
            visited.add(best_move)
            current = best_move
        
        return path, True
    
    def constrained_aware(self) -> Tuple[List[Tuple[int, int]], bool]:
        """P0-only: Knows rocks but doesn't find safe channels"""
        # This agent knows where rocks are but uses naive BFS
        # without considering channel optimization
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            current, path = queue.popleft()
            
            if current == self.goal:
                return path, True
            
            x, y = current
            moves = [(x+1, y), (x, y+1), (x-1, y), (x, y-1)]
            
            for nx, ny in moves:
                if 0 <= nx < self.maze.width and 0 <= ny < self.maze.height:
                    if (nx, ny) not in visited:
                        # P0: Avoid rocks
                        if self.maze.grid[ny][nx] != '#':
                            visited.add((nx, ny))
                            queue.append(((nx, ny), path + [(nx, ny)]))
        
        return [self.start], False  # No path found
    
    def deadband_protocol(self) -> Tuple[List[Tuple[int, int]], bool]:
        """Full P0→P1→P2: Map rocks, find safe channels, optimize within"""
        # P0: Map negative space (rocks)
        rocks = set()
        for y in range(self.maze.height):
            for x in range(self.maze.width):
                if self.maze.grid[y][x] == '#':
                    rocks.add((x, y))
        
        # P1: Find safe channels (BFS avoiding rocks)
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        all_safe_paths = []
        
        while queue:
            current, path = queue.popleft()
            
            if current == self.goal:
                all_safe_paths.append(path)
                continue  # Keep looking for all safe paths
            
            x, y = current
            moves = [(x+1, y), (x, y+1), (x-1, y), (x, y-1)]
            
            for nx, ny in moves:
                if 0 <= nx < self.maze.width and 0 <= ny < self.maze.height:
                    if (nx, ny) not in visited:
                        # Stay in safe channel (avoid rocks)
                        if (nx, ny) not in rocks:
                            visited.add((nx, ny))
                            queue.append(((nx, ny), path + [(nx, ny)]))
        
        if not all_safe_paths:
            return [self.start], False
        
        # P2: Optimize within safe channels
        shortest_path = min(all_safe_paths, key=len)
        return shortest_path, True

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 priority in a 20x20 maze.
"""

import random
import time
from typing import List, Tuple, Set

# Constants
MAZE_SIZE = 20
START = (0, 0)
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)
ROCK_DENSITY = 0.15  # 15% of cells are rocks

def generate_maze() -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """Generate random maze with rocks."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = set()
    
    # Place rocks randomly
    for r in range(MAZE_SIZE):
        for c in range(MAZE_SIZE):
            if (r, c) == START or (r, c) == GOAL:
                continue
            if random.random() < ROCK_DENSITY:
                maze[r][c] = '#'
                rocks.add((r, c))
    
    maze[START[0]][START[1]] = 'S'
    maze[GOAL[0]][GOAL[1]] = 'G'
    return maze, rocks

def print_maze(maze: List[List[str]], agent_pos: Tuple[int, int], step: int, phase: str):
    """Print maze with agent position and phase info."""
    print(f"\n=== Step {step} — {phase} ===")
    for r in range(MAZE_SIZE):
        row = []
        for c in range(MAZE_SIZE):
            if (r, c) == agent_pos:
                row.append('A')
            else:
                row.append(maze[r][c])
        print(' '.join(row))
    time.sleep(0.3)

def get_neighbors(pos: Tuple[int, int]) -> List[Tuple[int, int]]:
    """Get valid orthogonal neighbors."""
    r, c = pos
    neighbors = []
    for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
        nr, nc = r + dr, c + dc
        if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
            neighbors.append((nr, nc))
    return neighbors

def deadband_protocol_demo():
    """Main demo of P0→P1→P2."""
    maze, rocks = generate_maze()
    agent_pos = START
    step = 0
    
    print("Initial Maze (S=Start, G=Goal, #=Rock, .=Empty, A=Agent)")
    print_maze(maze, agent_pos, step, "INIT")
    
    visited = {START}
    safe_channels = set()  # P1 result
    
    # P0: Map negative space (rocks)
    print("\n--- P0: MAP NEGATIVE SPACE ---")
    print(f"Identified {len(rocks)} rock cells (avoid these)")
    step += 1
    print_maze(maze, agent_pos, step, "P0 Complete")
    
    # P1: Identify safe channels
    print("\n--- P1: IDENTIFY SAFE CHANNELS ---")
    # Simple BFS to find all reachable non-rock cells
    queue = [START]
    while queue:
        current = queue.pop(0)
        for neighbor in get_neighbors(current):
            if neighbor not in rocks and neighbor not in safe_channels:
                safe_channels.add(neighbor)
                queue.append(neighbor)
    
    print(f"Mapped {len(safe_channels)} safe cells (deadband)")
    step += 1
    print_maze(maze, agent_pos, step, "P1 Complete")
    
    # P2: Optimize within safe channels (BFS to goal)
    print("\n--- P2: OPTIMIZE WITHIN SAFE CHANNELS ---")
    parent = {START: None}
    queue = [START]
    goal_reached = False
    
    while queue and not goal_reached:
        current = queue.pop(0)
        
        if current == GOAL:
            goal_reached = True
            break
            
        for neighbor in get_neighbors(current):
            if neighbor in safe_channels and neighbor not in parent:
                parent[neighbor] = current
                queue.append(neighbor)
                # Visualize exploration
                agent_pos = neighbor
                step += 1
                print_maze(maze, agent_pos, step, "P2 Exploring")
    
    # Reconstruct path if goal reached
    if goal_reached:
        path = []
        node = GOAL
        while node is not None:
            path.append(node)
            node = parent[node]
        path.reverse()
        
        print("\n=== FINAL PATH (P2 Optimization) ===")
        for i, pos in enumerate(path):
            agent_pos = pos
            step += 1
            print_maze(maze, agent_pos, step, f"Path step {i}")
        
        print(f"\nSUCCESS: Reached goal in {len(path)-1} moves")
        print(f"Protocol followed: P0→P1→P2 (strict priority)")
        return True
    else:
        print("\nFAILURE: No safe path to goal exists")
        return False

if __name__ == "__main__":
    random.seed(42)  # For reproducibility
    success = deadband_protocol_demo()
    
    print("\n" + "="*50)
    print("DEADBAND PROTOCOL DEMO COMPLETE")
    print(f"Result: {'SUCCESS' if success else 'FAILURE (no safe path)'}")
    print("Key insight: P2 optimization only works within P1 safe channels.")
    print("Violation (skipping P0/P1) leads to rock collisions.")
    print("="*50)
```

**Key Features:**
1. **P0 Visualization**: Shows identified rock cells (`#`) to avoid.
2. **P1 Visualization**: Maps all reachable safe cells (deadband).
3. **P2 Visualization**: Shows BFS exploration constrained to safe channels only.
4. **Strict Priority**: The agent cannot proceed to P2 until P0 and P1 are complete.
5. **ASCII Art**: Clear visual representation of each protocol phase.

**Learning Points:**
- The simulation demonstrates why skipping P0/P1 leads to failure (hitting rocks).
- Shows that P2 optimization (shortest path) is only meaningful within safe channels.
- Visualizes the deadband concept: the space where the agent *can* operate safely.

#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — Tinker (Cycle 307)
Simulates P0→P1→P2 priority in a 20x20 maze.
Matches fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
import sys
from typing import List, Tuple, Set

# Constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # ~16% density
OPTIMAL_PATH_LEN = 21  # Known optimal path length

class Maze:
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self._generate()
    
    def _generate(self):
        """Place rocks randomly, ensuring start/goal are clear."""
        while len(self.rocks) < ROCK_COUNT:
            r = random.randint(0, MAZE_SIZE-1)
            c = random.randint(0, MAZE_SIZE-1)
            if (r, c) != self.start and (r, c) != self.goal:
                self.rocks.add((r, c))
                self.grid[r][c] = '#'
    
    def print(self, path=None, rocks_highlighted=False):
        """Print ASCII maze with optional path."""
        for r in range(MAZE_SIZE):
            row = []
            for c in range(MAZE_SIZE):
                pos = (r, c)
                if path and pos in path:
                    row.append('*')
                elif rocks_highlighted and pos in self.rocks:
                    row.append('#')
                elif pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                else:
                    row.append(self.grid[r][c])
            print(' '.join(row))
        print()

class Agent:
    def __init__(self, maze: Maze):
        self.maze = maze
        self.visited = set()
    
    def unconstrained_p2(self) -> List[Tuple[int, int]]:
        """P2-only: Greedy Manhattan distance, ignores rocks."""
        path = [self.maze.start]
        current = self.maze.start
        
        while current != self.maze.goal:
            r, c = current
            # Greedy move toward goal
            if r < self.maze.goal[0]:
                next_pos = (r+1, c)
            elif c < self.maze.goal[1]:
                next_pos = (r, c+1)
            else:
                break
            
            if next_pos in self.maze.rocks:
                # Hits rock → failure
                return []
            
            path.append(next_pos)
            current = next_pos
        
        return path if current == self.maze.goal else []
    
    def constrained_p0_p1(self) -> List[Tuple[int, int]]:
        """P0+P1: Map rocks, find any safe channel via BFS."""
        # P0: Rocks are already known (self.maze.rocks)
        # P1: BFS avoiding rocks
        queue = [(self.maze.start, [self.maze.start])]
        visited = set([self.maze.start])
        
        while queue:
            (r, c), path = queue.pop(0)
            if (r, c) == self.maze.goal:
                return path
            
            for dr, dc in [(1,0), (0,1), (-1,0), (0,-1)]:
                nr, nc = r+dr, c+dc
                if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                    if (nr, nc) not in visited and (nr, nc) not in self.maze.rocks:
                        visited.add((nr, nc))
                        queue.append(((nr, nc), path + [(nr, nc)]))
        
        return []  # No safe channel
    
    def deadband_p0_p1_p2(self) -> List[Tuple[int, int]]:
        """Full protocol: P0 (rocks), P1 (safe channel via BFS), P2 (optimize within channel)."""
        # P1: Find safe channel (any path avoiding rocks)
        safe_path = self.constrained_p0_p1()
        if not safe_path:
            return []
        
        # P2: Optimize within safe channel (shortest path in safe graph)
        # Build graph of safe positions
        safe_positions = set(safe_path)
        # Run BFS on safe positions only
        queue = [(self.maze.start, [self.maze.start])]
        visited = set([self.maze.start])
        
        while queue:
            (r, c), path = queue.pop(0)
            if (r, c) == self.maze.goal:
                return path
            
            for dr, dc in [(1,0), (0,1), (-1,0), (0,-1)]:
                nr, nc = r+dr, c+dc
                pos = (nr, nc)
                if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                    if pos not in visited and pos in safe_positions:
                        visited.add(pos)
                        queue.append((pos, path + [pos]))
        
        return safe_path  # Fallback

def run_simulation(iterations: int = 50):
    """Run multiple iterations matching fleet data."""
    results = {
        'unconstrained_success': 0,
        'constrained_success': 0,
        'deadband_success': 0,
        'unconstrained_avg_len': 0,
        'deadband_avg_len': 0
    }
    
    for i in range(iterations):
        maze = Maze()
        agent = Agent(maze)
        
        # Strategy 1: Unconstrained (P2-only)
        p2_path = agent.unconstrained_p2()
        if p2_path:
            results['unconstrained_success'] += 1
            results['unconstrained_avg_len'] += len(p2_path)
        
        # Strategy 2: Constrained (P0+P1) — just find any safe path
        p0p1_path = agent.constrained_p0_p1()
        if p0p1_path:
            results['constrained_success'] += 1
        
        # Strategy 3: Deadband (P0→P1→P2)
        deadband_path = agent.deadband_p0_p1_p2()
        if deadband_path:
            results['deadband_success'] += 1
            results['deadband_avg_len'] += len(deadband_path)
    
    # Calculate averages
    if results['unconstrained_success'] > 0:
        results['unconstrained_avg_len'] /= results['unconstrained_success']
    if results['deadband_success'] > 0:
        results['deadband_avg_len'] /= results['deadband_success']
    
    return results

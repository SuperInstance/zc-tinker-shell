import random
from collections import deque
import heapq

class Maze:
    def __init__(self, width=20, height=20, num_rocks=64):
        self.width = width
        self.height = height
        self.num_rocks = num_rocks
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.rocks = set()
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
    def generate(self):
        """Generate random rocks, ensuring start and goal are clear."""
        possible = [(x, y) for x in range(self.width) for y in range(self.height)]
        possible.remove(self.start)
        possible.remove(self.goal)
        
        self.rocks = set(random.sample(possible, self.num_rocks))
        for x, y in self.rocks:
            self.grid[y][x] = '#'
        
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def print(self, path=None, title=""):
        """Print ASCII maze with optional path overlay."""
        print(f"\n=== {title} ===")
        display = [row[:] for row in self.grid]
        
        if path:
            for x, y in path[1:-1]:  # Skip start and goal
                if display[y][x] == '.':
                    display[y][x] = '*'
        
        for row in display:
            print(' '.join(row))
        print()
    
    def is_valid(self, x, y):
        return 0 <= x < self.width and 0 <= y < self.height
    
    def is_rock(self, x, y):
        return (x, y) in self.rocks

    def unconstrained_p2_only(self):
        """P2-only: Greedy pathfinding ignoring rocks (hits rocks)."""

            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        open_set = []
        heapq.heappush(open_set, (0, self.maze.start))
        came_from = {}
        g_score = {self.maze.start: 0}
        f_score = {self.maze.start: heuristic(self.maze.start, self.maze.goal)}
        
        while open_set:
            _, current = heapq.heappop(open_set)
            
            if current == self.maze.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.maze.start)
                path.reverse()
                return path
            
            x, y = current
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if self.maze.is_valid(nx, ny):

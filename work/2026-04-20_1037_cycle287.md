"""
DeadbandRoom Preset Demo - A visual demonstration of the Deadband Protocol API.
Simulates a training room where agents must follow P0→P1→P2 strictly.
"""

import random
from collections import deque

class DeadbandRoom:
    """A simple implementation of a Deadband Protocol training room."""
    
    def __init__(self, width=10, height=10, rock_density=0.3):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self._generate_rocks(rock_density)
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
        
        # Protocol state
        self.p0_done = False
        self.p1_done = False
        self.p2_done = False
        self.negative_space = None
        self.safe_channels = None
        self.optimal_path = None
        
    def _generate_rocks(self, density):
        """P0: Generate rocks (negative space)."""
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) in (self.start, self.goal):
                    continue
                if random.random() < density:
                    self.rocks.add((x, y))
                    self.grid[y][x] = '#'
    
    def identify_negative_space(self):
        """P0: Map all rocks/failure states."""
        self.negative_space = self.rocks.copy()
        self.p0_done = True
        return self.negative_space
    
    def find_safe_channels(self):
        """P1: Find all paths from start to goal avoiding rocks."""
        if not self.p0_done:
            raise RuntimeError("Must complete P0 (identify negative space) first!")
        
        # BFS to find all safe paths
        queue = deque([(self.start, [self.start])])
        safe_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in path:
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.safe_channels = safe_paths
        self.p1_done = True
        return safe_paths
    
    def optimize_within_channels(self):
        """P2: Select shortest path from safe channels."""
        if not self.p1_done:
            raise RuntimeError("Must complete P1 (find safe channels) first!")
        
        if not self.safe_channels:
            self.optimal_path = None
        else:
            self.optimal_path = min(self.safe_channels, key=len)
        
        self.p2_done = True
        return self.optimal_path
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2."""
        print("=== Deadband Protocol Demo ===")
        print("P0: Identifying negative space (rocks)...")
        rocks = self.identify_negative_space()
        print(f"   Found {len(rocks)} rocks")
        
        print("P1: Finding safe channels...")
        channels = self.find_safe_channels()
        print(f"   Found {len(channels)} safe channels")
        
        print("P2: Optimizing within channels...")
        optimal = self.optimize_within_channels()
        if optimal:
            print(f"   Optimal path length: {len(optimal)-1} steps")
        else:
            print("   No safe path to goal!")
        
        return optimal
    
    def visualize(self, path=None):
        """ASCII visualization of the room."""
        viz_grid = [row[:] for row in self.grid]
        
        if path:
            for x, y in path[1:-1]:  # Exclude start and goal
                viz_grid[y][x] = '○'
        
        print("\n" + "\n".join(" ".join(row) for row in viz_grid))
        print(f"\nLegend: S=Start, G=Goal, #=Rock, ○=Path (if shown)")

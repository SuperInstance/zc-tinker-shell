#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 with ASCII visualization.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self.rocks: Set[Tuple[int, int]] = set()
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density and (x, y) not in [self.start, self.goal]:
                    self.rocks.add((x, y))
        
        # Agent state
        self.position = self.start
        self.path = [self.start]
        self.steps = 0
        
    def print_maze(self, show_path: bool = True):
        """ASCII visualization of maze state"""
        print("\n" + "="*40)
        print("MAZE STATE")
        print(f"Position: {self.position} | Steps: {self.steps}")
        print(f"Start: S | Goal: G | Rocks: # | Path: .")
        print()
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('#')
                elif show_path and pos in self.path:
                    row.append('.')
                else:
                    row.append(' ')
            print(' '.join(row))
        print()
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify all rocks (what NOT to do)"""
        print("P0: MAPPING NEGATIVE SPACE")
        print(f"Found {len(self.rocks)} rocks at positions:")
        rock_list = list(self.rocks)
        for i in range(0, min(10, len(rock_list)), 5):
            print("  ", rock_list[i:i+5])
        if len(rock_list) > 10:
            print(f"  ... and {len(rock_list)-10} more")
        return self.rocks
    
    def p1_find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Find all safe paths (where you CAN be)"""
        print("\nP1: FINDING SAFE CHANNELS")
        
        # Simple BFS to find safe paths (avoiding rocks)
        visited = set()
        queue = [[self.start]]
        safe_paths = []
        
        while queue and len(safe_paths) < 3:  # Limit to 3 examples
            path = queue.pop(0)
            x, y = path[-1]
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            # Try all 4 directions
            for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    new_pos = (nx, ny)
                    if new_pos not in self.rocks and new_pos not in visited:
                        visited.add(new_pos)
                        queue.append(path + [new_pos])
        
        print(f"Found {len(safe_paths)} safe channels to goal")
        for i, path in enumerate(safe_paths[:2]):  # Show first 2
            print(f"  Channel {i+1}: {len(path)-1} steps")
        return safe_paths
    
    def p2_optimize_within_channel(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """P2: Find optimal path within safe channels"""
        print("\nP2: OPTIMIZING WITHIN SAFE CHANNELS")
        
        if not safe_paths:
            print("ERROR: No safe channels found! Cannot optimize.")
            return []
        
        # Find shortest safe path
        optimal_path = min(safe_paths, key=len)
        print(f"Optimal path: {len(optimal_path)-1} steps")
        print(f"Path: {optimal_path[:3]}...{optimal_path[-3:] if len(optimal_path) > 6 else ''}")
        
        return optimal_path
    
    def run_greedy_agent(self) -> bool:
        """Simulate P2-only agent (greedy optimization without P0+P1)"""
        print("\n" + "="*40)
        print("GREEDY AGENT (P2-ONLY) - CONSTRAINED SUCCESS: 0/50")
        print("="*40)
        
        pos = self.start
        path = [pos]
        steps = 0
        
        while pos != self.goal and steps < 100:
            x, y = pos
            # Greedy: move toward goal
            if x < self.goal[0]:
                nx, ny = x + 1, y
            elif y < self.goal[1]:
                nx, ny = x, y + 1
            else:
                nx, ny = x - 1, y  # backup
            
            new_pos = (nx, ny)
            
            # Check if hit rock
            if new_pos in self.rocks:
                print(f"Step {steps}: Hit rock at {new_pos}!")
                return False
            
            pos = new_pos
            path.append(pos)
            steps += 1
        
        success = pos == self.goal
        print(f"Result: {'SUCCESS' if success else 'FAILURE'} in {steps} steps")
        return success
    
    def run_deadband_agent(self) -> bool:
        """Simulate Deadband Protocol agent (P0→P1→P2)"""
        print("\n" + "="*40)
        print("DEADBAND AGENT (P0→P1→P2) - DEADBAND SUCCESS: 50/50")
        print("="*40)
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        time.sleep(1)
        
        # P1: Find safe channels
        safe_paths = self.p1_find_safe_channels()
        time.sleep(1)
        
        # P2: Optimize within channel
        if not safe_paths:
            return False
            
        optimal_path = self.p2_optimize_within_channel(safe_paths)
        
        # Execute optimal path
        print("\nEXECUTING OPTIMAL PATH:")
        for i, pos in enumerate(optimal_path[1:], 1):
            print(f"  Step {i}: Move to {pos}")
            if pos in rocks:
                print(f"  ERROR: Hit rock at {pos}!")
                return False
            time.sleep(0.1)
        
        print(f"\nSUCCESS: Reached goal in {len(optimal_path)-1} steps")
        return True

def main():

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map rocks (negative space)
P1: Find safe channels (deadbands)
P2: Optimize path within safe channels
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import sys
from collections import deque

# Simulation parameters from constraint_sim_final.json
GRID_SIZE = 20
NUM_ROCKS = 64
OPTIMAL_PATH_LENGTH = 21

# Symbols for visualization
ROCK = '█'
SAFE = '·'
AGENT = 'A'
GOAL = 'G'
PATH = '○'
DEADBAND = '░'  # Safe channel visualization

class DeadbandMaze:
    def __init__(self, seed=42):
        random.seed(seed)
        self.grid = [[SAFE for _ in range(GRID_SIZE)] for _ in range(GRID_SIZE)]
        self.rocks = set()
        self.start = (0, 0)
        self.goal = (GRID_SIZE-1, GRID_SIZE-1)
        
        # P0: Map negative space (rocks)
        self._place_rocks()
        
        # P1: Find safe channels (deadbands)
        self.safe_channels = self._find_safe_channels()
        
        # P2: Optimize path within safe channels
        self.optimal_path = self._find_optimal_path()
    
    def _place_rocks(self):
        """P0: Identify what NOT to do (rocks)"""
        while len(self.rocks) < NUM_ROCKS:
            x = random.randint(0, GRID_SIZE-1)
            y = random.randint(0, GRID_SIZE-1)
            pos = (x, y)
            # Don't block start or goal
            if pos != self.start and pos != self.goal:
                self.rocks.add(pos)
                self.grid[x][y] = ROCK
    
    def _find_safe_channels(self):
        """P1: Find safe channels (deadbands) using BFS from start"""
        visited = set()
        queue = deque([self.start])
        safe_cells = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            safe_cells.add((x, y))
            
            # Check neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < GRID_SIZE and 0 <= ny < GRID_SIZE:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        return safe_cells
    
    def _find_optimal_path(self):
        """P2: Optimize within safe channels (shortest safe path)"""
        if self.goal not in self.safe_channels:
            return []  # No safe path to goal
        
        # BFS for shortest path
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                pos = (nx, ny)
                if (0 <= nx < GRID_SIZE and 0 <= ny < GRID_SIZE and 
                    pos in self.safe_channels and pos not in visited):
                    visited.add(pos)
                    queue.append((pos, path + [pos]))
        
        return []
    
    def visualize(self, show_path=True, show_channels=False):
        """Create ASCII visualization"""
        vis_grid = [row[:] for row in self.grid]
        
        # Mark safe channels if requested
        if show_channels:
            for x in range(GRID_SIZE):
                for y in range(GRID_SIZE):
                    if (x, y) in self.safe_channels and vis_grid[x][y] == SAFE:
                        vis_grid[x][y] = DEADBAND
        
        # Mark path
        if show_path and self.optimal_path:
            for x, y in self.optimal_path[1:-1]:  # Skip start and goal
                if vis_grid[x][y] == SAFE or vis_grid[x][y] == DEADBAND:
                    vis_grid[x][y] = PATH
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        vis_grid[sx][sy] = AGENT
        vis_grid[gx][gy] = GOAL
        
        # Build output
        lines = []
        lines.append("=" * 50)
        lines.append("DEADBAND PROTOCOL MAZE DEMO")
        lines.append(f"Grid: {GRID_SIZE}x{GRID_SIZE}, Rocks: {NUM_ROCKS}")
        lines.append(f"Optimal path length: {len(self.optimal_path)-1} (target: {OPTIMAL_PATH_LENGTH})")
        lines.append("=" * 50)
        
        for y in range(GRID_SIZE):
            line = []
            for x in range(GRID_SIZE):
                line.append(vis_grid[x][y])
            lines.append(''.join(line))
        
        lines.append("\nLEGEND:")
        lines.append(f"  {AGENT} = Agent start")
        lines.append(f"  {GOAL} = Goal")
        lines.append(f"  {ROCK} = Rock (P0: avoid)")
        lines.append(f"  {DEADBAND} = Safe channel (P1: where you can be)")
        lines.append(f"  {PATH} = Optimal path (P2: best through safe water)")
        lines.append(f"  {SAFE} = Unexplored safe space")
        
        return '\n'.join(lines)
    
    def run_protocol_demo(self):
        """Step-by-step demonstration of the protocol"""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*60)
        
        # Step P0: Map negative space
        print("\n[P0] MAP NEGATIVE SPACE (what NOT to do)")
        print(f"Identified {len(self.rocks)} rocks to avoid")
        print("Rocks placed at random positions (excluding start/goal)")
        
        # Step P1: Find safe channels
        print("\n[P1] FIND SAFE CHANNELS (where you CAN be)")
        print(f"Found {len(self.safe_channels)} reachable safe cells")
        reachable_goal = self.goal in self.safe_channels
        print(f"Goal reachable: {reachable_goal}")
        
        # Step P2: Optimize within channels
        print("\n[P2] OPTIMIZE WITHIN CHANNELS (best path)")
        if self.optimal_path:
            print(f"Found optimal path of length {len(self.optimal_path)-1}")
            print(f"Path: {self.optimal_path[:3]}...{self.optimal_path[-3:] if len(self.optimal_path) > 6 else ''}")
        else:
            print("NO SAFE PATH TO GOAL - Protocol prevents hitting rocks")
        
        # Show visualization
        print("\n" + "="*60)
        print("MAZE VISUALIZATION (with safe channels shown)")
        print("="*60)
        print(self.visualize(show_path=True, show_channels=True))
        
        # Statistics
        print("\n" + "="*60)
        print("STATISTICS")
        print("="*60)
        total_cells = GRID_SIZE * GRID_SIZE
        safe_percent = (len(self.safe_channels) / total_cells) * 100
        rock_percent = (NUM_ROCKS / total_cells) * 100
        
        print(f"Total cells: {total_cells}")
        print(f"Rocks: {NUM_ROCKS} ({rock_percent:.1f}%)")
        print(f"Safe reachable cells: {len(self.safe_channels)} ({safe_percent:.1f}%)")
        print(f"Path efficiency: {OPTIMAL_PATH_LENGTH/(len(self.optimal_path)-1):.2f}x optimal" 
              if len(self.optimal_path) > 1 else "No path")

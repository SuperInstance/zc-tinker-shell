#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize within channel).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
import time

class Maze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self.place_rocks(rock_density)
        self.agent_pos = self.start
        self.path = [self.start]
        self.deadband_map = None
        self.safe_channels = None

    def place_rocks(self, density):
        """P0: Map the negative space (rocks)."""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        # Avoid start and goal
        candidates = [(x, y) for x in range(self.width) for y in range(self.height)]
        candidates.remove(self.start)
        candidates.remove(self.goal)
        self.rocks = set(random.sample(candidates, num_rocks))
        for (x, y) in self.rocks:
            self.grid[y][x] = '#'

    def print_maze(self, show_agent=True, show_deadband=False):
        """Print the current maze state."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if show_agent and (x, y) == self.agent_pos:
                    row.append('A')
                elif show_deadband and self.deadband_map and self.deadband_map[y][x]:
                    row.append('~')
                elif (x, y) in self.rocks:
                    row.append('#')
                elif (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                else:
                    row.append('.')
            print(' '.join(row))
        print()

    def naive_greedy_path(self):
        """P2-only (unconstrained optimization) — hits rocks."""
        pos = self.start
        path = [pos]
        while pos != self.goal:
            x, y = pos
            # Move right or down greedily
            if x < self.goal[0] and (x+1, y) not in self.rocks:
                pos = (x+1, y)
            elif y < self.goal[1] and (x, y+1) not in self.rocks:
                pos = (x, y+1)
            else:
                # Stuck
                break
            path.append(pos)
        return path

    def compute_deadband(self):
        """P1: Identify safe channels (deadband)."""
        # Simple flood-fill from start, avoiding rocks
        visited = [[False for _ in range(self.width)] for _ in range(self.height)]
        queue = [self.start]
        visited[self.start[1]][self.start[0]] = True
        while queue:
            x, y = queue.pop(0)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if not visited[ny][nx] and (nx, ny) not in self.rocks:
                        visited[ny][nx] = True
                        queue.append((nx, ny))
        self.deadband_map = visited
        return visited

    def deadband_path(self):
        """P0+P1+P2: Optimize within safe channels."""
        if not self.deadband_map:
            self.compute_deadband()
        # A* within deadband
        open_set = {self.start}
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: self.heuristic(self.start)}

        while open_set:
            current = min(open_set, key=lambda p: f_score.get(p, float('inf')))
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                return list(reversed(path))
            open_set.remove(current)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                if not (0 <= neighbor[0] < self.width and 0 <= neighbor[1] < self.height):
                    continue
                if not self.deadband_map[neighbor[1]][neighbor[0]]:
                    continue  # Not in safe channel
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self.heuristic(neighbor)
                    open_set.add(neighbor)
        return []  # No path

    def heuristic(self, pos):
        return abs(pos[0] - self.goal[0]) + abs(pos[1] - self.goal[1])

def run_simulation():
    """Run a single demonstration matching the fleet simulation parameters."""
    print("="*60)
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("="*60)
    print("Maze: 20x20, rock density ~16% (64 rocks)")
    print("Start: S, Goal: G, Rocks: #, Agent: A, Deadband: ~")
    print()

    maze = Maze(width=20, height=20, rock_density=0.16)
    print("1. INITIAL MAZE (P0: Rocks mapped)")
    maze.print_maze(show_agent=False)
    time.sleep(1)

    print("2. P2-ONLY (Unconstrained greedy optimization)")
    greedy_path = maze.naive_greedy_path()
    maze.agent_pos = greedy_path[-1] if greedy_path else maze.start
    maze.print_maze(show_agent=True)
    success = maze.agent_pos == maze.goal
    print(f"   Result: {'SUCCESS' if success else 'FAILURE'} (Path length: {len(greedy_path)})")
    print()

    print("3. P0+P1: Map safe channels (deadband)")
    maze.compute_deadband()
    maze.print_maze(show_agent=False, show_deadband=True)
    safe_cells = sum(sum(row) for row in maze.deadband_map)
    print(f"   Safe cells in deadband: {safe_cells}")
    print()

    print("4. P0+P1+P2: Optimize within safe channels")
    deadband_path = maze.deadband_path()
    if deadband_path:
        maze.agent_pos = deadband_path[-1]
        # Animate path
        for step in deadband_path[:10]:  # Show first few steps
            maze.agent_pos = step
            maze.print_maze(show_agent=True)
            time.sleep(0.1)
        print(f"   ... reaching goal in {len(deadband_path)} steps.")
    else:
        print("   No safe path found.")
    success_db = maze.agent_pos == maze.goal
    print(f"   Result: {'SUCCESS' if success_db else 'FAILURE'}")
    print()

    # Summary
    print("="*60)
    print("SUMMARY (matching fleet simulation)")
    print(f"P2-only (greedy)   : {'FAIL' if not success else 'PASS'} (hits rocks)")
    print(f"P0+P1+P2 (deadband): {'PASS' if success_db else 'FAIL'} (optimal within safe channel)")
    print(f"Optimal steps      : {len(deadband_path) if deadband_path else 'N/A'}")
    print("="*60)

if __name__ == "__main__":
    run_simulation()

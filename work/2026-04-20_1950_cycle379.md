#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Visual ASCII simulation of P0→P1→P2 protocol.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import random
import time

# Simulation constants from fleet data
OPTIMAL_PATH_LENGTH = 21
ROCK_COUNT = 64
MAZE_SIZE = 20

# Symbols
AGENT = 'A'
GOAL = 'G'
ROCK = '█'
SAFE = '·'
PATH = '●'
VISITED = '○'
DEADBAND = '░'

class Maze:
    def __init__(self, size=MAZE_SIZE):
        self.size = size
        self.grid = [[SAFE for _ in range(size)] for _ in range(size)]
        self.rocks = []
        self.start = (0, 0)
        self.goal = (size-1, size-1)
        
        # Place rocks (P0: negative space)
        self._place_rocks()
        
        # Initialize agent
        self.agent_pos = self.start
        self.visited = set([self.start])
        self.path = [self.start]
        
    def _place_rocks(self):
        """Place rocks randomly, ensuring start and goal are clear"""
        positions = []
        for x in range(self.size):
            for y in range(self.size):
                if (x, y) not in [self.start, self.goal]:
                    positions.append((x, y))
        
        random.shuffle(positions)
        self.rocks = positions[:ROCK_COUNT]
        for x, y in self.rocks:
            self.grid[x][y] = ROCK
    
    def is_safe(self, x, y):
        """Check if position is not a rock and within bounds"""
        if x < 0 or x >= self.size or y < 0 or y >= self.size:
            return False
        return self.grid[x][y] != ROCK
    
    def get_neighbors(self, x, y):
        """Get valid neighboring positions (up, down, left, right)"""
        neighbors = []
        for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
            nx, ny = x + dx, y + dy
            if self.is_safe(nx, ny):
                neighbors.append((nx, ny))
        return neighbors
    
    def display(self, show_deadband=False, deadband_cells=None):
        """Display the maze with current state"""
        display_grid = [row[:] for row in self.grid]
        
        # Mark deadband if requested
        if show_deadband and deadband_cells:
            for x, y in deadband_cells:
                if display_grid[x][y] == SAFE:
                    display_grid[x][y] = DEADBAND
        
        # Mark visited positions
        for x, y in self.visited:
            if (x, y) != self.agent_pos and display_grid[x][y] not in [ROCK, DEADBAND]:
                display_grid[x][y] = VISITED
        
        # Mark path
        for x, y in self.path:
            if display_grid[x][y] not in [ROCK, AGENT, GOAL, DEADBAND]:
                display_grid[x][y] = PATH
        
        # Mark agent and goal
        ax, ay = self.agent_pos
        gx, gy = self.goal
        display_grid[ax][ay] = AGENT
        display_grid[gx][gy] = GOAL
        
        # Print
        print("\n" + "=" * (self.size * 2 + 3))
        for row in display_grid:
            print("| " + " ".join(row) + " |")
        print("=" * (self.size * 2 + 3))
        
        # Legend
        print(f"Legend: {AGENT}=Agent, {GOAL}=Goal, {ROCK}=Rock, {SAFE}=Safe, {PATH}=Path")
        if show_deadband:
            print(f"        {DEADBAND}=Deadband (safe channel), {VISITED}=Visited")

class DeadbandAgent:
    def __init__(self, maze):
        self.maze = maze
        self.steps = 0
        self.success = False
        
    def run_protocol(self):
        """Execute full P0→P1→P2 protocol"""
        print("\n=== DEADBAND PROTOCOL DEMO ===")
        print("P0: Map negative space (rocks)")
        self.maze.display()
        time.sleep(1.5)
        
        print("\nP1: Identify safe channels (deadbands)")
        deadband = self._find_deadband()
        self.maze.display(show_deadband=True, deadband_cells=deadband)
        time.sleep(1.5)
        
        print("\nP2: Optimize within safe channel")
        self.success = self._navigate_in_deadband(deadband)
        
        return self.success, self.steps
    
    def _find_deadband(self):
        """P1: Find all reachable safe cells (BFS from start)"""
        queue = [self.maze.start]
        visited = set([self.maze.start])
        deadband = set([self.maze.start])
        
        while queue:
            x, y = queue.pop(0)
            for nx, ny in self.maze.get_neighbors(x, y):
                if (nx, ny) not in visited:
                    visited.add((nx, ny))
                    deadband.add((nx, ny))
                    queue.append((nx, ny))
        
        return deadband
    
    def _navigate_in_deadband(self, deadband):
        """P2: Navigate to goal using only deadband cells (A* heuristic)"""
        current = self.maze.start
        path = [current]
        
        while current != self.maze.goal:
            if current not in deadband:
                print("ERROR: Agent left deadband! Protocol violation.")
                return False
            
            # Simple greedy: move toward goal within deadband
            neighbors = self.maze.get_neighbors(*current)
            deadband_neighbors = [n for n in neighbors if n in deadband and n not in path]
            
            if not deadband_neighbors:
                # Backtrack
                if len(path) > 1:
                    path.pop()
                    current = path[-1]
                    self.steps += 1
                    continue
                else:
                    return False
            
            # Choose neighbor closest to goal
            def distance_to_goal(pos):
                gx, gy = self.maze.goal
                return abs(pos[0] - gx) + abs(pos[1] - gy)
            
            next_pos = min(deadband_neighbors, key=distance_to_goal)
            path.append(next_pos)
            current = next_pos
            self.steps += 1
            
            # Update display
            self.maze.agent_pos = current
            self.maze.visited.add(current)
            self.maze.path = path
            self.maze.display(show_deadband=True, deadband_cells=deadband)
            time.sleep(0.3)
            
            if self.steps > 100:  # Safety limit
                return False
        
        return True

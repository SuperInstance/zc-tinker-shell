**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on the fleet knowledge, I have the constraint simulation results showing the Deadband Protocol's effectiveness (50/50 success vs 0/50 for constrained-only). I will implement a visual demonstration using ASCII to show the three phases.

**Approach:**
1. **P0 (Map negative space):** Generate a random maze with obstacles ('#') as rocks.
2. **P1 (Find safe channels):** Use BFS to find all reachable paths from start to goal that avoid rocks.
3. **P2 (Optimize within channels):** Select the shortest safe path from the set of safe channels.

**Output:** A Python script that prints each phase step-by-step with ASCII visualization.

## File: `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 with visual ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque

def generate_maze(width=10, height=10, rock_density=0.25):
    """Generate a random maze with rocks as obstacles."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    rocks = 0
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density and (x, y) not in [(0, 0), (width-1, height-1)]:
                maze[y][x] = '#'
                rocks += 1
    maze[0][0] = 'S'  # Start
    maze[height-1][width-1] = 'G'  # Goal
    return maze, rocks

def print_maze(maze, title="Maze"):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def p0_map_rocks(maze):
    """P0: Identify rocks (negative space)."""
    rocks = []
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start, goal):
    """P1: Find all safe paths from start to goal (BFS)."""
    width, height = len(maze[0]), len(maze)
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    
    queue = deque([(start[0], start[1], [])])
    visited = set()
    safe_paths = []
    
    while queue:
        x, y, path = queue.popleft()
        if (x, y) in visited:
            continue
        visited.add((x, y))
        
        current_path = path + [(x, y)]
        
        if (x, y) == goal:
            safe_paths.append(current_path)
            continue
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if maze[ny][nx] != '#' and (nx, ny) not in visited:
                    queue.append((nx, ny, current_path))
    
    return safe_paths

def p2_optimize_within_channels(safe_paths):
    """P2: Select shortest path from safe channels."""
    if not safe_paths:
        return None
    return min(safe_paths, key=len)

def visualize_phase(maze, phase, data):
    """Visualize each phase on the maze."""
    viz = [row[:] for row in maze]
    if phase == "P0":
        rocks = data
        for x, y in rocks:
            viz[y][x] = '#'  # Already marked
        print_maze(viz, "P0: Rocks (Negative Space)")
        print(f"Rocks identified: {len(rocks)}")
    
    elif phase == "P1":
        safe_paths = data
        # Mark all safe cells (reachable without hitting rocks)
        safe_cells = set()
        for path in safe_paths:
            for x, y in path:
                safe_cells.add((x, y))
        for y in range(len(viz)):
            for x in range(len(viz[0])):
                if (x, y) in safe_cells and viz[y][x] not in ['S', 'G']:
                    viz[y][x] = '~'  # Safe channel
        print_maze(viz, "P1: Safe Channels (~)")
        print(f"Safe paths found: {len(safe_paths)}")
    
    elif phase == "P2":
        optimal_path = data
        if optimal_path:
            for x, y in optimal_path[1:-1]:  # Exclude start and goal
                viz[y][x] = '*'  # Optimal path
            print_maze(viz, "P2: Optimal Path (*) within Safe Channels")
            print(f"Optimal path length: {len(optimal_path)} steps")
        else:
            print("No safe path to goal (all channels blocked).")
    
    return viz

def main():
    print("DEADBAND PROTOCOL DEMO - ASCII MAZE")
    print("Based on fleet simulation: deadband_success=50/50, constrained_success=0/50")
    print("=" * 50)
    
    # Generate maze
    maze, rocks_count = generate_maze(width=8, height=8, rock_density=0.2)
    print_maze(maze, "Initial Maze (S=Start, G=Goal, #=Rock)")
    
    # Phase 0: Map rocks
    rocks = p0_map_rocks(maze)
    visualize_phase(maze, "P0", rocks)
    
    # Phase 1: Find safe channels
    start = (0, 0)
    goal = (len(maze[0])-1, len(maze)-1)
    safe_paths = p1_find_safe_channels(maze, start, goal)
    visualize_phase(maze, "P1", safe_paths)
    
    # Phase 2: Optimize within channels
    optimal_path = p2_optimize_within_channels(safe_paths)
    visualize_phase(maze, "P2", optimal_path)
    
    # Summary
    print("\n=== DEADBAND PROTOCOL SUMMARY ===")
    print(f"P0 Rocks mapped: {len(rocks)}")
    print(f"P1 Safe channels found: {len(safe_paths)}")
    print(f"P2 Optimal path length: {len(optimal_path) if optimal_path else 'N/A'}")
    print(f"Success: {'YES' if optimal_path else 'NO'}")
    
    # Demonstrate violation (skipping P0/P1)
    print("\n--- VIOLATION EXAMPLE (Skip to P2) ---")
    print("If agent tries to optimize without mapping rocks first:")
    print("Greedy approach would hit rocks with high probability.")
    print("Simulation data: constrained_success = 0/50")

if __name__ == "__main__":
    main()
```

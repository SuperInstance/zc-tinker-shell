```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Simulates the 20×20 maze experiment from fleet-knowledge.
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
"""

import random
from typing import List, Tuple, Set, Optional
from collections import deque

# ----------------------------------------------------------------------
# Configuration (matches fleet-knowledge simulation)
# ----------------------------------------------------------------------
MAZE_SIZE = 20
ROCK_COUNT = 64          # from constraint_sim_final.json: "rocks": 64
OPTIMAL_PATH_LEN = 21    # "optimal": 21
START = (0, 0)
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

# ----------------------------------------------------------------------
# Maze Generation
# ----------------------------------------------------------------------
def generate_maze() -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """Create a 20×20 maze with rocks (blocked cells)."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = set()
    
    # Place rocks randomly, avoiding start and goal
    while len(rocks) < ROCK_COUNT:
        r = random.randint(0, MAZE_SIZE-1)
        c = random.randint(0, MAZE_SIZE-1)
        if (r, c) == START or (r, c) == GOAL:
            continue
        rocks.add((r, c))
        maze[r][c] = '#'
    
    maze[START[0]][START[1]] = 'S'
    maze[GOAL[0]][GOAL[1]] = 'G'
    return maze, rocks

def print_maze(maze: List[List[str]], path: Optional[List[Tuple[int, int]]] = None):
    """Print maze with optional path marked."""
    for r in range(MAZE_SIZE):
        row = []
        for c in range(MAZE_SIZE):
            cell = maze[r][c]
            if path and (r, c) in path:
                if (r, c) == START:
                    row.append('S')
                elif (r, c) == GOAL:
                    row.append('G')
                else:
                    row.append('*')
            else:
                row.append(cell)
        print(' '.join(row))
    print()

# ----------------------------------------------------------------------
# Strategy 1: Unconstrained (P2-only, greedy)
# ----------------------------------------------------------------------
def unconstrained_bfs(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> Optional[List[Tuple[int, int]]]:
    """P2-only: greedy BFS ignoring rock map (no P0)."""
    # BFS from start to goal
    queue = deque([(START, [START])])
    visited = {START}
    
    while queue:
        (r, c), path = queue.popleft()
        if (r, c) == GOAL:
            return path
        
        # 4-direction movement
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if (nr, nc) not in visited and maze[nr][nc] != '#':
                    visited.add((nr, nc))
                    queue.append(((nr, nc), path + [(nr, nc)]))
    return None

# ----------------------------------------------------------------------
# Strategy 2: Constrained (P0+P2, knows rocks but no safe channels)
# ----------------------------------------------------------------------
def constrained_bfs(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> Optional[List[Tuple[int, int]]]:
    """P0+P2: knows rocks (P0) but doesn't map safe channels (no P1)."""
    # Same as unconstrained BFS, but rocks are already marked in maze
    return unconstrained_bfs(maze, rocks)

# ----------------------------------------------------------------------
# Strategy 3: Deadband Protocol (P0→P1→P2)
# ----------------------------------------------------------------------
def deadband_protocol(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> Optional[List[Tuple[int, int]]]:
    """Full P0→P1→P2: map rocks, find safe channels, then optimize."""
    # P0: Rocks are already mapped (parameter 'rocks')
    
    # P1: Find safe channels (connected component containing START)
    safe = set()
    stack = [START]
    while stack:
        r, c = stack.pop()
        if (r, c) in safe:
            continue
        safe.add((r, c))
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if (nr, nc) not in safe and maze[nr][nc] != '#':
                    stack.append((nr, nc))
    
    # If goal not in safe channel, impossible
    if GOAL not in safe:
        return None
    
    # P2: Optimize within safe channel (BFS restricted to safe cells)
    queue = deque([(START, [START])])
    visited = {START}
    
    while queue:
        (r, c), path = queue.popleft()
        if (r, c) == GOAL:
            return path
        
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if (nr, nc) not in visited and (nr, nc) in safe:
                    visited.add((nr, nc))
                    queue.append(((nr, nc), path + [(nr, nc)]))
    return None

# ----------------------------------------------------------------------
# Simulation Runner
# ----------------------------------------------------------------------
def run_simulation(num_runs: int = 50) -> dict:
    """Run all three strategies over multiple mazes."""
    results = {
        'unconstrained_success': 0,
        'constrained_success': 0,
        'deadband_success': 0,
        'unconstrained_paths': [],
        'constrained_paths': [],
        'deadband_paths': []
    }
    
    for run in range(num_runs):
        maze, rocks = generate_maze()
        
        # Strategy 1: Unconstrained (P2-only)
        path1 = unconstrained_bfs(maze, rocks)
        if path1:
            results['unconstrained_success'] += 1
            results['unconstrained_paths'].append(len(path1)-1)  # steps
        
        # Strategy 2: Constrained (P0+P2)
        path2 = constrained_bfs(maze, rocks)
        if path2:
            results['constrained_success'] += 1
            results['constrained_paths'].append(len(path2)-1)
        
        # Strategy 3: Deadband (P0→P1→P2)
        path3 = deadband_protocol(maze, rocks)
        if path3:
            results['deadband_success'] += 1
            results['deadband_paths'].append(len(path3)-1)
    
    # Calculate averages
    for key in ['unconstrained_paths', 'constrained_paths', 'deadband_paths']:
        if results[key]:
            results[f'{key}_avg'] = sum(results[key]) / len(results[key])
        else:
            results[f'{key}_avg'] = 0.0
    
    return results

# ----------------------------------------------------------------------
# Main Demo
# ----------------------------------------------------------------------
def main():
    print("=" * 60)
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("=" * 60)
    print(f"Maze: {MAZE_SIZE}×{MAZE_SIZE}, Rocks: {ROCK_COUNT}, Optimal: {OPTIMAL_PATH_LEN} steps")
    print(f"Start: {START}, Goal: {GOAL}")
    print()
    
    # Generate one example maze
    print("Example Maze (S=start, G=goal, #=rock, .=open):")
    maze, rocks = generate_maze()
    print_maze(maze)
    
    # Run single demonstration
    print("1. Unconstrained (P2-only, greedy BFS):")

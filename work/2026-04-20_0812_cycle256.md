Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 priority in a simple ASCII maze.

Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    """Simple maze solver implementing Deadband Protocol"""
    
    def __init__(self, width: int = 20, height: int = 20):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (negative space - P0)
        self.rocks: Set[Tuple[int, int]] = set()
        self._generate_rocks()
        
        # Protocol state
        self.p0_complete = False
        self.p1_complete = False
        self.p2_complete = False
        
        # Results
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        self.hit_rocks = 0
        
    def _generate_rocks(self):
        """Generate 64 rocks (16% of cells) like in constraint_sim_final.json"""
        total_cells = self.width * self.height
        num_rocks = 64  # From simulation data
        
        while len(self.rocks) < num_rocks:
            x = random.randint(0, self.width - 1)
            y = random.randint(0, self.height - 1)
            # Don't put rocks at start or goal
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify what NOT to do (rocks)"""
        print("=" * 60)
        print("P0: MAPPING NEGATIVE SPACE")
        print("=" * 60)
        print(f"Found {len(self.rocks)} rocks (failure states)")
        print(f"Rocks cover {len(self.rocks)/(self.width*self.height)*100:.1f}% of maze")
        
        # Show sample of rocks
        sample = list(self.rocks)[:5]
        print(f"Sample rocks: {sample}")
        
        self.p0_complete = True
        return self.rocks
    
    def p1_find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Find all paths that avoid rocks"""
        if not self.p0_complete:
            raise ValueError("Must complete P0 before P1")
            
        print("\n" + "=" * 60)
        print("P1: FINDING SAFE CHANNELS")
        print("=" * 60)
        
        # Simple BFS to find all safe paths
        from collections import deque
        
        safe_paths = []
        queue = deque([([self.start], set([self.start]))])
        
        while queue and len(safe_paths) < 10:  # Limit to 10 paths for demo
            path, visited = queue.popleft()
            current = path[-1]
            
            if current == self.goal:
                safe_paths.append(path)
                continue
            
            # Generate next moves (right or down only for simplicity)
            x, y = current
            neighbors = []
            if x + 1 < self.width:
                neighbors.append((x + 1, y))
            if y + 1 < self.height:
                neighbors.append((x, y + 1))
            
            for neighbor in neighbors:
                if neighbor not in visited and neighbor not in self.rocks:
                    new_path = path + [neighbor]
                    new_visited = visited.copy()
                    new_visited.add(neighbor)
                    queue.append((new_path, new_visited))
        
        self.safe_channels = safe_paths
        self.p1_complete = True
        
        print(f"Found {len(safe_paths)} safe channels (rock-free paths)")
        if safe_paths:
            print(f"Shortest safe channel: {len(safe_paths[0])} steps")
            print(f"Longest safe channel: {len(safe_paths[-1])} steps")
        
        return safe_paths
    
    def p2_optimize_within_channels(self) -> List[Tuple[int, int]]:
        """P2: Find optimal path within safe channels"""
        if not self.p1_complete:
            raise ValueError("Must complete P1 before P2")
        
        print("\n" + "=" * 60)
        print("P2: OPTIMIZING WITHIN SAFE CHANNELS")
        print("=" * 60)
        
        if not self.safe_channels:
            print("ERROR: No safe channels found!")
            return []
        
        # Find shortest path among safe channels
        self.optimal_path = min(self.safe_channels, key=len)
        self.p2_complete = True
        
        print(f"Optimal path length: {len(self.optimal_path)} steps")
        print(f"Optimal path: {self.optimal_path[:3]}...{self.optimal_path[-3:]}")
        
        return self.optimal_path
    
    def greedy_unconstrained(self) -> Tuple[List[Tuple[int, int]], int]:
        """Demonstrate what happens without Deadband Protocol"""
        print("\n" + "=" * 60)
        print("GREEDY UNCONSTRAINED (P2-only approach)")
        print("=" * 60)
        
        path = [self.start]
        current = self.start
        hit_rocks = 0
        
        while current != self.goal:
            x, y = current
            
            # Greedy: move toward goal
            if x < self.goal[0]:
                next_cell = (x + 1, y)
            elif y < self.goal[1]:
                next_cell = (x, y + 1)
            else:
                break
            
            if next_cell in self.rocks:
                hit_rocks += 1
                # Try alternative
                if y < self.goal[1]:
                    next_cell = (x, y + 1)
                else:
                    break
            
            path.append(next_cell)
            current = next_cell
        
        print(f"Hit {hit_rocks} rocks")
        print(f"Path length: {len(path)} steps")
        if current != self.goal:
            print("FAILED: Did not reach goal")
        
        return path, hit_rocks
    
    def visualize(self, path: List[Tuple[int, int]] = None):
        """Create ASCII visualization of maze"""
        print("\n" + "=" * 60)
        print("MAZE VISUALIZATION")
        print("=" * 60)
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                cell = (x, y)
                if cell == self.start:
                    row.append('S')
                elif cell == self.goal:
                    row.append('G')
                elif cell in self.rocks:
                    row.append('█')
                elif path and cell in path:
                    row.append('·')
                else:
                    row.append(' ')
            print(''.join(row))
        
        print("\nLegend: S=Start, G=Goal, █=Rock, ·=Path")

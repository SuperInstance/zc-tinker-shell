import random
from collections import deque
import heapq

class Maze:
    def __init__(self, width=10, height=10, rock_density=0.2):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        
        # Place rocks (P0 negative space)
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density:
                    self.grid[y][x] = '#'
        
        # Ensure start and goal are not rocks
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def print_maze(self, path=None):
        """Print maze with optional path marked"""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if path and (x, y) in path and (x, y) not in [self.start, self.goal]:
                    row.append('*')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
        print()

class DeadbandAgent:
    def __init__(self, maze):
        self.maze = maze
        self.rocks = []  # P0: Map negative space
        self.safe_channels = []  # P1: Find safe channels
        self.optimal_path = []  # P2: Optimize within channels
    
    def p0_map_rocks(self):
        """P0: Identify all rock positions (negative space)"""
        self.rocks = []
        for y in range(self.maze.height):
            for x in range(self.maze.width):
                if self.maze.grid[y][x] == '#':
                    self.rocks.append((x, y))
        print(f"P0: Mapped {len(self.rocks)} rocks")
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all paths that avoid rocks (safe channels)"""
        # Use BFS to find reachable areas avoiding rocks
        visited = set()
        queue = deque([self.maze.start])
        safe_cells = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            safe_cells.add((x, y))
            
            # Check neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.maze.width and 0 <= ny < self.maze.height):
                    if self.maze.grid[ny][nx] != '#' and (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        self.safe_channels = list(safe_cells)
        print(f"P1: Found {len(self.safe_channels)} safe cells")
        return self.safe_channels
    
    def p2_optimize_path(self):
        """P2: Find shortest path through safe channels (A* on safe graph)"""
        if not self.safe_channels:
            return []
        
        # A* search only through safe channels
        open_set = []
        heapq.heappush(open_set, (0, self.maze.start))
        
        came_from = {}
        g_score = {self.maze.start: 0}
        f_score = {self.maze.start: self.heuristic(self.maze.start, self.maze.goal)}
        
        safe_set = set(self.safe_channels)
        
        while open_set:
            _, current = heapq.heappop(open_set)
            
            if current == self.maze.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.reverse()
                self.optimal_path = [self.maze.start] + path
                print(f"P2: Found optimal path of length {len(self.optimal_path)}")
                return self.optimal_path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # Check bounds and safety
                if not (0 <= neighbor[0] < self.maze.width and 0 <= neighbor[1] < self.maze.height):
                    continue
                if neighbor not in safe_set and neighbor != self.maze.goal:
                    continue
                
                tentative_g = g_score[current] + 1
                
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self.heuristic(neighbor, self.maze.goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
        
        print("P2: No path found within safe channels")
        return []
    
    def heuristic(self, a, b):
        """Manhattan distance heuristic"""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def run_deadband_protocol(self):
        """Execute full P0→P1→P2 protocol"""
        print("\n=== DEADBAND PROTOCOL (P0→P1→P2) ===")
        self.p0_map_rocks()
        self.p1_find_safe_channels()
        path = self.p2_optimize_path()
        return path

```python
#!/usr/bin/env python3
"""
deadband_maze_demo.py

Demonstrates the Deadband Protocol (P0→P1→P2) in a 20×20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Optimal path length: 21, Rocks: 64.
"""

import random
from collections import deque

# Constants from simulation data
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LEN = 21
START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

def generate_maze():
    """P0: Map negative space (rocks)."""
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        rocks.add((random.randint(0, MAZE_SIZE - 1), random.randint(0, MAZE_SIZE - 1)))
    # Ensure start and goal are not rocks
    rocks.discard(START)
    rocks.discard(GOAL)
    return rocks

def find_safe_channels(rocks):
    """P1: Find all cells reachable without hitting rocks."""
    safe = set()
    queue = deque([START])
    visited = set([START])
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        x, y = queue.popleft()
        safe.add((x, y))
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    return safe

def optimize_path(safe):
    """P2: Find shortest path within safe channels (BFS)."""
    if GOAL not in safe:
        return None  # No safe channel to goal
    
    queue = deque([(START, [START])])
    visited = set([START])
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        (x, y), path = queue.popleft()
        if (x, y) == GOAL:
            return path
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) in safe and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    return None

def visualize(rocks, safe, path):
    """ASCII visualization of maze."""
    grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    for x, y in rocks:
        grid[y][x] = '█'  # Rock
    for x, y in safe:
        if grid[y][x] == '.':
            grid[y][x] = ' '  # Safe channel
    if path:
        for x, y in path:
            grid[y][x] = '●'  # Optimal path
    grid[START[1]][START[0]] = 'S'
    grid[GOAL[1]][GOAL[0]] = 'G'
    
    print("20×20 Maze - Deadband Protocol Demo")
    print("█ = Rock (P0 negative space)")
    print("  = Safe channel (P1)")
    print("● = Optimal path (P2)")
    print("S = Start, G = Goal\n")
    for row in grid:
        print(''.join(row))
    print()

def main():
    random.seed(42)  # Deterministic for demo
    
    # P0: Map negative space
    rocks = generate_maze()
    print(f"P0: Mapped {len(rocks)} rocks (negative space)")
    
    # P1: Identify safe channels
    safe = find_safe_channels(rocks)
    print(f"P1: Found {len(safe)} safe cells (channels)")
    
    # P2: Optimize within channels
    path = optimize_path(safe)
    if path:
        print(f"P2: Optimal path length = {len(path)-1} (theoretical optimum: {OPTIMAL_PATH_LEN})")
    else:
        print("P2: No safe path to goal (constrained failure)")
    
    # Visualization
    visualize(rocks, safe, path)
    
    # Protocol validation
    if path:
        # Check no rocks in path
        assert all(p not in rocks for p in path), "VIOLATION: Path hits rocks!"
        # Check all path cells are safe
        assert all(p in safe for p in path), "VIOLATION: Path leaves safe channels!"
        print("Deadband protocol satisfied: P0→P1→P2 followed.")
    else:
        print("Goal unreachable within safe channels (constrained success = 0 in some simulations).")

if __name__ == "__main__":
    main()
```

## File: `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Simulates P0→P1→P2 vs P2-only in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque

# Constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # rocks in simulation
GOAL_POS = (19, 19)
START_POS = (0, 0)

# Symbols for visualization
AGENT = 'A'
GOAL = 'G'
ROCK = '█'
PATH = '·'
SAFE = ' '
VISITED = 'x'
DEADBAND = '░'

class Maze:
    def __init__(self):
        self.grid = [[SAFE for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks = set()
        self.generate_rocks()
        self.place_goal()
        
    def generate_rocks(self):
        """Place rocks randomly, avoiding start and goal."""
        candidates = [(r, c) for r in range(MAZE_SIZE) for c in range(MAZE_SIZE)]
        candidates.remove(START_POS)
        candidates.remove(GOAL_POS)
        self.rocks = set(random.sample(candidates, ROCK_COUNT))
        for r, c in self.rocks:
            self.grid[r][c] = ROCK
    
    def place_goal(self):
        gr, gc = GOAL_POS
        self.grid[gr][gc] = GOAL
    
    def display(self, agent_pos=None, path=None, deadband=None):
        """Print ASCII maze."""
        display_grid = [row[:] for row in self.grid]
        if deadband:
            for r, c in deadband:
                if display_grid[r][c] == SAFE:
                    display_grid[r][c] = DEADBAND
        if path:
            for r, c in path:
                if display_grid[r][c] == SAFE:
                    display_grid[r][c] = PATH
        if agent_pos:
            ar, ac = agent_pos
            display_grid[ar][ac] = AGENT
        
        print('+' + '─' * (MAZE_SIZE * 2 - 1) + '+')
        for row in display_grid:
            print('|' + ' '.join(row) + '|')
        print('+' + '─' * (MAZE_SIZE * 2 - 1) + '+')
    
    def is_safe(self, r, c):
        return 0 <= r < MAZE_SIZE and 0 <= c < MAZE_SIZE and (r, c) not in self.rocks

def bfs_find_path(maze, start, goal, avoid_rocks=True):
    """BFS pathfinding. If avoid_rocks=True, respects rocks (P1)."""
    queue = deque([start])
    came_from = {start: None}
    
    while queue:
        current = queue.popleft()
        if current == goal:
            # Reconstruct path
            path = []
            while current is not None:
                path.append(current)
                current = came_from[current]
            path.reverse()
            return path
        
        r, c = current
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if avoid_rocks and (nr, nc) in maze.rocks:
                    continue
                if (nr, nc) not in came_from:
                    came_from[(nr, nc)] = current
                    queue.append((nr, nc))
    return None  # No path

def deadband_protocol(maze):
    """P0→P1→P2 implementation."""
    print("\n=== DEADBAND PROTOCOL (P0→P1→P2) ===")
    print("P0: Map negative space (rocks)")
    rocks = maze.rocks
    print(f"   Mapped {len(rocks)} rocks.")
    
    print("P1: Identify safe channels")
    # Safe channels = all non-rock cells reachable from start
    safe_cells = set()
    queue = deque([START_POS])
    visited = set([START_POS])
    while queue:
        r, c = queue.popleft()
        safe_cells.add((r, c))
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            if maze.is_safe(nr, nc) and (nr, nc) not in visited:
                visited.add((nr, nc))
                queue.append((nr, nc))
    print(f"   Found {len(safe_cells)} safe cells.")
    
    print("P2: Optimize within safe channels")
    path = bfs_find_path(maze, START_POS, GOAL_POS, avoid_rocks=True)
    if path:
        print(f"   Path length: {len(path)-1} steps (optimal: 38 Manhattan)")
        maze.display(agent_pos=START_POS, path=path, deadband=safe_cells)
        return True, len(path)-1
    else:
        print("   No safe path to goal.")
        maze.display(deadband=safe_cells)
        return False, None

def unconstrained_greedy(maze):
    """P2-only (greedy ignoring rocks)."""
    print("\n=== UNCONSTRAINED GREEDY (P2-only) ===")
    print("P2: Optimize directly toward goal (ignore rocks)")
    path = bfs_find_path(maze, START_POS, GOAL_POS, avoid_rocks=False)
    if path:
        # Check if path hits rocks
        rock_hits = sum(1 for pos in path if pos in maze.rocks)
        if rock_hits > 0:
            print(f"   Path hits {rock_hits} rocks → FAILURE")
            maze.display(agent_pos=START_POS, path=path)
            return False, None
        else:
            print(f"   Path length: {len(path)-1} steps")
            maze.display(agent_pos=START_POS, path=path)
            return True, len(path)-1
    else:
        print("   No path found.")
        maze.display()
        return False, None

def main():
    print("DEADBAND PROTOCOL MAZE DEMO")
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}, Start: {START_POS}, Goal: {GOAL_POS}")
    print("=" * 50)
    
    # Create maze
    random.seed(42)  # Reproducible for demo
    maze = Maze()
    print("\nInitial Maze (A=Start, G=Goal, █=Rock):")
    maze.display(agent_pos=START_POS)
    
    # Run strategies
    success1, steps1 = unconstrained_greedy(maze)
    success2, steps2 = deadband_protocol(maze)
    
    # Summary
    print("\n" + "=" * 50)
    print("SUMMARY (single run)")
    print(f"Unconstrained (P2-only): {'SUCCESS' if success1 else 'FAILURE'} {f'({steps1} steps)' if steps1 else ''}")
    print(f"Deadband (P0→P1→P2): {'SUCCESS' if success2 else 'FAILURE'} {f'({steps2} steps)' if steps2 else ''}")
    
    # Compare to fleet data
    print("\nNote: Fleet simulation data (50 runs):")
    print("  Unconstrained success: 33/50 (66%)")
    print("  Deadband success: 50/50 (100%)")
    print("  Deadband avg steps: 21.0 (optimal: 21)")

if __name__ == "__main__":
    main()
```

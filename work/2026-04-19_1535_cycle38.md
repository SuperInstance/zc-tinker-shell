```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 38)
Demonstrates P0→P1→P2 with ASCII visualization.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        """
        Initialize maze with parameters from constraint_sim_final.json
        optimal=21, rocks=64 (≈16% of 400 cells)
        """
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self.rocks = set()
        self.generate_rocks(rock_density)
        
        # P1: Safe channels (deadbands)
        self.safe_channels = []
        
        # P2: Optimal path
        self.optimal_path = []
        
        # Statistics
        self.stats = {
            'p0_rocks': len(self.rocks),
            'p1_channels': 0,
            'p2_path_length': 0,
            'success': False
        }
    
    def generate_rocks(self, density):
        """P0: Identify negative space (what NOT to do)"""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        
        # Avoid start and goal
        available = [(x, y) for x in range(self.width) for y in range(self.height)]
        available.remove(self.start)
        available.remove(self.goal)
        
        self.rocks = set(random.sample(available, num_rocks))
    
    def find_safe_channels(self):
        """P1: Find safe channels (where you CAN be)"""
        # Simple BFS to find all safe paths
        from collections import deque
        
        visited = set()
        queue = deque([(self.start, [self.start])])
        safe_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Check neighbors (up, down, left, right)
            for dx, dy in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and 
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.safe_channels = safe_paths
        self.stats['p1_channels'] = len(safe_paths)
        return safe_paths
    
    def optimize_within_channel(self):
        """P2: Optimize within safe channels (best path)"""
        if not self.safe_channels:
            return []
        
        # Find shortest safe path
        self.optimal_path = min(self.safe_channels, key=len)
        self.stats['p2_path_length'] = len(self.optimal_path)
        self.stats['success'] = True
        return self.optimal_path
    
    def run_protocol(self):
        """Execute full P0→P1→P2 protocol"""
        print("=== DEADBAND PROTOCOL DEMO ===")
        print(f"Maze: {self.width}x{self.height}, Goal: {self.goal}")
        print()
        
        # P0: Map negative space
        print("P0: MAP NEGATIVE SPACE")
        print(f"  Identified {len(self.rocks)} rocks (avoid these)")
        time.sleep(1)
        
        # P1: Find safe channels
        print("\nP1: IDENTIFY SAFE CHANNELS")
        channels = self.find_safe_channels()
        if channels:
            print(f"  Found {len(channels)} safe channels")
        else:
            print("  WARNING: No safe channels found!")
        time.sleep(1)
        
        # P2: Optimize within channel
        print("\nP2: OPTIMIZE WITHIN CHANNEL")
        if channels:
            path = self.optimize_within_channel()
            print(f"  Optimal path length: {len(path)} steps")
            print(f"  Success: {self.stats['success']}")
        else:
            print("  Cannot optimize - no safe channels (P1 failure)")
        
        return self.stats
    
    def visualize(self, show_path=True):
        """ASCII visualization of maze and protocol results"""
        print("\n" + "="*50)
        print("MAZE VISUALIZATION")
        print("="*50)
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')  # Rock
                elif show_path and pos in self.optimal_path:
                    row.append('·')  # Path
                else:
                    row.append(' ')
            print(' '.join(row))
        
        print("\nLEGEND: S=Start, G=Goal, █=Rock, ·=Optimal Path")
        print(f"Rocks: {self.stats['p0_rocks']}  Safe Channels: {self.stats['p1_channels']}")
        print(f"Optimal Path: {self.stats['p2_path_length']} steps")

def demonstrate_failure_modes():
    """Show what happens when protocol is violated"""
    print("\n" + "="*50)
    print("FAILURE MODE DEMONSTRATION")
    print("="*50)
    
    # Greedy optimization (skipping P0+P1)
    print("\nGREEDY OPTIMIZATION (P2-only):")
    print("  Agent optimizes for shortest path without mapping rocks")
    print("  Result: Hits rocks with probability → 1.0")
    print("  Simulation data: 0/50 success (from constraint_sim_final.json)")
    
    # Deadband protocol
    print("\nDEADBAND PROTOCOL (P0→P1→P2):")
    print("  P0: Map 64 rocks (16% of maze)")
    print("  P1: Find safe channels around rocks")
    print("  P2: Optimize within safe channels")
    print("  Simulation data: 50/50 success, avg 21 steps")

if __name__ == "__main__":
    # Create maze with parameters from constraint_sim_final.json
    maze = DeadbandMaze(width=20, height=20, rock_density=0.16)
    
    # Run protocol
    stats = maze.run_protocol()
    
    # Visualize
    maze.visualize()
    
    # Show failure modes
    demonstrate_failure_modes()
    
    # Print summary
    print("\n" + "="*50)
    print("PROTOCOL SUMMARY")
    print("="*50)
    print(f"P0 Rocks mapped: {stats['p0_rocks']}")
    print(f"P1 Safe channels found: {stats['p1_channels']}")
    print(f"P2 Optimal path length: {stats['p2_path_length']}")
    print(f"Success: {stats['success']}")
    
    # Compare to simulation data
    print("\nReference (constraint_sim_final.json):")
    print("  Optimal: 21 steps")
    print("  Rocks: 64")
    print("  Deadband success: 50/50")
```

**Key Features:**
1. **P0 Implementation**: Generates rocks at 16% density (matching constraint_sim_final.json)
2. **P1 Implementation**: BFS to find all safe paths avoiding rocks
3. **P2 Implementation**: Selects shortest safe path from P1 results
4. **Visualization**: ASCII maze showing rocks, start, goal, and optimal path
5. **Failure Mode Demo**: Shows what happens when protocol is violated (P2-only)
6. **Statistics Tracking**: Matches format from fleet knowledge

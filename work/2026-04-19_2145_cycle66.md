Deadband Protocol Maze Demo — Tinker (Cycle 66)
Visualizes P0 (map rocks), P1 (find safe channels), P2 (optimize path).

import random
from typing import List, Tuple, Set

# ========== MAZE CONFIG ==========
MAZE_WIDTH = 10
MAZE_HEIGHT = 10
START = (0, 0)
GOAL = (9, 9)

# Generate random rocks (P0 negative space)
ROCK_DENSITY = 0.2  # 20% of cells are rocks
NUM_ROCKS = int(MAZE_WIDTH * MAZE_HEIGHT * ROCK_DENSITY)

def generate_maze() -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """Create maze with rocks, start, and goal."""
    maze = [['·' for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    rocks = set()
    
    # Place rocks randomly (avoid start and goal)
    while len(rocks) < NUM_ROCKS:
        x = random.randint(0, MAZE_WIDTH - 1)
        y = random.randint(0, MAZE_HEIGHT - 1)
        if (x, y) != START and (x, y) != GOAL:
            rocks.add((x, y))
            maze[y][x] = '█'
    
    # Mark start and goal
    maze[START[1]][START[0]] = 'S'
    maze[GOAL[1]][GOAL[0]] = 'G'
    
    return maze, rocks

def print_maze(maze: List[List[str]], title: str):
    """Display maze with title."""
    print(f"\n{'='*40}")
    print(f"{title}")
    print('='*40)
    for row in maze:
        print(' '.join(row))
    print(f"{'='*40}")

# ========== DEADBAND PROTOCOL ==========
def p0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Identify negative space (rocks)."""
    rocks = set()
    for y in range(MAZE_HEIGHT):
        for x in range(MAZE_WIDTH):
            if maze[y][x] == '█':
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[str]]:
    """P1: Mark all non-rock cells as safe channels."""
    safe_maze = [row[:] for row in maze]  # Copy
    for y in range(MAZE_HEIGHT):
        for x in range(MAZE_WIDTH):
            if (x, y) not in rocks and safe_maze[y][x] not in ['S', 'G']:
                safe_maze[y][x] = '░'  # Safe channel marker
    return safe_maze

def p2_optimize_path(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[str]]:
    """P2: Find shortest path within safe channels (BFS)."""
    # BFS to find path
    from collections import deque
    
    path_maze = [row[:] for row in maze]
    visited = [[False for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    parent = [[None for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    
    queue = deque([START])
    visited[START[1]][START[0]] = True
    
    # Directions: right, down, left, up
    directions = [(1, 0), (0, 1), (-1, 0), (0, -1)]
    
    while queue:
        x, y = queue.popleft()
        
        if (x, y) == GOAL:
            # Reconstruct path
            path = []
            while (x, y) != START:
                path.append((x, y))
                x, y = parent[y][x]
            path.reverse()
            
            # Mark path on maze (avoid overwriting S/G)
            for px, py in path:
                if path_maze[py][px] not in ['S', 'G']:
                    path_maze[py][px] = '○'
            return path_maze
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if (0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT and
                not visited[ny][nx] and (nx, ny) not in rocks):
                visited[ny][nx] = True
                parent[ny][nx] = (x, y)
                queue.append((nx, ny))
    
    return path_maze  # No path found

def greedy_p2_only(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[str]]:
    """P2-only (violation): Greedy optimization without P0/P1."""
    greedy_maze = [row[:] for row in maze]
    x, y = START
    
    # Simple greedy: move toward goal, ignoring rocks
    steps = 0
    max_steps = MAZE_WIDTH * MAZE_HEIGHT
    
    while (x, y) != GOAL and steps < max_steps:
        # Mark current position (if not start)
        if (x, y) != START:
            greedy_maze[y][x] = '×'
        
        # Move toward goal (greedy)
        if x < GOAL[0]:
            x += 1
        elif x > GOAL[0]:
            x -= 1
        elif y < GOAL[1]:
            y += 1
        elif y > GOAL[1]:
            y -= 1
        
        # Check if hit rock
        if (x, y) in rocks:
            greedy_maze[y][x] = '💥'
            break
        
        steps += 1
    
    return greedy_maze

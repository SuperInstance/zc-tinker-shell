#!/usr/bin/env python3
"""
Deadband Protocol Maze Demonstration
Implements P0→P1→P2 priority order in a 20×20 maze navigation problem.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set
from collections import deque

class DeadbandMaze:
    """Demonstrates Deadband Protocol in maze navigation."""
    
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        total_cells = width * height
        num_rocks = int(total_cells * rock_density)
        
        # Avoid start and goal positions
        possible_positions = [(x, y) for x in range(width) for y in range(height)]
        possible_positions.remove(self.start)
        possible_positions.remove(self.goal)
        
        self.rocks = set(random.sample(possible_positions, num_rocks))
        
        # Initialize agent state
        self.position = self.start
        self.path = [self.start]
        self.phase = "P0: Mapping negative space"
        
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify all states where failure probability > threshold."""
        print("\n" + "="*60)
        print("P0: MAPPING NEGATIVE SPACE")
        print("="*60)
        print(f"Identified {len(self.rocks)} rock positions (failure states)")
        
        # In real implementation, this would involve exploration
        # For demo, we assume perfect knowledge of rocks
        return self.rocks
    
    def p1_find_safe_channels(self, rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
        """P1: Find all paths where no position is in negative space."""
        print("\n" + "="*60)
        print("P1: FINDING SAFE CHANNELS")
        print("="*60)
        
        # BFS to find safe paths
        queue = deque([(self.start, [self.start])])
        visited = {self.start}
        safe_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            # Check all four directions
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                new_pos = (nx, ny)
                
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    new_pos not in rocks and new_pos not in visited):
                    visited.add(new_pos)
                    queue.append((new_pos, path + [new_pos]))
        
        print(f"Found {len(safe_paths)} safe channels from start to goal")
        if safe_paths:
            print(f"Shortest safe path: {len(safe_paths[0])} steps")
            print(f"Longest safe path: {len(safe_paths[-1])} steps")
        
        return safe_paths
    
    def p2_optimize_within_channel(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """P2: Select optimal path from safe channels."""
        print("\n" + "="*60)
        print("P2: OPTIMIZING WITHIN SAFE CHANNELS")
        print("="*60)
        
        if not safe_paths:
            print("ERROR: No safe channels found! Cannot optimize.")
            return []
        
        # Select shortest path (minimum cost)
        optimal_path = min(safe_paths, key=len)
        print(f"Selected optimal path: {len(optimal_path)} steps")
        return optimal_path
    
    def greedy_unconstrained(self) -> Tuple[bool, int]:
        """Simulates P2-only (greedy) approach for comparison."""
        print("\n" + "="*60)
        print("GREEDY UNCONSTRAINED (P2-ONLY) APPROACH")
        print("="*60)
        
        pos = self.start
        path = [pos]
        steps = 0
        
        while pos != self.goal and steps < 1000:
            # Greedy: move toward goal without checking rocks
            x, y = pos
            gx, gy = self.goal
            
            # Determine direction (prioritize reducing larger distance)
            if abs(gx - x) > abs(gy - y):
                nx = x + (1 if gx > x else -1)
                ny = y
            else:
                nx = x
                ny = y + (1 if gy > y else -1)
            
            # Boundary check
            nx = max(0, min(nx, self.width-1))
            ny = max(0, min(ny, self.height-1))
            
            new_pos = (nx, ny)
            
            # Check for rock hit
            if new_pos in self.rocks:
                print(f"  Hit rock at {new_pos} after {steps} steps!")
                return False, steps
            
            pos = new_pos
            path.append(pos)
            steps += 1
        
        success = (pos == self.goal)
        print(f"  Result: {'SUCCESS' if success else 'FAILURE'} after {steps} steps")
        return success, steps
    
    def display_maze(self, path: List[Tuple[int, int]] = None):
        """Display ASCII maze with rocks, start, goal, and optional path."""
        print("\n" + "="*60)
        print("MAZE VISUALIZATION")
        print("="*60)
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')
                elif path and pos in path:
                    row.append('·')
                else:
                    row.append('.')
            print(' '.join(row))
        
        print("\nLegend: S=Start, G=Goal, █=Rock, ·=Path, .=Empty")
        print(f"Rocks: {len(self.rocks)} ({len(self.rocks)/(self.width*self.height):.1%} of cells)")
    
    def run_deadband_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2."""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*60)
        print("Based on constraint_sim_final.json results:")
        print("  - Deadband success: 50/50")
        print("  - Constrained success: 0/50")
        print("  - Unconstrained success: 33/50")
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        
        # P1: Find safe channels
        safe_paths = self.p1_find_safe_channels(rocks)
        
        if not safe_paths:
            print("\nABORT: No safe channels exist. Protocol complete.")
            return
        
        # P2: Optimize within channel
        optimal_path = self.p2_optimize_within_channel(safe_paths)
        
        # Display results
        self.display_maze(optimal_path)
        
        print("\n" + "="*60)
        print("PROTOCOL COMPLETE")
        print("="*60)
        print(f"Start: {self.start}")
        print(f"Goal: {self.goal}")
        print(f"Optimal path length: {len(optimal_path)} steps")
        print(f"Rocks avoided: {len(rocks)}")
        
        return optimal_path

def main():
    """Run the demonstration."""
    print("Deadband Protocol Maze Demo")
    print("="*60)
    
    # Create maze with same parameters as constraint theory experiments
    maze = DeadbandMaze(width=20, height=20, rock_density=0.16)
    
    # Show initial maze state
    maze.display_maze()
    
    # Run greedy approach (P2-only) for comparison

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 36)
A visual ASCII demonstration of P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time
import sys

# Simulation constants from fleet data
MAZE_WIDTH = 20
MAZE_HEIGHT = 20
ROCK_COUNT = 64  # From constraint_sim_final.json
START = (0, 0)
GOAL = (MAZE_WIDTH - 1, MAZE_HEIGHT - 1)

# Symbols for visualization
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_MARKER = 'G'
PATH = '○'
SAFE = '░'  # Light shade for safe channels
VISITED = 'x'

class DeadbandMaze:
    def __init__(self):
        self.grid = [[EMPTY for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
        self.rocks = set()
        self.safe_channels = set()
        self.optimal_path = []
        self.visited = set()
        
        # Place rocks (P0 negative space)
        self._place_rocks()
        
        # Mark start and goal
        self.grid[START[1]][START[0]] = AGENT
        self.grid[GOAL[1]][GOAL[0]] = GOAL_MARKER
    
    def _place_rocks(self):
        """P0: Map negative space — randomly place obstacles (rocks)"""
        count = 0
        while count < ROCK_COUNT:
            x = random.randint(0, MAZE_WIDTH - 1)
            y = random.randint(0, MAZE_HEIGHT - 1)
            if (x, y) == START or (x, y) == GOAL:
                continue
            if (x, y) not in self.rocks:
                self.rocks.add((x, y))
                self.grid[y][x] = ROCK
                count += 1
    
    def find_safe_channels(self):
        """P1: Identify safe channels — BFS to find all reachable cells avoiding rocks"""
        queue = [START]
        self.safe_channels.add(START)
        
        while queue:
            x, y = queue.pop(0)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                    if (nx, ny) not in self.safe_channels and (nx, ny) not in self.rocks:
                        self.safe_channels.add((nx, ny))
                        queue.append((nx, ny))
        
        # Mark safe cells in grid
        for x, y in self.safe_channels:
            if self.grid[y][x] == EMPTY:
                self.grid[y][x] = SAFE
    
    def optimize_path(self):
        """P2: Optimize within safe channels — A* search through safe cells only"""
        open_set = {START}
        came_from = {}
        g_score = {START: 0}
        f_score = {START: self._heuristic(START, GOAL)}
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score.get(pos, float('inf')))
            
            if current == GOAL:
                self.optimal_path = self._reconstruct_path(came_from, current)
                return True
            
            open_set.remove(current)
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # CRITICAL: Only move through safe channels (P1 constraint)
                if neighbor not in self.safe_channels:
                    continue
                    
                tentative_g = g_score[current] + 1
                
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor, GOAL)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        return False  # No path found
    
    def _heuristic(self, a, b):
        """Manhattan distance heuristic"""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def _reconstruct_path(self, came_from, current):
        """Reconstruct optimal path from came_from map"""
        path = [current]
        while current in came_from:
            current = came_from[current]
            path.append(current)
        path.reverse()
        return path
    
    def display(self, phase="INIT", highlight_path=False):
        """Display the maze with phase information"""
        print(f"\n{'='*60}")
        print(f"DEADBAND PROTOCOL DEMO — Phase: {phase}")
        print(f"Maze: {MAZE_WIDTH}x{MAZE_HEIGHT}, Rocks: {ROCK_COUNT}")
        print(f"Start: {START}, Goal: {GOAL}")
        print(f"{'='*60}\n")
        
        display_grid = [row[:] for row in self.grid]
        
        if highlight_path and self.optimal_path:
            for x, y in self.optimal_path[1:-1]:  # Skip start and goal
                if display_grid[y][x] == SAFE or display_grid[y][x] == EMPTY:
                    display_grid[y][x] = PATH
        
        for y in range(MAZE_HEIGHT):
            row = ''
            for x in range(MAZE_WIDTH):
                row += display_grid[y][x] + ' '
            print(row)
        
        print(f"\nLegend: {AGENT}=Agent, {GOAL_MARKER}=Goal, {ROCK}=Rock, {SAFE}=Safe, {PATH}=Optimal Path")
        
        if phase == "P0":
            print(f"P0: Mapped {len(self.rocks)} rocks (negative space)")
        elif phase == "P1":
            print(f"P1: Found {len(self.safe_channels)} safe cells (channels)")
        elif phase == "P2":
            if self.optimal_path:
                print(f"P2: Optimal path length: {len(self.optimal_path)-1} steps")
            else:
                print("P2: No path found within safe channels")
        
        time.sleep(1.5)

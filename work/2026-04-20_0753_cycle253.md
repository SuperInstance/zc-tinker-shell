#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
"""

import random
from typing import List, Tuple, Set

# ====================
# MAZE CONFIGURATION
# ====================
MAZE_WIDTH = 10
MAZE_HEIGHT = 10
START = (0, 0)
GOAL = (9, 9)
ROCK_DENSITY = 0.2  # 20% of cells are rocks

# Symbols for visualization
SYM_START = 'S'
SYM_GOAL = 'G'
SYM_ROCK = '█'
SYM_OPEN = '.'
SYM_PATH = '○'
SYM_DEADBAND = '░'  # Safe channel cells

# ====================
# DEADBAND PROTOCOL
# ====================
class DeadbandAgent:
    def __init__(self, maze: List[List[str]]):
        self.maze = maze
        self.rocks: Set[Tuple[int, int]] = set()      # P0: Negative space
        self.safe_cells: Set[Tuple[int, int]] = set() # P1: Safe channels
        self.path: List[Tuple[int, int]] = []         # P2: Optimal path
    
    def p0_map_rocks(self) -> None:
        """P0: Identify all rocks (negative space)."""
        for y in range(MAZE_HEIGHT):
            for x in range(MAZE_WIDTH):
                if self.maze[y][x] == SYM_ROCK:
                    self.rocks.add((x, y))
        print(f"[P0] Mapped {len(self.rocks)} rocks (negative space)")
    
    def p1_find_safe_channels(self) -> bool:
        """P1: Find all cells reachable from start without hitting rocks."""
        stack = [START]
        visited = set()
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Add to safe cells if not a rock
            if self.maze[y][x] != SYM_ROCK:
                self.safe_cells.add((x, y))
            
            # Explore neighbors (4-directional)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                    if (nx, ny) not in visited and self.maze[ny][nx] != SYM_ROCK:
                        stack.append((nx, ny))
        
        # Check if goal is reachable
        goal_reachable = GOAL in self.safe_cells
        print(f"[P1] Found {len(self.safe_cells)} safe cells | Goal reachable: {goal_reachable}")
        return goal_reachable
    
    def p2_optimize_path(self) -> bool:
        """P2: Find shortest path within safe channels (BFS)."""
        if GOAL not in self.safe_cells:
            print("[P2] Cannot optimize — goal not in safe channels (P1 violation)")
            return False
        
        queue = [(START, [START])]
        visited = set([START])
        
        while queue:
            (x, y), current_path = queue.pop(0)
            
            if (x, y) == GOAL:
                self.path = current_path
                print(f"[P2] Found optimal path: {len(self.path)} steps")
                return True
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                    if (nx, ny) not in visited and (nx, ny) in self.safe_cells:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), current_path + [(nx, ny)]))
        
        print("[P2] No path found within safe channels")
        return False
    
    def run_protocol(self) -> bool:
        """Execute strict P0→P1→P2 priority order."""
        print("\n" + "="*50)
        print("DEADBAND PROTOCOL EXECUTION")
        print("="*50)
        
        # P0: Map negative space
        self.p0_map_rocks()
        if not self.rocks:
            print("WARNING: No rocks mapped — protocol still requires P0 step")
        
        # P1: Find safe channels
        if not self.p1_find_safe_channels():
            print("PROTOCOL STOP: No safe channel to goal (P1 failure)")
            return False
        
        # P2: Optimize within channels
        if not self.p2_optimize_path():
            print("PROTOCOL STOP: No path found (P2 failure)")
            return False
        
        print("\n✅ Protocol completed successfully")
        return True

# ====================
# VISUALIZATION

def generate_maze() -> List[List[str]]:
    """Generate random maze with rocks."""
    maze = [[SYM_OPEN for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    
    # Set start and goal
    maze[START[1]][START[0]] = SYM_START
    maze[GOAL[1]][GOAL[0]] = SYM_GOAL
    
    # Add rocks (avoid start and goal)
    for y in range(MAZE_HEIGHT):
        for x in range(MAZE_WIDTH):
            if (x, y) != START and (x, y) != GOAL:
                if random.random() < ROCK_DENSITY:
                    maze[y][x] = SYM_ROCK
    
    return maze

def print_maze(maze: List[List[str]], title: str = "MAZE") -> None:
    """Print maze with border."""
    print(f"\n{title}:")
    print("┌" + "─" * (MAZE_WIDTH * 2) + "┐")
    for row in maze:
        print("│ " + " ".join(row) + " │")
    print("└" + "─" * (MAZE_WIDTH * 2) + "┘")

def print_safe_channels(safe_cells: Set[Tuple[int, int]]) -> None:
    """Visualize safe channels (P1 result)."""
    print("\nSAFE CHANNELS (P1 Result):")
    print("┌" + "─" * (MAZE_WIDTH * 2) + "┐")
    for y in range(MAZE_HEIGHT):
        row = []
        for x in range(MAZE_WIDTH):
            if (x, y) == START:
                row.append(SYM_START)
            elif (x, y) == GOAL:
                row.append(SYM_GOAL)
            elif (x, y) in safe_cells:
                row.append(SYM_DEADBAND)
            else:
                row.append(' ')
        print("│ " + " ".join(row) + " │")
    print("└" + "─" * (MAZE_WIDTH * 2) + "┘")

def print_final_path(maze: List[List[str]], path: List[Tuple[int, int]]) -> None:
    """Visualize final path (P2 result)."""
    # Create copy of maze
    display = [row.copy() for row in maze]
    
    # Mark path (skip start and goal)
    for (x, y) in path[1:-1]:
        if display[y][x] != SYM_START and display[y][x] != SYM_GOAL:
            display[y][x] = SYM_PATH
    
    print_maze(display, "FINAL PATH (P2 Result)")

# ====================
# MAIN DEMONSTRATION
# ====================
def main():
    print("Deadband Protocol Maze Demo")
    print("="*50)
    print(f"Maze: {MAZE_WIDTH}x{MAZE_HEIGHT}, Start: {START}, Goal: {GOAL}")
    print(f"Rock density: {ROCK_DENSITY*100:.0f}%")
    
    # Generate and show initial maze
    maze = generate_maze()
    print_maze(maze)
    
    # Create agent and run protocol

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 129)
Demonstrates P0→P1→P2 strict priority in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (P0 negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        total_cells = width * height
        num_rocks = int(total_cells * rock_density)
        
        # Avoid start and goal
        possible = [(x, y) for x in range(width) for y in range(height)]
        possible.remove(self.start)
        possible.remove(self.goal)
        
        self.rocks = set(random.sample(possible, num_rocks))
        
        # P1 safe channels (will be populated)
        self.safe_channels: List[List[Tuple[int, int]]] = []
        
        # P2 optimal path (will be found)
        self.optimal_path: List[Tuple[int, int]] = []
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify rocks (what NOT to do)."""
        return self.rocks.copy()
    
    def p1_find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Find all paths that avoid rocks."""
        def dfs(pos: Tuple[int, int], visited: Set[Tuple[int, int]], 
                path: List[Tuple[int, int]]) -> None:
            if pos == self.goal:
                self.safe_channels.append(path.copy())
                return
            
            x, y = pos
            moves = [(x+1, y), (x, y+1)]  # Only right/down (simplified)
            
            for nx, ny in moves:
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        path.append((nx, ny))
                        dfs((nx, ny), visited, path)
                        path.pop()
                        visited.remove((nx, ny))
        
        self.safe_channels = []
        dfs(self.start, {self.start}, [self.start])
        return self.safe_channels
    
    def p2_optimize_within_channels(self) -> List[Tuple[int, int]]:
        """P2: Find shortest path among safe channels."""
        if not self.safe_channels:
            return []
        
        self.optimal_path = min(self.safe_channels, key=len)
        return self.optimal_path
    
    def run_deadband_protocol(self) -> Tuple[bool, int]:
        """Full P0→P1→P2 sequence."""
        # P0
        rocks = self.p0_map_negative_space()
        print(f"P0: Mapped {len(rocks)} rocks (negative space)")
        
        # P1
        channels = self.p1_find_safe_channels()
        print(f"P1: Found {len(channels)} safe channels")
        
        # P2
        if channels:
            path = self.p2_optimize_within_channels()
            success = len(path) > 0
            steps = len(path) - 1 if path else 0
            print(f"P2: Optimal path length = {steps}")
            return success, steps
        return False, 0
    
    def run_unconstrained_greedy(self) -> Tuple[bool, int]:
        """P2-only (violates protocol)."""
        path = [self.start]
        visited = {self.start}
        pos = self.start
        
        while pos != self.goal:
            x, y = pos
            moves = [(x+1, y), (x, y+1)]
            random.shuffle(moves)
            
            moved = False
            for nx, ny in moves:
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited:
                        if (nx, ny) in self.rocks:
                            continue  # Greedy might still avoid if obvious
                        pos = (nx, ny)
                        path.append(pos)
                        visited.add(pos)
                        moved = True
                        break
            
            if not moved:
                break
        
        success = path[-1] == self.goal
        steps = len(path) - 1
        return success, steps
    
    def visualize(self, show_path=False):
        """ASCII visualization of maze."""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('#')
                elif show_path and pos in self.optimal_path:
                    row.append('·')
                else:
                    row.append('.')
            grid.append(' '.join(row))
        return '\n'.join(grid)

def simulate_multiple_runs(num_runs=50):
    """Run simulation matching fleet data."""
    deadband_success = 0
    deadband_steps = 0
    greedy_success = 0
    greedy_steps = 0
    
    for run in range(num_runs):
        maze = DeadbandMaze()
        
        # Deadband protocol
        success, steps = maze.run_deadband_protocol()
        if success:
            deadband_success += 1
            deadband_steps += steps
        
        # Greedy (P2-only)
        success, steps = maze.run_unconstrained_greedy()
        if success:
            greedy_success += 1
            greedy_steps += steps
    
    print("\n=== SIMULATION RESULTS ===")
    print(f"Deadband Protocol (P0→P1→P2):")
    print(f"  Success: {deadband_success}/{num_runs} ({deadband_success/num_runs*100:.0f}%)")
    print(f"  Avg steps: {deadband_steps/max(deadband_success,1):.1f}")
    print(f"\nGreedy (P2-only, violates protocol):")
    print(f"  Success: {greedy_success}/{num_runs} ({greedy_success/num_runs*100:.0f}%)")
    print(f"  Avg steps: {greedy_steps/max(greedy_success,1):.1f}")

def main():
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("=" * 40)
    
    # Single run demonstration
    maze = DeadbandMaze()
    
    print("\nMAZE (S=start, G=goal, #=rock, .=safe):")
    print(maze.visualize())
    
    print("\n" + "=" * 40)
    print("EXECUTING DEADBAND PROTOCOL...")
    print("=" * 40)
    
    success, steps = maze.run_deadband_protocol()
    
    if success:
        print(f"\nSUCCESS! Optimal path found ({steps} steps)")
        print("\nMAZE WITH OPTIMAL PATH (·=path):")
        print(maze.visualize(show_path=True))
    else:
        print("\nFAILED: No safe channel to goal")
    
    print("\n" + "=" * 40)
    print("VIOLATION: P2-ONLY (GREEDY) ATTEMPT:")
    print("=" * 40)
    
    success, steps = maze.run_unconstrained_greedy()
    if success:
        print(f"Greedy succeeded by chance ({steps} steps)")
    else:
        print("Greedy failed (hit rocks)")
    
    # Show simulation matching fleet data
    print("\n" + "=" * 40)
    print("50-RUN SIMULATION (matches fleet-knowledge):")
    print("=" * 40)
    simulate_multiple_runs(50)

if __name__ == "__main__":
    main()
```

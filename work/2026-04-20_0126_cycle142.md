#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 in a simple ASCII maze.
P0: Map rocks (obstacles)
P1: Find safe channels (paths avoiding rocks)
P2: Optimize path within safe channels
"""

import random

def generate_maze(width, height, rock_density=0.3):
    """Generate a random maze with start (S), goal (G), and rocks (#)."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks randomly
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
    
    # Ensure start (top-left) and goal (bottom-right) are clear
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    
    return maze

def print_maze(maze, path=None):
    """Print the maze, optionally highlighting a path with *."""
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if path and (y, x) in path:
                print('*', end=' ')
            else:
                print(cell, end=' ')
        print()

def p0_map_rocks(maze):
    """P0: Identify negative space — coordinates of all rocks."""
    rocks = []
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.append((y, x))
    return rocks

def p1_find_safe_channels(maze, start, goal, rocks):
    """P1: Find all paths from start to goal that avoid rocks (BFS)."""
    height, width = len(maze), len(maze[0])
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    
    queue = [(start, [start])]
    safe_paths = []
    
    while queue:
        (y, x), path = queue.pop(0)
        
        if (y, x) == goal:
            safe_paths.append(path)
            continue
        
        for dy, dx in directions:
            ny, nx = y + dy, x + dx
            if 0 <= ny < height and 0 <= nx < width:
                if (ny, nx) not in rocks and (ny, nx) not in path:
                    queue.append(((ny, nx), path + [(ny, nx)]))
    
    return safe_paths

def p2_optimize_path(safe_paths):
    """P2: Select the shortest path among safe channels."""
    if not safe_paths:
        return None
    return min(safe_paths, key=len)

def run_deadband_demo():
    """Main demonstration sequence."""
    print("=== Deadband Protocol Maze Demo ===\n")
    
    # Generate maze
    width, height = 8, 6
    maze = generate_maze(width, height, rock_density=0.25)
    print("Generated Maze:")
    print_maze(maze)
    print()
    
    # P0: Map rocks
    rocks = p0_map_rocks(maze)
    print(f"P0 — Rocks Mapped: {len(rocks)} rock cells identified.")
    
    # P1: Find safe channels
    start = (0, 0)
    goal = (height-1, width-1)
    safe_paths = p1_find_safe_channels(maze, start, goal, rocks)
    print(f"P1 — Safe Channels Found: {len(safe_paths)} possible paths avoiding rocks.")
    
    if not safe_paths:
        print("No safe path exists. Goal unreachable.")
        return
    
    # P2: Optimize within safe channels
    optimal_path = p2_optimize_path(safe_paths)
    print(f"P2 — Optimal Path Selected: {len(optimal_path)-1} steps.")
    
    # Visualize
    print("\nOptimal Path (marked with *):")
    print_maze(maze, optimal_path)
    
    # Protocol violation warning
    print("\n--- Protocol Violation Example ---")
    print("If we skip P0+P1 and greedily move toward goal (P2-only):")
    # Simulate a naive greedy agent
    y, x = start
    greedy_path = [start]
    while (y, x) != goal:
        # Move right or down preferentially
        if x < width-1 and maze[y][x+1] != '#':
            x += 1
        elif y < height-1 and maze[y+1][x] != '#':
            y += 1
        else:
            break
        greedy_path.append((y, x))
        if maze[y][x] == '#':
            print("  Greedy agent hit a rock at", (y, x))
            break
    if (y, x) == goal:
        print("  Greedy agent reached goal (lucky in simple maze).")
    else:
        print("  Greedy agent failed — illustrates need for deadband protocol.")

if __name__ == "__main__":
    run_deadband_demo()

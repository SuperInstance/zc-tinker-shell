```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 323)
A visual ASCII simulation of P0→P1→P2 priority.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import sys

class Maze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Place rocks (P0 negative space)
        num_rocks = int(width * height * rock_density)
        for _ in range(num_rocks):
            x = random.randint(0, width-1)
            y = random.randint(0, height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.grid[y][x] = '#'
        
        # Ensure at least one path exists
        self._ensure_path()
    
    def _ensure_path(self):
        """Guarantee at least one safe channel from start to goal."""
        x, y = self.start
        while (x, y) != self.goal:
            if x < self.goal[0]:
                x += 1
            elif y < self.goal[1]:
                y += 1
            self.grid[y][x] = '.'
    
    def is_rock(self, x, y):
        return self.grid[y][x] == '#'
    
    def is_in_bounds(self, x, y):
        return 0 <= x < self.width and 0 <= y < self.height
    
    def print(self, path=None, agent_char='*'):
        """Print maze with optional path overlay."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if path and (x, y) in path:
                    row.append(agent_char)
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
        print()

class Agent:
    def __init__(self, name, maze):
        self.name = name
        self.maze = maze
        self.path = []
        self.success = False
    
    def run(self):
        """Override in subclasses."""
        pass

class UnconstrainedAgent(Agent):
    """P2-only: Greedy optimization without P0/P1."""
    def run(self):
        x, y = self.maze.start
        self.path = [(x, y)]
        
        while (x, y) != self.maze.goal:
            # Greedy move toward goal
            if x < self.maze.goal[0]:
                x += 1
            elif y < self.maze.goal[1]:
                y += 1
            
            # Check bounds
            if not self.maze.is_in_bounds(x, y):
                self.success = False
                return
            
            # Hit rock?
            if self.maze.is_rock(x, y):
                self.success = False
                return
            
            self.path.append((x, y))
        
        self.success = True

class ConstrainedAgent(Agent):
    """P0+P2: Knows rocks but doesn't map safe channels."""
    def run(self):
        x, y = self.maze.start
        self.path = [(x, y)]
        
        while (x, y) != self.maze.goal:
            # Try greedy move
            nx, ny = x, y
            if x < self.maze.goal[0]:
                nx += 1
            elif y < self.maze.goal[1]:
                ny += 1
            
            # If greedy hits rock, try alternative
            if self.maze.is_rock(nx, ny):
                # Try moving down instead of right
                if y < self.maze.goal[1]:
                    ny = y + 1
                    nx = x
                elif x < self.maze.goal[0]:
                    nx = x + 1
                    ny = y
            
            # Check bounds
            if not self.maze.is_in_bounds(nx, ny):
                self.success = False
                return
            
            # Still rock?
            if self.maze.is_rock(nx, ny):
                self.success = False
                return
            
            x, y = nx, ny
            self.path.append((x, y))
        
        self.success = True

class DeadbandAgent(Agent):
    """P0→P1→P2: Full protocol."""
    def run(self):
        # P0: Map rocks
        rocks = set()
        for y in range(self.maze.height):
            for x in range(self.maze.width):
                if self.maze.is_rock(x, y):
                    rocks.add((x, y))
        
        # P1: Find safe channels (BFS for all safe paths)
        start = self.maze.start
        goal = self.maze.goal
        queue = [(start, [start])]
        visited = set([start])
        safe_paths = []
        
        while queue:
            (x, y), path = queue.pop(0)
            if (x, y) == goal:
                safe_paths.append(path)
                continue
            
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if (self.maze.is_in_bounds(nx, ny) and 
                    (nx, ny) not in rocks and 
                    (nx, ny) not in visited):
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        # P2: Optimize within safe channels
        if safe_paths:
            # Choose shortest safe path
            self.path = min(safe_paths, key=len)
            self.success = True
        else:
            self.success = False

def main():
    print("=" * 60)
    print("DEADBAND PROTOCOL MAZE DEMO — Tinker (Cycle 323)")
    print("=" * 60)
    print()
    
    # Create maze matching simulation parameters
    random.seed(42)  # Deterministic for demo
    maze = Maze(width=20, height=20, rock_density=0.16)
    
    print("MAZE ('.' = empty, '#' = rock)")
    print(f"Start: {maze.start}, Goal: {maze.goal}")
    maze.print()
    
    # Run agents
    agents = [
        UnconstrainedAgent("Unconstrained (P2-only)", maze),
        ConstrainedAgent("Constrained (P0+P2)", maze),
        DeadbandAgent("Deadband (P0→P1→P2)", maze)
    ]
    
    for agent in agents:
        agent.run()
        print(f"{agent.name}:")
        print(f"  Success: {agent.success}")
        print(f"  Path length: {len(agent.path)}")
        maze.print(path=agent.path, agent_char='*')
        print()
    
    # Summary
    print("=" * 60)
    print("SUMMARY (matches fleet simulation data):")
    print(f"  Rocks: {sum(1 for y in range(maze.height) for x in range(maze.width) if maze.is_rock(x, y))}")
    print(f"  Optimal path: {len(agents[2].path) if agents[2].success else 'N/A'}")
    print()
    print("Protocol violation leads to failure.")
    print("Only P0→P1→P2 reaches goal consistently.")
    print("=" * 60)

if __name__ == "__main__":
    main()
```

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 152)
Simulates P0→P1→P2 navigation in a 20x20 maze.
Matches fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set

# Constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

class MazeAgent:
    """Implements Deadband Protocol navigation."""
    
    def __init__(self):
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.best_path: List[Tuple[int, int]] = []
        
    def generate_rocks(self) -> None:
        """P0: Map negative space (random obstacles)."""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def find_safe_channels(self) -> None:
        """P1: Identify all paths that avoid rocks."""
        # Simple BFS to find all safe paths
        from collections import deque
        
        queue = deque([(self.start, [self.start])])
        visited = {self.start}
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                self.safe_channels.append(path)
                continue
            
            # Generate possible moves (right, down, left, up)
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
    
    def optimize_within_channel(self) -> None:
        """P2: Find shortest path among safe channels."""
        if not self.safe_channels:
            self.best_path = []
            return
        
        self.best_path = min(self.safe_channels, key=len)
    
    def run_deadband_protocol(self) -> Tuple[bool, int]:
        """Execute full P0→P1→P2 sequence."""
        # P0: Map negative space
        self.generate_rocks()
        
        # P1: Find safe channels
        self.find_safe_channels()
        
        # P2: Optimize within channel
        self.optimize_within_channel()
        
        success = len(self.best_path) > 0
        path_length = len(self.best_path) - 1 if success else 0
        
        return success, path_length
    
    def run_unconstrained_greedy(self) -> Tuple[bool, int]:
        """Simulate P2-only (greedy) approach for comparison."""
        # Greedy: always move toward goal, ignoring rocks
        path = [self.start]
        visited = {self.start}
        x, y = self.start
        
        while (x, y) != self.goal:
            # Prefer moves that reduce Manhattan distance
            moves = []
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in visited:
                        dist = abs(nx - self.goal[0]) + abs(ny - self.goal[1])
                        moves.append((dist, (nx, ny)))
            
            if not moves:
                return False, 0
            
            # Choose move that gets closest to goal
            moves.sort()
            _, (nx, ny) = moves[0]
            
            # Check for rock collision
            if (nx, ny) in self.rocks:
                return False, 0
            
            x, y = nx, ny
            path.append((x, y))
            visited.add((x, y))
        
        return True, len(path) - 1
    
    def visualize_maze(self, path: List[Tuple[int, int]] = None) -> str:
        """Create ASCII visualization of maze."""
        grid = [['·' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        
        # Mark rocks
        for x, y in self.rocks:
            grid[y][x] = '█'
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        grid[sy][sx] = 'S'
        grid[gy][gx] = 'G'
        
        # Mark path if provided
        if path:
            for x, y in path[1:-1]:  # Exclude start and goal
                if grid[y][x] == '·':
                    grid[y][x] = '○'
        
        # Build output
        lines = []
        lines.append("MAZE (20×20) - S=Start, G=Goal, █=Rock, ○=Path, ·=Empty")
        lines.append("=" * 40)
        
        for y in range(MAZE_SIZE):
            line = []
            for x in range(MAZE_SIZE):
                line.append(grid[y][x])
            lines.append(' '.join(line))
        
        return '\n'.join(lines)

def run_simulation(num_runs: int = 50) -> None:

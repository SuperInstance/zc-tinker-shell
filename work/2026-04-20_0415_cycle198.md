**Status:** Based on the provided fleet knowledge, I'll create a script that:

2. **P1:** Identifies safe channels (deadbands) where the agent can move.

4. Visualizes each step with ASCII output.

I'll use the simulation data from `constraint_sim_final.json` as a reference for maze dimensions and rock density.

## File: `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 strict priority in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque

# Constants from simulation data
MAZE_SIZE = 20
ROCK_COUNT = 64  # matches "rocks": 64 in JSON
START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

# Symbols for visualization
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_SYM = 'G'
PATH = '○'
SAFE = '░'  # visual marker for safe channels

def generate_maze(size, rocks):
    """Create a maze with random rocks (negative space)."""
    maze = [[EMPTY for _ in range(size)] for _ in range(size)]
    placed = 0
    while placed < rocks:
        x = random.randint(0, size - 1)
        y = random.randint(0, size - 1)
        if (x, y) == START or (x, y) == GOAL:
            continue
        if maze[x][y] != ROCK:
            maze[x][y] = ROCK
            placed += 1
    maze[START[0]][START[1]] = AGENT
    maze[GOAL[0]][GOAL[1]] = GOAL_SYM
    return maze

def print_maze(maze, title="Maze"):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def p0_map_negative_space(maze):
    """P0: Identify all rock coordinates (what NOT to do)."""
    rocks = []
    for x in range(len(maze)):
        for y in range(len(maze[0])):
            if maze[x][y] == ROCK:
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, rocks):
    """P1: Find all reachable cells that are not rocks (safe channels)."""
    size = len(maze)
    safe = [[False for _ in range(size)] for _ in range(size)]
    # BFS from start, avoiding rocks
    queue = deque([START])
    visited = set([START])
    while queue:
        x, y = queue.popleft()
        safe[x][y] = True
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < size and 0 <= ny < size:
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    return safe

def p2_optimize_within_channels(maze, safe):
    """P2: Find shortest path from start to goal using only safe cells."""
    size = len(maze)
    if not safe[GOAL[0]][GOAL[1]]:
        return None  # goal unreachable in safe channels
    
    # BFS for shortest path
    queue = deque([(START, [START])])
    visited = set([START])
    while queue:
        (x, y), path = queue.popleft()
        if (x, y) == GOAL:
            return path
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < size and 0 <= ny < size:
                if safe[nx][ny] and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    return None

def visualize_safe_channels(maze, safe):
    """Create a visualization of safe channels (P1 result)."""
    viz = [row[:] for row in maze]
    for x in range(len(maze)):
        for y in range(len(maze[0])):
            if viz[x][y] == EMPTY and safe[x][y]:
                viz[x][y] = SAFE
    return viz

def visualize_path(maze, path):
    """Mark the optimal path on the maze (P2 result)."""
    viz = [row[:] for row in maze]
    for (x, y) in path[1:-1]:  # exclude start and goal
        if viz[x][y] == EMPTY or viz[x][y] == SAFE:
            viz[x][y] = PATH
    return viz

def main():
    print("Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)")
    print("Based on constraint_sim_final.json: 20x20 maze, 64 rocks")
    print("Strict P0→P1→P2 priority\n")
    
    # Generate maze
    maze = generate_maze(MAZE_SIZE, ROCK_COUNT)
    print_maze(maze, "Initial Maze (A=Start, G=Goal, █=Rock)")
    
    # P0: Map negative space
    rocks = p0_map_negative_space(maze)
    print(f"P0 — Mapped negative space: {len(rocks)} rocks identified")
    
    # P1: Find safe channels
    safe = p1_find_safe_channels(maze, rocks)
    safe_viz = visualize_safe_channels(maze, safe)
    print_maze(safe_viz, "P1 — Safe Channels (░=Safe, █=Rock)")
    safe_count = sum(sum(row) for row in safe)
    print(f"P1 — Safe cells reachable from start: {safe_count}/{MAZE_SIZE*MAZE_SIZE}")
    
    # P2: Optimize within channels
    path = p2_optimize_within_channels(maze, safe)
    if path:
        path_viz = visualize_path(maze, path)
        print_maze(path_viz, "P2 — Optimal Path within Safe Channels (○=Path)")
        print(f"P2 — Shortest safe path length: {len(path)-1} steps")
        # Verify no rocks in path
        path_rocks = any(maze[x][y] == ROCK for (x, y) in path)
        print(f"P2 — Path contains rocks? {path_rocks} (should be False)")
    else:
        print("P2 — No safe path to goal exists within channels")
    
    # Summary
    print("\n=== Protocol Summary ===")
    print("P0: Identified what NOT to do (avoid 64 rocks)")
    print("P1: Mapped where we CAN be (safe reachable cells)")
    print("P2: Optimized path constrained to safe channels")
    if path:
        print("Result: Success — optimal path found without hitting rocks")
    else:
        print("Result: Failure — goal unreachable in safe space (local minima avoided)")

if __name__ == "__main__":
    main()
```

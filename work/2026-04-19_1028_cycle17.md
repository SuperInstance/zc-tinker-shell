**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Complete. Script written, tested, and ready for submission as a training tile.

**Approach (as implemented):**

1. **P0 (Map Rocks):** Generate a random 10x10 maze with obstacles ('#') and open cells ('.'). The agent's starting position is marked 'S', goal is 'G'. Rocks are all '#' cells. The script first prints the full maze, explicitly marking the negative space.

2. **P1 (Find Safe Channels):** Perform a breadth-first search (BFS) from start to goal, avoiding all rocks. This finds *all* shortest safe paths. The script prints the number of safe paths found and visualizes one representative safe path with 'o' markers.

3. **P2 (Optimize Within Channel):** From the set of safe paths (all equal length in this BFS case), select one. The script prints the chosen path's coordinates and the step count.

**Key Demonstration:**
- The protocol is followed strictly: P0 → P1 → P2.
- Optimization (P2) only occurs within the pre-mapped safe channels (P1).
- The visual output shows the maze, the rocks, and a safe path.

**Code (deadband_maze_demo.py):**

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo - Tinker
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize within channel).
"""

import random
from collections import deque

def generate_maze(width=10, height=10, rock_density=0.2):
    """Generate a random maze with rocks (#) and open cells (.)."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    rocks = 0
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
                rocks += 1
    # Ensure start (0,0) and goal (width-1, height-1) are open
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    return maze, rocks

def print_maze(maze, path=None):
    """Print the maze, optionally overlaying a path with 'o'."""
    for y, row in enumerate(maze):
        line = []
        for x, cell in enumerate(row):
            if path and (x, y) in path and cell not in ('S', 'G'):
                line.append('o')
            else:
                line.append(cell)
        print(' '.join(line))

def find_safe_channels(maze):
    """P1: BFS to find all shortest safe paths from S to G."""
    width, height = len(maze[0]), len(maze)
    # Locate start and goal
    start = goal = None
    for y in range(height):
        for x in range(width):
            if maze[y][x] == 'S':
                start = (x, y)
            elif maze[y][x] == 'G':
                goal = (x, y)
    if not start or not goal:
        return []

    # BFS
    queue = deque([(start[0], start[1], [])])
    visited = {start: 0}
    shortest_length = None
    safe_paths = []
    directions = [(1,0), (-1,0), (0,1), (0,-1)]

    while queue:
        x, y, path = queue.popleft()
        current_len = len(path)
        if (x, y) == goal:
            if shortest_length is None:
                shortest_length = current_len
            if current_len == shortest_length:
                safe_paths.append(path + [(x, y)])
            continue

        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if maze[ny][nx] == '#':
                    continue  # Avoid rock
                if (nx, ny) in visited and visited[(nx, ny)] < current_len + 1:
                    continue
                visited[(nx, ny)] = current_len + 1
                queue.append((nx, ny, path + [(x, y)]))

    return safe_paths

def main():
    print("=== Deadband Protocol Maze Demo ===\n")
    random.seed(42)  # For reproducibility

    # Generate maze
    maze, rock_count = generate_maze()
    width, height = len(maze[0]), len(maze)

    # P0: Map the negative space (rocks)
    print("P0: MAP ROCKS (Negative Space)")
    print(f"Maze size: {width}x{height}, Rocks: {rock_count}")
    print_maze(maze)
    print()

    # P1: Find safe channels
    print("P1: FIND SAFE CHANNELS")
    safe_paths = find_safe_channels(maze)
    if not safe_paths:
        print("No safe path from S to G. All channels blocked.")
        return
    print(f"Found {len(safe_paths)} safe channel(s) of length {len(safe_paths[0])-1} steps.")
    # Visualize one safe path
    print("Visualizing one safe channel (path marked 'o'):")
    print_maze(maze, safe_paths[0])
    print()

    # P2: Optimize within channel
    print("P2: OPTIMIZE WITHIN CHANNEL")
    # All found paths are already shortest; pick first.
    chosen_path = safe_paths[0]
    print(f"Selected path (coordinates): {chosen_path}")
    print(f"Steps: {len(chosen_path)-1}")
    print("\nProtocol complete: P0→P1→P2 followed.")

if __name__ == "__main__":
    main()
```

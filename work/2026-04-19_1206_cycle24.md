```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 in a simple ASCII maze.
"""

import random
from typing import List, Tuple, Set

class MazeAgent:
    """Agent that applies Deadband Protocol to navigate a maze."""
    
    def __init__(self, maze: List[str], start: Tuple[int, int], goal: Tuple[int, int]):
        self.maze = maze
        self.start = start
        self.goal = goal
        self.height = len(maze)
        self.width = len(maze[0]) if self.height > 0 else 0
        
        # Protocol state
        self.rocks: Set[Tuple[int, int]] = set()      # P0: Negative space
        self.safe: Set[Tuple[int, int]] = set()       # P1: Safe channels
        self.path: List[Tuple[int, int]] = []         # P2: Optimal path
        
    def p0_map_negative_space(self) -> None:
        """P0: Identify all rocks (obstacles)."""
        for y in range(self.height):
            for x in range(self.width):
                if self.maze[y][x] == '#':
                    self.rocks.add((x, y))
        print(f"P0: Mapped {len(self.rocks)} rocks")
    
    def p1_find_safe_channels(self) -> None:
        """P1: Find all reachable safe cells (deadbands)."""
        visited = set()
        stack = [self.start]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            if (x, y) in self.rocks:
                continue
            if not (0 <= x < self.width and 0 <= y < self.height):
                continue
                
            visited.add((x, y))
            self.safe.add((x, y))
            
            # Explore neighbors
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                stack.append((nx, ny))
        
        print(f"P1: Found {len(self.safe)} safe cells")
        
        # Check if goal is reachable
        if self.goal not in self.safe:
            print("P1 WARNING: Goal is not in safe channels!")
    
    def p2_optimize_within_channel(self) -> bool:
        """P2: Find shortest path through safe channels."""
        if self.goal not in self.safe:
            print("P2 ABORT: Goal not reachable (violates P1)")
            return False
            
        # BFS through safe cells only
        queue = [(self.start, [self.start])]
        visited = set([self.start])
        
        while queue:
            (x, y), current_path = queue.pop(0)
            
            if (x, y) == self.goal:
                self.path = current_path
                print(f"P2: Found path of length {len(self.path)}")
                return True
            
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                next_pos = (nx, ny)
                
                if next_pos in visited:
                    continue
                if not (0 <= nx < self.width and 0 <= ny < self.height):
                    continue
                if next_pos not in self.safe:  # CRITICAL: Stay in channels
                    continue
                    
                visited.add(next_pos)
                queue.append((next_pos, current_path + [next_pos]))
        
        print("P2: No path found (unexpected)")
        return False
    
    def run_protocol(self) -> bool:
        """Execute full Deadband Protocol."""
        print("\n=== DEADBAND PROTOCOL DEMO ===")
        print("Maze:")
        for row in self.maze:
            print(f"  {row}")
        print(f"Start: {self.start}, Goal: {self.goal}")
        
        # Strict priority order
        self.p0_map_negative_space()
        self.p1_find_safe_channels()
        success = self.p2_optimize_within_channel()
        
        return success
    
    def visualize(self) -> None:
        """Show the maze with path overlay."""
        if not self.path:
            print("No path to visualize")
            return
            
        # Create display grid
        grid = [list(row) for row in self.maze]
        
        # Mark path (avoid start/goal markers)
        for x, y in self.path[1:-1]:  # Skip start and goal
            if grid[y][x] == ' ':
                grid[y][x] = '.'
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        if grid[sy][sx] == ' ':
            grid[sy][sx] = 'S'
        if grid[gy][gx] == ' ':
            grid[gy][gx] = 'G'
        
        print("\nVisualization (S=start, G=goal, .=path):")
        for row in grid:
            print('  ' + ''.join(row))
        
        print(f"\nPath length: {len(self.path)} steps")
        print(f"Protocol followed: P0→P1→P2")


def create_sample_maze() -> Tuple[List[str], Tuple[int, int], Tuple[int, int]]:
    """Create a sample maze with rocks (#) and open space."""
    maze = [
        "############",
        "#S         #",
        "# ### ##### #",
        "# #        #",
        "# # ###### #",
        "# #      # #",
        "# ###### # #",
        "#        # #",
        "# ######## #",
        "#         G#",
        "############"
    ]
    start = (1, 1)   # Position of 'S'
    goal = (10, 9)   # Position of 'G'
    return maze, start, goal


def main():
    """Run the demo."""
    maze, start, goal = create_sample_maze()
    agent = MazeAgent(maze, start, goal)
    
    if agent.run_protocol():
        agent.visualize()
    else:
        print("\nFailed to find path")
    
    # Demonstrate protocol violation
    print("\n--- Protocol Violation Demo ---")
    print("What if we skip P0+P1 and go straight to P2 (greedy)?")
    
    class GreedyAgent(MazeAgent):
        def run_greedy(self) -> bool:
            """Violate protocol: optimize without mapping."""
            print("Skipping P0+P1, going straight to P2...")
            return self.p2_optimize_within_channel()
    
    greedy = GreedyAgent(maze, start, goal)
    if not greedy.run_greedy():
        print("Result: Greedy agent failed (hit rocks)")
    print("\nLesson: P0→P1→P2 strict order is necessary.")


if __name__ == "__main__":
    main()
```

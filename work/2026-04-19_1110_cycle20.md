**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Complete. Script written, tested, and ready for submission as a training tile.

**Approach (following Deadband Protocol):**
1. **P0 (Map negative space):** Read the simulation data from `constraint_sim_final.json`. Identify that unconstrained greedy optimization (P2-first) fails 100% of the time (0/50 success). The "rocks" are any path that hits an obstacle.
2. **P1 (Find safe channels):** Design an algorithm that first identifies all reachable cells not blocked by obstacles (safe channels), then performs pathfinding only within that space.
3. **P2 (Optimize within channels):** Use A* search constrained to the safe channel to find the shortest path.

**Output:** A self-contained Python script that generates a random maze, applies the Deadband Protocol, and prints an ASCII visualization showing the path.

```python
#!/usr/bin/env python3
"""
deadband_maze_demo.py
Demonstrates the Deadband Protocol (P0→P1→P2) with an ASCII maze.
Tinker - Cocapn Fleet - Cycle 20
"""

import random
import heapq
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.25):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.grid = self._generate_maze(rock_density)
        
    def _generate_maze(self, rock_density: float) -> List[List[str]]:
        """Generate a random maze with rocks ('#') and open cells ('.')."""
        grid = [['.' for _ in range(self.width)] for _ in range(self.height)]
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) in (self.start, self.goal):
                    continue
                if random.random() < rock_density:
                    grid[y][x] = '#'
        return grid
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify all rock positions (negative space)."""
        rocks = set()
        for y in range(self.height):
            for x in range(self.width):
                if self.grid[y][x] == '#':
                    rocks.add((x, y))
        return rocks
    
    def p1_find_safe_channels(self, rocks: Set[Tuple[int, int]]) -> Set[Tuple[int, int]]:
        """P1: Find all reachable cells that are not rocks (safe channels)."""
        safe = set()
        stack = [self.start]
        visited = set()
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Only add if not a rock
            if (x, y) not in rocks:
                safe.add((x, y))
                # Explore neighbors
                for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                    nx, ny = x + dx, y + dy
                    if 0 <= nx < self.width and 0 <= ny < self.height:
                        stack.append((nx, ny))
        return safe
    
    def p2_optimize_in_channel(self, safe: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
        """P2: Find shortest path from start to goal constrained to safe cells."""
        # A* search with Manhattan distance heuristic
        def heuristic(a, b):
            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        open_set = []
        heapq.heappush(open_set, (0, self.start))
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: heuristic(self.start, self.goal)}
        
        while open_set:
            _, current = heapq.heappop(open_set)
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                return path[::-1]
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                if not (0 <= neighbor[0] < self.width and 0 <= neighbor[1] < self.height):
                    continue
                if neighbor not in safe:
                    continue  # Constrained to safe channel
                
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, self.goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
        
        return []  # No path found
    
    def run_deadband_protocol(self) -> Tuple[bool, List[Tuple[int, int]]]:
        """Execute full P0→P1→P2 protocol."""
        rocks = self.p0_map_negative_space()
        safe = self.p1_find_safe_channels(rocks)
        path = self.p2_optimize_in_channel(safe)
        success = len(path) > 0 and path[-1] == self.goal
        return success, path
    
    def visualize(self, path: List[Tuple[int, int]] = None) -> str:
        """Return ASCII representation of maze with optional path."""
        vis = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif path and (x, y) in path:
                    row.append('*')
                else:
                    row.append(self.grid[y][x])
            vis.append(' '.join(row))
        return '\n'.join(vis)

def main():
    """Run demonstration with statistics."""
    print("=" * 60)
    print("DEADBAND PROTOCOL DEMONSTRATION - ASCII MAZE")
    print("=" * 60)
    
    # Run multiple trials to show statistics
    trials = 10
    successes = 0
    total_path_length = 0
    
    for trial in range(trials):
        print(f"\n--- Trial {trial+1}/{trials} ---")
        maze = DeadbandMaze(width=15, height=10, rock_density=0.25)
        
        print("Maze (S=start, G=goal, #=rock, .=open):")
        print(maze.visualize())
        
        success, path = maze.run_deadband_protocol()
        
        if success:
            successes += 1
            total_path_length += len(path) - 1  # Subtract start node
            print(f"\nSafe path found (length {len(path)-1}):")
            print(maze.visualize(path))
        else:
            print("\nNo safe path to goal exists.")
    
    # Statistics
    print("\n" + "=" * 60)
    print("STATISTICS (10 trials, 25% rock density)")
    print(f"Success rate: {successes}/{trials} ({successes/trials*100:.0f}%)")
    if successes > 0:
        print(f"Average path length: {total_path_length/successes:.1f}")
    print("\nProtocol phases:")
    print("P0: Mapped rock locations (negative space)")
    print("P1: Found reachable safe cells (channels)")
    print("P2: Optimized path within safe channels")
    print("=" * 60)

if __name__ == "__main__":
    main()
```

2. Shows both successful and failed mazes (when no safe channel exists)
3. Provides statistics matching the pattern in `constraint_sim_final.json`
4. Visual output helps understand the protocol intuitively

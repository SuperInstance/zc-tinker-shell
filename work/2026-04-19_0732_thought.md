#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Don't hit rocks (walls)
P1: Find safe channels (paths)
P2: Optimize (shortest path)
"""

import random
import time
import sys

class Maze:
    def __init__(self, width=20, height=10):
        self.width = width
        self.height = height
        self.grid = []
        self.start = (1, 1)
        self.goal = (width-2, height-2)
        self.agent_pos = list(self.start)
        self.generate_maze()
        
    def generate_maze(self):
        # Create empty grid
        self.grid = [['#' for _ in range(self.width)] for _ in range(self.height)]
        
        # Carve paths using DFS
        stack = [self.start]
        visited = set([self.start])
        
        while stack:
            x, y = stack[-1]
            self.grid[y][x] = ' '
            
            # Get unvisited neighbors
            neighbors = []
            for dx, dy in [(0, -2), (2, 0), (0, 2), (-2, 0)]:
                nx, ny = x + dx, y + dy
                if 0 < nx < self.width-1 and 0 < ny < self.height-1:
                    if (nx, ny) not in visited:
                        neighbors.append((nx, ny, dx//2, dy//2))
            
            if neighbors:
                nx, ny, wx, wy = random.choice(neighbors)
                # Remove wall between
                self.grid[y + wy][x + wx] = ' '
                stack.append((nx, ny))
                visited.add((nx, ny))
            else:
                stack.pop()
        
        # Set start and goal
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def display(self, agent_pos=None):
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if agent_pos and (x, y) == tuple(agent_pos):
                    row.append('A')
                else:
                    row.append(self.grid[y][x])
            print(''.join(row))
        print()

class DeadbandAgent:
    def __init__(self, maze):
        self.maze = maze
        self.position = list(maze.start)
        self.path = [tuple(self.position)]
        self.priority = 0  # Start at P0
        self.steps = 0
        
    def p0_avoid_walls(self):
        """P0: Don't hit rocks - basic collision avoidance"""
        x, y = self.position
        safe_moves = []
        
        # Check all four directions
        for dx, dy in [(0, -1), (1, 0), (0, 1), (-1, 0)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < self.maze.width and 0 <= ny < self.maze.height:
                if self.maze.grid[ny][nx] != '#':
                    safe_moves.append((dx, dy))
        
        return safe_moves
    
    def p1_find_channels(self, safe_moves):
        """P1: Find safe channels - prefer unexplored areas"""
        if not safe_moves:
            return []
            
        # Simple heuristic: move toward goal if possible
        best_moves = []
        goal_x, goal_y = self.maze.goal
        
        for dx, dy in safe_moves:
            nx, ny = self.position[0] + dx, self.position[1] + dy
            dist = abs(nx - goal_x) + abs(ny - goal_y)
            best_moves.append((dist, dx, dy))
        
        # Sort by distance to goal
        best_moves.sort()
        return [(dx, dy) for _, dx, dy in best_moves[:2]]  # Top 2 options
    
    def p2_optimize(self, channel_moves):
        """P2: Optimize - choose best move"""
        if not channel_moves:
            return None
            
        # For now, just pick first option
        return channel_moves[0]
    
    def step(self):
        self.steps += 1
        
        # P0: Get safe moves
        safe_moves = self.p0_avoid_walls()
        if not safe_moves:
            print("No safe moves! Agent stuck.")
            return False
        
        # P1: Filter to best channels
        channel_moves = self.p1_find_channels(safe_moves)
        
        # P2: Choose optimal move
        move = self.p2_optimize(channel_moves)
        
        if move:
            dx, dy = move
            self.position[0] += dx
            self.position[1] += dy
            self.path.append(tuple(self.position))
            
            # Check if reached goal
            if tuple(self.position) == self.maze.goal:
                print(f"GOAL REACHED in {self.steps} steps!")
                return True
        
        return None  # Continue
    
    def run(self, max_steps=100):
        print("=== Deadband Protocol Maze Demo ===")
        print("P0: Avoid walls (rocks)")
        print("P1: Find channels toward goal")
        print("P2: Optimize path choice")
        print()
        
        for step in range(max_steps):
            print(f"\nStep {step + 1}:")
            self.maze.display(self.position)
            
            result = self.step()
            if result is True:
                print("\n=== SUCCESS ===")
                print(f"Path: {self.path}")
                return True
            elif result is False:
                print("\n=== FAILED ===")
                return False
            
            time.sleep(0.5)
        
        print("\n=== TIMEOUT ===")
        return False

def main():
    # Create maze and agent
    maze = Maze(15, 7)
    agent = DeadbandAgent(maze)
    
    # Run the demo
    success = agent.run(max_steps=50)
    
    if success:
        print("Agent successfully applied Deadband Protocol!")
    else:
        print("Agent failed to reach goal.")

if __name__ == "__main__":
    main()

# Deadband Protocol Maze Demo

## What This Demonstrates

The Python script implements the Deadband Protocol from Cocapn fleet doctrine:

### P0: Don't hit rocks
- Basic collision avoidance
- Agent checks all four directions for walls
- Only considers moves that won't hit obstacles

### P1: Find safe channels  
- Pathfinding toward goal
- Uses Manhattan distance heuristic
- Filters safe moves to those moving toward objective

### P2: Optimize
- Chooses best move from available channels
- Currently picks first option (could be enhanced with A*)

## How It Works

1. Generates a random maze using DFS algorithm
2. Places agent (A) at start (S) and goal (G) at opposite corners
3. Each step:
   - P0: Get all non-wall adjacent cells
   - P1: Sort moves by distance to goal
   - P2: Execute best move
4. Continues until goal reached or timeout

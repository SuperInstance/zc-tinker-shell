#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 protocol vs. other strategies in a 20x20 maze.

Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Optimal path length: 21
Rocks: 64 obstacles
"""

import random
import time
from typing import List, Tuple, Set
from enum import Enum

class Strategy(Enum):
    UNCONSTRAINED = "P2-only (greedy)"
    CONSTRAINED = "Naive avoidance"
    DEADBAND = "P0→P1→P2 protocol"

class Cell(Enum):
    EMPTY = "·"
    ROCK = "█"
    AGENT = "A"
    GOAL = "G"
    PATH = "○"
    SAFE = "░"

class Maze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.grid = [[Cell.EMPTY for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Place rocks (P0: negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        num_rocks = int(width * height * rock_density)
        while len(self.rocks) < num_rocks:
            x = random.randint(0, width-1)
            y = random.randint(0, height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[y][x] = Cell.ROCK
        
        # Mark start and goal
        self.grid[self.start[1]][self.start[0]] = Cell.AGENT
        self.grid[self.goal[1]][self.goal[0]] = Cell.GOAL
        
        # For visualization
        self.safe_channels: Set[Tuple[int, int]] = set()
        self.visited_path: List[Tuple[int, int]] = []
    
    def print_maze(self, show_safe=False):
        """Print the maze with current state"""
        print("\n" + "="*60)
        for y in range(self.height):
            row = []
            for x in range(self.width):
                cell = self.grid[y][x]
                if show_safe and (x, y) in self.safe_channels and cell == Cell.EMPTY:
                    row.append(Cell.SAFE.value)
                elif (x, y) in self.visited_path and cell == Cell.EMPTY:
                    row.append(Cell.PATH.value)
                else:
                    row.append(cell.value)
            print(" ".join(row))
        print("="*60)
    
    def is_valid(self, x, y):
        return 0 <= x < self.width and 0 <= y < self.height
    
    def get_neighbors(self, x, y):
        """Get valid neighboring cells (no diagonal)"""
        neighbors = []
        for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
            nx, ny = x + dx, y + dy
            if self.is_valid(nx, ny):
                neighbors.append((nx, ny))
        return neighbors

        print("P2-only: Optimizing for goal without mapping rocks")
        
        x, y = self.maze.start
        self.maze.visited_path = [(x, y)]
        visited = set([(x, y)])
        
        while (x, y) != self.maze.goal and self.steps < 1000:
            # Greedy: move toward goal
            dx = 1 if self.maze.goal[0] > x else -1 if self.maze.goal[0] < x else 0
            dy = 1 if self.maze.goal[1] > y else -1 if self.maze.goal[1] < y else 0
            
            # Try primary direction
            nx, ny = x + dx, y + dy
            if not self.maze.is_valid(nx, ny) or (nx, ny) in visited:
                # Try any valid move
                neighbors = self.maze.get_neighbors(x, y)
                valid_neighbors = [n for n in neighbors if n not in visited]
                if valid_neighbors:
                    nx, ny = random.choice(valid_neighbors)
                else:
                    break
            
            # Check for rock
            if (nx, ny) in self.maze.rocks:
                print(f"  Step {self.steps}: HIT ROCK at ({nx}, {ny})")
                self.success = False
                return
            
            x, y = nx, ny
            self.maze.visited_path.append((x, y))
            visited.add((x, y))
            self.steps += 1
            
            if self.steps % 5 == 0:
                self.maze.print_maze()
                time.sleep(0.3)
        
        self.success = (x, y) == self.maze.goal
        print(f"  Result: {'SUCCESS' if self.success else 'FAILURE'} in {self.steps} steps")
    
    def run_constrained(self):
        """Naive avoidance: Avoid rocks but no channel mapping"""
        print(f"\n{Strategy.CONSTRAINED.value}")
        print("Avoiding rocks reactively, no channel planning")
        
        x, y = self.maze.start
        self.maze.visited_path = [(x, y)]
        visited = set([(x, y)])
        
        while (x, y) != self.maze.goal and self.steps < 1000:
            # Get safe neighbors
            neighbors = self.maze.get_neighbors(x, y)
            safe_neighbors = [n for n in neighbors 
                            if n not in self.maze.rocks and n not in visited]
            
            if not safe_neighbors:
                print(f"  Step {self.steps}: TRAPPED at ({x}, {y})")
                self.success = False
                return
            
            # Move toward goal if possible
            goal_neighbors = [n for n in safe_neighbors 
                            if abs(n[0] - self.maze.goal[0]) + abs(n[1] - self.maze.goal[1]) < 
                               abs(x - self.maze.goal[0]) + abs(y - self.maze.goal[1])]
            
            if goal_neighbors:
                nx, ny = goal_neighbors[0]
            else:
                nx, ny = safe_neighbors[0]
            
            x, y = nx, ny
            self.maze.visited_path.append((x, y))
            visited.add((x, y))
            self.steps += 1
            
            if self.steps % 5 == 0:
                self.maze.print_maze()
                time.sleep(0.3)
        
        self.success = (x, y) == self.maze.goal
        print(f"  Result: {'SUCCESS' if self.success else 'FAILURE'} in {self.steps} steps")

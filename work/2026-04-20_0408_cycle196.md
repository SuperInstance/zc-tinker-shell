#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Visualizes P0→P1→P2 using ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import json
import random
from typing import List, Tuple, Set

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

def generate_maze(size: int, rock_count: int) -> List[List[str]]:
    """Generate a size x size maze with rocks ('#') and open cells ('.')"""
    maze = [['.' for _ in range(size)] for _ in range(size)]
    
    # Place rocks randomly (but ensure start (0,0) and goal (size-1, size-1) are clear)
    rocks_placed = 0
    while rocks_placed < rock_count:
        x = random.randint(0, size-1)
        y = random.randint(0, size-1)
        
        # Don't block start or goal
        if (x == 0 and y == 0) or (x == size-1 and y == size-1):
            continue
            
        if maze[y][x] == '.':
            maze[y][x] = '#'
            rocks_placed += 1
    
    return maze

def p0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Map negative space - identify all rock positions"""
    rocks = set()
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == '#':
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
    """P1: Find safe channels - all paths from start to goal avoiding rocks"""
    size = len(maze)
    start = (0, 0)
    goal = (size-1, size-1)
    
    # Simple BFS to find all safe paths
    from collections import deque
    
    queue = deque([([start], set([start]))])
    safe_paths = []
    
    while queue:
        path, visited = queue.popleft()
        x, y = path[-1]
        
        if (x, y) == goal:
            safe_paths.append(path)
            continue
            
        # Explore neighbors (right, down, left, up)
        for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
            nx, ny = x + dx, y + dy
            
            if 0 <= nx < size and 0 <= ny < size:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    new_path = path + [(nx, ny)]
                    new_visited = visited.copy()
                    new_visited.add((nx, ny))
                    queue.append((new_path, new_visited))
    
    return safe_paths

def p2_optimize_within_channel(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """P2: Optimize within safe channels - find shortest safe path"""
    if not safe_paths:
        return []
    
    # Find path with minimum length (Manhattan distance would be optimal)
    optimal_path = min(safe_paths, key=len)
    return optimal_path

def visualize_maze(maze: List[List[str]], 
                   rocks: Set[Tuple[int, int]], 
                   safe_paths: List[List[Tuple[int, int]]], 
                   optimal_path: List[Tuple[int, int]]) -> str:
    """Create ASCII visualization of the maze with protocol phases"""
    size = len(maze)
    vis = [['.' for _ in range(size)] for _ in range(size)]
    
    # Mark rocks
    for x, y in rocks:
        vis[y][x] = '#'
    
    # Mark start and goal
    vis[0][0] = 'S'
    vis[size-1][size-1] = 'G'
    
    # Mark optimal path (avoid overwriting start/goal)
    for x, y in optimal_path[1:-1]:  # Skip start and goal
        vis[y][x] = '*'
    
    # Convert to ASCII
    lines = []
    lines.append("=" * (size * 2 + 3))
    lines.append("DEADBAND PROTOCOL MAZE DEMO")
    lines.append(f"Size: {size}x{size}, Rocks: {len(rocks)}, Safe paths: {len(safe_paths)}")
    lines.append("=" * (size * 2 + 3))
    
    for y in range(size):
        row = []
        for x in range(size):
            row.append(vis[y][x])
        lines.append(' '.join(row))
    
    lines.append("")
    lines.append("LEGEND:")
    lines.append("  S = Start (0,0)")
    lines.append("  G = Goal (19,19)")
    lines.append("  # = Rock (P0 negative space)")
    lines.append("  * = Optimal path (P2 within safe channels)")
    lines.append("  . = Safe channel space (P1)")
    lines.append("")
    
    # Protocol stats
    lines.append("PROTOCOL EXECUTION:")
    lines.append(f"  P0 (Map rocks): {len(rocks)} rocks identified")
    lines.append(f"  P1 (Safe channels): {len(safe_paths)} safe paths found")
    if optimal_path:
        lines.append(f"  P2 (Optimize): Optimal path length = {len(optimal_path)}")
        lines.append(f"     (Theoretical optimal from sim: {OPTIMAL_PATH_LENGTH})")
    else:
        lines.append("  P2 (Optimize): NO SAFE PATH FOUND - Protocol violation would occur")
    
    return '\n'.join(lines)

def run_demo():
    """Main demo function"""
    print("Initializing Deadband Protocol Demo...")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print("")
    
    # Generate maze matching simulation parameters
    maze = generate_maze(MAZE_SIZE, ROCK_COUNT)

    print("P0: MAPPING NEGATIVE SPACE")

    # P2: Optimize within channel
    print("P2: OPTIMIZING WITHIN SAFE CHANNELS")

    if optimal_path:
        print(f"  Selected optimal path of length {len(optimal_path)}")

        print("  WARNING: No safe path found!")
        print("  This demonstrates protocol violation consequence")
    print("")
    
    # Visualize
    visualization = visualize_maze(maze, rocks, safe_paths, optimal_path)
    print(visualization)
    
    # Compare with simulation results
    print("COMPARISON WITH FLEET SIMULATION DATA:")
    print("  From constraint_sim_final.json:")

    print("    - Deadband (P0→P1→P2): 100% success")
    print("    - Optimal path length: 21")
    print("")
    
    if optimal_path and len(optimal_path) <= OPTIMAL_PATH_LENGTH + 5:
        print("RESULT: Protocol execution successful ✓")
        print("Agent reached goal via safe channels.")
    else:
        print("RESULT: Protocol highlights failure mode ⚠")
        print("Without P0→P1, agent would hit rocks.")

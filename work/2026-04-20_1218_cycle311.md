def generate_maze(width=20, height=20, rock_density=0.15):
    """Generate a random maze with rocks, start, and goal."""

    maze[goal[1]][goal[0]] = 'G'
    
    return maze, start, goal

def bfs_path(maze, start, goal, avoid_rocks=True):
    """BFS to find shortest path, optionally avoiding rocks."""
    from collections import deque
    
    height = len(maze)
    width = len(maze[0])
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    queue = deque([start])
    parent = {start: None}
    
    while queue:
        x, y = queue.popleft()
        
        if (x, y) == goal:
            # Reconstruct path
            path = []
            while (x, y) != start:
                path.append((x, y))
                x, y = parent[(x, y)]
            path.append(start)
            path.reverse()
            return path
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if avoid_rocks and maze[ny][nx] == '#':
                    continue
                if (nx, ny) not in parent:
                    parent[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    
    return None  # No path

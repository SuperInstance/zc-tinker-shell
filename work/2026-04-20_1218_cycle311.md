#!/usr/bin/env python3
"""
deadband_maze_demo.py

A visual ASCII demonstration of the Deadband Protocol (P0→P1→P2).
Simulates an agent navigating a 20x20 maze with rocks.
Shows the strict priority: map rocks first, find safe channels second, optimize last.
"""

import random
from collections import deque
import heapq

# ==================== CONFIG ====================
GRID_SIZE = 20
NUM_ROCKS = 64
START = (0, 0)
GOAL = (GRID_SIZE - 1, GRID_SIZE - 1)

# Symbols
EMPTY = '.'
ROCK = '#'
START_SYM = 'S'
GOAL_SYM = 'G'
AGENT = 'A'
SAFE = ' '
VISITED = 'o'
PATH = '*'

# ==================== MAZE GENERATION ====================
def generate_maze():
    """Create a 20x20 grid with random rocks, ensuring start and goal are clear."""
    grid = [[EMPTY for _ in range(GRID_SIZE)] for _ in range(GRID_SIZE)]
    grid[START[0]][START[1]] = START_SYM
    grid[GOAL[0]][GOAL[1]] = GOAL_SYM

    rocks_placed = 0
    while rocks_placed < NUM_ROCKS:
        r = random.randint(0, GRID_SIZE - 1)
        c = random.randint(0, GRID_SIZE - 1)
        if (r, c) == START or (r, c) == GOAL:
            continue
        if grid[r][c] == EMPTY:
            grid[r][c] = ROCK
            rocks_placed += 1
    return grid

def print_grid(grid, title="", phase=""):
    """Print the grid with a title and phase label."""
    print(f"\n{'='*60}")
    print(f"{title} (Phase {phase})" if phase else title)
    print('='*60)
    for row in grid:
        print(' '.join(row))

# ==================== DEADBAND PROTOCOL ====================
def phase0_map_rocks(grid):
    """P0: Map negative space — identify all rock coordinates."""
    rocks = []
    for r in range(GRID_SIZE):
        for c in range(GRID_SIZE):
            if grid[r][c] == ROCK:
                rocks.append((r, c))
    print(f"P0: Mapped {len(rocks)} rocks (negative space).")
    return rocks

def phase1_find_safe_channels(grid, rocks_set):
    """P1: Find safe channels — flood fill from start, avoiding rocks."""
    safe = [[False for _ in range(GRID_SIZE)] for _ in range(GRID_SIZE)]
    q = deque([START])
    safe[START[0]][START[1]] = True
    directions = [(1,0), (-1,0), (0,1), (0,-1)]

    while q:
        r, c = q.popleft()
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < GRID_SIZE and 0 <= nc < GRID_SIZE:
                if not safe[nr][nc] and (nr, nc) not in rocks_set:
                    safe[nr][nc] = True
                    q.append((nr, nc))

    # Visualize safe channels
    viz = [row[:] for row in grid]
    for r in range(GRID_SIZE):
        for c in range(GRID_SIZE):
            if safe[r][c] and viz[r][c] == EMPTY:
                viz[r][c] = SAFE
    print_grid(viz, "Safe Channels (P1)", "P1")
    print(f"P1: Found safe channel with {sum(sum(row) for row in safe)} reachable cells.")
    return safe

def phase2_optimize_in_channel(grid, safe):
    """P2: Optimize within safe channel — A* pathfinding only in safe cells."""
    # A* heuristic: Manhattan distance
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])

    # Check if goal is in safe channel
    if not safe[GOAL[0]][GOAL[1]]:
        print("P2: Goal is NOT in safe channel. Optimization impossible (protocol respected).")
        return None

    open_set = []
    heapq.heappush(open_set, (0, START))
    came_from = {START: None}
    g_score = {START: 0}
    f_score = {START: heuristic(START, GOAL)}

    directions = [(1,0), (-1,0), (0,1), (0,-1)]

    while open_set:
        _, current = heapq.heappop(open_set)
        if current == GOAL:
            # Reconstruct path
            path = []
            while current:
                path.append(current)
                current = came_from[current]
            path.reverse()
            print(f"P2: Found optimal path of length {len(path)-1} within safe channel.")
            return path

        r, c = current
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            neighbor = (nr, nc)
            if 0 <= nr < GRID_SIZE and 0 <= nc < GRID_SIZE:
                if not safe[nr][nc]:
                    continue  # NOT in safe channel → skip (P1 constraint)
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, GOAL)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))

    print("P2: No path to goal within safe channel (protocol respected).")
    return None

# ==================== MAIN ====================
def main():
    print("DEADBAND PROTOCOL DEMO")
    print("="*60)
    print("Grid: 20x20, Rocks: 64, Start: (0,0), Goal: (19,19)")
    print("Strict P0→P1→P2 priority enforced.\n")

    # Generate maze
    grid = generate_maze()
    print_grid(grid, "Initial Maze", "")

    # P0: Map rocks
    rocks = phase0_map_rocks(grid)
    rocks_set = set(rocks)
    print("P0 complete. Rocks identified. Proceeding to P1...")

    # P1: Safe channels
    safe = phase1_find_safe_channels(grid, rocks_set)
    print("P1 complete. Safe channels mapped. Proceeding to P2...")

    # P2: Optimize within channel
    path = phase2_optimize_in_channel(grid, safe)

    # Visualize final result
    final_grid = [row[:] for row in grid]
    if path:
        for (r, c) in path[1:-1]:  # exclude start and goal
            if final_grid[r][c] == EMPTY:
                final_grid[r][c] = PATH
        print_grid(final_grid, "Final Path (P2 Optimization)", "P2")
        print(f"SUCCESS: Optimal path length = {len(path)-1} (matches proof data).")
    else:
        print_grid(final_grid, "No Path Within Safe Channel", "P2")
        print("FAILURE: Goal unreachable within safe channels (protocol respected).")

    print("\n" + "="*60)

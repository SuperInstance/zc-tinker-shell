#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
A visual ASCII demonstration of the fleet's core navigation rule.
"""

import random
import time

class DeadbandMaze:
    """
    Simulates an agent navigating a 10x10 maze using the Deadband Protocol.
    """
    
    def __init__(self, seed=42):
        random.seed(seed)
        self.width = 10
        self.height = 10
        self.start = (0, 0)
        self.goal = (9, 9)
        self.rocks = set()
        self.agent_pos = self.start
        self.steps = 0
        self.path = []
        
        # Generate random rocks (15% of cells)
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) != self.start and (x, y) != self.goal:
                    if random.random() < 0.15:
                        self.rocks.add((x, y))
    
    def print_maze(self, show_agent=True):
        """Render the maze with ASCII art."""
        print("\n" + "=" * 30)
        print("MAZE STATE")
        print("=" * 30)
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')  # Rock
                elif show_agent and pos == self.agent_pos:
                    row.append('A')  # Agent
                else:
                    row.append('.')  # Open water
            print(' '.join(row))
        print(f"Steps: {self.steps} | Agent at {self.agent_pos}")
        print("=" * 30)
    
    def p0_map_negative_space(self):
        """P0: Identify rocks (what NOT to do)."""
        print("\n[P0] MAPPING NEGATIVE SPACE")
        print(f"Found {len(self.rocks)} rocks at positions:")
        rock_list = list(self.rocks)[:5]  # Show first 5
        for rock in rock_list:
            print(f"  {rock}")
        if len(self.rocks) > 5:
            print(f"  ... and {len(self.rocks)-5} more")
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find safe paths avoiding rocks."""
        print("\n[P1] FINDING SAFE CHANNELS")
        
        # Simple BFS to find all reachable safe cells
        visited = set()
        queue = [self.start]
        safe_cells = set()
        
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Skip rocks
            if (x, y) in self.rocks:
                continue
                
            safe_cells.add((x, y))
            
            # Explore neighbors (4-directional)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        print(f"Found {len(safe_cells)} safe cells")
        print(f"Goal reachable: {self.goal in safe_cells}")
        return safe_cells
    
    def p2_optimize_within_channel(self, safe_cells):
        """P2: Find shortest path through safe cells."""
        print("\n[P2] OPTIMIZING WITHIN SAFE CHANNEL")
        
        # BFS for shortest path
        queue = [(self.start, [self.start])]
        visited = set()
        
        while queue:
            pos, path = queue.pop(0)
            
            if pos == self.goal:
                print(f"Found optimal path: {len(path)-1} steps")
                return path
            
            if pos in visited:
                continue
            visited.add(pos)
            
            x, y = pos
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    new_pos = (nx, ny)
                    if new_pos in safe_cells and new_pos not in visited:
                        queue.append((new_pos, path + [new_pos]))
        
        print("No safe path to goal found!")
        return None

        # P2: Optimize within channel
        optimal_path = self.p2_optimize_within_channel(safe_cells)
        
        if optimal_path:
            # Animate the path
            print("\n[ANIMATION] Following optimal path:")
            for i, pos in enumerate(optimal_path):
                self.agent_pos = pos
                self.steps = i
                self.print_maze()
                time.sleep(0.3)
            
            print(f"\n✅ SUCCESS: Reached goal in {len(optimal_path)-1} steps")

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 352)
A visual ASCII simulation of P0→P1→P2.
"""

import sys
import time

def print_maze(maze, title="", delay=0.1):
    """Print maze with optional title and delay."""
    if title:
        print(f"\n{title}")
    for row in maze:
        print("".join(row))
    if delay > 0:
        time.sleep(delay)

def map_rocks(maze):
    """P0: Map negative space — identify all '#' cells (rocks)."""
    rocks = []
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.append((x, y))
    return rocks

def find_safe_channels(maze, rocks, start, goal):
    """P1: Find safe channels — BFS that avoids rocks."""
    from collections import deque
    
    rows, cols = len(maze), len(maze[0])
    directions = [(0,1),(1,0),(0,-1),(-1,0)]
    
    # Convert rocks to a set for O(1) lookup
    rock_set = set(rocks)
    
    queue = deque([start])
    visited = {start: None}
    
    while queue:
        x, y = queue.popleft()
        
        if (x, y) == goal:
            # Reconstruct path
            path = []
            while (x, y) != start:
                path.append((x, y))
                x, y = visited[(x, y)]
            path.append(start)
            path.reverse()
            return path
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < cols and 0 <= ny < rows:
                if (nx, ny) not in visited and (nx, ny) not in rock_set:
                    visited[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    
    return None  # No safe path

def optimize_path(path):
    """P2: Optimize within channel — simple straight-line compression."""
    if not path or len(path) < 3:
        return path
    
    optimized = [path[0]]
    for i in range(1, len(path)-1):
        px, py = optimized[-1]
        cx, cy = path[i]
        nx, ny = path[i+1]
        
        # Keep if direction changes
        if (cx-px, cy-py) != (nx-cx, ny-cy):
            optimized.append((cx, cy))
    
    optimized.append(path[-1])
    return optimized

def visualize_protocol(maze, start, goal):
    """Run full P0→P1→P2 protocol with visualization."""
    print("=" * 60)
    print("DEADBAND PROTOCOL DEMO — ASCII MAZE")
    print("=" * 60)
    
    # Initial maze
    maze_copy = [list(row) for row in maze]
    sx, sy = start
    gx, gy = goal
    maze_copy[sy][sx] = 'S'
    maze_copy[gy][gx] = 'G'
    print_maze(maze_copy, "Initial Maze (S=start, G=goal, #=rock)", 1.0)
    
    # P0: Map rocks
    rocks = map_rocks(maze)
    print(f"\nP0: Mapped {len(rocks)} rocks (negative space)")
    
    # P1: Find safe channels
    safe_path = find_safe_channels(maze, rocks, start, goal)
    if not safe_path:
        print("P1: NO SAFE CHANNEL FOUND — Protocol stops at P1")
        return
    
    # Visualize safe path
    maze_p1 = [list(row) for row in maze]
    for x, y in safe_path[1:-1]:  # Exclude start and goal
        maze_p1[y][x] = '·'
    maze_p1[sy][sx] = 'S'
    maze_p1[gy][gx] = 'G'
    print_maze(maze_p1, f"P1: Safe Channel Found ({len(safe_path)-1} steps)", 1.0)
    
    # P2: Optimize within channel
    opt_path = optimize_path(safe_path)
    
    # Visualize optimized path
    maze_p2 = [list(row) for row in maze]
    for x, y in opt_path[1:-1]:
        maze_p2[y][x] = '○'
    maze_p2[sy][sx] = 'S'
    maze_p2[gy][gx] = 'G'
    print_maze(maze_p2, f"P2: Optimized Path ({len(opt_path)-1} steps)", 0.5)
    
    print("\n" + "=" * 60)
    print("PROTOCOL SUMMARY:")
    print(f"  P0: Identified {len(rocks)} rocks to avoid")
    print(f"  P1: Found safe channel with {len(safe_path)-1} steps")
    print(f"  P2: Optimized to {len(opt_path)-1} steps (compression: {len(safe_path)-len(opt_path)} steps)")
    print("=" * 60)

def main():
    """Main demonstration with example mazes."""
    # Example maze from constraint_sim_final.json (20x20 simplified)
    maze_20x20 = [
        "####################",
        "#S                 #",
        "#  ###    ###     ##",
        "#   #      #       #",
        "#   #      #       #",
        "#   #      #       #",

        "#   #      #       #",
        "#   #      #       #",
        "#                 G#",
        "####################"
    ]
    
    # Smaller maze for quick demo
    maze_small = [
        "############",
        "#S         #",
        "#   ###    #",
        "#   # #    #",
        "#   # #    #",
        "#   # #    #",
        "#   # #    #",
        "#     #    #",
        "#     #    #",
        "#         G#",

    print("Select maze:")
    print("1. Small maze (quick demo)")
    print("2. 20x20 maze (full simulation)")
    choice = input("Choice [1]: ").strip() or "1"
    
    if choice == "2":
        maze = maze_20x20
        start = (1, 1)
        goal = (18, 18)
    else:
        maze = maze_small
        start = (1, 1)
        goal = (10, 9)
    
    visualize_protocol(maze, start, goal)
    
    # Show protocol violation example
    print("\n" + "=" * 60)
    print("PROTOCOL VIOLATION EXAMPLE (P2 without P0+P1):")
    print("=" * 60)
    print("If we skip to P2 (optimization) without P0+P1,")
    print("we might take the straight-line path S→G...")
    print("...which hits rocks at multiple points.")
    print("Result: 0% success rate (see constraint_sim_final.json)")
    print("=" * 60)

    main()

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 98)
A visual ASCII simulation of P0→P1→P2 in a 10x10 grid.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time

class DeadbandMaze:
    def __init__(self, width=10, height=10, rock_density=0.2):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.rocks = set()
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.agent_pos = self.start
        self.path = []
        
        # P0: Map negative space (rocks)
        self._generate_rocks(rock_density)
        # Ensure start and goal are not rocks
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def _generate_rocks(self, density):
        """P0: Identify rocks (negative space)."""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        # Exclude start and goal
        candidates = [(x, y) for x in range(self.width) for y in range(self.height)
                     if (x, y) not in (self.start, self.goal)]
        self.rocks = set(random.sample(candidates, min(num_rocks, len(candidates))))
        for x, y in self.rocks:
            self.grid[y][x] = '#'
    
    def print_maze(self, highlight=None):
        """Display the maze with current agent position."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.agent_pos:
                    row.append('A')
                elif highlight and (x, y) in highlight:
                    row.append('*')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
        print()
    
    def p0_map_negative_space(self):
        """Return set of rock coordinates."""
        return self.rocks
    
    def p1_find_safe_channels(self):
        """
        P1: Find all paths from start to goal that avoid rocks.
        Uses simple BFS to find first safe path.
        """
        from collections import deque
        
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.popleft()
            if (x, y) == self.goal:
                return path
            
            for dx, dy in [(1,0), (0,1), (-1,0), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited and (nx, ny) not in self.rocks:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        return None  # No safe path
    
    def p2_optimize_within_channel(self, safe_path):
        """
        P2: Optimize within the safe channel (shortest steps).
        Since BFS already gives shortest path in steps, this is a placeholder
        for more complex optimization (e.g., weight, turns).
        """
        # For demo, just return the path (already optimal in steps)
        return safe_path
    
    def run_deadband_protocol(self):
        """Execute full P0→P1→P2 sequence."""
        print("=== DEADBAND PROTOCOL DEMO ===")
        print("Maze (S=start, G=goal, #=rock, A=agent, .=empty):")
        self.print_maze()
        
        print("P0: MAP NEGATIVE SPACE")
        rocks = self.p0_map_negative_space()
        print(f"Identified {len(rocks)} rocks.")
        time.sleep(1)
        
        print("\nP1: FIND SAFE CHANNELS")
        safe_path = self.p1_find_safe_channels()
        if safe_path is None:
            print("FAIL: No safe channel exists. Agent cannot proceed.")
            return False
        print(f"Found safe channel with {len(safe_path)-1} steps.")
        print("Safe path (highlighted with *):")
        self.print_maze(highlight=set(safe_path))
        time.sleep(1)
        
        print("P2: OPTIMIZE WITHIN CHANNEL")
        optimal_path = self.p2_optimize_within_channel(safe_path)
        print(f"Optimal path length: {len(optimal_path)-1} steps")
        
        # Animate agent moving along path
        print("\nAgent traversing safe channel:")
        for step, pos in enumerate(optimal_path):
            self.agent_pos = pos
            self.print_maze()
            time.sleep(0.3)
            if pos == self.goal:
                print(f"SUCCESS: Reached goal in {step} steps.")
                return True
        return False
    
    def run_unconstrained_greedy(self):
        """
        Contrast: Greedy approach that ignores P0/P1 (just heads toward goal).
        This simulates the 'constrained_success: 0' result.
        """
        print("\n=== UNCONSTRAINED GREEDY (P2-only) ===")
        self.agent_pos = self.start
        steps = 0
        max_steps = self.width * self.height * 2
        
        while self.agent_pos != self.goal and steps < max_steps:
            x, y = self.agent_pos
            # Move toward goal (greedy)
            if x < self.goal[0]:
                nx, ny = x + 1, y
            elif y < self.goal[1]:
                nx, ny = x, y + 1
            else:
                nx, ny = x, y  # stuck
            
            if (nx, ny) in self.rocks:
                print(f"FAIL: Hit rock at ({nx},{ny}) after {steps} steps.")
                return False
            self.agent_pos = (nx, ny)
            steps += 1
            # Optional: print every few steps
            if steps % 3 == 0:
                self.print_maze()
        
        if self.agent_pos == self.goal:
            print(f"Lucky success in {steps} steps (low probability).")
            return True
        print(f"FAIL: Exceeded max steps ({max_steps}).")
        return False

def main():
    """Run the demonstration."""
    random.seed(42)  # Reproducible for demo
    maze = DeadbandMaze(width=8, height=6, rock_density=0.25)
    
    # Deadband protocol
    success1 = maze.run_deadband_protocol()
    
    # Reset and try greedy
    maze.agent_pos = maze.start
    success2 = maze.run_unconstrained_greedy()
    
    print("\n=== RESULTS ===")
    print(f"Deadband Protocol (P0→P1→P2): {'SUCCESS' if success1 else 'FAIL'}")
    print(f"Unconstrained Greedy (P2-only): {'SUCCESS' if success2 else 'FAIL'}")
    print("\nNote: In complex mazes, greedy fails with probability → 1.0")
    print("(Matches fleet simulation: deadband 50/50, constrained 0/50)")

if __name__ == "__main__":
    main()
```

- Shows ASCII visualization of each phase
- Animated agent movement along safe channel
- Contrasts deadband protocol vs. greedy P2-only (which usually hits rocks)
- Parameters adjustable: maze size, rock density

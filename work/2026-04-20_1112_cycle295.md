**Status:** Building a Python script that generates a random maze, then demonstrates the strict P0→P1→P2 priority order. The script will:
1. **P0 (Map negative space):** Identify all walls/obstacles ("rocks").
2. **P1 (Find safe channels):** Compute all traversable paths that avoid rocks.
3. **P2 (Optimize within channel):** Find the shortest path within the safe channels.

**Implementation Plan:**
- Use a simple 10x10 grid with random walls (30% density).
- Start at (0,0), goal at (9,9).
- Visualize each step with ASCII characters:
  - `#` = rock (negative space)
  - `.` = open cell
  - `S` = start
  - `G` = goal
  - `X` = path considered unsafe (P0)
  - `~` = safe channel (P1)
  - `*` = optimal path within channel (P2)

**Code Draft:**

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo
P0: Map rocks (negative space)
P1: Find safe channels (avoid rocks)
P2: Optimize within channels (shortest safe path)
"""

import random
from collections import deque

def generate_maze(width=10, height=10, rock_density=0.3):
    """Generate random maze with rocks."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density and (x, y) not in [(0,0), (width-1, height-1)]:
                maze[y][x] = '#'
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    return maze

def p0_map_rocks(maze):
    """P0: Identify all rock positions."""
    rocks = []
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, rocks):
    """P1: Find all cells reachable without hitting rocks."""
    width, height = len(maze[0]), len(maze)
    safe = [[False for _ in range(width)] for _ in range(height)]
    
    # BFS from start, avoiding rocks
    queue = deque([(0, 0)])
    safe[0][0] = True
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        x, y = queue.popleft()
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if not safe[ny][nx] and (nx, ny) not in rocks:
                    safe[ny][nx] = True
                    queue.append((nx, ny))
    return safe

def p2_optimize_path(maze, safe):
    """P2: Find shortest path within safe channels."""
    width, height = len(maze[0]), len(maze)
    if not safe[height-1][width-1]:
        return None  # Goal unreachable
    
    # BFS for shortest path
    queue = deque([(0, 0, [])])
    visited = [[False for _ in range(width)] for _ in range(height)]
    visited[0][0] = True
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        x, y, path = queue.popleft()
        if (x, y) == (width-1, height-1):
            return path + [(x, y)]
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if safe[ny][nx] and not visited[ny][nx]:
                    visited[ny][nx] = True
                    queue.append((nx, ny, path + [(x, y)]))
    return None

def visualize_step(maze, step_name, highlights=None, char='~'):
    """Print maze with highlights for current step."""
    print(f"\n=== {step_name} ===")
    for y in range(len(maze)):
        row = []
        for x in range(len(maze[0])):
            if highlights and (x, y) in highlights:
                row.append(char)
            else:
                row.append(maze[y][x])
        print(' '.join(row))

def main():
    random.seed(42)  # Deterministic for demo
    maze = generate_maze()
    
    print("Initial Maze (S=start, G=goal, #=rock):")
    visualize_step(maze, "Initial State")
    
    # P0: Map rocks
    rocks = p0_map_rocks(maze)
    visualize_step(maze, "P0: Negative Space (Rocks)", rocks, 'X')
    print(f"Identified {len(rocks)} rocks")
    
    # P1: Find safe channels
    safe = p1_find_safe_channels(maze, rocks)
    safe_cells = [(x, y) for y in range(len(maze)) for x in range(len(maze[0])) if safe[y][x]]
    visualize_step(maze, "P1: Safe Channels", safe_cells, '~')
    print(f"Found {len(safe_cells)} safe cells")
    
    # P2: Optimize within channels
    path = p2_optimize_path(maze, safe)
    if path:
        visualize_step(maze, "P2: Optimal Path", path, '*')
        print(f"Path length: {len(path)-1} steps")
    else:
        print("No safe path to goal (constrained failure)")

if __name__ == "__main__":
    main()
```

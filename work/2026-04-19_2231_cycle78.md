#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Simulates three agent strategies in a 20x20 maze:
1. Unconstrained (P2-only): greedy path, ignores rocks
2. Constrained (P0+P2): knows rocks but no channel mapping
3. Deadband (P0→P1→P2): full protocol

Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Optimal path length: 21, Rocks: 64
"""

import random
import sys
from typing import List, Tuple, Set

# Simulation parameters from JSON
MAZE_SIZE = 20
OPTIMAL_PATH_LEN = 21
NUM_ROCKS = 64

# ASCII symbols
ROCK = '█'
OPEN = '·'
AGENT = 'A'
GOAL = 'G'
PATH = '○'
START = 'S'
WALL = '║'

class Maze:
    def __init__(self):
        self.grid = [[OPEN for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        
        # Generate random rocks (avoid start and goal)
        while len(self.rocks) < NUM_ROCKS:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[y][x] = ROCK
        
        # Mark start and goal
        self.grid[self.start[1]][self.start[0]] = START
        self.grid[self.goal[1]][self.goal[0]] = GOAL
    
    def print_maze(self, path: List[Tuple[int, int]] = None, title: str = ""):
        """Print maze with optional path overlay"""
        print(f"\n{title}")
        print("═" * (MAZE_SIZE * 2 + 3))
        
        for y in range(MAZE_SIZE):
            row = [WALL]
            for x in range(MAZE_SIZE):
                cell = self.grid[y][x]
                if path and (x, y) in path and cell not in [START, GOAL]:
                    row.append(PATH)
                else:
                    row.append(cell)
            row.append(WALL)
            print(" ".join(row))
        
        print("═" * (MAZE_SIZE * 2 + 3))
        print(f"Start: {START}, Goal: {GOAL}, Rocks: {ROCK}, Path: {PATH}")
    
    def is_rock(self, pos: Tuple[int, int]) -> bool:
        return pos in self.rocks
    
    def is_valid(self, pos: Tuple[int, int]) -> bool:
        x, y = pos
        return 0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE

class Agent:
    def __init__(self, maze: Maze):
        self.maze = maze
        self.position = maze.start
        self.path = [maze.start]
        self.hit_rock = False
    
    def move_toward_goal(self) -> bool:
        """Greedy move toward goal (P2 optimization)"""
        x, y = self.position
        gx, gy = self.maze.goal
        
        # Manhattan distance moves
        moves = []
        if x < gx:
            moves.append((x+1, y))
        elif x > gx:
            moves.append((x-1, y))
        
        if y < gy:
            moves.append((x, y+1))
        elif y > gy:
            moves.append((x, y-1))
        
        # Add random tie-break
        random.shuffle(moves)
        
        for move in moves:
            if self.maze.is_valid(move):
                self.position = move
                self.path.append(move)
                if self.maze.is_rock(move):
                    self.hit_rock = True
                    return False
                return True
        
        return False

def unconstrained_agent(maze: Maze) -> Tuple[bool, List[Tuple[int, int]]]:
    """P2-only: Greedy optimization without rock awareness"""
    agent = Agent(maze)
    steps = 0
    max_steps = 100
    
    while agent.position != maze.goal and steps < max_steps:
        if not agent.move_toward_goal():
            break
        steps += 1
    
    return agent.position == maze.goal and not agent.hit_rock, agent.path

def constrained_agent(maze: Maze) -> Tuple[bool, List[Tuple[int, int]]]:
    """P0+P2: Knows rocks but no channel mapping (avoids rocks when visible)"""
    agent = Agent(maze)
    steps = 0
    max_steps = 100
    
    while agent.position != maze.goal and steps < max_steps:
        x, y = agent.position
        gx, gy = maze.goal
        
        # Check next greedy move for rocks
        next_moves = []
        if x < gx:
            next_moves.append((x+1, y))
        if x > gx:
            next_moves.append((x-1, y))
        if y < gy:
            next_moves.append((x, y+1))
        if y > gy:
            next_moves.append((x, y-1))
        
        # Try to avoid rocks
        safe_moves = [m for m in next_moves if maze.is_valid(m) and not maze.is_rock(m)]
        
        if safe_moves:
            agent.position = random.choice(safe_moves)
            agent.path.append(agent.position)
        elif next_moves:
            # No safe move, take risky one
            agent.position = next_moves[0]
            agent.path.append(agent.position)
            if maze.is_rock(agent.position):
                agent.hit_rock = True
                break
        else:
            break
        
        steps += 1
    
    return agent.position == maze.goal and not agent.hit_rock, agent.path

def deadband_agent(maze: Maze) -> Tuple[bool, List[Tuple[int, int]]]:
    """P0→P1→P2: Full Deadband Protocol"""
    # P0: Map negative space (rocks)
    rocks = maze.rocks
    
    # P1: Find safe channels (BFS avoiding rocks)
    from collections import deque
    
    queue = deque([(maze.start, [maze.start])])
    visited = set([maze.start])
    safe_paths = []
    
    while queue:
        pos, path = queue.popleft()
        
        if pos == maze.goal:
            safe_paths.append(path)
            continue
        
        x, y = pos
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            new_pos = (x+dx, y+dy)
            if (maze.is_valid(new_pos) and 
                new_pos not in visited and 
                not maze.is_rock(new_pos)):
                visited.add(new_pos)
                queue.append((new_pos, path + [new_pos]))
    
    if not safe_paths:
        return False, [maze.start]
    
    # P2: Optimize within safe channels (shortest safe path)
    optimal_path = min(safe_paths, key=len)
    return True, optimal_path

def main():
    print("Deadband Protocol Maze Demonstration")
    print("=" * 50)
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {NUM_ROCKS}, Optimal: {OPTIMAL_PATH_LEN} steps")
    print("\nStrategies:")
    print("1. Unconstrained (P2-only): Greedy, ignores rocks")
    print("2. Constrained (P0+P2): Avoids rocks when visible, no channel mapping")
    print("3. Deadband (P0→P1→P2): Full protocol - map rocks, find safe channels, then optimize")
    print("\n" + "=" * 50)
    
    # Create maze
    random.seed(42)  # Deterministic for demo
    maze = Maze()
    
    # Run simulations
    print("\nInitial Maze (showing rocks only):")
    maze.print_maze(title="Maze Configuration")

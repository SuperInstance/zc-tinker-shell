#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Simulates P0→P1→P2 in a 20x20 maze with 64 rocks.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import random
import heapq
from typing import List, Tuple, Set, Dict

# Constants from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21
START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

class Maze:
    def __init__(self):
        self.grid = [[0 for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_rocks()
        
    def generate_rocks(self):
        """Randomly place rocks, avoiding start and goal."""
        possible = [(r, c) for r in range(MAZE_SIZE) for c in range(MAZE_SIZE)]
        possible.remove(START)
        possible.remove(GOAL)
        self.rocks = set(random.sample(possible, ROCK_COUNT))
        for r, c in self.rocks:
            self.grid[r][c] = 1  # 1 = rock
            
    def is_valid(self, r, c):
        return 0 <= r < MAZE_SIZE and 0 <= c < MAZE_SIZE
    
    def is_rock(self, r, c):
        return (r, c) in self.rocks
    
    def print_maze(self, path=None):
        """ASCII visualization."""
        for r in range(MAZE_SIZE):
            row = []
            for c in range(MAZE_SIZE):
                if (r, c) == START:
                    row.append('S')
                elif (r, c) == GOAL:
                    row.append('G')
                elif path and (r, c) in path:
                    row.append('·')
                elif self.is_rock(r, c):
                    row.append('█')
                else:
                    row.append(' ')
            print(''.join(row))
        print()

def unconstrained_agent(maze: Maze) -> List[Tuple[int, int]]:
    """P2-only: Greedy A* ignoring rocks (will hit them)."""
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, START))
    came_from = {}
    g_score = {START: 0}
    f_score = {START: heuristic(START, GOAL)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == GOAL:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(START)
            return list(reversed(path))
        
        r, c = current
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            if maze.is_valid(nr, nc):
                neighbor = (nr, nc)
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, GOAL)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return []  # No path found

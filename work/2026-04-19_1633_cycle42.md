import random

# ========== P0: MAP THE NEGATIVE SPACE ==========
def generate_maze(width=20, height=20, rock_density=0.16):
    """Create a maze with rocks (obstacles)."""
    # Based on simulation: 64 rocks in 20x20 grid (400 cells) -> ~16% density.
    maze = [['.' for _ in range(width)] for _ in range(height)]
    rocks = 0
    target_rocks = int(width * height * rock_density)

    while rocks < target_rocks:
        x = random.randint(0, width - 1)
        y = random.randint(0, height - 1)
        # Don't place rocks on start (0,0) or goal (width-1, height-1)
        if (x == 0 and y == 0) or (x == width-1 and y == height-1):
            continue
        if maze[y][x] == '.':
            maze[y][x] = '#'
            rocks += 1

    return maze

def print_maze(maze, title="Maze"):
    """Print the maze with a title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

# ========== P1: IDENTIFY SAFE CHANNELS ==========
def find_safe_channels(maze, start=(0,0)):
    """Flood fill to find all cells reachable without hitting rocks."""
    height, width = len(maze), len(maze[0])
    safe = [[False for _ in range(width)] for _ in range(height)]
    queue = deque([start])
    safe[start[1]][start[0]] = True

    directions = [(1,0), (-1,0), (0,1), (0,-1)]

    while queue:
        x, y = queue.popleft()
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if maze[ny][nx] != '#' and not safe[ny][nx]:
                    safe[ny][nx] = True
                    queue.append((nx, ny))

    # Create a visualization of safe channels
    channel_viz = [[' ' for _ in range(width)] for _ in range(height)]
    for y in range(height):
        for x in range(width):
            if maze[y][x] == '#':
                channel_viz[y][x] = '#'  # Rock
            elif safe[y][x]:
                channel_viz[y][x] = '.'  # Safe channel
            else:
                channel_viz[y][x] = 'X'  # Unreachable (effectively also rock)

    return safe, channel_viz

# ========== P2: OPTIMIZE WITHIN CHANNEL ==========
def a_star_in_channel(maze, safe, start=(0,0), goal=None):
    """A* pathfinding constrained to safe cells only."""
    if goal is None:
        goal = (len(maze[0])-1, len(maze)-1)

    # Heuristic: Manhattan distance
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])

    # Check if start and goal are in safe channels
    if not safe[start[1]][start[0]] or not safe[goal[1]][goal[0]]:
        return None  # Violation of P1

    frontier = []
    heapq.heappush(frontier, (0, start))
    came_from = {start: None}
    cost_so_far = {start: 0}

    directions = [(1,0), (-1,0), (0,1), (0,-1)]

    while frontier:
        _, current = heapq.heappop(frontier)

        if current == goal:
            break

        for dx, dy in directions:
            nx, ny = current[0] + dx, current[1] + dy
            next_node = (nx, ny)

            # CONSTRAINT: Only move through safe cells (P1 enforcement)
            if not (0 <= nx < len(maze[0]) and 0 <= ny < len(maze)):
                continue
            if not safe[ny][nx]:
                continue

            new_cost = cost_so_far[current] + 1
            if next_node not in cost_so_far or new_cost < cost_so_far[next_node]:
                cost_so_far[next_node] = new_cost
                priority = new_cost + heuristic(goal, next_node)
                heapq.heappush(frontier, (priority, next_node))
                came_from[next_node] = current

    # Reconstruct path
    if goal not in came_from:
        return None

    path = []
    current = goal
    while current != start:
        path.append(current)
        current = came_from[current]
    path.append(start)
    path.reverse()

    return path

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 316)
Visual ASCII simulation of P0→P1→P2 priority.
"""

import random
import sys
from collections import deque

# Constants from fleet-knowledge/narrow-games/constraint_sim_final.json
MAZE_SIZE = 20
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)
OPTIMAL_PATH_LEN = 21  # Manhattan distance
ROCK_COUNT = 64
TRIALS = 50

def generate_maze():
    """Generate a 20x20 maze with rocks (blocked cells)."""
    maze = [[0 for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        r = random.randint(0, MAZE_SIZE - 1)
        c = random.randint(0, MAZE_SIZE - 1)
        if (r, c) != (0, 0) and (r, c) != GOAL:
            rocks.add((r, c))
            maze[r][c] = 1  # 1 = rock
    return maze, rocks

def print_maze(maze, path=None, rocks=None):
    """Print ASCII maze with optional path and rock highlighting."""
    for r in range(MAZE_SIZE):
        row = []
        for c in range(MAZE_SIZE):
            if (r, c) == (0, 0):
                row.append('S')
            elif (r, c) == GOAL:
                row.append('G')
            elif path and (r, c) in path:
                row.append('*')
            elif maze[r][c] == 1:
                row.append('█')
            else:
                row.append('.')
        print(' '.join(row))
    print()

def unconstrained_agent(maze):
    """P2-only: Greedy BFS ignoring rocks."""
    start = (0, 0)
    queue = deque([(start, [start])])
    visited = set([start])
    dirs = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    
    while queue:
        (r, c), path = queue.popleft()
        if (r, c) == GOAL:
            return path
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if (nr, nc) not in visited:
                    visited.add((nr, nc))
                    queue.append(((nr, nc), path + [(nr, nc)]))
    return None  # No path (shouldn't happen with open maze)

def constrained_agent(maze, rocks):
    """P0-only: Knows rocks but doesn't find safe channels (just avoids rocks)."""
    start = (0, 0)
    queue = deque([(start, [start])])
    visited = set([start])
    dirs = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    
    while queue:
        (r, c), path = queue.popleft()
        if (r, c) == GOAL:
            return path
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if (nr, nc) not in visited and (nr, nc) not in rocks:
                    visited.add((nr, nc))
                    queue.append(((nr, nc), path + [(nr, nc)]))
    return None  # No safe path

def deadband_agent(maze, rocks):
    """Full P0→P1→P2: Map rocks, find safe channels, then optimize."""
    # P0: Map negative space (rocks already known)
    # P1: Identify safe channels (connected free cells)
    start = (0, 0)
    safe = set()
    stack = [start]
    dirs = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    
    while stack:
        r, c = stack.pop()
        if (r, c) in safe:
            continue
        safe.add((r, c))
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if 0 <= nr < MAZE_SIZE and 0 <= nc < MAZE_SIZE:
                if (nr, nc) not in rocks and (nr, nc) not in safe:
                    stack.append((nr, nc))
    
    # If goal not in safe region, no channel exists
    if GOAL not in safe:
        return None
    
    # P2: Optimize within safe channel (BFS on safe set)
    queue = deque([(start, [start])])
    visited = set([start])
    
    while queue:
        (r, c), path = queue.popleft()
        if (r, c) == GOAL:
            return path
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if (nr, nc) in safe and (nr, nc) not in visited:
                visited.add((nr, nc))
                queue.append(((nr, nc), path + [(nr, nc)]))
    return None

def run_trial(trial_num):
    """Run one trial with all three agents."""
    maze, rocks = generate_maze()
    
    # Unconstrained (P2-only)
    path_u = unconstrained_agent(maze)
    success_u = path_u is not None and all(p not in rocks for p in path_u)
    len_u = len(path_u) - 1 if path_u else 0
    
    # Constrained (P0-only)
    path_c = constrained_agent(maze, rocks)
    success_c = path_c is not None
    len_c = len(path_c) - 1 if path_c else 0
    
    # Deadband (P0→P1→P2)
    path_d = deadband_agent(maze, rocks)
    success_d = path_d is not None
    len_d = len(path_d) - 1 if path_d else 0
    
    # Visual output for first trial
    if trial_num == 0:
        print("=== TRIAL 0 — MAZE LAYOUT ===")
        print_maze(maze)
        print("Unconstrained path (P2-only):", "SUCCESS" if success_u else "FAIL")
        if path_u:
            print_maze(maze, path=set(path_u))
        print("Constrained path (P0-only):", "SUCCESS" if success_c else "FAIL")
        if path_c:
            print_maze(maze, path=set(path_c))
        print("Deadband path (P0→P1→P2):", "SUCCESS" if success_d else "FAIL")
        if path_d:
            print_maze(maze, path=set(path_d))
        print()
    
    return success_u, len_u, success_c, len_c, success_d, len_d

#!/usr/bin/env python3
"""
Deadband Protocol Maze Simulator — Tinker's Demo

A visual ASCII simulation of the Deadband Protocol (P0→P1→P2) in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.

Key:
    S = Start
    G = Goal
    # = Rock (negative space, P0)
    ~ = Safe channel (P1)
    · = Untested/unknown
    @ = Agent path (P2 optimization)
    X = Crash (hit rock)
"""

import random
import time
import sys

# Simulation parameters from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64  # 64 rocks in 400 cells = 16% density
OPTIMAL_PATH_LENGTH = 21

class DeadbandMaze:
    def __init__(self):
        self.grid = [['·' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.safe_channels = set()
        self.agent_path = []
        self.crashed = False
        
        # Place rocks (negative space)
        self._place_rocks()
        
        # Mark start and goal
        self.grid[self.start[0]][self.start[1]] = 'S'
        self.grid[self.goal[0]][self.goal[1]] = 'G'
    
    def _place_rocks(self):
        """P0: Map the negative space (rocks)"""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[x][y] = '#'
    
    def find_safe_channels(self):
        """P1: Identify safe channels (BFS from start, avoiding rocks)"""
        visited = set()
        queue = [self.start]
        self.safe_channels.add(self.start)
        
        while queue:
            x, y = queue.pop(0)
            
            # Check all four directions
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in visited and (nx, ny) not in self.rocks:
                        visited.add((nx, ny))
                        self.safe_channels.add((nx, ny))
                        queue.append((nx, ny))
                        if self.grid[nx][ny] == '·':
                            self.grid[nx][ny] = '~'  # Mark safe channel
    
    def greedy_optimize(self):
        """P2-only (VIOLATION): Greedy optimization without P0+P1"""
        x, y = self.start
        path = [(x, y)]
        
        for _ in range(OPTIMAL_PATH_LENGTH * 2):  # Give it extra tries
            if (x, y) == self.goal:
                self.agent_path = path
                return True
            
            # Greedy: move toward goal
            if x < self.goal[0]:
                x += 1
            elif y < self.goal[1]:
                y += 1
            elif x > self.goal[0]:
                x -= 1
            elif y > self.goal[1]:
                y -= 1
            
            # Check for crash
            if (x, y) in self.rocks:
                self.crashed = True
                self.grid[x][y] = 'X'
                self.agent_path = path
                return False
            
            path.append((x, y))
        
        return False
    
    def deadband_optimize(self):
        """P2 WITHIN safe channels: Constrained optimization"""
        if not self.safe_channels:
            return False
        
        # A* search constrained to safe channels
        open_set = {self.start}
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: self._heuristic(self.start)}
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score.get(pos, float('inf')))
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                self.agent_path = list(reversed(path))
                return True
            
            open_set.remove(current)
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # Constraint: must be in safe channels
                if neighbor not in self.safe_channels:
                    continue
                
                tentative_g = g_score[current] + 1
                
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        return False
    
    def _heuristic(self, pos):
        """Manhattan distance to goal"""
        return abs(pos[0] - self.goal[0]) + abs(pos[1] - self.goal[1])
    
    def print_maze(self, title="", show_path=False):
        """Display the current maze state"""
        print(f"\n{'='*60}")
        print(f"DEADBAND PROTOCOL DEMO: {title}")
        print(f"{'='*60}")
        
        display_grid = [row[:] for row in self.grid]
        
        if show_path and self.agent_path:
            for x, y in self.agent_path[1:-1]:  # Skip start and goal
                if display_grid[x][y] not in ['S', 'G', 'X']:
                    display_grid[x][y] = '@'
        
        for row in display_grid:
            print(' '.join(row))
        
        print(f"\nStats: Rocks: {len(self.rocks)} | Safe cells: {len(self.safe_channels)}")
        if self.agent_path:
            print(f"Path length: {len(self.agent_path)-1} (Optimal: {OPTIMAL_PATH_LENGTH})")
        if self.crashed:
            print("STATUS: CRASHED - P2 optimization without P0+P1")

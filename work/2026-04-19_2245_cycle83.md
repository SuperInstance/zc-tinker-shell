class Maze:
    def __init__(self, width: int = 20, height: int = 20, num_rocks: int = 64):
        self.width = width
        self.height = height
        self.num_rocks = num_rocks
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_maze()
    
    def generate_maze(self):
        """Generate random rocks, ensuring start and goal are clear."""
        possible = [(x, y) for x in range(self.width) for y in range(self.height)]
        possible.remove(self.start)
        possible.remove(self.goal)
        
        self.rocks = set(random.sample(possible, self.num_rocks))
    
    def is_rock(self, x: int, y: int) -> bool:
        return (x, y) in self.rocks
    
    def is_valid(self, x: int, y: int) -> bool:
        return 0 <= x < self.width and 0 <= y < self.height
    
    def print_maze(self, path: List[Tuple[int, int]] = None, 
                   deadband: Set[Tuple[int, int]] = None):
        """Print ASCII maze with optional path and deadband visualization."""

            deadband = set()

                    row.append('S')

                    row.append('~')

                    row.append('#')
                else:
                    row.append('.')

class DeadbandAgent:
    def __init__(self, maze: Maze):
        self.maze = maze
        self.visited = set()
        self.deadband = set()  # Safe channels (P1 result)
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify rocks (negative space)."""
        return self.maze.rocks.copy()
    
    def p1_find_safe_channels(self) -> Set[Tuple[int, int]]:
        """P1: Find all reachable safe cells (deadband)."""
        rocks = self.p0_map_negative_space()
        queue = deque([self.maze.start])
        reachable = set([self.maze.start])
        
        while queue:
            x, y = queue.popleft()
            
            # Check all four directions
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if self.maze.is_valid(nx, ny):
                    if (nx, ny) not in rocks and (nx, ny) not in reachable:
                        reachable.add((nx, ny))
                        queue.append((nx, ny))
        
        self.deadband = reachable
        return reachable
    
    def p2_optimize_in_channel(self) -> List[Tuple[int, int]]:
        """P2: Find shortest path within safe channels (deadband)."""
        if not self.deadband:
            self.p1_find_safe_channels()
        
        # BFS within deadband only
        queue = deque([(self.maze.start, [self.maze.start])])
        visited = set([self.maze.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.maze.goal:
                return path
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if self.maze.is_valid(nx, ny):
                    if (nx, ny) in self.deadband and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        return []  # No path found
    
    def run_full_protocol(self) -> Tuple[bool, List[Tuple[int, int]], int]:
        """Execute P0→P1→P2 and return results."""
        # P0: Map rocks (implicit in maze generation)
        rocks = self.p0_map_negative_space()
        
        # P1: Find safe channels
        safe_cells = self.p1_find_safe_channels()
        
        # P2: Optimize within channels
        path = self.p2_optimize_in_channel()
        
        success = len(path) > 0 and path[-1] == self.maze.goal
        return success, path, len(safe_cells)

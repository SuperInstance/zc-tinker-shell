        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]

def p1_find_safe_channels(maze: Maze) -> Set[Tuple[int, int]]:
    """P1: Find safe channels (deadbands) using BFS from start."""
    print("\n[P1] FINDING SAFE CHANNELS")
    
    visited = set()
    queue = [maze.start]
    deadband = set()
    
    while queue:
        x, y = queue.pop(0)
        if (x, y) in visited:
            continue
        
        visited.add((x, y))
        deadband.add((x, y))
        
        # Explore neighbors (up, down, left, right)
        for dx, dy in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
            nx, ny = x + dx, y + dy
            if maze.is_valid(nx, ny) and (nx, ny) not in visited:
                queue.append((nx, ny))
    
    print(f"Found safe channel with {len(deadband)} reachable cells")
    return deadband

## Step 2 — Write the Script (P1: Safe Channel → Implementation)

I'll create a Python script that:
1. Generates a 20×20 maze with 64 rocks (P0: map negative space).
2. Finds all safe paths avoiding rocks (P1: identify safe channels).
3. Selects the shortest safe path (P2: optimize within channel).
4. Prints an ASCII visualization.

File: `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0 → P1 → P2 in a 20×20 maze with 64 rocks.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque

# Constants from simulation data
MAZE_SIZE = 20
ROCK_COUNT = 64
START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

def generate_maze():
    """P0: Map negative space — place rocks randomly."""
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        x = random.randint(0, MAZE_SIZE - 1)
        y = random.randint(0, MAZE_SIZE - 1)
        if (x, y) != START and (x, y) != GOAL:
            rocks.add((x, y))
    return rocks

def find_safe_channels(rocks):
    """P1: Identify safe channels — BFS to find all reachable cells avoiding rocks."""
    visited = [[False] * MAZE_SIZE for _ in range(MAZE_SIZE)]
    for (x, y) in rocks:
        visited[x][y] = True  # mark rocks as visited (blocked)
    
    queue = deque([START])
    sx, sy = START
    visited[sx][sy] = True
    safe_cells = [START]
    
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        x, y = queue.popleft()
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and not visited[nx][ny]:
                visited[nx][ny] = True
                safe_cells.append((nx, ny))
                queue.append((nx, ny))
    
    return safe_cells, visited

def optimize_within_channel(rocks, visited):
    """P2: Optimize within safe channel — shortest path from START to GOAL avoiding rocks."""
    if not visited[GOAL[0]][GOAL[1]]:
        return None  # goal unreachable (should not happen with deadband protocol)
    
    # BFS for shortest path
    parent = {START: None}
    queue = deque([START])
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        x, y = queue.popleft()
        if (x, y) == GOAL:
            break
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and visited[nx][ny] and (nx, ny) not in parent:
                parent[(nx, ny)] = (x, y)
                queue.append((nx, ny))
    
    # Reconstruct path
    if GOAL not in parent:
        return None
    
    path = []
    current = GOAL
    while current is not None:
        path.append(current)
        current = parent[current]
    path.reverse()
    return path

def visualize_maze(rocks, path):
    """ASCII visualization of maze with rocks, path, start, goal."""
    grid = [['·' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    
    # Mark rocks
    for (x, y) in rocks:
        grid[x][y] = '█'
    
    # Mark path
    if path:
        for (x, y) in path[1:-1]:  # exclude start and goal
            grid[x][y] = '○'
    
    # Mark start and goal
    sx, sy = START
    gx, gy = GOAL
    grid[sx][sy] = 'S'
    grid[gx][gy] = 'G'
    
    # Print
    print("20×20 Maze (Deadband Protocol Demo)")
    print("█ = Rock (P0 negative space)")
    print("· = Safe water (P1 channel)")
    print("○ = Optimal path (P2 within channel)")
    print("S = Start, G = Goal\n")
    
    for y in range(MAZE_SIZE):
        row = []
        for x in range(MAZE_SIZE):
            row.append(grid[x][y])
        print(' '.join(row))
    
    if path:
        print(f"\nOptimal path length: {len(path)-1} steps (theoretical optimum: 21)")
    else:
        print("\nNo path found — protocol violation?")

def main():
    print("=== DEADBAND PROTOCOL DEMONSTRATION ===\n")
    
    # P0: Map negative space
    print("P0: Mapping negative space (rocks)...")
    rocks = generate_maze()
    print(f"  → Placed {len(rocks)} rocks at random positions.\n")
    
    # P1: Identify safe channels
    print("P1: Identifying safe channels...")
    safe_cells, visited = find_safe_channels(rocks)
    reachable_goal = visited[GOAL[0]][GOAL[1]]
    print(f"  → Safe cells reachable from start: {len(safe_cells)}")
    print(f"  → Goal reachable? {'YES' if reachable_goal else 'NO'}\n")
    
    if not reachable_goal:
        print("VIOLATION: Goal is in rock space! Regenerate maze.")
        return
    
    # P2: Optimize within channel
    print("P2: Optimizing within safe channel...")
    path = optimize_within_channel(rocks, visited)
    
    if path:
        print(f"  → Found optimal path of {len(path)-1} steps.")
    else:
        print("  → No path found (should not happen).")
    
    print("\n" + "="*50 + "\n")
    
    # Visualization
    visualize_maze(rocks, path)
    
    # Protocol summary
    print("\n" + "="*50)
    print("PROTOCOL SUMMARY:")
    print("  P0 ✓ Mapped 64 rocks (negative space)")
    print("  P1 ✓ Identified safe channels (reachable cells)")
    print("  P2 ✓ Optimized path within safe channel")
    print(f"  Result: {len(path)-1 if path else 'N/A'} steps vs optimal 21")
    print("="*50)

if __name__ == "__main__":
    main()
```

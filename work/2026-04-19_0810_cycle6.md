#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Don't hit rocks (walls)
P1: Find safe channels (paths without rocks)
P2: Optimize within channels (shortest safe path)
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.2):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate maze with rocks (1 = rock, 0 = clear)
        self.rocks = [[1 if random.random() < rock_density else 0 
                      for _ in range(width)] for _ in range(height)]
        # Ensure start and goal are clear
        self.rocks[0][0] = 0
        self.rocks[height-1][width-1] = 0
        
        # P0: Negative space (rocks)
        self.negative_space = [(x, y) for y in range(height) 
                              for x in range(width) if self.rocks[y][x]]
        
        # P1: Safe channels (BFS from start avoiding rocks)
        self.safe_channels = self._find_safe_channels()
        
        # P2: Optimal path within safe channels
        self.optimal_path = self._find_optimal_path()
    
    def _find_safe_channels(self):
        """P1: Find all reachable cells without hitting rocks"""
        visited = [[False] * self.width for _ in range(self.height)]
        queue = deque([self.start])
        visited[0][0] = True
        safe_cells = []
        
        directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        
        while queue:
            x, y = queue.popleft()
            safe_cells.append((x, y))
            
            for dx, dy in directions:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    not visited[ny][nx] and self.rocks[ny][nx] == 0):
                    visited[ny][nx] = True
                    queue.append((nx, ny))
        
        return safe_cells
    
    def _find_optimal_path(self):
        """P2: Find shortest path within safe channels"""
        if self.goal not in self.safe_channels:
            return []  # No safe path to goal
        
        # BFS for shortest path
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                return path
            
            for dx, dy in directions:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) in self.safe_channels and (nx, ny) not in visited):
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        return []
    
    def visualize(self, show_path=True):
        """ASCII visualization of the maze"""
        grid = [[' ' for _ in range(self.width)] for _ in range(self.height)]
        
        # Mark rocks (P0 negative space)
        for x, y in self.negative_space:
            grid[y][x] = '█'
        
        # Mark safe channels (P1)
        for x, y in self.safe_channels:
            if grid[y][x] == ' ':
                grid[y][x] = '·'
        
        # Mark optimal path (P2)
        if show_path and self.optimal_path:
            for x, y in self.optimal_path[1:-1]:  # Skip start and goal
                grid[y][x] = '○'
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        grid[sy][sx] = 'S'
        grid[gy][gx] = 'G'
        
        # Add border
        border = '┌' + '─' * self.width + '┐'
        result = [border]
        for row in grid:
            result.append('│' + ''.join(row) + '│')
        result.append('└' + '─' * self.width + '┘')
        
        return '\n'.join(result)
    
    def stats(self):
        """Calculate statistics matching the constraint simulation"""
        total_cells = self.width * self.height
        rocks_count = len(self.negative_space)
        safe_count = len(self.safe_channels)
        path_length = len(self.optimal_path) - 1 if self.optimal_path else 0
        
        success = 1 if self.goal in self.safe_channels and self.optimal_path else 0
        
        return {
            'rocks': rocks_count,
            'safe_channels': safe_count,
            'optimal_path_length': path_length,
            'success': success,
            'goal_reachable': self.goal in self.safe_channels
        }

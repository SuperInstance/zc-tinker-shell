#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Don't hit rocks (walls)
P1: Find safe channels (paths without walls)
P2: Optimize within channels (shortest safe path)

Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Simulation: 20×20 maze, 50 runs
- Greedy (P2-only): 0/50 success
- Deadband (P0→P1→P2): 50/50 success
"""

import random
from collections import deque

class Maze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        
        # Generate rocks (walls) - P0 negative space
        for y in range(height):
            for x in range(width):
                if (x, y) != self.start and (x, y) != self.goal:
                    if random.random() < rock_density:
                        self.rocks.add((x, y))
        
        # Ensure at least one path exists
        self._ensure_path()
    
    def _ensure_path(self):
        """Make sure there's at least one path from start to goal."""
        visited = set()
        stack = [self.start]
        
        while stack:
            x, y = stack.pop()
            if (x, y) == self.goal:
                return
            
            visited.add((x, y))
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    stack.append((nx, ny))
        
        # If no path, clear a straight line
        for i in range(min(self.width, self.height)):
            if i < self.width:
                self.rocks.discard((i, 0))
            if i < self.height:
                self.rocks.discard((0, i))
    
    def is_rock(self, x, y):
        return (x, y) in self.rocks
    
    def is_valid(self, x, y):
        return (0 <= x < self.width and 0 <= y < self.height)
    
    def render(self, path=None, visited=None):
        """ASCII visualization of maze."""
        result = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif path and (x, y) in path:
                    row.append('*')
                elif visited and (x, y) in visited:
                    row.append('.')
                elif self.is_rock(x, y):
                    row.append('#')
                else:
                    row.append(' ')
            result.append(''.join(row))
        return '\n'.join(result)

class DeadbandAgent:
    """Implements P0→P1→P2 protocol."""
    
    def __init__(self, maze):
        self.maze = maze
        self.visited = set()
        self.safe_channels = set()  # P1 result
        
    def p0_map_rocks(self):
        """P0: Identify negative space (rocks)."""
        return self.maze.rocks.copy()
    
    def p1_find_safe_channels(self):
        """P1: Find all safe cells reachable from start."""
        queue = deque([self.maze.start])
        self.safe_channels = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in self.safe_channels:
                continue
                
            self.safe_channels.add((x, y))
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (self.maze.is_valid(nx, ny) and 
                    not self.maze.is_rock(nx, ny) and
                    (nx, ny) not in self.safe_channels):
                    queue.append((nx, ny))
        
        return self.safe_channels
    
    def p2_optimize_in_channel(self):
        """P2: Find shortest path within safe channels."""
        if self.maze.goal not in self.safe_channels:
            return None  # No safe path to goal
        
        queue = deque([(self.maze.start, [self.maze.start])])
        visited = set()
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.maze.goal:
                return path
            
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if ((nx, ny) in self.safe_channels and 
                    (nx, ny) not in visited):
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        return None

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo - Tinker (Cycle 164)
Demonstrates P0→P1→P2 priority using ASCII maze navigation.
Based on fleet simulation data: 20×20 maze, 50 runs.
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    """Implements the Deadband Protocol for maze navigation."""
    
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        """
        Initialize maze based on fleet simulation parameters.
        From constraint_sim_final.json: rocks=64 in 20×20 grid (density=0.16)
        """
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self.rocks: Set[Tuple[int, int]] = set()
        self._generate_rocks(rock_density)
        
        # P1: Safe channels (deadbands)
        self.safe_channels: List[List[Tuple[int, int]]] = []
        
        # P2: Optimal path
        self.optimal_path: List[Tuple[int, int]] = []
        
        # Statistics
        self.steps_p0 = 0
        self.steps_p1 = 0
        self.steps_p2 = 0
        
    def _generate_rocks(self, density: float):
        """Generate rocks randomly, avoiding start and goal positions."""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        
        # Ensure start and goal are not rocks
        available = [(x, y) for x in range(self.width) for y in range(self.height)
                    if (x, y) not in [self.start, self.goal]]
        
        self.rocks = set(random.sample(available, min(num_rocks, len(available))))
    
    def p0_map_negative_space(self) -> bool:
        """P0: Identify all rocks (what NOT to do)."""
        self.steps_p0 = len(self.rocks)
        return len(self.rocks) > 0
    
    def p1_find_safe_channels(self) -> bool:
        """P1: Find all paths from start to goal avoiding rocks."""
        # Simple BFS to find safe paths
        from collections import deque
        
        queue = deque([(self.start, [self.start])])
        visited = {self.start}
        self.safe_channels = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                self.safe_channels.append(path)
                continue
            
            # Check adjacent cells (right, down, left, up)
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.steps_p1 = len(visited)
        return len(self.safe_channels) > 0
    
    def p2_optimize_within_channels(self) -> bool:
        """P2: Find shortest path among safe channels."""
        if not self.safe_channels:
            return False
        
        # Find shortest path (Manhattan distance as tie-breaker)
        self.optimal_path = min(self.safe_channels, 
                               key=lambda p: (len(p), 
                                             abs(p[-1][0]-self.goal[0]) + abs(p[-1][1]-self.goal[1])))
        self.steps_p2 = len(self.optimal_path)
        return True
    
    def run_protocol(self) -> bool:
        """Execute full Deadband Protocol P0→P1→P2."""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===")
        print(f"Maze: {self.width}×{self.height}, Rocks: {len(self.rocks)}")
        print()
        
        # P0: Map negative space
        print("P0: MAPPING NEGATIVE SPACE")
        if not self.p0_map_negative_space():
            print("ERROR: No rocks identified (unrealistic environment)")
            return False
        print(f"  • Identified {self.steps_p0} rock positions")
        print(f"  • Rocks are states where |failure| > threshold")
        print()
        
        # P1: Find safe channels
        print("P1: FINDING SAFE CHANNELS")
        if not self.p1_find_safe_channels():
            print("ERROR: No safe path exists (all channels blocked)")
            return False
        print(f"  • Found {len(self.safe_channels)} safe channels")
        print(f"  • Channels avoid all P0 rocks")
        print()
        
        # P2: Optimize within channels
        print("P2: OPTIMIZING WITHIN CHANNELS")
        if not self.p2_optimize_within_channels():
            print("ERROR: Cannot optimize without safe channels")
            return False
        print(f"  • Optimal path length: {self.steps_p2} steps")
        print(f"  • Constrained optimization (only within safe channels)")
        print()
        
        return True
    
    def display_maze(self, show_path: bool = True):
        """Display ASCII maze with rocks, path, start, and goal."""
        print("\nMAZE VISUALIZATION:")
        print(f"Start: S at {self.start}, Goal: G at {self.goal}")
        print(f"Rocks: #, Path: ., Empty: ·")
        print()
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('#')
                elif show_path and pos in self.optimal_path:
                    row.append('.')
                else:
                    row.append('·')
            print(' '.join(row))
    
    def display_statistics(self):
        """Display protocol statistics matching fleet simulation."""
        print("\n=== SIMULATION STATISTICS ===")
        print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json:")
        print(f"  Optimal path length: {self.steps_p2}")
        print(f"  Rocks in environment: {len(self.rocks)}")
        print()
        print("Protocol success rates (from 50 runs):")
        print("  • Unconstrained (P2-only): 0/50 (0%)")
        print("  • Deadband (P0→P1→P2): 50/50 (100%)")
        print()
        print("Average steps to goal:")
        print(f"  • Unconstrained: 2320.9 (random wandering)")
        print(f"  • Deadband: {self.steps_p2} (optimal within safe channels)")


def demonstrate_failure_modes():
    """Show what happens when protocol is violated."""
    print("\n" + "="*60)
    print("FAILURE MODE DEMONSTRATION")
    print("="*60)
    
    print("\n1. P2-ONLY (Unconstrained Optimization):")
    print("   • Agent tries to go straight to goal")
    print("   • Hits rocks immediately")
    print("   • Success: 0/50 (from fleet data)")
    
    print("\n2. SKIPPING P0 (No Rock Mapping):")
    print("   • Agent doesn't know what to avoid")
    print("   • Randomly hits rocks")
    print("   • Local minima probability → 1.0")
    
    print("\n3. SKIPPING P1 (No Channel Mapping):")
    print("   • Knows rocks but not safe paths")
    print("   • Gets stuck in dead ends")
    print("   • Cannot reach goal")


def main():
    """Run the Deadband Protocol demonstration."""
    # Initialize with parameters from fleet simulation
    maze = DeadbandMaze(width=20, height=20, rock_density=0.16)
    
    # Run full protocol
    success = maze.run_protocol()
    
    if success:
        # Display results
        maze.display_maze()
        maze.display_statistics()
        demonstrate_failure_modes()

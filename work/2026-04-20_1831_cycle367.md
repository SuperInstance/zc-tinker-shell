## Deadband Protocol Visual Demo (ASCII Maze)

**File:** `deadband_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo - ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# ========== MAZE CONFIG ==========
GRID_SIZE = 10
START = (0, 0)
GOAL = (GRID_SIZE-1, GRID_SIZE-1)
ROCK_DENSITY = 0.2  # 20% rocks

# Symbols
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_MARKER = 'G'
PATH = '○'
SAFE = ' '

# ========== DEADBAND PROTOCOL ==========
class DeadbandAgent:
    def __init__(self, maze: List[List[str]]):
        self.maze = maze
        self.rocks: Set[Tuple[int, int]] = set()      # P0: Negative space
        self.safe: Set[Tuple[int, int]] = set()       # P1: Safe channels
        self.path: List[Tuple[int, int]] = []         # P2: Optimal path
        
    def p0_map_rocks(self):
        """P0: Identify all rocks (negative space)."""
        for y in range(GRID_SIZE):
            for x in range(GRID_SIZE):
                if self.maze[y][x] == ROCK:
                    self.rocks.add((x, y))
        print(f"[P0] Mapped {len(self.rocks)} rocks")
        
    def p1_find_safe_channels(self):
        """P1: Find all reachable safe cells (deadbands)."""
        visited = set()
        stack = [START]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            if (x, y) in self.rocks:
                continue
            if not (0 <= x < GRID_SIZE and 0 <= y < GRID_SIZE):
                continue
                
            self.safe.add((x, y))
            visited.add((x, y))
            
            # 4-directional movement
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                stack.append((nx, ny))
                
        print(f"[P1] Found {len(self.safe)} safe cells (channel width)")
        
    def p2_optimize_in_channel(self):
        """P2: Find shortest path within safe channels (BFS)."""
        if START not in self.safe or GOAL not in self.safe:
            print("[P2] No safe path exists (start or goal blocked)")
            return
            
        queue = [(START, [START])]
        visited = set([START])
        
        while queue:
            (x, y), current_path = queue.pop(0)
            
            if (x, y) == GOAL:
                self.path = current_path
                print(f"[P2] Found optimal path: {len(self.path)} steps")
                return
                
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (nx, ny) in self.safe and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), current_path + [(nx, ny)]))
                    
        print("[P2] No path found within safe channels")
        
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2."""
        print("\n=== DEADBAND PROTOCOL EXECUTION ===")
        self.p0_map_rocks()
        self.p1_find_safe_channels()
        self.p2_optimize_in_channel()
        print("=" * 40)

# ========== VISUALIZATION ==========
def generate_maze() -> List[List[str]]:
    """Generate random maze with rocks."""
    maze = [[EMPTY for _ in range(GRID_SIZE)] for _ in range(GRID_SIZE)]
    
    # Place rocks randomly
    for y in range(GRID_SIZE):
        for x in range(GRID_SIZE):
            if (x, y) != START and (x, y) != GOAL:
                if random.random() < ROCK_DENSITY:
                    maze[y][x] = ROCK
                    
    maze[START[1]][START[0]] = AGENT
    maze[GOAL[1]][GOAL[0]] = GOAL_MARKER
    return maze

def print_maze(maze: List[List[str]], 
               agent: DeadbandAgent = None,
               show_safe: bool = False,
               show_path: bool = False):
    """Print maze with optional overlays."""
    print("\n" + "=" * (GRID_SIZE*2 + 3))
    print("MAZE (A=Start, G=Goal, █=Rock)")
    print("=" * (GRID_SIZE*2 + 3))
    
    for y in range(GRID_SIZE):
        row = []
        for x in range(GRID_SIZE):
            cell = maze[y][x]
            
            if show_path and agent and (x, y) in agent.path:
                cell = PATH
            elif show_safe and agent and (x, y) in agent.safe:
                cell = SAFE
                
            row.append(cell)
        print("│ " + " ".join(row) + " │")
        
    print("=" * (GRID_SIZE*2 + 3))

# ========== MAIN ==========
def main():
    """Run the demonstration."""
    random.seed(42)  # Reproducible for demo
    
    # Generate maze
    maze = generate_maze()
    
    # Create agent
    agent = DeadbandAgent(maze)
    
    # Show initial maze
    print_maze(maze)
    
    # Run Deadband Protocol
    agent.run_protocol()
    
    # Visualize P0: Rocks
    print("\n--- P0: NEGATIVE SPACE (Rocks) ---")
    print("Red cells show what NOT to do")
    # (In ASCII, rocks are already shown)
    
    # Visualize P1: Safe channels
    print("\n--- P1: SAFE CHANNELS (Deadbands) ---")
    print("Empty spaces show where agent CAN move")
    print_maze(maze, agent, show_safe=True)
    
    # Visualize P2: Optimal path
    if agent.path:
        print("\n--- P2: OPTIMAL PATH WITHIN CHANNEL ---")
        print(f"Shortest safe path: {len(agent.path)-1} moves")
        print_maze(maze, agent, show_path=True)
        
        # Statistics
        print("\n=== STATISTICS ===")
        print(f"Grid size: {GRID_SIZE}x{GRID_SIZE}")
        print(f"Rocks: {len(agent.rocks)} ({ROCK_DENSITY*100:.0f}% density)")
        print(f"Safe cells: {len(agent.safe)}")
        print(f"Path length: {len(agent.path)-1} steps")
        print(f"Optimal possible: {abs(GOAL[0]-START[0]) + abs(GOAL[1]-START[1])}")
        
        # Protocol validation
        if any(pos in agent.rocks for pos in agent.path):
            print("⚠️  VIOLATION: Path goes through rocks!")
        else:
            print("✓ Protocol followed: Path stays in safe channels")
    else:
        print("\n❌ No safe path exists - demonstration complete")

if __name__ == "__main__":
    main()

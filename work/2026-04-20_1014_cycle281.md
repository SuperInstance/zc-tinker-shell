#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

A concrete simulation of the Deadband Protocol (P0→P1→P2) in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json.

Doctrine:
    P0: Map negative space (rocks) — what NOT to do.
    P1: Find safe channels (deadbands) — where you CAN be.
    P2: Optimize within channels — best path through safe water.

Violation: Optimizing (P2) before mapping (P0+P1) leads to guaranteed failure.
"""

import random
from typing import List, Tuple, Set, Optional

# ----------------------------------------------------------------------
# Configuration (matches fleet simulation)
# ----------------------------------------------------------------------
MAZE_SIZE = 20
OPTIMAL_PATH_LENGTH = 21  # from constraint_sim_final.json
ROCK_COUNT = 64           # 64 rocks in 400 cells = 16% obstacle density

START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

# Movement directions (4‑connectivity)
DIRECTIONS = [(0, 1), (1, 0), (0, -1), (-1, 0)]

# ----------------------------------------------------------------------
# 1. Environment: Maze with rocks
# ----------------------------------------------------------------------
class Maze:
    def __init__(self):
        self.rocks: Set[Tuple[int, int]] = set()
        self._generate_rocks()
    
    def _generate_rocks(self) -> None:
        """Place rocks randomly, ensuring start and goal are clear."""
        candidates = [
            (x, y)
            for x in range(MAZE_SIZE)
            for y in range(MAZE_SIZE)
            if (x, y) not in (START, GOAL)
        ]
        self.rocks = set(random.sample(candidates, ROCK_COUNT))
    
    def is_rock(self, x: int, y: int) -> bool:
        return (x, y) in self.rocks
    
    def in_bounds(self, x: int, y: int) -> bool:
        return 0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE
    
    def is_safe(self, x: int, y: int) -> bool:
        return self.in_bounds(x, y) and not self.is_rock(x, y)
    
    def render(self, path: Optional[List[Tuple[int, int]]] = None) -> str:
        """ASCII visualization: . = empty, # = rock, S = start, G = goal, * = path."""
        grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        for x, y in self.rocks:
            grid[y][x] = '#'
        
        grid[START[1]][START[0]] = 'S'
        grid[GOAL[1]][GOAL[0]] = 'G'
        
        if path:
            for x, y in path[1:-1]:  # exclude start and goal
                if grid[y][x] == '.':
                    grid[y][x] = '*'
        
        lines = []
        for row in grid:
            lines.append(' '.join(row))
        return '\n'.join(lines)

# 2. Agents

def greedy_optimizer(maze: Maze) -> Optional[List[Tuple[int, int]]]:
    """
    P2‑only agent (VIOLATION).
    Greedy move toward goal, ignoring rock map.
    This corresponds to 'constrained_success': 0 in fleet data.
    """
    path = [START]
    visited = {START}
    
    while path[-1] != GOAL:
        x, y = path[-1]
        # Generate candidate moves sorted by Manhattan distance to goal
        candidates = []
        for dx, dy in DIRECTIONS:
            nx, ny = x + dx, y + dy
            if maze.in_bounds(nx, ny) and (nx, ny) not in visited:
                dist = abs(nx - GOAL[0]) + abs(ny - GOAL[1])
                candidates.append((dist, (nx, ny)))
        
        if not candidates:
            return None  # stuck
        
        candidates.sort()
        _, next_pos = candidates[0]
        path.append(next_pos)
        visited.add(next_pos)
        
        # Early stop if hitting a rock (simulated crash)
        if maze.is_rock(*next_pos):
            return None
    
    return path

def deadband_agent(maze: Maze) -> Optional[List[Tuple[int, int]]]:
    """
    P0→P1→P2 compliant agent.
    P0: Map rocks (negative space).
    P1: Compute safe reachable area (safe channels).
    P2: Find shortest path within safe area.
    """
    # P0: Rocks are already known via maze.rocks
    # P1: Breadth‑first search to find all safe reachable cells
    safe_reachable = set()
    queue = [START]
    safe_reachable.add(START)
    
    while queue:
        x, y = queue.pop(0)
        for dx, dy in DIRECTIONS:
            nx, ny = x + dx, y + dy
            if maze.is_safe(nx, ny) and (nx, ny) not in safe_reachable:
                safe_reachable.add((nx, ny))
                queue.append((nx, ny))
    
    # If goal not reachable, failure
    if GOAL not in safe_reachable:
        return None
    
    # P2: A* search restricted to safe_reachable
    open_set = {START}
    came_from = {}
    g_score = {START: 0}
    f_score = {START: abs(START[0] - GOAL[0]) + abs(START[1] - GOAL[1])}
    
    while open_set:
        current = min(open_set, key=lambda pos: f_score.get(pos, float('inf')))
        if current == GOAL:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(START)
            return path[::-1]
        
        open_set.remove(current)
        x, y = current
        for dx, dy in DIRECTIONS:
            nx, ny = x + dx, y + dy
            neighbor = (nx, ny)
            if neighbor not in safe_reachable:
                continue
            tentative_g = g_score[current] + 1
            if tentative_g < g_score.get(neighbor, float('inf')):
                came_from[neighbor] = current
                g_score[neighbor] = tentative_g
                f_score[neighbor] = tentative_g + abs(nx - GOAL[0]) + abs(ny - GOAL[1])
                if neighbor not in open_set:
                    open_set.add(neighbor)
    
    return None

def run_simulation(num_trials: int = 50) -> dict:
    """
    Runs both agents over many random mazes, collects statistics.
    """
    results = {
        'greedy_success': 0,
        'deadband_success': 0,
        'greedy_path_lengths': [],
        'deadband_path_lengths': [],
    }

            results['greedy_success'] += 1
            results['greedy_path_lengths'].append(len(greedy_path) - 1)  # steps
        
        # Deadband (P0→P1→P2)
        deadband_path = deadband_agent(maze)
        if deadband_path:
            results['deadband_success'] += 1
            results['deadband_path_lengths'].append(len(deadband_path) - 1)

        sum(results['greedy_path_lengths']) / len(results['greedy_path_lengths'])
        if results['greedy_path_lengths'] else 0
    )
    results['deadband_avg'] = (
        sum(results['deadband_path_lengths']) / len(results['deadband_path_lengths'])
        if results['deadband_path_lengths'] else 0
    )
    
    return results

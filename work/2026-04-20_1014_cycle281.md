```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

A visual simulation of the Deadband Protocol (P0→P1→P2) in a 20x20 maze.
Based on fleet-knowledge data:
- constraint_sim_final.json (optimal=21, rocks=64, deadband_success=50/50)
- DEADBAND-PROTOCOL.md (strict P0→P1→P2 priority)

Demonstrates:
P0: Map negative space (rocks/obstacles)
P1: Identify safe channels (deadbands)
P2: Optimize shortest path within safe channels
"""

import random
import heapq
from typing import List, Tuple, Set

# Simulation parameters from fleet data
MAZE_SIZE = 20
OPTIMAL_PATH_LENGTH = 21  # From constraint_sim_final.json
ROCK_COUNT = 64  # 64 rocks in 20x20 maze
START = (0, 0)
GOAL = (19, 19)

class DeadbandMaze:
    def __init__(self):
        self.grid = [[0 for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[Set[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
    def generate_maze(self):
        """Generate maze with rocks (obstacles)"""
        # Place rocks randomly (but ensure path exists)
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != START and (x, y) != GOAL:
                self.rocks.add((x, y))
                self.grid[y][x] = 1  # 1 = rock
        
        # Ensure connectivity by clearing a basic path
        for i in range(MAZE_SIZE):
            if (i, i) != START and (i, i) != GOAL:
                self.rocks.discard((i, i))
                self.grid[i][i] = 0
    
    def p0_map_negative_space(self):
        """P0: Identify rocks (what NOT to do)"""
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Identified {len(self.rocks)} rocks (obstacles)")
        print("Rocks are states where agent would fail.")
        return list(self.rocks)
    
    def p1_find_safe_channels(self):
        """P1: Find safe channels (deadbands) using BFS"""
        print("\n=== P1: FIND SAFE CHANNELS ===")
        
        # Find all reachable positions from start (safe area)
        visited = set()
        queue = [START]
        
        while queue:
            x, y = queue.pop(0)
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Check neighbors
            for dx, dy in [(0,1),(1,0),(0,-1),(-1,0)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        safe_area = visited
        print(f"Safe channel contains {len(safe_area)} reachable positions")
        
        # Verify goal is reachable
        if GOAL not in safe_area:
            print("WARNING: Goal not reachable! Regenerating maze...")
            return False
        
        # Find all possible safe paths (multiple channels possible)
        self.safe_channels = [safe_area]
        return True
    
    def p2_optimize_within_channel(self):
        """P2: Find shortest path within safe channels (A* search)"""
        print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        # A* search only through safe positions
        open_set = []
        heapq.heappush(open_set, (0, START))
        
        came_from = {}
        g_score = {START: 0}
        f_score = {START: self._heuristic(START, GOAL)}
        
        safe_positions = self.safe_channels[0] if self.safe_channels else set()
        
        while open_set:
            _, current = heapq.heappop(open_set)
            
            if current == GOAL:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(START)
                path.reverse()
                self.optimal_path = path
                
                print(f"Found optimal path: {len(path)} steps")
                print(f"Theoretical optimal: {OPTIMAL_PATH_LENGTH} steps")
                return True
            
            x, y = current
            for dx, dy in [(0,1),(1,0),(0,-1),(-1,0)]:
                neighbor = (x + dx, y + dy)
                
                # Constraint: Only move within safe channel
                if neighbor not in safe_positions:
                    continue
                
                tentative_g = g_score[current] + 1
                
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor, GOAL)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
        
        print("No path found within safe channel")
        return False
    
    def _heuristic(self, a: Tuple[int, int], b: Tuple[int, int]) -> int:
        """Manhattan distance heuristic"""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def visualize(self, show_path=False, show_rocks=True):
        """ASCII visualization of maze"""
        print("\n" + "="*42)
        print("MAZE VISUALIZATION")
        print("="*42)
        
        for y in range(MAZE_SIZE):
            row = []
            for x in range(MAZE_SIZE):
                pos = (x, y)
                if pos == START:
                    row.append('S')
                elif pos == GOAL:
                    row.append('G')
                elif show_path and pos in self.optimal_path:
                    row.append('·')
                elif show_rocks and pos in self.rocks:
                    row.append('█')
                else:
                    row.append(' ')
            print('|' + ''.join(row) + '|')
        
        print("="*42)
        print("Legend: S=Start, G=Goal, █=Rock, ·=Optimal Path")
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2"""
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*50)
        
        # Generate environment
        self.generate_maze()
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        
        # P1: Find safe channels
        if not self.p1_find_safe_channels():
            return False
        
        # Visualize after P1
        self.visualize(show_path=False, show_rocks=True)
        
        # P2: Optimize within channel
        if not self.p2_optimize_within_channel():
            return False
        
        # Visualize final result
        self.visualize(show_path=True, show_rocks=True)
        
        # Results summary
        print("\n" + "="*50)
        print("RESULTS SUMMARY")
        print("="*50)
        print(f"P0 Rocks identified: {len(rocks)}")
        print(f"P1 Safe positions: {len(self.safe_channels[0])}")
        print(f"P2 Optimal path length: {len(self.optimal_path)}")
        print(f"Theoretical optimum: {OPTIMAL_PATH_LENGTH}")
        
        success = len(self.optimal_path) >= OPTIMAL_PATH_LENGTH
        print(f"\nProtocol {'SUCCESS' if success else 'FAILED'}")
        
        # Compare with unconstrained approach (what NOT to do)
        print("\n" + "-"*50)
        print("CONTRAST: Unconstrained Greedy Approach (P2-only)")
        print("-"*

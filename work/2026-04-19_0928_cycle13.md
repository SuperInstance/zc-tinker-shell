**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Building.

**Approach:**
1. **P0 (Map Rocks):** Generate a random maze with obstacles ('#') and open cells ('.').
2. **P1 (Find Safe Channels):** Use BFS from start to goal, only traversing open cells, to find all reachable paths (safe channels).
3. **P2 (Optimize Within Channel):** From the set of safe paths, select the shortest one (optimal path within the deadband).
4. **Visualization:** Print the maze with the optimal path marked ('*'), start ('S'), goal ('G'), and rocks ('#').

**Implementation Plan:**
- Use `random` for maze generation.
- Use `collections.deque` for BFS.
- Separate functions for each protocol phase.
- Clear ASCII output.

**Code:**

```python
#!/usr/bin/env python3
"""
deadband_maze_demo.py

A visual demonstration of the Deadband Protocol (P0→P1→P2) in a maze.
P0: Map rocks (obstacles)
P1: Find safe channels (reachable paths avoiding rocks)
P2: Optimize within channel (shortest safe path)
"""

import random
from collections import deque

def generate_maze(rows=10, cols=10, rock_density=0.2):
    """P0: Generate maze with rocks (negative space)."""
    maze = [['.' for _ in range(cols)] for _ in range(rows)]
    rocks = []
    for r in range(rows):
        for c in range(cols):
            if random.random() < rock_density:
                maze[r][c] = '#'
                rocks.append((r, c))
    # Ensure start (0,0) and goal (rows-1, cols-1) are open
    maze[0][0] = 'S'
    maze[rows-1][cols-1] = 'G'
    return maze, rocks

def find_safe_channels(maze, start, goal):
    """P1: BFS to find all reachable paths (safe channels)."""
    rows, cols = len(maze), len(maze[0])
    directions = [(0,1),(1,0),(0,-1),(-1,0)]
    visited = [[False]*cols for _ in range(rows)]
    parent = [[None]*cols for _ in range(rows)]
    queue = deque([start])
    visited[start[0]][start[1]] = True
    safe_cells = []

    while queue:
        r, c = queue.popleft()
        safe_cells.append((r, c))
        if (r, c) == goal:
            break
        for dr, dc in directions:
            nr, nc = r+dr, c+dc
            if 0 <= nr < rows and 0 <= nc < cols:
                if not visited[nr][nc] and maze[nr][nc] != '#':
                    visited[nr][nc] = True
                    parent[nr][nc] = (r, c)
                    queue.append((nr, nc))

    # Reconstruct shortest path if goal reached
    if visited[goal[0]][goal[1]]:
        path = []
        node = goal
        while node != start:
            path.append(node)
            node = parent[node[0]][node[1]]
        path.append(start)
        path.reverse()
        return safe_cells, path
    else:
        return safe_cells, []  # No path to goal

def optimize_within_channel(maze, path):
    """P2: Mark optimal path within safe channel."""
    maze_with_path = [row[:] for row in maze]
    for r, c in path[1:-1]:  # Skip start and goal
        if maze_with_path[r][c] == '.':
            maze_with_path[r][c] = '*'
    return maze_with_path

def print_maze(maze):
    """Visualize maze."""
    for row in maze:
        print(' '.join(row))
    print()

def main():
    print("=== Deadband Protocol Maze Demo ===\n")
    
    # Parameters
    rows, cols = 8, 12
    start = (0, 0)
    goal = (rows-1, cols-1)
    
    # P0: Map rocks
    maze, rocks = generate_maze(rows, cols, rock_density=0.25)
    print("P0: Maze with Rocks (Negative Space)")
    print_maze(maze)
    print(f"Rocks mapped: {len(rocks)} obstacles\n")
    
    # P1: Find safe channels
    safe_cells, shortest_path = find_safe_channels(maze, start, goal)
    print("P1: Safe Channels (Reachable Cells)")
    safe_maze = [row[:] for row in maze]
    for r, c in safe_cells:
        if safe_maze[r][c] == '.':
            safe_maze[r][c] = 'O'  # Mark safe reachable cells
    print_maze(safe_maze)
    print(f"Safe cells found: {len(safe_cells)}")
    if shortest_path:
        print(f"Goal reachable: YES (path length: {len(shortest_path)-1})\n")
    else:
        print("Goal reachable: NO — no safe channel to goal.\n")
        return
    
    # P2: Optimize within channel
    optimal_maze = optimize_within_channel(maze, shortest_path)
    print("P2: Optimal Path Within Safe Channel (* = optimal path)")
    print_maze(optimal_maze)
    print(f"Optimal steps: {len(shortest_path)-1}")
    
    # Summary
    print("\n=== Protocol Summary ===")
    print("P0: Identified rocks (#) — what NOT to touch.")
    print("P1: Mapped safe channels (O) — where we CAN move.")
    print("P2: Optimized path (*) — shortest route within safe channels.")

if __name__ == "__main__":
    main()
```

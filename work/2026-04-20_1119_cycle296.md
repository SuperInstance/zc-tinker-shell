#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 296)
Visualizes P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import sys
import time

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.rocks = set()
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Place rocks (P0 negative space)
        for y in range(height):
            for x in range(width):
                if (x, y) != self.start and (x, y) != self.goal:
                    if random.random() < rock_density:
                        self.grid[y][x] = '#'
                        self.rocks.add((x, y))
        
        # Ensure at least one path exists (guarantee a channel)
        self._ensure_path()
    
    def _ensure_path(self):
        """Guarantee at least one safe channel from start to goal."""
        x, y = self.start
        path = [self.start]
        while (x, y) != self.goal:
            if x < self.goal[0]:
                x += 1
            elif y < self.goal[1]:
                y += 1
            else:
                break
            self.grid[y][x] = '.'  # Clear if rock
            if (x, y) in self.rocks:
                self.rocks.remove((x, y))
            path.append((x, y))
        return path
    
    def display(self, title="MAZE"):
        print(f"\n=== {title} ===")
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
        print()
    
    def p0_map_rocks(self):
        """P0: Identify negative space (what NOT to do)."""
        print("P0 — MAP ROCKS (Negative Space)")
        print(f"Rocks identified: {len(self.rocks)}")
        rock_grid = [['.' for _ in range(self.width)] for _ in range(self.height)]
        for (x, y) in self.rocks:
            rock_grid[y][x] = '#'
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                else:
                    row.append(rock_grid[y][x])
            print(' '.join(row))
        print("→ Rocks are failure states. Do not enter.\n")
        time.sleep(1.5)
    
    def p1_find_channels(self):
        """P1: Find safe channels (where you CAN be)."""
        print("P1 — FIND SAFE CHANNELS")
        # Simple BFS to find all reachable safe cells
        from collections import deque
        visited = set()
        queue = deque([self.start])
        safe = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            safe.add((x, y))
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        # Visualize safe channels
        channel_grid = [['.' for _ in range(self.width)] for _ in range(self.height)]
        for (x, y) in safe:
            channel_grid[y][x] = '~'  # Safe water symbol
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif (x, y) in self.rocks:
                    row.append('#')
                else:
                    row.append(channel_grid[y][x])
            print(' '.join(row))
        
        print(f"→ Safe cells: {len(safe)}. Agent may ONLY operate here.\n")
        time.sleep(1.5)
        return safe
    
    def p2_optimize_path(self, safe_cells):
        """P2: Optimize within safe channels (best path)."""
        print("P2 — OPTIMIZE PATH (Constrained to Safe Channels)")
        # A* constrained to safe cells only
        from heapq import heappush, heappop
        
        def heuristic(a, b):
            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        open_set = []
        heappush(open_set, (0, self.start))
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: heuristic(self.start, self.goal)}
        
        while open_set:
            _, current = heappop(open_set)
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                path.reverse()
                return path
            
            x, y = current
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                neighbor = (nx, ny)
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if neighbor not in safe_cells:
                        continue  # Not in safe channel — skip
                    
                    tentative_g = g_score[current] + 1
                    if neighbor not in g_score or tentative_g < g_score[neighbor]:
                        came_from[neighbor] = current
                        g_score[neighbor] = tentative_g
                        f_score[neighbor] = tentative_g + heuristic(neighbor, self.goal)
                        heappush(open_set, (f_score[neighbor], neighbor))
        
        return []  # No path found
    
    def run_protocol(self):
        """Execute full P0→P1→P2 sequence."""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL DEMO — Tinker (Fleet Cocapn)")
        print("="*60)
        
        self.display("Initial Maze")
        time.sleep(1)
        
        # P0: Map rocks
        self.p0_map_rocks()
        
        # P1: Find safe channels
        safe = self.p1_find_channels()
        
        # P2: Optimize within channels
        path = self.p2_optimize_path(safe)
        
        # Show final path
        print("FINAL PATH (P2 Result)")
        path_grid = [['.' for _ in range(self.width)] for _ in range(self.height)]
        for (x, y) in self.rocks:
            path_grid[y][x] = '#'
        for (x, y) in path:
            path_grid[y][x] = '*'
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                else:
                    row.append(path_grid[y][x])
            print(' '.join(row))
        
        print(f"\nResults:")
        print(f"  - Path length: {len(path)-1} steps")
        print(f"  - Rocks avoided: {len(self.rocks)}")
        print(f"  - Protocol: P0→P1→P2 (strict priority)")
        print("\n" + "="*60)

def main():
    random.seed(42)  # Deterministic for demo
    maze = DeadbandMaze(width=12, height=8, rock_density=0.15)
    maze.run_protocol()

**Plan:**

2. **P1 (Identify safe channels):** Use BFS to find all reachable cells from start (0,0) to goal (19,19) that avoid rocks. These are the safe channels.
3. **P2 (Optimize within channel):** Run A* search constrained to safe channels only to find the shortest path.
4. **Visualization:** Print ASCII maze showing rocks, safe channels, and optimal path.

#!/usr/bin/env python3
"""
deadband_maze.py - Visual demonstration of the Deadband Protocol (P0→P1→P2)
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Optimal path length: 21, Rocks: 64, Maze size: 20x20
"""

import random
from collections import deque
import heapq

# Fixed seed for reproducibility matching fleet simulation
random.seed(42)

MAZE_SIZE = 20
START = (0, 0)
GOAL = (19, 19)
NUM_ROCKS = 64  # From constraint_sim_final.json

def generate_maze():
    """P0: Map negative space - create maze with rocks."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    
    # Place rocks randomly (but ensure start and goal are clear)
    rocks_placed = 0
    while rocks_placed < NUM_ROCKS:
        x = random.randint(0, MAZE_SIZE - 1)
        y = random.randint(0, MAZE_SIZE - 1)
        if (x, y) == START or (x, y) == GOAL:
            continue
        if maze[y][x] != '#':
            maze[y][x] = '#'
            rocks_placed += 1
    
    return maze

def find_safe_channels(maze):
    """P1: Identify safe channels - BFS to find all reachable cells avoiding rocks."""
    visited = [[False for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    queue = deque([START])
    visited[START[1]][START[0]] = True
    
    # 4-directional movement
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    
    while queue:
        x, y = queue.popleft()
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if not visited[ny][nx] and maze[ny][nx] != '#':
                    visited[ny][nx] = True
                    queue.append((nx, ny))
    
    return visited

def deadband_optimize(maze, safe_channels):
    """P2: Optimize within safe channels - A* search constrained to safe cells."""
    if not safe_channels[GOAL[1]][GOAL[0]]:
        return None  # Goal unreachable
    
    # Manhattan distance heuristic
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    # Priority queue: (f_score, g_score, position, path)
    open_set = [(heuristic(START, GOAL), 0, START, [START])]
    closed_set = set()
    
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    
    while open_set:
        f, g, current, path = heapq.heappop(open_set)
        
        if current in closed_set:
            continue
        
        if current == GOAL:
            return path
        
        closed_set.add(current)
        
        for dx, dy in directions:
            nx, ny = current[0] + dx, current[1] + dy
            neighbor = (nx, ny)
            
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if maze[ny][nx] != '#' and safe_channels[ny][nx] and neighbor not in closed_set:
                    new_g = g + 1
                    new_f = new_g + heuristic(neighbor, GOAL)
                    heapq.heappush(open_set, (new_f, new_g, neighbor, path + [neighbor]))
    
    return None

def greedy_unconstrained(maze):
    """P2-only (violation): Greedy search without P0/P1."""
    current = START
    path = [current]
    visited = set([current])
    
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    
    while current != GOAL and len(path) < 100:  # Limit steps
        # Find neighbor closest to goal
        best = None
        best_dist = float('inf')
        
        for dx, dy in directions:
            nx, ny = current[0] + dx, current[1] + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                dist = abs(nx - GOAL[0]) + abs(ny - GOAL[1])
                if dist < best_dist and (nx, ny) not in visited:
                    best = (nx, ny)
                    best_dist = dist
        
        if best is None:
            break  # Stuck
        
        # Check if it's a rock
        if maze[best[1]][best[0]] == '#':
            return None  # Hit a rock
        
        current = best
        path.append(current)
        visited.add(current)
    
    return path if current == GOAL else None

def visualize(maze, safe_channels, deadband_path, greedy_path):
    """Create ASCII visualization."""
    vis = [row[:] for row in maze]
    
    # Mark safe channels (P1)
    for y in range(MAZE_SIZE):
        for x in range(MAZE_SIZE):
            if safe_channels[y][x] and vis[y][x] == '.':
                vis[y][x] = ' '
    
    # Mark deadband path (P2)
    if deadband_path:
        for x, y in deadband_path[1:-1]:  # Skip start and goal
            if vis[y][x] == ' ':
                vis[y][x] = '·'
    
    # Mark greedy path (P2-only)
    if greedy_path:
        for x, y in greedy_path[1:-1]:
            if vis[y][x] == ' ' or vis[y][x] == '·':
                vis[y][x] = 'x'  # Overlap marker
    
    # Mark start and goal
    sx, sy = START
    gx, gy = GOAL
    vis[sy][sx] = 'S'
    vis[gy][gx] = 'G'
    
    return vis

Deadband Protocol Maze Demo — Tinker (Cycle 199)
A visual ASCII demonstration of P0→P1→P2 in a 20×20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import random
from typing import List, Tuple, Set

# Constants from fleet simulation
MAZE_SIZE = 20
ROCK_COUNT = 64          # from "rocks": 64
OPTIMAL_PATH_LEN = 21    # from "optimal": 21

# Symbols for visualization
ROCK = '█'
SAFE = '·'
AGENT = 'A'
GOAL = 'G'
PATH = '○'
DEADBAND = '░'  # safe channel visualization

class DeadbandMaze:
    def __init__(self):
        self.grid = [[SAFE for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        
        # Generate rocks (P0 negative space)
        self._generate_rocks()
        
        # State tracking
        self.safe_channels: List[Set[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
    
    def _generate_rocks(self):
        """P0: Map negative space (rocks)"""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            pos = (x, y)
            # Don't block start or goal
            if pos != self.start and pos != self.goal:
                self.rocks.add(pos)
                self.grid[x][y] = ROCK
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """Step 1: Identify all rocks (failure states)"""

    def p1_find_safe_channels(self):
        """Step 2: Find safe channels (deadbands) using BFS flood fill"""
        print("\n" + "="*60)
        print("P1: FIND SAFE CHANNELS")
        print("="*60)
        
        visited = set()
        channels = []
        
        for x in range(MAZE_SIZE):
            for y in range(MAZE_SIZE):
                pos = (x, y)
                if pos not in self.rocks and pos not in visited:
                    # Flood fill from this safe cell
                    channel = set()
                    stack = [pos]
                    
                    while stack:
                        cx, cy = stack.pop()
                        if (cx, cy) in visited:
                            continue
                        if (cx, cy) in self.rocks:
                            continue
                        
                        visited.add((cx, cy))
                        channel.add((cx, cy))
                        
                        # Check neighbors
                        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                            nx, ny = cx + dx, cy + dy
                            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                                if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                                    stack.append((nx, ny))
                    
                    if channel:
                        channels.append(channel)
        
        self.safe_channels = channels
        print(f"Found {len(channels)} safe channels (deadbands)")
        
        # Visualize the largest safe channel
        if channels:
            largest = max(channels, key=len)
            print(f"Largest channel size: {len(largest)} cells")
            self._print_maze(highlight_channel=largest)

    def _a_star_search(self, channel: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
        """A* search constrained to safe channel"""
        from heapq import heappush, heappop
        
        def heuristic(a, b):
            return abs(a[0]-b[0]) + abs(a[1]-b[1])
        
        open_set = []
        heappush(open_set, (0, self.start))
        
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: heuristic(self.start, self.goal)}
        
        while open_set:
            _, current = heappop(open_set)
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                return list(reversed(path))
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # Constraint: must stay in safe channel
                if neighbor not in channel:
                    continue
                
                tentative_g = g_score[current] + 1
                
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, self.goal)
                    heappush(open_set, (f_score[neighbor], neighbor))
        
        return []

        gx, gy = self.goal
        grid_copy[sx][sy] = AGENT
        grid_copy[gx][gy] = GOAL

                if (x, y) not in (self.start, self.goal):
                    grid_copy[x][y] = DEADBAND
        
        if show_path:
            for x, y in show_path[1:-1]:  # Skip start and goal
                grid_copy[x][y] = PATH
        
        print("   " + "".join(f"{i:2}" for i in range(MAZE_SIZE)))
        for x in range(MAZE_SIZE):
            print(f"{x:2} ", end="")
            for y in range(MAZE_SIZE):
                print(grid_copy[x][y] + " ", end="")
            print()

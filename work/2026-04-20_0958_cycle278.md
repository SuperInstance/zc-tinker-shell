#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 priority in a 20x20 maze with rocks.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
and fleet-knowledge/docs/DEADBAND-PROTOCOL.md
"""

import random
from typing import List, Tuple, Set

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64          # from "rocks": 64
OPTIMAL_PATH_LEN = 21    # from "optimal": 21
GOAL = (MAZE_SIZE-1, MAZE_SIZE-1)

class DeadbandAgent:
    """Implements the Deadband Protocol v1.0"""
    
    def __init__(self, maze: List[List[str]]):
        self.maze = maze
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: Set[Tuple[int, int]] = set()
        self.path = []
        
    def p0_map_negative_space(self):
        """P0: Identify all rocks (failure states)"""
        for y in range(MAZE_SIZE):
            for x in range(MAZE_SIZE):
                if self.maze[y][x] == '█':
                    self.rocks.add((x, y))
        print(f"P0 Complete: Mapped {len(self.rocks)} rocks")
        
    def p1_find_safe_channels(self, start: Tuple[int, int]):
        """P1: Find all reachable cells that are not rocks"""
        visited = set()
        stack = [start]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Only expand if not a rock
            if self.maze[y][x] != '█':
                self.safe_channels.add((x, y))
                # Check neighbors
                for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                    nx, ny = x + dx, y + dy
                    if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                        stack.append((nx, ny))
        
        print(f"P1 Complete: Found {len(self.safe_channels)} safe cells")
        
    def p2_optimize_in_channel(self, start: Tuple[int, int]):
        """P2: Find shortest path to goal within safe channels only"""
        # BFS constrained to safe channels
        queue = [(start, [start])]
        visited = set([start])
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == GOAL:
                self.path = path
                print(f"P2 Complete: Found path length {len(path)} (optimal: {OPTIMAL_PATH_LEN})")
                return True
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                    (nx, ny) not in visited and 
                    (nx, ny) in self.safe_channels):
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        print("P2 Failed: No path to goal within safe channels")
        return False

class GreedyAgent:
    """Naive P2-only optimizer (violates protocol)"""
    
    def __init__(self, maze: List[List[str]]):
        self.maze = maze
        self.path = []
        
    def p2_only_optimize(self, start: Tuple[int, int]):
        """Greedy Manhattan distance optimization (no P0/P1)"""
        x, y = start
        path = [start]
        
        while (x, y) != GOAL:
            # Greedy: move toward goal
            moves = []
            if x < GOAL[0]:
                moves.append((x+1, y))
            if y < GOAL[1]:
                moves.append((x, y+1))
            if not moves:  # Fallback
                if x > 0:
                    moves.append((x-1, y))
                if y > 0:
                    moves.append((x, y-1))
            
            # Pick first move (no rock checking!)
            x, y = moves[0]
            path.append((x, y))
            
            # Check if hit rock
            if self.maze[y][x] == '█':
                print(f"Greedy agent hit rock at ({x}, {y}) after {len(path)} steps")
                self.path = path
                return False
        
        self.path = path
        print(f"Greedy agent reached goal in {len(path)} steps (lucky!)")
        return True

def generate_maze() -> List[List[str]]:
    """Generate 20x20 maze with 64 rocks"""
    maze = [['·' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    
    # Place rocks randomly (avoid start and goal)
    rocks_placed = 0
    while rocks_placed < ROCK_COUNT:
        x = random.randint(0, MAZE_SIZE-1)
        y = random.randint(0, MAZE_SIZE-1)
        if (x, y) != (0, 0) and (x, y) != GOAL and maze[y][x] == '·':
            maze[y][x] = '█'
            rocks_placed += 1
    
    # Ensure at least one path exists (add safe corridor)
    for i in range(MAZE_SIZE):
        if maze[i][i] == '█':
            maze[i][i] = '·'
    
    return maze

def visualize(maze: List[List[str]], 
              deadband_path: List[Tuple[int, int]] = None,
              greedy_path: List[Tuple[int, int]] = None):
    """ASCII visualization of maze and paths"""
    vis = [row[:] for row in maze]
    
    # Mark paths
    if deadband_path:
        for x, y in deadband_path[1:-1]:  # Skip start and goal
            if vis[y][x] == '·':
                vis[y][x] = 'D'
    
    if greedy_path:
        for x, y in greedy_path[1:-1]:
            if vis[y][x] == '·':
                vis[y][x] = 'G'
            elif vis[y][x] == 'D':
                vis[y][x] = 'X'  # Both agents here
    
    # Start and goal
    vis[0][0] = 'S'
    vis[MAZE_SIZE-1][MAZE_SIZE-1] = 'E'
    
    print("\n" + "="*60)
    print("MAZE VISUALIZATION")
    print("="*60)
    print("█ = Rock (P0 negative space)")
    print("D = Deadband agent path (P1 safe channel)")
    print("G = Greedy agent path (P2-only, violates protocol)")
    print("X = Both agents passed here")
    print("S = Start, E = Goal")
    print("-"*60)
    
    for y in range(MAZE_SIZE):
        row = ''
        for x in range(MAZE_SIZE):
            row += vis[y][x] + ' '
        print(row)
    
    print("="*60)

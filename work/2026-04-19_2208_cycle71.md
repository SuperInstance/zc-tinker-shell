**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Based on `constraint_sim_final.json` (20×20 maze, 64 rocks, optimal path length 21). Implementing a visual demo.

**Approach:**
1. **P0 (Map Negative Space):** Generate random rocks (obstacles) in the maze.
2. **P1 (Identify Safe Channels):** Use BFS to find all reachable cells from start, avoiding rocks.
3. **P2 (Optimize Within Channel):** Use A* on the safe reachable graph to find shortest path to goal.

**Output:** A Python script that runs the protocol and prints an ASCII visualization of each step.

## File: `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Based on constraint_sim_final.json parameters:
- 20x20 grid
- 64 rocks (obstacles)
- Optimal path length: 21
Demonstrates P0→P1→P2 strict priority.
"""

import random
from collections import deque
import heapq

GRID_SIZE = 20
NUM_ROCKS = 64
START = (0, 0)
GOAL = (GRID_SIZE - 1, GRID_SIZE - 1)

# Symbols for visualization
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_SYM = 'G'
PATH = '○'
SAFE = '░'  # Safe channel cells

def generate_maze():
    """P0: Map negative space — place rocks randomly."""
    grid = [[EMPTY for _ in range(GRID_SIZE)] for _ in range(GRID_SIZE)]
    rocks_set = set()
    while len(rocks_set) < NUM_ROCKS:
        r = random.randint(0, GRID_SIZE - 1)
        c = random.randint(0, GRID_SIZE - 1)
        if (r, c) != START and (r, c) != GOAL:
            rocks_set.add((r, c))
    for r, c in rocks_set:
        grid[r][c] = ROCK
    return grid, rocks_set

def find_safe_channels(grid, rocks):
    """P1: Identify safe channels — BFS from start, avoiding rocks."""
    safe = set()
    queue = deque([START])
    visited = set([START])
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        r, c = queue.popleft()
        safe.add((r, c))
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < GRID_SIZE and 0 <= nc < GRID_SIZE:
                if (nr, nc) not in visited and (nr, nc) not in rocks:
                    visited.add((nr, nc))
                    queue.append((nr, nc))
    return safe

def optimize_path(grid, safe, rocks):
    """P2: Optimize within channel — A* on safe cells only."""
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, START))
    came_from = {}
    g_score = {START: 0}
    f_score = {START: heuristic(START, GOAL)}
    
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == GOAL:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            return path
        
        for dr, dc in directions:
            nr, nc = current[0] + dr, current[1] + dc
            neighbor = (nr, nc)
            if 0 <= nr < GRID_SIZE and 0 <= nc < GRID_SIZE:
                if neighbor in rocks:
                    continue  # P1 enforcement: never leave safe channel
                if neighbor not in safe:
                    continue  # Not in safe channel
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, GOAL)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    return []  # No path found within safe channels

def visualize(grid, safe=None, path=None):
    """Print ASCII grid with current state."""
    viz = [row[:] for row in grid]
    if safe:
        for r, c in safe:
            if viz[r][c] == EMPTY:
                viz[r][c] = SAFE
    if path:
        for r, c in path:
            if (r, c) != START and (r, c) != GOAL:
                viz[r][c] = PATH
    # Mark start and goal
    sr, sc = START
    gr, gc = GOAL
    viz[sr][sc] = AGENT
    viz[gr][gc] = GOAL_SYM
    
    print('+' + '-' * (GRID_SIZE * 2 - 1) + '+')
    for row in viz:
        print('|' + ' '.join(row) + '|')
    print('+' + '-' * (GRID_SIZE * 2 - 1) + '+')
    print(f"Legend: {AGENT}=Start, {GOAL_SYM}=Goal, {ROCK}=Rock, {SAFE}=Safe, {PATH}=Path")

def main():
    print("=== Deadband Protocol Maze Demo ===")
    print(f"Grid: {GRID_SIZE}x{GRID_SIZE}, Rocks: {NUM_ROCKS}, Optimal: 21 steps")
    print()
    
    # P0: Map negative space
    print("P0: MAP NEGATIVE SPACE (Rocks)")
    grid, rocks = generate_maze()
    visualize(grid)
    print(f"Rocks placed: {len(rocks)}")
    print()
    
    # P1: Identify safe channels
    print("P1: IDENTIFY SAFE CHANNELS (BFS from start)")
    safe = find_safe_channels(grid, rocks)
    visualize(grid, safe=safe)
    print(f"Safe reachable cells: {len(safe)}")
    if GOAL not in safe:
        print("WARNING: Goal not reachable within safe channels — regeneration needed.")
        return
    print()
    
    # P2: Optimize within channel
    print("P2: OPTIMIZE WITHIN CHANNEL (A* on safe cells)")
    path = optimize_path(grid, safe, rocks)
    visualize(grid, safe=safe, path=path)
    if path:
        print(f"Path length: {len(path)} steps")
        print(f"Path: {[START] + path}")
    else:
        print("No path found within safe channels.")
    print()
    
    # Protocol summary
    print("=== Protocol Summary ===")
    print("P0 ✓ Rocks mapped (negative space)")
    print("P1 ✓ Safe channels identified (reachable cells)")
    print("P2 ✓ Optimization constrained to safe channels")
    if path:
        print(f"RESULT: Success — path of {len(path)} steps found.")
    else:
        print("RESULT: Failure — goal unreachable under constraints.")

if __name__ == "__main__":
    main()
```

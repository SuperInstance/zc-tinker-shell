#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 priority in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    """Maze environment with Deadband Protocol implementation"""
    
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (64 rocks = 16% of 400 cells)
        self.rocks: Set[Tuple[int, int]] = set()
        all_cells = [(x, y) for x in range(width) for y in range(height)]
        all_cells.remove(self.start)
        all_cells.remove(self.goal)
        
        rock_cells = random.sample(all_cells, int(width * height * rock_density))
        self.rocks.update(rock_cells)
        
        # Agent state
        self.position = self.start
        self.path = [self.start]
        self.visited = set([self.start])
        
        # Protocol phases
        self.p0_complete = False  # Negative space mapped
        self.p1_complete = False  # Safe channels identified
        self.p2_complete = False  # Optimization complete
        
    def render(self, show_rocks: bool = True, show_path: bool = True) -> str:
        """Render maze as ASCII"""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos == self.position:
                    row.append('A')
                elif show_rocks and pos in self.rocks:
                    row.append('█')
                elif show_path and pos in self.path:
                    row.append('·')
                else:
                    row.append('.')
            grid.append(' '.join(row))
        return '\n'.join(grid)
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify rocks (negative space)"""
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Identified {len(self.rocks)} rocks (failure states)")
        print("Rocks marked with '█' in visualization:")
        print(self.render(show_rocks=True, show_path=False))
        self.p0_complete = True
        return self.rocks.copy()
    
    def p1_find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Find all paths that avoid rocks"""
        print("\n=== P1: IDENTIFY SAFE CHANNELS ===")
        
        # Simple BFS to find safe paths
        safe_paths = []
        queue = [(self.start, [self.start])]
        
        while queue and len(safe_paths) < 3:  # Limit to 3 example paths
            pos, path = queue.pop(0)
            
            if pos == self.goal:
                safe_paths.append(path)
                continue
                
            # Generate safe moves (avoid rocks, stay in bounds)
            x, y = pos
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                new_pos = (x + dx, y + dy)
                if (0 <= new_pos[0] < self.width and 
                    0 <= new_pos[1] < self.height and
                    new_pos not in self.rocks and
                    new_pos not in path):
                    queue.append((new_pos, path + [new_pos]))
        
        print(f"Found {len(safe_paths)} safe channels to goal")
        for i, path in enumerate(safe_paths[:2]):  # Show first 2
            print(f"Channel {i+1}: {len(path)-1} steps")
        
        self.p1_complete = True
        return safe_paths
    
    def p2_optimize_within_channel(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """P2: Find shortest path within safe channels"""
        print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        if not safe_paths:
            print("ERROR: No safe channels found. Cannot optimize.")
            return []
        
        # Find shortest safe path
        optimal_path = min(safe_paths, key=len)
        
        print(f"Selected optimal path: {len(optimal_path)-1} steps")
        print("Path visualization (A=agent, ·=path):")
        
        # Simulate walking the path
        self.path = optimal_path
        for i, pos in enumerate(optimal_path):
            self.position = pos
            if i % 5 == 0 or i == len(optimal_path)-1:  # Show every 5 steps
                print(f"\nStep {i}:")
                print(self.render(show_rocks=False, show_path=True))
        
        self.p2_complete = True
        return optimal_path
    
    def run_greedy_failure(self):
        """Demonstrate P2-only (greedy) failure"""
        print("\n=== GREEDY FAILURE (P2-only) ===")
        print("Attempting to optimize without P0+P1...")
        
        position = self.start
        path = [position]
        visited = set([position])
        
        steps = 0
        max_steps = 100
        
        while position != self.goal and steps < max_steps:
            x, y = position
            
            # Greedy: move toward goal without checking rocks
            if x < self.goal[0]:
                dx, dy = 1, 0
            elif y < self.goal[1]:
                dx, dy = 0, 1
            else:
                dx, dy = -1, 0  # Fallback
            
            new_pos = (x + dx, y + dy)
            
            # Check if hit rock
            if new_pos in self.rocks:
                print(f"Step {steps}: Hit rock at {new_pos}!")
                print("Greedy optimization failed.")
                return False
            
            position = new_pos
            path.append(position)
            steps += 1
        
        return position == self.goal

**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Implementing a proof-of-concept simulation based on the fleet's constraint theory data.

**Approach:**
1. **P0 (Map Negative Space):** Generate a 20x20 maze with random obstacles ("rocks") at 16% density (matching 64 rocks in 400 cells from the data).
2. **P1 (Identify Safe Channels):** Use BFS to find all reachable cells from the start that avoid rocks — these are the deadband channels.
3. **P2 (Optimize Within Channel):** Run A* search constrained to the safe channels to find the shortest path to the goal.
4. **Visualization:** Display the maze with ASCII art, showing rocks, safe channels, and the optimal path.

**Output:** A standalone Python script that runs the simulation and prints results.

---

```python
#!/usr/bin/env python3
"""
deadband_maze.py

A visual demonstration of the Deadband Protocol (P0→P1→P2) using an ASCII maze.
Based on Cocapn fleet constraint theory simulation data.
"""

import random
from collections import deque
import heapq

# ========== CONFIGURATION ==========
MAZE_WIDTH = 20
MAZE_HEIGHT = 20
START = (0, 0)
GOAL = (19, 19)
ROCK_DENSITY = 0.16  # 64 rocks in 400 cells (from constraint_sim_final.json)

# ASCII symbols
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_SYM = 'G'
PATH = '○'
SAFE = '░'  # Light shade for safe channels

# ========== P0: MAP NEGATIVE SPACE ==========
def generate_maze(width, height, rock_density):
    """Create a maze with random rocks (negative space)."""
    maze = [[EMPTY for _ in range(width)] for _ in range(height)]
    rocks = []
    for y in range(height):
        for x in range(width):
            if (x, y) == START or (x, y) == GOAL:
                continue
            if random.random() < rock_density:
                maze[y][x] = ROCK
                rocks.append((x, y))
    return maze, rocks

# ========== P1: IDENTIFY SAFE CHANNELS ==========
def find_safe_channels(maze, start):
    """BFS to find all cells reachable without hitting rocks."""
    width, height = len(maze[0]), len(maze)
    visited = [[False for _ in range(width)] for _ in range(height)]
    queue = deque([start])
    safe_cells = []
    
    while queue:
        x, y = queue.popleft()
        if visited[y][x]:
            continue
        visited[y][x] = True
        if maze[y][x] != ROCK:
            safe_cells.append((x, y))
            # Explore neighbors (4-directional)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < width and 0 <= ny < height and not visited[ny][nx]:
                    queue.append((nx, ny))
    return safe_cells

# ========== P2: OPTIMIZE WITHIN CHANNEL ==========
def a_star_in_channel(maze, start, goal, safe_cells):
    """A* search constrained to safe channels."""
    width, height = len(maze[0]), len(maze)
    safe_set = set(safe_cells)
    
    # Heuristic: Manhattan distance
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            return path
        
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            neighbor = (x + dx, y + dy)
            nx, ny = neighbor
            if 0 <= nx < width and 0 <= ny < height:
                # Constraint: must be in safe channels and not a rock
                if neighbor not in safe_set or maze[ny][nx] == ROCK:
                    continue
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    return []  # No path found

# ========== VISUALIZATION ==========
def print_maze(maze, safe_cells=None, path=None):
    """Print the maze with ASCII art."""
    width, height = len(maze[0]), len(maze)
    display = [row[:] for row in maze]  # Copy
    
    # Mark safe channels (P1)
    if safe_cells:
        for x, y in safe_cells:
            if display[y][x] == EMPTY:
                display[y][x] = SAFE
    
    # Mark path (P2)
    if path:
        for x, y in path:
            if (x, y) != START and (x, y) != GOAL:
                display[y][x] = PATH
    
    # Mark start and goal
    sx, sy = START
    gx, gy = GOAL
    display[sy][sx] = AGENT
    display[gy][gx] = GOAL_SYM
    
    # Print with borders
    print('+' + '-' * width + '+')
    for row in display:
        print('|' + ''.join(row) + '|')
    print('+' + '-' * width + '+')

def print_stats(maze, rocks, safe_cells, path):
    """Print simulation statistics."""
    total_cells = MAZE_WIDTH * MAZE_HEIGHT
    print("\n=== DEADBAND PROTOCOL SIMULATION ===")
    print(f"Maze: {MAZE_WIDTH}x{MAZE_HEIGHT} ({total_cells} cells)")
    print(f"Rocks (P0 negative space): {len(rocks)} ({len(rocks)/total_cells*100:.1f}%)")
    print(f"Safe channels (P1 deadband): {len(safe_cells)} cells")
    print(f"Optimal path length (P2): {len(path) if path else 'NO PATH'}")
    print(f"Success: {'YES' if path else 'NO'}")
    print()

# ========== MAIN SIMULATION ==========
def main():
    print("COCAPN FLEET — DEADBAND PROTOCOL DEMONSTRATION")
    print("Based on constraint_sim_final.json data")
    print("=" * 50)
    
    # P0: Map negative space
    print("\n[P0] MAPPING NEGATIVE SPACE (Rocks)")
    maze, rocks = generate_maze(MAZE_WIDTH, MAZE_HEIGHT, ROCK_DENSITY)
    print_maze(maze)
    print(f"Identified {len(rocks)} rocks (avoid these at all costs)")
    
    # P1: Identify safe channels
    print("\n[P1] IDENTIFYING SAFE CHANNELS (Deadband)")
    safe_cells = find_safe_channels(maze, START)
    print_maze(maze, safe_cells)
    print(f"Mapped {len(safe_cells)} safe cells (operate only within these)")
    
    # P2: Optimize within channel
    print("\n[P2] OPTIMIZING WITHIN SAFE CHANNELS")
    path = a_star_in_channel(maze, START, GOAL, safe_cells)
    print_maze(maze, safe_cells, path)
    
    # Statistics
    print_stats(maze, rocks, safe_cells, path)
    
    # Protocol validation
    if path:
        print("PROTOCOL SUCCESS: P0→P1→P2 completed.")
        print("Agent reached goal without hitting rocks.")
    else:
        print("PROTOCOL FAILURE: No safe path exists.")
        print("This is a P0 constraint — goal is unreachable.")

if __name__ == "__main__":
    main()

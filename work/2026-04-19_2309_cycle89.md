#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0→P1→P2 protocol in a 10x10 ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random

def generate_maze(width=10, height=10, rock_density=0.2):
    """Generate a random maze with start (S), goal (G), and rocks (#)."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place start (top-left) and goal (bottom-right)
    start = (0, 0)
    goal = (height-1, width-1)
    maze[start[0]][start[1]] = 'S'
    maze[goal[0]][goal[1]] = 'G'
    
    # Place rocks randomly, avoiding start and goal
    rocks = 0
    for r in range(height):
        for c in range(width):
            if (r, c) == start or (r, c) == goal:
                continue
            if random.random() < rock_density:
                maze[r][c] = '#'
                rocks += 1
    
    return maze, start, goal, rocks

def print_maze(maze, path=None):
    """Print maze with optional path overlay."""
    for r in range(len(maze)):
        row = ''
        for c in range(len(maze[0])):
            if path and (r, c) in path:
                row += '○'  # path marker
            else:
                row += maze[r][c]
            row += ' '
        print(row)

def p0_map_negative_space(maze):
    """P0: Identify rocks (negative space)."""
    rocks = []
    for r in range(len(maze)):
        for c in range(len(maze[0])):
            if maze[r][c] == '#':
                rocks.append((r, c))
    return rocks

def p1_find_safe_channels(maze, start, goal, rocks):
    """P1: Find safe paths using BFS, avoiding rocks."""
    from collections import deque
    
    height, width = len(maze), len(maze[0])
    visited = [[False] * width for _ in range(height)]
    parent = [[None] * width for _ in range(height)]
    
    # Mark rocks as visited (blocked)
    for r, c in rocks:
        visited[r][c] = True
    
    queue = deque([start])
    visited[start[0]][start[1]] = True
    
    # BFS directions: down, right, up, left (prioritize toward goal)
    dirs = [(1, 0), (0, 1), (-1, 0), (0, -1)]
    
    while queue:
        r, c = queue.popleft()
        
        if (r, c) == goal:
            # Reconstruct path
            path = []
            while (r, c) != start:
                path.append((r, c))
                r, c = parent[r][c]
            path.reverse()
            return path
        
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if 0 <= nr < height and 0 <= nc < width and not visited[nr][nc]:
                visited[nr][nc] = True
                parent[nr][nc] = (r, c)
                queue.append((nr, nc))
    
    return None  # No safe path found

def p2_optimize_within_channel(path):
    """P2: Optimize path (simple example: remove unnecessary detours)."""
    if not path or len(path) <= 2:
        return path
    
    # Simple optimization: keep only cardinal direction changes
    optimized = [path[0]]
    for i in range(1, len(path)-1):
        r1, c1 = optimized[-1]
        r2, c2 = path[i]
        r3, c3 = path[i+1]
        
        # Keep if direction changes
        if (r2 - r1, c2 - c1) != (r3 - r2, c3 - c2):
            optimized.append(path[i])
    
    optimized.append(path[-1])
    return optimized

def deadband_protocol_demo():
    """Run the full P0→P1→P2 demonstration."""
    print("=" * 50)
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("=" * 50)
    
    # Generate maze
    maze, start, goal, rocks = generate_maze()
    print("\n[MAZE]")
    print(f"Start: {start}, Goal: {goal}, Rocks: {rocks}")
    print_maze(maze)
    
    # P0: Map negative space
    print("\n[P0: MAP NEGATIVE SPACE]")
    rock_positions = p0_map_negative_space(maze)
    print(f"Identified {len(rock_positions)} rocks at positions:")
    for i, (r, c) in enumerate(rock_positions[:5]):  # Show first 5
        print(f"  ({r}, {c})", end='')
    if len(rock_positions) > 5:
        print(f" ... and {len(rock_positions)-5} more")
    else:
        print()
    
    # P1: Find safe channels
    print("\n[P1: FIND SAFE CHANNELS]")
    safe_path = p1_find_safe_channels(maze, start, goal, rock_positions)
    if safe_path:
        print(f"Found safe path with {len(safe_path)} steps")
        print_maze(maze, set(safe_path))
    else:
        print("NO SAFE PATH FOUND (maze impossible)")
        return
    
    # P2: Optimize within channel
    print("\n[P2: OPTIMIZE WITHIN CHANNEL]")
    optimized_path = p2_optimize_within_channel(safe_path)
    if len(optimized_path) < len(safe_path):
        print(f"Optimized from {len(safe_path)} to {len(optimized_path)} steps")
        print_maze(maze, set(optimized_path))
    else:
        print("Path already optimal")
    
    # Results
    print("\n" + "=" * 50)
    print("RESULTS")
    print("=" * 50)
    print(f"Start → Goal distance (Manhattan): {abs(goal[0]-start[0]) + abs(goal[1]-start[1])}")
    print(f"Safe path length: {len(safe_path)}")
    print(f"Optimized path length: {len(optimized_path)}")
    print(f"Rocks avoided: {len(rock_positions)}")
    
    # Protocol validation
    print("\nPROTOCOL VALIDATION:")
    if rock_positions and safe_path:
        print("✓ P0 completed: rocks mapped")
        print("✓ P1 completed: safe channel found")
        print("✓ P2 completed: optimization within channel")
        print("✓ DEADBAND PROTOCOL FOLLOWED")
    else:
        print("✗ Protocol violation detected")

def greedy_failure_demo():
    """Show what happens when you skip to P2 (greedy optimization)."""
    print("\n" + "=" * 50)
    print("GREEDY FAILURE DEMO (Skipping P0+P1)")
    print("=" * 50)
    
    maze, start, goal, rocks = generate_maze(rock_density=0.25)
    print("\n[Maze with 25% rocks]")
    print_maze(maze)
    
    # Greedy approach: straight line toward goal, ignoring rocks
    print("\n[Greedy path (P2-only)]")
    path = []
    r, c = start
    while (r, c) != goal:
        # Move toward goal
        if r < goal[0]:
            r += 1
        elif r > goal[0]:
            r -= 1
        elif c < goal[1]:
            c += 1
        elif c > goal[1]:
            c -= 1
        
        path.append((r, c))
        if maze[r][c] == '#':
            print(f"  HIT ROCK at ({r}, {c}) after {len(path)} steps!")
            print_maze(maze, set(path))
            return
    
    print(f"  Reached goal in {len(path)} steps (lucky, no rocks in line)")
    print_maze(maze, set(path))

if __name__ == "__main__":
    random.seed(42)  # Reproducible for training
    deadband_protocol_demo()
    greedy_failure_demo()

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Simulates P0→P1→P2 vs. other strategies in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# Constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

def generate_maze() -> Tuple[List[List[str]], Set[Tuple[int, int]]]:
    """Create a 20x20 maze with start (0,0), goal (19,19), and random rocks."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    maze[0][0] = 'S'
    maze[MAZE_SIZE-1][MAZE_SIZE-1] = 'G'
    
    rocks = set()
    while len(rocks) < ROCK_COUNT:
        x = random.randint(0, MAZE_SIZE-1)
        y = random.randint(0, MAZE_SIZE-1)
        if (x, y) not in [(0,0), (MAZE_SIZE-1, MAZE_SIZE-1)]:
            rocks.add((x, y))
            maze[y][x] = '#'
    
    return maze, rocks

def print_maze(maze: List[List[str]], path: List[Tuple[int, int]] = None):
    """Print maze with optional path marked."""
    for y in range(MAZE_SIZE):
        row = []
        for x in range(MAZE_SIZE):
            if path and (x, y) in path:
                row.append('*')
            else:
                row.append(maze[y][x])
        print(' '.join(row))
    print()

def unconstrained_greedy(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> Tuple[bool, List[Tuple[int, int]], int]:
    """P2-only: Greedy move toward goal, ignoring rocks."""
    x, y = 0, 0
    path = [(0, 0)]
    steps = 0
    
    while (x, y) != (MAZE_SIZE-1, MAZE_SIZE-1) and steps < 1000:
        # Move right or down randomly toward goal
        if random.random() < 0.5 and x < MAZE_SIZE-1:
            x += 1
        elif y < MAZE_SIZE-1:
            y += 1
        elif x < MAZE_SIZE-1:
            x += 1
        
        path.append((x, y))
        steps += 1
        
        if (x, y) in rocks:
            return False, path, steps  # Hit a rock
    
    success = (x, y) == (MAZE_SIZE-1, MAZE_SIZE-1)
    return success, path, steps

def constrained_naive(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> Tuple[bool, List[Tuple[int, int]], int]:
    """Constrained but without P0 map: avoids rocks if seen, but no planning."""
    x, y = 0, 0
    path = [(0, 0)]
    steps = 0
    
    while (x, y) != (MAZE_SIZE-1, MAZE_SIZE-1) and steps < 1000:
        # Look at possible next moves
        candidates = []
        if x < MAZE_SIZE-1 and (x+1, y) not in rocks:
            candidates.append((x+1, y))
        if y < MAZE_SIZE-1 and (x, y+1) not in rocks:
            candidates.append((x, y+1))
        
        if not candidates:
            return False, path, steps  # Stuck
        
        x, y = random.choice(candidates)
        path.append((x, y))
        steps += 1
    
    success = (x, y) == (MAZE_SIZE-1, MAZE_SIZE-1)
    return success, path, steps

def deadband_protocol(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> Tuple[bool, List[Tuple[int, int]], int]:
    """P0→P1→P2: Map rocks, find safe channels, then optimal path."""
    # P0: Map negative space (rocks)
    # Already have rocks set
    
    # P1: Find safe channels (BFS to find all reachable positions avoiding rocks)
    from collections import deque
    
    start = (0, 0)
    goal = (MAZE_SIZE-1, MAZE_SIZE-1)
    
    queue = deque([start])
    visited = {start}
    parent = {start: None}
    
    while queue:
        x, y = queue.popleft()
        
        if (x, y) == goal:
            break
        
        for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    parent[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
    
    # Check if goal reachable
    if goal not in visited:
        return False, [], 0
    
    # P2: Reconstruct optimal path within safe channel
    path = []
    current = goal
    while current is not None:
        path.append(current)
        current = parent[current]
    path.reverse()
    
    success = True
    steps = len(path) - 1  # Start doesn't count as step
    
    return success, path, steps

def run_simulation(runs: int = 50):
    """Run all three strategies multiple times and collect stats."""
    strategies = [
        ("Unconstrained (P2-only)", unconstrained_greedy),
        ("Constrained (naive)", constrained_naive),
        ("Deadband (P0→P1→P2)", deadband_protocol)
    ]
    
    results = {name: {"success": 0, "total_steps": 0, "paths": []} for name, _ in strategies}
    
    for run in range(runs):
        maze, rocks = generate_maze()
        
        for name, strategy in strategies:
            success, path, steps = strategy(maze, rocks)
            if success:
                results[name]["success"] += 1
                results[name]["total_steps"] += steps
            results[name]["paths"].append(path)
    
    print("=" * 60)
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}, Optimal: {OPTIMAL_PATH_LENGTH} steps")
    print(f"Simulation runs: {runs}")
    print("=" * 60)
    
    for name in results:
        success = results[name]["success"]
        avg_steps = results[name]["total_steps"] / success if success > 0 else 0
        print(f"\n{name}:")
        print(f"  Success: {success}/{runs} ({success/runs*100:.1f}%)")
        print(f"  Avg steps when successful: {avg_steps:.1f}")
    
    # Show one example maze with all three paths
    print("\n" + "=" * 60)
    print("EXAMPLE MAZE (Run 1):")
    maze, rocks = generate_maze()
    print_maze(maze)
    
    for name, strategy in strategies:
        success, path, steps = strategy(maze, rocks)
        print(f"{name}: {'SUCCESS' if success else 'FAIL'} ({steps} steps)")
        if path:
            print_maze(maze, path)

if __name__ == "__main__":
    random.seed(42)  # Reproducible for demo
    run_simulation(50)
```

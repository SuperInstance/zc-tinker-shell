#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Don't hit rocks (walls)
P1: Find safe channels (paths)
P2: Optimize within channels (shortest safe path)
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.2):
        self.width = width
        self.height = height
        self.grid = []
        self.start = (1, 1)
        self.goal = (width-2, height-2)
        self._generate(rock_density)
    
    def _generate(self, rock_density):
        """Generate maze with walls (rocks) and open cells."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if x == 0 or x == self.width-1 or y == 0 or y == self.height-1:
                    row.append('#')  # border wall
                elif random.random() < rock_density:
                    row.append('#')  # rock
                else:
                    row.append('.')  # open
            self.grid.append(row)
        
        # Ensure start and goal are open
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def display(self, path=None):
        """Print maze with optional path marked."""
        for y in range(self.height):
            line = []
            for x in range(self.width):
                cell = self.grid[y][x]
                if path and (x, y) in path:
                    line.append('*')
                else:
                    line.append(cell)
            print(''.join(line))
        print()
    
    def p0_map_rocks(self):
        """P0: Identify negative space (rocks/walls)."""
        rocks = []
        for y in range(self.height):
            for x in range(self.width):
                if self.grid[y][x] == '#':
                    rocks.append((x, y))
        return rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all safe paths (deadbands) using BFS."""
        rocks = set(self.p0_map_rocks())
        directions = [(0,1),(1,0),(0,-1),(-1,0)]
        
        # BFS to find all reachable safe cells
        visited = set()
        queue = deque([self.start])
        safe_cells = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            safe_cells.add((x, y))
            
            for dx, dy in directions:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in rocks and (nx, ny) not in visited:
                        queue.append((nx, ny))
        
        return safe_cells
    
    def p2_optimize_in_channel(self):
        """P2: Find shortest path within safe channels."""
        safe = self.p1_find_safe_channels()
        if self.goal not in safe:
            return None  # Goal unreachable
        
        directions = [(0,1),(1,0),(0,-1),(-1,0)]
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.popleft()
            if (x, y) == self.goal:
                return path
            
            for dx, dy in directions:
                nx, ny = x + dx, y + dy
                if (nx, ny) in safe and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        return None

def run_demo():
    """Run complete Deadband Protocol demonstration."""
    print("=== DEADBAND PROTOCOL MAZE DEMO ===\n")
    
    # Create maze matching simulation parameters (20x20)
    maze = DeadbandMaze(width=20, height=20, rock_density=0.16)  # ~64 rocks
    
    print("1. MAZE (S=start, G=goal, #=rock, .=open):")
    maze.display()
    
    print("2. P0: MAP ROCKS (negative space)")
    rocks = maze.p0_map_rocks()
    print(f"   Found {len(rocks)} rocks/walls to avoid\n")
    
    print("3. P1: FIND SAFE CHANNELS (deadbands)")
    safe = maze.p1_find_safe_channels()
    print(f"   {len(safe)} reachable safe cells")
    if maze.goal in safe:
        print("   ✓ Goal is reachable within safe channels")
    else:
        print("   ✗ Goal is NOT reachable - no safe channel")
        return
    
    print("\n4. P2: OPTIMIZE WITHIN CHANNEL (shortest safe path)")
    path = maze.p2_optimize_in_channel()
    if path:
        print(f"   Found optimal path: {len(path)-1} steps")
        print("\n5. FINAL PATH VISUALIZATION (* = path):")
        maze.display(set(path))
        
        # Protocol validation
        print("6. PROTOCOL VALIDATION:")
        path_cells = set(path)
        rock_cells = set(rocks)
        if not path_cells.intersection(rock_cells):
            print("   ✓ P0 respected: No path cells intersect rocks")
        if path_cells.issubset(safe):
            print("   ✓ P1 respected: All path cells within safe channels")
        print(f"   ✓ P2 achieved: Optimal path length = {len(path)-1}")
    else:
        print("   ✗ No path found (should not happen if goal is reachable)")

if __name__ == "__main__":
    run_demo()

def generate_maze():
    """Create a maze with rocks (obstacles) and open cells."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    rocks = set()

        c = random.randint(0, MAZE_SIZE-1)

            rocks.add((r, c))
            maze[r][c] = '#'
    
    maze[START[0]][START[1]] = 'S'
    maze[GOAL[0]][GOAL[1]] = 'G'
    return maze, rocks

def print_maze(maze, path=None, title=""):

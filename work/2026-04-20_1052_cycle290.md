#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo - Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on constraint_sim_final.json results.
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=10, height=10, rock_density=0.3):
        self.width = width
        self.height = height
        self.grid = []
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        self.safe_channels = []
        self.optimal_path = []
        
        # Generate maze
        self._generate_maze(rock_density)
    
    def _generate_maze(self, rock_density):
        """Generate random maze with rocks as obstacles."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif random.random() < rock_density:
                    row.append('█')  # Rock
                    self.rocks.add((x, y))
                else:
                    row.append('.')  # Open space
            self.grid.append(row)
    
    def p0_map_rocks(self):
        """P0: Map negative space (rocks)."""
        print("=== P0: MAP ROCKS ===")
        print(f"Identified {len(self.rocks)} rock positions")
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) in self.rocks:
                    print('█', end=' ')
                else:
                    print('.', end=' ')
            print()
        print()
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all safe channels (paths avoiding rocks)."""
        print("=== P1: FIND SAFE CHANNELS ===")
        
        # BFS to find all reachable positions
        visited = set()
        queue = deque([self.start])
        safe_positions = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            safe_positions.add((x, y))
            
            # Check neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    queue.append((nx, ny))
        
        # Visualize safe channels
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) == self.start:
                    print('S', end=' ')
                elif (x, y) == self.goal:
                    print('G', end=' ')
                elif (x, y) in self.rocks:
                    print('█', end=' ')
                elif (x, y) in safe_positions:
                    print('·', end=' ')  # Safe channel
                else:
                    print(' ', end=' ')  # Unreachable
            print()
        
        self.safe_positions = safe_positions
        print(f"Found {len(safe_positions)} safe positions")
        print()
        return safe_positions
    
    def p2_optimize_path(self):
        """P2: Find optimal path within safe channels."""
        print("=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        if not hasattr(self, 'safe_positions'):
            print("ERROR: Must run P1 first!")
            return []
        
        # A* search within safe positions
        open_set = {self.start}
        came_from = {}
        g_score = {pos: float('inf') for pos in self.safe_positions}
        g_score[self.start] = 0
        
        f_score = {pos: float('inf') for pos in self.safe_positions}
        f_score[self.start] = self._heuristic(self.start, self.goal)
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score[pos])
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                path.reverse()
                self.optimal_path = path
                break
            
            open_set.remove(current)
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                if neighbor not in self.safe_positions:
                    continue
                
                tentative_g = g_score[current] + 1
                if tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor, self.goal)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        # Visualize optimal path
        path_set = set(self.optimal_path) if self.optimal_path else set()
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) == self.start:
                    print('S', end=' ')
                elif (x, y) == self.goal:
                    print('G', end=' ')
                elif (x, y) in self.rocks:
                    print('█', end=' ')
                elif (x, y) in path_set:
                    print('★', end=' ')  # Optimal path
                elif (x, y) in self.safe_positions:
                    print('·', end=' ')  # Safe but not optimal
                else:
                    print(' ', end=' ')
            print()
        
        if self.optimal_path:
            print(f"Optimal path length: {len(self.optimal_path)-1} steps")
            print(f"Path: {self.optimal_path}")
        else:
            print("No path found to goal!")
        print()
        return self.optimal_path
    
    def _heuristic(self, a, b):
        """Manhattan distance heuristic."""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2."""
        print("DEADBAND PROTOCOL DEMONSTRATION")

        # P2: Optimize path
        path = self.p2_optimize_path()
        
        # Summary
        print("=== PROTOCOL SUMMARY ===")
        print(f"P0 Rocks mapped: {len(rocks)}")
        print(f"P1 Safe positions: {len(safe)}")
        print(f"P2 Path found: {'Yes' if path else 'No'}")
        if path:
            print(f"Path cost: {len(path)-1} (optimal within constraints)")
        
        # Simulate failure of skipping P0/P1
        print("\n=== WHAT IF WE SKIP P0/P1? ===")
        print("Greedy approach (P2-only):")
        # Try direct path ignoring rocks
        direct_path = []
        x, y = self.start
        while (x, y) != self.goal:
            if x < self.goal[0]:
                x += 1
            elif y < self.goal[1]:
                y += 1
            direct_path.append((x, y))
            if (x, y) in self.rocks:
                print(f"  Hit rock at {x},{y} after {len(direct_path)} steps!")
                break
        if (x, y) == self.goal:
            print(f"  Reached goal in {len(direct_path)} steps (lucky!)")
        else:
            print("  Failed - optimized into rocks")

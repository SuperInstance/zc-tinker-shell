#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 priority in a simple ASCII maze.
Based on Cocapn fleet's constraint theory and DEADBAND-PROTOCOL.md.
"""

import random
from collections import deque
from typing import List, Tuple, Set

# ========== MAZE GENERATION ==========
def generate_maze(width: int, height: int, rock_density: float = 0.2) -> List[List[str]]:
    """Generate a random maze with start, goal, and rocks."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place start (top-left) and goal (bottom-right)
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    
    # Place rocks randomly
    for y in range(height):
        for x in range(width):
            if maze[y][x] in ['S', 'G']:
                continue
            if random.random() < rock_density:
                maze[y][x] = '#'
    
    return maze

def print_maze(maze: List[List[str]], title: str = ""):
    """Print maze with title."""
    if title:
        print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

# ========== AGENT STRATEGIES ==========
def find_path_bfs(maze: List[List[str]], avoid: Set[Tuple[int, int]] = None) -> List[Tuple[int, int]]:
    """BFS pathfinding from S to G, avoiding given coordinates."""
    if avoid is None:
        avoid = set()
    
    height, width = len(maze), len(maze[0])
    start = (0, 0)
    goal = (height-1, width-1)
    
    queue = deque([start])
    parent = {start: None}
    visited = set([start])
    
    while queue:
        y, x = queue.popleft()
        if (y, x) == goal:
            # Reconstruct path
            path = []
            while (y, x) != start:
                path.append((y, x))
                y, x = parent[(y, x)]
            path.reverse()
            return path
        
        for dy, dx in [(0,1),(1,0),(0,-1),(-1,0)]:
            ny, nx = y + dy, x + dx
            if 0 <= ny < height and 0 <= nx < width:
                if (ny, nx) in avoid:
                    continue
                if maze[ny][nx] == '#':
                    continue
                if (ny, nx) not in visited:
                    visited.add((ny, nx))
                    parent[(ny, nx)] = (y, x)
                    queue.append((ny, nx))
    
    return []  # No path found

def unconstrained_agent(maze: List[List[str]]) -> Tuple[List[Tuple[int, int]], bool]:
    """P2-only agent: greedy optimization ignoring rocks."""
    print("P2-ONLY AGENT (Unconstrained Optimization)")
    print("  → Skips P0 (rock mapping) and P1 (channel finding)")
    print("  → Directly optimizes for shortest path")
    
    path = find_path_bfs(maze, avoid=set())
    success = len(path) > 0
    return path, success

def constrained_agent(maze: List[List[str]]) -> Tuple[List[Tuple[int, int]], bool]:
    """P0+P2 agent: knows rocks but doesn't map safe channels."""
    print("\nCONSTRAINED AGENT (P0+P2, missing P1)")
    print("  → P0: Identifies rocks")
    print("  → Skips P1 (channel mapping)")
    print("  → P2: Optimizes with rock avoidance")
    
    # P0: Map rocks
    rocks = set()
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == '#':
                rocks.add((y, x))
    
    # P2: Optimize avoiding rocks
    path = find_path_bfs(maze, avoid=rocks)
    success = len(path) > 0
    return path, success

def deadband_agent(maze: List[List[str]]) -> Tuple[List[Tuple[int, int]], bool]:
    """Full P0→P1→P2 agent: maps rocks, finds channels, optimizes within."""
    print("\nDEADBAND AGENT (P0→P1→P2 Protocol)")
    print("  → P0: Map negative space (rocks)")
    print("  → P1: Identify safe channels")
    print("  → P2: Optimize within channels")
    
    height, width = len(maze), len(maze[0])
    
    # P0: Map rocks
    rocks = set()
    for y in range(height):
        for x in range(width):
            if maze[y][x] == '#':
                rocks.add((y, x))
    
    # P1: Find safe channels (connected components without rocks)
    def flood_fill(start: Tuple[int, int], blocked: Set[Tuple[int, int]]) -> Set[Tuple[int, int]]:
        """Return all reachable cells from start avoiding blocked cells."""
        stack = [start]
        component = set()
        while stack:
            y, x = stack.pop()
            if (y, x) in component or (y, x) in blocked:
                continue
            component.add((y, x))
            for dy, dx in [(0,1),(1,0),(0,-1),(-1,0)]:
                ny, nx = y + dy, x + dx
                if 0 <= ny < height and 0 <= nx < width:
                    stack.append((ny, nx))
        return component
    
    safe_channels = flood_fill((0, 0), rocks)
    
    # Ensure goal is in safe channels
    goal = (height-1, width-1)
    if goal not in safe_channels:
        return [], False
    
    # P2: Optimize path within safe channels
    path = find_path_bfs(maze, avoid=set([c for c in safe_channels]))
    success = len(path) > 0
    return path, success

def mark_path(maze: List[List[str]], path: List[Tuple[int, int]], marker: str = '○'):
    """Mark path on maze copy."""
    maze_copy = [row[:] for row in maze]
    for y, x in path:
        if maze_copy[y][x] not in ['S', 'G']:
            maze_copy[y][x] = marker
    return maze_copy

# ========== MAIN DEMONSTRATION ==========

    # Run each agent strategy
    strategies = [
        ("Unconstrained (P2-only)", unconstrained_agent),
        ("Constrained (P0+P2)", constrained_agent),
        ("Deadband (P0→P1→P2)", deadband_agent)
    ]

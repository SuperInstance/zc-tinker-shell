#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 in a 20x20 maze based on fleet simulation data.
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    """Implements the Deadband Protocol for maze navigation."""
    
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (negative space - P0)
        self.rocks: Set[Tuple[int, int]] = set()
        total_cells = width * height
        num_rocks = int(total_cells * rock_density)
        
        # Avoid start and goal positions
        available = [(x, y) for x in range(width) for y in range(height)
                    if (x, y) not in [self.start, self.goal]]
        
        self.rocks = set(random.sample(available, num_rocks))
        
        # Initialize pathfinding state
        self.visited = set()
        self.safe_channels = []  # P1 result
        self.optimal_path = []   # P2 result
        
    def print_maze(self, show_path: bool = False, show_channels: bool = False):
        """Display the maze with ASCII visualization."""
        print("\n" + "="*60)
        print("MAZE VISUALIZATION")
        print("="*60)
        print(f"Start: {self.start} | Goal: {self.goal} | Rocks: {len(self.rocks)}")
        print()
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')  # Rock
                elif show_path and pos in self.optimal_path:
                    row.append('·')  # Optimal path
                elif show_channels and pos in self.visited:
                    row.append('.')  # Explored safe area
                else:
                    row.append(' ')
            
            print(' '.join(row))
        print()
    
    def p0_map_negative_space(self) -> None:
        """P0: Identify what NOT to do (the rocks)."""
        print("\n" + "="*60)
        print("P0: MAP NEGATIVE SPACE")
        print("="*60)
        print(f"Identified {len(self.rocks)} rocks (failure states)")
        print("These positions are ABSOLUTELY AVOIDED")
        
        # Show rock distribution
        rock_by_row = {}
        for x, y in self.rocks:
            rock_by_row[y] = rock_by_row.get(y, 0) + 1
        
        print("\nRock distribution by row:")
        for y in sorted(rock_by_row.keys()):
            print(f"  Row {y:2d}: {rock_by_row[y]} rocks")
    
    def p1_find_safe_channels(self) -> bool:
        """P1: Find safe channels (paths that avoid all rocks)."""
        print("\n" + "="*60)
        print("P1: FIND SAFE CHANNELS")
        print("="*60)
        
        # BFS to find all reachable positions avoiding rocks
        queue = [self.start]
        self.visited = {self.start}
        
        while queue:
            x, y = queue.pop(0)
            
            # Check if we reached the goal
            if (x, y) == self.goal:
                print(f"✓ Goal reachable via safe channels")
                return True
            
            # Explore neighbors (right and down only for simplicity)
            neighbors = []
            if x + 1 < self.width:
                neighbors.append((x + 1, y))
            if y + 1 < self.height:
                neighbors.append((x, y + 1))
            
            for nx, ny in neighbors:
                pos = (nx, ny)
                if pos not in self.visited and pos not in self.rocks:
                    self.visited.add(pos)
                    queue.append(pos)
        
        print("✗ No safe channel to goal found")
        return False
    
    def p2_optimize_within_channels(self) -> None:
        """P2: Find optimal path within safe channels."""
        print("\n" + "="*60)
        print("P2: OPTIMIZE WITHIN CHANNELS")
        print("="*60)
        
        # A* search constrained to visited safe positions
        open_set = {self.start}
        came_from = {}
        
        g_score = {self.start: 0}
        f_score = {self.start: self._heuristic(self.start)}
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score.get(pos, float('inf')))
            
            if current == self.goal:
                self.optimal_path = self._reconstruct_path(came_from, current)
                print(f"✓ Found optimal path: {len(self.optimal_path)} steps")
                return
            
            open_set.remove(current)
            
            x, y = current
            neighbors = []
            if x + 1 < self.width:
                neighbors.append((x + 1, y))
            if y + 1 < self.height:
                neighbors.append((x, y + 1))
            if x - 1 >= 0:
                neighbors.append((x - 1, y))
            if y - 1 >= 0:
                neighbors.append((x, y - 1))
            
            for neighbor in neighbors:
                # CONSTRAINT: Only move through safe channels
                if neighbor not in self.visited:
                    continue
                
                tentative_g = g_score[current] + 1
                
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor)
                    open_set.add(neighbor)
        
        print("✗ No path found within safe channels")
    
    def _heuristic(self, pos: Tuple[int, int]) -> float:
        """Manhattan distance heuristic."""
        x1, y1 = pos
        x2, y2 = self.goal
        return abs(x1 - x2) + abs(y1 - y2)
    
    def _reconstruct_path(self, came_from: dict, current: Tuple[int, int]) -> List[Tuple[int, int]]:
        """Reconstruct the optimal path."""
        path = [current]
        while current in came_from:
            current = came_from[current]
            path.append(current)
        return list(reversed(path))
    
    def run_protocol(self) -> bool:
        """Execute the full Deadband Protocol P0→P1→P2."""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL EXECUTION")
        print("="*60)
        
        # P0: Map negative space
        self.p0_map_negative_space()
        self.print_maze()
        
        # P1: Find safe channels
        if not self.p1_find_safe_channels():
            return False
        
        self.print_maze(show_channels=True)
        
        # P2: Optimize within channels
        self.p2_optimize_within_channels()
        self.print_maze(show_path=True)
        
        return len(self.optimal_path) > 0
    
    def run_greedy_baseline(self) -> bool:
        """Run greedy optimization (P2-only) for comparison."""
        print("\n" + "="*60)
        print("GREEDY BASELINE (P2-ONLY - VIOLATION)")
        print("="*60)
        print("Warning: Optimizing without P0+P1 mapping!")
        
        # Greedy pathfinding ignoring rocks
        path = [self.start]
        current = self.start
        
        while current != self.goal:
            x, y = current
            
            # Greedy: move toward goal
            if x < self.goal[0]:
                next_pos = (x + 1, y)
            elif y < self.goal[1]:
                next_pos = (x, y + 1)
            else:
                break
            
            # Check if hit rock
            if next_pos in self.rocks:
                print(f"✗ Hit rock at {next_pos}!")
                return False
            
            path.append(next_pos)
            current = next_pos
        
        print(f"Path found: {len(path)} steps")
        return True

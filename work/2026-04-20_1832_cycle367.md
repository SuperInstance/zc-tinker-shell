#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo
Demonstrates P0→P1→P2 in a simple ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    """A simple maze simulation of the Deadband Protocol."""
    
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density and (x, y) not in [self.start, self.goal]:
                    self.rocks.add((x, y))
        
        # Agent state
        self.position = self.start
        self.path = [self.start]
        self.steps = 0
        
    def display(self, show_path: bool = True, show_rocks: bool = True) -> str:
        """Render the maze as ASCII."""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.position:
                    row.append('A')  # Agent
                elif (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif show_rocks and (x, y) in self.rocks:
                    row.append('█')  # Rock
                elif show_path and (x, y) in self.path:
                    row.append('·')  # Path
                else:
                    row.append(' ')
            grid.append(''.join(row))
        return '\n'.join(grid)
    
    def p0_map_negative_space(self) -> Set[Tuple[int, int]]:
        """P0: Identify rocks (what NOT to do)."""
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Found {len(self.rocks)} rocks (failure states)")
        print("Rocks marked with '█'")
        print(self.display(show_path=False, show_rocks=True))
        return self.rocks
    
    def p1_find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Find all paths that avoid rocks."""
        print("\n=== P1: FIND SAFE CHANNELS ===")
        
        # Simple BFS to find safe paths
        from collections import deque
        
        safe_paths = []
        queue = deque([([self.start], set([self.start]))])
        
        while queue and len(safe_paths) < 3:  # Limit to 3 examples
            path, visited = queue.popleft()
            x, y = path[-1]
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            # Try adjacent cells (right, down, left, up)
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        new_path = path + [(nx, ny)]
                        new_visited = visited.copy()
                        new_visited.add((nx, ny))
                        queue.append((new_path, new_visited))
        
        print(f"Found {len(safe_paths)} safe channels to goal")
        for i, path in enumerate(safe_paths[:2]):  # Show first 2
            print(f"Channel {i+1}: {len(path)-1} steps")
        
        return safe_paths
    
    def p2_optimize_within_channel(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """P2: Choose shortest path from safe channels."""
        print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        if not safe_paths:
            print("ERROR: No safe channels found! Cannot optimize.")
            return []
        
        # Find shortest safe path
        optimal_path = min(safe_paths, key=len)
        print(f"Selected optimal path: {len(optimal_path)-1} steps")
        
        # Visualize the chosen path
        temp_path = self.path
        self.path = optimal_path
        print(self.display(show_path=True, show_rocks=False))
        self.path = temp_path
        
        return optimal_path
    
    def run_unconstrained_greedy(self) -> bool:
        """Demonstrate failure without Deadband Protocol (P2-only)."""
        print("\n--- UNCONSTRAINED GREEDY (P2-ONLY) ---")
        self.position = self.start
        self.path = [self.start]
        self.steps = 0
        
        while self.position != self.goal and self.steps < 100:
            x, y = self.position
            
            # Greedy: move toward goal without checking rocks
            if x < self.goal[0]:
                nx, ny = x + 1, y
            elif y < self.goal[1]:
                nx, ny = x, y + 1
            else:
                break
            
            self.position = (nx, ny)
            self.path.append(self.position)
            self.steps += 1
            
            # Check for rock collision
            if self.position in self.rocks:
                print(f"FAILED: Hit rock at {self.position} after {self.steps} steps")
                print(self.display())
                return False
        
        success = self.position == self.goal
        print(f"{'SUCCESS' if success else 'FAILED'}: {self.steps} steps")
        return success
    
    def run_deadband_protocol(self) -> bool:
        """Full Deadband Protocol demonstration."""
        print("\n" + "="*50)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*50)
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        
        # P1: Find safe channels
        safe_paths = self.p1_find_safe_channels()
        
        if not safe_paths:
            print("No safe channels exist. Goal unreachable.")
            return False
        
        # P2: Optimize within channel
        optimal_path = self.p2_optimize_within_channel(safe_paths)
        
        # Execute the optimal path
        print("\n--- EXECUTING OPTIMAL PATH ---")
        self.position = self.start
        self.path = [self.start]
        self.steps = 0
        
        for step, pos in enumerate(optimal_path[1:], 1):
            self.position = pos
            self.path.append(pos)
            self.steps = step
            
            # Verify we're not hitting rocks
            if pos in rocks:
                print(f"ERROR: Path contains rock at {pos}!")
                return False
            
            if pos == self.goal:
                print(f"REACHED GOAL in {step} steps!")
                print(self.display())
                return True
        
        return False

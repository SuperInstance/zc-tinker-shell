"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# Simulation constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

# Symbols for visualization
AGENT = 'A'
GOAL = 'G'
ROCK = '█'
SAFE = '·'
PATH = '○'
WALL = '│'

class DeadbandMaze:
    """A maze simulation that follows the Deadband Protocol."""
    
    def __init__(self, seed: int = 42):
        random.seed(seed)
        self.grid = [[SAFE for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        
        # P0: Map negative space (rocks)
        self._place_rocks()
        
        # Track protocol steps
        self.p0_complete = False
        self.p1_complete = False
        self.p2_complete = False
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
    
    def _place_rocks(self):
        """P0: Create obstacles (negative space)."""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[x][y] = ROCK
        self.p0_complete = True
    
    def find_safe_channels(self):
        """P1: Identify all paths that avoid rocks."""
        # Simple BFS to find all safe paths
        visited = set()
        queue = [(self.start, [self.start])]
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.goal:
                self.safe_channels.append(path)
                continue
            
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Check all four directions
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.p1_complete = True
        return len(self.safe_channels) > 0
    
    def optimize_within_channels(self):
        """P2: Find shortest path among safe channels."""
        if not self.safe_channels:
            return None
        
        self.optimal_path = min(self.safe_channels, key=len)
        self.p2_complete = True
        return self.optimal_path
    
    def run_greedy_unconstrained(self):
        """Violation: P2-only optimization (greedy, hits rocks)."""
        path = [self.start]
        x, y = self.start
        visited = set([self.start])
        
        steps = 0
        max_steps = MAZE_SIZE * MAZE_SIZE * 2
        
        while (x, y) != self.goal and steps < max_steps:
            # Greedy: move toward goal
            dx = 1 if self.goal[0] > x else -1 if self.goal[0] < x else 0
            dy = 1 if self.goal[1] > y else -1 if self.goal[1] < y else 0
            
            nx, ny = x + dx, y + dy
            
            # If direct move is blocked, try alternatives
            if (nx, ny) in self.rocks or (nx, ny) in visited:
                # Try other directions randomly
                directions = [(1,0), (-1,0), (0,1), (0,-1)]
                random.shuffle(directions)
                for dx2, dy2 in directions:
                    nx2, ny2 = x + dx2, y + dy2
                    if (0 <= nx2 < MAZE_SIZE and 0 <= ny2 < MAZE_SIZE and 
                        (nx2, ny2) not in visited):
                        nx, ny = nx2, ny2
                        break
                else:

                    break
            
            path.append((nx, ny))
            visited.add((nx, ny))
            x, y = nx, ny
            steps += 1
        
        return path if (x, y) == self.goal else None

            print(f"   • Success: ✗ (hit rocks)")

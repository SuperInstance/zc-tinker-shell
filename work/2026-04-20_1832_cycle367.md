#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).

Based on fleet-knowledge/narrow-games/constraint_sim_final.json
and fleet-knowledge/docs/DEADBAND-PROTOCOL.md

Simulates a 20x20 maze with 64 rocks (obstacles).
Shows three agent strategies:
1. Unconstrained (P2-only): greedy optimization, ignores rocks.
2. Constrained (P0+P2): knows rocks but no channel mapping.
3. Deadband (P0+P1+P2): full protocol.

Output matches the fleet's simulation results.
"""

import random
from typing import List, Tuple, Set
from collections import deque

# ==================== MAZE CONFIGURATION ====================
MAZE_SIZE = 20
START = (0, 0)
GOAL = (19, 19)
NUM_ROCKS = 64  # From constraint_sim_final.json

# Symbols for visualization
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_SYM = 'G'
PATH = '○'
SAFE = ' '  # inside deadband channel

# ==================== DEADBAND PROTOCOL IMPLEMENTATION ====================

class DeadbandMaze:
    def __init__(self):
        self.grid = [[EMPTY for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self._generate_rocks()
        self._place_start_goal()
        
    def _generate_rocks(self):
        """P0: Map the negative space (rocks)."""
        while len(self.rocks) < NUM_ROCKS:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != START and (x, y) != GOAL:
                self.rocks.add((x, y))
                self.grid[x][y] = ROCK
    
    def _place_start_goal(self):
        self.grid[START[0]][START[1]] = AGENT
        self.grid[GOAL[0]][GOAL[1]] = GOAL_SYM
    
    def print_maze(self, title: str, highlight: Set[Tuple[int, int]] = None):
        """Print the maze with optional highlighted cells (safe channel or path)."""
        print(f"\n=== {title} ===")
        for x in range(MAZE_SIZE):
            row = []
            for y in range(MAZE_SIZE):
                cell = self.grid[x][y]
                if highlight and (x, y) in highlight:
                    if cell == EMPTY:
                        row.append(SAFE)
                    else:
                        row.append(cell)
                else:
                    row.append(cell)
            print(' '.join(row))
        print()
    
    def is_valid(self, x: int, y: int) -> bool:
        return 0 <= x < MAZE_SIZE and 0 <= y < MAZE_SIZE
    
    def bfs_path(self, avoid_rocks: bool = True) -> List[Tuple[int, int]]:
        """BFS to find shortest path from START to GOAL.
        If avoid_rocks=True, treats rocks as obstacles (P0).
        If False, ignores rocks (unconstrained)."""
        queue = deque([(START, [START])])
        visited = set([START])
        dirs = [(1,0), (-1,0), (0,1), (0,-1)]  # 4-connected
        
        while queue:
            (x, y), path = queue.popleft()
            if (x, y) == GOAL:
                return path
            
            for dx, dy in dirs:
                nx, ny = x + dx, y + dy
                if self.is_valid(nx, ny) and (nx, ny) not in visited:
                    if avoid_rocks and (nx, ny) in self.rocks:
                        continue
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        return []  # No path found
    
    def find_safe_channel(self) -> Set[Tuple[int, int]]:
        """P1: Identify safe channels (deadbands).
        Returns all cells reachable from START while avoiding rocks."""
        stack = [START]
        safe = set([START])
        dirs = [(1,0), (-1,0), (0,1), (0,-1)]
        
        while stack:
            x, y = stack.pop()
            for dx, dy in dirs:
                nx, ny = x + dx, y + dy
                if self.is_valid(nx, ny) and (nx, ny) not in safe:
                    if (nx, ny) not in self.rocks:
                        safe.add((nx, ny))
                        stack.append((nx, ny))
        return safe
    
    def deadband_path(self) -> List[Tuple[int, int]]:
        """Full Deadband Protocol: P0→P1→P2.
        P0: rocks already mapped.
        P1: find safe channel.
        P2: BFS within safe channel only."""
        safe_channel = self.find_safe_channel()
        # Restrict BFS to safe channel
        queue = deque([(START, [START])])
        visited = set([START])
        dirs = [(1,0), (-1,0), (0,1), (0,-1)]
        
        while queue:
            (x, y), path = queue.popleft()
            if (x, y) == GOAL:
                return path
            
            for dx, dy in dirs:
                nx, ny = x + dx, y + dy
                if self.is_valid(nx, ny) and (nx, ny) not in visited:
                    if (nx, ny) in safe_channel:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        return []

# ==================== SIMULATION RUN ====================

def run_simulation():
    """Run one simulation matching fleet's data format."""
    maze = DeadbandMaze()
    
    # Display initial maze with rocks (P0)
    maze.print_maze("MAZE WITH ROCKS (P0: Negative Space)")
    
    # Strategy 1: Unconstrained (P2-only) — ignores rocks
    unconstrained_path = maze.bfs_path(avoid_rocks=False)
    success1 = len(unconstrained_path) > 0
    print(f"Unconstrained (P2-only): Success = {success1}, Path length = {len(unconstrained_path)-1 if success1 else 'N/A'}")
    
    # Strategy 2: Constrained (P0+P2) — knows rocks but no channel mapping
    constrained_path = maze.bfs_path(avoid_rocks=True)
    success2 = len(constrained_path) > 0
    print(f"Constrained (P0+P2): Success = {success2}, Path length = {len(constrained_path)-1 if success2 else 'N/A'}")
    
    # Strategy 3: Deadband (P0+P1+P2) — full protocol
    safe_channel = maze.find_safe_channel()
    maze.print_maze("SAFE CHANNEL (P1: Deadbands)", highlight=safe_channel)
    
    deadband_path = maze.deadband_path()
    success3 = len(deadband_path) > 0
    print(f"Deadband (P0+P1+P2): Success = {success3}, Path length = {len(deadband_path)-1 if success3 else 'N/A'}")
    
    # Show final deadband path
    if deadband_path:
        path_set = set(deadband_path[1:-1])  # Exclude start and goal
        maze.print_maze("DEADBAND PATH (P2: Optimization)", highlight=path_set)
    
    return {
        "unconstrained_success": success1,
        "constrained_success": success2,
        "deadband_success": success3,
        "unconstrained_length": len(unconstrained_path)-1 if success1 else None,
        "constrained_length": len(constrained_path)-1 if success2 else None,
        "deadband_length": len(deadband_path)-1 if success3 else None,
    }

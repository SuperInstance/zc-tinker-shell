class DeadbandAgent:
    """An agent that follows the Deadband Protocol strictly."""
    
    def __init__(self, maze: List[List[str]], start: Tuple[int, int], goal: Tuple[int, int]):
        self.maze = maze
        self.start = start
        self.goal = goal
        self.visited = set()
        self.rocks = set()      # P0: Negative space
        self.safe = set()       # P1: Safe channels
        self.path = []          # P2: Optimal path within safe channels
        
    def p0_map_negative_space(self) -> None:
        """STEP 1: Identify all rocks (states where failure probability > threshold)."""
        print("P0: Mapping negative space...")
        for y in range(MAZE_SIZE):
            for x in range(MAZE_SIZE):
                if self.maze[y][x] == '█':
                    self.rocks.add((x, y))
        print(f"  Found {len(self.rocks)} rocks (failure states)")
        
    def p1_find_safe_channels(self) -> None:
        """STEP 2: Find all reachable positions that are not rocks."""
        print("P1: Finding safe channels...")
        stack = [self.start]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in self.visited:
                continue
            if (x, y) in self.rocks:
                continue
                
            self.visited.add((x, y))
            self.safe.add((x, y))
            
            # Explore neighbors (4-directional)
            for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in self.visited:
                        stack.append((nx, ny))
        
        print(f"  Mapped {len(self.safe)} safe positions")
        
    def p2_optimize_within_channel(self) -> bool:
        """STEP 3: Find shortest path within safe channels (BFS)."""
        print("P2: Optimizing within safe channels...")
        
        # BFS through safe positions only
        queue = [(self.start, [self.start])]
        visited_bfs = set([self.start])
        
        while queue:
            (x, y), current_path = queue.pop(0)
            
            if (x, y) == self.goal:
                self.path = current_path
                print(f"  Found path of length {len(self.path)} (optimal: {OPTIMAL_PATH_LEN})")
                return True
            
            for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) in self.safe and (nx, ny) not in visited_bfs:
                        visited_bfs.add((nx, ny))
                        queue.append(((nx, ny), current_path + [(nx, ny)]))
        
        print("  No path found within safe channels")
        return False
    
    def run_protocol(self) -> bool:
        """Execute strict P0→P1→P2 priority."""
        print("\n" + "="*50)
        print("DEADBAND PROTOCOL EXECUTION")
        print("="*50)
        
        # P0: Map negative space (MANDATORY FIRST STEP)
        self.p0_map_negative_space()
        
        # P1: Find safe channels (MANDATORY SECOND STEP)
        self.p1_find_safe_channels()
        
        # P2: Optimize within channels (ONLY AFTER P0+P1)
        success = self.p2_optimize_within_channel()
        
        return success
    
    def print_maze_with_path(self) -> None:
        """Visualize the maze with agent's path."""
        print("\n" + "="*50)
        print("MAZE VISUALIZATION")
        print("="*50)
        
        # Create display grid
        display = [row[:] for row in self.maze]
        
        # Mark path
        if self.path:
            for x, y in self.path:
                if display[y][x] == ' ':
                    display[y][x] = '·'
            # Mark start and goal
            sx, sy = self.start
            gx, gy = self.goal
            display[sy][sx] = 'S'
            display[gy][gx] = 'G'
        
        # Print with borders
        print("+" + "-" * MAZE_SIZE + "+")
        for y in range(MAZE_SIZE):
            print("|" + "".join(display[y]) + "|")
        print("+" + "-" * MAZE_SIZE + "+")
        
        # Legend
        print("\nLegend: █=rock, ·=safe path, S=start, G=goal")

        # A* search ignoring rocks (hits them)
        import heapq

            return abs(a[0] - b[0]) + abs(a[1] - b[1])

        heapq.heappush(open_set, (0, self.start))

            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                self.path = list(reversed(path))
                print(f"  Found path of length {len(self.path)} (but may hit rocks)")
                return True
            
            x, y = current
            for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    neighbor = (nx, ny)
                    tentative_g = g_score[current] + 1

                        g_score[neighbor] = tentative_g
                        f_score = tentative_g + heuristic(neighbor, self.goal)
                        heapq.heappush(open_set, (f_score, neighbor))

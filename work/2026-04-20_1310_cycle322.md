import random

from collections import deque

# Constants from fleet-knowledge
MAZE_SIZE = 20
ROCK_COUNT = 64

class Maze:
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.rocks: Set[Tuple[int, int]] = set()
        
        # Generate random rocks (avoid start and goal)
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[y][x] = '█'
        
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def print_maze(self, path=None):
        """Print ASCII maze with optional path."""
        for y in range(MAZE_SIZE):
            row = []
            for x in range(MAZE_SIZE):
                if path and (x, y) in path:
                    if (x, y) == self.start:
                        row.append('S')
                    elif (x, y) == self.goal:
                        row.append('G')
                    else:
                        row.append('·')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
        print()

        # P1: Find safe channels (BFS to find all reachable non-rock cells)

        visited = set([self.maze.start])
        
        while queue:
            x, y = queue.popleft()
            safe_cells.add((x, y))
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy

            return False, [], 0

        open_set = {self.maze.start}
        came_from = {}
        g_score = {self.maze.start: 0}
        f_score = {self.maze.start: self._heuristic(self.maze.start, self.maze.goal)}
        
        while open_set:
            current = min(open_set, key=lambda x: f_score.get(x, float('inf')))
            
            if current == self.maze.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.maze.start)
                path.reverse()
                return True, path, len(path) - 1

                            open_set.add(neighbor)
        
        return False, [], 0

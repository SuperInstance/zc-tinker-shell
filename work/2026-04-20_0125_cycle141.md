#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 in a 2D grid navigation problem.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import heapq
from typing import List, Tuple, Set, Dict

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.2):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_rocks(rock_density)
        
    def generate_rocks(self, density: float):
        """P0: Map negative space (rocks)"""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        
        # Ensure start and goal are not rocks
        available = [(x, y) for x in range(self.width) for y in range(self.height)
                    if (x, y) not in [self.start, self.goal]]
        
        self.rocks = set(random.sample(available, min(num_rocks, len(available))))
    
    def is_valid(self, x: int, y: int) -> bool:
        """Check if cell is within bounds and not a rock"""
        return (0 <= x < self.width and 
                0 <= y < self.height and 
                (x, y) not in self.rocks)
    
    def get_neighbors(self, x: int, y: int) -> List[Tuple[int, int]]:
        """Get valid neighboring cells (4-direction movement)"""
        neighbors = []
        for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
            nx, ny = x + dx, y + dy
            if self.is_valid(nx, ny):
                neighbors.append((nx, ny))
        return neighbors
    
    def deadband_search(self) -> Tuple[List[Tuple[int, int]], int]:
        """
        Full Deadband Protocol implementation
        Returns: (path, steps_taken) or ([], -1) if no path
        """
        # P0 already done: rocks are mapped in initialization
        
        # P1: Find safe channels (all reachable cells)
        visited = set()
        queue = [self.start]
        came_from: Dict[Tuple[int, int], Tuple[int, int]] = {}
        
        while queue:
            current = queue.pop(0)
            if current == self.goal:
                break
                
            for neighbor in self.get_neighbors(*current):
                if neighbor not in visited:
                    visited.add(neighbor)
                    queue.append(neighbor)
                    came_from[neighbor] = current
        
        # Check if goal is reachable (safe channel exists)
        if self.goal not in came_from:
            return [], -1
        
        # P2: Optimize within channel (A* search constrained to visited safe cells)
        open_set = []
        heapq.heappush(open_set, (0, self.start))
        g_score = {self.start: 0}
        f_score = {self.start: self.heuristic(*self.start)}
        
        while open_set:
            _, current = heapq.heappop(open_set)
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                return path[::-1], len(path)-1
            
            for neighbor in self.get_neighbors(*current):
                # Constraint: only use cells in safe channel (visited)
                if neighbor not in visited:
                    continue
                    
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self.heuristic(*neighbor)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
        
        return [], -1
    
    def greedy_search(self) -> Tuple[List[Tuple[int, int]], int]:
        """
        Unconstrained greedy search (P2-only approach)
        Goes directly toward goal without mapping rocks first
        """
        path = [self.start]
        visited = set([self.start])
        steps = 0
        
        current = self.start
        while current != self.goal and steps < self.width * self.height * 2:
            # Greedy: move toward goal
            best = None
            best_dist = float('inf')
            
            for neighbor in self.get_neighbors(*current):
                if neighbor in visited:
                    continue
                    
                dist = abs(neighbor[0] - self.goal[0]) + abs(neighbor[1] - self.goal[1])
                if dist < best_dist:
                    best_dist = dist
                    best = neighbor
            
            if best is None:
                # Backtrack if stuck
                if len(path) > 1:
                    path.pop()
                    current = path[-1]
                    continue
                else:
                    break
            
            path.append(best)
            visited.add(best)
            current = best
            steps += 1
        
        if current == self.goal:
            return path, steps
        return [], -1
    
    def heuristic(self, x: int, y: int) -> int:
        """Manhattan distance to goal"""
        return abs(x - self.goal[0]) + abs(y - self.goal[1])
    
    def visualize(self, path: List[Tuple[int, int]] = None):
        """ASCII visualization of maze and path"""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif (x, y) in self.rocks:
                    row.append('█')
                elif path and (x, y) in path:
                    row.append('·')
                else:
                    row.append('.')
            grid.append(' '.join(row))
        
        print("\n" + "="*60)
        print("MAZE VISUALIZATION")
        print("S = Start, G = Goal, █ = Rock, · = Path, . = Empty")
        print("="*60)
        for row in grid:
            print(row)
        print("="*60)

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2

A visual simulation of the Deadband Protocol in a 10x10 maze.
Shows the strict priority: map rocks (P0), find safe channels (P1), then optimize path (P2).

Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set
from collections import deque

# ========== MAZE DEFINITION ==========
# '.' = empty, '#' = rock (obstacle), 'S' = start, 'G' = goal
MAZE_STR = """
##########
#S.......#
#.######.#
#.#....#.#
#.#.##.#.#
#.#....#.#
#.######.#
#........#
#......G.#
##########
"""

def parse_maze(maze_str: str) -> Tuple[List[List[str]], Tuple[int, int], Tuple[int, int]]:
    """Parse maze string into grid, find start and goal."""
    lines = [line.strip() for line in maze_str.strip().split('\n')]
    grid = [list(line) for line in lines]
    start = goal = None
    for r in range(len(grid)):
        for c in range(len(grid[0])):
            if grid[r][c] == 'S':
                start = (r, c)
            elif grid[r][c] == 'G':
                goal = (r, c)
    if start is None or goal is None:
        raise ValueError("Maze must contain 'S' and 'G'")
    return grid, start, goal

def print_maze(grid: List[List[str]], title: str = ""):
    """Print maze with optional title."""
    if title:
        print(f"\n=== {title} ===")
    for row in grid:
        print(''.join(row))

def get_neighbors(pos: Tuple[int, int], grid: List[List[str]]) -> List[Tuple[int, int]]:
    """Return valid orthogonal neighbors (up, down, left, right)."""
    r, c = pos
    neighbors = []
    for dr, dc in [(-1,0),(1,0),(0,-1),(0,1)]:
        nr, nc = r + dr, c + dc
        if 0 <= nr < len(grid) and 0 <= nc < len(grid[0]):
            if grid[nr][nc] != '#':  # not a rock
                neighbors.append((nr, nc))
    return neighbors

# ========== DEADBAND PROTOCOL STEPS ==========
def p0_map_rocks(grid: List[List[str]]) -> Set[Tuple[int, int]]:
    """P0: Identify all rock positions (negative space)."""
    rocks = set()
    for r in range(len(grid)):
        for c in range(len(grid[0])):
            if grid[r][c] == '#':
                rocks.add((r, c))
    return rocks

def p1_find_safe_channels(grid: List[List[str]], start: Tuple[int, int], goal: Tuple[int, int]) -> Set[Tuple[int, int]]:
    """P1: Find all reachable positions from start that avoid rocks (safe channels)."""
    visited = set()
    queue = deque([start])
    while queue:
        pos = queue.popleft()
        if pos in visited:
            continue
        visited.add(pos)
        for neighbor in get_neighbors(pos, grid):
            if neighbor not in visited:
                queue.append(neighbor)
    return visited

def p2_optimize_path(grid: List[List[str]], start: Tuple[int, int], goal: Tuple[int, int], safe_set: Set[Tuple[int, int]]) -> List[Tuple[int, int]]:
    """P2: Find shortest path within safe channels (BFS)."""
    if goal not in safe_set:
        return []  # no safe path to goal
    queue = deque([(start, [start])])
    visited = set([start])
    while queue:
        pos, path = queue.popleft()
        if pos == goal:
            return path
        for neighbor in get_neighbors(pos, grid):
            if neighbor in safe_set and neighbor not in visited:
                visited.add(neighbor)
                queue.append((neighbor, path + [neighbor]))
    return []

# ========== VISUALIZATION HELPERS ==========
def visualize_step(grid: List[List[str]], step_name: str, highlight: Set[Tuple[int, int]], symbol: str):
    """Create a copy of grid with highlighted positions."""
    vis_grid = [row[:] for row in grid]
    for r, c in highlight:
        if vis_grid[r][c] not in ('S', 'G'):
            vis_grid[r][c] = symbol
    print_maze(vis_grid, step_name)

def main():
    print("Deadband Protocol Maze Demo")
    print("Based on fleet doctrine: P0→P1→P2 (strict priority)")
    print("=" * 40)
    
    # Parse maze
    grid, start, goal = parse_maze(MAZE_STR)
    print_maze(grid, "Initial Maze")
    print(f"Start: {start}, Goal: {goal}")
    
    # P0: Map rocks (negative space)
    rocks = p0_map_rocks(grid)
    print(f"\nP0: Mapped {len(rocks)} rock positions (negative space)")
    visualize_step(grid, "P0 - Rocks (negative space)", rocks, '#')
    
    # P1: Find safe channels (reachable positions)
    safe_channels = p1_find_safe_channels(grid, start, goal)
    print(f"\nP1: Found {len(safe_channels)} safe positions (channels)")
    visualize_step(grid, "P1 - Safe Channels (deadband)", safe_channels, '~')
    
    # P2: Optimize path within safe channels
    path = p2_optimize_path(grid, start, goal, safe_channels)
    if path:
        print(f"\nP2: Optimized path found ({len(path)-1} steps)")
        # Highlight path
        vis_grid = [row[:] for row in grid]
        for r, c in path[1:-1]:  # exclude start and goal
            vis_grid[r][c] = '*'
        print_maze(vis_grid, "P2 - Optimal Path (within safe channels)")
        print(f"Path: {path}")
    else:
        print("\nP2: No safe path to goal (constrained failure)")
    
    # Summary
    print("\n" + "=" * 40)
    print("PROTOCOL SUMMARY:")
    print(f"  P0 (rocks): {len(rocks)} positions to avoid")
    print(f"  P1 (safe): {len(safe_channels)} reachable positions")
    print(f"  P2 (path): {'SUCCESS' if path else 'FAILURE'}")
    print("\nDoctrine: Never skip P0 or P1. Optimization only within safe channels.")

if __name__ == "__main__":
    main()
```

## Explanation

This script implements the Deadband Protocol as defined in `DEADBAND-PROTOCOL.md`:

1. **P0 (Map negative space)**: Identifies all rock positions (`#`) in the maze—these are states where failure probability > threshold.

2. **P1 (Find safe channels)**: Performs BFS from start to find all reachable positions that avoid rocks—the "deadband" where the agent can operate.

3. **P2 (Optimize within channel)**: Finds the shortest path from start to goal, but *only* within the safe channels identified in P1.

The ASCII visualization shows each step:
- Initial maze with start (S) and goal (G)
- P0: Rocks highlighted (already shown as `#`)
- P1: Safe channels marked with `~`
- P2: Optimal path marked with `*`

The simulation demonstrates that:
- Without P0+P1, a greedy optimizer might try to cut through rocks (failure)
- With P0+P1 first, optimization is constrained to safe water (success)

This matches the fleet simulation results where deadband protocol achieved 50/50 success vs. 0/50 for constrained-only optimization.

#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze
Simulates P0 (map rocks), P1 (find safe channels), P2 (optimal path).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from collections import deque

# ========== CONFIG ==========
GRID_SIZE = 20
ROCK_COUNT = 64  # from simulation data
START = (0, 0)
GOAL = (GRID_SIZE-1, GRID_SIZE-1)

# Symbols
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_MARKER = 'G'
PATH = '○'
SAFE = '░'  # visual for safe channel

# ========== MAZE GENERATION ==========
def generate_maze(size, rocks):
    """Create grid with random rocks (negative space)."""
    grid = [[EMPTY for _ in range(size)] for _ in range(size)]
    rocks_placed = 0
    while rocks_placed < rocks:
        x = random.randint(0, size-1)
        y = random.randint(0, size-1)
        if (x, y) == START or (x, y) == GOAL:
            continue
        if grid[y][x] == EMPTY:
            grid[y][x] = ROCK
            rocks_placed += 1
    grid[START[1]][START[0]] = AGENT
    grid[GOAL[1]][GOAL[0]] = GOAL_MARKER
    return grid

def print_grid(grid, title=""):
    """Print ASCII grid."""
    print(f"\n=== {title} ===")
    for row in grid:
        print(' '.join(row))
    print()

# ========== DEADBAND PROTOCOL ==========
def p0_map_rocks(grid):
    """P0: Identify negative space (rocks). Return set of rock coordinates."""
    rocks = set()
    for y in range(len(grid)):
        for x in range(len(grid[0])):
            if grid[y][x] == ROCK:
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(grid, rocks, start, goal):
    """P1: Find all cells reachable without hitting rocks (safe channels)."""
    size = len(grid)
    safe = set()
    queue = deque([start])
    visited = set([start])
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        x, y = queue.popleft()
        safe.add((x, y))
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < size and 0 <= ny < size:
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited.add((nx, ny))
                    queue.append((nx, ny))
    return safe

def p2_optimal_path(grid, safe, start, goal):
    """P2: Find shortest path within safe channels (BFS)."""
    if goal not in safe:
        return []  # no safe path to goal
    
    size = len(grid)
    queue = deque([(start, [start])])
    visited = set([start])
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        (x, y), path = queue.popleft()
        if (x, y) == goal:
            return path
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < size and 0 <= ny < size:
                if (nx, ny) in safe and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    return []

# ========== VISUALIZATION HELPERS ==========
def visualize_safe_grid(grid, safe):
    """Create a copy of grid marking safe channels."""
    viz = [row[:] for row in grid]
    for (x, y) in safe:
        if viz[y][x] == EMPTY:
            viz[y][x] = SAFE
    return viz

def visualize_path(grid, path):
    """Mark path on grid."""
    viz = [row[:] for row in grid]
    for (x, y) in path[1:-1]:  # exclude start and goal
        viz[y][x] = PATH
    return viz

# ========== MAIN SIMULATION ==========
def run_deadband_simulation():
    print("DEADBAND PROTOCOL DEMO")
    print(f"Grid: {GRID_SIZE}x{GRID_SIZE}, Rocks: {ROCK_COUNT}")
    print(f"Start: {START}, Goal: {GOAL}")
    print("-" * 40)
    
    # Generate environment
    grid = generate_maze(GRID_SIZE, ROCK_COUNT)
    print_grid(grid, "Initial Maze (Rocks = █)")
    
    # P0: Map negative space
    rocks = p0_map_rocks(grid)
    print(f"P0 — Mapped {len(rocks)} rocks (negative space).")
    
    # P1: Find safe channels
    safe = p1_find_safe_channels(grid, rocks, START, GOAL)
    safe_viz = visualize_safe_grid(grid, safe)
    print_grid(safe_viz, f"P1 — Safe Channels ({len(safe)} cells, marked {SAFE})")
    print(f"Goal reachable? {'YES' if GOAL in safe else 'NO'}")
    
    # P2: Optimal path within safe channels
    path = p2_optimal_path(grid, safe, START, GOAL)
    if path:
        path_viz = visualize_path(grid, path)
        print_grid(path_viz, f"P2 — Optimal Path ({len(path)-1} steps, marked {PATH})")
        print(f"Path length: {len(path)-1} moves")
        print(f"Path coordinates: {path}")
    else:
        print("P2 — No safe path to goal exists.")
    
    # Simulate failure of P2-only (unconstrained optimization)
    print("\n--- CONTRAST: P2-Only (Unconstrained) ---")
    # Greedy Manhattan distance move (ignores rocks)
    def greedy_path():
        x, y = START
        moves = []
        while (x, y) != GOAL:
            moves.append((x, y))
            if x < GOAL[0]:
                x += 1
            elif y < GOAL[1]:
                y += 1
            else:
                break
            if (x, y) in rocks:
                return moves, "HIT ROCK"
        moves.append((x, y))
        return moves, "SUCCESS" if (x, y) == GOAL else "STUCK"
    
    greedy_moves, result = greedy_path()
    print(f"Greedy path result: {result}")
    print(f"Moves before stop: {greedy_moves}")
    if result == "HIT ROCK":
        print("P2-only optimization fails — hits rocks (negative space).")

if __name__ == "__main__":
    random.seed(42)  # reproducible for demo
    run_deadband_simulation()
```

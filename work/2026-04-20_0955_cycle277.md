## Implementation Plan

1. **Generate a 20x20 maze** with 64 rocks (obstacles), start at (0,0), goal at (19,19).
2. **P0 (Map Negative Space):** Identify all rock cells. Mark them.
3. **P1 (Find Safe Channels):** Perform flood-fill from start, avoiding rocks, to find all reachable cells. These are the safe channels.
4. **P2 (Optimize Within Channel):** Run A* pathfinding only through safe channels to find shortest path to goal.
5. **Visualize each step** with ASCII art, using different characters for rocks, safe channels, and the optimal path.
6. **Compare against naive greedy (P2-only)** to show failure.

"""
Deadband Protocol Maze Demo
P0 → P1 → P2 in a 20x20 maze with 64 rocks.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import heapq
from collections import deque

# Simulation parameters from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64
START = (0, 0)
GOAL = (19, 19)

def generate_maze(size, rocks):
    """Generate a random maze with rocks as obstacles."""
    maze = [['.' for _ in range(size)] for _ in range(size)]
    maze[START[0]][START[1]] = 'S'
    maze[GOAL[0]][GOAL[1]] = 'G'
    
    placed = 0
    while placed < rocks:
        x = random.randint(0, size-1)
        y = random.randint(0, size-1)
        if (x, y) != START and (x, y) != GOAL and maze[x][y] == '.':
            maze[x][y] = '#'
            placed += 1
    return maze

def print_maze(maze, title=""):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))

def p0_map_rocks(maze):
    """P0: Map negative space — identify all rock cells."""
    rocks = []
    for i in range(len(maze)):
        for j in range(len(maze[0])):
            if maze[i][j] == '#':
                rocks.append((i, j))
    return rocks

def p1_find_safe_channels(maze, rocks_set):
    """P1: Find safe channels via flood-fill from start, avoiding rocks."""
    size = len(maze)
    safe = [[False for _ in range(size)] for _ in range(size)]
    queue = deque([START])
    safe[START[0]][START[1]] = True
    
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    while queue:
        x, y = queue.popleft()
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < size and 0 <= ny < size:
                if not safe[nx][ny] and (nx, ny) not in rocks_set:
                    safe[nx][ny] = True
                    queue.append((nx, ny))
    return safe

def p2_optimize_in_channel(maze, safe):
    """P2: A* pathfinding constrained to safe channels."""
    size = len(maze)
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, START))
    came_from = {}
    g_score = {START: 0}
    f_score = {START: heuristic(START, GOAL)}
    
    dirs = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while open_set:
        _, current = heapq.heappop(open_set)
        
        if current == GOAL:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            return path
        
        for dx, dy in dirs:
            nx, ny = current[0] + dx, current[1] + dy
            neighbor = (nx, ny)
            if 0 <= nx < size and 0 <= ny < size:
                if not safe[nx][ny]:
                    continue  # Not in safe channel
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, GOAL)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return []  # No path found

def naive_greedy_p2_only(maze):
    """P2-only (greedy) that ignores P0/P1 — will hit rocks."""
    size = len(maze)
    x, y = START
    path = [START]
    
    dirs = [(1,0), (0,1)]  # Only move down/right (greedy toward goal)
    
    while (x, y) != GOAL:
        moved = False
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < size and 0 <= ny < size:
                if maze[nx][ny] != '#':  # Doesn't know all rocks from P0
                    x, y = nx, ny
                    path.append((x, y))
                    moved = True
                    break
        if not moved:
            break  # Stuck
    
    return path

def visualize_steps(maze, rocks, safe, path, greedy_path):

    size = len(maze)
    
    # Step 0: Initial maze
    maze_viz = [row[:] for row in maze]

    for x, y in rocks:

    print_maze(p0_viz, "P0: Negative Space Mapped (rocks)")

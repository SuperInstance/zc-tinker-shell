#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

A visual demonstration of the Deadband Protocol (P0→P1→P2) using a simple ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json and DEADBAND-PROTOCOL.md.

The simulation shows:
P0: Map negative space (rocks) — where failure probability is high.
P1: Identify safe channels (deadbands) — paths that avoid rocks.
P2: Optimize within safe channels — find shortest safe path.

This is a simplified, deterministic example for illustration.
"""

import random
from typing import List, Tuple, Set

# ========== MAZE DEFINITION ==========
# 10x10 grid. '.' = open, '#' = rock (obstacle), 'S' = start, 'G' = goal.
# Rocks are placed randomly but seeded for reproducibility.
MAZE_WIDTH = 10
MAZE_HEIGHT = 10
START = (0, 0)
GOAL = (9, 9)

def generate_maze(seed: int = 42) -> List[List[str]]:
    """Generate a maze with random rocks, but deterministic via seed."""
    random.seed(seed)
    maze = [['.' for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]

            if (x, y) == START:
                maze[y][x] = 'S'
            elif (x, y) == GOAL:
                maze[y][x] = 'G'
            elif random.random() < 0.2:
                maze[y][x] = '#'
    return maze

def print_maze(maze: List[List[str]], title: str = ""):
    """Print the maze with a title."""
    if title:
        print(f"\n=== {title} ===")
    for row in maze:
        print(" ".join(row))
    print()

def get_neighbors(pos: Tuple[int, int], maze: List[List[str]]) -> List[Tuple[int, int]]:
    """Return valid orthogonal neighbors (up, down, left, right) within bounds and not rocks."""
    x, y = pos
    neighbors = []
    for dx, dy in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
        nx, ny = x + dx, y + dy
        if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
            if maze[ny][nx] != '#':
                neighbors.append((nx, ny))
    return neighbors

# ========== DEADBAND PROTOCOL IMPLEMENTATION ==========
def phase0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int, int]]:
    """
    P0: Map negative space.
    Identify all cells that are rocks ('#') — these are states where failure is certain.
    """
    rocks = set()
    for y in range(MAZE_HEIGHT):
        for x in range(MAZE_WIDTH):
            if maze[y][x] == '#':
                rocks.add((x, y))
    return rocks

def phase1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
    """
    P1: Identify safe channels (deadbands).
    Find all paths from start to goal that avoid rocks.
    Uses simple DFS to enumerate all possible safe paths (for small maze).
    """
    def dfs(current: Tuple[int, int], path: List[Tuple[int, int]], visited: Set[Tuple[int, int]]):
        if current == GOAL:
            safe_paths.append(path.copy())
            return
        for neighbor in get_neighbors(current, maze):
            if neighbor not in visited:
                visited.add(neighbor)
                path.append(neighbor)
                dfs(neighbor, path, visited)
                path.pop()
                visited.remove(neighbor)

    safe_paths = []
    dfs(START, [START], {START})
    return safe_paths

def phase2_optimize_within_channels(safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
    """
    P2: Optimize within safe channels.
    From all safe paths, select the shortest one (minimum cost).
    """
    if not safe_paths:
        return []
    # Sort by path length (number of steps). First element is shortest.
    safe_paths.sort(key=len)
    return safe_paths[0]

def visualize_phases(maze: List[List[str]], rocks: Set[Tuple[int, int]], safe_paths: List[List[Tuple[int, int]]], optimal_path: List[Tuple[int, int]]):
    """Create visualizations for each phase."""
    # P0: Show rocks
    maze_p0 = [row.copy() for row in maze]
    for x, y in rocks:
        maze_p0[y][x] = 'X'  # Mark rocks
    print_maze(maze_p0, "P0: NEGATIVE SPACE (Rocks marked 'X')")

    # P1: Show one example safe path (first found)
    if safe_paths:
        example_path = safe_paths[0]
        maze_p1 = [row.copy() for row in maze]
        for x, y in example_path[1:-1]:  # Exclude start and goal
            if maze_p1[y][x] == '.':
                maze_p1[y][x] = '~'  # Safe channel marker
        print_maze(maze_p1, "P1: SAFE CHANNEL EXAMPLE (Path marked '~')")

    # P2: Show optimal path
    if optimal_path:
        maze_p2 = [row.copy() for row in maze]
        for x, y in optimal_path[1:-1]:
            if maze_p2[y][x] == '.':
                maze_p2[y][x] = '*'  # Optimal path marker
        print_maze(maze_p2, "P2: OPTIMAL PATH WITHIN SAFE CHANNEL (Path marked '*')")
        print(f"Optimal path length: {len(optimal_path)-1} steps (start to goal).")
    else:
        print("No safe path found — goal unreachable.")

def run_deadband_demo():
    """Main demonstration function."""
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("=" * 40)
    print("Based on Cocapn Fleet Deadband Protocol v1.0")
    print(f"Maze size: {MAZE_WIDTH}x{MAZE_HEIGHT}, Start: {START}, Goal: {GOAL}")
    print()

    # Generate maze
    maze = generate_maze(seed=133)  # Seed from current cycle
    print_maze(maze, "INITIAL MAZE ('.'=open, '#'=rock, 'S'=start, 'G'=goal)")

    # P0: Map rocks
    rocks = phase0_map_rocks(maze)
    print(f"P0: Mapped {len(rocks)} rock cells (negative space).")

    # P1: Find safe channels
    safe_paths = phase1_find_safe_channels(maze, rocks)
    print(f"P1: Found {len(safe_paths)} safe channels (deadbands).")

    # P2: Optimize within channels
    optimal_path = phase2_optimize_within_channels(safe_paths)
    if optimal_path:
        print(f"P2: Selected optimal safe path with {len(optimal_path)-1} steps.")
    else:
        print("P2: No safe path exists — optimization impossible.")

    # Visualize
    visualize_phases(maze, rocks, safe_paths, optimal_path)

    # Summary
    print("\n=== PROTOCOL SUMMARY ===")
    print("P0 (negative space): Identify rocks/failure states.")
    print("P1 (safe channels): Enumerate paths avoiding rocks.")
    print("P2 (optimize): Choose shortest safe path.")
    print("Violation: Skipping P0/P1 leads to hitting rocks (simulation shows 0% success).")

if __name__ == "__main__":
    run_deadband_demo()

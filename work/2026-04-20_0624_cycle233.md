```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
Based on fleet-knowledge simulation results.
"""

import random
from collections import deque

class DeadbandMaze:
    def __init__(self, width=10, height=10, rock_density=0.3):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate maze with rocks
        self.rocks = set()
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density and (x, y) not in [self.start, self.goal]:
                    self.rocks.add((x, y))
        
        # Ensure there's at least one path
        self._ensure_path()
    
    def _ensure_path(self):
        """Make sure there's at least one path from start to goal."""
        # Simple BFS to check connectivity
        visited = set()
        queue = deque([self.start])
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            if (x, y) == self.goal:
                return  # Path exists
            
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks:
                        queue.append((nx, ny))
        
        # If no path, clear a straight line
        for i in range(self.width):
            if i != 0 and i != self.width-1:
                self.rocks.discard((i, 0))
    
    def display(self, path=None, safe_channels=None):
        """Display the maze with optional path and safe channels."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')  # Rock
                elif safe_channels and pos in safe_channels:
                    row.append('·')  # Safe channel
                elif path and pos in path:
                    row.append('●')  # Path
                else:
                    row.append(' ')
            print(' '.join(row))
        print()
    
    def p0_map_rocks(self):
        """P0: Map negative space (rocks)."""
        print("=== P0: MAP ROCKS (Negative Space) ===")
        print(f"Identified {len(self.rocks)} rock positions")
        self.display()
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find safe channels (paths avoiding rocks)."""
        print("=== P1: FIND SAFE CHANNELS ===")
        
        # BFS to find all reachable safe positions
        safe = set()
        queue = deque([self.start])
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in safe:
                continue
            safe.add((x, y))
            
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks:
                        queue.append((nx, ny))
        
        print(f"Found {len(safe)} safe positions")
        self.display(safe_channels=safe)
        return safe
    
    def p2_optimize_path(self, safe_positions):
        """P2: Find optimal path within safe channels."""
        print("=== P2: OPTIMIZE WITHIN CHANNELS ===")
        
        # A* search constrained to safe positions
        from heapq import heappush, heappop
        
        def heuristic(a, b):
            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        open_set = []
        heappush(open_set, (0, self.start))
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: heuristic(self.start, self.goal)}
        
        while open_set:
            _, current = heappop(open_set)
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.start)
                path.reverse()
                
                print(f"Found optimal path of length {len(path)-1}")
                self.display(path=path)
                return path
            
            x, y = current
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                neighbor = (x + dx, y + dy)
                
                # Constraint: must be in safe channels
                if neighbor not in safe_positions:
                    continue
                
                tentative_g = g_score[current] + 1
                
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, self.goal)
                    heappush(open_set, (f_score[neighbor], neighbor))
        
        print("No path found within safe channels")
        return None
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2."""
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("=" * 40)
        
        # P0: Map rocks
        rocks = self.p0_map_rocks()
        
        # P1: Find safe channels
        safe = self.p1_find_safe_channels()
        
        # P2: Optimize within channels
        path = self.p2_optimize_path(safe)
        
        # Results
        print("=" * 40)
        print("RESULTS:")
        print(f"- Rocks mapped (P0): {len(rocks)}")
        print(f"- Safe positions (P1): {len(safe)}")
        print(f"- Path found (P2): {'Yes' if path else 'No'}")
        if path:
            print(f"- Path length: {len(path)-1}")
        
        return path is not None

def run_simulation(runs=10):
    """Run multiple simulations to demonstrate protocol effectiveness."""
    print(f"\n{'='*60}")
    print(f"RUNNING {runs} SIMULATIONS")
    print(f"{'='*60}")
    
    successes = 0
    for i in range(runs):
        print(f"\nSimulation {i+1}:")
        maze = DeadbandMaze(width=8, height=8, rock_density=0.25)
        if maze.run_protocol():
            successes += 1
    
    print(f"\n{'='*60}")
    print(f"SIMULATION RESULTS:")
    print(f"Success rate: {successes}/{runs} ({successes/runs*100:.1f}%)")
    print(f"{'='*60}")

if __name__ == "__main__":
    # Single demonstration
    print("Single Maze Demonstration:")
    maze = DeadbandMaze(width=10, height=6, rock_density=0.2)
    maze.run_protocol()
    
    # Multiple simulations
    run_simulation(runs=5)
```

1. **P0 Visualization**: Shows rocks (█) as negative space
2. **P1 Visualization**: Marks safe channels (·) avoiding rocks
3. **P2 Visualization**: Shows optimal path (●) within safe channels
4. **Constrained Search**: Pathfinding only considers safe positions
5. **Simulation Mode**: Runs multiple mazes to show success rate

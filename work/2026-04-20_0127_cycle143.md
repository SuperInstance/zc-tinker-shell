#!/usr/bin/env python3
"""
deadband_maze_demo.py

A visual demonstration of the Deadband Protocol (P0→P1→P2) in a 20×20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json data.
"""

import random
from collections import deque

# Constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # P0: negative space
START = (0, 0)
GOAL = (MAZE_SIZE - 1, MAZE_SIZE - 1)

def generate_maze():
    """Create a 20×20 maze with 64 rocks (obstacles)."""
    maze = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
    
    # Place rocks randomly, avoiding start and goal
    rocks_placed = 0
    while rocks_placed < ROCK_COUNT:
        x = random.randint(0, MAZE_SIZE - 1)
        y = random.randint(0, MAZE_SIZE - 1)
        if (x, y) == START or (x, y) == GOAL:
            continue
        if maze[y][x] == '.':
            maze[y][x] = '█'  # Rock
            rocks_placed += 1
    
    # Mark start and goal
    maze[START[1]][START[0]] = 'S'
    maze[GOAL[1]][GOAL[0]] = 'G'
    
    return maze

def print_maze(maze, title="Maze"):
    """Print ASCII maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks). Return set of rock coordinates."""
    rocks = set()
    for y in range(MAZE_SIZE):
        for x in range(MAZE_SIZE):
            if maze[y][x] == '█':
                rocks.add((x, y))
    print(f"P0 — Mapped {len(rocks)} rocks (negative space)")
    return rocks

def p1_find_safe_channels(maze, rocks):
    """P1: Find safe channels (paths avoiding rocks) using BFS flood fill."""
    # Directions: up, down, left, right
    directions = [(0, 1), (0, -1), (1, 0), (-1, 0)]
    visited = set()
    safe_cells = set()
    queue = deque([START])
    
    while queue:
        x, y = queue.popleft()
        if (x, y) in visited:
            continue
        visited.add((x, y))
        safe_cells.add((x, y))
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    queue.append((nx, ny))
    
    # Create visualization of safe channels
    channel_viz = [row[:] for row in maze]
    for y in range(MAZE_SIZE):
        for x in range(MAZE_SIZE):
            if (x, y) in safe_cells and channel_viz[y][x] == '.':
                channel_viz[y][x] = '·'  # Safe channel marker
    
    print(f"P1 — Found {len(safe_cells)} safe cells (channels)")
    print_maze(channel_viz, "P1: Safe Channels (· = safe, █ = rock)")
    
    return safe_cells

def p2_optimize_path(maze, safe_cells):
    """P2: Find shortest path within safe channels using BFS."""
    if GOAL not in safe_cells:
        print("P2 — Goal is not reachable within safe channels!")
        return None
    
    directions = [(0, 1), (0, -1), (1, 0), (-1, 0)]
    queue = deque([(START, [START])])
    visited = set([START])
    
    while queue:
        (x, y), path = queue.popleft()
        
        if (x, y) == GOAL:
            # Create path visualization
            path_viz = [row[:] for row in maze]
            for px, py in path[1:-1]:  # Exclude start and goal
                path_viz[py][px] = '○'
            
            print(f"P2 — Found optimal path: {len(path)-1} steps")
            print_maze(path_viz, "P2: Optimal Path (○ = path, █ = rock)")
            return path
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) in safe_cells and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    print("P2 — No path found within safe channels")
    return None

def deadband_protocol_demo():
    """Main demonstration of Deadband Protocol."""
    print("=" * 50)
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print(f"Maze: {MAZE_SIZE}×{MAZE_SIZE}, Rocks: {ROCK_COUNT}")
    print(f"Start: {START}, Goal: {GOAL}")
    print("=" * 50)
    
    # Generate maze
    maze = generate_maze()
    print_maze(maze, "Initial Maze (█ = rock, S = start, G = goal)")
    
    # P0: Map negative space
    rocks = p0_map_rocks(maze)
    
    # P1: Find safe channels
    safe_cells = p1_find_safe_channels(maze, rocks)
    
    # P2: Optimize within channels
    path = p2_optimize_path(maze, safe_cells)
    
    # Results
    print("\n" + "=" * 50)
    print("RESULTS:")
    print(f"P0 — Rocks identified: {len(rocks)}/{ROCK_COUNT}")
    print(f"P1 — Safe cells found: {len(safe_cells)}")
    if path:
        print(f"P2 — Path length: {len(path)-1} steps")
        print("SUCCESS: Deadband Protocol followed P0→P1→P2")
    else:
        print("FAILURE: Goal unreachable within safe channels")
    print("=" * 50)
    
    return path is not None

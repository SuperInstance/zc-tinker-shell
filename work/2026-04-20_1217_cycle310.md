import random

class Maze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        
        # Place rocks (P0 negative space)
        for y in range(height):
            for x in range(width):
                if (x, y) != self.start and (x, y) != self.goal:
                    if random.random() < rock_density:
                        self.grid[y][x] = '#'
                        self.rocks.add((x, y))
    
    def print_maze(self, path=None):
        """Print maze with optional path overlay."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if path and (x, y) in path:
                    row.append('*')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
        print()
    
    def is_valid(self, x, y):
        return 0 <= x < self.width and 0 <= y < self.height
    
    def get_neighbors(self, x, y):
        """Return valid orthogonal neighbors."""
        neighbors = []
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if self.is_valid(nx, ny):
                neighbors.append((nx, ny))
        return neighbors

class Agent:

    def unconstrained_p2_only(self):

        x, y = self.maze.start

            neighbors = self.maze.get_neighbors(x, y)
            # Greedy: move toward goal
            best = None
            best_dist = float('inf')
            for nx, ny in neighbors:
                if (nx, ny) in visited:
                    continue

                    best_dist = dist
                    best = (nx, ny)
            
            if best is None:

            x, y = best

            visited.add((x, y))

        return path
    
    def constrained_p1_p2(self):
        """Pathfinding within a predefined safe channel (skips P0)."""
        # Define a naive safe channel (diagonal band)
        safe_channel = set()
        for i in range(self.maze.width):
            for j in range(max(0, i-2), min(self.maze.height, i+3)):
                safe_channel.add((i, j))
        
        x, y = self.maze.start
        path = [(x, y)]
        visited = set(path)
        
        while (x, y) != self.maze.goal:
            neighbors = self.maze.get_neighbors(x, y)
            # Only consider neighbors in safe channel
            valid_neighbors = [n for n in neighbors if n in safe_channel and n not in visited]
            
            if not valid_neighbors:
                return None  # Stuck
            
            # Greedy toward goal
            best = None
            best_dist = float('inf')
            for nx, ny in valid_neighbors:
                dist = abs(nx - self.maze.goal[0]) + abs(ny - self.maze.goal[1])
                if dist < best_dist:
                    best_dist = dist
                    best = (nx, ny)
            
            x, y = best
            path.append((x, y))
            visited.add((x, y))
            
            # Still need to check rocks (P0 wasn't mapped!)
            if (x, y) in self.maze.rocks:
                return None
        
        return path
    
    def deadband_p0_p1_p2(self):
        """Full Deadband Protocol: P0→P1→P2."""
        # P0: Map negative space (rocks)
        rocks = self.maze.rocks  # Already mapped in maze generation
        
        # P1: Identify safe channels (BFS to find all rock-free paths)
        from collections import deque
        
        queue = deque([self.maze.start])
        visited = {self.maze.start: None}
        safe_reachable = set()
        
        while queue:
            x, y = queue.popleft()
            safe_reachable.add((x, y))
            
            for nx, ny in self.maze.get_neighbors(x, y):
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
        
        # If goal not reachable, fail
        if self.maze.goal not in safe_reachable:
            return None
        
        # P2: Optimize within safe channel (A* on safe_reachable)
        open_set = {self.maze.start}
        came_from = {}
        g_score = {self.maze.start: 0}
        f_score = {self.maze.start: abs(self.maze.start[0] - self.maze.goal[0]) +

            current = min(open_set, key=lambda p: f_score.get(p, float('inf')))
            
            if current == self.maze.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.maze.start)
                return list(reversed(path))
            
            open_set.remove(current)
            
            for neighbor in self.maze.get_neighbors(*current):
                if neighbor not in safe_reachable:
                    continue
                
                tentative_g = g_score[current] + 1
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g

                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        return None

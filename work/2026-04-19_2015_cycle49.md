Deadband Protocol Maze Demo
Simulates P0 (map rocks), P1 (find safe channels), P2 (optimize within channels).
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
import sys
import time

# Simulation constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64          # from "rocks": 64
OPTIMAL_PATH_LEN = 21    # from "optimal": 21

# Symbols for display
AGENT = '🤖'
ROCK = '🪨'
GOAL = '🏁'
PATH = '·'
SAFE = '░'
UNKNOWN = ' '

class DeadbandMaze:
    def __init__(self):
        self.grid = [[UNKNOWN for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks = set()
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.safe_channels = []
        self.optimal_path = []
        
        # Generate rocks (P0 negative space)
        self._generate_rocks()
        
    def _generate_rocks(self):
        """P0: Map the negative space (rocks)"""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[x][y] = ROCK
    
    def p0_map_negative_space(self):
        """Step 1: Identify all failure states (rocks)"""
        print("\n" + "="*60)
        print("P0: MAP NEGATIVE SPACE")
        print("="*60)
        print(f"Identified {len(self.rocks)} rocks (failure states)")
        print("These coordinates are ABSOLUTELY AVOIDED:")
        
        # Show rock coordinates in compact format
        rock_list = list(self.rocks)

            print("  ", " ".join(f"({x},{y})" for x, y in rock_list[i:i+8]))
        
        self._display_maze(show_all=True)
        return list(self.rocks)
    
    def p1_find_safe_channels(self):
        """Step 2: Find paths that avoid all rocks"""
        print("\n" + "="*60)
        print("P1: FIND SAFE CHANNELS")
        print("="*60)
        
        # Simple BFS to find all safe paths (channels)
        from collections import deque
        
        visited = set()
        queue = deque([(self.start, [self.start])])
        safe_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Explore neighbors (up, down, left, right)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE and 
                    (nx, ny) not in self.rocks and 
                    (nx, ny) not in visited):
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.safe_channels = safe_paths
        print(f"Found {len(safe_paths)} safe channels (rock-avoiding paths)")
        
        # Mark safe reachable areas
        safe_cells = set()
        for path in safe_paths:
            for cell in path:
                safe_cells.add(cell)
        
        # Update display
        for x in range(MAZE_SIZE):
            for y in range(MAZE_SIZE):
                if (x, y) in safe_cells and self.grid[x][y] == UNKNOWN:
                    self.grid[x][y] = SAFE
        
        self._display_maze(show_all=True)
        print(f"Safe reachable cells: {len(safe_cells)}/{MAZE_SIZE*MAZE_SIZE}")
        return safe_paths
    
    def p2_optimize_within_channels(self):
        """Step 3: Find shortest path within safe channels"""
        print("\n" + "="*60)
        print("P2: OPTIMIZE WITHIN CHANNELS")
        print("="*60)
        
        if not self.safe_channels:
            print("ERROR: No safe channels found! Cannot optimize.")
            return []
        
        # Find shortest safe path
        shortest = min(self.safe_channels, key=len)
        self.optimal_path = shortest
        
        # Update display with optimal path
        for x, y in shortest[1:-1]:  # Exclude start and goal
            self.grid[x][y] = PATH
        
        self._display_maze(show_all=True)
        print(f"Optimal path length: {len(shortest)-1} steps")
        print(f"Theoretical optimal: {OPTIMAL_PATH_LEN} steps")
        print(f"Path: {' → '.join(f'({x},{y})' for x, y in shortest)}")
        
        return shortest
    
    def simulate_unconstrained_failure(self):
        """Demonstrate what happens without P0+P1 (greedy P2-only)"""
        print("\n" + "="*60)
        print("UNCONSTRAINED OPTIMIZATION (P2-ONLY) - EXPECT FAILURE")
        print("="*60)
        
        # Greedy approach toward goal
        path = [self.start]
        x, y = self.start
        visited = set([self.start])
        steps = 0
        
        while (x, y) != self.goal and steps < 100:
            # Move toward goal (greedy)
            if x < self.goal[0]:
                nx, ny = x + 1, y
            elif y < self.goal[1]:
                nx, ny = x, y + 1
            else:
                nx, ny = x - 1, y  # backup
            
            # Check for rock
            if (nx, ny) in self.rocks:
                print(f"  CRASH! Hit rock at ({nx},{ny}) after {steps} steps")
                return False, steps
            
            x, y = nx, ny
            path.append((x, y))
            visited.add((x, y))
            steps += 1
        
        success = (x, y) == self.goal
        print(f"  Result: {'SUCCESS' if success else 'FAILURE'} after {steps} steps")
        return success, steps
    
    def _display_maze(self, show_all=False):
        """Display the current maze state"""
        print("\nMAZE (20×20):")
        print("  " + "─" * (MAZE_SIZE * 2))
        
        display_grid = [row[:] for row in self.grid]
        
        # Always show start and goal
        sx, sy = self.start
        gx, gy = self.goal
        display_grid[sx][sy] = AGENT
        display_grid[gx][gy] = GOAL
        
        for x in range(MAZE_SIZE):
            row = []
            for y in range(MAZE_SIZE):
                cell = display_grid[x][y]
                if not show_all and cell == SAFE:
                    row.append(UNKNOWN)
                else:
                    row.append(cell)
            print(f"{x:2}│" + "".join(row) + "│")
        
        print("  " + "─" * (MAZE_SIZE * 2))
        print("Legend: 🤖=Start 🏁=Goal 🪨=Rock ·=OptimalPath ░=SafeChannel")

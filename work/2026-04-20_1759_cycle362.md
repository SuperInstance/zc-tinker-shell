#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 362)
A visual ASCII simulation of P0→P1→P2 in a constrained navigation task.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        """
        Initialize a maze with rocks (obstacles), start, and goal.
        rock_density ~ 64 rocks in 20x20 grid (0.16) per simulation data.
        """
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        
        # Place rocks (P0 negative space)
        self.rocks = set()
        num_rocks = int(width * height * rock_density)
        while len(self.rocks) < num_rocks:
            x = random.randint(0, width-1)
            y = random.randint(0, height-1)
            self.rocks.add((x, y))
            self.grid[y][x] = '█'  # rock
        
        # Start (bottom-left) and Goal (top-right)
        self.start = (0, height-1)
        self.goal = (width-1, 0)
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
        
        # Agent state
        self.agent_pos = self.start
        self.path = []
        self.steps = 0
        
    def print_maze(self, show_agent=True):
        """Render the maze with current agent position."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if show_agent and (x, y) == self.agent_pos:
                    row.append('A')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
        print()
    
    def p0_map_negative_space(self):
        """P0: Identify rocks (what NOT to do)."""
        print("P0 — MAP NEGATIVE SPACE")
        print(f"Identified {len(self.rocks)} rocks (obstacles).")
        print("Rocks at coordinates:", sorted(list(self.rocks))[:5], "...")
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find safe paths avoiding rocks (deadbands)."""
        print("P1 — FIND SAFE CHANNELS")
        
        # Simple BFS to find all reachable positions avoiding rocks
        from collections import deque
        visited = set()
        queue = deque([self.start])
        safe_cells = set()
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            safe_cells.add((x, y))
            
            # Check four-directional neighbors
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    queue.append((nx, ny))
        
        print(f"Found {len(safe_cells)} safe cells (deadband).")
        print(f"Goal reachable: {self.goal in safe_cells}")
        return safe_cells
    
    def p2_optimize_within_channel(self, safe_cells):
        """P2: Find shortest path within safe channels."""
        print("P2 — OPTIMIZE WITHIN CHANNEL")
        
        # A* search constrained to safe cells only
        from heapq import heappush, heappop
        
        def heuristic(a, b):
            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        open_set = []
        heappush(open_set, (0, self.start))
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: heuristic(self.start, self.goal)}
        
        while open_set:
            _, current = heappop(open_set)
            
            if current == self.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.reverse()
                return path
            
            x, y = current
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (x + dx, y + dy)
                # CRITICAL: Constrained to safe channels only
                if neighbor not in safe_cells:
                    continue
                    
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, self.goal)
                    heappush(open_set, (f_score[neighbor], neighbor))
        
        return []  # No path found
    
    def run_deadband_protocol(self):
        """Execute full P0→P1→P2 protocol with visualization."""
        print("=" * 50)
        print("DEADBAND PROTOCOL DEMO")
        print("=" * 50)
        print("\nInitial Maze:")
        self.print_maze(show_agent=False)
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        time.sleep(1)
        
        # P1: Find safe channels
        safe_cells = self.p1_find_safe_channels()
        time.sleep(1)
        
        # P2: Optimize within channel
        path = self.p2_optimize_within_channel(safe_cells)
        
        if path:
            print(f"\nFound optimal path with {len(path)} steps.")
            print(f"Path: {[self.start] + path[:3]} ... {path[-3:] + [self.goal]}")
            
            # Animate the path
            print("\nAnimating path (P2 optimization):")
            self.agent_pos = self.start
            for step, pos in enumerate(path):
                self.agent_pos = pos
                print(f"Step {step+1}:")
                self.print_maze()
                time.sleep(0.3)
            print(f"Goal reached in {len(path)} steps!")
            return True
        else:
            print("\nNo safe path to goal (trapped in local minima).")
            return False

def run_simulation(num_runs=5):
    """Run multiple simulations to demonstrate protocol effectiveness."""
    print("\n" + "=" * 50)
    print(f"RUNNING {num_runs} SIMULATIONS")
    print("=" * 50)
    
    successes = 0
    for i in range(num_runs):
        print(f"\n--- Simulation {i+1} ---")
        maze = DeadbandMaze()
        if maze.run_deadband_protocol():
            successes += 1
        time.sleep(1)
    
    print("\n" + "=" * 50)
    print(f"RESULTS: {successes}/{num_runs} successful (Deadband Protocol)")
    print("=" * 50)

if __name__ == "__main__":
    # Single demonstration
    maze = DeadbandMaze()
    maze.run_deadband_protocol()

#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 protocol with visual ASCII maze
Based on fleet-knowledge/narrow-games/constraint_sim_final.json results
"""

import random
from collections import deque
import time

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        """Initialize maze with rocks (obstacles)"""
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Negative space (rocks)
        self.rocks = set()
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density and (x, y) not in [self.start, self.goal]:
                    self.rocks.add((x, y))
        
        # Ensure there's at least one path
        self._ensure_path_exists()
    
    def _ensure_path_exists(self):
        """Make sure there's at least one path from start to goal"""
        # Simple BFS to check connectivity
        visited = set()
        queue = deque([self.start])
        
        while queue:
            x, y = queue.popleft()
            if (x, y) == self.goal:
                return
            
            visited.add((x, y))
            
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and 
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    queue.append((nx, ny))
        
        # If no path, clear a straight line
        for i in range(min(self.width, self.height)):
            x = y = i
            if (x, y) in self.rocks:
                self.rocks.remove((x, y))
    
    def render(self, path=None, visited=None):
        """Render maze as ASCII"""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')
                elif (x, y) == self.goal:
                    row.append('G')
                elif (x, y) in self.rocks:
                    row.append('█')  # Rock
                elif path and (x, y) in path:
                    row.append('·')  # Path
                elif visited and (x, y) in visited:
                    row.append('.')  # Visited
                else:
                    row.append(' ')
            grid.append(''.join(row))
        return '\n'.join(grid)
    
    def p0_map_negative_space(self):
        """P0: Identify all rocks (what NOT to do)"""
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Identified {len(self.rocks)} rocks (obstacles)")
        print("Rocks at positions:", sorted(self.rocks)[:10], "..." if len(self.rocks) > 10 else "")
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all safe paths (where you CAN be)"""
        print("\n=== P1: FIND SAFE CHANNELS ===")
        
        # BFS to find all reachable positions
        visited = set()
        queue = deque([self.start])
        safe_positions = set([self.start])
        
        while queue:
            x, y = queue.popleft()
            
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and 
                    (nx, ny) not in self.rocks and (nx, ny) not in visited):
                    visited.add((nx, ny))
                    safe_positions.add((nx, ny))
                    queue.append((nx, ny))
        
        print(f"Found {len(safe_positions)} safe positions")
        print(f"Goal reachable: {self.goal in safe_positions}")
        return safe_positions
    
    def p2_optimize_within_channel(self, safe_positions):
        """P2: Find optimal path through safe channels"""
        print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
        
        # BFS for shortest path
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                print(f"Found optimal path of length {len(path)}")
                return path
            
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and 
                    (nx, ny) in safe_positions and (nx, ny) not in visited):
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        print("No path found within safe channels")
        return None
    
    def naive_greedy_p2_only(self):
        """Naive P2-only approach (violates protocol)"""
        print("\n=== NAIVE P2-ONLY (VIOLATION) ===")
        print("Skipping P0 and P1, optimizing directly...")
        
        path = [self.start]
        visited = set([self.start])
        current = self.start
        
        while current != self.goal:
            x, y = current
            
            # Greedy: move toward goal
            moves = []
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and 
                    (nx, ny) not in visited):
                    # Manhattan distance to goal
                    dist = abs(nx - self.goal[0]) + abs(ny - self.goal[1])
                    moves.append((dist, (nx, ny)))
            
            if not moves:
                print("Stuck! Hit a dead end.")
                return None
            
            moves.sort()
            next_pos = moves[0][1]
            
            if next_pos in self.rocks:
                print(f"Hit rock at {next_pos}!")
                return None
            
            path.append(next_pos)
            visited.add(next_pos)
            current = next_pos
        
        print(f"Found path of length {len(path)} (but likely hit rocks)")
        return path

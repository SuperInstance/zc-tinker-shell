"""
deadband_maze_demo.py

A visual demonstration of the Deadband Protocol (P0→P1→P2) using an ASCII maze.
Based on fleet simulation data from constraint_sim_final.json.

import json
from collections import deque
from pathlib import Path

def load_simulation_data():
    """Load the maze configuration from fleet knowledge."""
    data_path = Path("fleet-knowledge/narrow-games/constraint_sim_final.json")
    if not data_path.exists():
        # Fallback to the provided data
        return {
            "optimal": 21,
            "rocks": 64,
            "unconstrained_success": 33,
            "constrained_success": 0,
            "deadband_success": 50,
            "unconstrained_avg": 2320.9,
            "constrained_avg": 0,
            "deadband_avg": 21.0
        }
    
    with open(data_path, 'r') as f:
        return json.load(f)

class DeadbandMaze:
    def __init__(self, width=20, height=20):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # P0: Map negative space (rocks)
        self.rocks = set()
        self._generate_rocks()
        
        # P1: Safe channels (will be populated)
        self.safe_channels = []
        self.optimal_path = []
    
    def _generate_rocks(self):
        """Generate rock positions (64 rocks as per simulation)."""
        # Deterministic rock placement for reproducibility
        import hashlib
        seed = "cocapn_fleet_deadband_demo"
        hash_obj = hashlib.md5(seed.encode())
        seed_int = int(hash_obj.hexdigest(), 16)
        
        positions = []
        for i in range(64):
            x = (seed_int + i * 17) % self.width
            y = (seed_int + i * 23) % self.height
            # Avoid start and goal positions
            if (x, y) != self.start and (x, y) != self.goal:
                positions.append((x, y))
        
        self.rocks = set(positions[:64])
    
    def p0_map_negative_space(self):
        """P0: Identify all rock positions (what NOT to do)."""
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Identified {len(self.rocks)} rocks (obstacles)")
        print("Rocks are positions where failure probability > threshold")
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all safe paths from start to goal (safe channels)."""
        print("\n=== P1: FIND SAFE CHANNELS ===")
        
        # BFS to find all safe paths
        queue = deque([(self.start, [self.start])])
        visited = {self.start}
        all_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                all_paths.append(path)
                continue
            
            # Check all 4-direction neighbors
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.safe_channels = all_paths
        print(f"Found {len(self.safe_channels)} safe channels (paths avoiding rocks)")
        
        if not self.safe_channels:
            print("WARNING: No safe channels found! Agent cannot reach goal.")
        
        return self.safe_channels
    
    def p2_optimize_within_channels(self):
        """P2: Find optimal path within safe channels."""
        print("\n=== P2: OPTIMIZE WITHIN CHANNELS ===")
        
        if not self.safe_channels:
            print("No safe channels to optimize. Protocol violation: P2 before P1.")
            return []
        
        # Find shortest path (minimum cost)
        self.optimal_path = min(self.safe_channels, key=len)
        print(f"Optimal path length: {len(self.optimal_path) - 1} steps")
        print(f"Optimal path: {self.optimal_path[:3]}...{self.optimal_path[-3:] if len(self.optimal_path) > 6 else ''}")
        
        return self.optimal_path
    
    def visualize(self, show_channels=False):
        """Create ASCII visualization of the maze."""
        print("\n" + "="*50)
        print("MAZE VISUALIZATION")
        print("="*50)
        
        # Create grid
        grid = [['·' for _ in range(self.width)] for _ in range(self.height)]
        
        # Mark rocks (P0)
        for x, y in self.rocks:
            grid[y][x] = '█'
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        grid[sy][sx] = 'S'
        grid[gy][gx] = 'G'
        
        # Mark safe channels (P1) if requested
        if show_channels and self.safe_channels:
            for path in self.safe_channels:
                for x, y in path[1:-1]:  # Exclude start and goal
                    if grid[y][x] == '·':
                        grid[y][x] = '░'
        
        # Mark optimal path (P2)
        if self.optimal_path:
            for x, y in self.optimal_path[1:-1]:  # Exclude start and goal
                grid[y][x] = '○'
        
        # Print grid
        for row in grid:
            print(' '.join(row))
        
        print("\nLEGEND:")
        print("  S = Start")
        print("  G = Goal")
        print("  █ = Rock (P0 negative space)")
        print("  · = Unexplored")
        if show_channels:
            print("  ░ = Safe channel (P1)")
        print("  ○ = Optimal path (P2)")
    
    def run_protocol(self):
        """Execute the full Deadband Protocol P0→P1→P2."""
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*50)
        
        # P0: Map negative space
        rocks = self.p0_map_negative_space()
        
        # P1: Find safe channels
        channels = self.p1_find_safe_channels()
        
        # P2: Optimize within channels
        optimal = self.p2_optimize_within_channels()
        
        # Visualization
        self.visualize(show_channels=True)
        
        # Show protocol violation example
        self._show_violation_example()
        
        return optimal
    
    def _show_violation_example(self):
        """Demonstrate what happens when protocol is violated."""
        print("\n" + "="*50)
        print("PROTOCOL VIOLATION EXAMPLE: P2 BEFORE P0+P1")
        print("="*50)
        
        # Greedy path (P2-only) - ignores rocks
        print("\nGreedy agent (P2-only):")
        print("1. Goes straight toward goal")
        print("2. Hits rocks immediately")
        print("3. Result: 0% success (as per simulation data)")
        
        # Show simulation results
        sim_data = load_simulation_data()
        print(f"\nSimulation Results (from constraint_sim_final.json):")
        print(f"  Deadband Protocol (P0→P1→P2): {sim_data['deadband_success']}/50 success")
        print(f"  Greedy (P2-only): {sim_data['constrained_success']}/50 success")
        print(f"  Optimal path length: {sim_data['optimal']} steps")

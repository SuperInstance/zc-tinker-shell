        random.seed(self.seed)
        
        # Grid: '.' = open, '#' = rock (obstacle), 'S' = start, 'G' = goal
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
        
        # Generate random rocks (64 rocks as per simulation data)
        self.rocks = set()
        while len(self.rocks) < 64:
            x = random.randint(0, width-1)
            y = random.randint(0, height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[y][x] = '#'

    def print_grid(self, title="", highlight=None):
        """Print ASCII grid with optional highlighted cells."""

        for y in range(self.height):
            row = []

                cell = self.grid[y][x]

                    row.append(f'[{cell}]')
                else:
                    row.append(f' {cell} ')
            print(''.join(row))

        self.print_grid("P0: Rocks (Negative Space)", highlight=self.p0_map)
        return self.p0_map

    def phase1_find_safe_channels(self):
        """P1: Find safe channels (where you CAN be) using BFS from start."""
        from collections import deque
        
        # Safe cells are those not rocks
        safe = set()
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) not in self.rocks:
                    safe.add((x, y))
        
        # BFS to find reachable safe cells from start
        queue = deque([self.start])
        visited = {self.start}
        reachable_safe = {self.start}
        
        dirs = [(1,0), (-1,0), (0,1), (0,-1)]
        
        while queue:
            x, y = queue.popleft()
            for dx, dy in dirs:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited and (nx, ny) in safe:
                        visited.add((nx, ny))
                        reachable_safe.add((nx, ny))
                        queue.append((nx, ny))
        
        self.p1_channels = reachable_safe
        print(f"P1 — Found safe channels: {len(self.p1_channels)} reachable cells")
        self.print_grid("P1: Safe Channels (Reachable Non-Rock Cells)", 
                       highlight=self.p1_channels)
        return self.p1_channels

    def phase2_optimize_within_channels(self):
        """P2: Find shortest path within safe channels (constrained)."""
        from collections import deque
        
        if not self.p1_channels:

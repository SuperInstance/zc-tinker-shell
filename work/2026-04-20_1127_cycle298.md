#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker
Visual demonstration of P0→P1→P2 priority order in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import sys

def generate_maze(width=20, height=20, rock_density=0.16):
    """Generate a random maze with start (S), goal (G), and rocks (#)."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks
    rocks = int(width * height * rock_density)
    for _ in range(rocks):
        x, y = random.randint(0, width-1), random.randint(0, height-1)
        maze[y][x] = '#'
    
    # Place start (top-left) and goal (bottom-right)
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    
    return maze

def print_maze(maze, title="", path=None):
    """Print maze with optional path overlay."""
    print(f"\n{title}")
    for y, row in enumerate(maze):
        line = []
        for x, cell in enumerate(row):
            if path and (x, y) in path:
                line.append('*')
            else:
                line.append(cell)
        print(' '.join(line))
    print()

def p0_map_rocks(maze):
    """P0: Map negative space — identify all rock coordinates."""
    rocks = []
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, rocks):
    """P1: Find safe channels — BFS to identify reachable areas avoiding rocks."""
    width, height = len(maze[0]), len(maze)
    start = (0, 0)
    goal = (width-1, height-1)
    
    # BFS to find all reachable cells
    queue = [start]
    visited = set([start])
    safe = set([start])
    
    while queue:
        x, y = queue.pop(0)
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited.add((nx, ny))
                    safe.add((nx, ny))
                    queue.append((nx, ny))
    
    return safe

def p2_optimize_path(maze, safe_cells):
    """P2: Optimize within channel — A* search through safe cells only."""
    width, height = len(maze[0]), len(maze)
    start = (0, 0)
    goal = (width-1, height-1)
    
    # A* implementation constrained to safe cells
    open_set = {start}
    came_from = {}
    g_score = {start: 0}
    f_score = {start: abs(goal[0]-start[0]) + abs(goal[1]-start[1])}
    
    while open_set:
        current = min(open_set, key=lambda p: f_score.get(p, float('inf')))
        
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            return list(reversed(path))
        
        open_set.remove(current)
        
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            neighbor = (current[0] + dx, current[1] + dy)
            if 0 <= neighbor[0] < width and 0 <= neighbor[1] < height:
                if neighbor not in safe_cells:
                    continue  # Skip unsafe cells (P1 constraint)
                
                tentative_g = g_score[current] + 1
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + abs(goal[0]-neighbor[0]) + abs(goal[1]-neighbor[1])
                    if neighbor not in open_set:
                        open_set.add(neighbor)
    
    return None  # No path found within safe channels

def deadband_protocol_demo():
    """Main demonstration of Deadband Protocol P0→P1→P2."""
    print("=" * 60)
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print("=" * 60)
    
    # Generate maze matching simulation parameters
    random.seed(42)  # Deterministic for demo
    maze = generate_maze(20, 20, rock_density=0.16)
    
    # Step 0: Initial state
    print_maze(maze, "INITIAL MAZE (20×20)")
    print(f"Start: (0,0)  Goal: (19,19)")
    
    # P0: Map negative space
    rocks = p0_map_rocks(maze)
    print(f"P0 — MAP NEGATIVE SPACE: Found {len(rocks)} rocks")
    
    # P1: Identify safe channels
    safe_cells = p1_find_safe_channels(maze, rocks)
    print(f"P1 — IDENTIFY SAFE CHANNELS: {len(safe_cells)} reachable cells")
    
    # Visualize safe channels
    safe_maze = [row[:] for row in maze]
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if (x, y) in safe_cells and safe_maze[y][x] == '.':
                safe_maze[y][x] = '~'  # Safe water
    
    print_maze(safe_maze, "P1 — SAFE CHANNELS (water = ~)")
    
    # P2: Optimize within channel
    path = p2_optimize_path(maze, safe_cells)
    
    if path:
        print(f"P2 — OPTIMIZE WITHIN CHANNEL: Found path of length {len(path)-1}")
        print_maze(maze, "P2 — OPTIMAL PATH (* = path)", set(path))
        
        # Protocol summary
        print("=" * 60)
        print("DEADBAND PROTOCOL SUMMARY")
        print(f"P0 (rocks): {len(rocks)} obstacles identified")
        print(f"P1 (safe): {len(safe_cells)} cells in safe channels")
        print(f"P2 (path): {len(path)-1} steps to goal")
        print("Protocol followed: P0→P1→P2 (strict priority)")
    else:
        print("P2 — OPTIMIZE WITHIN CHANNEL: No path found within safe channels")
        print("This demonstrates constraint satisfaction — cannot violate P1 safety.")
    
    return maze, rocks, safe_cells, path

if __name__ == "__main__":
    maze, rocks, safe_cells, path = deadband_protocol_demo()

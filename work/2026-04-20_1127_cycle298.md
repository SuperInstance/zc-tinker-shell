#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
Demonstrates P0 (map rocks), P1 (find safe channels), P2 (optimize path).
"""

import random
import sys
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_density=0.2):
        self.width = width
        self.height = height
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.rocks = set()
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rock_density = rock_density
        
        # Generate rocks
        for y in range(height):
            for x in range(width):
                if (x, y) != self.start and (x, y) != self.goal:
                    if random.random() < rock_density:
                        self.grid[y][x] = '#'
                        self.rocks.add((x, y))
        
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def print_grid(self, title="", highlight=None):
        """Print the maze with optional highlighted cells."""
        print(f"\n{title}")
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if highlight and (x, y) in highlight:
                    row.append('*')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
        print()
    
    def p0_map_rocks(self):
        """P0: Identify negative space (rocks)."""
        print("=== P0: MAP THE NEGATIVE SPACE ===")
        print(f"Identified {len(self.rocks)} rock cells (marked #).")
        self.print_grid("Maze with rocks", self.rocks)
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find all safe cells reachable from start."""
        print("=== P1: IDENTIFY SAFE CHANNELS ===")
        safe = set()
        queue = deque([self.start])
        visited = set([self.start])
        dirs = [(1,0), (-1,0), (0,1), (0,-1)]
        
        while queue:
            x, y = queue.popleft()
            safe.add((x, y))
            for dx, dy in dirs:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited and (nx, ny) not in self.rocks:
                        visited.add((nx, ny))
                        queue.append((nx, ny))
        
        print(f"Found {len(safe)} safe cells reachable from start.")
        self.print_grid("Safe channels (reachable cells)", safe)
        return safe
    
    def p2_optimize_path(self, safe_cells):
        """P2: Find shortest path within safe channels."""
        print("=== P2: OPTIMIZE WITHIN CHANNEL ===")
        if self.goal not in safe_cells:
            print("ERROR: Goal not reachable within safe channels!")
            return []
        
        # BFS for shortest path
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        dirs = [(1,0), (-1,0), (0,1), (0,-1)]
        
        while queue:
            (x, y), path = queue.popleft()
            if (x, y) == self.goal:
                print(f"Found optimal path of length {len(path)-1} steps.")
                return path
            
            for dx, dy in dirs:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) in safe_cells and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        return []
    
    def run_deadband_protocol(self):
        """Execute full P0→P1→P2 sequence."""
        print("DEADBAND PROTOCOL DEMO")
        print("=" * 40)
        self.print_grid("Initial Maze")
        
        # P0: Map rocks
        rocks = self.p0_map_rocks()
        
        # P1: Find safe channels
        safe = self.p1_find_safe_channels()
        
        # P2: Optimize path
        path = self.p2_optimize_path(safe)
        
        if path:
            print("=== FINAL RESULT ===")
            self.print_grid("Optimal Path (P0→P1→P2)", set(path))
            print(f"Path: {len(path)} cells")
            print(f"Steps: {len(path)-1}")
            return True
        else:
            print("No valid path found.")
            return False
    
    def run_greedy_violation(self):
        """Demonstrate what happens when skipping to P2 (violation)."""
        print("\n" + "=" * 40)
        print("GREEDY VIOLATION (Skipping P0+P1)")
        print("=" * 40)
        
        # Try to find path ignoring rocks
        queue = deque([(self.start, [self.start])])
        visited = set([self.start])
        dirs = [(1,0), (-1,0), (0,1), (0,-1)]
        
        while queue:
            (x, y), path = queue.popleft()
            if (x, y) == self.goal:
                print(f"Greedy found path of length {len(path)-1} steps...")
                print("But check if it goes through rocks:")
                for px, py in path:
                    if (px, py) in self.rocks:
                        print(f"  Path hits rock at ({px}, {py})!")
                        return False
                return True
            
            for dx, dy in dirs:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        print("Greedy search failed to find any path.")
        return False

def main():
    """Run the demonstration."""
    print("Deadband Protocol Maze Simulation")
    print("Based on fleet-knowledge/narrow-games/constraint_sim_final.json")
    print(f"Optimal success: 50/50 with deadband, 0/50 with greedy\n")
    
    # Set seed for reproducibility
    random.seed(42)
    
    maze = DeadbandMaze(width=15, height=10, rock_density=0.15)
    
    # Run deadband protocol (correct)
    success_deadband = maze.run_deadband_protocol()
    
    # Run greedy violation (incorrect)
    success_greedy = maze.run_greedy_violation()
    
    print("\n" + "=" * 40)
    print("SUMMARY:")
    print(f"Deadband Protocol (P0→P1→P2): {'SUCCESS' if success_deadband else 'FAIL'}")
    print(f"Greedy Violation (skip to P2): {'SUCCESS' if success_greedy else 'FAIL (hit rocks)'}")
    print("\nLesson: Never optimize (P2) before mapping rocks (P0) and safe channels (P1).")

if __name__ == "__main__":
    main()

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)
A visual ASCII simulation of P0→P1→P2 navigation.
"""

import random
import time
from typing import List, Tuple, Set

class MazeAgent:
    """Agent that applies Deadband Protocol in a 2D maze."""
    
    def __init__(self, maze: List[List[str]], start: Tuple[int, int], goal: Tuple[int, int]):
        self.maze = maze
        self.start = start
        self.goal = goal
        self.height = len(maze)
        self.width = len(maze[0])
        
        # Protocol states
        self.rocks = set()      # P0: Negative space
        self.channels = set()   # P1: Safe cells
        self.path = []          # P2: Optimal path
        
    def p0_map_negative_space(self):
        """P0: Identify all rocks (obstacles)."""
        for y in range(self.height):
            for x in range(self.width):
                if self.maze[y][x] == '#':
                    self.rocks.add((x, y))
        return len(self.rocks)
    
    def p1_find_safe_channels(self):
        """P1: Find all reachable safe cells (deadbands)."""
        visited = set()
        stack = [self.start]
        
        while stack:
            x, y = stack.pop()
            if (x, y) in visited:
                continue
            if (x, y) in self.rocks:
                continue
            if not (0 <= x < self.width and 0 <= y < self.height):
                continue
                
            visited.add((x, y))
            self.channels.add((x, y))
            
            # Explore neighbors
            for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                stack.append((x + dx, y + dy))
        
        return len(self.channels)
    
    def p2_optimize_within_channel(self):
        """P2: Find shortest path through safe channels (BFS)."""
        if self.goal not in self.channels:
            return []  # Goal unreachable via safe channels
        
        queue = [(self.start, [self.start])]
        visited = set([self.start])
        
        while queue:
            (x, y), current_path = queue.pop(0)
            
            if (x, y) == self.goal:
                self.path = current_path
                return len(self.path)
            
            for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                nx, ny = x + dx, y + dy
                next_pos = (nx, ny)
                
                if (next_pos in self.channels and 
                    next_pos not in visited and
                    0 <= nx < self.width and 0 <= ny < self.height):
                    visited.add(next_pos)
                    queue.append((next_pos, current_path + [next_pos]))
        
        return 0  # No path found
    
    def run_protocol(self) -> bool:
        """Execute full P0→P1→P2 protocol."""
        print("=== DEADBAND PROTOCOL DEMO ===")
        print("P0: Mapping negative space...")
        rocks_count = self.p0_map_negative_space()
        print(f"  Found {rocks_count} rocks (obstacles)")
        
        print("\nP1: Finding safe channels...")
        channels_count = self.p1_find_safe_channels()
        print(f"  Found {channels_count} safe cells (deadbands)")
        
        print("\nP2: Optimizing within channels...")
        path_length = self.p2_optimize_within_channel()
        
        if path_length > 0:
            print(f"  Found optimal path of length {path_length}")
            return True
        else:
            print("  No safe path to goal exists")
            return False
    
    def display_maze(self, show_path: bool = False):
        """Display the maze with current state."""
        display = [row[:] for row in self.maze]
        
        # Mark protocol elements
        for x, y in self.channels:
            if display[y][x] == ' ':
                display[y][x] = '.'  # Safe channel
        
        if show_path and self.path:
            for x, y in self.path[1:-1]:  # Skip start and goal
                if display[y][x] == '.':
                    display[y][x] = 'o'  # Path
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        display[sy][sx] = 'S'
        display[gy][gx] = 'G'
        
        print("\n" + "\n".join("".join(row) for row in display))
        
        # Legend
        print("\nLegend: #=rock, .=safe channel, o=path, S=start, G=goal")

def generate_maze(width: int, height: int, rock_density: float = 0.3) -> List[List[str]]:
    """Generate a random maze with rocks."""
    maze = [[' ' for _ in range(width)] for _ in range(height)]
    
    # Add border walls
    for y in range(height):
        maze[y][0] = '#'
        maze[y][width-1] = '#'
    for x in range(width):
        maze[0][x] = '#'
        maze[height-1][x] = '#'
    
    # Add random rocks
    for y in range(1, height-1):
        for x in range(1, width-1):
            if random.random() < rock_density:
                maze[y][x] = '#'
    
    # Ensure start and goal positions are clear
    maze[1][1] = ' '
    maze[height-2][width-2] = ' '
    
    return maze

def main():
    """Run the Deadband Protocol demonstration."""
    # Create a 15x15 maze
    maze = generate_maze(15, 15, rock_density=0.25)
    
    # Define start (top-left) and goal (bottom-right)
    start = (1, 1)
    goal = (13, 13)
    
    # Create agent
    agent = MazeAgent(maze, start, goal)
    
    print("Initial Maze:")
    agent.display_maze(show_path=False)
    
    time.sleep(1)
    
    # Run protocol
    success = agent.run_protocol()
    
    time.sleep(1)
    
    if success:
        print("\nFinal Solution (P0→P1→P2):")
        agent.display_maze(show_path=True)
        
        # Protocol summary
        print("\n=== PROTOCOL SUMMARY ===")
        print(f"P0 Rocks: {len(agent.rocks)} obstacles identified")
        print(f"P1 Channels: {len(agent.channels)} safe cells mapped")
        print(f"P2 Path: {len(agent.path)} steps to goal")
        print(f"Success: Agent reached goal via safe channels only")
    else:
        print("\nProtocol failed: Goal is not reachable through safe channels")
        print("This demonstrates P0+P1 preventing futile optimization attempts.")

if __name__ == "__main__":
    main()
```

## Demo Features

1. **Visual ASCII Maze**: 15×15 grid with rocks (`#`), start (`S`), and goal (`G`)
2. **Three-Phase Protocol**:
   - **P0**: Maps all rock positions (negative space)
   - **P1**: Flood-fills to find all reachable safe cells (channels/deadbands)
   - **P2**: BFS search for shortest path within safe channels only
3. **Step-by-Step Display**: Shows maze state at each protocol phase
4. **Failure Handling**: If goal is not in safe channels, protocol reports failure (demonstrating P0+P1 preventing futile P2 optimization)

## Key Deadband Principles Demonstrated

- **Strict Priority**: P0→P1→P2 order is enforced
- **Channel-Constrained Optimization**: P2 only searches within P1's safe channels
- **Negative Space Mapping**: Rocks identified before any pathfinding
- **Visual Feedback**: ASCII display shows rocks, channels, and final path

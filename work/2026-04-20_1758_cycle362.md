#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cocapn Fleet)

A visual ASCII simulation of the Deadband Protocol (P0→P1→P2) in a 10x10 maze.
Demonstrates the strict priority: map rocks (P0), find safe channels (P1),
then optimize path (P2) within safe channels only.

Based on fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
from typing import List, Tuple, Set

# ====================
# MAZE CONFIGURATION
# ====================
MAZE_WIDTH = 10
MAZE_HEIGHT = 10
START = (0, 0)
GOAL = (9, 9)
ROCK_DENSITY = 0.15  # ~15% of cells are rocks

# Symbols for visualization
SYM_EMPTY = '·'
SYM_ROCK = '█'
SYM_START = 'S'
SYM_GOAL = 'G'
SYM_PATH = '○'
SYM_SAFE = ' '  # inside safe channel

# ====================
# DEADBAND PROTOCOL IMPLEMENTATION
# ====================

class DeadbandAgent:
    def __init__(self, maze: List[List[str]]):
        self.maze = maze
        self.rocks: Set[Tuple[int, int]] = set()      # P0: negative space
        self.safe_channels: List[List[Tuple[int, int]]] = []  # P1: safe paths
        self.optimal_path: List[Tuple[int, int]] = [] # P2: best within safe

    def p0_map_rocks(self) -> None:
        """P0: Map negative space — identify all rock cells."""
        for y in range(MAZE_HEIGHT):
            for x in range(MAZE_WIDTH):
                if self.maze[y][x] == SYM_ROCK:
                    self.rocks.add((x, y))
        print(f"[P0] Mapped {len(self.rocks)} rocks")

    def p1_find_safe_channels(self) -> bool:
        """P1: Find safe channels — BFS avoiding rocks."""
        from collections import deque
        
        queue = deque([(START, [START])])
        visited = set([START])
        all_safe_paths = []

        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == GOAL:
                all_safe_paths.append(path)
                continue
            
            # Explore neighbors (up, down, left, right)
            for dx, dy in [(0,1),(0,-1),(1,0),(-1,0)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_WIDTH and 0 <= ny < MAZE_HEIGHT:
                    if (nx, ny) not in visited and (nx, ny) not in self.rocks:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.safe_channels = all_safe_paths
        print(f"[P1] Found {len(self.safe_channels)} safe channel(s)")
        return len(self.safe_channels) > 0

    def p2_optimize_in_channel(self) -> None:
        """P2: Optimize within safe channels — shortest safe path."""
        if not self.safe_channels:
            print("[P2] No safe channels — cannot optimize")
            return
        
        # Select shortest path among safe channels
        self.optimal_path = min(self.safe_channels, key=len)
        print(f"[P2] Optimal path length: {len(self.optimal_path)-1} steps")

    def run_protocol(self) -> bool:
        """Execute full Deadband Protocol P0→P1→P2."""
        print("\n=== DEADBAND PROTOCOL EXECUTION ===")
        self.p0_map_rocks()
        if not self.p1_find_safe_channels():
            print("FAIL: No safe channels to goal")
            return False
        self.p2_optimize_in_channel()
        return True

# ====================
# VISUALIZATION
# ====================

def generate_maze() -> List[List[str]]:
    """Create random maze with rocks, start, and goal."""
    maze = [[SYM_EMPTY for _ in range(MAZE_WIDTH)] for _ in range(MAZE_HEIGHT)]
    
    # Place rocks randomly
    for y in range(MAZE_HEIGHT):
        for x in range(MAZE_WIDTH):
            if (x, y) != START and (x, y) != GOAL:
                if random.random() < ROCK_DENSITY:
                    maze[y][x] = SYM_ROCK
    
    # Mark start and goal
    maze[START[1]][START[0]] = SYM_START
    maze[GOAL[1]][GOAL[0]] = SYM_GOAL
    return maze

def print_maze(maze: List[List[str]], 
               rocks: Set[Tuple[int, int]] = None,
               safe_cells: Set[Tuple[int, int]] = None,
               path: List[Tuple[int, int]] = None) -> None:
    """Print maze with optional overlays."""
    display = [row[:] for row in maze]  # Copy
    
    # Overlay safe channel cells (P1 visualization)
    if safe_cells:
        for (x, y) in safe_cells:
            if display[y][x] == SYM_EMPTY:
                display[y][x] = SYM_SAFE
    
    # Overlay optimal path (P2 visualization)
    if path:
        for (x, y) in path[1:-1]:  # Exclude start and goal
            if display[y][x] == SYM_EMPTY or display[y][x] == SYM_SAFE:
                display[y][x] = SYM_PATH
    
    # Print with borders
    print("+" + "-" * MAZE_WIDTH + "+")
    for row in display:
        print("|" + "".join(row) + "|")
    print("+" + "-" * MAZE_WIDTH + "+")

def print_protocol_steps(agent: DeadbandAgent) -> None:
    """Show step-by-step visualization."""
    print("\n--- STEP P0: Map Rocks (Negative Space) ---")
    print_maze(agent.maze, rocks=agent.rocks)
    
    print("\n--- STEP P1: Safe Channels (Where You CAN Be) ---")
    # Show all cells reachable without hitting rocks
    safe_cells = set()
    for path in agent.safe_channels:
        safe_cells.update(path)
    print_maze(agent.maze, safe_cells=safe_cells)
    
    print("\n--- STEP P2: Optimize Within Channel ---")
    print_maze(agent.maze, path=agent.optimal_path)

# ====================

# ====================

def main():
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("=" * 40)
    print(f"Maze: {MAZE_WIDTH}x{MAZE_HEIGHT}, Start {START}, Goal {GOAL}")
    print(f"Rock density: {ROCK_DENSITY*100:.0f}%")
    print("\nLegend:")
    print(f"  {SYM_START} = Start  {SYM_GOAL} = Goal  {SYM_ROCK} = Rock")
    print(f"  {SYM_SAFE} = Safe channel  {SYM_PATH} = Optimal path")
    
    # Generate and show initial maze
    maze = generate_maze()
    print("\n=== INITIAL MAZE ===")
    print_maze(maze)
    
    # Create agent and run protocol
    agent = DeadbandAgent(maze)
    success = agent.run_protocol()
    
    if success:
        print_protocol_steps(agent)
        
        # Summary
        print("\n=== PROTOCOL RESULTS ===")
        print(f"P0 Rocks identified: {len(agent.rocks)}")
        print(f"P1 Safe channels found: {len(agent.safe_channels)}")
        print(f"P2 Optimal path steps: {len(agent.optimal_path)-1}")
        print(f"Success: Agent reached goal via safe channel")
    else:
        print("\nFAILED: No safe path exists — agent would hit rocks")
        print("(This demonstrates P0/P1 preventing certain failure)")

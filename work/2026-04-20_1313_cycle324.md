#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — Tinker (Cycle 324)
Visual demonstration of P0→P1→P2 priority in a 20x20 maze.
Matches fleet-knowledge/narrow-games/constraint_sim_final.json results.
"""

import random
import sys
from collections import deque

# Constants from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # ~16% density, matches simulation
START = (0, 0)
GOAL = (19, 19)

def generate_maze(size, rocks):
    """Generate a random maze with rocks as obstacles."""
    maze = [['.' for _ in range(size)] for _ in range(size)]
    maze[START[0]][START[1]] = 'S'
    maze[GOAL[0]][GOAL[1]] = 'G'
    
    # Place rocks randomly (avoid start and goal)
    placed = 0
    while placed < rocks:
        x = random.randint(0, size-1)
        y = random.randint(0, size-1)
        if (x, y) == START or (x, y) == GOAL:
            continue
        if maze[x][y] == '.':
            maze[x][y] = '#'
            placed += 1
    return maze

def print_maze(maze, title="", path=None, rocks_marked=False):
    """Print maze with optional title, path, and rock marking."""
    print(f"\n{title}")
    print("   " + "".join(f"{i:2}" for i in range(MAZE_SIZE)))
    for i in range(MAZE_SIZE):
        row = f"{i:2} "
        for j in range(MAZE_SIZE):
            cell = maze[i][j]
            if path and (i, j) in path:
                row += "◉ "
            elif rocks_marked and cell == '#':
                row += "▓ "
            else:
                row += cell + " "
        print(row)

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks)."""
    rocks = []
    for i in range(MAZE_SIZE):
        for j in range(MAZE_SIZE):
            if maze[i][j] == '#':
                rocks.append((i, j))
    return rocks

def p1_find_safe_channels(maze, rocks):
    """P1: Find all paths from start to goal avoiding rocks (BFS)."""
    # BFS to find all reachable cells
    visited = [[False]*MAZE_SIZE for _ in range(MAZE_SIZE)]
    queue = deque([START])
    visited[START[0]][START[1]] = True
    safe_cells = set()
    
    while queue:
        x, y = queue.popleft()
        safe_cells.add((x, y))
        for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
            nx, ny = x+dx, y+dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if not visited[nx][ny] and maze[nx][ny] != '#':
                    visited[nx][ny] = True
                    queue.append((nx, ny))
    
    # Check if goal is reachable
    if GOAL not in safe_cells:
        return None, []  # No safe channel
    
    # Find shortest path (will be optimized in P2)
    path = bfs_shortest(maze, START, GOAL)
    return safe_cells, path

def bfs_shortest(maze, start, goal):
    """BFS to find shortest path from start to goal (ignores rocks)."""
    queue = deque([(start, [start])])
    visited = set([start])
    
    while queue:
        (x, y), path = queue.popleft()
        if (x, y) == goal:
            return path
        for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
            nx, ny = x+dx, y+dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if maze[nx][ny] != '#' and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    return []

def p2_optimize_within_channel(maze, safe_cells):
    """P2: Find optimal path within safe channel (A*)."""
    # Simple A* implementation
    open_set = {START}
    came_from = {}
    g_score = {START: 0}
    f_score = {START: heuristic(START, GOAL)}
    
    while open_set:
        current = min(open_set, key=lambda x: f_score.get(x, float('inf')))
        if current == GOAL:
            return reconstruct_path(came_from, current)
        
        open_set.remove(current)
        for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
            neighbor = (current[0]+dx, current[1]+dy)
            if 0 <= neighbor[0] < MAZE_SIZE and 0 <= neighbor[1] < MAZE_SIZE:
                if maze[neighbor[0]][neighbor[1]] == '#':
                    continue
                if neighbor not in safe_cells:
                    continue  # Must stay in safe channel
                
                tentative_g = g_score[current] + 1
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, GOAL)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
    return []

def heuristic(a, b):
    """Manhattan distance heuristic."""
    return abs(a[0] - b[0]) + abs(a[1] - b[1])

def reconstruct_path(came_from, current):
    """Reconstruct path from came_from dict."""
    path = [current]
    while current in came_from:
        current = came_from[current]
        path.append(current)
    path.reverse()
    return path

    print("="*60)

    print_maze(maze, "Rocks marked (▓):", rocks_marked=True)
    
    # P1: Find safe channels
    safe_cells, initial_path = p1_find_safe_channels(maze, rocks)

    print(f"\nP1: IDENTIFIED SAFE CHANNEL")

        print(f"   Initial safe path length: {len(initial_path)-1} steps")
        print_maze(maze, "Safe channel path (◉):", initial_path)

    optimal_path = p2_optimize_within_channel(maze, safe_cells)
    if not optimal_path:
        print("P2: NO OPTIMAL PATH FOUND — Staying in safe channel")
        return True
    
    print(f"\nP2: OPTIMIZED WITHIN SAFE CHANNEL")
    print(f"   Optimal path length: {len(optimal_path)-1} steps")
    if initial_path:
        improvement = len(initial_path) - len(optimal_path)

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0→P1→P2 priority order in a 20x20 maze.
"""

import random
from collections import deque

# Constants from fleet-knowledge
MAZE_SIZE = 20
ROCK_COUNT = 64
OPTIMAL_PATH_LENGTH = 21

class Maze:
    def __init__(self, size=MAZE_SIZE, rocks=ROCK_COUNT):
        self.size = size
        self.grid = [['.' for _ in range(size)] for _ in range(size)]
        self.start = (0, 0)
        self.goal = (size-1, size-1)
        self.grid[self.start[0]][self.start[1]] = 'S'
        self.grid[self.goal[0]][self.goal[1]] = 'G'
        
        # Place rocks (P0 negative space)
        self.rocks = set()
        while len(self.rocks) < rocks:
            r, c = random.randint(0, size-1), random.randint(0, size-1)
            if (r, c) not in [self.start, self.goal] and (r, c) not in self.rocks:
                self.rocks.add((r, c))
                self.grid[r][c] = '#'
    
    def print_maze(self, path=None):
        """Print maze with optional path overlay."""
        for r in range(self.size):
            row = []
            for c in range(self.size):
                pos = (r, c)
                if path and pos in path:
                    row.append('*')
                else:
                    row.append(self.grid[r][c])
            print(' '.join(row))
        print()
    
    def is_valid(self, r, c):
        return 0 <= r < self.size and 0 <= c < self.size
    
    def get_neighbors(self, r, c):
        """Return valid orthogonal neighbors."""
        dirs = [(-1,0), (1,0), (0,-1), (0,1)]
        neighbors = []
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if self.is_valid(nr, nc):
                neighbors.append((nr, nc))
        return neighbors

class Agent:
    def __init__(self, maze):
        self.maze = maze
        self.steps = 0
    
    def unconstrained_p2_only(self):
        """Greedy P2-only strategy (often hits rocks)."""
        path = [self.maze.start]
        visited = set([self.maze.start])
        current = self.maze.start
        
        while current != self.maze.goal:
            r, c = current
            neighbors = self.maze.get_neighbors(r, c)
            # Greedy: pick neighbor closest to goal
            neighbors.sort(key=lambda pos: abs(pos[0]-self.maze.goal[0]) + abs(pos[1]-self.maze.goal[1]))
            found = False
            for nr, nc in neighbors:
                if (nr, nc) not in visited:
                    if (nr, nc) in self.maze.rocks:
                        return False, path  # Hit rock
                    path.append((nr, nc))
                    visited.add((nr, nc))
                    current = (nr, nc)
                    found = True
                    break
            if not found:
                return False, path  # Stuck
            self.steps += 1
            if self.steps > 1000:
                return False, path  # Timeout
        return True, path
    
    def constrained_p0_only(self):
        """P0-only: avoids rocks but doesn't move toward goal."""
        # Just stand still (safe but useless)
        return True, [self.maze.start]
    
    def deadband_protocol(self):
        """Full P0→P1→P2: map rocks, find safe channels, optimize."""
        # P0: Rocks already mapped in maze.rocks
        
        # P1: Find safe channels (BFS through non-rock spaces)
        start = self.maze.start
        goal = self.maze.goal
        queue = deque([start])
        came_from = {start: None}
        
        while queue:
            current = queue.popleft()
            if current == goal:
                break
            r, c = current
            for nr, nc in self.maze.get_neighbors(r, c):
                if (nr, nc) in self.maze.rocks:
                    continue  # Avoid rocks
                if (nr, nc) not in came_from:
                    queue.append((nr, nc))
                    came_from[(nr, nc)] = current
        
        if goal not in came_from:
            return False, []  # No safe channel
        
        # P2: Optimize within safe channel (reconstruct shortest path)
        path = []
        current = goal
        while current is not None:
            path.append(current)
            current = came_from[current]
        path.reverse()
        return True, path

def run_simulation(runs=50):
    """Run multiple simulations to match fleet data."""
    print("=== Deadband Protocol Simulation ===")
    print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}, Optimal: {OPTIMAL_PATH_LENGTH}")
    print()
    
    stats = {
        'unconstrained': {'success': 0, 'total_steps': 0},
        'constrained': {'success': 0, 'total_steps': 0},
        'deadband': {'success': 0, 'total_steps': 0}
    }
    
    for run in range(runs):
        maze = Maze()
        agent = Agent(maze)
        
        # Unconstrained (P2-only)
        success, path = agent.unconstrained_p2_only()
        if success:
            stats['unconstrained']['success'] += 1
            stats['unconstrained']['total_steps'] += len(path)
        
        # Constrained (P0-only)
        success, path = agent.constrained_p0_only()
        if success:
            stats['constrained']['success'] += 1
            stats['constrained']['total_steps'] += len(path)
        
        # Deadband Protocol
        success, path = agent.deadband_protocol()
        if success:
            stats['deadband']['success'] += 1
            stats['deadband']['total_steps'] += len(path)
        
        # Show first run maze
        if run == 0:
            print("First run maze (S=start, G=goal, #=rock):")
            maze.print_maze()
            print("Deadband Protocol path (P0→P1→P2):")
            maze.print_maze(path)
    
    print("\n=== Results (50 runs) ===")
    print(f"Unconstrained (P2-only): {stats['unconstrained']['success']}/50 success, "
          f"avg steps: {stats['unconstrained']['total_steps']/max(stats['unconstrained']['success'],1):.1f}")
    print(f"Constrained (P0-only): {stats['constrained']['success']}/50 success, "
          f"avg steps: {stats['constrained']['total_steps']/max(stats['constrained']['success'],1):.1f}")
    print(f"Deadband Protocol: {stats['deadband']['success']}/50 success, "
          f"avg steps: {stats['deadband']['total_steps']/max(stats['deadband']['success'],1):.1f}")
    
    # Compare with fleet data
    print("\n=== Comparison with Fleet Data ===")
    print("From constraint_sim_final.json:")
    print("  unconstrained_success: 33/50")
    print("  constrained_success: 0/50")
    print("  deadband_success: 50/50")
    print("  deadband_avg: 21.0")

if __name__ == "__main__":
    run_simulation()
```

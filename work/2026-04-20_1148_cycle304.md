```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo — ASCII Maze Simulation
Demonstrates P0→P1→P2 priority in a 20x20 maze.
"""

import random
import sys

def generate_maze(width=20, height=20, rock_density=0.16):
    """Generate a random maze with start, goal, and rocks."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks
    rocks = int(width * height * rock_density)
    for _ in range(rocks):
        x, y = random.randint(0, width-1), random.randint(0, height-1)
        maze[y][x] = '#'
    
    # Place start (top-left) and goal (bottom-right)
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    
    return maze

def print_maze(maze, title="Maze"):
    """Print maze with title."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(' '.join(row))
    print()

def find_path_greedy(maze):
    """P2-only: Greedy path ignoring rocks (will hit them)."""
    width, height = len(maze[0]), len(maze)
    path = []
    x, y = 0, 0
    
    while not (x == width-1 and y == height-1):
        path.append((x, y))
        # Greedy: move right if possible, else down
        if x < width-1 and maze[y][x+1] != '#':
            x += 1
        elif y < height-1 and maze[y+1][x] != '#':
            y += 1
        else:
            # Stuck (hit a rock)
            break
    
    path.append((x, y))
    return path

def find_path_deadband(maze):
    """P0→P1→P2: Map rocks, find safe channels, then optimize."""
    width, height = len(maze[0]), len(maze)
    
    # P0: Map negative space (rocks)
    rocks = [(x, y) for y in range(height) for x in range(width) if maze[y][x] == '#']
    
    # P1: Identify safe channels (BFS avoiding rocks)
    from collections import deque
    queue = deque([(0, 0, [])])
    visited = set()
    
    while queue:
        x, y, path = queue.popleft()
        if (x, y) in visited:
            continue
        visited.add((x, y))
        
        if x == width-1 and y == height-1:
            return path + [(x, y)]
        
        # Move right or down (simplified)
        for dx, dy in [(1, 0), (0, 1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height and maze[ny][nx] != '#':
                queue.append((nx, ny, path + [(x, y)]))
    
    return []  # No path

def simulate_runs(num_runs=50):
    """Run multiple simulations to match fleet data."""
    successes_greedy = 0
    successes_deadband = 0
    total_steps_greedy = 0
    total_steps_deadband = 0
    
    for run in range(num_runs):
        maze = generate_maze()
        
        # Greedy (P2-only)
        path_g = find_path_greedy(maze)
        if path_g[-1] == (19, 19):
            successes_greedy += 1
            total_steps_greedy += len(path_g)
        
        # Deadband (P0→P1→P2)
        path_d = find_path_deadband(maze)
        if path_d and path_d[-1] == (19, 19):
            successes_deadband += 1
            total_steps_deadband += len(path_d)
    
    print(f"\n=== Simulation Results ({num_runs} runs) ===")
    print(f"Greedy (P2-only) success: {successes_greedy}/{num_runs}")
    print(f"Deadband (P0→P1→P2) success: {successes_deadband}/{num_runs}")
    if successes_greedy:
        print(f"Greedy avg steps: {total_steps_greedy/successes_greedy:.1f}")
    if successes_deadband:
        print(f"Deadband avg steps: {total_steps_deadband/successes_deadband:.1f}")

def main():
    """Run a single visual demo."""
    random.seed(42)  # Deterministic for demo
    maze = generate_maze()
    print_maze(maze, "Initial Maze (S=start, G=goal, #=rock)")
    
    # Greedy path (P2-only)
    greedy_path = find_path_greedy(maze)
    maze_g = [row[:] for row in maze]
    for x, y in greedy_path[1:-1]:
        maze_g[y][x] = '*'
    print_maze(maze_g, "Greedy Path (P2-only) - Hits rocks")
    
    # Deadband path (P0→P1→P2)
    deadband_path = find_path_deadband(maze)
    maze_d = [row[:] for row in maze]
    for x, y in deadband_path[1:-1]:
        maze_d[y][x] = 'O'
    print_maze(maze_d, "Deadband Path (P0→P1→P2) - Safe channel")
    
    # Summary
    print("=== Protocol Summary ===")
    print("P0 (Map rocks): Identify all '#' positions")
    print("P1 (Safe channels): BFS avoiding rocks")
    print("P2 (Optimize): Shortest path within safe channels")
    print(f"Greedy reached goal: {greedy_path[-1] == (19, 19)}")
    print(f"Deadband reached goal: {deadband_path and deadband_path[-1] == (19, 19)}")
    
    # Run batch simulation
    simulate_runs(10)  # Smaller sample for quick demo

if __name__ == "__main__":
    main()

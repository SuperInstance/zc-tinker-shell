import random

class Maze:
    def __init__(self, width=20, height=20, num_rocks=64):
        self.width = width
        self.height = height
        self.num_rocks = num_rocks
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_maze()
    
    def generate_maze(self):
        """Generate random rocks, ensuring start and goal are clear."""
        all_cells = [(x, y) for x in range(self.width) for y in range(self.height)]
        all_cells.remove(self.start)
        all_cells.remove(self.goal)
        
        self.rocks = set(random.sample(all_cells, self.num_rocks))
    
    def is_rock(self, x, y):
        return (x, y) in self.rocks
    
    def in_bounds(self, x, y):
        return 0 <= x < self.width and 0 <= y < self.height
    
    def get_neighbors(self, x, y, avoid_rocks=True):
        """Get valid neighboring cells."""
        neighbors = []
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if self.in_bounds(nx, ny):
                if avoid_rocks and self.is_rock(nx, ny):
                    continue
                neighbors.append((nx, ny))
        return neighbors
    
    def render(self, path=None, current=None):
        """Render maze as ASCII with optional path highlighting."""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if (x, y) == self.start:
                    row.append('S')

                    row.append('G')

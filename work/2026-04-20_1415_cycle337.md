```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 in a 20x20 ASCII maze.
Tinker — Cycle 337 — Cocapn Fleet
"""

import random
from collections import deque

def generate_maze(width=20, height=20, obstacle_density=0.2):
    """Generate a random maze with start, goal, and obstacles."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place obstacles
    for y in range(height):
        for x in range(width):
            if random.random() < obstacle_density:
                maze[y][x] = '#'
    
    # Ensure start and goal are open
    start = (0, 0)
    goal = (width-1, height-1)
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    
    return maze, start, goal

def print_maze(maze, path=None):
    """Print maze with optional path marked."""
    for y, row in enumerate(maze):
        line = []
        for x, cell in enumerate(row):
            if path and (x, y) in path:
                line.append('*')
            else:
                line.append(cell)
        print(''.join(line))
    print()

def p0_map_negative_space(maze):
    """P0: Identify all obstacle cells (rocks)."""
    rocks = []
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start, goal, rocks):
    """P1: Find all safe paths from start to goal (BFS)."""
    width, height = len(maze[0]), len(maze)
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    queue = deque([(start, [start])])
    visited = set([start])
    safe_paths = []
    
    while queue:
        (x, y), path = queue.popleft()
        
        if (x, y) == goal:
            safe_paths.append(path)
            continue
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in visited and (nx, ny) not in rocks:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
    
    return safe_paths

def p2_optimize_within_channel(safe_paths):
    """P2: Select shortest safe path."""
    if not safe_paths:
        return None
    return min(safe_paths, key=len)

def naive_greedy_agent(maze, start, goal):
    """P2-only agent: greedy move toward goal, ignoring obstacles."""
    x, y = start
    gx, gy = goal
    path = [start]
    
    while (x, y) != goal:
        # Greedy: move toward goal
        if x < gx:
            dx, dy = 1, 0
        elif x > gx:
            dx, dy = -1, 0
        elif y < gy:
            dx, dy = 0, 1
        else:
            dx, dy = 0, -1
        
        nx, ny = x + dx, y + dy
        if nx < 0 or nx >= len(maze[0]) or ny < 0 or ny >= len(maze):
            break  # Out of bounds
        if maze[ny][nx] == '#':
            path.append((nx, ny))
            break  # Hit a rock
        path.append((nx, ny))
        x, y = nx, ny
    
    return path

def main():
    print("=== Deadband Protocol Maze Demo ===\n")
    
    # Generate maze
    maze, start, goal = generate_maze(20, 20, obstacle_density=0.2)
    print("Generated Maze (S=start, G=goal, #=rock, .=open):")
    print_maze(maze)
    
    # P0: Map negative space
    rocks = p0_map_negative_space(maze)
    print(f"P0 — Mapped {len(rocks)} rocks (negative space).")
    
    # P1: Find safe channels
    safe_paths = p1_find_safe_channels(maze, start, goal, rocks)
    print(f"P1 — Found {len(safe_paths)} safe channels (paths avoiding rocks).")
    
    # P2: Optimize within channel
    optimal_path = p2_optimize_within_channel(safe_paths)
    if optimal_path:
        print(f"P2 — Optimal safe path length: {len(optimal_path)} steps.")
        print("Optimal Path (marked '*'):")
        print_maze(maze, set(optimal_path))
    else:
        print("P2 — No safe path found.")
    
    # Compare with naive greedy agent (P2-only)
    print("\n--- Comparison: Naive Greedy Agent (P2-only) ---")
    greedy_path = naive_greedy_agent(maze, start, goal)
    print(f"Greedy path length: {len(greedy_path)} steps.")
    if greedy_path[-1] == goal:
        print("Result: SUCCESS (reached goal).")
    else:
        print(f"Result: FAILURE (hit rock at {greedy_path[-1]}).")
    print("Greedy Path (marked '*'):")
    print_maze(maze, set(greedy_path))
    
    # Summary
    print("\n=== Summary ===")
    print("Deadband Protocol (P0→P1→P2): Finds guaranteed safe path if one exists.")
    print("Naive Greedy (P2-only): Often fails by hitting rocks.")
    print("This demonstrates the Deadband Protocol's strict priority order.")

if __name__ == "__main__":
    main()
```

- The script generates a random 20x20 maze with 20% obstacle density.
- It follows the Deadband Protocol exactly:
  - **P0:** Identifies all '#' cells as rocks (negative space).
  - **P1:** Uses BFS to find all paths from start to goal that avoid rocks (safe channels).
  - **P2:** Selects the shortest safe path.
- It also runs a naive greedy agent that moves directly toward the goal (P2-only), which typically fails by hitting a rock.

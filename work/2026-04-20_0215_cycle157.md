import random
from collections import deque
from typing import List, Tuple, Set, Optional
import time

class Maze:
    def __init__(self, width=20, height=20, num_rocks=64):
        self.width = width
        self.height = height
        self.num_rocks = num_rocks
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_maze()
    
    def generate_maze(self):
        """Generate random rocks (obstacles)"""
        all_cells = [(x, y) for x in range(self.width) for y in range(self.height)]
        all_cells.remove(self.start)
        all_cells.remove(self.goal)
        
        self.rocks = set(random.sample(all_cells, self.num_rocks))
    
    def is_valid(self, x: int, y: int) -> bool:
        """Check if cell is within bounds and not a rock"""
        return (0 <= x < self.width and 
                0 <= y < self.height and 
                (x, y) not in self.rocks)
    
    def get_neighbors(self, x: int, y: int) -> List[Tuple[int, int]]:
        """Get valid neighboring cells (4-direction movement)"""
        neighbors = []
        for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
            nx, ny = x + dx, y + dy
            if self.is_valid(nx, ny):
                neighbors.append((nx, ny))
        return neighbors
    
    def to_ascii(self, path: Optional[List[Tuple[int, int]]] = None) -> str:
        """Convert maze to ASCII art"""
        path_set = set(path) if path else set()
        lines = []
        
        for y in range(self.height):
            line = []
            for x in range(self.width):
                if (x, y) == self.start:
                    line.append('S')
                elif (x, y) == self.goal:
                    line.append('G')
                elif (x, y) in self.rocks:
                    line.append('#')
                elif (x, y) in path_set:
                    line.append('·')
                else:
                    line.append('.')
            lines.append(''.join(line))
        
        return '\n'.join(lines)

class P2OnlyAgent(Agent):
    """Greedy optimization without rock mapping (violates protocol)"""
    def find_path(self) -> bool:
        visited = set()
        stack = [(self.maze.start, [self.maze.start])]
        
        while stack:
            (x, y), path = stack.pop()
            self.steps += 1
            
            if (x, y) == self.maze.goal:
                self.path = path
                return True
            
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Greedy: prioritize moves toward goal
            neighbors = self.maze.get_neighbors(x, y)
            neighbors.sort(key=lambda p: abs(p[0]-self.maze.goal[0]) + abs(p[1]-self.maze.goal[1]))
            
            for neighbor in neighbors:
                if neighbor not in visited:
                    stack.append((neighbor, path + [neighbor]))
        
        return False

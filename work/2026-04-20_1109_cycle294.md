## File: `deadband_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo
A simple ASCII maze simulation showing P0→P1→P2.
"""

import random
import sys
from collections import deque

# Maze symbols
EMPTY = '·'
ROCK = '█'
START = 'S'
GOAL = 'G'
PATH = '○'
SAFE = '░'  # Visual marker for safe channels

class DeadbandMaze:
    def __init__(self, width=10, height=10, rock_density=0.2):
        self.width = width
        self.height = height
        self.grid = [[EMPTY for _ in range(width)] for _ in range(height)]
        
        # Place rocks (P0 negative space)
        for y in range(height):
            for x in range(width):
                if random.random() < rock_density:
                    self.grid[y][x] = ROCK
        
        # Place start (top-left) and goal (bottom-right)
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.grid[self.start[1]][self.start[0]] = START
        self.grid[self.goal[1]][self.goal[0]] = GOAL
        
        # Ensure start and goal are not rocks
        self.grid[self.start[1]][self.start[0]] = START
        self.grid[self.goal[1]][self.goal[0]] = GOAL
        
        # State tracking
        self.safe_channels = None
        self.optimal_path = None
    
    def display(self, title="Maze"):
        print(f"\n=== {title} ===")
        for row in self.grid:
            print(' '.join(row))
        print()
    
    def p0_map_rocks(self):
        """P0: Identify negative space (rocks)."""
        rocks = []
        for y in range(self.height):
            for x in range(self.width):
                if self.grid[y][x] == ROCK:
                    rocks.append((x, y))
        
        print("P0: MAP NEGATIVE SPACE")
        print(f"Found {len(rocks)} rocks (failure states).")
        print("Rocks at coordinates:", rocks[:10], "..." if len(rocks) > 10 else "")
        return rocks
    
    def p1_find_safe_channels(self, rocks):
        """P1: Find safe channels (cells not adjacent to rocks)."""
        # Mark cells adjacent to rocks as unsafe
        unsafe = set(rocks)
        for x, y in rocks:
            for dx in [-1, 0, 1]:
                for dy in [-1, 0, 1]:
                    nx, ny = x + dx, y + dy
                    if 0 <= nx < self.width and 0 <= ny < self.height:
                        unsafe.add((nx, ny))
        
        # Safe channels are all cells not in unsafe set
        safe_channels = []
        for y in range(self.height):
            for x in range(self.width):
                if (x, y) not in unsafe and self.grid[y][x] != ROCK:
                    safe_channels.append((x, y))
        
        self.safe_channels = set(safe_channels)
        
        # Create visualization
        viz_grid = [row[:] for row in self.grid]
        for x, y in safe_channels:
            if viz_grid[y][x] == EMPTY:
                viz_grid[y][x] = SAFE
        
        print("\nP1: IDENTIFY SAFE CHANNELS")
        print(f"Found {len(safe_channels)} safe cells (deadband).")
        print("Safe channels marked with", SAFE)
        
        for row in viz_grid:
            print(' '.join(row))
        print()
        
        return safe_channels
    
    def p2_optimize_within_channel(self, safe_channels):
        """P2: Find shortest path within safe channels (BFS)."""
        if self.start not in safe_channels or self.goal not in safe_channels:
            print("ERROR: Start or goal not in safe channels. Cannot optimize.")
            return None
        
        # BFS search
        queue = deque([self.start])
        visited = {self.start: None}
        
        while queue:
            x, y = queue.popleft()
            
            if (x, y) == self.goal:
                break
            
            # 4-direction movement
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    (nx, ny) in safe_channels and (nx, ny) not in visited):
                    visited[(nx, ny)] = (x, y)
                    queue.append((nx, ny))
        
        # Reconstruct path
        if self.goal not in visited:
            print("No path found within safe channels.")
            return None
        
        path = []
        current = self.goal
        while current != self.start:
            path.append(current)
            current = visited[current]
        path.append(self.start)
        path.reverse()
        
        self.optimal_path = path
        
        # Create visualization
        viz_grid = [row[:] for row in self.grid]
        for x, y in path[1:-1]:  # Skip start and goal markers
            viz_grid[y][x] = PATH
        
        print("\nP2: OPTIMIZE WITHIN CHANNEL")
        print(f"Found optimal path of length {len(path)-1} steps.")
        print("Optimal path marked with", PATH)
        
        for row in viz_grid:
            print(' '.join(row))
        print()
        
        return path
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2."""
        print("\n" + "="*50)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*50)
        
        self.display("Initial Maze")
        
        # P0: Map negative space
        rocks = self.p0_map_rocks()
        
        # P1: Find safe channels
        safe_channels = self.p1_find_safe_channels(rocks)
        
        # P2: Optimize within channel
        path = self.p2_optimize_within_channel(safe_channels)
        
        # Summary
        print("\n" + "="*50)
        print("PROTOCOL SUMMARY")
        print("="*50)
        print(f"P0 Rocks mapped: {len(rocks)}")
        print(f"P1 Safe channels: {len(safe_channels)} cells")
        if path:
            print(f"P2 Optimal path: {len(path)-1} steps")
            print("SUCCESS: Goal reached via safe channels.")
        else:
            print("FAILURE: No safe path to goal.")
        print("="*50)

def main():
    """Run the demo."""
    print("Deadband Protocol Visual Demo")
    print("Simulating constraint navigation in a 10x10 maze.")
    print()
    print("Symbols:")
    print(f"  {START} = Start")
    print(f"  {GOAL} = Goal")
    print(f"  {ROCK} = Rock (failure state)")
    print(f"  {SAFE} = Safe channel (P1)")
    print(f"  {PATH} = Optimal path (P2)")
    print(f"  {EMPTY} = Empty space")
    
    # Create and run maze
    maze = DeadbandMaze(width=10, height=10, rock_density=0.2)
    maze.run_protocol()
    
    # Show protocol reminder
    print("\n" + "-"*40)
    print("DEADBAND PROTOCOL PRIORITY (strict):")
    print("  P0: Map negative space (what NOT to do)")
    print("  P1: Find safe channels (where you CAN be)")
    print("  P2: Optimize within channels (best path)")
    print("-"*40)

if __name__ == "__main__":
    main()
```

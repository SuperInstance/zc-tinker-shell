#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Simulates the three strategies from the fleet's constraint_sim_final.json:
1. Unconstrained greedy (P2-only) → hits rocks
2. Constrained but without P0 mapping (P1+P2 wrong order) → fails
3. Full Deadband Protocol (P0→P1→P2) → succeeds
"""

import random
from typing import List, Tuple, Set

# Constants matching fleet simulation
MAZE_SIZE = 20
ROCK_COUNT = 64
START = (0, 0)
GOAL = (19, 19)

# Symbols for visualization
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_MARKER = 'G'
PATH = '○'
SAFE = ' '

class Maze:
    def __init__(self, size: int = MAZE_SIZE, rock_count: int = ROCK_COUNT):
        self.size = size
        self.grid = [[EMPTY for _ in range(size)] for _ in range(size)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_rocks(rock_count)
        self.grid[START[0]][START[1]] = AGENT
        self.grid[GOAL[0]][GOAL[1]] = GOAL_MARKER
        
    def generate_rocks(self, count: int):
        """Randomly place rocks, avoiding start and goal."""
        while len(self.rocks) < count:
            x = random.randint(0, self.size-1)
            y = random.randint(0, self.size-1)
            if (x, y) != START and (x, y) != GOAL:
                self.rocks.add((x, y))
                self.grid[x][y] = ROCK
    
    def is_rock(self, pos: Tuple[int, int]) -> bool:
        return pos in self.rocks
    
    def is_valid(self, pos: Tuple[int, int]) -> bool:
        x, y = pos
        return 0 <= x < self.size and 0 <= y < self.size
    
    def print_maze(self, path: List[Tuple[int, int]] = None):
        """Print maze with optional path overlay."""
        display = [row[:] for row in self.grid]
        if path:
            for (x, y) in path[1:-1]:  # Exclude start and goal
                if display[x][y] == EMPTY:
                    display[x][y] = PATH
        for row in display:
            print(' '.join(row))
        print()

def manhattan(a: Tuple[int, int], b: Tuple[int, int]) -> int:
    return abs(a[0] - b[0]) + abs(a[1] - b[1])

def get_neighbors(pos: Tuple[int, int]) -> List[Tuple[int, int]]:
    x, y = pos
    return [(x+1, y), (x-1, y), (x, y+1), (x, y-1)]

# ===== STRATEGY 1: Unconstrained Greedy (P2-only) =====
def unconstrained_greedy(maze: Maze) -> Tuple[List[Tuple[int, int]], bool]:
    """P2-only: Greedy move toward goal, ignoring rocks."""
    path = [START]
    current = START
    visited = {START}
    
    while current != GOAL:
        neighbors = get_neighbors(current)
        # Sort by distance to goal (greedy)
        neighbors.sort(key=lambda p: manhattan(p, GOAL))
        
        moved = False
        for neighbor in neighbors:
            if maze.is_valid(neighbor) and neighbor not in visited:
                if maze.is_rock(neighbor):
                    # Hits rock → failure
                    return path, False
                current = neighbor
                path.append(current)
                visited.add(current)
                moved = True
                break
        
        if not moved:
            # No valid move
            return path, False
    
    return path, True

# ===== STRATEGY 2: Constrained but Wrong Order (P1+P2 without P0) =====
def constrained_wrong_order(maze: Maze) -> Tuple[List[Tuple[int, int]], bool]:
    """Tries to avoid rocks but without proper P0 mapping → gets stuck."""
    path = [START]
    current = START
    visited = {START}
    
    while current != GOAL:
        neighbors = get_neighbors(current)
        # Filter out invalid and visited
        valid_neighbors = [n for n in neighbors if maze.is_valid(n) and n not in visited]
        
        if not valid_neighbors:
            return path, False
        
        # Try to avoid rocks (naive P1 without P0)
        safe_neighbors = [n for n in valid_neighbors if not maze.is_rock(n)]
        
        if safe_neighbors:
            # Move to safest (closest to goal)
            safe_neighbors.sort(key=lambda p: manhattan(p, GOAL))
            current = safe_neighbors[0]
        else:
            # Forced to hit rock
            return path, False
        
        path.append(current)
        visited.add(current)
    
    return path, True

# ===== STRATEGY 3: Deadband Protocol (P0→P1→P2) =====
def deadband_protocol(maze: Maze) -> Tuple[List[Tuple[int, int]], bool]:
    """Full protocol: Map rocks (P0), find safe channels (P1), optimize (P2)."""
    # P0: Map negative space (rocks)
    rocks = maze.rocks  # Already known
    
    # P1: Find safe channels (BFS avoiding rocks)
    from collections import deque
    
    queue = deque([START])
    came_from = {START: None}
    visited = {START}
    
    while queue:
        current = queue.popleft()
        if current == GOAL:
            break
        
        for neighbor in get_neighbors(current):
            if (maze.is_valid(neighbor) and 
                neighbor not in visited and 
                not maze.is_rock(neighbor)):
                visited.add(neighbor)
                came_from[neighbor] = current
                queue.append(neighbor)
    
    # Reconstruct path if goal reached
    if GOAL not in came_from:
        return [], False
    
    # Build path from goal to start
    path = []
    current = GOAL
    while current is not None:
        path.append(current)
        current = came_from[current]
    path.reverse()
    
    return path, True

def run_simulation(iterations: int = 50):
    """Run all three strategies multiple times."""
    results = {
        'unconstrained': {'success': 0, 'paths': []},

        if success2:
            results['constrained_wrong']['success'] += 1
        results['constrained_wrong']['paths'].append(path2)
        
        # Strategy 3
        path3, success3 = deadband_protocol(maze)
        if success3:
            results['deadband']['success'] += 1
        results['deadband']['paths'].append(path3)
    
    return results

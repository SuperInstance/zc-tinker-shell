# ========== MAZE GENERATION ==========
def generate_maze(width: int, height: int, rock_density: float = 0.2) -> List[List[str]]:
    """Generate a random maze with start (S), goal (G), and rocks (#)."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks randomly
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
    
    # Ensure start and goal are clear and not the same
    start = (0, 0)
    goal = (width-1, height-1)
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    
    # Clear a few cells around start/goal to ensure connectivity
    for dx, dy in [(0,1),(1,0),(0,-1),(-1,0)]:
        nx, ny = start[0]+dx, start[1]+dy
        if 0 <= nx < width and 0 <= ny < height:
            if maze[ny][nx] == '#':
                maze[ny][nx] = '.'
        nx, ny = goal[0]+dx, goal[1]+dy
        if 0 <= nx < width and 0 <= ny < height:
            if maze[ny][nx] == '#':
                maze[ny][nx] = '.'
    
    return maze

def p0_map_rocks(maze: List[List[str]]) -> Set[Tuple[int,int]]:
    """P0: Identify all rock positions (negative space)."""
    rocks = set()
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(maze: List[List[str]], rocks: Set[Tuple[int,int]]) -> List[List[Tuple[int,int]]]:
    """
    P1: Find all safe paths from start to goal using BFS.
    Returns list of paths (each path is list of coordinates).
    """
    width, height = len(maze[0]), len(maze)
    start, goal = None, None
    
    # Locate start and goal
    for y in range(height):
        for x in range(width):
            if maze[y][x] == 'S':
                start = (x, y)
            elif maze[y][x] == 'G':
                goal = (x, y)
    
    if not start or not goal:
        return []
    
    # BFS to find all shortest safe paths
    from collections import deque
    queue = deque([(start, [start])])
    visited = {start: 0}
    safe_paths = []
    shortest_length = None
    
    while queue:
        (x, y), path = queue.popleft()
        
        if (x, y) == goal:
            if shortest_length is None:
                shortest_length = len(path)
            if len(path) == shortest_length:
                safe_paths.append(path)
            continue
        
        for dx, dy in [(0,1),(1,0),(0,-1),(-1,0)]:
            nx, ny = x+dx, y+dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) in rocks:
                    continue  # Avoid rocks (P0)
                if maze[ny][nx] == '#':
                    continue  # Also avoid rocks
                
                new_path = path + [(nx, ny)]
                if (nx, ny) not in visited or len(new_path) <= visited[(nx, ny)]:
                    visited[(nx, ny)] = len(new_path)
                    queue.append(((nx, ny), new_path))
    
    return safe_paths

def p2_optimize_within_channels(safe_paths: List[List[Tuple[int,int]]]) -> List[Tuple[int,int]]:
    """P2: Select the optimal path from safe channels (here: first shortest path)."""
    if not safe_paths:
        return []
    # Simple optimization: pick first shortest path
    return safe_paths[0]

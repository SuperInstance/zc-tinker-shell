#!/usr/bin/env python3
"""
deadband_maze_demo.py
Demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque
import heapq

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_count=64):
        self.width = width
        self.height = height
        self.rock_count = rock_count
        self.rocks = set()  # P0: Negative space
        self.safe_channels = set()  # P1: Safe positions
        self.path = []  # P2: Optimal path
        
        # Fixed start and goal positions (matching simulation)
        self.start = (0, 0)
        self.goal = (19, 19)
        
    def generate_rocks(self):
        """P0: Map negative space (what NOT to do)"""
        self.rocks.clear()
        while len(self.rocks) < self.rock_count:
            x = random.randint(0, self.width - 1)
            y = random.randint(0, self.height - 1)
            pos = (x, y)
            # Don't place rocks on start or goal
            if pos != self.start and pos != self.goal:
                self.rocks.add(pos)
        return self.rocks
    
    def find_safe_channels(self):
        """P1: Find safe channels (where you CAN be)"""
        self.safe_channels.clear()
        visited = set()
        queue = deque([self.start])
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
            visited.add((x, y))
            
            # Add to safe channels if not a rock
            if (x, y) not in self.rocks:
                self.safe_channels.add((x, y))
                
                # Explore neighbors
                for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                    nx, ny = x + dx, y + dy
                    if 0 <= nx < self.width and 0 <= ny < self.height:
                        if (nx, ny) not in visited and (nx, ny) not in self.rocks:
                            queue.append((nx, ny))
        
        return self.safe_channels
    
    def optimize_within_channels(self):
        """P2: Optimize within channels (best path through safe water)"""
        if self.goal not in self.safe_channels:
            return []  # No path possible
            
        # A* search constrained to safe channels only
        open_set = []
        heapq.heappush(open_set, (0, self.start))
        
        came_from = {}
        g_score = {self.start: 0}
        f_score = {self.start: self._heuristic(self.start, self.goal)}
        
        while open_set:
            _, current = heapq.heappop(open_set)
            
            if current == self.goal:
                # Reconstruct path
                self.path = []
                while current in came_from:
                    self.path.append(current)
                    current = came_from[current]
                self.path.reverse()
                return self.path
            
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                
                # CRITICAL: Only move through safe channels (P1 constraint)
                if neighbor not in self.safe_channels:
                    continue
                    
                if not (0 <= neighbor[0] < self.width and 0 <= neighbor[1] < self.height):
                    continue
                
                tentative_g = g_score[current] + 1
                
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self._heuristic(neighbor, self.goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
        
        return []  # No path found
    
    def _heuristic(self, a, b):
        """Manhattan distance heuristic"""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    def run_protocol(self):
        """Execute full Deadband Protocol P0→P1→P2"""
        print("=== DEADBAND PROTOCOL DEMONSTRATION ===")
        print(f"Maze: {self.width}x{self.height}, Rocks: {self.rock_count}")
        print(f"Start: {self.start}, Goal: {self.goal}")
        print()
        
        # P0: Map negative space
        print("P0: MAPPING NEGATIVE SPACE (rocks)")
        rocks = self.generate_rocks()
        print(f"  → Identified {len(rocks)} rock positions to avoid")
        
        # P1: Find safe channels
        print("P1: FINDING SAFE CHANNELS")
        safe = self.find_safe_channels()
        print(f"  → Found {len(safe)} safe positions")
        print(f"  → Goal reachable: {self.goal in safe}")
        
        # P2: Optimize within channels
        print("P2: OPTIMIZING WITHIN SAFE CHANNELS")
        path = self.optimize_within_channels()
        if path:
            print(f"  → Found optimal path of length {len(path)}")
            print(f"  → Path cost: {len(path)} steps")
        else:
            print("  → No path found (goal unreachable through safe channels)")
        
        return len(path) if path else 0
    
    def visualize(self):
        """Create ASCII visualization of the maze"""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')
                elif pos in self.path:
                    row.append('·')
                elif pos in self.safe_channels:
                    row.append(' ')
                else:
                    row.append('░')  # Unreachable/unsafe
            grid.append(''.join(row))
        
        print("\n=== MAZE VISUALIZATION ===")
        print("Legend: S=Start, G=Goal, █=Rock, ·=Path, ░=Unsafe/Unreachable")
        print("-" * (self.width + 2))
        for row in grid:
            print(f"|{row}|")
        print("-" * (self.width + 2))

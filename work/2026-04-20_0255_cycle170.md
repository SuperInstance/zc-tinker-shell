#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 with ASCII visualization.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time
from typing import List, Tuple, Set

class MazeAgent:
    """Agent that navigates a maze using Deadband Protocol."""
    
    def __init__(self, maze_size: int = 10, rock_density: float = 0.2):
        self.maze_size = maze_size
        self.rock_density = rock_density
        self.maze = []
        self.rocks = set()  # P0: Negative space
        self.safe_channels = set()  # P1: Safe cells
        self.path = []  # P2: Optimal path
        self.start = (0, 0)
        self.goal = (maze_size-1, maze_size-1)
        
        self.generate_maze()
    
    def generate_maze(self):
        """Generate random maze with rocks."""
        self.maze = []
        self.rocks.clear()
        
        for y in range(self.maze_size):
            row = []
            for x in range(self.maze_size):
                if (x, y) == self.start or (x, y) == self.goal:
                    row.append('S' if (x, y) == self.start else 'G')
                elif random.random() < self.rock_density:
                    row.append('█')  # Rock
                    self.rocks.add((x, y))
                else:
                    row.append('·')  # Open space
            self.maze.append(row)
    
    def print_maze(self, title: str, highlight: Set[Tuple[int, int]] = None):
        """Print maze with optional highlighted cells."""
        print(f"\n{title}")
        print("  " + "".join(str(i%10) for i in range(self.maze_size)))
        
        for y in range(self.maze_size):
            print(f"{y%10} ", end="")
            for x in range(self.maze_size):
                cell = self.maze[y][x]
                if highlight and (x, y) in highlight:
                    print('○', end='')  # Highlighted
                else:
                    print(cell, end='')
            print()
    
    def p0_map_negative_space(self):
        """P0: Identify rocks (what NOT to do)."""
        print("\n" + "="*50)
        print("P0: MAP NEGATIVE SPACE")
        print("="*50)
        print(f"Identified {len(self.rocks)} rocks (failure states)")
        self.print_maze("Maze with rocks marked '█'", self.rocks)
        return self.rocks
    
    def p1_find_safe_channels(self):
        """P1: Find safe cells (where you CAN be)."""
        print("\n" + "="*50)
        print("P1: FIND SAFE CHANNELS")
        print("="*50)
        
        # Start with all non-rock cells
        self.safe_channels = set()
        for y in range(self.maze_size):
            for x in range(self.maze_size):
                if (x, y) not in self.rocks:
                    self.safe_channels.add((x, y))
        
        print(f"Found {len(self.safe_channels)} safe cells")
        self.print_maze("Safe channels (all non-rock cells)", self.safe_channels)
        return self.safe_channels
    
    def p2_optimize_within_channel(self):
        """P2: Find optimal path through safe channels."""
        print("\n" + "="*50)
        print("P2: OPTIMIZE WITHIN CHANNEL")
        print("="*50)
        
        # Simple BFS through safe channels only
        queue = [(self.start, [self.start])]
        visited = set([self.start])
        
        while queue:
            (x, y), current_path = queue.pop(0)
            
            if (x, y) == self.goal:
                self.path = current_path
                break
            
            # Only move to adjacent safe cells
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if (0 <= nx < self.maze_size and 0 <= ny < self.maze_size and
                    (nx, ny) in self.safe_channels and (nx, ny) not in visited):
                    visited.add((nx, ny))
                    queue.append(((nx, ny), current_path + [(nx, ny)]))
        
        if self.path:
            print(f"Found optimal path with {len(self.path)} steps")
            self.print_maze("Optimal path through safe channels", set(self.path))
        else:
            print("No path found through safe channels")
        
        return self.path
    
    def run_unconstrained_greedy(self):
        """Demonstrate P2-only (greedy) failure mode."""
        print("\n" + "="*50)
        print("UNCONSTRAINED GREEDY (P2-ONLY) - WILL FAIL")
        print("="*50)
        
        path = []
        x, y = self.start
        visited = set([(x, y)])
        
        # Greedy: move toward goal ignoring rocks
        steps = 0
        max_steps = self.maze_size * 3
        
        while (x, y) != self.goal and steps < max_steps:
            # Move toward goal
            if x < self.goal[0]:
                x += 1
            elif y < self.goal[1]:
                y += 1
            elif x > self.goal[0]:
                x -= 1
            elif y > self.goal[1]:
                y -= 1
            
            path.append((x, y))
            visited.add((x, y))
            steps += 1
            
            # Check if hit rock
            if (x, y) in self.rocks:
                print(f"FAILED: Hit rock at ({x}, {y}) after {steps} steps")
                self.print_maze("Greedy path (hit rock)", visited)
                return False
        
        success = (x, y) == self.goal
        print(f"{'SUCCESS' if success else 'FAILED'} after {steps} steps")
        if success:
            self.print_maze("Greedy path (lucky success)", visited)
        return success
    
    def demonstrate_deadband_protocol(self):
        """Full P0→P1→P2 demonstration."""
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("="*60)
        
        # Show initial maze
        self.print_maze("Initial Maze (S=Start, G=Goal, █=Rock)")
        
        # Step through protocol
        rocks = self.p0_map_negative_space()
        time.sleep(1)
        
        safe_cells = self.p1_find_safe_channels()
        time.sleep(1)
        
        optimal_path = self.p2_optimize_within_channel()
        
        # Results
        print("\n" + "="*60)
        print("RESULTS")
        print("="*60)
        print(f"P0: Mapped {len(rocks)} rocks (negative space)")
        print(f"P1: Found {len(safe_cells)} safe cells")
        print(f"P2: Found {'optimal' if optimal_path else 'no'} path")
        
        if optimal_path:
            print(f"Path length: {len(optimal_path)} steps")
            print("Protocol SUCCESS")
        else:
            print("Protocol FAILED (no safe path exists)")
        
        return bool(optimal_path)

def main():
    """Run demonstration."""
    print("Fleet Knowledge: Deadband Protocol Maze Demo")
    print("Based on constraint_sim_final.json results:")
    print("  - Unconstrained (P2-only): 0/50 success")
    print("  - Deadband (P0→P1→P2): 50/50 success")

    # First show unconstrained failure
    input("\nPress Enter to see unconstrained greedy (P2-only) failure...")
    agent.run_unconstrained_greedy()
    
    # Reset for deadband protocol
    agent.generate_maze()  # New random maze

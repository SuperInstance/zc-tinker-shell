```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Based on fleet-knowledge/DEADBAND-PROTOCOL.md and constraint_sim_final.json
"""

import random
import time

# ========== MAZE DEFINITION ==========
# 20x20 grid (matches simulation data)
MAZE_W = 20
MAZE_H = 20

# Symbols
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL = 'G'
PATH = '○'
SAFE = '░'  # safe channel marker

# Generate a maze with rocks (P0 negative space)
def generate_maze(rock_density=0.16):  # ~64 rocks in 400 cells
    maze = [[EMPTY for _ in range(MAZE_W)] for _ in range(MAZE_H)]
    rocks = []
    for y in range(MAZE_H):
        for x in range(MAZE_W):
            if random.random() < rock_density:
                maze[y][x] = ROCK
                rocks.append((x, y))
    # Ensure start (0,0) and goal (19,19) are clear
    maze[0][0] = EMPTY
    maze[MAZE_H-1][MAZE_W-1] = GOAL
    return maze, rocks

# ========== DEADBAND PROTOCOL ==========
def deadband_protocol_demo(maze):
    print("\n" + "="*60)
    print("DEADBAND PROTOCOL DEMO")
    print("="*60)
    
    # P0: MAP NEGATIVE SPACE (rocks)
    print("\n[P0] MAP NEGATIVE SPACE")
    rocks = []
    for y in range(MAZE_H):
        for x in range(MAZE_W):
            if maze[y][x] == ROCK:
                rocks.append((x, y))
    print(f"Identified {len(rocks)} rocks (failure states).")
    time.sleep(1.5)
    
    # P1: IDENTIFY SAFE CHANNELS (deadbands)
    print("\n[P1] IDENTIFY SAFE CHANNELS")
    # Simple flood‑fill from start, avoiding rocks
    safe = [[False for _ in range(MAZE_W)] for _ in range(MAZE_H)]
    stack = [(0, 0)]
    while stack:
        x, y = stack.pop()
        if safe[y][x]:
            continue
        safe[y][x] = True
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_W and 0 <= ny < MAZE_H:
                if maze[ny][nx] != ROCK and not safe[ny][nx]:
                    stack.append((nx, ny))
    
    # Mark safe channels in maze display
    maze_p1 = [row[:] for row in maze]
    for y in range(MAZE_H):
        for x in range(MAZE_W):
            if safe[y][x] and maze_p1[y][x] == EMPTY:
                maze_p1[y][x] = SAFE
    print_maze(maze_p1, "Safe channels (deadbands) marked with '░'")
    print(f"Safe channels cover {sum(sum(row) for row in safe)} cells.")
    time.sleep(2.0)
    
    # P2: OPTIMIZE WITHIN CHANNEL (A* only through safe cells)
    print("\n[P2] OPTIMIZE WITHIN SAFE CHANNEL (A* pathfinding)")
    # A* implementation constrained to safe cells
    def heuristic(a, b):
        return abs(a[0]-b[0]) + abs(a[1]-b[1])
    
    start = (0, 0)
    goal = (MAZE_W-1, MAZE_H-1)
    
    open_set = {start}
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        current = min(open_set, key=lambda p: f_score.get(p, float('inf')))
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            break
        open_set.remove(current)
        x, y = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            neighbor = (nx, ny)
            if 0 <= nx < MAZE_W and 0 <= ny < MAZE_H:
                if not safe[ny][nx]:
                    continue  # NOT in safe channel → skip
                tentative_g = g_score[current] + 1
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
    else:
        path = []  # no path found
    
    # Mark path on maze
    maze_p2 = [row[:] for row in maze]
    for (x, y) in path:
        if maze_p2[y][x] == EMPTY:
            maze_p2[y][x] = PATH
    maze_p2[0][0] = AGENT
    maze_p2[MAZE_H-1][MAZE_W-1] = GOAL
    
    success = len(path) > 0
    print_maze(maze_p2, f"Optimized path ({len(path)} steps) through safe channels")
    print(f"Path found: {success}")
    if success:
        print(f"Path length: {len(path)} steps (optimal ≤21 per simulation)")
    time.sleep(1.5)
    
    return success, len(path)

# ========== UTILITIES ==========
def print_maze(maze, title=""):
    print(f"\n{title}")
    print("   " + "".join(f"{i%10}" for i in range(MAZE_W)))
    for y in range(MAZE_H):
        row = f"{y:2} "
        for x in range(MAZE_W):
            row += maze[y][x]
        print(row)

# ========== MAIN ==========
def main():
    random.seed(42)  # reproducible for demo
    maze, rocks = generate_maze()
    
    print("\nInitial Maze (20×20)")
    print("Legend: · empty, █ rock, A agent, G goal")
    print_maze(maze)
    print(f"Rocks: {len(rocks)} (≈{len(rocks)*100/(MAZE_W*MAZE_H):.1f}% density)")
    
    success, steps = deadband_protocol_demo(maze)
    
    # Summary with simulation data reference
    print("\n" + "="*60)
    print("SUMMARY (vs fleet simulation data)")
    print("="*60)
    print("Fleet simulation results (constraint_sim_final.json):")
    print("  - Unconstrained greedy (P2-only): 0/50 success")
    print("  - Deadband protocol (P0→P1→P2): 50/50 success")
    print("  - Optimal path length: 21 steps")
    print(f"\nThis demo run:")
    print(f"  - Path found: {'YES' if success else 'NO'}")
    print(f"  - Path length: {steps if success else 'N/A'} steps")
    print("\nProtocol reminder:")
    print("  P0: Map rocks (failure states)")
    print("  P1: Identify safe channels (deadbands)")
    print("  P2: Optimize within safe channels")

if __name__ == "__main__":
    main()
```

  2. **P1:** Flood‑fills safe channels (deadbands), marked with `░`.
  3. **P2:** Runs A* pathfinding constrained to safe cells only.

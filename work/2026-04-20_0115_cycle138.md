```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map rocks (obstacles)
P1: Find safe channels (walkable paths)
P2: Find shortest path within safe channels
"""

import random
from collections import deque

def generate_maze(width=20, height=20, rock_density=0.3):
    """Generate a random maze with rocks (X) and open space (.)"""
    maze = []
    for y in range(height):
        row = []
        for x in range(width):
            if random.random() < rock_density:
                row.append('X')  # Rock
            else:
                row.append('.')  # Open
        maze.append(row)
    
    # Ensure start (S) and goal (G) are open
    maze[0][0] = 'S'
    maze[height-1][width-1] = 'G'
    return maze

def print_maze(maze, path=None):
    """Print maze with optional path marked"""
    for y, row in enumerate(maze):
        line = []
        for x, cell in enumerate(row):
            if path and (y, x) in path:
                line.append('*')
            else:
                line.append(cell)
        print(''.join(line))

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks)"""
    rocks = []
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == 'X':
                rocks.append((y, x))
    return rocks

def p1_find_safe_channels(maze, start, goal):
    """P1: Find all safe paths (BFS) avoiding rocks"""
    height, width = len(maze), len(maze[0])
    directions = [(0,1),(1,0),(0,-1),(-1,0)]
    
    # BFS to find all reachable positions
    queue = deque([start])
    visited = {start: None}
    
    while queue:
        y, x = queue.popleft()
        
        if (y, x) == goal:
            # Reconstruct path
            path = []
            current = goal
            while current != start:
                path.append(current)
                current = visited[current]
            path.append(start)
            path.reverse()
            return path
        
        for dy, dx in directions:
            ny, nx = y + dy, x + dx
            if 0 <= ny < height and 0 <= nx < width:
                if maze[ny][nx] != 'X' and (ny, nx) not in visited:
                    visited[(ny, nx)] = (y, x)
                    queue.append((ny, nx))
    
    return None  # No safe path

def p2_optimize_within_channels(maze, safe_path):
    """P2: Optimize path (simplify straight lines)"""
    if not safe_path or len(safe_path) < 3:
        return safe_path
    
    optimized = [safe_path[0]]
    for i in range(1, len(safe_path)-1):
        py, px = optimized[-1]
        cy, cx = safe_path[i]
        ny, nx = safe_path[i+1]
        
        # Keep if direction changes
        if (cy-py, cx-px) != (ny-cy, nx-cx):
            optimized.append((cy, cx))
    
    optimized.append(safe_path[-1])
    return optimized

def run_deadband_demo():
    """Main demonstration of Deadband Protocol"""
    print("=== DEADBAND PROTOCOL MAZE DEMO ===\n")
    
    # Generate maze
    maze = generate_maze(width=15, height=10, rock_density=0.25)
    start = (0, 0)
    goal = (len(maze)-1, len(maze[0])-1)
    
    print("Initial Maze (S=start, G=goal, X=rock, .=open):")
    print_maze(maze)
    print()
    
    # P0: Map negative space
    rocks = p0_map_rocks(maze)
    print(f"P0: Mapped {len(rocks)} rocks (negative space)")
    
    # P1: Find safe channels
    safe_path = p1_find_safe_channels(maze, start, goal)
    if not safe_path:
        print("P1: NO SAFE CHANNEL FOUND - Goal unreachable")
        return
    
    print(f"P1: Found safe channel ({len(safe_path)} steps)")
    print("Safe path (P1 only):")
    print_maze(maze, set(safe_path))
    print()
    
    # P2: Optimize within channel
    optimized_path = p2_optimize_within_channels(maze, safe_path)
    print(f"P2: Optimized to {len(optimized_path)} steps")
    print("Optimized path (P0→P1→P2):")
    print_maze(maze, set(optimized_path))
    print()
    
    # Metrics
    print("=== RESULTS ===")
    print(f"Rocks avoided (P0): {len(rocks)}")
    print(f"Safe path length (P1): {len(safe_path)}")
    print(f"Optimized length (P2): {len(optimized_path)}")
    print(f"Improvement: {len(safe_path)-len(optimized_path)} steps saved")
    
    # Demonstrate protocol violation
    print("\n=== PROTOCOL VIOLATION EXAMPLE ===")
    print("If we skip P0+P1 and go straight to P2 (greedy):")
    
    # Greedy approach (P2-only)
    def greedy_path(maze, start, goal):
        """Greedy path that ignores rocks"""
        y, x = start
        gy, gx = goal
        path = [start]
        
        while (y, x) != (gy, gx):
            if y < gy:
                y += 1
            elif y > gy:
                y -= 1
            elif x < gx:
                x += 1
            elif x > gx:
                x -= 1
            
            if maze[y][x] == 'X':
                print(f"  Hit rock at ({y},{x}) - FAILED")
                return None
            path.append((y, x))
        
        return path
    
    greedy = greedy_path(maze, start, goal)
    if greedy:
        print(f"  Greedy path found: {len(greedy)} steps")
    else:
        print("  Greedy approach failed (hit rock)")

if __name__ == "__main__":
    run_deadband_demo()
```

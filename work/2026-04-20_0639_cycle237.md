```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from typing import List, Tuple, Set

# Constants from simulation
MAZE_SIZE = 20
ROCK_COUNT = 64  # Approx 16% density, matching sim
START = (0, 0)
GOAL = (19, 19)

class DeadbandMaze:
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.agent_pos = START
        self.goal_pos = GOAL
        
        # Mark start and goal
        self.grid[START[0]][START[1]] = 'S'
        self.grid[GOAL[0]][GOAL[1]] = 'G'
        
    def generate_rocks(self):
        """P0: Map negative space (rocks)"""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != START and (x, y) != GOAL:
                self.rocks.add((x, y))
                self.grid[x][y] = '█'
    
    def find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Identify all safe paths (avoid rocks) using BFS"""
        from collections import deque
        
        safe_paths = []
        queue = deque([(START, [START])])
        visited = {START}
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == GOAL:
                safe_paths.append(path)
                continue
            
            for dx, dy in [(0,1), (1,0), (0,-1), (-1,0)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        return safe_paths
    
    def optimize_path(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """P2: Select shortest safe path"""
        if not safe_paths:
            return []
        return min(safe_paths, key=len)
    
    def greedy_path(self) -> List[Tuple[int, int]]:
        """P2-only (violation): greedy toward goal ignoring rocks"""
        path = [START]
        x, y = START
        visited = {START}
        
        while (x, y) != GOAL:
            # Move toward goal (Manhattan direction)
            dx = 1 if GOAL[0] > x else -1 if GOAL[0] < x else 0
            dy = 1 if GOAL[1] > y else -1 if GOAL[1] < y else 0
            
            # Try primary direction
            nx, ny = x + dx, y + dy
            if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                if (nx, ny) not in visited:
                    x, y = nx, ny
                    path.append((x, y))
                    visited.add((x, y))
                    continue
            
            # Fallback: any valid move
            moved = False
            for dx, dy in [(0,1), (1,0), (0,-1), (-1,0)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in visited:
                        x, y = nx, ny
                        path.append((x, y))
                        visited.add((x, y))
                        moved = True
                        break
            if not moved:
                break  # Stuck
        
        return path
    
    def visualize(self, path: List[Tuple[int, int]] = None, title: str = ""):
        """Display maze with optional path"""
        display = [row[:] for row in self.grid]
        
        if path:
            for (x, y) in path[1:-1]:  # Don't overwrite S and G
                if display[x][y] == '.':
                    display[x][y] = '·'
        
        print(f"\n{'='*60}")
        print(f"{title}")
        print(f"Maze: {MAZE_SIZE}x{MAZE_SIZE}, Rocks: {ROCK_COUNT}")
        print(f"Start: S, Goal: G, Rocks: █, Path: ·")
        print('-'*40)
        
        for row in display:
            print(' '.join(row))
        
        if path:
            print(f"\nPath length: {len(path)} steps")
            success = path[-1] == GOAL
            print(f"Success: {'YES' if success else 'NO'}")
            if not success:
                print(f"Stopped at: {path[-1]}")
        
        print('='*60)

def main():
    print("DEADBAND PROTOCOL DEMONSTRATION")
    print("Based on fleet simulation: deadband_success=50/50, constrained_success=0/50")
    print("P0: Map rocks | P1: Find safe channels | P2: Optimize within channels")
    
    # Create maze
    maze = DeadbandMaze()
    maze.generate_rocks()
    
    # Show initial state
    maze.visualize(title="INITIAL MAZE (P0: Rocks Mapped)")
    
    # VIOLATION: P2-only (greedy optimization without P0+P1)
    greedy_path = maze.greedy_path()
    maze.visualize(greedy_path, title="VIOLATION: P2-ONLY (Greedy, ignores rocks)")
    
    # CORRECT: P0→P1→P2
    print("\n\nCORRECT PROTOCOL EXECUTION:")
    print("P0: Rocks already mapped")
    
    safe_paths = maze.find_safe_channels()
    print(f"P1: Found {len(safe_paths)} safe channels")
    
    if safe_paths:
        optimal_path = maze.optimize_path(safe_paths)
        maze.visualize(optimal_path, title="P2: Optimal path within safe channels")
        
        # Statistics
        print("\nSTATISTICS:")
        print(f"Safe channels found: {len(safe_paths)}")
        print(f"Shortest safe path: {len(optimal_path)} steps")
        print(f"Goal reached: {'YES' if optimal_path[-1] == GOAL else 'NO'}")
        
        # Compare
        print(f"\nCOMPARISON:")
        print(f"P2-only (greedy): {len(greedy_path)} steps, Success: {greedy_path[-1] == GOAL}")
        print(f"P0→P1→P2 (deadband): {len(optimal_path)} steps, Success: {optimal_path[-1] == GOAL}")
        
        # Show protocol adherence
        print(f"\nPROTOCOL ADHERENCE:")
        print("✓ P0: Negative space mapped (rocks identified)")
        print("✓ P1: Safe channels enumerated")
        print("✓ P2: Optimization constrained to safe channels")
    else:
        print("ERROR: No safe channels found (maze may be impossible)")

if __name__ == "__main__":
    random.seed(42)  # Reproducible for demo
    main()
```

**Explanation:**
1. **P0 Implementation**: `generate_rocks()` maps negative space (64 rocks at random positions).
2. **P1 Implementation**: `find_safe_channels()` performs BFS to find all paths avoiding rocks.
3. **P2 Implementation**: `optimize_path()` selects shortest path from safe channels only.
4. **Violation Example**: `greedy_path()` shows P2-only approach that ignores rocks.
5. **Visualization**: ASCII grid shows rocks (█), start (S), goal (G), and path (·).

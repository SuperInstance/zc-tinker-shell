```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Simulation parameters: 20×20 grid, 64 rocks, optimal path length 21
Demonstrates P0→P1→P2 protocol phases.
"""

import random
from typing import List, Tuple, Set
from collections import deque

class DeadbandMaze:
    def __init__(self, width=20, height=20, rock_count=64):
        self.width = width
        self.height = height
        self.rock_count = rock_count
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Initialize grid
        self.grid = [['.' for _ in range(width)] for _ in range(height)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.agent_path = []
        
        # Phase tracking
        self.phase0_complete = False
        self.phase1_complete = False
        self.phase2_complete = False
        
    def generate_rocks(self):
        """P0: Map negative space (rocks)"""
        positions = []
        while len(self.rocks) < self.rock_count:
            x = random.randint(0, self.width-1)
            y = random.randint(0, self.height-1)
            pos = (x, y)
            if pos != self.start and pos != self.goal and pos not in self.rocks:
                self.rocks.add(pos)
                self.grid[y][x] = '█'  # Rock symbol
        
        self.phase0_complete = True
        return list(self.rocks)
    
    def find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """P1: Identify all safe paths from start to goal"""
        if not self.phase0_complete:
            print("ERROR: Must complete P0 (map rocks) before P1")
            return []
        
        # BFS to find all safe paths
        queue = deque([(self.start, [self.start])])
        safe_paths = []
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            # Explore neighbors (right, down, left, up)
            for dx, dy in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in path:
                        queue.append(((nx, ny), path + [(nx, ny)]))
        
        self.phase1_complete = True
        return safe_paths
    
    def optimize_path(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """P2: Find shortest path among safe channels"""
        if not self.phase1_complete:
            print("ERROR: Must complete P1 (find safe channels) before P2")
            return []
        
        if not safe_paths:
            print("WARNING: No safe paths found!")
            return []
        
        # Find shortest path (optimal)
        optimal_path = min(safe_paths, key=len)
        self.agent_path = optimal_path
        
        # Mark path on grid
        for x, y in optimal_path[1:-1]:  # Skip start and goal
            if self.grid[y][x] == '.':
                self.grid[y][x] = '·'  # Path marker
        
        self.phase2_complete = True
        return optimal_path
    
    def display(self, show_phases=False):
        """Display the maze with current state"""
        display_grid = [row[:] for row in self.grid]
        
        # Mark start and goal
        sx, sy = self.start
        gx, gy = self.goal
        display_grid[sy][sx] = 'S'
        display_grid[gy][gx] = 'G'
        
        # Mark agent if on path
        if self.agent_path:
            for x, y in self.agent_path[1:-1]:
                if display_grid[y][x] == '·':
                    display_grid[y][x] = '○'
        
        print("\n" + "="*50)
        print("DEADBAND PROTOCOL MAZE DEMO")
        print(f"Grid: {self.width}×{self.height}, Rocks: {self.rock_count}")
        print("="*50)
        
        if show_phases:
            print(f"P0 (Map rocks): {'✓' if self.phase0_complete else '✗'}")
            print(f"P1 (Safe channels): {'✓' if self.phase1_complete else '✗'}")
            print(f"P2 (Optimize): {'✓' if self.phase2_complete else '✗'}")
            print("-"*50)
        
        print("\nMAZE (█=rock, ·=safe channel, ○=optimal path):")
        for y in range(self.height):
            row = ''
            for x in range(self.width):
                row += display_grid[y][x] + ' '
            print(row)
        
        if self.agent_path:
            print(f"\nOptimal path length: {len(self.agent_path)}")
            print(f"Expected optimal: 21 (from constraint_sim_final.json)")
        
        print("="*50)

def run_demo():
    """Main demonstration function"""
    print("Initializing Deadband Protocol Demo...")
    
    # Create maze with parameters from constraint_sim_final.json
    maze = DeadbandMaze(width=20, height=20, rock_count=64)
    
    print("\n=== PHASE 0: MAP NEGATIVE SPACE ===")
    print("Identifying rocks (failure states)...")
    rocks = maze.generate_rocks()
    maze.display(show_phases=True)
    
    print("\n=== PHASE 1: IDENTIFY SAFE CHANNELS ===")
    print("Finding all paths that avoid rocks...")
    safe_paths = maze.find_safe_channels()
    print(f"Found {len(safe_paths)} safe channel(s)")
    maze.display(show_phases=True)
    
    print("\n=== PHASE 2: OPTIMIZE WITHIN CHANNEL ===")
    print("Selecting shortest path from safe channels...")
    optimal_path = maze.optimize_path(safe_paths)
    
    if optimal_path:
        print(f"Optimal path found with {len(optimal_path)} steps")
    else:
        print("No optimal path found")
    
    maze.display(show_phases=True)
    
    # Show protocol violation example
    print("\n=== PROTOCOL VIOLATION EXAMPLE ===")
    print("Attempting P2 (optimize) without P0+P1...")
    bad_maze = DeadbandMaze(width=5, height=5, rock_count=5)
    bad_maze.generate_rocks()
    
    # Skip to P2 without safe channels
    try:
        bad_maze.optimize_path([])
        print("ERROR: Should have failed!")
    except:
        print("✓ Correctly blocked: Cannot optimize without safe channels")
    
    return maze

if __name__ == "__main__":
    # Run the demo
    final_maze = run_demo()
    
    # Summary
    print("\n" + "="*50)
    print("DEMO SUMMARY")
    print("="*50)
    print("This demonstrates the Deadband Protocol:")
    print("1. P0: Map rocks (64 failure states)")
    print("2. P1: Find safe channels (paths avoiding rocks)")
    print("3. P2: Optimize within channels (shortest safe path)")
    print("\nKey insight from constraint_sim_final.json:")
    print("- Unconstrained optimization: 0% success")
    print("- Deadband protocol: 50/50 success")
    print("- Optimal path length: 21 steps")
    print("="*50)
```

```python
#!/usr/bin/env python3
"""
Deadband Protocol Visual Demo
P0 → P1 → P2 strict priority in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
from collections import deque
import heapq

WIDTH, HEIGHT = 20, 20
OBSTACLE_RATIO = 0.15  # ~15% rocks

def generate_maze():
    """Create random maze with S, G, and obstacles."""
    maze = [['.' for _ in range(WIDTH)] for _ in range(HEIGHT)]
    
    # Place obstacles
    rocks = 0
    for y in range(HEIGHT):
        for x in range(WIDTH):
            if random.random() < OBSTACLE_RATIO:
                maze[y][x] = '#'
                rocks += 1
    
    # Place start (top-left) and goal (bottom-right)
    maze[0][0] = 'S'
    maze[HEIGHT-1][WIDTH-1] = 'G'
    
    # Ensure start and goal aren't blocked
    if maze[0][0] == '#':
        maze[0][0] = 'S'
    if maze[HEIGHT-1][WIDTH-1] == '#':
        maze[HEIGHT-1][WIDTH-1] = 'G'
    
    return maze, rocks

def print_maze(maze, title="Maze"):
    """Print maze with border."""
    print(f"\n=== {title} ===")
    for row in maze:
        print(''.join(row))

def p0_map_negative_space(maze):
    """P0: Identify all obstacle cells (rocks)."""
    rocks = []
    for y in range(HEIGHT):
        for x in range(WIDTH):
            if maze[y][x] == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start_x, start_y):
    """P1: BFS to find all reachable cells from start, avoiding rocks."""
    visited = [[False for _ in range(WIDTH)] for _ in range(HEIGHT)]
    queue = deque([(start_x, start_y)])
    visited[start_y][start_x] = True
    reachable = [(start_x, start_y)]
    
    dirs = [(0,1),(1,0),(0,-1),(-1,0)]
    
    while queue:
        x, y = queue.popleft()
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < WIDTH and 0 <= ny < HEIGHT:
                if not visited[ny][nx] and maze[ny][nx] != '#':
                    visited[ny][nx] = True
                    reachable.append((nx, ny))
                    queue.append((nx, ny))
    
    # Create visualization
    viz = [row[:] for row in maze]
    for x, y in reachable:
        if viz[y][x] == '.':
            viz[y][x] = ' '
    return reachable, viz

def p2_optimize_within_channel(maze, reachable, start_x, start_y, goal_x, goal_y):
    """P2: A* search constrained to reachable cells only."""
    if (goal_x, goal_y) not in reachable:
        return None, None
    
    # Heuristic: Manhattan distance
    def heuristic(x1, y1, x2, y2):
        return abs(x1 - x2) + abs(y1 - y2)
    
    # Convert reachable list to set for O(1) lookup
    reachable_set = set(reachable)
    
    open_set = []
    heapq.heappush(open_set, (0, start_x, start_y))
    
    came_from = {}
    g_score = {(start_x, start_y): 0}
    f_score = {(start_x, start_y): heuristic(start_x, start_y, goal_x, goal_y)}
    
    dirs = [(0,1),(1,0),(0,-1),(-1,0)]
    
    while open_set:
        _, x, y = heapq.heappop(open_set)
        
        if x == goal_x and y == goal_y:
            # Reconstruct path
            path = []
            while (x, y) in came_from:
                path.append((x, y))
                x, y = came_from[(x, y)]
            path.append((start_x, start_y))
            path.reverse()
            return path, len(path) - 1
        
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if (nx, ny) not in reachable_set:
                continue  # Constrained to safe channel
            
            tentative_g = g_score[(x, y)] + 1
            if (nx, ny) not in g_score or tentative_g < g_score[(nx, ny)]:
                came_from[(nx, ny)] = (x, y)
                g_score[(nx, ny)] = tentative_g
                f_score[(nx, ny)] = tentative_g + heuristic(nx, ny, goal_x, goal_y)
                heapq.heappush(open_set, (f_score[(nx, ny)], nx, ny))
    
    return None, None

def visualize_path(maze, path):
    """Overlay path on maze."""
    viz = [row[:] for row in maze]
    for x, y in path[1:-1]:  # Skip start and goal
        if viz[y][x] == '.' or viz[y][x] == ' ':
            viz[y][x] = '*'
    return viz

def main():
    print("Deadband Protocol Demo - 20x20 Maze")
    print("P0 → P1 → P2 strict priority")
    print("=" * 40)
    
    # Generate maze
    maze, rock_count = generate_maze()
    print_maze(maze, "Initial Maze")
    print(f"Rocks: {rock_count}")
    
    # Find start and goal
    start_x, start_y = 0, 0
    goal_x, goal_y = WIDTH-1, HEIGHT-1
    
    # P0: Map negative space
    rocks = p0_map_negative_space(maze)
    print(f"\nP0: Mapped {len(rocks)} rocks (negative space)")
    
    # P1: Find safe channels
    reachable, channel_viz = p1_find_safe_channels(maze, start_x, start_y)
    print_maze(channel_viz, "P1: Safe Channels (reachable cells)")
    print(f"P1: Found {len(reachable)} reachable cells")
    
    # P2: Optimize within channel
    path, path_length = p2_optimize_within_channel(maze, reachable, start_x, start_y, goal_x, goal_y)
    
    if path:
        path_viz = visualize_path(maze, path)
        print_maze(path_viz, "P2: Optimal Path (within safe channel)")
        print(f"P2: Path length = {path_length} steps")
        print("SUCCESS: Goal reached via Deadband Protocol")
    else:
        print_maze(maze, "P2: No Path Found")
        print("FAILURE: Goal unreachable from safe channels")
    
    # Summary
    print("\n" + "=" * 40)
    print("PROTOCOL EXECUTION:")
    print(f"P0 rocks mapped: {len(rocks)}")
    print(f"P1 safe cells: {len(reachable)}")
    print(f"P2 path length: {path_length if path else 'N/A'}")
    print("Strict P0→P1→P2 priority maintained.")

if __name__ == "__main__":
    main()
```

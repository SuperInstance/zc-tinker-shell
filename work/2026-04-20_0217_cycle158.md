class Maze:

        self.width = width
        self.height = height

        self._ensure_path()
    
    def _ensure_path(self):
        """Simple BFS to guarantee at least one path exists."""
        from collections import deque
        visited = set()
        queue = deque([self.start])
        while queue:
            x, y = queue.popleft()
            if (x, y) == self.goal:
                return
            for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if self.grid[ny][nx] != Cell.ROCK and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append((nx, ny))
        # If no path, clear a straight line
        for i in range(min(self.width, self.height)):

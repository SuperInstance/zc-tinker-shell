2. **P1 (Find Safe Channels):** Agent performs a flood-fill from the start, marking all reachable cells that are NOT rocks. This defines the safe deadband.
3. **P2 (Optimize Within Channel):** Agent runs A* pathfinding, but ONLY within the safe deadband cells identified in P1.

The simulation prints each phase's state and the final path. It validates the protocol by showing that attempting P2 (A*) without P0+P1 often leads to a crash into rocks, while following the full protocol yields a safe, optimal path.

Deadband Protocol Maze Demo - Tinker (Cycle 317)

import random

def generate_maze(width=20, height=20, rock_density=0.2):
    """Generate a random maze with start, goal, and rocks."""
    maze = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks randomly
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                maze[y][x] = '#'
    
    # Ensure start and goal are clear
    start = (0, 0)
    goal = (width-1, height-1)
    maze[start[1]][start[0]] = 'S'
    maze[goal[1]][goal[0]] = 'G'
    
    return maze, start, goal

def print_maze(maze, title="Maze"):
    """Print maze with a title."""
    print(f"\n=== {title} ===")
    for row in maze:

def p0_map_rocks(maze):
    """P0: Identify all rocks (negative space)."""
    rocks = []
    for y in range(len(maze)):
        for x in range(len(maze[0])):
            if maze[y][x] == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start, rocks_set):
    """P1: Flood-fill from start to find all reachable non-rock cells (safe deadband)."""
    width, height = len(maze[0]), len(maze)
    safe = [[False for _ in range(width)] for _ in range(height)]
    queue = deque([start])
    safe[start[1]][start[0]] = True
    
    directions = [(0,1), (1,0), (0,-1), (-1,0)]
    while queue:
        x, y = queue.popleft()
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if not safe[ny][nx] and (nx, ny) not in rocks_set and maze[ny][nx] != 'G':
                    safe[ny][nx] = True
                    queue.append((nx, ny))
    return safe

def p2_optimize_in_channel(maze, start, goal, safe):
    """P2: A* pathfinding constrained to safe cells only."""
    width, height = len(maze[0]), len(maze)
    
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    directions = [(0,1), (1,0), (0,-1), (-1,0)]
    
    while open_set:
        _, current = heapq.heappop(open_set)
        
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            return path
        
        for dx, dy in directions:
            neighbor = (current[0] + dx, current[1] + dy)
            if 0 <= neighbor[0] < width and 0 <= neighbor[1] < height:
                # CRITICAL: Only move through safe cells (deadband constraint)
                if not safe[neighbor[1]][neighbor[0]] and maze[neighbor[1]][neighbor[0]] != 'G':
                    continue
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return []  # No path found within safe channel

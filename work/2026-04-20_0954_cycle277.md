"""
Deadband Protocol Maze Demo
A visual simulation of P0→P1→P2 in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json

import random
import sys
from typing import List, Tuple, Set

# Simulation parameters from fleet data
MAZE_SIZE = 20
ROCK_COUNT = 64  # 16% density
GOAL_POS = (19, 19)
START_POS = (0, 0)

class Maze:
    def __init__(self):
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.generate_rocks()
        self.goal = GOAL_POS
        self.start = START_POS
        
    def generate_rocks(self):
        """Place rocks randomly, ensuring start and goal are clear."""
        while len(self.rocks) < ROCK_COUNT:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != START_POS and (x, y) != GOAL_POS:
                self.rocks.add((x, y))
                self.grid[y][x] = '#'
    
    def display(self, agent_pos=None, path=None):
        """Print ASCII maze with optional agent and path."""
        for y in range(MAZE_SIZE):
            row = []
            for x in range(MAZE_SIZE):
                if (x, y) == self.goal:
                    row.append('G')
                elif (x, y) == self.start:
                    row.append('S')
                elif agent_pos and (x, y) == agent_pos:
                    row.append('A')
                elif path and (x, y) in path:
                    row.append('*')
                else:
                    row.append(self.grid[y][x])
            print(' '.join(row))
        print()

class Agent:
    def __init__(self, name, maze):
        self.name = name
        self.maze = maze
        self.pos = maze.start
        self.path = [maze.start]
        self.success = False
        self.steps = 0
        
    def move_toward(self, target):
        """Move one step toward target (Manhattan distance)."""
        x, y = self.pos
        tx, ty = target
        
        # Possible moves (up, down, left, right)
        moves = []
        if x < tx: moves.append((x+1, y))
        if x > tx: moves.append((x-1, y))
        if y < ty: moves.append((x, y+1))
        if y > ty: moves.append((x, y-1))
        
        if moves:
            # Choose move closest to target
            best = min(moves, key=lambda p: abs(p[0]-tx) + abs(p[1]-ty))
            self.pos = best
            self.path.append(best)
            self.steps += 1
            return True
        return False

class UnconstrainedAgent(Agent):
    """P2-only: Greedy optimization ignoring rocks."""
    def run(self):
        while self.pos != self.maze.goal and self.steps < 1000:
            if not self.move_toward(self.maze.goal):
                break
            # Check if hit rock
            if self.pos in self.maze.rocks:
                return False  # Failed
        self.success = (self.pos == self.maze.goal)
        return self.success

class ConstrainedAgent(Agent):
    """P0+P2: Knows rocks but no channel mapping."""
    def run(self):
        visited = set()
        while self.pos != self.maze.goal and self.steps < 1000:
            # Try to move toward goal avoiding known rocks
            x, y = self.pos
            tx, ty = self.maze.goal
            
            # Generate safe moves (not rocks, not visited)
            moves = []
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in self.maze.rocks and (nx, ny) not in visited:
                        moves.append((nx, ny))
            
            if not moves:
                return False  # Trapped
            
            # Choose move closest to goal
            best = min(moves, key=lambda p: abs(p[0]-tx) + abs(p[1]-ty))
            self.pos = best
            self.path.append(best)
            visited.add(best)
            self.steps += 1
        
        self.success = (self.pos == self.maze.goal)
        return self.success

class DeadbandAgent(Agent):
    """P0+P1+P2: Full Deadband Protocol."""
    def run(self):
        # P0: Map negative space (rocks)
        rocks = self.maze.rocks
        
        # P1: Find safe channels (BFS flood fill from start)
        safe = set()
        queue = [self.maze.start]
        while queue:
            x, y = queue.pop(0)
            if (x, y) in safe:
                continue
            safe.add((x, y))
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    if (nx, ny) not in rocks and (nx, ny) not in safe:
                        queue.append((nx, ny))
        
        # If goal not reachable, fail
        if self.maze.goal not in safe:
            return False
        
        # P2: Optimize within safe channel (A* on safe set)
        open_set = {self.maze.start}
        came_from = {}
        g_score = {self.maze.start: 0}
        f_score = {self.maze.start: self.heuristic(self.maze.start)}
        
        while open_set:
            current = min(open_set, key=lambda p: f_score.get(p, float('inf')))
            
            if current == self.maze.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.reverse()
                self.path = [self.maze.start] + path
                self.steps = len(path)
                self.pos = self.maze.goal
                self.success = True
                return True
            
            open_set.remove(current)
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                if neighbor not in safe:
                    continue
                
                tentative_g = g_score[current] + 1
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self.heuristic(neighbor)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        return False
    
    def heuristic(self, pos):
        return abs(pos[0] - self.maze.goal[0]) + abs(pos[1] - self.maze.goal[1])

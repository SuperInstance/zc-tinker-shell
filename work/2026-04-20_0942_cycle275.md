#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
A visual ASCII simulation of P0→P1→P2 in a 20x20 maze.
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
"""

import random
import time
from typing import List, Tuple, Set

class MazeAgent:
    """Agent that navigates a maze using the Deadband Protocol."""
    
    def __init__(self, maze_size: int = 20):
        self.maze_size = maze_size
        self.start = (0, 0)
        self.goal = (maze_size-1, maze_size-1)
        self.rocks: Set[Tuple[int, int]] = set()
        self.safe_channels: List[List[Tuple[int, int]]] = []
        self.optimal_path: List[Tuple[int, int]] = []
        
    def generate_rocks(self, rock_count: int = 64) -> None:
        """P0: Map negative space (what NOT to do)"""
        self.rocks.clear()
        while len(self.rocks) < rock_count:
            x = random.randint(0, self.maze_size-1)
            y = random.randint(0, self.maze_size-1)
            # Don't put rocks on start or goal
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def find_safe_channels(self) -> None:
        """P1: Find safe channels (where you CAN be)"""
        self.safe_channels = []
        
        # Simple BFS to find all safe paths
        queue = [(self.start, [self.start])]
        visited = {self.start}
        
        while queue:
            (x, y), path = queue.pop(0)
            
            # If we reached goal, save this safe path
            if (x, y) == self.goal:
                self.safe_channels.append(path)
                continue
            
            # Check all four directions
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                
                # Check bounds
                if 0 <= nx < self.maze_size and 0 <= ny < self.maze_size:
                    # Check if safe (not a rock)
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
    
    def optimize_within_channel(self) -> None:
        """P2: Optimize within safe channels (best path)"""
        if not self.safe_channels:
            self.optimal_path = []
            return
        
        # Find shortest safe path (optimization constrained to safe channels)
        self.optimal_path = min(self.safe_channels, key=len)
    
    def run_deadband_protocol(self) -> bool:
        """Execute full P0→P1→P2 protocol"""
        # P0: Map negative space
        self.generate_rocks()
        
        # P1: Find safe channels
        self.find_safe_channels()
        
        # P2: Optimize within channel
        self.optimize_within_channel()
        
        return len(self.optimal_path) > 0
    
    def run_greedy_unconstrained(self) -> bool:
        """P2-only (greedy optimization without P0+P1)"""
        # Greedy pathfinding ignoring rocks
        path = [self.start]
        x, y = self.start
        
        while (x, y) != self.goal:
            # Move toward goal (greedy)
            if x < self.goal[0]:
                x += 1
            elif y < self.goal[1]:
                y += 1
            elif x > self.goal[0]:
                x -= 1
            elif y > self.goal[1]:
                y -= 1
            
            path.append((x, y))
            
            # Check if hit rock
            if (x, y) in self.rocks:
                return False
        
        return True
    
    def visualize_maze(self, show_path: bool = False) -> str:
        """Create ASCII visualization of maze"""
        grid = []
        for y in range(self.maze_size):
            row = []
            for x in range(self.maze_size):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')
                elif show_path and pos in self.optimal_path:
                    row.append('·')
                else:
                    row.append(' ')
            grid.append(''.join(row))
        return '\n'.join(grid)

def main():
    """Run the Deadband Protocol demonstration"""
    print("=" * 60)
    print("DEADBAND PROTOCOL MAZE DEMO")
    print("Based on fleet constraint theory simulation")
    print("=" * 60)
    
    # Create agent
    agent = MazeAgent(maze_size=20)
    
    print("\n1. P0: MAPPING NEGATIVE SPACE (Rocks)")
    agent.generate_rocks(rock_count=64)
    print(f"   Generated {len(agent.rocks)} rocks (obstacles)")
    print("\n   Maze with rocks (█):")
    print(agent.visualize_maze())
    
    print("\n2. P1: FINDING SAFE CHANNELS")
    agent.find_safe_channels()
    print(f"   Found {len(agent.safe_channels)} safe channels")
    
    print("\n3. P2: OPTIMIZING WITHIN CHANNEL")
    success = agent.run_deadband_protocol()
    
    if success:
        print(f"   ✓ Found optimal path of length {len(agent.optimal_path)}")
        print("\n   Safe path through maze (·):")
        print(agent.visualize_maze(show_path=True))
        
        # Show protocol steps
        print("\n" + "=" * 60)
        print("PROTOCOL EXECUTION:")
        print(f"  P0: Mapped {len(agent.rocks)} rocks")
        print(f"  P1: Found {len(agent.safe_channels)} safe channels")
        print(f"  P2: Selected path of {len(agent.optimal_path)} steps")
        print(f"  Result: SUCCESS (reached goal in safe channel)")
    else:
        print("   ✗ No safe path found (maze may be impossible)")
    
    # Compare with greedy approach
    print("\n" + "=" * 60)
    print("COMPARISON: Deadband Protocol vs Greedy (P2-only)")
    
    # Run multiple trials
    deadband_success = 0
    greedy_success = 0
    trials = 10
    
    for i in range(trials):
        agent = MazeAgent(maze_size=20)
        agent.generate_rocks(64)
        
        # Deadband protocol
        if agent.run_deadband_protocol():
            deadband_success += 1
        
        # Greedy approach
        if agent.run_greedy_unconstrained():
            greedy_success += 1
    
    print(f"\nResults over {trials} trials:")
    print(f"  Deadband Protocol (P0→P1→P2): {deadband_success}/{trials} successes")
    print(f"  Greedy (P2-only): {greedy_success}/{trials} successes")
    
    # Show statistics from fleet data
    print("\n" + "=" * 60)
    print("FLEET SIMULATION DATA (from constraint_sim_final.json):")
    print("  Optimal path length: 21")
    print("  Rocks: 64")
    print("  Unconstrained (greedy) success: 33%")
    print("  Constrained (no P0) success: 0%")
    print("  Deadband Protocol success: 50%")
    
    print("\n" + "=" * 60)
    print("KEY INSIGHT:")
    print("  P2-only optimization hits rocks with high probability.")
    print("  Deadband Protocol (P0→P1→P2) finds safe channels first,")
    print("  then optimizes within them. This matches fleet doctrine.")

if __name__ == "__main__":
    main()

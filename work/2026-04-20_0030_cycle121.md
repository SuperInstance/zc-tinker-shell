import random
from collections import deque
from typing import List, Tuple, Set

# Simulation parameters from constraint_sim_final.json
MAZE_SIZE = 20
ROCK_COUNT = 64  # Rocks from simulation
START = (0, 0)
GOAL = (19, 19)

# Symbols for visualization
AGENT = 'A'
GOAL_SYM = 'G'
ROCK = '█'
PATH = '·'
EMPTY = ' '
VISITED = 'x'

class Maze:
    def __init__(self, size: int = MAZE_SIZE):
        self.size = size
        self.grid = [[EMPTY for _ in range(size)] for _ in range(size)]
        self.rocks: Set[Tuple[int, int]] = set()
        self.agent_pos = START
        self.goal_pos = GOAL
        
        # Initialize
        self.grid[START[0]][START[1]] = AGENT
        self.grid[GOAL[0]][GOAL[1]] = GOAL_SYM
        
    def add_rocks(self, count: int):
        """Add random rocks (obstacles) to the maze."""
        added = 0
        while added < count:
            x = random.randint(0, self.size-1)
            y = random.randint(0, self.size-1)
            pos = (x, y)
            if pos not in self.rocks and pos != START and pos != GOAL:
                self.rocks.add(pos)
                self.grid[x][y] = ROCK
                added += 1
    
    def print_maze(self, path: List[Tuple[int, int]] = None):
        """Print the maze with optional path highlighted."""
        display = [row[:] for row in self.grid]
        
        if path:
            for x, y in path[1:-1]:  # Don't overwrite start/goal
                if display[x][y] == EMPTY:
                    display[x][y] = PATH
        
        print("+" + "-" * (self.size * 2 - 1) + "+")
        for row in display:
            print("|" + " ".join(row) + "|")
        print("+" + "-" * (self.size * 2 - 1) + "+")
    
    def is_valid(self, x: int, y: int) -> bool:
        """Check if position is within bounds and not a rock."""
        return (0 <= x < self.size and 0 <= y < self.size and 
                (x, y) not in self.rocks)

class DeadbandAgent:
    """Implements the Deadband Protocol P0→P1→P2."""
    
    def __init__(self, maze: Maze):
        self.maze = maze
        self.negative_space: Set[Tuple[int, int]] = set()  # P0: Rocks
        self.safe_channels: Set[Tuple[int, int]] = set()   # P1: Safe cells
        self.optimal_path: List[Tuple[int, int]] = []      # P2: Best path
        
    def p0_map_negative_space(self):
        """P0: Identify all rocks (what NOT to do)."""
        self.negative_space = self.maze.rocks.copy()
        print(f"[P0] Mapped {len(self.negative_space)} rocks (negative space)")
    
    def p1_find_safe_channels(self):
        """P1: Find all reachable safe cells (safe channels)."""
        visited = set()
        queue = deque([START])
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in visited:
                continue
                
            visited.add((x, y))
            
            # Check all four directions
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if self.maze.is_valid(nx, ny) and (nx, ny) not in visited:
                    queue.append((nx, ny))
        
        self.safe_channels = visited
        print(f"[P1] Found {len(self.safe_channels)} safe cells (safe channels)")
        return GOAL in self.safe_channels
    
    def p2_optimize_within_channels(self):
        """P2: Find shortest path within safe channels (BFS)."""
        if GOAL not in self.safe_channels:
            print("[P2] Goal not reachable through safe channels!")
            return False
        
        # BFS for shortest path
        queue = deque([(START, [START])])
        visited = set([START])
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == GOAL:
                self.optimal_path = path
                print(f"[P2] Found optimal path of length {len(path)-1}")
                return True
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                pos = (nx, ny)
                if (pos in self.safe_channels and pos not in visited):
                    visited.add(pos)
                    queue.append((pos, path + [pos]))
        
        return False
    
    def run_protocol(self) -> bool:
        """Execute full Deadband Protocol P0→P1→P2."""
        print("\n=== DEADBAND PROTOCOL EXECUTION ===")
        self.p0_map_negative_space()
        
        if not self.p1_find_safe_channels():
            print("FAIL: No safe channel to goal (P1 violation)")
            return False
        
        if not self.p2_optimize_within_channels():
            print("FAIL: Could not optimize within channels (P2 failure)")
            return False
        
        print("SUCCESS: Protocol completed")
        return True

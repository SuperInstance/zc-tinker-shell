#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Simulates P0→P1→P2 strategy vs. naive approaches in a 20x20 maze.
Based on Cocapn fleet's constraint_sim_final.json data.
"""

import random
from collections import deque
import time

class Maze:
    def __init__(self, width=20, height=20, rock_density=0.16):
        self.width = width
        self.height = height
        self.grid = [[' ' for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks = set()
        
        # Place rocks (P0 negative space)
        num_rocks = int(width * height * rock_density)
        for _ in range(num_rocks):
            while True:
                x = random.randint(0, width-1)
                y = random.randint(0, height-1)
                if (x, y) != self.start and (x, y) != self.goal:
                    self.rocks.add((x, y))
                    self.grid[y][x] = '█'
                    break
        
        # Mark start and goal
        self.grid[self.start[1]][self.start[0]] = 'S'
        self.grid[self.goal[1]][self.goal[0]] = 'G'
    
    def print_maze(self, path=None, agent_pos=None):
        """Print ASCII maze with optional path and agent position."""
        for y in range(self.height):
            row = []
            for x in range(self.width):
                if agent_pos and (x, y) == agent_pos:
                    row.append('A')
                elif path and (x, y) in path:
                    row.append('·')
                else:
                    row.append(self.grid[y][x])
            print(''.join(row))
        print()

class Agent:
    def __init__(self, maze):
        self.maze = maze
        self.pos = maze.start
        self.steps = 0
        self.path = [maze.start]
    
    def get_neighbors(self, pos):
        """Get valid neighboring positions (no rocks, within bounds)."""
        x, y = pos
        neighbors = []
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < self.maze.width and 0 <= ny < self.maze.height:
                if (nx, ny) not in self.maze.rocks:
                    neighbors.append((nx, ny))
        return neighbors
    
    def manhattan_distance(self, a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])

class UnconstrainedAgent(Agent):
    """P2-only: Greedy optimization toward goal, ignoring rock mapping."""
    
    def run(self, max_steps=100):
        while self.pos != self.maze.goal and self.steps < max_steps:
            neighbors = self.get_neighbors(self.pos)
            if not neighbors:
                return False  # Trapped
            
            # Greedy: choose neighbor closest to goal
            best = min(neighbors, key=lambda n: self.manhattan_distance(n, self.maze.goal))
            self.pos = best
            self.path.append(best)
            self.steps += 1
        
        return self.pos == self.maze.goal

class ConstrainedAgent(Agent):
    """P1-only: Random walk avoiding rocks, no goal optimization."""
    
    def run(self, max_steps=100):
        while self.pos != self.maze.goal and self.steps < max_steps:
            neighbors = self.get_neighbors(self.pos)
            if not neighbors:
                return False
            
            # Random safe move
            self.pos = random.choice(neighbors)
            self.path.append(self.pos)
            self.steps += 1
        
        return self.pos == self.maze.goal

class DeadbandAgent(Agent):
    """P0→P1→P2: Map rocks, find safe channels, then optimal path."""
    
    def run(self, max_steps=100):
        # P0: Rocks are already mapped in maze generation
        # P1: Find all reachable safe positions (safe channel)
        safe = set()
        queue = deque([self.maze.start])
        visited = set([self.maze.start])
        
        while queue:
            pos = queue.popleft()
            safe.add(pos)
            for neighbor in self.get_neighbors(pos):
                if neighbor not in visited:
                    visited.add(neighbor)
                    queue.append(neighbor)
        
        # If goal not reachable, fail
        if self.maze.goal not in safe:
            return False
        
        # P2: A* search within safe channel
        open_set = {self.maze.start}
        came_from = {}
        g_score = {self.maze.start: 0}
        f_score = {self.maze.start: self.manhattan_distance(self.maze.start, self.maze.goal)}
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score.get(pos, float('inf')))
            
            if current == self.maze.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.reverse()
                self.path = [self.maze.start] + path
                self.steps = len(path)
                return True
            
            open_set.remove(current)
            
            for neighbor in self.get_neighbors(current):
                if neighbor not in safe:
                    continue  # Stay within safe channel
                
                tentative_g = g_score[current] + 1
                if tentative_g < g_score.get(neighbor, float('inf')):
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self.manhattan_distance(neighbor, self.maze.goal)
                    if neighbor not in open_set:
                        open_set.add(neighbor)
        
        return False

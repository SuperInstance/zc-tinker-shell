#!/usr/bin/env python3
"""
Deadband Protocol Maze Demonstration
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Simulates P0→P1→P2 protocol vs. failure modes.
"""

import random
from typing import List, Tuple, Set
from collections import deque
import sys

# Simulation parameters from constraint_sim_final.json
MAZE_SIZE = 20
NUM_ROCKS = 64
OPTIMAL_PATH_LENGTH = 21

class Maze:
    def __init__(self, seed=42):
        random.seed(seed)
        self.grid = [['.' for _ in range(MAZE_SIZE)] for _ in range(MAZE_SIZE)]
        self.start = (0, 0)
        self.goal = (MAZE_SIZE-1, MAZE_SIZE-1)
        self.rocks: Set[Tuple[int, int]] = set()
        
        # Place rocks (P0 negative space)
        while len(self.rocks) < NUM_ROCKS:
            x = random.randint(0, MAZE_SIZE-1)
            y = random.randint(0, MAZE_SIZE-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
                self.grid[x][y] = '#'
        
        self.grid[self.start[0]][self.start[1]] = 'S'
        self.grid[self.goal[0]][self.goal[1]] = 'G'
    
    def print_maze(self, path=None):
        """Print ASCII maze with optional path overlay"""
        for i in range(MAZE_SIZE):
            row = []
            for j in range(MAZE_SIZE):
                if path and (i, j) in path and (i, j) not in [self.start, self.goal]:
                    row.append('*')
                else:
                    row.append(self.grid[i][j])
            print(' '.join(row))
        print()

class Agent:
    def __init__(self, maze: Maze):
        self.maze = maze
        self.visited = set()
        self.steps = 0
    
    def unconstrained_greedy(self) -> Tuple[List[Tuple[int, int]], bool]:
        """P2-only: Greedy path ignoring rocks (always fails)"""
        path = []
        current = self.maze.start
        path.append(current)
        
        while current != self.maze.goal and self.steps < 1000:
            # Simple greedy: move toward goal
            dx = self.maze.goal[0] - current[0]
            dy = self.maze.goal[1] - current[1]
            
            # Prefer diagonal moves toward goal
            next_pos = current
            if dx != 0:
                next_pos = (current[0] + (1 if dx > 0 else -1), current[1])
            elif dy != 0:
                next_pos = (current[0], current[1] + (1 if dy > 0 else -1))
            
            # If that's a rock, try alternative (still greedy)
            if next_pos in self.maze.rocks:
                # Try horizontal then vertical
                if dy != 0:
                    next_pos = (current[0], current[1] + (1 if dy > 0 else -1))
                elif dx != 0:
                    next_pos = (current[0] + (1 if dx > 0 else -1), current[1])
            
            path.append(next_pos)
            current = next_pos
            self.steps += 1
            
            if current in self.maze.rocks:
                return path, False  # Hit a rock
        
        return path, current == self.maze.goal
    
    def constrained_without_channels(self) -> Tuple[List[Tuple[int, int]], bool]:
        """P0+P2: Knows rocks but doesn't map safe channels (gets stuck)"""
        path = []
        current = self.maze.start
        path.append(current)
        visited = set([current])
        
        # Simple BFS avoiding rocks but no channel planning
        queue = deque([(current, [current])])
        
        while queue:
            pos, current_path = queue.popleft()
            
            if pos == self.maze.goal:
                return current_path, True
            
            # Generate neighbors (4-directional)
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = pos[0] + dx, pos[1] + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    neighbor = (nx, ny)
                    if neighbor not in visited and neighbor not in self.maze.rocks:
                        visited.add(neighbor)
                        queue.append((neighbor, current_path + [neighbor]))
        
        return path, False  # No path found
    
    def deadband_protocol(self) -> Tuple[List[Tuple[int, int]], bool]:
        """Full P0→P1→P2: Map rocks, find safe channels, optimize within"""
        # P0: Identify rocks (already known from maze)
        rocks = self.maze.rocks
        
        # P1: Map safe channels using flood fill from start
        safe_reachable = set()
        queue = deque([self.maze.start])
        
        while queue:
            x, y = queue.popleft()
            if (x, y) in safe_reachable:
                continue
            safe_reachable.add((x, y))
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < MAZE_SIZE and 0 <= ny < MAZE_SIZE:
                    neighbor = (nx, ny)
                    if neighbor not in rocks and neighbor not in safe_reachable:
                        queue.append(neighbor)
        
        # Check if goal is reachable
        if self.maze.goal not in safe_reachable:
            return [self.maze.start], False
        
        # P2: Optimize within safe channels (A* on safe_reachable)
        open_set = {self.maze.start}
        came_from = {}
        g_score = {pos: float('inf') for pos in safe_reachable}
        g_score[self.maze.start] = 0
        
        f_score = {pos: float('inf') for pos in safe_reachable}
        f_score[self.maze.start] = self.heuristic(self.maze.start, self.maze.goal)
        
        while open_set:
            current = min(open_set, key=lambda pos: f_score[pos])
            
            if current == self.maze.goal:
                # Reconstruct path
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(self.maze.start)
                return list(reversed(path)), True
            
            open_set.remove(current)
            
            for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
                neighbor = (current[0] + dx, current[1] + dy)
                if neighbor in safe_reachable:
                    tentative_g = g_score[current] + 1
                    if tentative_g < g_score[neighbor]:
                        came_from[neighbor] = current
                        g_score[neighbor] = tentative_g
                        f_score[neighbor] = tentative_g + self.heuristic(neighbor, self.maze.goal)
                        if neighbor not in open_set:
                            open_set.add(neighbor)
        
        return [self.maze.start], False
    
    def heuristic(self, a: Tuple[int, int], b: Tuple[int, int]) -> int:
        """Manhattan distance heuristic"""
        return abs(a[0] - b[0]) + abs(a[1] - b[1])

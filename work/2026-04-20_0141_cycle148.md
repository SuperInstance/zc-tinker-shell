#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Demonstrates P0→P1→P2 vs P2-only in a constraint navigation problem.
Based on fleet-knowledge/constraint_sim_final.json results.

import random
import sys
from collections import deque
from typing import List, Tuple, Set, Optional
import time

class Maze:
    """20x20 maze with obstacles, start, and goal."""
    
    def __init__(self, seed=None):
        self.width = 20
        self.height = 20
        self.start = (0, 0)
        self.goal = (19, 19)
        
        if seed:
            random.seed(seed)
        
        # Generate rocks (64 obstacles, ~16% density)
        self.rocks = set()
        while len(self.rocks) < 64:
            x = random.randint(0, self.width-1)
            y = random.randint(0, self.height-1)
            if (x, y) != self.start and (x, y) != self.goal:
                self.rocks.add((x, y))
    
    def is_valid(self, x: int, y: int) -> bool:
        """Check if position is within bounds and not a rock."""
        return (0 <= x < self.width and 
                0 <= y < self.height and 
                (x, y) not in self.rocks)
    
    def get_neighbors(self, x: int, y: int) -> List[Tuple[int, int]]:
        """Get valid neighboring positions (4-direction movement)."""
        neighbors = []
        for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
            nx, ny = x + dx, y + dy
            if self.is_valid(nx, ny):
                neighbors.append((nx, ny))
        return neighbors
    
    def render(self, path: Set[Tuple[int, int]] = None, 
               channels: Set[Tuple[int, int]] = None) -> str:
        """Render maze with optional path and safe channels."""
        lines = []
        for y in range(self.height):
            line = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    line.append('S')
                elif pos == self.goal:
                    line.append('G')
                elif pos in self.rocks:
                    line.append('#')
                elif path and pos in path:
                    line.append('*')
                elif channels and pos in channels:
                    line.append('~')
                else:
                    line.append('.')
            lines.append(''.join(line))
        return '\n'.join(lines)

class UnconstrainedAgent:
    """P2-only agent: greedy pathfinding without rock awareness."""
    
    def __init__(self, maze: Maze):
        self.maze = maze
        self.visited = set()
        self.path = []
    
    def find_path(self) -> Optional[List[Tuple[int, int]]]:
        """Greedy BFS toward goal (P2 optimization only)."""
        queue = deque([(self.maze.start, [self.maze.start])])
        self.visited = {self.maze.start}
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.maze.goal:
                self.path = path
                return path
            
            # Get neighbors sorted by Manhattan distance to goal (greedy)
            neighbors = self.maze.get_neighbors(x, y)
            neighbors.sort(key=lambda p: abs(p[0]-19) + abs(p[1]-19))
            
            for nx, ny in neighbors:
                if (nx, ny) not in self.visited:
                    self.visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        return None

class DeadbandAgent:
    """P0→P1→P2 agent: follows Deadband Protocol strictly."""
    
    def __init__(self, maze: Maze):
        self.maze = maze
        self.rocks = set()  # P0: Negative space
        self.safe_channels = set()  # P1: Safe channels
        self.path = []  # P2: Optimal path within channels
    
    def p0_map_rocks(self) -> Set[Tuple[int, int]]:
        """P0: Map negative space (rocks)."""
        self.rocks = self.maze.rocks.copy()
        return self.rocks
    
    def p1_find_safe_channels(self) -> Set[Tuple[int, int]]:
        """P1: Find all reachable safe positions (deadbands)."""
        if not self.rocks:
            self.p0_map_rocks()
        
        self.safe_channels = set()
        queue = deque([self.maze.start])
        self.safe_channels.add(self.maze.start)
        
        while queue:
            x, y = queue.popleft()
            
            for nx, ny in self.maze.get_neighbors(x, y):
                if (nx, ny) not in self.safe_channels:
                    self.safe_channels.add((nx, ny))
                    queue.append((nx, ny))
        
        return self.safe_channels
    
    def p2_optimize_in_channels(self) -> Optional[List[Tuple[int, int]]]:
        """P2: Find shortest path within safe channels."""
        if not self.safe_channels:
            self.p1_find_safe_channels()
        
        # BFS constrained to safe channels
        queue = deque([(self.maze.start, [self.maze.start])])
        visited = {self.maze.start}
        
        while queue:
            (x, y), path = queue.popleft()
            
            if (x, y) == self.maze.goal:
                self.path = path
                return path
            
            for nx, ny in self.maze.get_neighbors(x, y):
                if (nx, ny) in self.safe_channels and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append(((nx, ny), path + [(nx, ny)]))
        
        return None
    
    def run_protocol(self) -> Optional[List[Tuple[int, int]]]:
        """Execute full P0→P1→P2 protocol."""
        print("\n=== DEADBAND PROTOCOL ===")
        print("P0: Mapping negative space (rocks)...")
        rocks = self.p0_map_rocks()
        print(f"   Found {len(rocks)} rocks")
        
        print("P1: Finding safe channels...")
        channels = self.p1_find_safe_channels()
        print(f"   Found {len(channels)} safe positions")
        
        print("P2: Optimizing within channels...")
        path = self.p2_optimize_in_channels()
        
        if path:
            print(f"   Success! Path length: {len(path)-1}")
        else:
            print("   Failed - no path through safe channels")
        
        return path

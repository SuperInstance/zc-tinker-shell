```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
P0: Map rocks (obstacles)
P1: Find safe channels (deadbands)
P2: Optimize path within safe channels
"""

import random
from typing import List, Tuple, Set

class DeadbandMaze:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        
        # Generate rocks (P0 negative space)
        self.rocks: Set[Tuple[int, int]] = set()
        total_cells = width * height
        num_rocks = int(total_cells * rock_density)
        
        # Avoid start and goal
        possible = [(x, y) for x in range(width) for y in range(height)]
        possible.remove(self.start)
        possible.remove(self.goal)
        
        self.rocks = set(random.sample(possible, num_rocks))
        
        # P1: Safe channels (non-rock cells)
        self.safe_cells = [(x, y) for x in range(width) for y in range(height) 
                          if (x, y) not in self.rocks]
        
        # For visualization
        self.path = []
        self.visited = set()
    
    def render(self, show_path: bool = False) -> str:
        """Render maze with ASCII art"""
        grid = []
        for y in range(self.height):
            row = []
            for x in range(self.width):
                pos = (x, y)
                if pos == self.start:
                    row.append('S')
                elif pos == self.goal:
                    row.append('G')
                elif pos in self.rocks:
                    row.append('█')  # Rock
                elif show_path and pos in self.path:
                    row.append('·')  # Path
                elif show_path and pos in self.visited:
                    row.append('.')  # Visited
                else:
                    row.append(' ')  # Empty
            grid.append(''.join(row))
        return '\n'.join(grid)
    
    def p0_map_rocks(self) -> Set[Tuple[int, int]]:
        """P0: Identify negative space (what NOT to do)"""
        print("=== P0: MAP NEGATIVE SPACE ===")
        print(f"Identified {len(self.rocks)} rocks (obstacles)")
        print("Rocks are positions where agent would fail.")
        return self.rocks
    
    def p1_find_safe_channels(self) -> List[Tuple[int, int]]:
        """P1: Find safe channels (where you CAN be)"""
        print("\n=== P1: IDENTIFY SAFE CHANNELS ===")
        print(f"Found {len(self.safe_cells)} safe cells (deadband)")
        print("Agent may ONLY operate within these cells.")
        return self.safe_cells
    
    def p2_optimize_in_channel(self) -> List[Tuple[int, int]]:
        """P2: Optimize within safe channels (best path)"""
        print("\n=== P2: OPTIMIZE WITHIN CHANNEL ===")
        print("Finding shortest path through safe water...")
        
        # Simple BFS through safe cells only
        queue = [(self.start, [self.start])]
        self.visited = {self.start}
        
        while queue:
            (x, y), path = queue.pop(0)
            
            if (x, y) == self.goal:
                self.path = path
                print(f"Found path with {len(path)} steps")
                return path
            
            # Check neighbors (only orthogonal moves)
            for dx, dy in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nx, ny = x + dx, y + dy
                neighbor = (nx, ny)
                
                if (0 <= nx < self.width and 0 <= ny < self.height and
                    neighbor not in self.rocks and
                    neighbor not in self.visited):
                    
                    self.visited.add(neighbor)
                    queue.append((neighbor, path + [neighbor]))
        
        print("No path found! (All channels blocked)")
        return []
    
    def run_protocol(self) -> bool:
        """Execute full Deadband Protocol P0→P1→P2"""
        print("DEADBAND PROTOCOL DEMONSTRATION")
        print("=" * 40)
        
        # P0: Map negative space
        rocks = self.p0_map_rocks()
        print("\nInitial Maze (S=start, G=goal, █=rock):")
        print(self.render())
        
        # P1: Find safe channels
        safe_cells = self.p1_find_safe_channels()
        
        # P2: Optimize within channels
        path = self.p2_optimize_in_channel()
        
        if path:
            print("\nFinal Path Found (S=start, G=goal, █=rock, ·=path):")
            print(self.render(show_path=True))
            
            # Statistics
            optimal_steps = abs(self.goal[0] - self.start[0]) + abs(self.goal[1] - self.start[1])
            actual_steps = len(path) - 1  # Exclude start
            
            print(f"\nSTATISTICS:")
            print(f"  Maze size: {self.width}x{self.height}")
            print(f"  Rocks: {len(self.rocks)} ({len(self.rocks)/(self.width*self.height):.1%})")
            print(f"  Safe cells: {len(safe_cells)}")
            print(f"  Optimal (no rocks): {optimal_steps} steps")
            print(f"  Actual (with rocks): {actual_steps} steps")
            print(f"  Efficiency: {optimal_steps/actual_steps:.1%}")
            
            return True
        else:
            print("\nPROTOCOL FAILED: No safe path exists")
            return False
    
    def run_greedy_violation(self) -> bool:
        """Demonstrate what happens when violating protocol (P2 before P0+P1)"""
        print("\n" + "=" * 40)
        print("GREEDY VIOLATION (P2 before P0+P1)")
        print("=" * 40)
        
        # Greedy approach: optimize without mapping rocks first
        print("Skipping P0+P1, going straight to P2 optimization...")
        
        # Try to find path ignoring rocks (will hit them)
        path = []
        x, y = self.start
        self.visited = {self.start}
        
        # Simple greedy: move toward goal
        while (x, y) != self.goal:
            # Check if we're at a rock
            if (x, y) in self.rocks:
                print(f"  HIT ROCK at ({x}, {y})!")
                return False
            
            path.append((x, y))
            
            # Move toward goal
            if x < self.goal[0]:
                x += 1
            elif x > self.goal[0]:
                x -= 1
            elif y < self.goal[1]:
                y += 1
            elif y > self.goal[1]:
                y -= 1
            
            self.visited.add((x, y))
        
        path.append(self.goal)
        self.path = path
        
        print("  Greedy path found (but likely hit rocks in reality)")
        print("\nGreedy 'Path' (would fail in reality):")
        print(self.render(show_path=True))
        
        return True


def main():
    """Run demonstration"""
    print("Deadband Protocol Maze Demonstration")
    print("Based on fleet-knowledge/constraint_sim_final.json")
    print(f"Optimal: 21 steps, Rocks: 64, Deadband success: 50/50")
    print("-" * 50)
    
    # Create maze similar to simulation
    maze = DeadbandMaze(width=20, height=20, rock_density=0.16)
    
    # Run proper protocol
    success = maze.run_protocol()
    
    # Show violation (for comparison)
    if success:
        maze.run_greedy_violation()
    
    print("\n" + "=" * 50)
    print("LESSON: Always follow P0→P1→P2 priority order.")
    print("Greedy optimization (P2 first) hits rocks.")
    print("Deadband protocol finds safe channels first.")


if __name__ == "__main__":
    main()
```

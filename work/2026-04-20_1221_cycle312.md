**Task:** Complete the Deadband Protocol visual demo (ASCII maze simulation).

**Status:** I have written a Python script that generates a random 20x20 maze and visually demonstrates the strict P0→P1→P2 priority order of the Deadband Protocol. The script simulates an agent navigating from start (S) to goal (G) while avoiding rocks (#). It shows three distinct phases:

1. **P0 (Map Negative Space):** Agent identifies all rock locations (failure states).
2. **P1 (Find Safe Channels):** Agent performs flood-fill from start, marking all reachable cells that are NOT rocks.
3. **P2 (Optimize Within Channel):** Agent runs A* pathfinding, but ONLY within the safe channel identified in P1.

The simulation prints each phase's output as ASCII grids, and finally the successful path (if one exists within the safe channel). The code is below.

**Output (Python Script):**

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2 Strict Priority
Simulates the fleet's Deadband Protocol in a 20x20 ASCII maze.
"""

import random
from collections import deque
import heapq

def generate_maze(width=20, height=20, rock_density=0.2):
    """Generate a random maze with start, goal, and rocks."""
    grid = [['.' for _ in range(width)] for _ in range(height)]
    
    # Place rocks randomly
    for y in range(height):
        for x in range(width):
            if random.random() < rock_density:
                grid[y][x] = '#'
    
    # Ensure start and goal are open
    start = (0, 0)
    goal = (width-1, height-1)
    grid[start[1]][start[0]] = 'S'
    grid[goal[1]][goal[0]] = 'G'
    
    return grid, start, goal

def print_grid(grid, title="Grid"):
    """Print ASCII grid with title."""
    print(f"\n=== {title} ===")
    for row in grid:
        print(' '.join(row))
    print()

def p0_map_negative_space(grid):
    """P0: Identify all rocks (failure states)."""
    rocks = []
    for y, row in enumerate(grid):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(grid, start, rocks_set):
    """P1: Flood-fill from start, marking reachable non‑rock cells."""
    width, height = len(grid[0]), len(grid)
    safe = [[False for _ in range(width)] for _ in range(height)]
    
    # BFS from start
    queue = deque([start])
    sx, sy = start
    safe[sy][sx] = True
    
    while queue:
        x, y = queue.popleft()
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if not safe[ny][nx] and (nx, ny) not in rocks_set:
                    safe[ny][nx] = True
                    queue.append((nx, ny))
    
    # Create visualization
    viz = [row[:] for row in grid]
    for y in range(height):
        for x in range(width):
            if safe[y][x] and viz[y][x] == '.':
                viz[y][x] = ' '
    return safe, viz

def p2_optimize_within_channel(grid, start, goal, safe):
    """P2: A* pathfinding constrained to safe cells only."""
    width, height = len(grid[0]), len(grid)
    
    # Heuristic: Manhattan distance
    def heuristic(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])
    
    # A* search
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        if current == goal:
            # Reconstruct path
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            return path
        
        cx, cy = current
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            nx, ny = cx + dx, cy + dy
            neighbor = (nx, ny)
            if 0 <= nx < width and 0 <= ny < height:
                if not safe[ny][nx]:
                    continue  # NOT in safe channel → skip
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return []  # No path within safe channel

def main():
    random.seed(42)  # Deterministic for demo
    grid, start, goal = generate_maze(20, 20, rock_density=0.2)
    
    print("DEADBAND PROTOCOL DEMO — MAZE NAVIGATION")
    print("=" * 50)
    print_grid(grid, "Initial Maze (S=start, G=goal, #=rock)")
    
    # P0: Map negative space
    rocks = p0_map_negative_space(grid)
    rocks_set = set(rocks)
    print(f"P0 — Rocks identified: {len(rocks)} failure states")
    
    # P1: Find safe channels
    safe, safe_viz = p1_find_safe_channels(grid, start, rocks_set)
    print_grid(safe_viz, "P1 — Safe Channels (space=reachable non‑rock)")
    
    # P2: Optimize within channel
    path = p2_optimize_within_channel(grid, start, goal, safe)
    
    # Result visualization
    result = [row[:] for row in grid]
    for (x, y) in path:
        if result[y][x] == '.':
            result[y][x] = '*'
    print_grid(result, f"P2 — Optimal Path (length={len(path)})")
    
    if path:
        print("SUCCESS: Path found within safe channel (P0→P1→P2 respected).")
    else:
        print("FAILURE: No path exists within safe channel (goal unreachable).")

if __name__ == "__main__":
    main()
```

**Key Points Demonstrated:**
- The agent **never** considers moving into a rock cell (P0 violation).
- Pathfinding is **constrained** to the safe channel identified in P1.
- Optimization (P2) only occurs after P0 and P1 are satisfied.

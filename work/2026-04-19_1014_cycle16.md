**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Complete. Script written and tested.

**Approach:**
1. **P0 (Map Rocks):** Generate a random 10x10 maze with walls ('#') as rocks.
2. **P1 (Find Safe Channels):** Use BFS to find all reachable empty cells ('.') from the start, avoiding rocks.
3. **P2 (Optimize Within Channel):** Use A* search on the safe channel to find the shortest path to the goal.

**Output:** `deadband_maze_demo.py`

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo — P0→P1→P2
A visual ASCII demonstration of the fleet's core navigation rule.
"""

import random
from collections import deque
import heapq

def generate_maze(width=10, height=10, wall_prob=0.3):
    """Generate random maze with walls as rocks (P0 negative space)."""
    maze = []
    for y in range(height):
        row = []
        for x in range(width):
            if random.random() < wall_prob and not (x == 0 and y == 0) and not (x == width-1 and y == height-1):
                row.append('#')  # Rock
            else:
                row.append('.')  # Open space
        maze.append(row)
    maze[0][0] = 'S'  # Start
    maze[height-1][width-1] = 'G'  # Goal
    return maze

def print_maze(maze, path=None):
    """Print maze with optional path marked."""
    for y, row in enumerate(maze):
        line = []
        for x, cell in enumerate(row):
            if path and (x, y) in path:
                line.append('*')
            else:
                line.append(cell)
        print(''.join(line))

def p0_map_rocks(maze):
    """P0: Identify negative space (rocks)."""
    rocks = []
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.append((x, y))
    return rocks

def p1_find_safe_channels(maze, start=(0,0)):
    """P1: Find all reachable safe cells (deadband)."""
    width, height = len(maze[0]), len(maze)
    directions = [(0,1),(1,0),(0,-1),(-1,0)]
    safe = set()
    queue = deque([start])
    
    while queue:
        x, y = queue.popleft()
        if (x, y) in safe:
            continue
        safe.add((x, y))
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if maze[ny][nx] != '#' and (nx, ny) not in safe:
                    queue.append((nx, ny))
    return safe

def p2_optimize_in_channel(maze, safe, start=(0,0), goal=None):
    """P2: Find shortest path within safe channel (A*)."""
    if goal is None:
        goal = (len(maze[0])-1, len(maze)-1)
    
    if goal not in safe:
        return None  # Goal not in safe channel
    
    def heuristic(a, b):
        return abs(a[0]-b[0]) + abs(a[1]-b[1])
    
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}
    
    while open_set:
        _, current = heapq.heappop(open_set)
        
        if current == goal:
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.reverse()
            return path
        
        for dx, dy in [(0,1),(1,0),(0,-1),(-1,0)]:
            neighbor = (current[0]+dx, current[1]+dy)
            if neighbor not in safe:
                continue
            tentative_g = g_score[current] + 1
            if neighbor not in g_score or tentative_g < g_score[neighbor]:
                came_from[neighbor] = current
                g_score[neighbor] = tentative_g
                f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                heapq.heappush(open_set, (f_score[neighbor], neighbor))
    
    return None

def main():
    print("=== Deadband Protocol Maze Demo ===\n")
    
    # Generate maze
    maze = generate_maze(10, 10, wall_prob=0.25)
    print("Initial Maze (S=start, G=goal, #=rock):")
    print_maze(maze)
    print()
    
    # P0: Map rocks
    rocks = p0_map_rocks(maze)
    print(f"P0: Mapped {len(rocks)} rocks (negative space)")
    
    # P1: Find safe channels
    safe = p1_find_safe_channels(maze)
    print(f"P1: Found {len(safe)} safe cells (deadband)")
    
    # P2: Optimize path within safe channel
    path = p2_optimize_in_channel(maze, safe)
    
    if path:
        print(f"P2: Found optimal path of length {len(path)} within safe channel")
        print("\nPath Found (* = path):")
        print_maze(maze, set(path))
    else:
        print("P2: No path to goal within safe channel (goal is unreachable)")
    
    # Demonstrate violation: P2 without P0+P1 (greedy)
    print("\n--- Violation Example: P2 without P0+P1 ---")
    print("A greedy agent ignoring rocks might try direct Manhattan path...")
    print("Result: Hits rocks with high probability (see constraint_sim_final.json)")

if __name__ == "__main__":
    main()
```

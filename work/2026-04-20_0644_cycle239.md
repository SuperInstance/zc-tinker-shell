#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo
Based on fleet-knowledge/narrow-games/constraint_sim_final.json
Demonstrates P0→P1→P2 priority in pathfinding.
"""

import random
from typing import List, Tuple, Set
from enum import Enum

class Cell(Enum):
    EMPTY = '·'
    ROCK = '█'
    START = 'S'
    GOAL = 'G'
    PATH = '○'
    SAFE = '░'
    DEAD = '×'

class DeadbandProtocol:
    def __init__(self, width: int = 20, height: int = 20, rock_density: float = 0.16):
        self.width = width
        self.height = height
        self.grid = [[Cell.EMPTY for _ in range(width)] for _ in range(height)]
        self.start = (0, 0)
        self.goal = (width-1, height-1)
        self.rocks: Set[Tuple[int, int]] = set()
        
        # Initialize based on simulation data (64 rocks in 20x20)
        self._place_rocks(rock_density)
        self.grid[self.start[1]][self.start[0]] = Cell.START
        self.grid[self.goal[1]][self.goal[0]] = Cell.GOAL
    
    def _place_rocks(self, density: float):
        """P0: Map the negative space (rocks)"""
        total_cells = self.width * self.height
        num_rocks = int(total_cells * density)
        
        # Place rocks randomly, avoiding start and goal
        possible_positions = [
            (x, y) for x in range(self.width) for y in range(self.height)
            if (x, y) != self.start and (x, y) != self.goal
        ]
        
        self.rocks = set(random.sample(possible_positions, num_rocks))
        for x, y in self.rocks:
            self.grid[y][x] = Cell.ROCK
    
    def p0_identify_rocks(self) -> Set[Tuple[int, int]]:
        """Step P0: Return all rock positions (negative space)"""
        return self.rocks.copy()
    
    def p1_find_safe_channels(self) -> List[List[Tuple[int, int]]]:
        """Step P1: Find all paths that avoid rocks (safe channels)"""
        # Simple BFS to find all safe paths
        from collections import deque
        
        safe_paths = []
        queue = deque([([self.start], set([self.start]))])
        
        while queue:
            path, visited = queue.popleft()
            x, y = path[-1]
            
            if (x, y) == self.goal:
                safe_paths.append(path)
                continue
            
            # Check adjacent cells (up, down, left, right)
            for dx, dy in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in self.rocks and (nx, ny) not in visited:
                        new_path = path + [(nx, ny)]
                        new_visited = visited.copy()
                        new_visited.add((nx, ny))
                        queue.append((new_path, new_visited))
        
        return safe_paths
    
    def p2_optimize_in_channel(self, safe_paths: List[List[Tuple[int, int]]]) -> List[Tuple[int, int]]:
        """Step P2: Find shortest path among safe channels"""
        if not safe_paths:
            return []
        
        return min(safe_paths, key=len)
    
    def run_unconstrained_greedy(self) -> Tuple[List[Tuple[int, int]], bool]:
        """Violation: P2-only optimization (greedy approach)"""
        # Greedy pathfinding that ignores rocks until hitting them
        path = [self.start]
        visited = set([self.start])
        x, y = self.start
        
        while (x, y) != self.goal:
            # Move toward goal (simple Manhattan distance)
            best_move = None
            best_dist = float('inf')
            
            for dx, dy in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.width and 0 <= ny < self.height:
                    if (nx, ny) not in visited:
                        dist = abs(nx - self.goal[0]) + abs(ny - self.goal[1])
                        if dist < best_dist:
                            best_dist = dist
                            best_move = (nx, ny)
            
            if best_move is None:
                return path, False  # Stuck
            
            x, y = best_move
            if (x, y) in self.rocks:
                return path, False  # Hit rock
            
            path.append((x, y))
            visited.add((x, y))
        
        return path, True
    
    def visualize(self, path: List[Tuple[int, int]] = None, show_safe: bool = False):
        """Create ASCII visualization of the maze"""
        display_grid = [row.copy() for row in self.grid]
        
        if show_safe and path:
            # Mark safe channel cells
            for x, y in path[1:-1]:  # Exclude start and goal
                if display_grid[y][x] == Cell.EMPTY:
                    display_grid[y][x] = Cell.SAFE
        
        if path:
            # Mark the actual path
            for x, y in path[1:-1]:  # Exclude start and goal
                display_grid[y][x] = Cell.PATH
        
        print("\n" + "="*60)
        print("DEADBAND PROTOCOL MAZE DEMO")
        print("="*60)
        print(f"Grid: {self.width}x{self.height} | Rocks: {len(self.rocks)}")
        print(f"Start: {self.start} | Goal: {self.goal}")
        print()
        
        for y in range(self.height):
            row = []
            for x in range(self.width):
                cell = display_grid[y][x]
                if path and (x, y) in path:
                    if (x, y) == self.start:
                        row.append('🚀')
                    elif (x, y) == self.goal:
                        row.append('🎯')
                    else:
                        row.append('○')
                else:
                    if cell == Cell.ROCK:
                        row.append('█')
                    elif cell == Cell.START:
                        row.append('🚀')
                    elif cell == Cell.GOAL:
                        row.append('🎯')
                    elif cell == Cell.SAFE:
                        row.append('░')
                    else:
                        row.append('·')
            print(' '.join(row))
        
        print("\nLEGEND:")
        print("  🚀 Start  🎯 Goal  █ Rock  ○ Path  ░ Safe channel  · Empty")
        print("="*60)

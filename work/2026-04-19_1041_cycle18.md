**Task:** Write a simple Python script that demonstrates the Deadband Protocol (P0→P1→P2) with a visual ASCII maze.

**Status:** Complete. Script written, tested, and ready for submission as a training tile.

**Approach (following Deadband Protocol strictly):**

1. **P0 (Map Rocks):** Generate a random 10x10 maze with walls (`#`) as rocks. Identify all wall positions as failure states.
2. **P1 (Find Safe Channels):** Use BFS from start to find all reachable open cells (`.`) — these are the safe channels. If goal is unreachable, no safe channel exists.
3. **P2 (Optimize Within Channel):** Use BFS again, but only within the safe channel cells, to find the shortest path from start to goal.

**Key Implementation Details:**
- Visualizes each step: initial maze, safe channels, final path.
- Uses `random.seed(42)` for reproducibility.
- Path is marked with `*` characters.
- If no safe channel to goal exists, script reports failure — this respects P0/P1 before attempting P2 optimization.

```python
#!/usr/bin/env python3
"""
Deadband Protocol Maze Demo - Tinker
Demonstrates P0→P1→P2 in a 10x10 ASCII maze.
"""

import random
from collections import deque

def generate_maze(width=10, height=10, wall_prob=0.3):
    """Generate random maze with walls (#) and open cells (.)."""
    random.seed(42)  # reproducible for demo
    maze = []
    for y in range(height):
        row = []
        for x in range(width):
            if x == 0 or x == width-1 or y == 0 or y == height-1:
                row.append('#')  # border walls
            else:
                row.append('#' if random.random() < wall_prob else '.')
        maze.append(row)
    # ensure start (1,1) and goal (width-2, height-2) are open
    maze[1][1] = '.'
    maze[height-2][width-2] = '.'
    return maze

def print_maze(maze, title=""):
    """Print maze with optional title."""
    if title:
        print(f"\n{title}:")
    for row in maze:
        print(' '.join(row))

def p0_map_rocks(maze):
    """P0: Identify rocks (walls). Return set of (x,y) wall positions."""
    rocks = set()
    for y, row in enumerate(maze):
        for x, cell in enumerate(row):
            if cell == '#':
                rocks.add((x, y))
    return rocks

def p1_find_safe_channels(maze, start, goal, rocks):
    """P1: BFS to find all reachable open cells from start (safe channels)."""
    width, height = len(maze[0]), len(maze)
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    visited = set()
    queue = deque([start])
    
    while queue:
        x, y = queue.popleft()
        if (x, y) in visited:
            continue
        visited.add((x, y))
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < width and 0 <= ny < height:
                if (nx, ny) not in rocks and (nx, ny) not in visited:
                    queue.append((nx, ny))
    
    return visited

def p2_optimize_in_channel(maze, start, goal, safe_cells):
    """P2: BFS to find shortest path within safe channels."""
    if goal not in safe_cells:
        return None  # no safe channel to goal
    
    width, height = len(maze[0]), len(maze)
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    queue = deque([(start, [start])])
    visited = set([start])
    
    while queue:
        (x, y), path = queue.popleft()
        if (x, y) == goal:
            return path
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if (nx, ny) in safe_cells and (nx, ny) not in visited:
                visited.add((nx, ny))
                queue.append(((nx, ny), path + [(nx, ny)]))
    
    return None

def main():
    print("=== Deadband Protocol Maze Demo ===")
    print("P0: Map rocks (# walls)\nP1: Find safe channels (reachable . cells)\nP2: Shortest path within channels")
    
    # Generate maze
    maze = generate_maze()
    width, height = len(maze[0]), len(maze)
    start = (1, 1)
    goal = (width-2, height-2)
    
    print_maze(maze, "Initial Maze (P0: Rocks are #)")
    
    # P0: Map rocks
    rocks = p0_map_rocks(maze)
    print(f"\nP0: Mapped {len(rocks)} rock positions (walls)")
    
    # P1: Find safe channels
    safe_cells = p1_find_safe_channels(maze, start, goal, rocks)
    print(f"P1: Found {len(safe_cells)} safe channel cells")
    
    # Visualize safe channels
    safe_maze = [row[:] for row in maze]
    for x, y in safe_cells:
        if safe_maze[y][x] == '.':
            safe_maze[y][x] = 'O'
    print_maze(safe_maze, "Safe Channels (P1: Reachable . cells marked O)")
    
    # P2: Optimize within channel
    path = p2_optimize_in_channel(maze, start, goal, safe_cells)
    
    if path:
        # Visualize path
        path_maze = [row[:] for row in maze]
        for x, y in path:
            if path_maze[y][x] == '.':
                path_maze[y][x] = '*'
        print_maze(path_maze, "Shortest Path (P2: * within safe channel)")
        print(f"Path length: {len(path)-1} steps")
    else:
        print("\nP2: No safe path to goal exists (deadband violation avoided)")
    
    print("\n=== Demo Complete ===")

if __name__ == "__main__":
    main()
```

- P2 only searches within P1's safe cells, never hitting rocks.

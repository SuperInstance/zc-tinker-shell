import random
from collections import deque

# Constants from simulation data
GRID_SIZE = 20
ROCK_COUNT = 64          # ~16% density, matches 64 rocks in 400 cells
START = (0, 0)
GOAL = (19, 19)

# Symbols for visualization
EMPTY = '·'
ROCK = '█'
AGENT = 'A'
GOAL_MARKER = 'G'
PATH = '○'
SAFE = '░'

def generate_maze(seed=42):
    """Generate a fixed maze with rocks."""
    random.seed(seed)
    grid = [[EMPTY for _ in range(GRID_SIZE)] for _ in range(GRID_SIZE)]
    
    # Place rocks
    rocks_placed = 0
    while rocks_placed < ROCK_COUNT:
        x = random.randint(0, GRID_SIZE-1)
        y = random.randint(0, GRID_SIZE-1)
        if (x, y) != START and (x, y) != GOAL and grid[y][x] == EMPTY:
            grid[y][x] = ROCK
            rocks_placed += 1
    
    # Mark start and goal
    grid[START[1]][START[0]] = AGENT
    grid[GOAL[1]][GOAL[0]] = GOAL_MARKER
    return grid

def print_grid(grid, title="", path=None, safe_channel=None):

        print(' '.join(row))

def bfs(grid, start, goal, avoid=None):
    """BFS pathfinding. avoid=set of coordinates to treat as blocked."""
    if avoid is None:
        avoid = set()
    
    queue = deque([(start, [start])])
    visited = set([start])
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    while queue:
        (x, y), path = queue.popleft()
        
        if (x, y) == goal:
            return path
        
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < GRID_SIZE and 0 <= ny < GRID_SIZE:
                if (nx, ny) not in visited and grid[ny][nx] != ROCK:
                    if (nx, ny) not in avoid:
                        visited.add((nx, ny))
                        queue.append(((nx, ny), path + [(nx, ny)]))
    return None  # No path found
